2//! N-axis linear coordinate.
3//!
4//! [`LinearCoordinate`] implements a general linear transformation between
5//! pixel and world coordinates:
6//!
7//! ```text
8//! world = crval + cdelt * PC * (pixel - crpix)
9//! ```
10//!
11//! where `PC` is a rotation/coupling matrix (identity by default), `cdelt`
12//! is the axis increment vector, `crval` is the reference world value, and
13//! `crpix` is the reference pixel position.
14//!
15//! This corresponds to C++ `LinearCoordinate`.
16
17use casacore_types::{RecordValue, ScalarValue, Value};
18use ndarray::Array2;
19
20use crate::coordinate::{Coordinate, CoordinateType};
21use crate::error::CoordinateError;

23/// An N-axis linear coordinate with an affine pixel-to-world mapping.
24///
25/// The transformation follows the FITS WCS linear convention:
26///
27/// ```text
28/// world[i] = crval[i] + cdelt[i] * sum_j(PC[i][j] * (pixel[j] - crpix[j]))
29/// ```
30///
31/// The PC matrix defaults to the identity, making the transformation a simple
32/// scale-and-shift along each axis independently.
33///
34/// Corresponds to C++ `LinearCoordinate`.
35#[derive(Debug, Clone)]
36pub struct LinearCoordinate {
37    crval: Vec<f64>,
38    cdelt: Vec<f64>,
39    crpix: Vec<f64>,
40    pc: Array2<f64>,
41    names: Vec<String>,
42    units: Vec<String>,
43}
44
45impl LinearCoordinate {
46    /// Creates a new linear coordinate with `naxes` axes and default values.
47    ///
48    /// The reference value, reference pixel, and all increments default to 0.0
49    /// (increments default to 1.0). Axis names default to `"Linear1"`,
50    /// `"Linear2"`, etc. Units default to empty strings. The PC matrix
51    /// defaults to the identity.
52    ///
53    /// # Panics
54    ///
55    /// Panics if `naxes` is 0.
56    pub fn new(naxes: usize, names: Vec<String>, units: Vec<String>) -> Self {
57        assert!(naxes > 0, "LinearCoordinate requires at least 1 axis");
58        let names = if names.len() == naxes {
59            names
60        } else {
61            (1..=naxes).map(|i| format!("Linear{i}")).collect()
62        };
63        let units = if units.len() == naxes {
64            units
65        } else {
66            vec![String::new(); naxes]
67        };
68        Self {
69            crval: vec![0.0; naxes],
70            cdelt: vec![1.0; naxes],
71            crpix: vec![0.0; naxes],
72            pc: Array2::eye(naxes),
73            names,
74            units,
75        }
76    }
77
78    /// Sets the reference world values. Returns `self` for chaining.
79    ///
80    /// # Panics
81    ///
82    /// Panics if `crval.len()` does not match the number of axes.
83    pub fn with_reference_value(mut self, crval: Vec<f64>) -> Self {
84        assert_eq!(crval.len(), self.n_axes());
85        self.crval = crval;
86        self
87    }
88
89    /// Sets the reference pixel positions. Returns `self` for chaining.
90    ///
91    /// # Panics
92    ///
93    /// Panics if `crpix.len()` does not match the number of axes.
94    pub fn with_reference_pixel(mut self, crpix: Vec<f64>) -> Self {
95        assert_eq!(crpix.len(), self.n_axes());
96        self.crpix = crpix;
97        self
98    }
99
100    /// Sets the axis increments. Returns `self` for chaining.
101    ///
102    /// # Panics
103    ///
104    /// Panics if `cdelt.len()` does not match the number of axes.
105    pub fn with_increment(mut self, cdelt: Vec<f64>) -> Self {
106        assert_eq!(cdelt.len(), self.n_axes());
107        self.cdelt = cdelt;
108        self
109    }
110
111    /// Sets the PC (rotation/coupling) matrix. Returns `self` for chaining.
112    ///
113    /// # Panics
114    ///
115    /// Panics if the matrix shape is not `(naxes, naxes)`.
116    pub fn with_pc_matrix(mut self, pc: Array2<f64>) -> Self {
117        let n = self.n_axes();
118        assert_eq!(pc.shape(), [n, n]);
119        self.pc = pc;
120        self
121    }
122
123    /// Returns the number of axes.
124    fn n_axes(&self) -> usize {
125        self.crval.len()
126    }
127
128    /// Returns a reference to the PC rotation/coupling matrix.
129    pub fn pc_matrix(&self) -> &Array2<f64> {
130        &self.pc
131    }
132}
133
134impl Coordinate for LinearCoordinate {
135    fn coordinate_type(&self) -> CoordinateType {
136        CoordinateType::Linear
137    }
138
139    fn n_pixel_axes(&self) -> usize {
140        self.n_axes()
141    }
142
143    fn n_world_axes(&self) -> usize {
144        self.n_axes()
145    }
146
147    fn to_world(&self, pixel: &[f64]) -> Result<Vec<f64>, CoordinateError> {
148        let n = self.n_axes();
149        if pixel.len() != n {
150            return Err(CoordinateError::DimensionMismatch {
151                expected: n,
152                got: pixel.len(),
153            });
154        }
155
156        // delta_pixel = pixel - crpix
157        let delta: Vec<f64> = pixel.iter().zip(&self.crpix).map(|(p, r)| p - r).collect();
158
159        // Apply PC matrix: rotated = PC * delta
160        let mut rotated = vec![0.0; n];
161        for (i, rotated_i) in rotated.iter_mut().enumerate() {
162            for (j, delta_j) in delta.iter().enumerate() {
163                *rotated_i += self.pc[[i, j]] * delta_j;
164            }
165        }
166
167        // world = crval + cdelt * rotated
168        let world: Vec<f64> = (0..n)
169            .map(|i| self.crval[i] + self.cdelt[i] * rotated[i])
170            .collect();
171
172        Ok(world)
173    }
174
175    fn to_pixel(&self, world: &[f64]) -> Result<Vec<f64>, CoordinateError> {
176        let n = self.n_axes();
177        if world.len() != n {
178            return Err(CoordinateError::DimensionMismatch {
179                expected: n,
180                got: world.len(),
181            });
182        }
183
184        // Compute (world - crval) / cdelt
185        let mut scaled = vec![0.0; n];
186        for i in 0..n {
187            if self.cdelt[i].abs() < 1e-300 {
188                return Err(CoordinateError::ConversionFailed(format!(
189                    "zero increment on axis {i}"
190                )));
191            }
192            scaled[i] = (world[i] - self.crval[i]) / self.cdelt[i];
193        }
194
195        // Invert PC matrix: pixel = crpix + PC^-1 * scaled
196        // For small N, use direct inversion via cofactor / Gauss elimination.
197        let inv_pc = invert_matrix(&self.pc)?;
198
199        let mut pixel = vec![0.0; n];
200        for i in 0..n {
201            let mut sum = 0.0;
202            for j in 0..n {
203                sum += inv_pc[[i, j]] * scaled[j];
204            }
205            pixel[i] = sum + self.crpix[i];
206        }
207
208        Ok(pixel)
209    }
210
211    fn reference_value(&self) -> Vec<f64> {
212        self.crval.clone()
213    }
214
215    fn reference_pixel(&self) -> Vec<f64> {
216        self.crpix.clone()
217    }
218
219    fn increment(&self) -> Vec<f64> {
220        self.cdelt.clone()
221    }
222
223    fn axis_names(&self) -> Vec<String> {
224        self.names.clone()
225    }
226
227    fn axis_units(&self) -> Vec<String> {
228        self.units.clone()
229    }
230
231    fn to_record(&self) -> RecordValue {
232        let mut rec = RecordValue::default();
233        let n = self.n_axes();
234
235        rec.upsert(
236            "coordinate_type",
237            Value::Scalar(ScalarValue::String("Linear".into())),
238        );
239        rec.upsert("naxes", Value::Scalar(ScalarValue::Int32(n as i32)));
240        rec.upsert(
241            "crval",
242            Value::Array(casacore_types::ArrayValue::from_f64_vec(self.crval.clone())),
243        );
244        rec.upsert(
245            "cdelt",
246            Value::Array(casacore_types::ArrayValue::from_f64_vec(self.cdelt.clone())),
247        );
248        rec.upsert(
249            "crpix",
250            Value::Array(casacore_types::ArrayValue::from_f64_vec(self.crpix.clone())),
251        );
252        // PC matrix as flattened row-major array
253        let pc_flat: Vec<f64> = self.pc.iter().copied().collect();
254        rec.upsert(
255            "pc",
256            Value::Array(casacore_types::ArrayValue::from_f64_vec(pc_flat)),
257        );
258        rec.upsert(
259            "axes",
260            Value::Array(casacore_types::ArrayValue::from_string_vec(
261                self.names.clone(),
262            )),
263        );
264        rec.upsert(
265            "units",
266            Value::Array(casacore_types::ArrayValue::from_string_vec(
267                self.units.clone(),
268            )),
269        );
270
271        rec
272    }
273
274    fn clone_box(&self) -> Box<dyn Coordinate> {
275        Box::new(self.clone())
276    }
277}

279/// Inverts a square matrix using Gauss-Jordan elimination with partial pivoting.
280///
281/// Returns an error if the matrix is singular.
282pub(crate) fn invert_matrix(m: &Array2<f64>) -> Result<Array2<f64>, CoordinateError> {
283    let n = m.nrows();
284    assert_eq!(n, m.ncols());
285
286    // Augmented matrix [M | I]
287    let mut aug = Array2::zeros((n, 2 * n));
288    for i in 0..n {
289        for j in 0..n {
290            aug[[i, j]] = m[[i, j]];
291        }
292        aug[[i, n + i]] = 1.0;
293    }
294
295    // Forward elimination with partial pivoting
296    for col in 0..n {
297        // Find pivot
298        let mut max_val = aug[[col, col]].abs();
299        let mut max_row = col;
300        for row in (col + 1)..n {
301            let val = aug[[row, col]].abs();
302            if val > max_val {
303                max_val = val;
304                max_row = row;
305            }
306        }
307
308        if max_val < 1e-15 {
309            return Err(CoordinateError::ConversionFailed(
310                "singular PC matrix".into(),
311            ));
312        }
313
314        // Swap rows
315        if max_row != col {
316            for j in 0..2 * n {
317                let tmp = aug[[col, j]];
318                aug[[col, j]] = aug[[max_row, j]];
319                aug[[max_row, j]] = tmp;
320            }
321        }
322
323        // Scale pivot row
324        let pivot = aug[[col, col]];
325        for j in 0..2 * n {
326            aug[[col, j]] /= pivot;
327        }
328
329        // Eliminate column
330        for row in 0..n {
331            if row != col {
332                let factor = aug[[row, col]];
333                for j in 0..2 * n {
334                    aug[[row, j]] -= factor * aug[[col, j]];
335                }
336            }
337        }
338    }
339
340    // Extract inverse from right half
341    let mut inv = Array2::zeros((n, n));
342    for i in 0..n {
343        for j in 0..n {
344            inv[[i, j]] = aug[[i, n + j]];
345        }
346    }
347
348    Ok(inv)
349}

355    #[test]
356    fn default_linear_identity() {
357        let coord = LinearCoordinate::new(2, vec![], vec![]);
358        let pixel = vec![10.0, 20.0];
359        let world = coord.to_world(&pixel).unwrap();
360        // crval=0, cdelt=1, crpix=0, PC=I => world = pixel
361        assert!((world[0] - 10.0).abs() < 1e-12);
362        assert!((world[1] - 20.0).abs() < 1e-12);
363    }

365    #[test]
366    fn linear_with_offset_and_scale() {
367        let coord = LinearCoordinate::new(1, vec!["Frequency".into()], vec!["Hz".into()])
368            .with_reference_value(vec![1.4e9])
369            .with_reference_pixel(vec![100.0])
370            .with_increment(vec![1e6]);
371
372        // At crpix, world = crval
373        let world = coord.to_world(&[100.0]).unwrap();
374        assert!((world[0] - 1.4e9).abs() < 1.0);
375
376        // 10 pixels above crpix
377        let world = coord.to_world(&[110.0]).unwrap();
378        assert!((world[0] - (1.4e9 + 10.0 * 1e6)).abs() < 1.0);
379    }

382    fn linear_roundtrip() {
383        let coord = LinearCoordinate::new(2, vec![], vec![])
384            .with_reference_value(vec![100.0, 200.0])
385            .with_reference_pixel(vec![50.0, 50.0])
386            .with_increment(vec![2.0, 3.0]);

388        let pixel = vec![75.0, 30.0];
389        let world = coord.to_world(&pixel).unwrap();
390        let back = coord.to_pixel(&world).unwrap();

398        // 90-degree rotation

470        for i in 0..3 {
