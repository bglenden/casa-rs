2//! Virtual column engine trait and registry.
3//!
4//! Virtual columns are computed from other data rather than stored on disk.
5//! During loading, stored columns are populated first (pass 1), then virtual
6//! engines materialize their columns from the already-loaded data (pass 2).
7//!
8//! # C++ equivalent
9//!
10//! `VirtualColumnEngine` / `DataManager::isVirtualColumn()` in
11//! `casacore/tables/DataMan/VirtualColEngine.h`.
12
13use std::fmt;

17
18use super::StorageError;

21/// Context provided to virtual engines during materialization.
22///
23/// Contains the column descriptors, already-loaded row data (from pass 1),
24/// the table's filesystem path, and the total row count.
25pub(crate) struct VirtualContext<'a> {
26    /// All column descriptors from the table descriptor.
27    pub col_descs: &'a [ColumnDescContents],
28    /// Already-loaded rows from storage managers (pass 1).
29    /// Virtual engines read stored column values from here.
30    pub rows: &'a [RecordValue],
31    /// Filesystem path of the table being loaded.
32    pub table_path: &'a Path,
33    /// Total number of rows in the table.
34    pub nrrow: usize,
35}
36
37/// Trait for virtual column engines that compute column values from other data.
38///
39/// Implementors materialize virtual column values into the row records during
40/// table loading. The engine reads configuration from column keywords and
41/// computes values based on stored columns or external tables.
42///
43/// # C++ equivalent
44///
45/// `VirtualColumnEngine` in `casacore/tables/DataMan/VirtualColEngine.h`.
46pub(crate) trait VirtualColumnEngine: fmt::Debug {
47    /// Returns the C++ data manager type name (e.g. `"ForwardColumnEngine"`).
48    fn type_name(&self) -> &str;
49
50    /// Materialize virtual column values into the row records.
51    ///
52    /// `bound_cols` contains `(desc_index, PlainColumnEntry)` pairs for the
53    /// columns bound to this engine instance. The engine should insert
54    /// computed values into each row of `rows` for these columns.
55    fn materialize(
56        &self,
57        ctx: &VirtualContext,
58        bound_cols: &[(usize, &PlainColumnEntry)],
59        rows: &mut [RecordValue],
60    ) -> Result<(), StorageError>;
61}

63/// Look up a virtual engine implementation by its C++ type name.
64///
65/// Returns `Some(engine)` for supported virtual engine types, `None` otherwise.
66/// Uses prefix matching for parameterized engines like `ScaledArrayEngine<...>`.
67pub(crate) fn lookup_engine(type_name: &str) -> Option<Box<dyn VirtualColumnEngine>> {
68    use super::virtual_compress::{
69        CompressComplexEngine, CompressComplexVariant, CompressFloatEngine,
70    };
71    use super::virtual_scaled_array::{ScaledColumnEngine, ScaledVariant};
72    if type_name == "ForwardColumnEngine" {
73        Some(Box::new(super::virtual_forward::ForwardColumnEngine))
74    } else if type_name == "ForwardColumnIndexedRowEngine" {
75        Some(Box::new(
76            super::virtual_forward::ForwardColumnIndexedRowEngine,
77        ))
78    } else if type_name.starts_with("ScaledArrayEngine") {
79        Some(Box::new(ScaledColumnEngine {
80            variant: ScaledVariant::Array,
81        }))
82    } else if type_name.starts_with("ScaledComplexData") {
83        Some(Box::new(ScaledColumnEngine {
84            variant: ScaledVariant::ComplexData,
85        }))
86    } else if type_name.starts_with("BitFlagsEngine") {
87        Some(Box::new(super::virtual_bitflags::BitFlagsEngine))
88    } else if type_name == "CompressFloat" {
89        Some(Box::new(CompressFloatEngine))
90    } else if type_name == "CompressComplex" {
91        Some(Box::new(CompressComplexEngine {
92            variant: CompressComplexVariant::Standard,
93        }))
94    } else if type_name == "CompressComplexSD" {
95        Some(Box::new(CompressComplexEngine {
96            variant: CompressComplexVariant::SingleDish,
97        }))
98    } else if type_name == "VirtualTaQLColumn" {
99        Some(Box::new(
100            super::virtual_taql_column::VirtualTaQLColumnEngine,
101        ))
102    } else {
103        None
104    }
105}

107/// Returns `true` if the given DM type name is a recognized virtual engine.
108pub(crate) fn is_virtual_engine(type_name: &str) -> bool {
109    type_name == "ForwardColumnEngine"
110        || type_name == "ForwardColumnIndexedRowEngine"
111        || type_name.starts_with("ScaledArrayEngine")
112        || type_name.starts_with("ScaledComplexData")
113        || type_name.starts_with("BitFlagsEngine")
114        || type_name == "CompressFloat"
115        || type_name == "CompressComplex"
116        || type_name == "CompressComplexSD"
117        || type_name == "VirtualTaQLColumn"
118}

120/// Metadata for a virtual column binding, used during save to produce
121/// the correct DM entries and column keywords in `table.dat`.
122#[derive(Debug, Clone)]
123pub(crate) enum VirtualColumnBinding {
124    /// A ForwardColumnEngine binding: column `col_name` reads from `ref_table`.
125    Forward {
126        col_name: String,
127        ref_table: PathBuf,
128    },
129    /// A ScaledArrayEngine binding: column `virtual_col` = `stored_col * scale + offset`.
130    ScaledArray {
131        virtual_col: String,
132        stored_col: String,
133        scale: f64,
134        offset: f64,
135    },
136    /// A ScaledComplexData binding: stored `[2, ...]` → virtual Complex with
137    /// complex-valued scale/offset applied per re/im component.
138    ///
139    /// # C++ equivalent
140    ///
141    /// `ScaledComplexData<Complex,Short>(virtualCol, storedCol, scale, offset)`.
142    ScaledComplexData {
143        virtual_col: String,
144        stored_col: String,
145        scale: Complex64,
146        offset: Complex64,
147    },
148    /// A BitFlagsEngine binding: stored integer → virtual Bool via bitmask.
149    ///
150    /// # C++ equivalent
151    ///
152    /// `BitFlagsEngine<uChar|Short|Int>`.
153    BitFlags {
154        virtual_col: String,
155        stored_col: String,
156        read_mask: u32,
157        write_mask: u32,
158    },
159    /// A CompressFloat binding: stored Int16 → virtual Float via FITS-style scaling.
160    ///
161    /// # C++ equivalent
162    ///
163    /// `CompressFloat` in `casacore/tables/DataMan/CompressFloat.h`.
164    CompressFloat {
165        virtual_col: String,
166        stored_col: String,
167        scale: f32,
168        offset: f32,
169    },
170    /// A CompressComplex binding: stored Int32 → virtual Complex.
171    ///
172    /// # C++ equivalent
173    ///
174    /// `CompressComplex` / `CompressComplexSD`.
175    CompressComplex {
176        virtual_col: String,
177        stored_col: String,
178        scale: f32,
179        offset: f32,
180        single_dish: bool,
181    },
182    /// A ForwardColumnIndexedRowEngine binding: forwarding with row remapping.
183    ///
184    /// # C++ equivalent
185    ///
186    /// `ForwardColumnIndexedRowEngine`.
187    ForwardIndexedRow {
188        col_name: String,
189        ref_table: PathBuf,
190        row_column: String,
191    },
192    /// A VirtualTaQLColumn binding: column computed from a TaQL expression.
193    ///
194    /// # C++ equivalent
195    ///
196    /// `VirtualTaQLColumn` in `casacore/tables/DataMan/VirtualTaQLColumn.h`.
197    TaQLColumn {
198        col_name: String,
199        expression: String,
200    },
201}

207    #[test]
208    fn lookup_engine_matches_all_supported_virtual_engines() {
209        let cases = [
210            (
211                "ForwardColumnEngine",
212                "ForwardColumnEngine",
213                "ForwardColumnEngine",
214            ),
215            (
216                "ForwardColumnIndexedRowEngine",
217                "ForwardColumnIndexedRowEngine",
218                "ForwardColumnIndexedRowEngine",
219            ),
220            (
221                "ScaledArrayEngine<Float,Short>",
222                "ScaledArrayEngine",
223                "ScaledColumnEngine",
224            ),
225            (
226                "ScaledComplexData<Complex,Short>",
227                "ScaledComplexData",
228                "ScaledColumnEngine",
229            ),
230            ("BitFlagsEngine<Int>", "BitFlagsEngine", "BitFlagsEngine"),
231            ("CompressFloat", "CompressFloat", "CompressFloatEngine"),
232            (
233                "CompressComplex",
234                "CompressComplex",
235                "CompressComplexEngine",
236            ),
237            (
238                "CompressComplexSD",
239                "CompressComplexSD",
240                "CompressComplexEngine",
241            ),
242            (
243                "VirtualTaQLColumn",
244                "VirtualTaQLColumn",
245                "VirtualTaQLColumnEngine",
246            ),
247        ];
248
249        for (type_name, expected_type_name, expected_debug_name) in cases {
250            let engine = lookup_engine(type_name)
251                .unwrap_or_else(|| panic!("expected virtual engine for {type_name}"));
252            assert!(
253                is_virtual_engine(type_name),
254                "{type_name} should be recognized"
255            );
256            assert_eq!(engine.type_name(), expected_type_name);
257            assert!(
258                format!("{engine:?}").contains(expected_debug_name),
259                "debug output for {type_name} should identify {expected_debug_name}"
260            );
261        }
262    }

266        let cases = [
