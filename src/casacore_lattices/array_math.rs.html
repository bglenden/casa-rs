2//! Array-level mathematical utilities mirroring C++ `casacore/casa/Arrays/ArrayMath.h`.
3//!
4//! `ndarray` already covers element-wise arithmetic and basic reductions
5//! (`sum`, `mean`). The functions here fill the gaps needed for idiomatic
6//! casa-rs code:
7//!
8//! | Rust                 | C++ equivalent                                     |
9//! |----------------------|----------------------------------------------------|
10//! | [`array_median`]     | `casacore::median(arr)`                            |
11//! | [`array_fractile`]   | `casacore::fractile(arr, fraction)`                |
12//! | [`near`]             | `casacore::near(a, b)` (f64 default tol=1e-13)     |
13//! | [`near_f32`]         | `casacore::near(a, b)` (f32 default tol=1e-5)      |
14//! | [`near_tol`]         | `casacore::near(a, b, tol)`                        |
15//! | [`near_abs`]         | `casacore::nearAbs(a, b, tol)`                     |
16

25/// Compute the median of all elements in `arr`.
26///
27/// For even-length arrays the two middle elements are averaged, matching C++
28/// `casacore::median(arr, takeEvenMean=true)`.
29///
30/// # Example
31///
32/// ```rust
33/// use casacore_lattices::array_median;
34/// use ndarray::{ArrayD, IxDyn};
35///
36/// let a = ArrayD::from_shape_vec(IxDyn(&[4]), vec![3.0_f32, 1.0, 4.0, 2.0]).unwrap();
37/// // sorted: [1,2,3,4], median = (2+3)/2 = 2.5
38/// assert_eq!(array_median(&a), 2.5);
39/// ```
40pub fn array_median<T: StatsElement>(arr: &ArrayD<T>) -> f64 {
41    let values: Vec<f64> = arr.iter().map(|v| v.to_f64_stats()).collect();
42    casacore_median(&values)
43}
44
45/// Compute the `fraction`-th fractile (percentile) of all elements in `arr`.
46///
47/// Uses C++ `casacore::fractile` semantics: returns
48/// `sorted[floor((n-1)*fraction + 0.01)]` — nearest-rank selection with the
49/// C++ epsilon offset.
50///
51/// # Example
52///
53/// ```rust
54/// use casacore_lattices::array_fractile;
55/// use ndarray::{ArrayD, IxDyn};
56///
57/// // indgen 0..63
58/// let a = ArrayD::from_shape_fn(IxDyn(&[64]), |idx| idx[0] as f32);
59/// // C++ fractile(arr, 0.25) = sorted[floor(63*0.25+0.01)] = sorted[15] = 15
60/// assert_eq!(array_fractile(&a, 0.25), 15.0);
61/// assert_eq!(array_fractile(&a, 0.75), 47.0);
62/// ```
63pub fn array_fractile<T: StatsElement>(arr: &ArrayD<T>, fraction: f64) -> f64 {
64    let values: Vec<f64> = arr.iter().map(|v| v.to_f64_stats()).collect();
65    casacore_fractile(&values, fraction)
66}

68/// Compute the median absolute deviation from the median.
69///
70/// Matches C++ `casacore::madfm(arr)`.
71///
72/// # Example
73///
74/// ```rust
75/// use casacore_lattices::array_madfm;
76/// use ndarray::{ArrayD, IxDyn};
77///
78/// let a = ArrayD::from_shape_fn(IxDyn(&[64]), |idx| idx[0] as f32);
79/// // For indgen 0..63: median=31.5, madfm=16.0
80/// assert_eq!(array_madfm(&a), 16.0);
81/// ```
82pub fn array_madfm<T: StatsElement>(arr: &ArrayD<T>) -> f64 {
83    let values: Vec<f64> = arr.iter().map(|v| v.to_f64_stats()).collect();
84    casacore_madfm(&values)
85}

91/// Relative-tolerance comparison for `f64` values.
92///
93/// Returns `true` if `|a - b| <= tol * max(|a|, |b|)`.
94///
95/// Default tolerance for `f64` is `1e-13`, matching C++
96/// `casacore::near(Double, Double)`.
97///
98/// # Special cases
99///
100/// - Returns `true` if `a == b` (handles infinities).
101/// - Returns `false` if `a` and `b` have opposite signs.
102/// - Returns `true` if `a == 0` and `|b|` is below the floating-point minimum
103///   (matching C++ behaviour).
104///
105/// # Example
106///
107/// ```rust
108/// use casacore_lattices::near;
109/// assert!(near(1.0, 1.0 + 1e-14));
110/// assert!(!near(1.0, 1.001));
111/// ```
112pub fn near(a: f64, b: f64) -> bool {
113    near_tol(a, b, 1e-13)
114}
115
116/// Relative-tolerance comparison for `f32`-precision values.
117///
118/// Default tolerance `1e-5`, matching C++ `casacore::near(Float, Float)`.
119pub fn near_f32(a: f64, b: f64) -> bool {
120    near_tol(a, b, 1e-5)
121}
122
123/// Relative-tolerance comparison with explicit tolerance.
124///
125/// Mirrors C++ `casacore::near(a, b, tol)`.
126pub fn near_tol(a: f64, b: f64, tol: f64) -> bool {
127    if tol <= 0.0 {
128        return a == b;
129    }
130    if a == b {
131        return true;
132    }
133    if a == 0.0 {
134        return b.abs() <= (1.0 + tol) * f64::MIN_POSITIVE;
135    }
136    if b == 0.0 {
137        return a.abs() <= (1.0 + tol) * f64::MIN_POSITIVE;
138    }
139    if (a > 0.0) != (b > 0.0) {
140        return false;
141    }
142    (a - b).abs() <= tol * a.abs().max(b.abs())
143}
144
145/// Absolute-tolerance comparison.
146///
147/// Returns `true` if `|a - b| <= tol`.
148/// Mirrors C++ `casacore::nearAbs(a, b, tol)`.
149pub fn near_abs(a: f64, b: f64, tol: f64) -> bool {
150    (a - b).abs() <= tol
151}

153#[cfg(test)]
154mod tests {
155    use super::*;
156    use ndarray::{ArrayD, IxDyn};
157
158    #[test]
159    fn order_statistics_match_casacore_semantics() {
160        let values = ArrayD::from_shape_vec(IxDyn(&[2, 3]), vec![9.0_f32, 1.0, 4.0, 7.0, 3.0, 6.0])
161            .expect("valid test array");
162
163        assert_eq!(array_median(&values), 5.0);
164        assert_eq!(array_fractile(&values, 0.25), 3.0);
165        assert_eq!(array_fractile(&values, 0.75), 6.0);
166        assert_eq!(array_madfm(&values), 2.0);
167    }
168
169    #[test]
170    fn near_wrappers_use_expected_default_tolerances() {
171        assert!(near(1.0, 1.0 + 5.0e-14));
172        assert!(!near(1.0, 1.0 + 5.0e-12));
173
174        assert!(near_f32(1.0, 1.0 + 5.0e-6));
175        assert!(!near_f32(1.0, 1.0 + 5.0e-4));
176    }
177
178    #[test]
179    fn near_tol_handles_special_cases() {
180        assert!(near_tol(3.0, 3.0, 0.0));
181        assert!(!near_tol(3.0, 4.0, 0.0));
182
183        assert!(near_tol(f64::INFINITY, f64::INFINITY, 1.0));
184
185        let tiny = f64::MIN_POSITIVE;
186        assert!(near_tol(0.0, tiny, 0.5));
187        assert!(near_tol(tiny, 0.0, 0.5));
188        assert!(!near_tol(0.0, 1.0e-100, 0.5));
189
190        assert!(!near_tol(1.0, -1.0, 0.1));
191        assert!(near_tol(10.0, 10.5, 0.05));
192        assert!(!near_tol(10.0, 10.6, 0.05));
193    }
194
195    #[test]
196    fn near_abs_uses_absolute_tolerance() {
197        assert!(near_abs(1.0, 1.049, 0.05));
198        assert!(!near_abs(1.0, 1.06, 0.05));
199    }
200}
