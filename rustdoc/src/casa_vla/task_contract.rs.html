18/// Stable protocol name advertised by `importvla --protocol-info`.
19pub const IMPORTVLA_TASK_PROTOCOL_NAME: &str = "casa_importvla_task";
20/// Stable protocol version advertised by `importvla --protocol-info`.
21pub const IMPORTVLA_TASK_PROTOCOL_VERSION: u32 = 1;
22
23/// Build the current shared `importvla` protocol descriptor.
24pub fn importvla_protocol_descriptor() -> ProviderProtocolDescriptor {
25    ProviderProtocolDescriptor::new(
26        IMPORTVLA_TASK_PROTOCOL_NAME,
27        IMPORTVLA_TASK_PROTOCOL_VERSION,
28        ProviderSurfaceKind::Task,
29        env!("CARGO_PKG_VERSION"),
30    )
31}

33/// Request for scanning VLA export archives without writing an MS.

40/// Request for importing VLA export archives into a MeasurementSet.
41#[derive(Debug, Clone, PartialEq, Serialize, Deserialize, JsonSchema)]
42pub struct ImportVlaImportTaskRequest {
43    /// Task-style import options.
44    pub options: ImportVlaOptions,
45}

47/// Canonical `importvla` task request envelope.
48#[derive(Debug, Clone, PartialEq, Serialize, Deserialize, JsonSchema)]
49#[serde(tag = "kind", content = "request", rename_all = "snake_case")]
50pub enum ImportVlaTaskRequest {
51    /// Scan one or more disk archives.
52    Scan(ImportVlaScanTaskRequest),
53    /// Import one or more disk archives into an MS.
54    Import(ImportVlaImportTaskRequest),
55}
56
57impl ImportVlaTaskRequest {
58    /// Execute the request and return the canonical task result envelope.
59    pub fn execute(&self) -> Result<ImportVlaTaskResult, String> {
60        match self {
61            Self::Scan(request) => scan_disk_archive_files_from_options(&request.options)
62                .map(ImportVlaTaskResult::Scan)
63                .map_err(|error| error.to_string()),
64            Self::Import(request) => {
65                import_archive_files_to_measurement_set_from_options(&request.options)
66                    .map(ImportVlaTaskResult::Import)
67                    .map_err(|error| error.to_string())
68            }
69        }
70    }
71}

83/// Build the current request/result schema bundle with the shared envelope.
84pub fn importvla_task_schema_bundle() -> TaskProviderContract {
85    let request_schema = schema_for!(ImportVlaTaskRequest);
86    let result_schema = schema_for!(ImportVlaTaskResult);
87    TaskProviderContract {
88        protocol: importvla_protocol_descriptor(),
89        semantic: TaskSemanticContract {
90            request_schema: request_schema.clone(),
91            result_schema: result_schema.clone(),
92            operations: importvla_task_operations(),
93        },
94        components: merged_components([&request_schema, &result_schema]),
95        annotations: serde_json::json!({}),
96        projections: ProviderProjectionMetadata {
97            cli: Some(ProviderCliProjection {
98                machine_actions: ProviderCliMachineActions {
99                    json_schema: Some("--json-schema".to_string()),
100                    protocol_info: Some("--protocol-info".to_string()),
101                    json_run: Some("--json-run <SOURCE>".to_string()),
102                    session: None,
103                },
104            }),
105            python: None,
106        },
107        parameter_surfaces: vec![
108            builtin_surface_bundle("importvla")
109                .expect("built-in importvla parameter surface must remain valid"),
110        ],
111        domain_schemas: TaskProviderSchemas {
112            request_schema,
113            result_schema,
114            additional: NoAdditionalProviderSchemas {},
115        },
116    }
117}

132}

134#[cfg(test)]
135mod tests {
136    use super::*;
137
138    #[test]
139    fn schema_bundle_advertises_importvla_protocol_and_actions() {
140        let bundle = importvla_task_schema_bundle();
141        bundle.validate().expect("shared provider envelope");
142        assert_eq!(bundle.protocol.protocol_name, IMPORTVLA_TASK_PROTOCOL_NAME);
143        assert_eq!(
144            bundle.protocol.protocol_version,
145            IMPORTVLA_TASK_PROTOCOL_VERSION
146        );
147        assert_eq!(bundle.protocol.surface_kind, ProviderSurfaceKind::Task);
148        assert_eq!(bundle.semantic.operations.len(), 2);
149        assert_eq!(bundle.parameter_surfaces.len(), 1);
150        assert_eq!(bundle.parameter_surfaces[0].surface.id(), "importvla");
151        bundle.parameter_surfaces[0]
152            .validate()
153            .expect("embedded importvla parameter surface");
154        assert_eq!(
155            serde_json::to_value(&bundle).unwrap()["parameter_surfaces"]
156                .as_array()
157                .unwrap()
158                .len(),
159            1
160        );
161        assert_eq!(
162            bundle
163                .projections
164                .cli
165                .as_ref()
166                .expect("cli projection")
167                .machine_actions
168                .json_run
169                .as_deref(),
170            Some("--json-run <SOURCE>")
171        );
172    }
173
174    #[test]
175    fn canonical_surface_projects_presentation_form() {
176        let bundle = importvla_task_schema_bundle();
177        let form = casa_provider_contracts::project_ui_form(&bundle.parameter_surfaces[0]);
178        assert_eq!(form["command_id"], "importvla");
179        assert!(form.to_string().contains("--archivefiles"));
180    }
