30/// Input visibility column used for continuum subtraction.
31#[derive(Debug, Clone, Copy, Default, PartialEq, Eq, Serialize, Deserialize, JsonSchema)]
32pub enum ContinuumSubtractionDataColumn {
33    /// MAIN.DATA.
34    Data,
35    /// MAIN.CORRECTED_DATA.
36    #[default]
37    CorrectedData,
38}
39
40impl ContinuumSubtractionDataColumn {
41    fn visibility_column(self) -> VisibilityDataColumn {
42        match self {
43            Self::Data => VisibilityDataColumn::Data,
44            Self::CorrectedData => VisibilityDataColumn::CorrectedData,
45        }
46    }
47
48    fn name(self) -> &'static str {
49        self.visibility_column().name()
50    }
51}

53/// Request to produce a continuum-subtracted MeasurementSet.
54#[derive(Debug, Clone)]
55pub struct ContinuumSubtractionRequest {
56    /// Input MeasurementSet root path.
57    pub input_ms: PathBuf,
58    /// Output MeasurementSet root path.
59    pub output_ms: PathBuf,
60    /// CASA-style line-free channel selector, e.g. `0:0~500;900~1919`.
61    pub fit_spw: String,
62    /// Polynomial order fitted independently to real and imaginary visibilities.
63    pub fit_order: usize,
64    /// Input data column to subtract.
65    pub data_column: ContinuumSubtractionDataColumn,
66    /// Optional row selection to materialize in the output MS.
67    pub selection: MsSelection,
68}

70/// Report returned after UV continuum subtraction.
71#[derive(Debug, Clone, PartialEq, Serialize, Deserialize, JsonSchema)]
72pub struct ContinuumSubtractionReport {
73    /// Input MeasurementSet root path.
74    pub input_ms: PathBuf,
75    /// Output MeasurementSet root path.
76    pub output_ms: PathBuf,
77    /// Number of MAIN rows selected for output.
78    pub row_count: usize,
79    /// Number of rows with at least one fitted correlation.
80    pub fitted_row_count: usize,
81    /// Number of per-correlation fits that had too few unflagged line-free samples.
82    pub skipped_fit_count: usize,
83    /// Source column read from the input MS.
84    pub source_column: String,
85    /// Output visibility data column populated for line imaging.
86    pub output_column: String,
87    /// CASA-style line-free channel selector used for fitting.
88    pub fit_spw: String,
89    /// Polynomial order used for the row/correlation fits.
90    pub fit_order: usize,
91    /// Spectral-window ids touched by the request.
92    pub spectral_window_ids: Vec<i32>,
93    /// End-to-end runtime in nanoseconds.
94    pub elapsed_ns: u64,
95}

97/// Errors returned by UV continuum subtraction.
98#[derive(Debug, Error)]
99pub enum ContinuumSubtractionError {
100    /// Opening an MS failed.
101    #[error("failed to open MeasurementSet {path}: {source}")]
102    OpenMeasurementSet {
103        /// Path being opened.
104        path: String,
105        /// Underlying MS error.
106        #[source]
107        source: MsError,
108    },
109
110    /// Input and output roots are identical.
111    #[error("input and output MeasurementSet paths must differ: {path}")]
112    SameInputOutput {
113        /// Duplicated path.
114        path: String,
115    },
116
117    /// The requested data column is absent.
118    #[error("MeasurementSet {path} does not contain {column}")]
119    MissingDataColumn {
120        /// Input MS path.
121        path: String,
122        /// Missing column.
123        column: String,
124    },
125
126    /// The row selection failed.
127    #[error("failed to select rows from MeasurementSet {path}: {source}")]
128    SelectRows {
129        /// Input MS path.
130        path: String,
131        /// Underlying MS error.
132        #[source]
133        source: MsError,
134    },
135
136    /// The row selection was empty.
137    #[error("continuum-subtraction selection produced no rows for {path}")]
138    EmptySelection {
139        /// Input MS path.
140        path: String,
141    },
142
143    /// Parsing or resolving `fit_spw` failed.
144    #[error("invalid line-free channel selection {selector:?}: {reason}")]
145    InvalidFitSpw {
146        /// CASA-style selector.
147        selector: String,
148        /// Human-readable reason.
149        reason: String,
150    },
151
152    /// Required spectral metadata is missing or inconsistent.
153    #[error("missing or invalid spectral metadata in {path}: {reason}")]
154    SpectralMetadata {
155        /// Input MS path.
156        path: String,
157        /// Human-readable reason.
158        reason: String,
159    },
160
161    /// MAIN-table data/flag mutation failed.
162    #[error("failed to continuum-subtract MAIN data for {path}: {source}")]
163    MutateMeasurementSet {
164        /// MS path.
165        path: String,
166        /// Underlying table error.
167        #[source]
168        source: Box<TableError>,
169    },
170
171    /// Output filesystem preparation failed.
172    #[error("failed to prepare output MeasurementSet {path}: {reason}")]
173    PrepareOutput {
174        /// Output path.
175        path: String,
176        /// Error context.
177        reason: String,
178    },
179
180    /// Saving the output MS failed.
181    #[error("failed to save MeasurementSet {path}: {source}")]
182    SaveMeasurementSet {
183        /// Output path.
184        path: String,
185        /// Underlying MS error.
186        #[source]
187        source: MsError,
188    },
189}

191/// Create a continuum-subtracted MS whose output `DATA` column contains line residuals.
192pub fn continuum_subtract(
193    request: &ContinuumSubtractionRequest,
194) -> Result<ContinuumSubtractionReport, ContinuumSubtractionError> {
195    if request.input_ms == request.output_ms {
196        return Err(ContinuumSubtractionError::SameInputOutput {
197            path: request.input_ms.display().to_string(),
198        });
199    }
200    let target =
201        MeasurementSetCreateTarget::prepare(&request.output_ms, true).map_err(|error| {
202            ContinuumSubtractionError::PrepareOutput {
203                path: request.output_ms.display().to_string(),
204                reason: error.to_string(),
205            }
206        })?;
207    let mut physical_request = request.clone();
208    physical_request.output_ms = target.staging_path().to_path_buf();
209    let mut report = continuum_subtract_staged(&physical_request)?;
210    target
211        .commit()
212        .map_err(|error| ContinuumSubtractionError::PrepareOutput {
213            path: request.output_ms.display().to_string(),
214            reason: error.to_string(),
215        })?;
216    report.output_ms = request.output_ms.clone();
217    Ok(report)
218}

220fn continuum_subtract_staged(
221    request: &ContinuumSubtractionRequest,
222) -> Result<ContinuumSubtractionReport, ContinuumSubtractionError> {
223    let started_at = Instant::now();
224    if request.fit_order > 3 {
225        return Err(ContinuumSubtractionError::InvalidFitSpw {
226            selector: request.fit_spw.clone(),
227            reason: "fit_order above 3 is not supported by the initial uvcontsub path".to_string(),
228        });
229    }
230
231    let input = MeasurementSet::open(&request.input_ms).map_err(|source| {
232        ContinuumSubtractionError::OpenMeasurementSet {
233            path: request.input_ms.display().to_string(),
234            source,
235        }
236    })?;
237    let source_column = request.data_column.visibility_column();
238    if !input
239        .main_table()
240        .schema()
241        .is_some_and(|schema| schema.contains_column(source_column.name()))
242    {
243        return Err(ContinuumSubtractionError::MissingDataColumn {
244            path: request.input_ms.display().to_string(),
245            column: source_column.name().to_string(),
246        });
247    }
248    let input_row_count = input.row_count();
249    let selected_rows = resolve_calibration_selection(&input, &request.selection)
250        .map_err(|error| ContinuumSubtractionError::SelectRows {
251            path: request.input_ms.display().to_string(),
252            source: match error {
253                casa_ms::MsSelectionError::Domain(source) => source,
254                other => MsError::InvalidInput(other.to_string()),
255            },
256        })?
257        .row_indices()
258        .collect::<Vec<_>>();
259    if selected_rows.is_empty() {
260        return Err(ContinuumSubtractionError::EmptySelection {
261            path: request.input_ms.display().to_string(),
262        });
263    }
264    let fit_channels_by_spw = resolve_fit_channels(&input, &request.fit_spw)?;
265    drop(input);
266
267    let full_selection = selected_rows.len() == input_row_count;
268    if full_selection {
269        prepare_output_root(&request.output_ms)?;
270        copy_dir_recursive(&request.input_ms, &request.output_ms).map_err(|error| {
271            ContinuumSubtractionError::PrepareOutput {
272                path: request.output_ms.display().to_string(),
273                reason: error.to_string(),
274            }
275        })?;
276    } else {
277        mstransform(&MsTransformRequest {
278            input_ms: request.input_ms.clone(),
279            output_ms: request.output_ms.clone(),
280            spw: String::new(),
281            width: 1,
282            data_column: match request.data_column {
283                ContinuumSubtractionDataColumn::Data => TransformDataColumn::Data,
284                ContinuumSubtractionDataColumn::CorrectedData => TransformDataColumn::CorrectedData,
285            },
286            selection: request.selection.clone(),
287            keep_flags: true,
288        })
289        .map_err(|error| ContinuumSubtractionError::SaveMeasurementSet {
290            path: request.output_ms.display().to_string(),
291            source: MsError::InvalidInput(error.to_string()),
292        })?;
293    }
294
295    let mut output = MeasurementSet::open(&request.output_ms).map_err(|source| {
296        ContinuumSubtractionError::OpenMeasurementSet {
297            path: request.output_ms.display().to_string(),
298            source,
299        }
300    })?;
301    let processing_source_column = if full_selection {
302        source_column
303    } else {
304        VisibilityDataColumn::Data
305    };
306    let output_ddid_to_spw = data_description_spw_map(&output)?;
307    let row_indices = (0..output.row_count()).collect::<Vec<_>>();
308    let maximum_data_bytes = maximum_visibility_cell_elements(&output)
309        .and_then(|elements| {
310            elements
311                .checked_mul(8)
312                .ok_or(casa_ms::MeasurementSetWriteError::ByteOverflow)
313        })
314        .map_err(|error| ContinuumSubtractionError::MutateMeasurementSet {
315            path: request.output_ms.display().to_string(),
316            source: Box::new(TableError::Storage(error.to_string())),
317        })?;
318    let resources = MeasurementSetWriteResources::from_system_memory(2).map_err(|error| {
319        ContinuumSubtractionError::MutateMeasurementSet {
320            path: request.output_ms.display().to_string(),
321            source: Box::new(TableError::Storage(error.to_string())),
322        }
323    })?;
324    let write_plan = MeasurementSetWritePlan::selected_row_mutation(
325        row_indices,
326        vec![MeasurementSetWriteColumnPlan {
327            name: VisibilityDataColumn::Data.name().to_string(),
328            bytes_per_row: maximum_data_bytes,
329            mode: MeasurementSetColumnWriteMode::Replace,
330            storage_manager: MeasurementSetColumnStorage::Persisted,
331            tile_shape: None,
332            create_source_column: None,
333        }],
334        resources,
335    )
336    .map_err(|error| ContinuumSubtractionError::MutateMeasurementSet {
337        path: request.output_ms.display().to_string(),
338        source: Box::new(TableError::Storage(error.to_string())),
339    })?;
340    let mut write_session =
341        MeasurementSetWriteSession::start_selected_row_mutation(&mut output, write_plan).map_err(
342            |error| ContinuumSubtractionError::MutateMeasurementSet {
343                path: request.output_ms.display().to_string(),
344                source: Box::new(TableError::Storage(error.to_string())),
345            },
346        )?;
347
348    let mut fitted_row_count = 0usize;
349    let mut skipped_fit_count = 0usize;
350    let mut touched_spws = BTreeSet::new();
351    loop {
352        let rows = write_session
353            .next_mutation_rows()
354            .map_err(|error| ContinuumSubtractionError::MutateMeasurementSet {
355                path: request.output_ms.display().to_string(),
356                source: Box::new(TableError::Storage(error.to_string())),
357            })?
358            .to_vec();
359        if rows.is_empty() {
360            break;
361        }
362        let data_values = output
363            .main_table()
364            .column_accessor(processing_source_column.name())
365            .and_then(|column| column.array_cells_owned(&rows))
366            .map_err(|source| ContinuumSubtractionError::MutateMeasurementSet {
367                path: request.output_ms.display().to_string(),
368                source: Box::new(source),
369            })?;
370        let flag_values = output
371            .main_table()
372            .column_accessor("FLAG")
373            .and_then(|column| column.array_cells_owned(&rows))
374            .map_err(|source| ContinuumSubtractionError::MutateMeasurementSet {
375                path: request.output_ms.display().to_string(),
376                source: Box::new(source),
377            })?;
378        let weight_values = output
379            .main_table()
380            .column_accessor("WEIGHT")
381            .and_then(|column| column.array_cells_owned(&rows))
382            .map_err(|source| ContinuumSubtractionError::MutateMeasurementSet {
383                path: request.output_ms.display().to_string(),
384                source: Box::new(source),
385            })?;
386        let weight_spectrum_values = if output
387            .main_table()
388            .schema()
389            .is_some_and(|schema| schema.contains_column("WEIGHT_SPECTRUM"))
390        {
391            Some(
392                output
393                    .main_table()
394                    .column_accessor("WEIGHT_SPECTRUM")
395                    .and_then(|column| column.array_cells_owned(&rows))
396                    .map_err(|source| ContinuumSubtractionError::MutateMeasurementSet {
397                        path: request.output_ms.display().to_string(),
398                        source: Box::new(source),
399                    })?,
400            )
401        } else {
402            None
403        };
404        let flag_row_values = output
405            .main_table()
406            .column_accessor("FLAG_ROW")
407            .and_then(|column| column.scalar_cells_owned_for_rows(&rows))
408            .map_err(|source| ContinuumSubtractionError::MutateMeasurementSet {
409                path: request.output_ms.display().to_string(),
410                source: Box::new(source),
411            })?;
412        let data_desc_ids = output
413            .main_table()
414            .column_accessor("DATA_DESC_ID")
415            .and_then(|column| column.scalar_cells_owned_for_rows(&rows))
416            .map_err(|source| ContinuumSubtractionError::MutateMeasurementSet {
417                path: request.output_ms.display().to_string(),
418                source: Box::new(source),
419            })?;
420        let mut weight_spectrum_values = weight_spectrum_values.map(Vec::into_iter);
421        let mut residuals = Vec::with_capacity(rows.len());
422        for (((((row_index, data), flags), weights), flag_row), ddid) in rows
423            .iter()
424            .copied()
425            .zip(data_values)
426            .zip(flag_values)
427            .zip(weight_values)
428            .zip(flag_row_values)
429            .zip(data_desc_ids)
430        {
431            let data = data.ok_or_else(|| {
432                missing_column_error(
433                    &request.output_ms,
434                    row_index,
435                    processing_source_column.name(),
436                )
437            })?;
438            let flags =
439                flags.ok_or_else(|| missing_column_error(&request.output_ms, row_index, "FLAG"))?;
440            let weights = weights
441                .ok_or_else(|| missing_column_error(&request.output_ms, row_index, "WEIGHT"))?;
442            let weight_spectrum = weight_spectrum_values
443                .as_mut()
444                .map(|values| {
445                    values.next().flatten().ok_or_else(|| {
446                        missing_column_error(&request.output_ms, row_index, "WEIGHT_SPECTRUM")
447                    })
448                })
449                .transpose()?;
450            let flag_row = scalar_bool(flag_row.as_ref(), "FLAG_ROW", row_index)?;
451            let ddid = scalar_i32(ddid.as_ref(), "DATA_DESC_ID", row_index)?;
452            let spw_id = output_ddid_to_spw.get(&ddid).copied().ok_or_else(|| {
453                ContinuumSubtractionError::SpectralMetadata {
454                    path: request.output_ms.display().to_string(),
455                    reason: format!("MAIN row {row_index} references DATA_DESC_ID {ddid}, which has no DATA_DESCRIPTION row"),
456                }
457            })?;
458            let fit = fit_channels_by_spw.get(&spw_id).ok_or_else(|| {
459                ContinuumSubtractionError::InvalidFitSpw {
460                    selector: request.fit_spw.clone(),
461                    reason: format!(
462                        "selected row {row_index} maps to spectral window {spw_id}, but fit_spw does not include that SPW"
463                    ),
464                }
465            })?;
466            touched_spws.insert(spw_id);
467            let (residual, fitted, skipped) = subtract_row(SubtractRowRequest {
468                data,
469                flags,
470                weights,
471                weight_spectrum,
472                flag_row,
473                fit,
474                fit_order: request.fit_order,
475                row_index,
476            })?;
477            fitted_row_count += usize::from(fitted);
478            skipped_fit_count += skipped;
479            residuals.push(residual);
480        }
481        write_session
482            .write_mutation_batch(
483                &mut output,
484                MeasurementSetMutationBatch {
485                    row_indices: rows,
486                    columns: vec![MeasurementSetMutationColumnBatch {
487                        name: VisibilityDataColumn::Data.name().to_string(),
488                        values: MeasurementSetMutationColumnValues::Arrays(residuals),
489                    }],
490                },
491            )
492            .map_err(|error| ContinuumSubtractionError::MutateMeasurementSet {
493                path: request.output_ms.display().to_string(),
494                source: Box::new(TableError::Storage(error.to_string())),
495            })?;
496    }
497    write_session.finish_mutation().map_err(|error| {
498        ContinuumSubtractionError::MutateMeasurementSet {
499            path: request.output_ms.display().to_string(),
500            source: Box::new(TableError::Storage(error.to_string())),
501        }
502    })?;
503
504    Ok(ContinuumSubtractionReport {
505        input_ms: request.input_ms.clone(),
506        output_ms: request.output_ms.clone(),
507        row_count: output.row_count(),
508        fitted_row_count,
509        skipped_fit_count,
510        source_column: request.data_column.name().to_string(),
511        output_column: VisibilityDataColumn::Data.name().to_string(),
512        fit_spw: request.fit_spw.clone(),
513        fit_order: request.fit_order,
514        spectral_window_ids: touched_spws.into_iter().collect(),
515        elapsed_ns: started_at.elapsed().as_nanos() as u64,
516    })
517}

519fn resolve_fit_channels(
520    ms: &MeasurementSet,
521    fit_spw: &str,
522) -> Result<BTreeMap<i32, FitChannels>, ContinuumSubtractionError> {
523    let selectors =
524        parse_spw_selector(fit_spw).map_err(|source| ContinuumSubtractionError::InvalidFitSpw {
525            selector: fit_spw.to_string(),
526            reason: source.to_string(),
527        })?;
528    let spw =
529        ms.spectral_window()
530            .map_err(|source| ContinuumSubtractionError::SpectralMetadata {
531                path: "<measurement-set/SPECTRAL_WINDOW>".to_string(),
532                reason: source.to_string(),
533            })?;
534    let mut by_spw = BTreeMap::new();
535    for selector in selectors {
536        if selector.spw_id < 0 {
537            return Err(ContinuumSubtractionError::InvalidFitSpw {
538                selector: fit_spw.to_string(),
539                reason: format!("negative spectral-window id {}", selector.spw_id),
540            });
541        }
542        let row = selector.spw_id as usize;
543        if row >= spw.row_count() {
544            return Err(ContinuumSubtractionError::InvalidFitSpw {
545                selector: fit_spw.to_string(),
546                reason: format!(
547                    "spectral-window id {} is outside SPECTRAL_WINDOW with {} rows",
548                    selector.spw_id,
549                    spw.row_count()
550                ),
551            });
552        }
553        let frequencies_hz =
554            spw.chan_freq(row)
555                .map_err(|source| ContinuumSubtractionError::SpectralMetadata {
556                    path: "<measurement-set/SPECTRAL_WINDOW>".to_string(),
557                    reason: source.to_string(),
558                })?;
559        if frequencies_hz.is_empty() {
560            return Err(ContinuumSubtractionError::SpectralMetadata {
561                path: "<measurement-set/SPECTRAL_WINDOW>".to_string(),
562                reason: format!(
563                    "spectral-window {} has no channel frequencies",
564                    selector.spw_id
565                ),
566            });
567        }
568        let indices = match selector.channels {
569            Some(ChannelSelection { segments }) => ChannelSelection { segments }
570                .indices(frequencies_hz.len())
571                .map_err(|source| ContinuumSubtractionError::InvalidFitSpw {
572                    selector: fit_spw.to_string(),
573                    reason: source.to_string(),
574                })?,
575            None => (0..frequencies_hz.len()).collect(),
576        };
577        if indices.is_empty() {
578            return Err(ContinuumSubtractionError::InvalidFitSpw {
579                selector: fit_spw.to_string(),
580                reason: format!(
581                    "spectral-window {} selection produced no channels",
582                    selector.spw_id
583                ),
584            });
585        }
586        by_spw.insert(
587            selector.spw_id,
588            FitChannels {
589                indices,
590                frequencies_hz,
591            },
592        );
593    }
594    Ok(by_spw)
595}

597fn data_description_spw_map(
598    ms: &MeasurementSet,
599) -> Result<BTreeMap<i32, i32>, ContinuumSubtractionError> {
600    let data_description =
601        ms.data_description()
602            .map_err(|source| ContinuumSubtractionError::SpectralMetadata {
603                path: "<measurement-set/DATA_DESCRIPTION>".to_string(),
604                reason: source.to_string(),
605            })?;
606    let mut map = BTreeMap::new();
607    for row in 0..data_description.row_count() {
608        let spw = data_description.spectral_window_id(row).map_err(|source| {
609            ContinuumSubtractionError::SpectralMetadata {
610                path: "<measurement-set/DATA_DESCRIPTION>".to_string(),
611                reason: source.to_string(),
612            }
613        })?;
614        map.insert(row as i32, spw);
615    }
616    Ok(map)
617}

619struct SubtractRowRequest<'a> {
620    data: ArrayValue,
621    flags: ArrayValue,
622    weights: ArrayValue,
623    weight_spectrum: Option<ArrayValue>,
624    flag_row: bool,
625    fit: &'a FitChannels,
626    fit_order: usize,
627    row_index: usize,
628}
629
630fn subtract_row(
631    request: SubtractRowRequest<'_>,
632) -> Result<(ArrayValue, bool, usize), ContinuumSubtractionError> {
633    let SubtractRowRequest {
634        data,
635        flags,
636        weights,
637        weight_spectrum,
638        flag_row,
639        fit,
640        fit_order,
641        row_index,
642    } = request;
643    let ArrayValue::Complex32(values) = data else {
644        return Err(ContinuumSubtractionError::SpectralMetadata {
645            path: "<measurement-set/DATA>".to_string(),
646            reason: format!("row {row_index} DATA must be Complex32 array"),
647        });
648    };
649    let shape = values.shape().to_vec();
650    if shape.len() != 2 {
651        return Err(ContinuumSubtractionError::SpectralMetadata {
652            path: "<measurement-set/DATA>".to_string(),
653            reason: format!("row {row_index} DATA must be rank-2 [corr, chan], found {shape:?}"),
654        });
655    }
656    let ArrayValue::Bool(flag_values) = flags else {
657        return Err(ContinuumSubtractionError::SpectralMetadata {
658            path: "<measurement-set/FLAG>".to_string(),
659            reason: format!("row {row_index} FLAG must be Bool array"),
660        });
661    };
662    if flag_values.shape() != values.shape() {
663        return Err(ContinuumSubtractionError::SpectralMetadata {
664            path: "<measurement-set/FLAG>".to_string(),
665            reason: format!(
666                "row {row_index} FLAG shape {:?} does not match DATA shape {:?}",
667                flag_values.shape(),
668                values.shape()
669            ),
670        });
671    }
672    let num_corr = shape[0];
673    let num_chan = shape[1];
674    let ArrayValue::Float32(weight_values) = weights else {
675        return Err(ContinuumSubtractionError::SpectralMetadata {
676            path: "<measurement-set/WEIGHT>".to_string(),
677            reason: format!("row {row_index} WEIGHT must be Float32 array"),
678        });
679    };
680    if weight_values.shape() != [num_corr] {
681        return Err(ContinuumSubtractionError::SpectralMetadata {
682            path: "<measurement-set/WEIGHT>".to_string(),
683            reason: format!(
684                "row {row_index} WEIGHT shape {:?} does not match correlation count {num_corr}",
685                weight_values.shape()
686            ),
687        });
688    }
689    let weight_spectrum_values = match weight_spectrum {
690        Some(ArrayValue::Float32(values)) => {
691            if values.shape() != [num_corr, num_chan] {
692                return Err(ContinuumSubtractionError::SpectralMetadata {
693                    path: "<measurement-set/WEIGHT_SPECTRUM>".to_string(),
694                    reason: format!(
695                        "row {row_index} WEIGHT_SPECTRUM shape {:?} does not match DATA shape {:?}",
696                        values.shape(),
697                        [num_corr, num_chan]
698                    ),
699                });
700            }
701            Some(values)
702        }
703        Some(_) => {
704            return Err(ContinuumSubtractionError::SpectralMetadata {
705                path: "<measurement-set/WEIGHT_SPECTRUM>".to_string(),
706                reason: format!("row {row_index} WEIGHT_SPECTRUM must be Float32 array"),
707            });
708        }
709        None => None,
710    };
711    if fit.frequencies_hz.len() != num_chan {
712        return Err(ContinuumSubtractionError::SpectralMetadata {
713            path: "<measurement-set/SPECTRAL_WINDOW>".to_string(),
714            reason: format!(
715                "row {row_index} DATA has {num_chan} channels but its SPECTRAL_WINDOW has {} frequencies",
716                fit.frequencies_hz.len()
717            ),
718        });
719    }
720
721    let mut residual = values;
722    if flag_row {
723        return Ok((ArrayValue::Complex32(residual), false, num_corr));
724    }
725    let x_scale = normalized_frequency_axis(&fit.frequencies_hz);
726    let mut fitted_any = false;
727    let mut skipped = 0usize;
728    for corr in 0..num_corr {
729        let mut samples = Vec::with_capacity(fit.indices.len());
730        for &chan in &fit.indices {
731            if chan >= num_chan {
732                return Err(ContinuumSubtractionError::InvalidFitSpw {
733                    selector: format!("{:?}", fit.indices),
734                    reason: format!(
735                        "fit channel {chan} exceeds row {row_index} with {num_chan} channels"
736                    ),
737                });
738            }
739            if !flag_values[[corr, chan]] {
740                let weight = weight_spectrum_values
741                    .as_ref()
742                    .map(|values| values[[corr, chan]])
743                    .unwrap_or_else(|| weight_values[[corr]]);
744                if weight.is_finite() && weight > 0.0 {
745                    samples.push((x_scale[chan], residual[[corr, chan]], weight as f64));
746                }
747            }
748        }
749        if samples.len() < fit_order + 1 {
750            skipped += 1;
751            continue;
752        }
753        let real_coeffs = fit_polynomial(
754            samples.iter().map(|(x, y, weight)| (*x, y.re as f64, *weight)),
755            fit_order,
756        )
757        .ok_or_else(|| ContinuumSubtractionError::InvalidFitSpw {
758            selector: format!("{:?}", fit.indices),
759            reason: format!("line-free channels for row {row_index} correlation {corr} are singular for order {fit_order}"),
760        })?;
761        let imag_coeffs = fit_polynomial(
762            samples.iter().map(|(x, y, weight)| (*x, y.im as f64, *weight)),
763            fit_order,
764        )
765        .ok_or_else(|| ContinuumSubtractionError::InvalidFitSpw {
766            selector: format!("{:?}", fit.indices),
767            reason: format!("line-free channels for row {row_index} correlation {corr} are singular for order {fit_order}"),
768        })?;
769        for chan in 0..num_chan {
770            let continuum = Complex32::new(
771                evaluate_polynomial(&real_coeffs, x_scale[chan]) as f32,
772                evaluate_polynomial(&imag_coeffs, x_scale[chan]) as f32,
773            );
774            residual[[corr, chan]] -= continuum;
775        }
776        fitted_any = true;
777    }
778
779    Ok((ArrayValue::Complex32(residual), fitted_any, skipped))
780}
781
782fn normalized_frequency_axis(frequencies_hz: &[f64]) -> Vec<f64> {
783    let mean = frequencies_hz.iter().sum::<f64>() / frequencies_hz.len() as f64;
784    let max_abs = frequencies_hz
785        .iter()
786        .map(|frequency| (frequency - mean).abs())
787        .fold(0.0_f64, f64::max);
788    let scale = if max_abs > 0.0 { max_abs } else { 1.0 };
789    frequencies_hz
790        .iter()
791        .map(|frequency| (frequency - mean) / scale)
792        .collect()
793}
794
795fn fit_polynomial(
796    samples: impl IntoIterator<Item = (f64, f64, f64)>,
797    order: usize,
798) -> Option<Vec<f64>> {
799    let size = order + 1;
800    let mut rows = Vec::new();
801    for (x, y, weight) in samples {
802        let mut powers = vec![1.0_f64; size];
803        for index in 1..size {
804            powers[index] = powers[index - 1] * x;
805        }
806        rows.push((powers, y, weight));
807    }
808    solve_weighted_least_squares(&rows, size)
809}
810
811fn evaluate_polynomial(coeffs: &[f64], x: f64) -> f64 {
812    coeffs
813        .iter()
814        .rev()
815        .fold(0.0_f64, |accumulator, coefficient| {
816            accumulator * x + coefficient
817        })
818}

864fn prepare_output_root(path: &Path) -> Result<(), ContinuumSubtractionError> {
865    if path.exists() {
866        fs::remove_dir_all(path)
867            .or_else(|_| fs::remove_file(path))
868            .map_err(|error| ContinuumSubtractionError::PrepareOutput {
869                path: path.display().to_string(),
870                reason: error.to_string(),
871            })?;
872    }
873    if let Some(parent) = path.parent() {
874        fs::create_dir_all(parent).map_err(|error| ContinuumSubtractionError::PrepareOutput {
875            path: path.display().to_string(),
876            reason: error.to_string(),
877        })?;
878    }
879    Ok(())
880}
881
882fn copy_dir_recursive(source: &Path, destination: &Path) -> std::io::Result<()> {
883    fs::create_dir_all(destination)?;
884    for entry in fs::read_dir(source)? {
885        let entry = entry?;
886        let source_path = entry.path();
887        let destination_path = destination.join(entry.file_name());
888        let file_type = entry.file_type()?;
889        if file_type.is_dir() {
890            copy_dir_recursive(&source_path, &destination_path)?;
891        } else if file_type.is_file() {
892            fs::copy(&source_path, &destination_path)?;
893        } else if file_type.is_symlink() {
894            let target = fs::read_link(&source_path)?;
895            std::os::unix::fs::symlink(target, destination_path)?;
896        }
897    }
898    Ok(())
899}
