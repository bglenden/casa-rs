6fn compile_prepared_row_slots(
7    table: &Table,
8    columns: &[&str],
9) -> Result<
10    (
11        Vec<PreparedRowSlot>,
12        std::collections::HashMap<String, usize>,
13    ),
14    TableError,
15> {
16    let schema = table
17        .schema()
18        .ok_or(TableError::PreparedRowRequiresSchema)?;
19    let mut slots = Vec::with_capacity(columns.len());
20    let mut column_indices = std::collections::HashMap::with_capacity(columns.len());
21
22    for &column in columns {
23        table.require_column(column)?;
24        let schema_column = schema
25            .columns()
26            .iter()
27            .find(|candidate| candidate.name() == column)
28            .ok_or_else(|| TableError::SchemaColumnUnknown {
29                column: column.to_string(),
30            })?;
31        let kind = match schema_column.column_type() {
32            ColumnType::Scalar => PreparedRowSlotKind::Scalar,
33            ColumnType::Array(_) => PreparedRowSlotKind::Array,
34            ColumnType::Record => {
35                return Err(TableError::PreparedRowRecordColumnUnsupported {
36                    column: column.to_string(),
37                });
38            }
39        };
40        let slot_index = slots.len();
41        slots.push(PreparedRowSlot {
42            column: column.to_string(),
43            kind,
44        });
45        column_indices
46            .entry(column.to_string())
47            .or_insert(slot_index);
48    }
49
50    Ok((slots, column_indices))
51}

69fn load_prepared_row_value(
70    table: &Table,
71    row_index: usize,
72    slot: &PreparedRowSlot,
73) -> Result<Value, TableError> {
74    match slot.kind {
75        PreparedRowSlotKind::Scalar => table
76            .get_scalar_cell(row_index, &slot.column)
77            .map(|value| Value::Scalar(value.clone())),
78        PreparedRowSlotKind::Array => table
79            .get_array_cell(row_index, &slot.column)
80            .map(|value| Value::Array(value.clone())),
81    }

85    table: &Table,
86    slots: &[PreparedRowSlot],
87    row: &mut RecordValue,
88    row_index: usize,
89) -> Result<(), TableError> {

104fn flush_prepared_row_buffer(
105    table: &mut Table,
106    slots: &[PreparedRowSlot],
107    row: &RecordValue,
108    row_index: usize,
109) -> Result<(), TableError> {
110    for (slot, field) in slots.iter().zip(row.fields().iter()) {
111        match (slot.kind, &field.value) {
112            (PreparedRowSlotKind::Scalar, Value::Scalar(value)) => {
113                table.set_scalar_cell_assuming_valid(row_index, &slot.column, value.clone())?;
114            }
115            (PreparedRowSlotKind::Array, Value::Array(value)) => {
116                table.set_array_cell_assuming_valid(row_index, &slot.column, value.clone())?;
117            }
118            (PreparedRowSlotKind::Scalar, value) => {
119                return Err(TableError::ColumnTypeMismatch {
120                    row_index,
121                    column: slot.column.clone(),
122                    expected: "scalar",
123                    found: value.kind(),
124                });
125            }
126            (PreparedRowSlotKind::Array, value) => {
127                return Err(TableError::ColumnTypeMismatch {
128                    row_index,
129                    column: slot.column.clone(),
130                    expected: "array",
131                    found: value.kind(),
132                });
133            }
134        }
135    }
136    Ok(())
137}

148    /// Attaches or replaces the schema, validating all existing rows.
149    ///
150    /// If the new schema is incompatible with any existing row the schema is
151    /// not updated and the original is restored. Returns the first
152    /// [`TableError`] encountered during validation.
153    pub fn set_schema(&mut self, schema: TableSchema) -> Result<(), TableError> {
154        let auto_unlock = self.begin_write_operation("set_schema")?;
155        let result = (|| {
156            let previous = self.inner.schema().cloned();
157            self.inner.set_schema(Some(schema));
158            if let Err(err) = self.validate() {
159                self.inner.set_schema(previous);
160                return Err(err);
161            }
162            let undefined =
163                collect_undefined_cells_for_schema(self.rows()?, self.schema().unwrap());
164            self.inner
165                .undefined_cells_mut()?
166                .clone_from_slice(&undefined);
167            Ok(())
168        })();
169        self.finish_write_operation(auto_unlock, result)
170    }

181    /// Returns `Ok(())` immediately when no schema is attached. Otherwise,

269    /// Appends a row to the table.
270    ///
271    /// If a schema is attached, the row is validated before insertion.
272    /// Returns [`TableError`] if the row violates the schema; the table is
273    /// left unchanged in that case.
274    pub fn add_row(&mut self, row: RecordValue) -> Result<(), TableError> {
275        let auto_unlock = self.begin_write_operation("add_row")?;
276        let result = (|| {
277            let mut undefined = None;
278            if let Some(schema) = self.schema() {
279                validate_row_against_schema(self.row_count(), &row, schema)?;
280                undefined = Some(undefined_columns_for_row(&row, schema));
281            }
282            self.inner.add_row(row)?;
283            if let Some(undefined) = undefined {
284                if let Some(set) = self.inner.undefined_for_row_mut(self.row_count() - 1)? {
285                    *set = undefined;
286                }
287            }
288            Ok(())
289        })();
290        self.finish_write_operation(auto_unlock, result)
291    }

294    ///

296    /// the current schema because it was synthesized directly from the schema
297    /// or validated earlier in the same write path. Undefined scalar-column
298    /// tracking is still updated when a schema is attached.
299    ///
300    /// Callers that are not certain the row is schema-valid should keep using
301    /// [`add_row`](Table::add_row).

306                .schema()
307                .map(|schema| undefined_columns_for_row(&row, schema));
308            self.inner.add_row(row)?;
309            if let Some(undefined) = undefined
310                && let Some(set) = self.inner.undefined_for_row_mut(self.row_count() - 1)?
311            {
312                *set = undefined;
313            }
314            Ok(())
315        })();
316        self.finish_write_operation(auto_unlock, result)
317    }

394    /// Returns the record value at `(row_index, column)`.
395    ///
396    /// When a schema is attached and the column is declared as a record column,
397    /// a missing cell is treated as an empty [`RecordValue`]. Without a schema,
398    /// an absent cell returns [`TableError::ColumnNotFound`].
399    ///
400    /// Returns [`TableError::SchemaColumnNotRecord`] if the schema declares the
401    /// column with a non-record type.
402    pub fn record_cell(&self, row_index: usize, column: &str) -> Result<RecordValue, TableError> {
403        self.require_row(row_index)?;
404        if let Some(schema) = self.schema() {
405            let column_schema =
406                schema
407                    .column(column)
408                    .ok_or_else(|| TableError::SchemaColumnUnknown {
409                        column: column.to_string(),
410                    })?;
411            if !matches!(column_schema.column_type(), ColumnType::Record) {
412                return Err(TableError::SchemaColumnNotRecord {
413                    column: column.to_string(),
414                });
415            }
416            return match self.cell(row_index, column)? {
417                Some(Value::Record(record)) => Ok(record.clone()),
418                Some(value) => Err(TableError::ColumnTypeMismatch {
419                    row_index,
420                    column: column.to_string(),
421                    expected: "record",
422                    found: value.kind(),
423                }),
424                None => Ok(RecordValue::default()),
425            };
426        }
427
428        match self.cell(row_index, column)? {
429            Some(Value::Record(record)) => Ok(record.clone()),
430            Some(value) => Err(TableError::ColumnTypeMismatch {
431                row_index,
432                column: column.to_string(),
433                expected: "record",
434                found: value.kind(),
435            }),
436            None => Err(TableError::ColumnNotFound {
437                row_index,
438                column: column.to_string(),
439            }),
440        }
441    }

470    ///
471    /// Each item is a [`RecordColumnCell`] whose `value` is always populated:
472    /// absent cells are defaulted to an empty [`RecordValue`] when the schema
473    /// permits it. Returns [`TableError`] if the column is unknown, not a
474    /// record column, or a cell has the wrong type.

476        &'a self,
477        column: &str,
478        row_range: RowRange,
479    ) -> Result<RecordColumnIter<'a>, TableError> {
480        self.require_column(column)?;
481        row_range.validate(self.row_count())?;
482
483        let default_missing = if let Some(schema) = self.schema() {
484            let column_schema =
485                schema
486                    .column(column)
487                    .ok_or_else(|| TableError::SchemaColumnUnknown {
488                        column: column.to_string(),
489                    })?;
490            if !matches!(column_schema.column_type(), ColumnType::Record) {
491                return Err(TableError::SchemaColumnNotRecord {
492                    column: column.to_string(),
493                });
494            }
495            true
496        } else {
497            false
498        };
499
500        let row_data = self.rows()?;
501        for row_index in row_range.iter() {
502            match row_data[row_index].get(column) {
503                Some(Value::Record(_)) => {}
504                Some(value) => {
505                    return Err(TableError::ColumnTypeMismatch {
506                        row_index,
507                        column: column.to_string(),
508                        expected: "record",
509                        found: value.kind(),
510                    });
511                }
512                None => {
513                    if !default_missing {
514                        return Err(TableError::ColumnNotFound {
515                            row_index,
516                            column: column.to_string(),
517                        });
518                    }
519                }
520            }
521        }
522
523        Ok(RecordColumnIter {
524            row_data,
525            column: column.to_string(),
526            rows: row_range.iter(),
527            default_missing,
528        })
529    }

546    /// Writes values from an iterator into `column` for the rows in `row_range`.
547    ///
548    /// The iterator must produce exactly as many values as there are rows in
549    /// `row_range`; otherwise [`TableError::ColumnWriteTooFewValues`] or
550    /// [`TableError::ColumnWriteTooManyValues`] is returned. Returns the
551    /// number of cells written on success.
552    pub fn put_column_range<I>(
553        &mut self,
554        column: &str,
555        row_range: RowRange,
556        values: I,
557    ) -> Result<usize, TableError>
558    where
559        I: IntoIterator<Item = Value>,
560    {
561        let auto_unlock = self.begin_write_operation("put_column_range")?;
562        let result = (|| {
563            self.require_column(column)?;
564            row_range.validate(self.row_count())?;
565
566            let expected = row_range.len();
567            let row_iter = row_range.iter();
568            let mut value_iter = values.into_iter();
569            let mut provided = 0usize;
570            for row_index in row_iter {
571                let Some(value) = value_iter.next() else {
572                    return Err(TableError::ColumnWriteTooFewValues { expected, provided });
573                };
574                self.set_cell_impl(row_index, column, value)?;
575                provided += 1;
576            }
577            if value_iter.next().is_some() {
578                return Err(TableError::ColumnWriteTooManyValues { expected });
579            }
580            Ok(provided)
581        })();
582        self.finish_write_operation(auto_unlock, result)
583    }

585    /// Returns `true` if the cell at `(row_index, column)` is considered defined.
586    ///
587    /// A cell is defined if a value is present in the row. For record columns
588    /// with a schema, a missing cell is still considered defined because it
589    /// defaults to an empty record. Returns [`TableError`] if `row_index` is
590    /// out of bounds or the column is unknown per the schema.
591    pub fn is_cell_defined(&self, row_index: usize, column: &str) -> Result<bool, TableError> {
592        self.require_row(row_index)?;
593        self.require_column(column)?;
594        if let Some(undefined) = self
595            .inner
596            .undefined_cells()?
597            .get(row_index)
598            .map(|set| set.contains(column))
599        {
600            if undefined {
601                return Ok(false);
602            }
603        }
604        if self.cell(row_index, column)?.is_some() {
605            return Ok(true);
606        }
607        if let Some(schema) = self.schema()
608            && matches!(
609                schema.column(column).map(ColumnSchema::column_type),
610                Some(ColumnType::Record)
611            )
612        {
613            return Ok(true);
614        }
615        Ok(false)
616    }

669    ) -> Result<(), TableError> {
670        let schema_column = if let Some(schema) = self.schema() {
671            Some(
672                schema
673                    .column(column)
674                    .ok_or_else(|| TableError::SchemaColumnUnknown {
675                        column: column.to_string(),
676                    })?
677                    .clone(),
678            )
679        } else {
680            None
681        };
682        if let Some(column_schema) = &schema_column {
683            validate_cell_against_schema_column(row_index, column_schema, Some(&value))?;
684        }
685
686        {
687            if let Some(set) = self.inner.undefined_for_row_mut(row_index)? {
688                set.remove(column);
689            }
690        }
691
692        let row = self.row_mut(row_index)?;
693
694        if schema_column.is_some() {
695            row.upsert(column.to_string(), value);
696            return Ok(());
697        }
698
699        let target = row
700            .get_mut(column)
701            .ok_or_else(|| TableError::ColumnNotFound {
702                row_index,
703                column: column.to_string(),
704            })?;
705        *target = value;
706        Ok(())
707    }

1332    /// Loads `row_index` into the reusable row buffer.
1333    ///
1334    /// If the current row buffer is dirty and `row_index` differs from the
1335    /// loaded row, the current row is flushed first.
1336    pub fn load(&mut self, row_index: usize) -> Result<&RecordValue, TableError> {
1337        self.table
1338            .require_row_index_without_loading_rows(row_index)?;
1339        if self.row_index != Some(row_index) || !self.row_materialized {
1340            self.flush()?;
1341            fill_prepared_row_buffer(self.table, &self.slots, &mut self.row, row_index)?;
1342            self.row_index = Some(row_index);
1343            self.row_materialized = true;
1344        }
1345        Ok(&self.row)
1346    }
1347
1348    /// Selects `row_index` as the current row for direct indexed writes.
1349    ///
1350    /// This avoids loading the reusable row buffer unless [`row_mut`](Self::row_mut)
1351    /// is used afterwards.
1352    pub fn seek(&mut self, row_index: usize) -> Result<(), TableError> {
1353        self.table
1354            .require_row_index_without_loading_rows(row_index)?;
1355        if self.row_index != Some(row_index) && self.dirty {
1356            self.flush()?;
1357        }
1358        if self.row_index != Some(row_index) {
1359            self.row_materialized = false;
1360        }
1361        self.row_index = Some(row_index);
1362        Ok(())
1363    }
1364
1365    /// Flushes the current row buffer through the table accessor layer.
1366    pub fn flush(&mut self) -> Result<(), TableError> {
1367        let Some(row_index) = self.row_index else {
1368            self.dirty = false;
1369            return Ok(());
1370        };
1371        if self.dirty {
1372            flush_prepared_row_buffer(self.table, &self.slots, &self.row, row_index)?;
1373            self.dirty = false;
1374        }
1375        Ok(())
1376    }
1377
1378    /// Writes `value` to `slot_index` in the currently selected row.
1379    ///
1380    /// This is the fast path for callers that already computed replacement
1381    /// values and do not need a materialized row buffer.
1382    pub fn set_value_at(&mut self, slot_index: usize, value: Value) -> Result<(), TableError> {
1383        let row_index = current_prepared_row_index(self.row_index)?;
1384        if self.dirty && self.row_materialized {
1385            if let Some(field) = self.row.fields_mut().get_mut(slot_index) {
1386                field.value = value;
1387                return Ok(());
1388            }
1389        }
1390        let slot = self
1391            .slots
1392            .get(slot_index)
1393            .ok_or_else(|| TableError::SchemaColumnUnknown {
1394                column: format!("#{slot_index}"),
1395            })?;
1396        if self.row_materialized
1397            && let Some(field) = self.row.fields_mut().get_mut(slot_index)
1398        {
1399            field.value = value.clone();
1400        }
1401        match (slot.kind, value) {
1402            (PreparedRowSlotKind::Scalar, Value::Scalar(value)) => self
1403                .table
1404                .set_scalar_cell_assuming_valid(row_index, &slot.column, value),
1405            (PreparedRowSlotKind::Array, Value::Array(value)) => self
1406                .table
1407                .set_array_cell_assuming_valid(row_index, &slot.column, value),
1408            (PreparedRowSlotKind::Scalar, value) => Err(TableError::ColumnTypeMismatch {
1409                row_index,
1410                column: slot.column.clone(),
1411                expected: "scalar",
1412                found: value.kind(),
1413            }),
1414            (PreparedRowSlotKind::Array, value) => Err(TableError::ColumnTypeMismatch {
1415                row_index,
1416                column: slot.column.clone(),
1417                expected: "array",
1418                found: value.kind(),
1419            }),
1420        }
1421    }

1772pub(super) fn validate_cell_against_schema_column(
1773    row_index: usize,
1774    column: &ColumnSchema,
1775    value: Option<&Value>,
1776) -> Result<(), TableError> {
1777    match (column.column_type(), value) {
1778        (ColumnType::Scalar, Some(Value::Scalar(_))) => Ok(()),
1779        (ColumnType::Scalar, Some(value)) => Err(TableError::ColumnTypeMismatch {
1780            row_index,
1781            column: column.name().to_string(),
1782            expected: "scalar",
1783            found: value.kind(),
1784        }),
1785        (ColumnType::Scalar, None) => {
1786            if column.options().undefined {
1787                Ok(())
1788            } else {
1789                Err(TableError::SchemaColumnMissing {
1790                    row_index,
1791                    column: column.name().to_string(),
1792                })
1793            }
1794        }
1795        (ColumnType::Record, Some(Value::Record(_))) => Ok(()),
1796        (ColumnType::Record, Some(value)) => Err(TableError::ColumnTypeMismatch {
1797            row_index,
1798            column: column.name().to_string(),
1799            expected: "record",
1800            found: value.kind(),
1801        }),
1802        (ColumnType::Record, None) => Ok(()),
1803        (ColumnType::Array(contract), Some(Value::Array(array))) => {
1804            validate_array_contract(row_index, column.name(), contract, array)
1805        }
1806        (ColumnType::Array(_), Some(value)) => Err(TableError::ColumnTypeMismatch {
1807            row_index,
1808            column: column.name().to_string(),
1809            expected: "array",
1810            found: value.kind(),
1811        }),
1812        (ColumnType::Array(ArrayShapeContract::Fixed { .. }), None) => {
1813            Err(TableError::SchemaColumnMissing {
1814                row_index,
1815                column: column.name().to_string(),
1816            })
1817        }
1818        (ColumnType::Array(ArrayShapeContract::Variable { .. }), None) => Ok(()),
1819    }
1820}
1821
1822fn validate_array_contract(
1823    row_index: usize,
1824    column_name: &str,
1825    contract: &ArrayShapeContract,
1826    array: &ArrayValue,
1827) -> Result<(), TableError> {
1828    match contract {
1829        ArrayShapeContract::Fixed { shape } => {
1830            let found = array.shape().to_vec();
1831            if found == *shape {
1832                Ok(())
1833            } else {
1834                Err(TableError::ArrayShapeMismatch {
1835                    row_index,
1836                    column: column_name.to_string(),
1837                    expected: shape.clone(),
1838                    found,
1839                })
1840            }
1841        }
1842        ArrayShapeContract::Variable {
1843            ndim: Some(expected),
1844        } => {
1845            let found = array.ndim();
1846            if found == *expected {
1847                Ok(())
1848            } else {
1849                Err(TableError::ArrayNdimMismatch {
1850                    row_index,
1851                    column: column_name.to_string(),
1852                    expected: *expected,
1853                    found,
1854                })
1855            }
1856        }
1857        ArrayShapeContract::Variable { ndim: None } => Ok(()),
1858    }
