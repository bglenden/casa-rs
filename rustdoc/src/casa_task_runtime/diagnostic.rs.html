3use serde::{Deserialize, Serialize};
4
5/// Severity of a parameter/profile diagnostic.
6#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
7#[serde(rename_all = "snake_case")]
8pub enum DiagnosticLevel {
9    /// Informational resolution detail.
10    Info,
11    /// The profile is usable but deserves user attention.
12    Warning,
13    /// Resolution or validation cannot continue.
14    Error,
15}
16
17/// Stable machine-readable diagnostic category.
18#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
19#[serde(rename_all = "snake_case")]
20pub enum DiagnosticCode {
21    UnknownParameter,
22    DuplicateParameter,
23    InactiveParameter,
24    WrongSurface,
25    WrongKind,
26    FutureFormat,
27    FutureContract,
28    MissingRequired,
29    InvalidType,
30    InvalidValue,
31    UnsafeValue,
32    DefaultChanged,
33    Migrated,
34    AutomaticSaveFailed,
35}
36
37/// One-based source location within a human profile.
38#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
39pub struct SourceLocation {
40    pub line: usize,
41    pub column: usize,
42}
43
44/// One actionable profile diagnostic shared by every UI surface.
45#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
46pub struct Diagnostic {
47    pub level: DiagnosticLevel,
48    pub code: DiagnosticCode,
49    pub message: String,
50    #[serde(skip_serializing_if = "Option::is_none")]
51    pub parameter: Option<String>,
52    #[serde(skip_serializing_if = "Option::is_none")]
53    pub location: Option<SourceLocation>,
54    #[serde(default, skip_serializing_if = "Vec::is_empty")]
55    pub suggestions: Vec<String>,
56}
57
58impl Diagnostic {
59    pub(crate) fn error(code: DiagnosticCode, message: impl Into<String>) -> Self {
60        Self {
61            level: DiagnosticLevel::Error,
62            code,
63            message: message.into(),
64            parameter: None,
65            location: None,
66            suggestions: Vec::new(),
67        }
68    }
69
70    pub(crate) fn warning(code: DiagnosticCode, message: impl Into<String>) -> Self {
71        Self {
72            level: DiagnosticLevel::Warning,
73            code,
74            message: message.into(),
75            parameter: None,
76            location: None,
77            suggestions: Vec::new(),
78        }
79    }
80
81    pub(crate) fn for_parameter(mut self, parameter: impl Into<String>) -> Self {
82        self.parameter = Some(parameter.into());
83        self
84    }
85}
