2//! BitFlagsEngine — virtual Bool columns derived from integer bit-flag columns.
3//!
4//! Maps stored integer columns (UInt8, Int16, Int32) to Bool arrays by applying
5//! a configurable bitmask: `virtual_bool[i] = (stored_int[i] & read_mask) != 0`.
6//!
7//! Nearly every MeasurementSet uses this engine for FLAG columns, where the
8//! underlying FLAG_ROW or BIT_FLAGS column stores per-correlation bit flags
9//! and the FLAG column presents a boolean view for downstream processing.
10//!
11//! # On-disk keywords (on the virtual Bool column)
12//!
13//! - `_BitFlagsEngine_ReadMask` (UInt32) — bitmask for read direction
14//! - `_BitFlagsEngine_WriteMask` (UInt32) — bitmask for write direction
15//! - `_BitFlagsEngine_ReadMaskKeys` (String array) — symbolic mask names
16//! - `_BitFlagsEngine_WriteMaskKeys` (String array) — symbolic mask names
17//! - `_BaseMappedArrayEngine_StoredColumnName` (String) — stored column name
18//!
19//! # DM type strings
20//!
21//! `"BitFlagsEngine<uChar"`, `"BitFlagsEngine<Short"`, `"BitFlagsEngine<Int"`
22//! (note: no closing `>` — this is a C++ quirk).
23//!
24//! # C++ equivalent
25//!
26//! `BitFlagsEngine<uChar|Short|Int>` in
27//! `casacore/tables/DataMan/BitFlagsEngine.h`.
28

36// -- Keyword name constants --------------------------------------------------
37
38/// Keyword storing the name of the underlying integer column.
39const KW_STORED_COL: &str = "_BaseMappedArrayEngine_StoredColumnName";
40
41/// Keyword storing the read-direction bitmask (UInt32).
42const KW_READ_MASK: &str = "_BitFlagsEngine_ReadMask";
43
44// -- Engine struct -----------------------------------------------------------
45
46/// Virtual column engine that maps integer bit-flag columns to Bool arrays.
47///
48/// For each element in the stored integer column, the virtual Bool value is
49/// `(stored & read_mask) != 0`. If the read mask keyword is absent, all bits
50/// are considered (mask = `0xFFFF_FFFF`).
51///
52/// # C++ equivalent
53///
54/// `BitFlagsEngine<uChar|Short|Int>` in
55/// `casacore/tables/DataMan/BitFlagsEngine.h`.
56#[derive(Debug)]
57pub(crate) struct BitFlagsEngine;
58
59impl VirtualColumnEngine for BitFlagsEngine {
60    fn type_name(&self) -> &str {
61        "BitFlagsEngine"
62    }
63
64    fn materialize(
65        &self,
66        ctx: &VirtualContext,
67        bound_cols: &[(usize, &PlainColumnEntry)],
68        rows: &mut [RecordValue],
69    ) -> Result<(), StorageError> {
70        for &(desc_idx, _pc) in bound_cols {
71            let col_desc = &ctx.col_descs[desc_idx];
72            let col_name = &col_desc.col_name;
73            let kw = &col_desc.keywords;
74
75            // Read stored column name from keywords.
76            let stored_col_name = get_string_keyword(kw, KW_STORED_COL, col_name)?;
77
78            // Read the bitmask (default: all bits set).
79            let read_mask = get_uint32_keyword(kw, KW_READ_MASK).unwrap_or(0xFFFF_FFFF);
80
81            for (row_idx, row) in rows.iter_mut().enumerate() {
82                let stored_value = ctx.rows[row_idx].get(&stored_col_name);
83                let virtual_value = match stored_value {
84                    Some(v) => apply_bitmask(v, read_mask)?,
85                    None => {
86                        return Err(StorageError::FormatMismatch(format!(
87                            "BitFlagsEngine: stored column '{stored_col_name}' \
88                             not found in row {row_idx}"
89                        )));
90                    }
91                };
92                row.push(RecordField::new(col_name.clone(), virtual_value));
93            }
94        }
95        Ok(())
96    }
97}
98
99// -- Transform logic ---------------------------------------------------------
100
101/// Apply the bitmask transform: `(stored & mask) != 0`.
102///
103/// Handles both scalar and array stored values of integer types
104/// (UInt8, Int16, Int32, UInt16, UInt32, Int64).
105fn apply_bitmask(stored: &Value, mask: u32) -> Result<Value, StorageError> {
106    match stored {
107        Value::Scalar(sv) => {
108            let bits = scalar_to_u32(sv)?;
109            Ok(Value::Scalar(ScalarValue::Bool((bits & mask) != 0)))
110        }
111        Value::Array(av) => {
112            let bool_arr = array_to_bool(av, mask)?;
113            Ok(Value::Array(ArrayValue::Bool(bool_arr)))
114        }
115        Value::Record(_) => Err(StorageError::FormatMismatch(
116            "BitFlagsEngine: cannot apply bitmask to a Record value".to_string(),
117        )),
118        Value::TableRef(_) => Err(StorageError::FormatMismatch(
119            "BitFlagsEngine: cannot apply bitmask to a table reference".to_string(),
120        )),
121    }
122}
123
124/// Convert a scalar integer value to u32 for bitmask operations.
125fn scalar_to_u32(sv: &ScalarValue) -> Result<u32, StorageError> {
126    match sv {
127        ScalarValue::UInt8(v) => Ok(*v as u32),
128        ScalarValue::Int16(v) => Ok(*v as u32),
129        ScalarValue::Int32(v) => Ok(*v as u32),
130        ScalarValue::UInt16(v) => Ok(*v as u32),
131        ScalarValue::UInt32(v) => Ok(*v),
132        ScalarValue::Int64(v) => Ok(*v as u32),
133        other => Err(StorageError::FormatMismatch(format!(
134            "BitFlagsEngine: unsupported stored scalar type: {other:?}"
135        ))),
136    }
137}
138
139/// Convert an integer array to a Bool array via bitmask.
140fn array_to_bool(av: &ArrayValue, mask: u32) -> Result<ArrayD<bool>, StorageError> {
141    match av {
142        ArrayValue::UInt8(a) => Ok(a.mapv(|x| (x as u32 & mask) != 0)),
143        ArrayValue::Int16(a) => Ok(a.mapv(|x| (x as u32 & mask) != 0)),
144        ArrayValue::Int32(a) => Ok(a.mapv(|x| (x as u32 & mask) != 0)),
145        ArrayValue::UInt16(a) => Ok(a.mapv(|x| (x as u32 & mask) != 0)),
146        ArrayValue::UInt32(a) => Ok(a.mapv(|x| (x & mask) != 0)),
147        ArrayValue::Int64(a) => Ok(a.mapv(|x| (x as u32 & mask) != 0)),
148        other => Err(StorageError::FormatMismatch(format!(
149            "BitFlagsEngine: unsupported stored array type: {:?}",
150            std::mem::discriminant(other)
151        ))),
152    }
153}
154
155// -- Keyword helpers ---------------------------------------------------------
156
157/// Extract a string keyword value.
158fn get_string_keyword(kw: &RecordValue, key: &str, col_name: &str) -> Result<String, StorageError> {
159    match kw.get(key) {
160        Some(Value::Scalar(ScalarValue::String(s))) => Ok(s.clone()),
161        Some(_) => Err(StorageError::FormatMismatch(format!(
162            "column '{col_name}': keyword '{key}' is not a string"
163        ))),
164        None => Err(StorageError::FormatMismatch(format!(
165            "column '{col_name}': missing keyword '{key}'"
166        ))),
167    }
168}
169
170/// Extract a UInt32 keyword value, supporting promotion from smaller int types.
171fn get_uint32_keyword(kw: &RecordValue, key: &str) -> Option<u32> {
172    match kw.get(key)? {
173        Value::Scalar(ScalarValue::UInt32(v)) => Some(*v),
174        Value::Scalar(ScalarValue::Int32(v)) => Some(*v as u32),
175        Value::Scalar(ScalarValue::UInt8(v)) => Some(*v as u32),
176        Value::Scalar(ScalarValue::Int16(v)) => Some(*v as u32),
177        Value::Scalar(ScalarValue::UInt16(v)) => Some(*v as u32),
178        _ => None,
179    }
180}

188    /// Build a minimal VirtualContext for testing.
189    fn make_context<'a>(
190        col_descs: &'a [super::super::table_control::ColumnDescContents],
191        rows: &'a [RecordValue],
192        nrrow: usize,
193    ) -> VirtualContext<'a> {
194        VirtualContext {
195            col_descs,
196            rows,
197            table_path: std::path::Path::new("/tmp/test_bitflags"),
198            nrrow,
199        }
200    }

202    fn make_col_desc(
203        col_name: &str,
204        keywords: RecordValue,
205    ) -> super::super::table_control::ColumnDescContents {
206        use super::super::data_type::CasacoreDataType;
207        super::super::table_control::ColumnDescContents {
208            class_name: String::new(),
209            col_name: col_name.to_string(),
210            comment: String::new(),
211            data_manager_type: "BitFlagsEngine<uChar".to_string(),
212            data_manager_group: "BitFlagsEngine".to_string(),
213            data_type: CasacoreDataType::TpBool,
214            option: 0,
215            nrdim: -1,
216            shape: Vec::new(),
217            max_length: 0,
218            keywords,
219            is_array: true,
220            primitive_type: Some(casacore_types::PrimitiveType::Bool),
221        }
222    }

233    fn test_bitflags_uint8_array() {
234        // Stored column: UInt8 array [0x01, 0x02, 0x04, 0x08]
235        // Read mask: 0x05 (bits 0 and 2)
236        // Expected: [true, false, true, false]
237        let mut kw = RecordValue::default();
238        kw.push(RecordField::new(
239            KW_STORED_COL.to_string(),
240            Value::Scalar(ScalarValue::String("BIT_FLAGS".to_string())),
241        ));
242        kw.push(RecordField::new(
243            KW_READ_MASK.to_string(),
244            Value::Scalar(ScalarValue::UInt32(0x05)),
245        ));
246
247        let col_desc = make_col_desc("FLAG", kw);
248        let pc = make_plain_col(1);
249
250        let stored_arr =
251            ArrayD::from_shape_vec(ndarray::IxDyn(&[4]), vec![0x01u8, 0x02, 0x04, 0x08]).unwrap();
252        let mut stored_row = RecordValue::default();
253        stored_row.push(RecordField::new(
254            "BIT_FLAGS".to_string(),
255            Value::Array(ArrayValue::UInt8(stored_arr)),
256        ));
257
258        let stored_rows = vec![stored_row];
259        let mut rows = vec![RecordValue::default()];
260
261        let descs = [col_desc.clone()];
262        let ctx = make_context(&descs, &stored_rows, 1);
263        let engine = BitFlagsEngine;
264        engine
265            .materialize(&ctx, &[(0, &pc)], &mut rows)
266            .expect("materialize should succeed");
267
268        let flag_val = rows[0].get("FLAG").expect("FLAG column missing");
269        match flag_val {
270            Value::Array(ArrayValue::Bool(arr)) => {
271                assert_eq!(arr.as_slice().unwrap(), &[true, false, true, false]);
272            }
273            other => panic!("expected Bool array, got {other:?}"),
274        }
275    }

392    #[test]
393    fn test_bitflags_multiple_rows() {
394        let mut kw = RecordValue::default();
395        kw.push(RecordField::new(
396            KW_STORED_COL.to_string(),
397            Value::Scalar(ScalarValue::String("BF".to_string())),
398        ));
399        kw.push(RecordField::new(
400            KW_READ_MASK.to_string(),
401            Value::Scalar(ScalarValue::UInt32(0x01)),
402        ));
403
404        let col_desc = make_col_desc("FLAG", kw);
405        let pc = make_plain_col(1);
406
407        let stored_rows: Vec<RecordValue> = (0..3)
408            .map(|i| {
409                let mut r = RecordValue::default();
410                r.push(RecordField::new(
411                    "BF".to_string(),
412                    Value::Scalar(ScalarValue::UInt8(i as u8)),
413                ));
414                r
415            })
416            .collect();
417
418        let mut rows: Vec<RecordValue> = (0..3).map(|_| RecordValue::default()).collect();
419
420        let descs = [col_desc.clone()];
421        let ctx = make_context(&descs, &stored_rows, 3);
422        BitFlagsEngine
423            .materialize(&ctx, &[(0, &pc)], &mut rows)
424            .unwrap();
425
426        // Row 0: stored=0, 0&1=0 → false
427        // Row 1: stored=1, 1&1=1 → true
428        // Row 2: stored=2, 2&1=0 → false
429        let expected = [false, true, false];
430        for (i, exp) in expected.iter().enumerate() {
431            match rows[i].get("FLAG").unwrap() {
432                Value::Scalar(ScalarValue::Bool(b)) => assert_eq!(b, exp, "row {i}"),
433                other => panic!("row {i}: expected Bool scalar, got {other:?}"),
434            }
435        }
436    }
