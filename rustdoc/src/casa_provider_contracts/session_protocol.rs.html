15pub const JSONL_STDIO_TRANSPORT: &str = "jsonl_stdio";

17/// Build the common provider bundle for a typed JSONL session protocol.
18pub fn jsonl_session_contract<Request, Response>(
19    protocol_name: &str,
20    protocol_version: u32,
21    binary_version: &str,
22    parameter_surface: SurfaceContractBundle,
23) -> SessionProviderContract
24where
25    Request: JsonSchema,
26    Response: JsonSchema,
27{
28    let request_schema = schema_for!(Request);
29    let response_schema = schema_for!(Response);
30    SessionProviderContract {
31        protocol: ProviderProtocolDescriptor::new(
32            protocol_name,
33            protocol_version,
34            ProviderSurfaceKind::Session,
35            binary_version,
36        ),
37        semantic: SessionSemanticContract {
38            transport: JSONL_STDIO_TRANSPORT.to_string(),
39            request_schema: request_schema.clone(),
40            response_schema: response_schema.clone(),
41        },
42        components: merged_components([&request_schema, &response_schema]),
43        annotations: serde_json::json!({}),
44        projections: ProviderProjectionMetadata {
45            cli: Some(ProviderCliProjection {
46                machine_actions: ProviderCliMachineActions {
47                    json_schema: Some("--json-schema".to_string()),
48                    protocol_info: Some("--protocol-info".to_string()),
49                    json_run: None,
50                    session: Some("--session".to_string()),
51                },
52            }),
53            python: None,
54        },
55        parameter_surfaces: vec![parameter_surface],
56        domain_schemas: SessionProviderSchemas {
57            request_schema,
58            response_schema,
59            additional: NoAdditionalProviderSchemas::default(),
60        },
61    }
62}

64/// Render the deterministic JSON schema for a session request or response.
65pub fn session_schema_json<T>() -> Result<String, serde_json::Error>
66where
67    T: JsonSchema,
68{
69    serde_json::to_string_pretty(&schema_for!(T))

79/// Common version accessor used by generic JSONL clients.
80pub trait VersionedSessionEnvelope {
81    /// Protocol version carried on the wire.
82    fn protocol_version(&self) -> u32;
83}

85/// Define concrete, schema-stable request/response/error types for a JSONL
86/// session protocol while keeping their implementation in this shared owner.
87#[macro_export]
88macro_rules! define_jsonl_session_envelopes {
89    (
90        request $request:ident for $command:ty;
91        response $response:ident for $payload:ty;
92        error $error:ident;
93    ) => {
94        /// JSON Lines request envelope sent from `casars` to the browser backend.
95        #[derive(
96            Debug, Clone, serde::Serialize, serde::Deserialize, schemars::JsonSchema, PartialEq, Eq,
97        )]
98        pub struct $request {
99            /// Protocol version expected by the client.
100            pub version: u32,
101            /// Requested command.
102            pub command: $command,
103        }
104
105        impl $request {
106            /// Wrap a command using the current protocol version.
107            pub fn new(command: $command) -> Self {
108                Self {
109                    version: PROTOCOL_VERSION,
110                    command,
111                }
112            }
113        }
114
115        impl $crate::VersionedSessionEnvelope for $request {
116            fn protocol_version(&self) -> u32 {
117                self.version
118            }
119        }
120
121        /// JSON Lines response envelope sent from the backend to `casars`.
122        #[derive(
123            Debug, Clone, serde::Serialize, serde::Deserialize, schemars::JsonSchema, PartialEq,
124        )]
125        pub struct $response {
126            /// Protocol version returned by the backend.
127            pub version: u32,
128            /// Response payload.
129            pub response: $payload,
130        }
131
132        impl $crate::VersionedSessionEnvelope for $response {
133            fn protocol_version(&self) -> u32 {
134                self.version
135            }
136        }
137
138        /// Structured error payload returned by the session backend.
139        #[derive(
140            Debug, Clone, serde::Serialize, serde::Deserialize, schemars::JsonSchema, PartialEq, Eq,
141        )]
142        pub struct $error {
143            /// Stable machine-readable error code.
144            pub code: String,
145            /// Human-readable explanation.
146            pub message: String,
147        }
148
149        impl $error {
150            /// Construct a provider-owned error payload.
151            pub fn new(code: impl Into<String>, message: impl Into<String>) -> Self {
152                Self {
153                    code: code.into(),
154                    message: message.into(),
155                }
156            }
157        }
158    };
159}

161/// Define deterministic stale checks (and an explicit regeneration path) for
162/// checked-in request and response schemas.
163#[macro_export]
164macro_rules! committed_session_schema_tests {
165    (
166        request $request:ty => $request_file:literal;
167        response $response:ty => $response_file:literal;
168    ) => {
169        #[cfg(test)]
170        mod committed_session_schemas {
171            use std::path::Path;
172
173            fn regenerate_if_requested(path: &Path, schema: &str) {
174                if std::env::var_os("CASA_RS_REGENERATE_SESSION_SCHEMAS").is_none() {
175                    return;
176                }
177                if let Some(parent) = path.parent() {
178                    std::fs::create_dir_all(parent).expect("create protocol schema directory");
179                }
180                std::fs::write(path, format!("{}\n", schema.trim_end()))
181                    .expect("write regenerated protocol schema");
182            }
183
184            #[test]
185            fn request_schema_matches_committed_artifact() {
186                let generated =
187                    $crate::session_schema_json::<$request>().expect("generate request schema");
188                let path = Path::new(env!("CARGO_MANIFEST_DIR")).join($request_file);
189                regenerate_if_requested(&path, &generated);
190                let committed = std::fs::read_to_string(&path).expect("read request schema");
191                assert_eq!(
192                    serde_json::from_str::<serde_json::Value>(&generated)
193                        .expect("parse generated request schema"),
194                    serde_json::from_str::<serde_json::Value>(&committed)
195                        .expect("parse committed request schema")
196                );
197            }
198
199            #[test]
200            fn response_schema_matches_committed_artifact() {
201                let generated =
202                    $crate::session_schema_json::<$response>().expect("generate response schema");
203                let path = Path::new(env!("CARGO_MANIFEST_DIR")).join($response_file);
204                regenerate_if_requested(&path, &generated);
205                let committed = std::fs::read_to_string(&path).expect("read response schema");
206                assert_eq!(
207                    serde_json::from_str::<serde_json::Value>(&generated)
208                        .expect("parse generated response schema"),
209                    serde_json::from_str::<serde_json::Value>(&committed)
210                        .expect("parse committed response schema")
211                );
212            }
213        }
214    };
215}
