150/// Read table.dat and dispatch based on the table type marker.
151pub(crate) fn read_table_dat_dispatch(path: &Path) -> Result<TableDatResult, StorageError> {
152    let mut io = AipsIo::open(path, AipsOpenOption::Old)?;
153    let table_version = io.getstart("Table")?;
154
155    let nrrow = if table_version >= 3 {
156        io.get_u64()?
157    } else {
158        io.get_u32()? as u64
159    };
160
161    let format = io.get_u32()?;
162    let big_endian = format == 0;
163
164    let table_type = io.get_string()?;
165
166    match table_type.as_str() {
167        "PlainTable" => {
168            let table_desc = read_table_desc(&mut io)?;
169            let ncol = table_desc.columns.len();
170            let col_is_array: Vec<bool> = table_desc.columns.iter().map(|c| c.is_array).collect();
171            let column_set = read_column_set(&mut io, table_version, ncol, &col_is_array)?;
172
173            io.getend()?;
174            io.close()?;
175
176            Ok(TableDatResult::Plain(TableDatContents {
177                nrrow,
178                big_endian,
179                table_desc,
180                column_set,
181            }))
182        }
183        "RefTable" => {
184            let ref_contents = read_ref_table_payload(&mut io, nrrow, big_endian)?;
185            io.getend()?;
186            io.close()?;
187            Ok(TableDatResult::Ref(ref_contents))
188        }
189        "ConcatTable" => {
190            let concat_contents = read_concat_table_payload(&mut io, nrrow, big_endian)?;
191            io.getend()?;
192            io.close()?;
193            Ok(TableDatResult::Concat(concat_contents))
194        }
195        other => Err(StorageError::FormatMismatch(format!(
196            "unsupported table type: {other}"
197        ))),
198    }
199}

201/// Read table.dat assuming it contains a PlainTable (backward-compat wrapper).
202pub(crate) fn read_table_dat(path: &Path) -> Result<TableDatContents, StorageError> {
203    match read_table_dat_dispatch(path)? {
204        TableDatResult::Plain(contents) => Ok(contents),
205        TableDatResult::Ref(_) => Err(StorageError::FormatMismatch(
206            "expected PlainTable but found RefTable".to_string(),
207        )),
208        TableDatResult::Concat(_) => Err(StorageError::FormatMismatch(
209            "expected PlainTable but found ConcatTable".to_string(),
210        )),
211    }
212}

214/// Read the AipsIO "RefTable" object payload from inside the "Table" envelope.
215fn read_ref_table_payload(
216    io: &mut AipsIo,
217    nrrow: u64,
218    big_endian: bool,
219) -> Result<RefTableDatContents, StorageError> {
220    let version = io.getstart("RefTable")?;
221    if version > 3 {
222        return Err(StorageError::FormatMismatch(format!(
223            "unsupported RefTable version: {version}"
224        )));
225    }
226
227    // Parent table relative path.
228    let parent_relative_path = io.get_string()?;
229
230    // Column name map: C++ serializes std::map<String, String> as a
231    // "SimpleOrderedMap" AipsIO object with a default value, count, old incr,
232    // then (key, value) pairs.
233    let _som_version = io.getstart("SimpleOrderedMap")?;
234    let _default_value = io.get_string()?; // old default value (unused)
235    let map_count = io.get_u32()? as usize;
236    let _old_incr = io.get_u32()?; // old increment (unused)
237    let mut column_name_map = Vec::with_capacity(map_count);
238    for _ in 0..map_count {
239        let view_name = io.get_string()?;
240        let parent_name = io.get_string()?;
241        column_name_map.push((view_name, parent_name));
242    }
243    io.getend()?;
244
245    // Column names (version > 1 only).
246    // C++ writes Vector<String> as an "Array" AipsIO object (version 3):
247    //   ndim(u32), shape[ndim](u32...), count(u32), then strings.
248    let column_names = if version > 1 {
249        let _array_version = io.getstart("Array")?;
250        let ndim = io.get_u32()?;
251        let mut _shape = Vec::with_capacity(ndim as usize);
252        for _ in 0..ndim {
253            _shape.push(io.get_u32()?);
254        }
255        let n = io.get_u32()? as usize;
256        let mut names = Vec::with_capacity(n);
257        for _ in 0..n {
258            names.push(io.get_string()?);
259        }
260        io.getend()?;
261        names
262    } else {
263        // Version 1: derive from map keys.
264        column_name_map.iter().map(|(k, _)| k.clone()).collect()
265    };
266
267    // Parent row count and row-order flag.
268    let (parent_nrrow, row_order, ref_nrrow) = if version > 2 {
269        let pnr = io.get_u64()?;
270        let ro = io.get_bool()?;
271        let rnr = io.get_u64()?;
272        (pnr, ro, rnr)
273    } else {
274        let pnr = io.get_u32()? as u64;
275        let ro = io.get_bool()?;
276        let rnr = io.get_u32()? as u64;
277        (pnr, ro, rnr)
278    };
279
280    // Row number vector (chunked at 2^20 per read, no count prefix).
281    let max_chunk: u64 = 1 << 20;
282    let mut row_map = vec![0u64; ref_nrrow as usize];
283    let mut done: u64 = 0;
284    while done < ref_nrrow {
285        let todo = std::cmp::min(ref_nrrow - done, max_chunk) as usize;
286        if version > 2 {
287            io.get_u64_into(&mut row_map[done as usize..done as usize + todo])?;
288        } else {
289            let mut buf32 = vec![0u32; todo];
290            io.get_u32_into(&mut buf32)?;
291            for (i, &v) in buf32.iter().enumerate() {
292                row_map[done as usize + i] = v as u64;
293            }
294        }
295        done += todo as u64;
296    }
297
298    io.getend()?;
299
300    Ok(RefTableDatContents {
301        nrrow,
302        big_endian,
303        parent_relative_path,
304        column_name_map,
305        column_names,
306        parent_nrrow,
307        row_order,
308        row_map,
309    })
310}

312/// Read the AipsIO "ConcatTable" object payload from inside the "Table" envelope.
313fn read_concat_table_payload(
314    io: &mut AipsIo,
315    nrrow: u64,
316    big_endian: bool,
317) -> Result<ConcatTableDatContents, StorageError> {
318    let version = io.getstart("ConcatTable")?;
319    if version > 0 {
320        return Err(StorageError::FormatMismatch(format!(
321            "unsupported ConcatTable version: {version}"
322        )));
323    }
324
325    // Number of constituent tables.
326    let ntable = io.get_u32()? as usize;
327    let mut table_paths = Vec::with_capacity(ntable);
328    for _ in 0..ntable {
329        table_paths.push(io.get_string()?);
330    }
331
332    // Block<String> subtable names: C++ serializes via operator<<(AipsIO, Block<String>)
333    // which wraps in an AipsIO "Block" object envelope (version 1).
334    let _block_version = io.getstart("Block")?;
335    let nsub = io.get_u32()? as usize;
336    let mut sub_table_names = Vec::with_capacity(nsub);
337    for _ in 0..nsub {
338        sub_table_names.push(io.get_string()?);
339    }
340    io.getend()?; // Block
341
342    io.getend()?; // ConcatTable
343
344    Ok(ConcatTableDatContents {
345        nrrow,
346        big_endian,
347        table_paths,
348        sub_table_names,
349    })
350}

352fn read_table_desc(io: &mut AipsIo) -> Result<TableDescContents, StorageError> {
353    let version = io.getstart("TableDesc")?;
354
355    let name = io.get_string()?;
356    let vers = io.get_string()?;
357    let comment = io.get_string()?;
358
359    let table_keywords = read_table_record(io)?;
360
361    let private_keywords = if version >= 2 {
362        read_table_record(io)?
363    } else {
364        RecordValue::default()
365    };
366
367    // ColumnDescSet: read ncol then each column
368    let ncol = io.get_u32()?;
369    let mut columns = Vec::with_capacity(ncol as usize);
370    for _ in 0..ncol {
371        columns.push(read_column_desc(io)?);
372    }
373
374    io.getend()?;
375
376    Ok(TableDescContents {
377        name,
378        version: vers,
379        comment,
380        table_keywords,
381        private_keywords,
382        columns,
383    })
384}

386fn read_column_desc(io: &mut AipsIo) -> Result<ColumnDescContents, StorageError> {
387    // ColumnDesc wrapper
388    let _col_desc_version = io.get_u32()?; // version=1
389    let class_name = io.get_string()?;
390
391    let class_info = parse_column_class_name(&class_name).ok_or_else(|| {
392        StorageError::FormatMismatch(format!("unknown column class name: {class_name}"))
393    })?;
394
395    let (primitive_type, is_array) = match class_info {
396        ColumnClassInfo::Typed {
397            primitive_type,
398            is_array,
399        } => (Some(primitive_type), is_array),
400        ColumnClassInfo::Record => (None, false),
401    };
402
403    // BaseColumnDesc
404    let _base_version = io.get_u32()?; // version=1
405    let col_name = io.get_string()?;
406    let comment = io.get_string()?;
407    let data_manager_type = io.get_string()?;
408    let data_manager_group = io.get_string()?;
409    let data_type_i32 = io.get_i32()?;
410    let data_type = CasacoreDataType::from_i32(data_type_i32).ok_or_else(|| {
411        StorageError::FormatMismatch(format!("unknown data type: {data_type_i32}"))
412    })?;
413    let option = io.get_i32()?;
414    let nrdim = io.get_i32()?;
415
416    // C++ writes shape for ALL non-scalar columns (if !isScalar_p),
417    // regardless of nrdim. This includes array columns with nrdim=0.
418    let shape = if is_array {
419        read_iposition(io)?
420    } else {
421        vec![]
422    };
423
424    let max_length = io.get_u32()?;
425
426    let keywords = read_table_record(io)?;
427
428    // Type-specific descriptor (ScalarColumnDesc or ArrayColumnDesc)
429    let _type_version = io.get_u32()?; // version=1
430    if is_array {
431        // ArrayColumnDesc: hasDefault (Bool)
432        let _has_default = io.get_bool()?;
433    } else {
434        // ScalarColumnDesc: default value
435        skip_default_value(io, data_type)?;
436    }
437
438    Ok(ColumnDescContents {
439        class_name,
440        col_name,
441        comment,
442        data_manager_type,
443        data_manager_group,
444        data_type,
445        option,
446        nrdim,
447        shape,
448        max_length,
449        keywords,
450        is_array,
451        primitive_type,
452    })
453}

455fn skip_default_value(io: &mut AipsIo, dt: CasacoreDataType) -> Result<(), StorageError> {
456    match dt {
457        CasacoreDataType::TpBool => {
458            io.get_bool()?;
459        }
460        CasacoreDataType::TpUChar => {
461            io.get_u8()?;
462        }
463        CasacoreDataType::TpShort => {
464            io.get_i16()?;
465        }
466        CasacoreDataType::TpUShort => {
467            io.get_u16()?;
468        }
469        CasacoreDataType::TpInt => {
470            io.get_i32()?;
471        }
472        CasacoreDataType::TpUInt => {
473            io.get_u32()?;
474        }
475        CasacoreDataType::TpFloat => {
476            io.get_f32()?;
477        }
478        CasacoreDataType::TpDouble => {
479            io.get_f64()?;
480        }
481        CasacoreDataType::TpComplex => {
482            io.get_complex32()?;
483        }
484        CasacoreDataType::TpDComplex => {
485            io.get_complex64()?;
486        }
487        CasacoreDataType::TpString => {
488            io.get_string()?;
489        }
490        CasacoreDataType::TpInt64 => {
491            io.get_i64()?;
492        }
493        // ScalarRecordColumnDesc has no default value.
494        CasacoreDataType::TpRecord => {}
495        _ => {
496            return Err(StorageError::FormatMismatch(format!(
497                "cannot skip default value for type {:?}",
498                dt
499            )));
500        }
501    }
502    Ok(())
503}

505fn read_column_set(
506    io: &mut AipsIo,
507    _table_version: u32,
508    ncol: usize,
509    col_is_array: &[bool],
510) -> Result<ColumnSetContents, StorageError> {
511    let version = io.get_i32()?; // negative: -2 or -3
512
513    let nrrow = if version == -3 {
514        let nr = io.get_u64()?;
515        let _storage_opt = io.get_i32()?;
516        let _block_size = io.get_i32()?;
517        nr
518    } else {
519        io.get_u32()? as u64
520    };
521
522    let seq_count = io.get_u32()?;
523
524    let num_dm = io.get_u32()?;
525    let mut data_managers = Vec::with_capacity(num_dm as usize);
526    for _ in 0..num_dm {
527        let type_name = io.get_string()?;
528        let seq_nr = io.get_u32()?;
529        data_managers.push(DataManagerEntry {
530            type_name,
531            seq_nr,
532            data: Vec::new(),
533        });
534    }
535
536    // Read PlainColumn entries (one per column in the table)
537    let mut columns = Vec::with_capacity(ncol);
538    for &is_array in &col_is_array[..ncol] {
539        columns.push(read_plain_column(io, is_array)?);
540    }
541
542    // Read DM data blobs (one per data manager) — preserve for StandardStMan
543    for dm in &mut data_managers {
544        let len = io.get_u32()?;
545        if len > 0 {
546            let mut buf = vec![0u8; len as usize];
547            io.get_u8_into(&mut buf)?;
548            dm.data = buf;
549        }
550    }
551
552    Ok(ColumnSetContents {
553        nrrow,
554        seq_count,
555        data_managers,
556        columns,
557    })
558}

560fn read_plain_column(io: &mut AipsIo, is_array: bool) -> Result<PlainColumnEntry, StorageError> {
561    let _version = io.get_u32()?; // version=2
562    let original_name = io.get_string()?;
563
564    // Type-specific derived data (ScaColData or ArrColData)
565    let _derived_version = io.get_u32()?; // version=1
566    let dm_seq_nr = io.get_u32()?;
567
568    if is_array {
569        // ArrColData: shapeColDef (bool) + optional shapeCol (IPosition)
570        let shape_col_def = io.get_bool()?;
571        if shape_col_def {
572            // C++ writes IPosition via ios << shapeCol_p (AipsIO object, not string)
573            let _shape_col = read_iposition(io)?;
574        }
575    }
576
577    Ok(PlainColumnEntry {
578        original_name,
579        dm_seq_nr,
580        is_array,
581    })
582}

588pub(crate) fn read_iposition(io: &mut AipsIo) -> Result<Vec<i32>, StorageError> {
589    let _version = io.getstart("IPosition")?;
590    let n = io.get_u32()?;
591    let mut values = vec![0i32; n as usize];
592    io.get_i32_into(&mut values)?;
593    io.getend()?;
594    Ok(values)
595}

601/// Read a C++ `Record` (serialized as `"RecordRep"`) from AipsIO.
602///
603/// This is the format used by `TSMCube::values_p` in tiled storage managers.
604/// Structurally identical to `TableRecord` but uses a different AipsIO type name.
605pub(crate) fn read_record_rep(io: &mut AipsIo) -> Result<RecordValue, StorageError> {
606    let _version = io.getstart("RecordRep")?;
607    let desc = read_record_desc(io)?;
608    let _record_type = io.get_i32()?;
609    let mut fields = Vec::with_capacity(desc.len());
610    for (name, field_type) in &desc {
611        let value = read_record_field_value(io, *field_type)?;
612        fields.push(RecordField::new(name.clone(), value));
613    }
614    io.getend()?;
615    Ok(RecordValue::new(fields))
616}
617
618/// Read a C++ `Record` object.
619///
620/// C++ `RecordRep::putRecord` writes a single `"Record"` v1 envelope containing
621/// `RecordDesc`, `recordType` (Int), then the raw field data. There is no nested
622/// `"RecordRep"` envelope — that name is only used in `TableRecord` context.
623///
624/// The TSMCube `values` field uses this format (`ios << values_p` in TSMCube.cc).
625pub(crate) fn read_record(io: &mut AipsIo) -> Result<RecordValue, StorageError> {
626    let _version = io.getstart("Record")?;
627    let desc = read_record_desc(io)?;
628    let _record_type = io.get_i32()?;
629    let mut fields = Vec::with_capacity(desc.len());
630    for (name, field_type) in &desc {
631        let value = read_record_field_value(io, *field_type)?;
632        fields.push(RecordField::new(name.clone(), value));
633    }
634    io.getend()?;
635    Ok(RecordValue::new(fields))
636}
637
638/// Write a C++ `Record` object.
639///
640/// Writes: `putstart("Record", 1)`, `RecordDesc`, `recordType`, field data, `putend()`.
641/// This matches C++ `RecordRep::putRecord`.
642pub(crate) fn write_record(io: &mut AipsIo, record: &RecordValue) -> Result<(), StorageError> {
643    io.putstart("Record", 1)?;
644    write_record_desc(io, record)?;
645    io.put_i32(1)?; // recordType = RecordInterface::Variable
646    for field in record.fields() {
647        write_record_field_value(io, &field.value)?;
648    }
649    io.putend()?;
650    Ok(())
651}

665fn read_table_record(io: &mut AipsIo) -> Result<RecordValue, StorageError> {
666    let _version = io.getstart("TableRecord")?;
667
668    let desc = read_record_desc(io)?;
669    let _record_type = io.get_i32()?; // RecordType enum
670
671    let mut fields = Vec::with_capacity(desc.len());
672    for (name, field_type) in &desc {
673        let value = read_record_field_value(io, *field_type)?;
674        fields.push(RecordField::new(name.clone(), value));
675    }
676
677    io.getend()?;
678    Ok(RecordValue::new(fields))
679}

681fn read_record_desc(io: &mut AipsIo) -> Result<Vec<(String, CasacoreDataType)>, StorageError> {
682    let _version = io.getstart("RecordDesc")?;
683    let nfields = io.get_i32()?;
684    let mut desc = Vec::with_capacity(nfields as usize);
685
686    for _ in 0..nfields {
687        let name = io.get_string()?;
688        let type_i32 = io.get_i32()?;
689        let dt = CasacoreDataType::from_i32(type_i32).ok_or_else(|| {
690            StorageError::FormatMismatch(format!("unknown RecordDesc field type: {type_i32}"))
691        })?;
692
693        // Sub-record, array shape, or table name depending on type
694        if dt == CasacoreDataType::TpRecord {
695            // Recursive RecordDesc — skip it (we don't use sub-record shapes)
696            let _sub_desc = read_record_desc(io)?;
697        } else if is_array_data_type(dt) {
698            // Array field: read IPosition shape
699            let _shape = read_iposition(io)?;
700        } else if dt == CasacoreDataType::TpTable {
701            let _table_desc_name = io.get_string()?;
702        }
703
704        let _comment = io.get_string()?;
705        desc.push((name, dt));
706    }
707
708    io.getend()?;
709    Ok(desc)
710}

712fn read_record_field_value(io: &mut AipsIo, dt: CasacoreDataType) -> Result<Value, StorageError> {
713    match dt {
714        CasacoreDataType::TpBool => Ok(Value::Scalar(ScalarValue::Bool(io.get_bool()?))),
715        CasacoreDataType::TpUChar => Ok(Value::Scalar(ScalarValue::UInt8(io.get_u8()?))),
716        CasacoreDataType::TpShort => Ok(Value::Scalar(ScalarValue::Int16(io.get_i16()?))),
717        CasacoreDataType::TpUShort => Ok(Value::Scalar(ScalarValue::UInt16(io.get_u16()?))),
718        CasacoreDataType::TpInt => Ok(Value::Scalar(ScalarValue::Int32(io.get_i32()?))),
719        CasacoreDataType::TpUInt => Ok(Value::Scalar(ScalarValue::UInt32(io.get_u32()?))),
720        CasacoreDataType::TpFloat => Ok(Value::Scalar(ScalarValue::Float32(io.get_f32()?))),
721        CasacoreDataType::TpDouble => Ok(Value::Scalar(ScalarValue::Float64(io.get_f64()?))),
722        CasacoreDataType::TpComplex => {
723            Ok(Value::Scalar(ScalarValue::Complex32(io.get_complex32()?)))
724        }
725        CasacoreDataType::TpDComplex => {
726            Ok(Value::Scalar(ScalarValue::Complex64(io.get_complex64()?)))
727        }
728        CasacoreDataType::TpString => Ok(Value::Scalar(ScalarValue::String(io.get_string()?))),
729        CasacoreDataType::TpInt64 => Ok(Value::Scalar(ScalarValue::Int64(io.get_i64()?))),
730        CasacoreDataType::TpRecord => {
731            let record = read_table_record(io)?;
732            Ok(Value::Record(record))
733        }
734        // Table reference keyword — stores a subtable path as a string.
735        // C++ casacore writes these via `TableKeywordSet::toRecord()` and
736        // the `TableAttr` class. We represent them as plain string values
737        // so the keyword name→path mapping is preserved.
738        CasacoreDataType::TpTable => Ok(Value::TableRef(io.get_string()?)),
739        // Array types — used in hypercolumn definitions and cube metadata.
740        CasacoreDataType::TpArrayBool
741        | CasacoreDataType::TpArrayUChar
742        | CasacoreDataType::TpArrayShort
743        | CasacoreDataType::TpArrayUShort
744        | CasacoreDataType::TpArrayInt
745        | CasacoreDataType::TpArrayUInt
746        | CasacoreDataType::TpArrayFloat
747        | CasacoreDataType::TpArrayDouble
748        | CasacoreDataType::TpArrayComplex
749        | CasacoreDataType::TpArrayDComplex
750        | CasacoreDataType::TpArrayString
751        | CasacoreDataType::TpArrayInt64 => read_array_record_field(io, dt),
752        _ => Err(StorageError::FormatMismatch(format!(
753            "unsupported record field type: {dt:?}"
754        ))),
755    }
756}

758/// Read an array-typed record field from AipsIO.
759///
760/// C++ RecordRep uses type-specific names: `"Array<String>"`, `"Array<float>"`, etc.
761/// The reader peeks the actual type name and uses it for `getstart`.
762fn read_array_record_field(io: &mut AipsIo, dt: CasacoreDataType) -> Result<Value, StorageError> {
763    // C++ writes different type names depending on the element type.
764    // Peek the actual type name and use it for getstart.
765    let type_name = io.get_next_type()?;
766    let version = io.getstart(&type_name)?;
767    let ndim = io.get_u32()? as usize;
768
769    if ndim == 0 {
770        io.getend()?;
771        // Return a zero-element array of the appropriate type.
772        return match dt {
773            CasacoreDataType::TpArrayString => Ok(Value::Array(ArrayValue::String(
774                ArrayD::from_shape_vec(IxDyn(&[0]).f(), vec![]).unwrap(),
775            ))),
776            _ => Ok(Value::Array(ArrayValue::Float32(
777                ArrayD::from_shape_vec(IxDyn(&[0]).f(), vec![]).unwrap(),
778            ))),
779        };
780    }
781
782    // Older versions (< 3) include an origin before the shape.
783    if version < 3 {
784        for _ in 0..ndim {
785            let _origin = io.get_i32()?;
786        }
787    }
788
789    // Shape: ndim × u32 (NOT an IPosition AipsIO object).
790    let mut shape = Vec::with_capacity(ndim);
791    for _ in 0..ndim {
792        shape.push(io.get_u32()? as usize);
793    }
794    let nelem: usize = shape.iter().product();
795
796    let av = match dt {
797        CasacoreDataType::TpArrayBool => {
798            let count = io.get_u32()? as usize;
799            let mut vals = vec![false; count.min(nelem)];
800            for v in vals.iter_mut() {
801                *v = io.get_bool()?;
802            }
803            ArrayValue::Bool(
804                ArrayD::from_shape_vec(IxDyn(&shape).f(), vals)
805                    .map_err(|e| StorageError::FormatMismatch(format!("array shape: {e}")))?,
806            )
807        }
808        CasacoreDataType::TpArrayUChar => {
809            let count = io.get_u32()? as usize;
810            let mut vals = vec![0u8; count.min(nelem)];
811            for v in vals.iter_mut() {
812                *v = io.get_u8()?;
813            }
814            ArrayValue::UInt8(
815                ArrayD::from_shape_vec(IxDyn(&shape).f(), vals)
816                    .map_err(|e| StorageError::FormatMismatch(format!("array shape: {e}")))?,
817            )
818        }
819        CasacoreDataType::TpArrayShort => {
820            let count = io.get_u32()? as usize;
821            let mut vals = vec![0i16; count.min(nelem)];
822            for v in vals.iter_mut() {
823                *v = io.get_i16()?;
824            }
825            ArrayValue::Int16(
826                ArrayD::from_shape_vec(IxDyn(&shape).f(), vals)
827                    .map_err(|e| StorageError::FormatMismatch(format!("array shape: {e}")))?,
828            )
829        }
830        CasacoreDataType::TpArrayUShort => {
831            let count = io.get_u32()? as usize;
832            let mut vals = vec![0u16; count.min(nelem)];
833            for v in vals.iter_mut() {
834                *v = io.get_u16()?;
835            }
836            ArrayValue::UInt16(
837                ArrayD::from_shape_vec(IxDyn(&shape).f(), vals)
838                    .map_err(|e| StorageError::FormatMismatch(format!("array shape: {e}")))?,
839            )
840        }
841        CasacoreDataType::TpArrayInt => {
842            let count = io.get_u32()? as usize;
843            let mut vals = vec![0i32; count.min(nelem)];
844            for v in vals.iter_mut() {
845                *v = io.get_i32()?;
846            }
847            ArrayValue::Int32(
848                ArrayD::from_shape_vec(IxDyn(&shape).f(), vals)
849                    .map_err(|e| StorageError::FormatMismatch(format!("array shape: {e}")))?,
850            )
851        }
852        CasacoreDataType::TpArrayUInt => {
853            let count = io.get_u32()? as usize;
854            let mut vals = vec![0u32; count.min(nelem)];
855            for v in vals.iter_mut() {
856                *v = io.get_u32()?;
857            }
858            ArrayValue::UInt32(
859                ArrayD::from_shape_vec(IxDyn(&shape).f(), vals)
860                    .map_err(|e| StorageError::FormatMismatch(format!("array shape: {e}")))?,
861            )
862        }
863        CasacoreDataType::TpArrayFloat => {
864            let count = io.get_u32()? as usize;
865            let mut vals = vec![0f32; count.min(nelem)];
866            for v in vals.iter_mut() {
867                *v = io.get_f32()?;
868            }
869            ArrayValue::Float32(
870                ArrayD::from_shape_vec(IxDyn(&shape).f(), vals)
871                    .map_err(|e| StorageError::FormatMismatch(format!("array shape: {e}")))?,
872            )
873        }
874        CasacoreDataType::TpArrayDouble => {
875            let count = io.get_u32()? as usize;
876            let mut vals = vec![0f64; count.min(nelem)];
877            for v in vals.iter_mut() {
878                *v = io.get_f64()?;
879            }
880            ArrayValue::Float64(
881                ArrayD::from_shape_vec(IxDyn(&shape).f(), vals)
882                    .map_err(|e| StorageError::FormatMismatch(format!("array shape: {e}")))?,
883            )
884        }
885        CasacoreDataType::TpArrayComplex => {
886            let count = io.get_u32()? as usize;
887            let mut vals = vec![Complex32::new(0.0, 0.0); count.min(nelem)];
888            for v in vals.iter_mut() {
889                *v = io.get_complex32()?;
890            }
891            ArrayValue::Complex32(
892                ArrayD::from_shape_vec(IxDyn(&shape).f(), vals)
893                    .map_err(|e| StorageError::FormatMismatch(format!("array shape: {e}")))?,
894            )
895        }
896        CasacoreDataType::TpArrayDComplex => {
897            let count = io.get_u32()? as usize;
898            let mut vals = vec![Complex64::new(0.0, 0.0); count.min(nelem)];
899            for v in vals.iter_mut() {
900                *v = io.get_complex64()?;
901            }
902            ArrayValue::Complex64(
903                ArrayD::from_shape_vec(IxDyn(&shape).f(), vals)
904                    .map_err(|e| StorageError::FormatMismatch(format!("array shape: {e}")))?,
905            )
906        }
907        CasacoreDataType::TpArrayString => {
908            let count = io.get_u32()? as usize;
909            let mut vals = Vec::with_capacity(count.min(nelem));
910            for _ in 0..count.min(nelem) {
911                vals.push(io.get_string()?);
912            }
913            ArrayValue::String(
914                ArrayD::from_shape_vec(IxDyn(&shape).f(), vals)
915                    .map_err(|e| StorageError::FormatMismatch(format!("array shape: {e}")))?,
916            )
917        }
918        CasacoreDataType::TpArrayInt64 => {
919            let count = io.get_u32()? as usize;
920            let mut vals = vec![0i64; count.min(nelem)];
921            for v in vals.iter_mut() {
922                *v = io.get_i64()?;
923            }
924            ArrayValue::Int64(
925                ArrayD::from_shape_vec(IxDyn(&shape).f(), vals)
926                    .map_err(|e| StorageError::FormatMismatch(format!("array shape: {e}")))?,
927            )
928        }
929        _ => {
930            return Err(StorageError::FormatMismatch(format!(
931                "unsupported array record field type: {dt:?}"
932            )));
933        }
934    };
935
936    io.getend()?;
937    Ok(Value::Array(av))
938}

983/// Write a RefTable table.dat file.
984pub(crate) fn write_ref_table_dat(
985    path: &Path,
986    contents: &RefTableDatContents,
987) -> Result<(), StorageError> {
988    let mut io = AipsIo::open(path, AipsOpenOption::New)?;
989
990    // Table envelope — version 2 (nrrow as u32).
991    io.putstart("Table", 2)?;
992    io.put_u32(contents.nrrow as u32)?;
993    io.put_u32(if contents.big_endian { 0 } else { 1 })?;
994    io.put_string("RefTable")?;
995
996    // RefTable payload — version 2 (32-bit row numbers).
997    io.putstart("RefTable", 2)?;
998
999    // Parent table relative path.
1000    io.put_string(&contents.parent_relative_path)?;
1001
1002    // Column name map: C++ serializes std::map<String, String> as a
1003    // "SimpleOrderedMap" AipsIO object.
1004    io.putstart("SimpleOrderedMap", 1)?;
1005    io.put_string("")?; // old default value (unused)
1006    io.put_u32(contents.column_name_map.len() as u32)?;
1007    io.put_u32(1)?; // old increment (unused)
1008    for (view_name, parent_name) in &contents.column_name_map {
1009        io.put_string(view_name)?;
1010        io.put_string(parent_name)?;
1011    }
1012    io.putend()?;
1013
1014    // Column names (version > 1): C++ writes as an "Array" AipsIO object (v3).
1015    // Format: ndim(u32), shape[ndim](u32...), count(u32), then strings.
1016    io.putstart("Array", 3)?;
1017    io.put_u32(1)?; // ndim = 1 (Vector)
1018    io.put_u32(contents.column_names.len() as u32)?; // shape[0]
1019    io.put_u32(contents.column_names.len() as u32)?; // element count
1020    for name in &contents.column_names {
1021        io.put_string(name)?;
1022    }
1023    io.putend()?;
1024
1025    // Parent row count at save time.
1026    io.put_u32(contents.parent_nrrow as u32)?;
1027
1028    // Row order flag.
1029    io.put_bool(contents.row_order)?;
1030
1031    // This RefTable's row count.
1032    io.put_u32(contents.nrrow as u32)?;
1033
1034    // Row number vector (chunked at 2^20, no count prefix).
1035    let max_chunk: usize = 1 << 20;
1036    let mut done: usize = 0;
1037    let total = contents.row_map.len();
1038    while done < total {
1039        let todo = std::cmp::min(total - done, max_chunk);
1040        let chunk32: Vec<u32> = contents.row_map[done..done + todo]
1041            .iter()
1042            .map(|&r| r as u32)
1043            .collect();
1044        io.put_u32_slice(&chunk32, false)?;
1045        done += todo;
1046    }
1047
1048    io.putend()?; // RefTable
1049
1050    io.putend()?; // Table
1051    io.close()?;
1052    Ok(())
1053}

1055/// Write table.dat for a ConcatTable.
1056///
1057/// Binary layout matches C++ `ConcatTable::writeConcatTable`:
1058/// ```text
1059/// putstart("Table", 2)
1060///   nrrow (u32), endian (u32), "ConcatTable" (String)
1061///   putstart("ConcatTable", 0)
1062///     ntable (u32), table_path[0..ntable] (String each)
1063///     putstart("Block", 1)                 // Block<String> subtable names
1064///       count (u32), sub_table_name[0..count] (String each)
1065///     putend()
1066///   putend()
1067/// putend()
1068/// ```
1069pub(crate) fn write_concat_table_dat(
1070    path: &Path,
1071    contents: &ConcatTableDatContents,
1072) -> Result<(), StorageError> {
1073    let mut io = AipsIo::open(path, AipsOpenOption::New)?;
1074
1075    // Table envelope — version 2 (nrrow as u32).
1076    io.putstart("Table", 2)?;
1077    io.put_u32(contents.nrrow as u32)?;
1078    io.put_u32(if contents.big_endian { 0 } else { 1 })?;
1079    io.put_string("ConcatTable")?;
1080
1081    // ConcatTable payload — version 0.
1082    io.putstart("ConcatTable", 0)?;
1083
1084    // Number of constituent tables and their relative paths.
1085    io.put_u32(contents.table_paths.len() as u32)?;
1086    for table_path in &contents.table_paths {
1087        io.put_string(table_path)?;
1088    }
1089
1090    // Block<String> subtable names: C++ serializes as "Block" AipsIO object (v1).
1091    io.putstart("Block", 1)?;
1092    io.put_u32(contents.sub_table_names.len() as u32)?;
1093    for name in &contents.sub_table_names {
1094        io.put_string(name)?;
1095    }
1096    io.putend()?; // Block
1097
1098    io.putend()?; // ConcatTable
1099
1100    io.putend()?; // Table
1101    io.close()?;
1102    Ok(())
1103}

1105fn write_table_desc(io: &mut AipsIo, desc: &TableDescContents) -> Result<(), StorageError> {
1106    io.putstart("TableDesc", 2)?;
1107    io.put_string(&desc.name)?;
1108    io.put_string(&desc.version)?;
1109    io.put_string(&desc.comment)?;
1110

1112    write_table_record(io, &desc.private_keywords)?;
1113
1114    // ColumnDescSet
1115    io.put_u32(desc.columns.len() as u32)?;
1116    for col in &desc.columns {
1117        write_column_desc(io, col)?;
1118    }
1119
1120    io.putend()?;
1121    Ok(())

1124fn write_column_desc(io: &mut AipsIo, col: &ColumnDescContents) -> Result<(), StorageError> {
1125    // ColumnDesc wrapper
1126    io.put_u32(1)?; // version=1
1127    io.put_string(&col.class_name)?;
1128
1129    // BaseColumnDesc
1130    io.put_u32(1)?; // version=1
1131    io.put_string(&col.col_name)?;
1132    io.put_string(&col.comment)?;
1133    io.put_string(&col.data_manager_type)?;
1134    io.put_string(&col.data_manager_group)?;
1135    io.put_i32(col.data_type as i32)?;
1136    io.put_i32(col.option)?;
1137    io.put_i32(col.nrdim)?;
1138
1139    // C++ writes shape for all non-scalar columns (if !isScalar_p)
1140    if col.is_array {
1141        write_iposition(io, &col.shape)?;
1142    }
1143
1144    io.put_u32(col.max_length)?;
1145    write_table_record(io, &col.keywords)?;
1146
1147    // Type-specific descriptor
1148    io.put_u32(1)?; // version=1
1149    if col.is_array {
1150        io.put_bool(false)?; // hasDefault=false
1151    } else {
1152        write_default_value(io, col.data_type)?;
1153    }
1154
1155    Ok(())
1156}

1158fn write_default_value(io: &mut AipsIo, dt: CasacoreDataType) -> Result<(), StorageError> {
1159    match dt {
1160        CasacoreDataType::TpBool => io.put_bool(false)?,
1161        CasacoreDataType::TpUChar => io.put_u8(0)?,
1162        CasacoreDataType::TpShort => io.put_i16(0)?,
1163        CasacoreDataType::TpUShort => io.put_u16(0)?,
1164        CasacoreDataType::TpInt => io.put_i32(0)?,
1165        CasacoreDataType::TpUInt => io.put_u32(0)?,
1166        CasacoreDataType::TpFloat => io.put_f32(0.0)?,
1167        CasacoreDataType::TpDouble => io.put_f64(0.0)?,
1168        CasacoreDataType::TpComplex => io.put_complex32(Complex32::new(0.0, 0.0))?,
1169        CasacoreDataType::TpDComplex => io.put_complex64(Complex64::new(0.0, 0.0))?,
1170        CasacoreDataType::TpString => io.put_string("")?,
1171        CasacoreDataType::TpInt64 => io.put_i64(0)?,
1172        // ScalarRecordColumnDesc has no default value.
1173        CasacoreDataType::TpRecord => {}
1174        _ => {
1175            return Err(StorageError::FormatMismatch(format!(
1176                "cannot write default for type {:?}",
1177                dt
1178            )));
1179        }
1180    }
1181    Ok(())
1182}

1184fn write_column_set(io: &mut AipsIo, contents: &TableDatContents) -> Result<(), StorageError> {
1185    let cs = &contents.column_set;
1186
1187    // Version -2 (nrrow as u32)
1188    io.put_i32(-2)?;
1189    io.put_u32(cs.nrrow as u32)?;
1190    io.put_u32(cs.seq_count)?;
1191
1192    // Data managers
1193    io.put_u32(cs.data_managers.len() as u32)?;
1194    for dm in &cs.data_managers {
1195        io.put_string(&dm.type_name)?;
1196        io.put_u32(dm.seq_nr)?;
1197    }
1198
1199    // PlainColumn entries
1200    for (i, col) in cs.columns.iter().enumerate() {
1201        io.put_u32(2)?; // version=2
1202        io.put_string(&col.original_name)?;
1203        // ScaColData or ArrColData derived data
1204        io.put_u32(1)?; // version=1
1205        io.put_u32(col.dm_seq_nr)?;
1206        if col.is_array {
1207            // ArrColData: write shapeColDef + optional shapeCol (IPosition)
1208            let col_desc = &contents.table_desc.columns[i];
1209            let has_shape = col_desc.nrdim > 0 && !col_desc.shape.is_empty();
1210            io.put_bool(has_shape)?;
1211            if has_shape {
1212                write_iposition(io, &col_desc.shape)?;
1213            }
1214        }
1215    }
1216
1217    // DM data blobs (empty for StManAipsIO, populated for StandardStMan)
1218    for dm in &cs.data_managers {
1219        io.put_u32(dm.data.len() as u32)?;
1220        if !dm.data.is_empty() {
1221            io.put_u8_slice(&dm.data, false)?;
1222        }
1223    }
1224
1225    Ok(())
1226}

1249/// Deserialize a `RecordValue` from a uChar byte buffer (AipsIO `TableRecord` format).
1250///
1251/// C++ `ScalarRecordColumnData` stores records as `Vector<uChar>` containing
1252/// AipsIO-serialized `TableRecord` objects. This function reverses that encoding.
1253pub(crate) fn deserialize_record_from_uchar(bytes: &[u8]) -> Result<RecordValue, StorageError> {

1255    use std::io::Cursor;

1258        AipsIo::new_read_only_with_order(Cursor::new(bytes.to_vec()), ByteOrder::BigEndian);

1262/// Serialize a `RecordValue` to a uChar byte buffer (AipsIO `TableRecord` format).
1263///
1264/// Produces the same encoding that C++ `ScalarRecordColumnData::putRecord` writes:
1265/// a `TableRecord` AipsIO object serialized to a `Vector<uChar>`.
1266pub(crate) fn serialize_record_to_uchar(record: &RecordValue) -> Result<Vec<u8>, StorageError> {
1267    use casacore_aipsio::ByteOrder;
1268    use std::io::Cursor;
1269
1270    let buf = Vec::new();
1271    let mut io = AipsIo::new_write_only_with_order(Cursor::new(buf), ByteOrder::BigEndian);

1273    let cursor: Cursor<Vec<u8>> = io
1274        .into_inner_typed()
1275        .map_err(|e| StorageError::FormatMismatch(format!("serialize_record_to_uchar: {e}")))?;

1286fn write_record_desc(io: &mut AipsIo, record: &RecordValue) -> Result<(), StorageError> {
1287    io.putstart("RecordDesc", 2)?;
1288    io.put_i32(record.fields().len() as i32)?;
1289
1290    for field in record.fields() {
1291        io.put_string(&field.name)?;
1292        let dt = value_to_casacore_data_type(&field.value)?;
1293        io.put_i32(dt as i32)?;
1294
1295        if dt == CasacoreDataType::TpRecord {
1296            // C++ writes an empty sub-record schema here (nfields=0).
1297            // The actual fields are written in write_record_field_value as
1298            // a full TableRecord (Variable record type).
1299            write_empty_record_desc(io)?;
1300        } else if dt == CasacoreDataType::TpTable {
1301            // Table keyword entries carry an additional table-desc name slot.
1302            // We do not model named table descriptors, so persist an empty name
1303            // while keeping the wire layout aligned with casacore.
1304            io.put_string("")?;
1305        } else if is_array_data_type(dt) {
1306            // Array field: write IPosition shape.
1307            if let Value::Array(av) = &field.value {
1308                let shape: Vec<i32> = av.shape().iter().map(|&d| d as i32).collect();
1309                write_iposition(io, &shape)?;
1310            }
1311        }
1312
1313        io.put_string("")?; // comment
1314    }
1315
1316    io.putend()?;
1317    Ok(())
1318}

1320fn write_record_field_value(io: &mut AipsIo, value: &Value) -> Result<(), StorageError> {
1321    match value {
1322        Value::Scalar(sv) => match sv {
1323            ScalarValue::Bool(v) => io.put_bool(*v)?,
1324            ScalarValue::UInt8(v) => io.put_u8(*v)?,
1325            ScalarValue::UInt16(v) => io.put_u16(*v)?,
1326            ScalarValue::UInt32(v) => io.put_u32(*v)?,
1327            ScalarValue::Int16(v) => io.put_i16(*v)?,
1328            ScalarValue::Int32(v) => io.put_i32(*v)?,
1329            ScalarValue::Int64(v) => io.put_i64(*v)?,
1330            ScalarValue::Float32(v) => io.put_f32(*v)?,
1331            ScalarValue::Float64(v) => io.put_f64(*v)?,
1332            ScalarValue::Complex32(v) => io.put_complex32(*v)?,
1333            ScalarValue::Complex64(v) => io.put_complex64(*v)?,
1334            ScalarValue::String(v) => io.put_string(v)?,
1335        },
1336        Value::TableRef(path) => io.put_string(path)?,
1337        Value::Record(record) => {
1338            write_table_record(io, record)?;
1339        }
1340        Value::Array(av) => {
1341            write_array_record_field(io, av)?;
1342        }
1343    }
1344    Ok(())
1345}

1347/// Write an array-typed record field to AipsIO.
1348///
1349/// C++ serializes `Array<T>` as: `"Array"` v3 → Int ndim → IPosition shape → count + data.
1350fn write_array_record_field(io: &mut AipsIo, av: &ArrayValue) -> Result<(), StorageError> {
1351    // C++ RecordRep uses type-specific AipsIO names for arrays.
1352    let type_name = match av {
1353        ArrayValue::Bool(_) => "Array<void>",
1354        ArrayValue::UInt8(_) => "Array<uChar>",
1355        ArrayValue::Int16(_) => "Array<short>",
1356        ArrayValue::UInt16(_) => "Array<short>",
1357        ArrayValue::Int32(_) => "Array<Int>",
1358        ArrayValue::UInt32(_) => "Array<uInt>",
1359        ArrayValue::Int64(_) => "Array<Int64>",
1360        ArrayValue::Float32(_) => "Array<float>",
1361        ArrayValue::Float64(_) => "Array<double>",
1362        ArrayValue::Complex32(_) => "Array<void>",
1363        ArrayValue::Complex64(_) => "Array<void>",
1364        ArrayValue::String(_) => "Array<String>",
1365    };
1366    io.putstart(type_name, 3)?;
1367
1368    // Write ndim as u32, then shape as ndim × u32 (NOT as IPosition).
1369    let ndim = av.ndim() as u32;
1370    io.put_u32(ndim)?;
1371    if ndim > 0 {
1372        for &d in av.shape() {
1373            io.put_u32(d as u32)?;
1374        }
1375        let nelem = av.len() as u32;
1376        io.put_u32(nelem)?;
1377        match av {
1378            ArrayValue::String(a) => {
1379                for s in a.as_slice_memory_order().unwrap_or(&[]) {
1380                    io.put_string(s)?;
1381                }
1382            }
1383            _ => {
1384                write_array_elements(io, av)?;
1385            }
1386        }
1387    }
1388
1389    io.putend()?;
1390    Ok(())
1391}

1469        }

2074impl ColumnDescContents {
2075    fn to_column_schema(&self) -> Result<ColumnSchema, StorageError> {
2076        let options = crate::schema::ColumnOptions {
2077            direct: (self.option & 1) != 0,
2078            undefined: (self.option & 2) != 0,
2079        };
2080
2081        // Record columns have data_type = TpRecord and no primitive type.
2082        if self.data_type == CasacoreDataType::TpRecord {
2083            return ColumnSchema::record(&self.col_name)
2084                .with_options(options)
2085                .map_err(|err| StorageError::FormatMismatch(err.to_string()));
2086        }
2087
2088        let pt = self.primitive_type.ok_or_else(|| {
2089            StorageError::FormatMismatch(format!(
2090                "column '{}' has no primitive type",
2091                self.col_name
2092            ))
2093        })?;
2094
2095        if self.is_array {
2096            if self.nrdim > 0 && !self.shape.is_empty() {
2097                let shape: Vec<usize> = self.shape.iter().map(|&s| s as usize).collect();
2098                ColumnSchema::array_fixed(&self.col_name, pt, shape)
2099                    .with_options(options)
2100                    .map_err(|err| StorageError::FormatMismatch(err.to_string()))
2101            } else if self.nrdim > 0 {
2102                ColumnSchema::array_variable(&self.col_name, pt, Some(self.nrdim as usize))
2103                    .with_options(options)
2104                    .map_err(|err| StorageError::FormatMismatch(err.to_string()))
2105            } else {
2106                ColumnSchema::array_variable(&self.col_name, pt, None)
2107                    .with_options(options)
2108                    .map_err(|err| StorageError::FormatMismatch(err.to_string()))
2109            }
2110        } else {
2111            ColumnSchema::scalar(&self.col_name, pt)
2112                .with_options(options)
2113                .map_err(|err| StorageError::FormatMismatch(err.to_string()))
2114        }
2115    }

2117    fn from_column_schema(col: &ColumnSchema) -> Self {
2118        use crate::schema::{ArrayShapeContract, ColumnType};
2119
2120        // Record columns have no primitive type; use TpRecord directly.
2121        if matches!(col.column_type(), ColumnType::Record) {
2122            return ColumnDescContents {
2123                class_name: RECORD_COLUMN_CLASS_NAME.to_string(),
2124                col_name: col.name().to_string(),
2125                comment: String::new(),
2126                data_manager_type: "StandardStMan".to_string(),
2127                data_manager_group: "StandardStMan".to_string(),
2128                data_type: CasacoreDataType::TpRecord,
2129                option: 0,
2130                nrdim: 0,
2131                shape: vec![],
2132                max_length: 0,
2133                keywords: RecordValue::default(),
2134                is_array: false,
2135                primitive_type: None,
2136            };
2137        }
2138
2139        let pt = col.data_type().unwrap_or(PrimitiveType::Int32);
2140        let is_array = matches!(col.column_type(), ColumnType::Array(_));
2141
2142        let (nrdim, shape) = match col.column_type() {
2143            ColumnType::Array(ArrayShapeContract::Fixed { shape }) => {
2144                let s: Vec<i32> = shape.iter().map(|&d| d as i32).collect();
2145                (s.len() as i32, s)
2146            }
2147            ColumnType::Array(ArrayShapeContract::Variable { ndim: Some(n) }) => {
2148                (*n as i32, vec![])
2149            }
2150            _ => (0, vec![]),
2151        };
2152
2153        let class_name = if is_array {
2154            array_column_class_name(pt).to_string()
2155        } else {
2156            scalar_column_class_name(pt).to_string()
2157        };
2158
2159        // C++ casacore ColumnDesc stores the scalar element type (e.g. TpFloat)
2160        // in the data_type field, even for array columns. The array-ness is
2161        // conveyed by the className (ArrayColumnDesc<T>).
2162        let dt = CasacoreDataType::from_primitive_type(pt, false);
2163
2164        // Column option flags per casacore ColumnDesc.h:
2165        //   Direct = 1, Undefined = 2, FixedShape = 4
2166        //
2167        // Fixed-shape arrays are not necessarily direct in casacore; preserve
2168        // the caller's Direct flag instead of forcing it on every fixed array.
2169        let mut option = 0;
2170        if col.options().direct {
2171            option |= 1;
2172        }
2173        if col.options().undefined {
2174            option |= 2;
2175        }
2176        if is_array
2177            && matches!(
2178                col.column_type(),
2179                ColumnType::Array(ArrayShapeContract::Fixed { .. })
2180            )
2181        {
2182            option |= 4;
2183        }
2184
2185        ColumnDescContents {
2186            class_name,
2187            col_name: col.name().to_string(),
2188            comment: String::new(),
2189            data_manager_type: "StandardStMan".to_string(),
2190            data_manager_group: "StandardStMan".to_string(),
2191            data_type: dt,
2192            option,
2193            nrdim,
2194            shape,
2195            max_length: 0,
2196            keywords: RecordValue::default(),
2197            is_array,
2198            primitive_type: Some(pt),
2199        }
2200    }
