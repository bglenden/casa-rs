151pub(crate) fn build_single_plane_execution_plan(
152    config: &CliConfig,
153    force_standard_gridder: bool,
154    ms_count: usize,
155) -> SinglePlaneExecutionPlan {
156    let output_channel_count = output_channel_count(config);
157    let spectral = spectral_plan(config.spectral_mode, output_channel_count);
158    let projection = projection_plan(config, force_standard_gridder);
159    let deconvolver = deconvolver_plan(config.deconvolver);
160    let primary_beam_products = needs_single_field_primary_beam_products(config);
161    let primary_beam_requirement =
162        primary_beam_requirement(config, projection, primary_beam_products);
163    let output_products = output_products(config, primary_beam_requirement);
164    let standard_mfs_eligible =
165        can_plan_standard_mfs_acceleration(config, force_standard_gridder, ms_count);
166    let mosaic_mfs_eligible = can_plan_mosaic_mfs_acceleration(config, ms_count);
167    let one_output_channel = spectral.is_one_output_channel();
168    let standard_mfs_regression_sentinel = matches!(spectral, SinglePlaneSpectralPlan::Mfs)
169        && matches!(
170            projection,
171            SinglePlaneProjectionPlan::Standard
172                | SinglePlaneProjectionPlan::StandardOrMosaicInferred
173        )
174        && matches!(
175            deconvolver,
176            SinglePlaneDeconvolverPlan::SingleTerm
177                | SinglePlaneDeconvolverPlan::Multiscale
178                | SinglePlaneDeconvolverPlan::Mtmfs
179        );
180
181    SinglePlaneExecutionPlan {
182        spectral,
183        projection,
184        deconvolver,
185        weighting: weighting_label(&config.weighting),
186        output_channel_count,
187        primary_beam_products,
188        primary_beam_requirement,
189        output_products,
190        cpu_multi_worker: cpu_multi_worker_eligibility(
191            config,
192            standard_mfs_eligible,
193            mosaic_mfs_eligible,
194            one_output_channel,
195        ),
196        gpu_metal: gpu_metal_eligibility(
197            config,
198            standard_mfs_eligible,
199            mosaic_mfs_eligible,
200            one_output_channel,
201        ),
202        stage_timing_attribution: "frontend-core-product-stages",
203        standard_mfs_regression_sentinel,
204    }
205}

207fn output_channel_count(config: &CliConfig) -> usize {
208    match config.spectral_mode {
209        SpectralMode::Mfs => 1,
210        SpectralMode::Cube | SpectralMode::Cubedata => config.channel_count.unwrap_or(1),
211    }

214fn spectral_plan(mode: SpectralMode, output_channel_count: usize) -> SinglePlaneSpectralPlan {
215    match mode {
216        SpectralMode::Mfs => SinglePlaneSpectralPlan::Mfs,
217        SpectralMode::Cube | SpectralMode::Cubedata if output_channel_count == 1 => {
218            SinglePlaneSpectralPlan::CubeLikeOneChannel
219        }
220        SpectralMode::Cube | SpectralMode::Cubedata => {
221            SinglePlaneSpectralPlan::CubeLikeMultiChannel
222        }
223    }
224}

226fn projection_plan(config: &CliConfig, force_standard_gridder: bool) -> SinglePlaneProjectionPlan {
227    if matches!(config.w_term_mode, WTermMode::WProject | WTermMode::Direct) {
228        return SinglePlaneProjectionPlan::WProjection;
229    }
230    if config.use_pointing || needs_single_field_primary_beam_products(config) {
231        return SinglePlaneProjectionPlan::Mosaic;
232    }
233    if !force_standard_gridder
234        && !config.force_standard_gridder
235        && (config.field_ids.as_ref().is_some_and(|ids| ids.len() > 1)
236            || config.phasecenter_field.is_some()
237            || config.phasecenter.is_some())
238    {
239        return SinglePlaneProjectionPlan::Mosaic;
240    }
241    if force_standard_gridder || config.force_standard_gridder {
242        SinglePlaneProjectionPlan::Standard
243    } else {
244        SinglePlaneProjectionPlan::StandardOrMosaicInferred
245    }
246}

248fn deconvolver_plan(deconvolver: Deconvolver) -> SinglePlaneDeconvolverPlan {
249    match deconvolver {
250        Deconvolver::Mtmfs => SinglePlaneDeconvolverPlan::Mtmfs,
251        Deconvolver::Multiscale => SinglePlaneDeconvolverPlan::Multiscale,
252        Deconvolver::Hogbom | Deconvolver::Clark => SinglePlaneDeconvolverPlan::SingleTerm,
253    }
254}

256fn primary_beam_requirement(
257    config: &CliConfig,
258    projection: SinglePlaneProjectionPlan,
259    primary_beam_products: bool,
260) -> SinglePlanePrimaryBeamRequirement {
261    if matches!(projection, SinglePlaneProjectionPlan::Mosaic) {
262        SinglePlanePrimaryBeamRequirement::MosaicProjection
263    } else if primary_beam_products && matches!(projection, SinglePlaneProjectionPlan::WProjection)
264    {
265        SinglePlanePrimaryBeamRequirement::WProjectionProducts
266    } else if primary_beam_products {
267        SinglePlanePrimaryBeamRequirement::SingleFieldProducts
268    } else if config.use_pointing {
269        SinglePlanePrimaryBeamRequirement::MosaicProjection
270    } else {
271        SinglePlanePrimaryBeamRequirement::None
272    }
273}

275fn output_products(
276    config: &CliConfig,
277    primary_beam_requirement: SinglePlanePrimaryBeamRequirement,
278) -> Vec<String> {
279    let mut products = Vec::new();
280    if config.deconvolver == Deconvolver::Mtmfs {
281        let nterms = config.nterms.max(1);
282        for term in 0..nterms {
283            products.push(format!(".image.tt{term}"));
284            products.push(format!(".residual.tt{term}"));
285            products.push(format!(".model.tt{term}"));
286            products.push(format!(".psf.tt{term}"));
287            products.push(format!(".sumwt.tt{term}"));
288        }
289        if nterms > 1 {
290            products.push(".alpha".to_string());
291            products.push(".alpha.error".to_string());
292        }
293        if config.write_pb || config.pbcor || config.mosaic_pb_limit < 0.0 {
294            for term in 0..nterms {
295                products.push(format!(".pb.tt{term}"));
296            }
297        }
298        if config.pbcor {
299            for term in 0..nterms {
300                products.push(format!(".image.tt{term}.pbcor"));
301            }
302            if nterms > 1 {
303                products.push(".alpha.pbcor".to_string());
304            }
305        }
306        return products;
307    }
308
309    for product in [".image", ".residual", ".model", ".psf", ".sumwt"] {
310        products.push(product.to_string());
311    }
312    if matches!(
313        primary_beam_requirement,
314        SinglePlanePrimaryBeamRequirement::MosaicProjection
315    ) {
316        products.push(".weight".to_string());
317    }
318    if config.write_pb || config.pbcor || config.mosaic_pb_limit < 0.0 {
319        products.push(".pb".to_string());
320    }
321    if config.pbcor {
322        products.push(".image.pbcor".to_string());
323    }
324    products
325}

336fn cpu_multi_worker_eligibility(
337    config: &CliConfig,
338    standard_mfs_eligible: bool,
339    mosaic_mfs_eligible: bool,
340    one_output_channel: bool,
341) -> BackendEligibility {
342    if !one_output_channel {
343        return BackendEligibility::ineligible("not-one-output-channel");
344    }
345    if standard_mfs_eligible || mosaic_mfs_eligible {
346        let workers = match config.standard_mfs_acceleration {
347            StandardMfsAccelerationPolicy::Cpu => 1,
348            StandardMfsAccelerationPolicy::Auto
349            | StandardMfsAccelerationPolicy::MultiCpu
350            | StandardMfsAccelerationPolicy::Metal => {
351                if let Some(override_workers) =
352                    single_plane_grid_threads_override(config.standard_mfs_grid_threads.as_deref())
353                {
354                    override_workers
355                } else if matches!(config.w_term_mode, WTermMode::WProject) {
356                    standard_mfs_wproject_auto_grid_threads()
357                } else {
358                    standard_mfs_auto_grid_threads()
359                }
360            }
361        };
362        if workers > 1 {
363            if mosaic_mfs_eligible {
364                if config.spectral_mode.is_cube_like() && config.channel_count == Some(1) {
365                    BackendEligibility::eligible(format!(
366                        "mosaic-cube-like-one-channel-parallel-prepare-workers-{workers}-single-grid-owner"
367                    ))
368                } else if config.standard_mfs_acceleration
369                    == StandardMfsAccelerationPolicy::MultiCpu
370                {
371                    BackendEligibility::eligible(format!(
372                        "mosaic-sample-range-workers-{workers}-diagnostic"
373                    ))
374                } else {
375                    BackendEligibility::ineligible(
376                        "mosaic-auto-uses-single-grid-owner-or-metal-groups",
377                    )
378                }
379            } else if matches!(config.w_term_mode, WTermMode::WProject) {
380                BackendEligibility::eligible(format!("wproject-streaming-workers-{workers}"))
381            } else if config.spectral_mode.is_cube_like() {
382                BackendEligibility::eligible(format!(
383                    "standard-cube-like-one-channel-parallel-prepare-workers-{workers}"
384                ))
385            } else {
386                BackendEligibility::eligible(format!("standard-mfs-fixed-tile-workers-{workers}"))
387            }
388        } else {
389            BackendEligibility::ineligible("standard-mfs-policy-selected-single-worker")
390        }
391    } else {
392        BackendEligibility::ineligible(shared_strategy_gap_reason(config))
393    }
394}
395
396fn single_plane_grid_threads_override(value: Option<&str>) -> Option<usize> {
397    let value = value?.trim();
398    if value.eq_ignore_ascii_case("auto") {
399        return None;
400    }
401    value.parse::<usize>().ok().filter(|threads| *threads > 0)
402}

404fn gpu_metal_eligibility(
405    config: &CliConfig,
406    standard_mfs_eligible: bool,
407    mosaic_mfs_eligible: bool,
408    one_output_channel: bool,
409) -> BackendEligibility {
410    if !one_output_channel {
411        return BackendEligibility::ineligible("not-one-output-channel");
412    }
413    if !(standard_mfs_eligible || mosaic_mfs_eligible) {
414        return BackendEligibility::ineligible(shared_strategy_gap_reason(config));
415    }
416    if config.standard_mfs_acceleration == StandardMfsAccelerationPolicy::Cpu
417        || config.standard_mfs_acceleration == StandardMfsAccelerationPolicy::MultiCpu
418    {
419        return BackendEligibility::ineligible("standard-mfs-policy-disabled-metal");
420    }
421    if casa_imaging::standard_mfs_metal_device_available() {
422        if mosaic_mfs_eligible {
423            BackendEligibility::eligible("mosaic-screen-projector-metal-single-grid-owner")
424        } else if matches!(config.w_term_mode, WTermMode::WProject) {
425            BackendEligibility::eligible("wproject-metal-kernel")
426        } else if config.spectral_mode.is_cube_like() {
427            BackendEligibility::eligible("standard-cube-like-one-channel-grouped-metal")
428        } else if config.deconvolver == Deconvolver::Mtmfs {
429            BackendEligibility::eligible("mtmfs-metal-sample-cache")
430        } else {
431            BackendEligibility::eligible("standard-mfs-grouped-metal")
432        }
433    } else {
434        BackendEligibility::ineligible("metal-device-unavailable")
435    }
436}

438fn shared_strategy_gap_reason(config: &CliConfig) -> String {
439    if config.spectral_mode.is_cube_like() && config.channel_count != Some(1) {
440        return "not-one-output-channel".to_string();
441    }
442    if config.spectral_mode.is_cube_like() && !matches!(config.w_term_mode, WTermMode::None) {
443        return "cube-like-wterm-requires-cube-path".to_string();
444    }
445    if config.spectral_mode.is_cube_like()
446        && config.cube_axis.interpolation != CubeInterpolation::Nearest
447        && !(config.dirty_only || config.niter == 0)
448    {
449        return "cleaned-linear-cube-like-requires-cube-path".to_string();
450    }
451    if config.use_pointing || matches!(config.w_term_mode, WTermMode::Direct) {
452        return "shared-strategy-not-yet-adapted-to-projection-family".to_string();
453    }
454    if config.save_model != SaveModelMode::None {
455        return "savemodel-requires-traced-path".to_string();
456    }
457    if config.use_mask != CleanMaskMode::User {
458        return "automask-not-supported-by-shared-strategy".to_string();
459    }
460    "standard-mfs-eligibility-check-rejected".to_string()
461}
