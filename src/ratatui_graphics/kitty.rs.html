110/// Errors returned by direct Kitty layer operations.
111#[derive(Debug, Error)]
112pub enum KittyLayerError {
113    /// The manager ran out of image or placement ids.
114    #[error("kitty layer id space is exhausted")]
115    IdExhausted,
116    /// Animation gaps must fit in Kitty's signed 32-bit control range.
117    #[error("kitty animation gap exceeds the supported control range")]
118    AnimationGapOverflow,
119    /// The placement was invalid.
120    #[error(transparent)]
121    InvalidPlacement(#[from] KittyPlacementError),
122    /// Writing Kitty escape sequences failed.
123    #[error("failed to write Kitty graphics sequence")]
124    Io(#[from] std::io::Error),
125    /// The requested pane slot is unknown.
126    #[error("unknown kitty pane slot {0}")]
127    UnknownSlot(NonZeroU32),
128    /// The requested stored image is unknown.
129    #[error("unknown stored kitty image {0}")]
130    UnknownImage(NonZeroU32),
131}

153/// Terminal-driven animation state for a Kitty image.
154#[derive(Clone, Copy, Debug, PartialEq, Eq)]
155pub enum KittyAnimationPlaybackState {
156    /// Stop playback and leave the current frame visible.
157    Stopped,
158    /// Play while waiting for more frames at the end of the sequence.
159    Loading,
160    /// Play and loop over the uploaded frame sequence.
161    Looping,
162}
163
164impl KittyAnimationPlaybackState {
165    fn protocol_value(self) -> u8 {
166        match self {
167            Self::Stopped => 1,
168            Self::Loading => 2,
169            Self::Looping => 3,
170        }
171    }
172}

174/// A typed animation frame gap for Kitty playback controls.
175#[derive(Clone, Copy, Debug, PartialEq, Eq)]
176pub enum KittyAnimationGap {
177    /// Display the frame for the specified amount of time.
178    Timed(Duration),
179    /// Store the frame without showing it to the user.
180    Gapless,
181}
182
183impl KittyAnimationGap {
184    fn protocol_value(self) -> Result<i32, KittyLayerError> {
185        match self {
186            Self::Timed(duration) => {
187                let millis = duration.as_millis();
188                if millis > i32::MAX as u128 {
189                    return Err(KittyLayerError::AnimationGapOverflow);
190                }
191                Ok(millis as i32)
192            }
193            Self::Gapless => Ok(-1),
194        }
195    }
196}

213impl KittyLayerManager {
214    /// Create a new layer manager with monotonic id allocation.
215    pub fn new() -> Self {
216        Self::with_starting_ids(1, 1).expect("default Kitty layer ids are valid")
217    }

219    /// Create a new layer manager with explicit starting ids.
220    pub fn with_starting_ids(
221        image_id_start: u32,
222        placement_id_start: u32,
223    ) -> Result<Self, KittyLayerError> {
224        if image_id_start == 0 || placement_id_start == 0 {
225            return Err(KittyLayerError::IdExhausted);
226        }
227        Ok(Self {
228            next_image_id: image_id_start,
229            next_placement_id: placement_id_start,
230        })
231    }
232
233    /// Allocate a fresh Kitty image id.
234    pub fn allocate_image_id(&mut self) -> Result<NonZeroU32, KittyLayerError> {
235        let image_id = NonZeroU32::new(self.next_image_id).ok_or(KittyLayerError::IdExhausted)?;
236        self.next_image_id = self
237            .next_image_id
238            .checked_add(1)
239            .ok_or(KittyLayerError::IdExhausted)?;
240        Ok(image_id)
241    }
242
243    /// Allocate a fresh Kitty placement id.
244    pub fn allocate_placement_id(&mut self) -> Result<NonZeroU32, KittyLayerError> {
245        let placement_id =
246            NonZeroU32::new(self.next_placement_id).ok_or(KittyLayerError::IdExhausted)?;
247        self.next_placement_id = self
248            .next_placement_id
249            .checked_add(1)
250            .ok_or(KittyLayerError::IdExhausted)?;
251        Ok(placement_id)
252    }
253
254    /// Allocate a fresh image/placement handle pair.
255    pub fn allocate(&mut self) -> Result<KittyLayerHandle, KittyLayerError> {
256        Ok(KittyLayerHandle::new(
257            self.allocate_image_id()?,
258            self.allocate_placement_id()?,
259        ))
260    }

262    /// Upload RGBA image data to the terminal using the supplied handle.
263    pub fn upload_rgba<W: Write>(
264        &self,
265        out: &mut W,
266        handle: KittyLayerHandle,
267        image: &RgbaImage,
268    ) -> Result<(), KittyLayerError> {
269        self.upload_rgba_with_compression(out, handle.image_id(), image, false)
270    }
271
272    fn upload_rgba_with_compression<W: Write>(
273        &self,
274        out: &mut W,
275        image_id: NonZeroU32,
276        image: &RgbaImage,
277        compressed: bool,
278    ) -> Result<(), KittyLayerError> {
279        let intro = format!(
280            "\x1b_Ga=t,q=1,f=32,s={},v={},i={}",
281            image.width(),
282            image.height(),
283            image_id
284        );
285        write_direct_data_chunks(out, &intro, image.as_raw(), compressed)?;
286        Ok(())
287    }

289    /// Place a previously uploaded image on the terminal surface.
290    pub fn place<W: Write>(
291        &self,
292        out: &mut W,
293        handle: KittyLayerHandle,
294        placement: KittyPlacement,
295    ) -> Result<(), KittyLayerError> {
296        validate_placement(placement)?;
297
298        if placement.preserve_cursor {
299            execute!(
300                out,
301                SavePosition,
302                MoveTo(placement.rect.x, placement.rect.y)
303            )?;
304        } else {
305            execute!(out, MoveTo(placement.rect.x, placement.rect.y))?;
306        }
307
308        write!(
309            out,
310            "\x1b_Ga=p,q=1,i={},p={},c={},r={}",
311            handle.image_id(),
312            handle.placement_id(),
313            placement.rect.width,
314            placement.rect.height
315        )?;
316        if placement.preserve_cursor {
317            write!(out, ",C=1")?;
318        }
319        if placement.z_index != 0 {
320            write!(out, ",z={}", placement.z_index)?;
321        }
322        write!(out, "\x1b\\")?;
323
324        if placement.preserve_cursor {
325            execute!(out, RestorePosition)?;
326        }
327
328        out.flush()?;
329
330        Ok(())
331    }

345    /// Append a full RGBA animation frame to an existing Kitty image.
346    ///
347    /// The image itself must already have been created with [`Self::upload_rgba`] or
348    /// [`Self::upload_and_place_rgba`]. The appended frame uses Kitty's `a=f` animation-frame
349    /// transfer mode and is associated with `handle.image_id()`.
350    pub fn append_animation_frame_rgba<W: Write>(
351        &self,
352        out: &mut W,
353        handle: KittyLayerHandle,
354        image: &RgbaImage,
355        gap: Option<KittyAnimationGap>,
356    ) -> Result<(), KittyLayerError> {
357        let mut intro = format!(
358            "\x1b_Ga=f,i={},q=2,f=32,s={},v={}",
359            handle.image_id(),
360            image.width(),
361            image.height()
362        );
363        if let Some(gap) = gap {
364            intro.push_str(&format!(",z={}", gap.protocol_value()?));
365        }
366        write_direct_data_chunks(out, &intro, image.as_raw(), false)?;
367        Ok(())
368    }

370    /// Send a typed animation control command for an existing Kitty image.
371    pub fn control_animation<W: Write>(
372        &self,
373        out: &mut W,
374        handle: KittyLayerHandle,
375        control: KittyAnimationControl,
376    ) -> Result<(), KittyLayerError> {
377        write!(out, "\x1b_Ga=a,i={},q=2", handle.image_id())?;
378        if let Some(current_frame) = control.current_frame {
379            write!(out, ",c={}", current_frame.get())?;
380        }
381        if let Some(frame_number) = control.frame_number {
382            write!(out, ",r={}", frame_number.get())?;
383        }
384        if let Some(gap) = control.gap {
385            write!(out, ",z={}", gap.protocol_value()?)?;
386        }
387        if let Some(state) = control.state {
388            write!(out, ",s={}", state.protocol_value())?;
389        }
390        if let Some(loops) = control.loops {
391            write!(out, ",v={}", loops.get())?;
392        }
393        write!(out, "\x1b\\")?;
394        out.flush()?;
395        Ok(())
396    }

451impl KittyStoredImageStore {
452    /// Create a new stored-image store with monotonic id allocation.
453    pub fn new() -> Self {
454        Self::with_starting_ids(1, 1).expect("default Kitty store ids are valid")
455    }
456
457    /// Create a new stored-image store with explicit starting ids.
458    pub fn with_starting_ids(
459        image_id_start: u32,
460        placement_id_start: u32,
461    ) -> Result<Self, KittyLayerError> {
462        Ok(Self {
463            manager: KittyLayerManager::with_starting_ids(image_id_start, placement_id_start)?,
464            slots: HashMap::new(),
465            images: HashMap::new(),
466            total_bytes: 0,
467        })
468    }
469
470    /// Allocate a stable pane slot id.
471    pub fn allocate_slot(&mut self) -> Result<KittyPaneSlotId, KittyLayerError> {
472        let slot = KittyPaneSlotId(self.manager.allocate_placement_id()?);
473        self.slots.insert(slot, None);
474        Ok(slot)
475    }
476
477    /// Upload an RGBA bitmap into terminal memory and return its stored id.
478    pub fn store_rgba<W: Write>(
479        &mut self,
480        out: &mut W,
481        image: &RgbaImage,
482    ) -> Result<(KittyStoredImageId, KittyStoredImageInfo), KittyLayerError> {
483        let image_id = KittyStoredImageId(self.manager.allocate_image_id()?);
484        let info = KittyStoredImageInfo {
485            pixel_width: image.width(),
486            pixel_height: image.height(),
487            bytes: image.as_raw().len(),
488        };
489        self.manager
490            .upload_rgba_with_compression(out, image_id.0, image, true)?;
491        self.total_bytes = self.total_bytes.saturating_add(info.bytes);
492        self.images.insert(image_id, info);
493        Ok((image_id, info))
494    }
495
496    /// Place a stored image into a stable pane slot.
497    pub fn place_in_slot<W: Write>(
498        &mut self,
499        out: &mut W,
500        slot: KittyPaneSlotId,
501        image: KittyStoredImageId,
502        placement: KittyPlacement,
503    ) -> Result<(), KittyLayerError> {
504        if !self.images.contains_key(&image) {
505            return Err(KittyLayerError::UnknownImage(image.0));
506        }
507        let Some(current) = self.slots.get_mut(&slot) else {
508            return Err(KittyLayerError::UnknownSlot(slot.0));
509        };
510        self.manager
511            .place(out, KittyLayerHandle::new(image.0, slot.0), placement)?;
512        *current = Some(image);
513        Ok(())
514    }
515
516    /// Show a stored image in a pane slot using delete-before-place semantics when switching images.
517    pub fn show_in_slot<W: Write>(
518        &mut self,
519        out: &mut W,
520        slot: KittyPaneSlotId,
521        image: KittyStoredImageId,
522        placement: KittyPlacement,
523    ) -> Result<(), KittyLayerError> {
524        if !self.images.contains_key(&image) {
525            return Err(KittyLayerError::UnknownImage(image.0));
526        }
527        let Some(current) = self.slots.get(&slot).copied().flatten() else {
528            if self.slots.contains_key(&slot) {
529                return self.place_in_slot(out, slot, image, placement);
530            }
531            return Err(KittyLayerError::UnknownSlot(slot.0));
532        };
533        if current != image {
534            self.manager
535                .clear_placement(out, KittyLayerHandle::new(current.0, slot.0))?;
536        }
537        self.place_in_slot(out, slot, image, placement)
538    }
539
540    /// Clear the current placement for a pane slot while leaving stored images resident.
541    pub fn clear_slot<W: Write>(
542        &mut self,
543        out: &mut W,
544        slot: KittyPaneSlotId,
545    ) -> Result<(), KittyLayerError> {
546        let Some(current) = self.slots.get_mut(&slot) else {
547            return Err(KittyLayerError::UnknownSlot(slot.0));
548        };
549        if let Some(image) = *current {
550            self.manager
551                .clear_placement(out, KittyLayerHandle::new(image.0, slot.0))?;
552            *current = None;
553        }
554        Ok(())
555    }
556
557    /// Delete a stored image and clear any pane slots that currently show it.
558    pub fn delete_image<W: Write>(
559        &mut self,
560        out: &mut W,
561        image: KittyStoredImageId,
562    ) -> Result<(), KittyLayerError> {
563        let Some(info) = self.images.remove(&image) else {
564            return Err(KittyLayerError::UnknownImage(image.0));
565        };
566        for (slot, current) in &mut self.slots {
567            if *current == Some(image) {
568                self.manager
569                    .clear_placement(out, KittyLayerHandle::new(image.0, slot.0))?;
570                *current = None;
571            }
572        }
573        self.manager
574            .delete_image(out, KittyLayerHandle::new(image.0, image.0))?;
575        self.total_bytes = self.total_bytes.saturating_sub(info.bytes);
576        Ok(())
577    }

625fn write_direct_data_chunks<W: Write>(
626    out: &mut W,
627    intro: &str,
628    data: &[u8],
629    compressed: bool,
630) -> Result<(), std::io::Error> {
631    const ENCODED_CHUNK_BYTES: usize = 4096;
632    let compressed_data;
633    let payload = if compressed {
634        let mut encoder = ZlibEncoder::new(Vec::new(), Compression::fast());
635        encoder.write_all(data)?;
636        compressed_data = encoder.finish()?;
637        compressed_data.as_slice()
638    } else {
639        data
640    };
641    let encoded = STANDARD.encode_to_string(payload);
642    if encoded.is_empty() {
643        return Ok(());
644    }
645    let total_chunks = encoded.len().div_ceil(ENCODED_CHUNK_BYTES);
646    for (index, chunk) in encoded.as_bytes().chunks(ENCODED_CHUNK_BYTES).enumerate() {
647        let more = u8::from(index + 1 < total_chunks);
648        if index == 0 {
649            write!(out, "{intro}")?;
650            if compressed {
651                write!(out, ",o=z")?;
652            }
653            write!(out, ",m={more};")?;
654        } else {
655            write!(out, "\x1b_Gm={more};")?;
656        }
657        out.write_all(chunk)?;
658        write!(out, "\x1b\\")?;
659    }
660    out.flush()?;
661    Ok(())
662}

873        let image_a = RgbaImage::from_pixel(2, 1, Rgba([1, 2, 3, 255]));

876            rect: Rect::new(0, 0, 2, 1),
877            z_index: 9,
878            preserve_cursor: true,
879        };
880        let mut out = Vec::new();
881        let (stored_a, _) = store.store_rgba(&mut out, &image_a).unwrap();
882        let (stored_b, _) = store.store_rgba(&mut out, &image_b).unwrap();
883        store
884            .show_in_slot(&mut out, slot, stored_a, placement)
885            .unwrap();
886        store
887            .show_in_slot(&mut out, slot, stored_b, placement)
