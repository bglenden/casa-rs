42fn lazy_array_column_store(
43    schema: Option<&TableSchema>,
44) -> HashMap<String, OnceCell<Vec<Option<ArrayValue>>>> {
45    schema
46        .into_iter()
47        .flat_map(|schema| schema.columns())
48        .filter(|column| matches!(column.column_type(), ColumnType::Array(_)))
49        .map(|column| (column.name().to_string(), OnceCell::new()))
50        .collect()
51}
52
53fn eager_loaded_rows(
54    rows: Vec<RecordValue>,
55    mut undefined_cells: Vec<HashSet<String>>,
56) -> LoadedRows {
57    if undefined_cells.len() != rows.len() {
58        undefined_cells = (0..rows.len()).map(|_| HashSet::new()).collect();
59    }
60    LoadedRows {
61        rows,
62        undefined_cells,
63    }
64}

79    pub(crate) fn new() -> Self {
80        let loaded_rows = OnceCell::new();
81        loaded_rows
82            .set(LoadedRows {
83                rows: Vec::new(),
84                undefined_cells: Vec::new(),
85            })
86            .expect("initialize empty row store");
87        Self {

163    fn load_rows_now(source: &LazyRowsSource) -> Result<LoadedRows, TableError> {
164        let storage = CompositeStorage;
165        let snapshot = storage
166            .load_with_row_hint(&source.path, Some(source.row_count_hint as u64))
167            .map_err(|err| {
168                TableError::Storage(format!(
169                    "failed to materialize rows for table {}: {err}",
170                    source.path.display()
171                ))
172            })?;
173        Ok(eager_loaded_rows(snapshot.rows, snapshot.undefined_cells))
174    }
175
176    fn load_scalar_columns_now(source: &LazyRowsSource) -> Result<LoadedScalarColumns, TableError> {
177        let storage = CompositeStorage;
178        let snapshot = storage
179            .load_scalar_columns_with_row_hint(&source.path, Some(source.row_count_hint as u64))
180            .map_err(|err| {
181                TableError::Storage(format!(
182                    "failed to load scalar columns for table {}: {err}",
183                    source.path.display()
184                ))
185            })?;
186        Ok(LoadedScalarColumns {
187            columns: snapshot.columns,
188        })
189    }
190
191    fn load_array_column_now(
192        source: &LazyRowsSource,
193        column: &str,
194    ) -> Result<Vec<Option<ArrayValue>>, TableError> {
195        let storage = CompositeStorage;
196        storage
197            .load_array_column_with_row_hint(
198                &source.path,
199                column,
200                Some(source.row_count_hint as u64),
201            )
202            .map_err(|err| {
203                TableError::Storage(format!(
204                    "failed to load array column '{column}' for table {}: {err}",
205                    source.path.display()
206                ))
207            })
208    }

210    fn ensure_loaded(&self) -> Result<&LoadedRows, TableError> {
211        if let Some(loaded) = self.loaded_rows.get() {
212            return Ok(loaded);
213        }
214        let loaded = match &self.lazy_rows {
215            Some(source) => Self::load_rows_now(source)?,
216            None => LoadedRows {
217                rows: Vec::new(),
218                undefined_cells: Vec::new(),
219            },
220        };
221        self.loaded_rows
222            .set(loaded)
223            .expect("initialize immutable row store");
224        Ok(self
225            .loaded_rows
226            .get()
227            .expect("row store initialized before shared access"))
228    }
229
230    fn ensure_loaded_mut(&mut self) -> Result<&mut LoadedRows, TableError> {
231        if self.loaded_rows.get().is_none() {
232            let loaded = match &self.lazy_rows {
233                Some(source) => Self::load_rows_now(source)?,
234                None => LoadedRows {
235                    rows: Vec::new(),
236                    undefined_cells: Vec::new(),
237                },
238            };
239            self.loaded_rows
240                .set(loaded)
241                .expect("initialize mutable row store");
242        }
243        Ok(self
244            .loaded_rows
245            .get_mut()
246            .expect("row store initialized before mutable access"))
247    }

249    pub(crate) fn add_row(&mut self, row: RecordValue) -> Result<(), TableError> {
250        let loaded = self.ensure_loaded_mut()?;
251        loaded.rows.push(row);
252        loaded.undefined_cells.push(HashSet::new());
253        self.persisted_row_count = loaded.rows.len();
254        self.lazy_rows = None;
255        Ok(())

280    pub(crate) fn scalar_cell(
281        &self,
282        row_index: usize,
283        column: &str,
284    ) -> Result<LazyScalarLookup<'_>, TableError> {
285        if let Some(loaded) = self.loaded_rows.get() {
286            let Some(row) = loaded.rows.get(row_index) else {
287                return Ok(LazyScalarLookup::Missing);
288            };
289            return Ok(match row.get(column) {
290                Some(Value::Scalar(scalar)) => LazyScalarLookup::Hit(scalar),
291                Some(_) => LazyScalarLookup::Unknown,
292                None => LazyScalarLookup::Missing,
293            });
294        }
295
296        let Some(source) = &self.lazy_rows else {
297            return Ok(LazyScalarLookup::Unknown);
298        };
299
300        if self.loaded_scalar_columns.get().is_none() {
301            let loaded = Self::load_scalar_columns_now(source)?;
302            let _ = self.loaded_scalar_columns.set(loaded);
303        }
304        let columns = self
305            .loaded_scalar_columns
306            .get()
307            .expect("scalar columns initialized before access");
308        let Some(values) = columns.columns.get(column) else {
309            return Ok(LazyScalarLookup::Unknown);
310        };
311        match values.get(row_index) {
312            Some(Some(value)) => Ok(LazyScalarLookup::Hit(value)),
313            Some(None) | None => Ok(LazyScalarLookup::Missing),
314        }
315    }

317    pub(crate) fn array_cell(
318        &self,
319        row_index: usize,
320        column: &str,
321    ) -> Result<LazyArrayLookup<'_>, TableError> {
322        if let Some(loaded) = self.loaded_rows.get() {
323            let Some(row) = loaded.rows.get(row_index) else {
324                return Ok(LazyArrayLookup::Missing);
325            };
326            return Ok(match row.get(column) {
327                Some(Value::Array(array)) => LazyArrayLookup::Hit(array),
328                Some(_) => LazyArrayLookup::Unknown,
329                None => LazyArrayLookup::Missing,
330            });
331        }
332
333        let Some(source) = &self.lazy_rows else {
334            return Ok(LazyArrayLookup::Unknown);
335        };
336        let Some(cached_column) = self.loaded_array_columns.get(column) else {
337            return Ok(LazyArrayLookup::Unknown);
338        };
339        if cached_column.get().is_none() {
340            let loaded = Self::load_array_column_now(source, column)?;
341            let _ = cached_column.set(loaded);
342        }
343        let values = cached_column
344            .get()
345            .expect("array column initialized before access");
346        match values.get(row_index) {
347            Some(Some(value)) => Ok(LazyArrayLookup::Hit(value)),
348            Some(None) | None => Ok(LazyArrayLookup::Missing),
349        }
350    }

370    pub(crate) fn remove_row(&mut self, index: usize) -> Result<RecordValue, TableError> {
371        let loaded = self.ensure_loaded_mut()?;
372        loaded.undefined_cells.remove(index);
373        let removed = loaded.rows.remove(index);
374        self.persisted_row_count = loaded.rows.len();
375        self.lazy_rows = None;
376        Ok(removed)
377    }

380        let loaded = self.ensure_loaded_mut()?;
381        loaded.rows.insert(index, row);
382        loaded.undefined_cells.insert(index, HashSet::new());
383        self.persisted_row_count = loaded.rows.len();
384        self.lazy_rows = None;
385        Ok(())

426    /// Replace all inner state from a storage snapshot.
427    ///
428    /// Used by `Table::lock()` when sync data indicates another process
429    /// modified the table and a full reload is needed.
430    pub(crate) fn replace_from_snapshot(
431        &mut self,
432        rows: Vec<RecordValue>,
433        undefined_cells: Vec<HashSet<String>>,
434        keywords: RecordValue,
435        column_keywords: HashMap<String, RecordValue>,
436        schema: Option<TableSchema>,
437    ) {
438        self.loaded_rows = {
439            let loaded_rows = OnceCell::new();
440            loaded_rows
441                .set(eager_loaded_rows(rows, undefined_cells))
442                .expect("replace eager row store");
443            loaded_rows
444        };
445        self.loaded_scalar_columns = OnceCell::new();
446        self.loaded_array_columns = lazy_array_column_store(schema.as_ref());
447        self.persisted_row_count = self.loaded_rows.get().map_or(0, |loaded| loaded.rows.len());
448        self.lazy_rows = None;
449        self.keywords = keywords;
450        self.column_keywords = column_keywords;
451        self.schema = schema;
452    }
