106#[derive(Debug)]
107struct RecentCache<K, V> {
108    capacity: usize,
109    values: HashMap<K, V>,
110    order: VecDeque<K>,
111}
112
113impl<K, V> RecentCache<K, V>
114where
115    K: Clone + Eq + Hash,
116{
117    fn new(capacity: usize) -> Self {
118        Self {
119            capacity: capacity.max(1),
120            values: HashMap::new(),
121            order: VecDeque::new(),
122        }
123    }
124
125    fn get(&mut self, key: &K) -> Option<V>
126    where
127        V: Clone,
128    {
129        let value = self.values.get(key).cloned()?;
130        self.touch(key);
131        Some(value)
132    }
133
134    fn insert(&mut self, key: K, value: V) {
135        if self.values.contains_key(&key) {
136            self.values.insert(key.clone(), value);
137            self.touch(&key);
138            return;
139        }
140        self.values.insert(key.clone(), value);
141        self.order.push_back(key);
142        self.evict_if_needed();
143    }
144
145    fn contains_key(&self, key: &K) -> bool {
146        self.values.contains_key(key)
147    }
148
149    fn touch(&mut self, key: &K) {
150        if let Some(index) = self.order.iter().position(|existing| existing == key)
151            && let Some(existing) = self.order.remove(index)
152        {
153            self.order.push_back(existing);
154        }
155    }
156
157    fn evict_if_needed(&mut self) {
158        while self.values.len() > self.capacity {
159            let Some(oldest) = self.order.pop_front() else {
160                break;
161            };
162            self.values.remove(&oldest);
163        }
164    }
165}

223struct PlanePrefetchWorker {
224    request_txs: Vec<Sender<PlanePrefetchRequest>>,
225    response_rx: Receiver<PlanePrefetchResult>,
226    queued: HashSet<PlaneCacheKey>,
227    next_worker: usize,

230impl PlanePrefetchWorker {
231    fn new(path: &Path) -> Result<Self, ImageError> {
232        let (response_tx, response_rx) = mpsc::channel::<PlanePrefetchResult>();
233        let path = path.to_path_buf();
234        let worker_count = thread::available_parallelism()
235            .map(|parallelism| parallelism.get().clamp(1, PREFETCH_MAX_WORKERS))
236            .unwrap_or(1);
237        let mut request_txs = Vec::with_capacity(worker_count);
238        for worker_index in 0..worker_count {
239            let (request_tx, request_rx) = mpsc::channel::<PlanePrefetchRequest>();
240            let worker_path = path.clone();
241            let worker_response_tx = response_tx.clone();
242            thread::Builder::new()
243                .name(format!("imexplore-plane-prefetch-{worker_index}"))
244                .spawn(move || {
245                    run_plane_prefetch_worker(worker_path, request_rx, worker_response_tx)
246                })
247                .map_err(|error| {
248                    ImageError::InvalidMetadata(format!("failed to spawn prefetch worker: {error}"))
249                })?;
250            request_txs.push(request_tx);
251        }
252        Ok(Self {
253            request_txs,
254            response_rx,
255            queued: HashSet::new(),
256            next_worker: 0,
257        })
258    }
259
260    fn submit(
261        &mut self,
262        request: PlanePrefetchRequest,
263        cache: &RecentCache<PlaneCacheKey, PlaneRaster>,
264    ) {
265        if cache.contains_key(&request.key) || self.queued.contains(&request.key) {
266            return;
267        }
268        let worker_count = self.request_txs.len().max(1);
269        let worker_index = self.next_worker % worker_count;
270        self.next_worker = (self.next_worker + 1) % worker_count;
271        if self.request_txs[worker_index].send(request.clone()).is_ok() {
272            self.queued.insert(request.key);
273        }
274    }
275
276    fn drain_into(
277        &mut self,
278        cache: &mut RecentCache<PlaneCacheKey, PlaneRaster>,
279        prefetched_keys: &mut Vec<PlaneCacheKey>,
280    ) {
281        while let Ok(result) = self.response_rx.try_recv() {
282            self.queued.remove(&result.key);
283            if let Ok(raster) = result.raster {
284                prefetched_keys.push(result.key.clone());
285                cache.insert(result.key, raster);
286            }
287        }
288    }
289}

291fn run_plane_prefetch_worker(
292    path: PathBuf,
293    request_rx: Receiver<PlanePrefetchRequest>,
294    response_tx: Sender<PlanePrefetchResult>,
295) {
296    let Ok(view) = OpenedImageView::open_with_cache_bytes(&path, PREFETCH_CACHE_BYTES) else {
297        return;
298    };
299    while let Ok(request) = request_rx.recv() {
300        let raster = view
301            .render_plane_with_window_and_axes_and_stretch(
302                request.viewport,
303                &request.window,
304                &request.non_display_indices,
305                &request.stretch,
306                request.clip_override,
307                request.region.as_ref(),
308            )
309            .map_err(|error| error.to_string());
310        if response_tx
311            .send(PlanePrefetchResult {
312                key: request.key,
313                raster,
314            })
315            .is_err()
316        {
317            break;
318        }
319    }
320}

345fn normalize_non_display_indices(
346    view: &OpenedImageView,
347    window: &ImageViewWindow,
348    requested: &[usize],
349) -> Vec<usize> {
350    view.axis_model()
351        .non_display_axes
352        .iter()
353        .enumerate()
354        .map(|(position, axis)| {
355            let length = window.sampled_axis_len(*axis);
356            let requested_index = requested.get(position).copied().unwrap_or_default();
357            if length == 0 {
358                0
359            } else {
360                requested_index.min(length.saturating_sub(1))
361            }
362        })
363        .collect()
364}

375    /// Opens a new browser session rooted at the provided image path and parameters.
376    pub fn open_with_parameters(
377        path: impl AsRef<Path>,
378        viewport: ImageBrowserViewport,
379        parameters: Option<&ImageBrowserParameters>,
380    ) -> Result<Self, ImageError> {
381        let path = path.as_ref();
382        let view = OpenedImageView::open(path)?;
383        let non_display_axis_count = session_non_display_axis_count(&view);
384        let stretch = parameters
385            .map(parse_stretch_parameters)
386            .transpose()?
387            .unwrap_or_default();
388        let window = match parameters {
389            Some(parameters) => {
390                view.window_from_text(&parameters.blc, &parameters.trc, &parameters.inc)?
391            }
392            None => view.default_window(),
393        };
394        let default_display_pixels = centered_display_pixels(&view, &window);
395        let default_non_display_pixels = centered_non_display_pixels(&view, &window);
396        let active_view = if view.capabilities().renderable_plane {
397            ImageBrowserView::Plane
398        } else {
399            ImageBrowserView::Metadata
400        };
401        let perf_enabled = image_browser_perf_enabled();
402        let mut session = Self {
403            prefetch_worker: Some(PlanePrefetchWorker::new(view.path())?),
404            view,
405            window,
406            stretch,
407            frozen_clip_bounds: None,
408            active_view,
409            focus: ImageBrowserFocus::Content,
410            viewport,
411            cursor_x: 0,
412            cursor_y: 0,
413            non_display_indices: vec![0; non_display_axis_count],
414            selected_profile_axis: None,
415            content_offset: 0,
416            plane_content_mode: default_plane_content_mode(viewport),
417            region: None,
418            region_revision: 0,
419            mask_revision: 0,
420            active_region_definition_name: None,
421            saved_region_cycle_index: 0,
422            perf_enabled,
423            plane_cache: RecentCache::new(SESSION_PLANE_CACHE_CAPACITY),
424            prefetched_plane_keys: HashSet::new(),
425            profile_cache: RecentCache::new(SESSION_PROFILE_CACHE_CAPACITY),
426            profile_perf: ProfilePerfAggregate::default(),
427            last_non_display_step: None,
428        };
429        session.selected_profile_axis = session.view.preferred_profile_axis();
430        session.clamp_cursor_to_window(
431            None,
432            None,
433            default_display_pixels,
434            Some(default_non_display_pixels),
435        );
436        Ok(session)
437    }

450    /// Builds a stateless preview snapshot for a specific non-display-axis occurrence.
451    pub fn preview_occurrence(
452        &mut self,
453        request: &ImageBrowserPreviewRequest,
454    ) -> Result<ImageBrowserPreviewPayload, ImageError> {
455        let visible_display_pixels = self.current_display_pixels();
456        let visible_non_display_pixels = self.current_non_display_pixels();
457        let saved = SessionPreviewState {
458            viewport: self.viewport,
459            window: self.window.clone(),
460            stretch: self.stretch.clone(),
461            frozen_clip_bounds: self.frozen_clip_bounds,
462            plane_content_mode: self.plane_content_mode,
463            non_display_indices: self.non_display_indices.clone(),
464            content_offset: self.content_offset,
465        };
466
467        self.viewport = request.viewport;
468        self.stretch = parse_stretch_parameters(&request.parameters)?;
469        self.frozen_clip_bounds = None;
470        self.window = self.view.window_from_text(
471            &request.parameters.blc,
472            &request.parameters.trc,
473            &request.parameters.inc,
474        )?;
475        self.plane_content_mode = request.plane_content_mode;
476        self.content_offset = 0;
477        let preview_non_display_indices =
478            normalize_non_display_indices(&self.view, &self.window, &request.non_display_indices);
479        self.non_display_indices = preview_non_display_indices.clone();
480        self.clamp_cursor_to_window(visible_display_pixels, None, None, None);
481        self.non_display_indices = preview_non_display_indices.clone();
482
483        let snapshot = self.snapshot_with_profile(request.include_profile);
484
485        self.viewport = saved.viewport;
486        self.window = saved.window;
487        self.stretch = saved.stretch;
488        self.frozen_clip_bounds = saved.frozen_clip_bounds;
489        self.plane_content_mode = saved.plane_content_mode;
490        self.non_display_indices = saved.non_display_indices.clone();
491        self.content_offset = saved.content_offset;
492        self.clamp_cursor_to_window(
493            visible_display_pixels,
494            visible_non_display_pixels,
495            None,
496            None,
497        );
498        self.non_display_indices = saved.non_display_indices;
499
500        let snapshot = snapshot?;
501        Ok(ImageBrowserPreviewPayload {
502            non_display_indices: preview_non_display_indices,
503            snapshot: Box::new(snapshot),
504        })
505    }

538            }

643    fn snapshot_with_profile(
644        &mut self,
645        include_profile: bool,
646    ) -> Result<ImageBrowserSnapshot, ImageError> {
647        self.drain_prefetched_planes();
648        let mut backend_timing = self.perf_enabled.then_some(ImageBackendTimingState {
649            plane_cache_result: ImageBackendPlaneCacheResult::Miss,
650            cached_plane_lookup_ns: 0,
651            plane_extract_ns: 0,
652            stat_collection_ns: 0,
653            histogram_ns: 0,
654            rasterize_ns: 0,
655            total_plane_ns: 0,
656            profile_cache_hits: self.profile_perf.cache_hits,
657            profile_cache_misses: self.profile_perf.cache_misses,
658            profile_extract_total_ns: self.profile_perf.extract_total_ns,
659        });
660        let plane_raster = if self.active_view == ImageBrowserView::Plane
661            && self.view.capabilities().renderable_plane
662        {
663            let clip_override = match self.stretch.autoscale {
664                PlaneAutoscaleMode::PerPlane => None,
665                PlaneAutoscaleMode::Frozen => self.frozen_clip_bounds,
666            };
667            let raster = self.cached_plane_raster(clip_override, backend_timing.as_mut())?;
668            if self.stretch.autoscale == PlaneAutoscaleMode::Frozen
669                && self.frozen_clip_bounds.is_none()
670                && !raster.no_finite_values
671            {
672                self.frozen_clip_bounds = Some((raster.clip_min, raster.clip_max));
673            }
674            Some(raster)
675        } else {
676            None
677        };
678        let mut inspector_lines = self.inspector_lines()?;
679        inspector_lines.extend(self.plane_display_lines(plane_raster.as_ref()));
680        let profile = if include_profile
681            && self.view.capabilities().renderable_plane
682            && matches!(
683                self.active_view,
684                ImageBrowserView::Plane | ImageBrowserView::Spectrum
685            ) {
686            let closed_region = self.region.as_ref().and_then(|region| {
687                region
688                    .shapes
689                    .iter()
690                    .any(|shape| shape.closed && shape.vertices.len() >= 3)
691                    .then_some(region.clone())
692            });
693            self.selected_profile_axis()
694                .map(|profile_axis| {
695                    self.cached_profile(
696                        (self.cursor_x, self.cursor_y),
697                        profile_axis,
698                        closed_region.as_ref(),
699                        backend_timing.as_mut(),
700                    )
701                })
702                .transpose()?
703        } else {
704            None
705        };
706        let inspector_height = if self.viewport.inspector_height == 0 {
707            inspector_lines.len().max(1)
708        } else {
709            self.viewport.inspector_height as usize
710        };
711        let content_height = self.viewport.height.max(1) as usize;
712        let all_content_lines = self.all_content_lines(plane_raster.as_ref(), profile.as_ref())?;
713        let total_items = all_content_lines.len();
714        let max_offset = total_items.saturating_sub(content_height);
715        let content_offset = self.content_offset.min(max_offset);
716        let content_lines = clip_lines(&all_content_lines, content_offset, content_height);
717        let plane = plane_raster.map(map_plane_raster);
718        let display_axes = self
719            .view
720            .display_axes_with_window(&self.window)
721            .into_iter()
722            .map(map_display_axis)
723            .collect::<Vec<_>>();
724        let probe = if self.active_view == ImageBrowserView::Plane
725            && self.view.capabilities().renderable_plane
726        {
727            Some(map_probe(self.view.probe_with_window_and_axes(
728                (self.cursor_x, self.cursor_y),
729                &self.window,
730                &self.non_display_indices,
731            )?))
732        } else {
733            None
734        };
735        let non_display_axes = self
736            .view
737            .non_display_axes_with_window(&self.window, &self.non_display_indices)?
738            .into_iter()
739            .map(map_non_display_axis)
740            .collect::<Vec<_>>();
741        let region = self.region_state()?;
742
743        self.schedule_plane_prefetch();
744
745        let navigation = if self.active_view == ImageBrowserView::Plane
746            && self.view.capabilities().renderable_plane
747        {
748            if let Some(display_axes) = self.view.axis_model().display_axes {
749                ImageNavigationMetrics {
750                    selected_index: self.cursor_y.min(
751                        self.window
752                            .sampled_axis_len(display_axes[1])
753                            .saturating_sub(1),
754                    ),
755                    total_items: self.window.sampled_axis_len(display_axes[1]),
756                    viewport_items: content_height,
757                }
758            } else {
759                ImageNavigationMetrics {
760                    selected_index: 0,
761                    total_items: 0,
762                    viewport_items: content_height,
763                }
764            }
765        } else {
766            ImageNavigationMetrics {
767                selected_index: content_offset,
768                total_items,
769                viewport_items: content_height,
770            }
771        };
772
773        Ok(ImageBrowserSnapshot {
774            status_line: self.view.status_line(),
775            active_view: self.active_view,
776            focus: self.focus,
777            shape: self.view.shape().to_vec(),
778            parameters: self.parameter_state(),
779            inspector_lines: clip_lines(&inspector_lines, 0, inspector_height),
780            content_lines,
781            navigation,
782            plane,
783            probe,
784            profile: profile.as_ref().map(map_profile),
785            display_axes,
786            plane_cursor: self.current_plane_cursor(),
787            non_display_axes,
788            region,
789            saved_region_names: self.view.saved_region_names(),
790            active_region_definition_name: self.active_region_definition_name.clone(),
791            mask_names: self.view.mask_names(),

833    fn move_cursor(&mut self, dx: i32, dy: i32) -> Result<(), ImageError> {
834        if self.active_view == ImageBrowserView::Plane
835            && self.focus == ImageBrowserFocus::Content
836            && self.view.capabilities().renderable_plane
837        {
838            let Some(display_axes) = self.view.axis_model().display_axes else {
839                return Ok(());
840            };
841            let max_x = self
842                .window
843                .sampled_axis_len(display_axes[0])
844                .saturating_sub(1) as i32;
845            let max_y = self
846                .window
847                .sampled_axis_len(display_axes[1])
848                .saturating_sub(1) as i32;
849            self.cursor_x = (self.cursor_x as i32 + dx).clamp(0, max_x) as usize;
850            self.cursor_y = (self.cursor_y as i32 + dy).clamp(0, max_y) as usize;
851            return Ok(());
852        }
853

855        let total_items = self.all_content_lines(None, None)?.len();
856        let max_offset = total_items.saturating_sub(content_height) as i32;
857        self.content_offset = (self.content_offset as i32 + dy).clamp(0, max_offset) as usize;
858        Ok(())
859    }

861    fn step_non_display_axis(&mut self, axis: usize, delta: i32) {
862        let Some(position) = self
863            .view
864            .axis_model()
865            .non_display_axes
866            .iter()
867            .position(|candidate| *candidate == axis)
868        else {
869            return;
870        };
871        let length = self.window.sampled_axis_len(axis);
872        if length == 0 {
873            self.non_display_indices[position] = 0;
874            return;
875        }
876        self.non_display_indices[position] = (self.non_display_indices[position] as i32 + delta)
877            .clamp(0, length.saturating_sub(1) as i32)
878            as usize;
879        self.last_non_display_step = (delta != 0).then_some((axis, delta.signum()));
880    }

1130    fn cached_plane_raster(
1131        &mut self,
1132        clip_override: Option<(f64, f64)>,
1133        backend_timing: Option<&mut ImageBackendTimingState>,
1134    ) -> Result<PlaneRaster, ImageError> {
1135        let key = self.plane_cache_key(self.non_display_indices.clone(), clip_override);
1136        let lookup_started_at = Instant::now();
1137        if let Some(raster) = self.plane_cache.get(&key) {
1138            if let Some(timing) = backend_timing {
1139                timing.plane_cache_result = if self.prefetched_plane_keys.contains(&key) {
1140                    ImageBackendPlaneCacheResult::PrefetchHit
1141                } else {
1142                    ImageBackendPlaneCacheResult::Hit
1143                };
1144                timing.cached_plane_lookup_ns = duration_ns(lookup_started_at.elapsed());
1145                timing.total_plane_ns = timing.cached_plane_lookup_ns;
1146            }
1147            return Ok(raster);
1148        }
1149        self.prefetched_plane_keys.remove(&key);
1150        let (raster, telemetry) = self
1151            .view
1152            .render_plane_with_window_and_axes_and_stretch_timed(
1153                self.plane_pixel_viewport(),
1154                &self.window,
1155                &self.non_display_indices,
1156                &self.stretch.plane_settings(),
1157                clip_override,
1158                self.region.as_ref(),
1159            )?;
1160        if let Some(timing) = backend_timing {
1161            apply_plane_timing(
1162                timing,
1163                ImageBackendPlaneCacheResult::Miss,
1164                duration_ns(lookup_started_at.elapsed()),
1165                telemetry,
1166            );
1167        }
1168        self.plane_cache.insert(key, raster.clone());
1169        Ok(raster)
1170    }

1172    fn cached_profile(
1173        &mut self,
1174        pixel_xy: (usize, usize),
1175        profile_axis: usize,
1176        region: Option<&ImageRegion>,
1177        backend_timing: Option<&mut ImageBackendTimingState>,
1178    ) -> Result<ImageProfile, ImageError> {
1179        let key = self.profile_cache_key(pixel_xy, profile_axis, region);
1180        if let Some(mut profile) = self.profile_cache.get(&key) {
1181            self.profile_perf.cache_hits = self.profile_perf.cache_hits.saturating_add(1);
1182            if let Some(timing) = backend_timing {
1183                timing.profile_cache_hits = self.profile_perf.cache_hits;
1184                timing.profile_cache_misses = self.profile_perf.cache_misses;
1185                timing.profile_extract_total_ns = self.profile_perf.extract_total_ns;
1186            }
1187            profile.selected_sample_index = self.current_profile_selected_index(profile_axis);
1188            return Ok(profile);
1189        }
1190        self.profile_perf.cache_misses = self.profile_perf.cache_misses.saturating_add(1);
1191        let normalized_non_display_indices =
1192            self.normalized_profile_non_display_indices(profile_axis);
1193        let extract_started_at = Instant::now();
1194        let mut profile = if let Some(region) = region {
1195            if let Some(profile) = self.view.region_profile_with_window_and_axes(
1196                region,
1197                &self.window,
1198                &normalized_non_display_indices,
1199                profile_axis,
1200            )? {
1201                profile
1202            } else {
1203                self.view.profile_with_window_and_axes(
1204                    pixel_xy,
1205                    &self.window,
1206                    &normalized_non_display_indices,
1207                    profile_axis,
1208                )?
1209            }
1210        } else {
1211            self.view.profile_with_window_and_axes(
1212                pixel_xy,
1213                &self.window,
1214                &normalized_non_display_indices,
1215                profile_axis,
1216            )?
1217        };
1218        self.profile_perf.extract_total_ns = self
1219            .profile_perf
1220            .extract_total_ns
1221            .saturating_add(duration_ns(extract_started_at.elapsed()));
1222        profile.selected_sample_index = self.current_profile_selected_index(profile_axis);
1223        self.profile_cache.insert(key, profile.clone());
1224        if let Some(timing) = backend_timing {
1225            timing.profile_cache_hits = self.profile_perf.cache_hits;
1226            timing.profile_cache_misses = self.profile_perf.cache_misses;
1227            timing.profile_extract_total_ns = self.profile_perf.extract_total_ns;
1228        }
1229        Ok(profile)
1230    }

1307    fn schedule_plane_prefetch(&mut self) {
1308        if !self.view.capabilities().renderable_plane {
1309            return;
1310        }
1311        let Some(selected_axis) = self.selected_profile_axis() else {
1312            return;
1313        };
1314        let Ok(non_display_axes) = self
1315            .view
1316            .non_display_axes_with_window(&self.window, &self.non_display_indices)
1317        else {
1318            return;
1319        };
1320        let Some(axis_state) = non_display_axes
1321            .into_iter()
1322            .find(|axis_state| axis_state.axis == selected_axis)
1323        else {
1324            return;
1325        };
1326        let Some(axis_position) = self
1327            .view
1328            .axis_model()
1329            .non_display_axes
1330            .iter()
1331            .position(|axis| *axis == axis_state.axis)
1332        else {
1333            return;
1334        };
1335        let clip_override = match self.stretch.autoscale {
1336            PlaneAutoscaleMode::PerPlane => None,
1337            PlaneAutoscaleMode::Frozen => self.frozen_clip_bounds,
1338        };
1339        let viewport = self.plane_pixel_viewport();
1340        let window = self.window.clone();
1341        let stretch = self.stretch.plane_settings();
1342        let region = self.region.clone();
1343        let preferred_direction = self
1344            .last_non_display_step
1345            .filter(|(axis, direction)| *axis == axis_state.axis && *direction != 0)
1346            .map(|(_, direction)| direction);
1347        let prefetch_offsets = if let Some(direction) = preferred_direction {
1348            let primary_direction = if direction < 0 { -1isize } else { 1isize };
1349            (1..=PREFETCH_FORWARD_DISTANCE)
1350                .map(|offset| (offset, primary_direction))
1351                .chain(
1352                    (1..=PREFETCH_REVERSE_DISTANCE).map(move |offset| (offset, -primary_direction)),
1353                )
1354                .collect::<Vec<_>>()
1355        } else {
1356            (1..=PREFETCH_DISTANCE)
1357                .flat_map(|offset| {
1358                    [-1isize, 1isize]
1359                        .into_iter()
1360                        .map(move |direction| (offset, direction))
1361                })
1362                .collect::<Vec<_>>()
1363        };
1364        let plane_cache_keys = prefetch_offsets
1365            .into_iter()
1366            .filter_map(|(offset, direction)| {
1367                let next_index = if direction.is_negative() {
1368                    axis_state.index.checked_sub(offset)
1369                } else {
1370                    axis_state.index.checked_add(offset)
1371                }?;
1372                (next_index < axis_state.length).then_some(next_index)
1373            })
1374            .map(|next_index| {
1375                let mut indices = self.non_display_indices.clone();
1376                if let Some(index) = indices.get_mut(axis_position) {
1377                    *index = next_index;
1378                }
1379                let key = self.plane_cache_key(indices.clone(), clip_override);
1380                PlanePrefetchRequest {
1381                    key,
1382                    viewport,
1383                    window: window.clone(),
1384                    non_display_indices: indices,
1385                    stretch: stretch.clone(),
1386                    clip_override,
1387                    region: region.clone(),
1388                }
1389            })
1390            .collect::<Vec<_>>();
1391        let Some(worker) = self.prefetch_worker.as_mut() else {
1392            return;
1393        };
1394        for request in plane_cache_keys {
1395            worker.submit(request, &self.plane_cache);
1396        }
1397    }

2426fn parse_crtf_region_coordinate(text: &str) -> Result<CrtfRegionCoordinate, ImageError> {
2427    let trimmed = text.trim();
2428    let lowercase = trimmed.to_ascii_lowercase();
2429    if let Some(value) = lowercase.strip_suffix("pix") {
2430        let parsed = value.trim().parse::<f64>().map_err(|error| {
2431            ImageError::InvalidMetadata(format!(
2432                "invalid CRTF pixel coordinate {trimmed:?}: {error}"
2433            ))
2434        })?;
2435        return rounded_region_pixel(parsed, "CRTF").map(CrtfRegionCoordinate::Pixel);
2436    }
2437    let (value, scale) = if let Some(value) = lowercase.strip_suffix("arcsec") {
2438        (value, std::f64::consts::PI / 180.0 / 3600.0)
2439    } else if let Some(value) = lowercase.strip_suffix("arcmin") {
2440        (value, std::f64::consts::PI / 180.0 / 60.0)
2441    } else if let Some(value) = lowercase.strip_suffix("deg") {
2442        (value, std::f64::consts::PI / 180.0)
2443    } else if let Some(value) = lowercase.strip_suffix("rad") {
2444        (value, 1.0)
2445    } else {
2446        (lowercase.as_str(), 1.0)
2447    };
2448    let parsed = value.trim().parse::<f64>().map_err(|error| {
2449        ImageError::InvalidMetadata(format!(
2450            "invalid CRTF world coordinate {trimmed:?}: {error}"
2451        ))
2452    })?;
2453    if !parsed.is_finite() {
2454        return Err(ImageError::InvalidMetadata(format!(
2455            "invalid CRTF world coordinate {trimmed:?}: value is not finite"
2456        )));
2457    }
2458    Ok(CrtfRegionCoordinate::World(parsed * scale))
2459}

2509fn parse_stretch_parameters(
2510    parameters: &ImageBrowserParameters,
2511) -> Result<SessionStretchState, ImageError> {
2512    let preset = match parameters.stretch.trim() {
2513        "" | "percentile99" => PlaneStretchPreset::Percentile99,
2514        "percentile95" => PlaneStretchPreset::Percentile95,
2515        "minmax" => PlaneStretchPreset::MinMax,
2516        "zscale" => PlaneStretchPreset::ZScale,
2517        "manual" => PlaneStretchPreset::Manual,
2518        other => {
2519            return Err(ImageError::InvalidMetadata(format!(
2520                "unsupported stretch preset: {other}"
2521            )));
2522        }
2523    };
2524    let autoscale = match parameters.autoscale.trim() {
2525        "" | "per_plane" => PlaneAutoscaleMode::PerPlane,
2526        "frozen" => PlaneAutoscaleMode::Frozen,
2527        other => {
2528            return Err(ImageError::InvalidMetadata(format!(
2529                "unsupported autoscale mode: {other}"
2530            )));
2531        }
2532    };
2533    let clip_low = parse_optional_clip("clip_low", &parameters.clip_low)?;
2534    let clip_high = parse_optional_clip("clip_high", &parameters.clip_high)?;
2535    let manual_clip = match (clip_low, clip_high) {
2536        (Some(low), Some(high)) if low < high => Some((low, high)),
2537        (Some(_), Some(_)) => {
2538            return Err(ImageError::InvalidMetadata(
2539                "clip_low must be smaller than clip_high".into(),
2540            ));
2541        }
2542        (None, None) => None,
2543        _ => {
2544            return Err(ImageError::InvalidMetadata(
2545                "manual clip requires both clip_low and clip_high".into(),
2546            ));
2547        }
2548    };
2549    if preset == PlaneStretchPreset::Manual && manual_clip.is_none() {
2550        return Err(ImageError::InvalidMetadata(
2551            "manual stretch requires clip_low and clip_high".into(),
2552        ));
2553    }
2554    Ok(SessionStretchState {
2555        preset,
2556        autoscale,
2557        manual_clip,
2558    })
2559}

2561fn parse_optional_clip(field: &str, value: &str) -> Result<Option<f64>, ImageError> {
2562    let value = value.trim();
2563    if value.is_empty() {
2564        return Ok(None);
2565    }
2566    let parsed = value.parse::<f64>().map_err(|error| {
2567        ImageError::InvalidMetadata(format!("invalid {field} value '{value}': {error}"))
2568    })?;
2569    if !parsed.is_finite() {
2570        return Err(ImageError::InvalidMetadata(format!(
2571            "{field} must be finite"
2572        )));
2573    }
2574    Ok(Some(parsed))
2575}
