2//! Canonical image-expression operator semantics.
3
4use crate::{ImageExprBinaryOp, ImageExprUnaryOp, image_expr::ImageExprValue};
5
6pub(crate) fn apply_unary<T: ImageExprValue>(op: ImageExprUnaryOp, value: T) -> T {
7    match op {
8        ImageExprUnaryOp::Negate => -value,
9        ImageExprUnaryOp::Exp => value.expr_exp(),
10        ImageExprUnaryOp::Sin => value.expr_sin(),
11        ImageExprUnaryOp::Cos => value.expr_cos(),
12        ImageExprUnaryOp::Tan => value.expr_tan(),
13        ImageExprUnaryOp::Asin => value.expr_asin(),
14        ImageExprUnaryOp::Acos => value.expr_acos(),
15        ImageExprUnaryOp::Atan => value.expr_atan(),
16        ImageExprUnaryOp::Sinh => value.expr_sinh(),
17        ImageExprUnaryOp::Cosh => value.expr_cosh(),
18        ImageExprUnaryOp::Tanh => value.expr_tanh(),
19        ImageExprUnaryOp::Log => value.expr_log(),
20        ImageExprUnaryOp::Log10 => value.expr_log10(),
21        ImageExprUnaryOp::Sqrt => value.expr_sqrt(),
22        ImageExprUnaryOp::Abs => value.expr_abs(),
23        ImageExprUnaryOp::Ceil => value.expr_ceil(),
24        ImageExprUnaryOp::Floor => value.expr_floor(),
25        ImageExprUnaryOp::Round => value.expr_round(),
26        ImageExprUnaryOp::Sign => value.expr_sign(),
27        ImageExprUnaryOp::Conj => value.expr_conj(),
28    }
29}
30
31pub(crate) fn apply_binary<T: ImageExprValue>(op: ImageExprBinaryOp, lhs: T, rhs: T) -> T {
32    match op {
33        ImageExprBinaryOp::Add => lhs + rhs,
34        ImageExprBinaryOp::Subtract => lhs - rhs,
35        ImageExprBinaryOp::Multiply => lhs * rhs,
36        ImageExprBinaryOp::Divide => lhs / rhs,
37        ImageExprBinaryOp::Pow => lhs.expr_pow(rhs),
38        ImageExprBinaryOp::Fmod => lhs.expr_fmod(rhs),
39        ImageExprBinaryOp::Atan2 => lhs.expr_atan2(rhs),
40        ImageExprBinaryOp::Min => lhs.expr_min(rhs),
41        ImageExprBinaryOp::Max => lhs.expr_max(rhs),
42    }
43}
