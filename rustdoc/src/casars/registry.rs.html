79fn task_catalog_entries() -> &'static [TaskCatalogEntry] {
80    static CATALOG: OnceLock<Vec<TaskCatalogEntry>> = OnceLock::new();
81    CATALOG.get_or_init(|| {
82        serde_json::from_str::<TaskCatalog>(TASK_CATALOG_JSON)
83            .expect("resources/task-catalog.json should parse")
84            .tasks
85    })
86}

88fn registry_app_from_catalog(entry: &TaskCatalogEntry) -> Option<RegistryApp> {
89    if !entry.show_in_tui {
90        return None;
91    }
92    let shell_kind = match entry.shell_kind.as_str() {
93        "inspect" => AppShellKind::Inspect,
94        "browser" => AppShellKind::Browser,
95        "workflow" => AppShellKind::Workflow,
96        _ => return None,
97    };
98    let interaction = match entry.interaction.as_str() {
99        "one_shot" => AppInteraction::OneShot,
100        "browser_session" => {
101            let browser_kind = match entry.browser_kind.as_deref() {
102                Some("table") => BrowserAppKind::Table,
103                Some("image") => BrowserAppKind::Image,
104                _ => return None,
105            };
106            AppInteraction::BrowserSession(browser_kind)
107        }
108        _ => return None,
109    };
110    Some(RegistryApp {
111        id: entry.id.clone(),
112        category: entry.category.clone(),
113        display_name: entry.display_name.clone(),
114        shell_kind,
115        kind: RegistryAppKind::Subprocess {
116            binary_name: entry.binary_name.clone(),
117            cargo_package: entry.cargo_package.clone(),
118            override_env: entry.override_env.clone(),
119            interaction,
120        },
121    })
122}

124#[derive(Debug, Clone)]
125pub(crate) struct ResolvedCommand {
126    program: OsString,
127    prefix_args: Vec<OsString>,
128}
129
130impl ResolvedCommand {
131    pub(crate) fn command(&self) -> Command {
132        let mut command = Command::new(&self.program);
133        command.args(&self.prefix_args);
134        command
135    }
136
137    #[cfg(test)]
138    pub(crate) fn direct(program: impl Into<OsString>) -> Self {
139        Self {
140            program: program.into(),
141            prefix_args: Vec::new(),
142        }
143    }
144}

146impl RegistryApp {
147    /// Derive the form schema from the canonical built-in surface definition.
148    /// Provider subprocesses no longer supply a fallback schema or alias table.
149    pub(crate) fn load_schema(&self) -> Result<UiCommandSchema, String> {
150        let bundle = builtin_surface_bundle(&self.id)?;
151        serde_json::from_value(project_ui_schema(&bundle))
152            .map_err(|error| format!("project {} parameter definition: {error}", self.id))
153    }
154
155    pub(crate) fn resolve_command(&self) -> Result<ResolvedCommand, String> {
156        let RegistryAppKind::Subprocess {
157            binary_name,
158            cargo_package,
159            override_env,
160            ..
161        } = &self.kind;
162
163        if let Some(path) = env::var_os(override_env) {
164            return Ok(ResolvedCommand {
165                program: path,
166                prefix_args: Vec::new(),
167            });
168        }
169        if let Some(path) = env::var_os(format!("CARGO_BIN_EXE_{binary_name}")) {
170            return Ok(ResolvedCommand {
171                program: path,
172                prefix_args: Vec::new(),
173            });
174        }
175        if let Some(path) = sibling_binary(binary_name) {
176            if !self.prefers_cargo_workspace_fallback_for_stale_sibling()
177                || !sibling_binary_is_stale_for_current_process(&path)
178            {
179                return Ok(ResolvedCommand {
180                    program: path.into_os_string(),
181                    prefix_args: Vec::new(),
182                });
183            }
184        }
185
186        let cargo = env::var_os("CARGO").unwrap_or_else(|| OsString::from("cargo"));
187        Ok(ResolvedCommand {
188            program: cargo,
189            prefix_args: vec![
190                OsString::from("run"),
191                OsString::from("--manifest-path"),
192                workspace_manifest_path().into_os_string(),
193                OsString::from("-q"),
194                OsString::from("-p"),
195                OsString::from(cargo_package),
196                OsString::from("--bin"),
197                OsString::from(binary_name),
198                OsString::from("--"),
199            ],
200        })
201    }
202
203    fn prefers_cargo_workspace_fallback_for_stale_sibling(&self) -> bool {
204        matches!(self.id.as_str(), "msexplore" | "calibrate" | "importvla")
205    }
206
207    pub(crate) fn is_browser_session(&self) -> bool {
208        matches!(
209            self.kind,
210            RegistryAppKind::Subprocess {
211                interaction: AppInteraction::BrowserSession(_),
212                ..
213            }
214        )
215    }
216
217    pub(crate) fn shell_kind(&self) -> AppShellKind {
218        self.shell_kind
219    }
220
221    pub(crate) fn browser_kind(&self) -> Option<BrowserAppKind> {
222        match self.kind {
223            RegistryAppKind::Subprocess {
224                interaction: AppInteraction::BrowserSession(kind),
225                ..
226            } => Some(kind),
227            RegistryAppKind::Subprocess {
228                interaction: AppInteraction::OneShot,
229                ..
230            } => None,
231        }
232    }
233
234    pub(crate) fn browser_path_field_id(&self) -> Option<&'static str> {
235        match self.browser_kind()? {
236            BrowserAppKind::Table => Some("table"),
237            BrowserAppKind::Image => Some("image"),
238        }
239    }
240
241    pub(crate) fn ready_status_line(&self) -> &'static str {
242        match (self.shell_kind, &self.kind) {
243            (
244                AppShellKind::Browser,
245                RegistryAppKind::Subprocess {
246                    interaction: AppInteraction::BrowserSession(_),
247                    ..
248                },
249            ) => "Ready. Press r to open the browser session.",
250            (AppShellKind::Inspect, _) => {
251                "Ready. Press r to run and refresh the current inspection view."
252            }
253            (AppShellKind::Workflow, _) => "Ready. Press r to run the selected workflow stage.",
254            (_, _) => "Ready. Press r to run the selected command.",
255        }
256    }
257}

259pub(crate) fn resolve_app(id: Option<&str>) -> Result<RegistryApp, String> {
260    let requested = id.unwrap_or("msexplore");
261    registered_apps()
262        .into_iter()
263        .find(|app| app.id == requested)
264        .ok_or_else(|| {
265            let expected = registered_apps()
266                .into_iter()
267                .map(|app| app.id)
268                .collect::<Vec<_>>()
269                .join(", ");
270            format!("unknown casars app {requested:?}; expected one of: {expected}")
271        })
272}
273
274pub(crate) fn registered_apps() -> Vec<RegistryApp> {
275    task_catalog_entries()
276        .iter()
277        .filter_map(registry_app_from_catalog)
278        .collect()
279}

311fn sibling_binary(binary_name: &str) -> Option<PathBuf> {
312    let mut path = env::current_exe().ok()?;
313    path.pop();
314    path.push(binary_name);
315    path.set_extension(env::consts::EXE_EXTENSION);
316    if path.exists() { Some(path) } else { None }
317}
318
319fn workspace_manifest_path() -> PathBuf {
320    let manifest_dir = PathBuf::from(env!("CARGO_MANIFEST_DIR"));
321    manifest_dir
322        .parent()
323        .and_then(|path| path.parent())
324        .unwrap_or_else(|| {
325            panic!("casars manifest dir should live under <workspace>/crates/casars")
326        })
327        .join("Cargo.toml")
328}
329
330fn sibling_binary_is_stale_for_current_process(sibling_path: &std::path::Path) -> bool {
331    let current_exe = match env::current_exe() {
332        Ok(path) => path,
333        Err(_) => return false,
334    };
335    is_binary_stale(
336        file_modified_time(sibling_path),
337        file_modified_time(&current_exe),
338    )
339}
340
341fn file_modified_time(path: &std::path::Path) -> Option<SystemTime> {
342    fs::metadata(path).ok()?.modified().ok()
343}
344
345fn is_binary_stale(
346    binary_modified: Option<SystemTime>,
347    reference_modified: Option<SystemTime>,
348) -> bool {
349    match (binary_modified, reference_modified) {
350        (Some(binary_modified), Some(reference_modified)) => binary_modified < reference_modified,
351        _ => false,
352    }
353}

359    #[test]
360    fn inventory_and_parameter_catalog_resolve_the_same_42_surfaces() {

362        assert_eq!(apps.len(), 42);
363        for app in apps {

365            assert_eq!(schema.command_id, app.id);

369    #[test]
370    fn schema_projection_uses_casa_names_without_alias_enrichment() {
371        let imager = imager_app().load_schema().unwrap();
372        assert!(imager.argument("vis").is_some());
373        assert!(imager.argument("cell").is_some());
374        assert!(imager.argument("ms").is_none());
375        assert!(imager.argument("cell_arcsec").is_none());
376
377        // These catalog surfaces intentionally share executable routes without
378        // a provider surface selector. Their live family binary cannot
379        // distinguish the alias for `--ui-schema`, so frontend schema loading
380        // must project each catalog surface directly instead of inferring alias
381        // semantics from the provider-family schema.
382        let split_app = resolve_app(Some("split")).unwrap();
383        let mstransform_app = resolve_app(Some("mstransform")).unwrap();
384        let RegistryAppKind::Subprocess {
385            binary_name: split_binary,
386            ..
387        } = &split_app.kind;
388        let RegistryAppKind::Subprocess {
389            binary_name: mstransform_binary,
390            ..
391        } = &mstransform_app.kind;
392        assert_eq!(split_binary, mstransform_binary);
393        let split = split_app.load_schema().unwrap();
394        let mstransform = mstransform_app.load_schema().unwrap();
395        assert_eq!(split.command_id, "split");
396        assert_eq!(mstransform.command_id, "mstransform");
397        assert!(split.argument("vis").is_some());
398
399        let plotms_app = resolve_app(Some("plotms")).unwrap();
400        let msexplore_app = resolve_app(Some("msexplore")).unwrap();
401        let RegistryAppKind::Subprocess {
402            binary_name: plotms_binary,
403            ..
404        } = &plotms_app.kind;
405        let RegistryAppKind::Subprocess {
406            binary_name: msexplore_binary,
407            ..
408        } = &msexplore_app.kind;
409        assert_eq!(plotms_binary, msexplore_binary);
410        let plotms = plotms_app.load_schema().unwrap();
411        let msexplore = msexplore_app.load_schema().unwrap();
412        assert_eq!(plotms.command_id, "plotms");
413        assert_eq!(msexplore.command_id, "msexplore");
414    }
