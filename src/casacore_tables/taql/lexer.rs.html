2//! Peekable wrapper around the logos-generated lexer.
3//!
4//! [`Lexer`] adds single-token lookahead, span tracking, and
5//! source-position computation (line/column) for error messages.
6
7use logos::Logos;
8use std::ops::Range;
9
10use super::error::{SourcePos, TaqlError};
11use super::token::Token;
12
13/// A peekable wrapper around the logos-generated lexer.
14///
15/// Provides `peek()`, `next_token()`, `expect()`, `eat_if()` and
16/// position tracking for parser error messages.
17pub struct Lexer<'src> {
18    inner: logos::Lexer<'src, Token>,
19    peeked: Option<Option<(Token, Range<usize>)>>,
20    source: &'src str,
21}
22
23impl<'src> Lexer<'src> {
24    /// Creates a new lexer over the given TaQL source string.
25    pub fn new(source: &'src str) -> Self {
26        Self {
27            inner: Token::lexer(source),
28            peeked: None,
29            source,
30        }
31    }
32
33    /// Peeks at the next token without consuming it.
34    ///
35    /// Returns `None` at end-of-input or on a lex error.
36    pub fn peek(&mut self) -> Option<&Token> {
37        if self.peeked.is_none() {
38            self.peeked = Some(self.advance());
39        }
40        self.peeked.as_ref().unwrap().as_ref().map(|(t, _)| t)
41    }
42
43    /// Consumes and returns the next token and its span.
44    pub fn next_token(&mut self) -> Option<(Token, Range<usize>)> {
45        if let Some(peeked) = self.peeked.take() {
46            peeked
47        } else {
48            self.advance()
49        }
50    }
51
52    /// Returns the slice of source text for the given span.
53    pub fn slice(&self, span: &Range<usize>) -> &'src str {
54        &self.source[span.clone()]
55    }
56
57    /// Consumes the next token if it matches `expected`, returning its span.
58    pub fn eat_if(&mut self, expected: &Token) -> Option<Range<usize>> {
59        if self.peek() == Some(expected) {
60            self.next_token().map(|(_, s)| s)
61        } else {
62            None
63        }
64    }
65
66    /// Consumes the next token, requiring it to match `expected`.
67    pub fn expect(&mut self, expected: &Token) -> Result<Range<usize>, TaqlError> {
68        match self.next_token() {
69            Some((ref tok, span)) if tok == expected => Ok(span),
70            Some((tok, span)) => Err(TaqlError::parse(
71                self.position(span.start),
72                format!("expected {expected}, found {tok}"),
73            )),
74            None => Err(TaqlError::unexpected_end(format!("expected {expected}"))),
75        }
76    }
77
78    /// Returns true if the lexer has been exhausted.
79    pub fn is_eof(&mut self) -> bool {
80        self.peek().is_none()
81    }
82
83    /// Computes (1-based) line and column for a byte offset.
84    pub fn position(&self, offset: usize) -> SourcePos {
85        let prefix = &self.source[..offset.min(self.source.len())];
86        let line = prefix.matches('\n').count() + 1;
87        let col = prefix.rfind('\n').map_or(offset, |nl| offset - nl - 1) + 1;
88        SourcePos { line, col }
89    }
90
91    /// Advances the inner logos lexer by one token.
92    fn advance(&mut self) -> Option<(Token, Range<usize>)> {
93        let result = self.inner.next()?;
94        let span = self.inner.span();
95        match result {
96            Ok(token) => Some((token, span)),
97            Err(()) => {
98                // logos skip failures: return None to signal a lex error.
99                // The parser will interpret this as unexpected EOF or error.
100                None
101            }
102        }
103    }
104}
