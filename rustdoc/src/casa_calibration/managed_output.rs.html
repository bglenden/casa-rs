2//! Calibration task result envelope shared by launcher- and JSON-facing workflows.
3
4use schemars::JsonSchema;
5use serde::{Deserialize, Serialize};
6
7use crate::{
8    ApplyExecutionReport, ApplyPlan, BandpassSolveReport, CalibrationStatsReport,
9    CalibrationTableSummary, ContinuumSubtractionReport, ExportCorrectedDataReport,
10    FluxScaleReport, GainSolveReport, GencalReport,
11};
12
13/// Canonical structured result for one calibration task execution.
14#[derive(Debug, Clone, PartialEq, Serialize, Deserialize, JsonSchema)]
15#[serde(tag = "kind", content = "report", rename_all = "snake_case")]
16pub enum CalibrationTaskResult {
17    /// `calibrate execute_apply`
18    Apply(ApplyExecutionReport),
19    /// `calibrate export-corrected`
20    ExportCorrectedData(ExportCorrectedDataReport),
21    /// `calibrate uvcontsub`
22    ContinuumSubtract(ContinuumSubtractionReport),
23    /// `calibrate summary`
24    Summary(Vec<CalibrationTableSummary>),
25    /// `calibrate plan-apply`
26    PlanApply(ApplyPlan),
27    /// `calibrate stats`
28    Stats(CalibrationStatsReport),
29    /// `calibrate solve-gain`
30    SolveGain(GainSolveReport),
31    /// `calibrate solve-bandpass`
32    SolveBandpass(BandpassSolveReport),
33    /// `calibrate fluxscale`
34    FluxScale(FluxScaleReport),
35    /// `calibrate gencal`
36    Gencal(GencalReport),
37}
38
39/// Backward-compatible alias for the launcher-managed calibration output type.
40pub type ManagedCalibrationOutput = CalibrationTaskResult;
