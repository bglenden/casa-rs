4impl Table {
5    /// Opens an existing table from disk with locking.
6    ///
7    /// Behaves like [`open`](Table::open) but also creates or opens the
8    /// `table.lock` file and acquires a lock according to the given
9    /// [`LockOptions`].
10    ///
11    /// - [`LockMode::PermanentLocking`]: acquires a write lock immediately;
12    ///   fails if unavailable.
13    /// - [`LockMode::PermanentLockingWait`]: acquires a write lock, waiting
14    ///   indefinitely.
15    /// - [`LockMode::AutoLocking`]: acquires a read lock immediately; write
16    ///   operations temporarily acquire/release a write lock.
17    /// - [`LockMode::UserLocking`]: no lock is acquired until
18    ///   [`lock()`](Table::lock) is called.
19    /// - [`LockMode::NoLocking`]: equivalent to [`open()`](Table::open).
20    ///
21    /// C++ equivalent: `Table(name, TableLock(...), Table::Old)`.
22    #[cfg(unix)]
23    pub fn open_with_lock(
24        options: TableOptions,
25        lock_opts: LockOptions,
26    ) -> Result<Self, TableError> {
27        if lock_opts.mode == LockMode::NoLocking {
28            return Self::open(options);
29        }
30
31        let storage = CompositeStorage;
32        let row_hint = crate::lock::read_sync_data_from_table_dir(&options.path)
33            .map_err(|e| TableError::LockIo {
34                path: options.path.display().to_string(),
35                message: e.to_string(),
36            })?
37            .map(|sync| sync.nrrow);
38        let snapshot = storage.load_with_row_hint(&options.path, row_hint)?;
39        let info = snapshot.table_info;
40        let mut table = Self {
41            inner: TableImpl::with_rows_keywords_and_schema(
42                snapshot.rows,
43                snapshot.undefined_cells,
44                snapshot.keywords,
45                snapshot.column_keywords,
46                snapshot.schema,
47            ),
48            source_path: Some(options.path.clone()),
49            kind: TableKind::Plain,
50            virtual_columns: snapshot.virtual_columns,
51            virtual_bindings: Vec::new(),
52            table_info: info,
53            dm_info: snapshot.dm_info,
54            external_sync: None,
55            marked_for_delete: false,
56            lock_state: None,
57        };
58        table.validate()?;
59
60        let perm = matches!(
61            lock_opts.mode,
62            LockMode::PermanentLocking | LockMode::PermanentLockingWait
63        );
64        let mut lock_file =
65            LockFile::create_or_open(&options.path, false, lock_opts.inspection_interval, perm)
66                .map_err(|e| TableError::LockIo {
67                    path: options.path.display().to_string(),
68                    message: e.to_string(),
69                })?;
70
71        // Acquire initial lock based on mode.
72        match lock_opts.mode {
73            LockMode::PermanentLocking => {
74                if !lock_file
75                    .acquire(LockType::Write, 1)
76                    .map_err(|e| TableError::LockIo {
77                        path: options.path.display().to_string(),
78                        message: e.to_string(),
79                    })?
80                {
81                    return Err(TableError::LockFailed {
82                        path: options.path.display().to_string(),
83                        message: "table is locked by another process".into(),
84                    });
85                }
86            }
87            LockMode::PermanentLockingWait => {
88                if !lock_file
89                    .acquire(LockType::Write, 0)
90                    .map_err(|e| TableError::LockIo {
91                        path: options.path.display().to_string(),
92                        message: e.to_string(),
93                    })?
94                {
95                    return Err(TableError::LockFailed {
96                        path: options.path.display().to_string(),
97                        message: "could not acquire permanent lock".into(),
98                    });
99                }
100            }
101            LockMode::AutoLocking | LockMode::DefaultLocking => {
102                let _ = lock_file.acquire(LockType::Read, 1);
103            }
104            LockMode::AutoNoReadLocking => {
105                // Skip read lock on open — only write locks are acquired.
106            }
107            LockMode::UserLocking | LockMode::UserNoReadLocking | LockMode::NoLocking => {}
108        }
109
110        // Read sync data if available.
111        let sync_data = lock_file
112            .read_sync_data()
113            .map_err(|e| TableError::LockIo {
114                path: options.path.display().to_string(),
115                message: e.to_string(),
116            })?
117            .unwrap_or_else(SyncData::new);
118
119        table.lock_state = Some(LockState {
120            path: options.path.clone(),
121            lock_file,
122            sync_data,
123            options: lock_opts,
124            data_manager: options.data_manager,
125            endian_format: options.endian_format,
126        });
127
128        Ok(table)
129    }
130
131    /// Acquires a lock on the table.
132    ///
133    /// Re-reads the table data from disk if another process modified it
134    /// since the last lock was held.
135    ///
136    /// `nattempts`: number of lock attempts. 0 means wait indefinitely,
137    /// 1 means try once without waiting.
138    ///
139    /// Returns `true` if the lock was acquired, `false` if it could not
140    /// be acquired within the given attempts.
141    ///
142    /// C++ equivalent: `Table::lock(type, nattempts)`.
143    #[cfg(unix)]
144    pub fn lock(&mut self, lock_type: LockType, nattempts: u32) -> Result<bool, TableError> {
145        // Memory tables always succeed — no file-based locking needed.
146        // C++ equivalent: MemoryTable::lock() returns True.
147        if self.kind == TableKind::Memory {
148            return Ok(true);
149        }
150
151        // Notify external sync hook before acquiring.
152        if let Some(sync) = &self.external_sync {
153            match lock_type {
154                LockType::Read => sync.acquire_read(),
155                LockType::Write => sync.acquire_write(),
156            }
157        }
158
159        let state = self
160            .lock_state
161            .as_mut()
162            .ok_or_else(|| TableError::NotLocked {
163                operation: "lock".into(),
164            })?;
165
166        // NoRead modes skip the file-level read lock entirely.
167        if lock_type == LockType::Read && state.options.mode.skip_read_lock() {
168            return Ok(true);
169        }
170
171        let acquired =
172            state
173                .lock_file
174                .acquire(lock_type, nattempts)
175                .map_err(|e| TableError::LockIo {
176                    path: state.path.display().to_string(),
177                    message: e.to_string(),
178                })?;
179
180        if acquired {
181            // Read sync data and check if we need to reload.
182            if let Some(new_sync) =
183                state
184                    .lock_file
185                    .read_sync_data()
186                    .map_err(|e| TableError::LockIo {
187                        path: state.path.display().to_string(),
188                        message: e.to_string(),
189                    })?
190            {
191                if state.sync_data.needs_reload(&new_sync) {
192                    // Another process modified the table — reload.
193                    let storage = CompositeStorage;
194                    let snapshot = storage
195                        .load_with_row_hint(&state.path, Some(new_sync.nrrow))
196                        .map_err(|e| TableError::LockIo {
197                            path: state.path.display().to_string(),
198                            message: e.to_string(),
199                        })?;
200                    self.virtual_columns = snapshot.virtual_columns;
201                    self.inner.replace_from_snapshot(
202                        snapshot.rows,
203                        snapshot.undefined_cells,
204                        snapshot.keywords,
205                        snapshot.column_keywords,
206                        snapshot.schema,
207                    );
208                    // Update our stored sync data.
209                    if let Some(s) = self.lock_state.as_mut() {
210                        s.sync_data = new_sync;
211                    }
212                }
213            }
214        }
215
216        Ok(acquired)
217    }

219    /// Releases the current lock.
220    ///
221    /// If a write lock was held, the table is flushed to disk first and
222    /// sync data is updated in the lock file.
223    ///
224    /// In permanent locking modes, this is a no-op (lock is held until close).
225    ///
226    /// C++ equivalent: `Table::unlock()`.
227    #[cfg(unix)]
228    pub fn unlock(&mut self) -> Result<(), TableError> {
229        // Memory tables have no lock to release.
230        // C++ equivalent: MemoryTable::unlock() is a no-op.
231        if self.kind == TableKind::Memory {
232            return Ok(());
233        }
234        // Extract the info we need before borrowing self for save/schema.
235        let (is_write_locked, save_opts, mode) = {
236            let state = self
237                .lock_state
238                .as_ref()
239                .ok_or_else(|| TableError::NotLocked {
240                    operation: "unlock".into(),
241                })?;
242            let wl = state.lock_file.has_lock(LockType::Write);
243            let opts = TableOptions::new(&state.path)
244                .with_data_manager(state.data_manager)
245                .with_endian_format(state.endian_format);
246            (wl, opts, state.options.mode)
247        };
248
249        if matches!(
250            mode,
251            LockMode::PermanentLocking | LockMode::PermanentLockingWait
252        ) {
253            return Ok(());
254        }
255
256        // If write-locked, flush data to disk.
257        if is_write_locked {
258            self.save(save_opts)?;
259
260            // Gather sync info from immutable borrows.
261            let nrrow = self.row_count() as u64;
262            let nrcolumn = self.schema().map(|s| s.columns().len() as u32).unwrap_or(0);
263
264            // Now borrow lock_state mutably for sync data update.
265            let state = self.lock_state.as_mut().expect("lock_state present");
266            state.sync_data.record_write(nrrow, nrcolumn, true, &[true]);
267
268            state
269                .lock_file
270                .write_sync_data(&state.sync_data)
271                .map_err(|e| TableError::LockIo {
272                    path: state.path.display().to_string(),
273                    message: e.to_string(),
274                })?;
275        }
276
277        let state = self.lock_state.as_mut().expect("lock_state present");
278        state.lock_file.release().map_err(|e| TableError::LockIo {
279            path: state.path.display().to_string(),
280            message: e.to_string(),
281        })?;
282
283        // Notify external sync hook after release.
284        if let Some(sync) = &self.external_sync {
285            sync.release();
286        }
287
288        Ok(())
289    }

291    /// Returns `true` if the given lock type is currently held.
292    ///
293    /// Returns `false` if the table was not opened with locking.
294    ///
295    /// C++ equivalent: `Table::hasLock(type)`.
296    #[cfg(unix)]
297    pub fn has_lock(&self, lock_type: LockType) -> bool {
298        // Memory tables always report holding the lock.
299        // C++ equivalent: MemoryTable::hasLock() returns True.
300        if self.kind == TableKind::Memory {
301            return true;
302        }
303        self.lock_state
304            .as_ref()
305            .map(|s| s.lock_file.has_lock(lock_type))
306            .unwrap_or(false)
307    }

309    /// Tests if the table is opened by another process.
310    ///
311    /// Checks the in-use indicator in the lock file. Returns `false` if the
312    /// table was not opened with locking.
313    ///
314    /// C++ equivalent: `Table::isMultiUsed()`.
315    #[cfg(unix)]
316    pub fn is_multi_used(&self) -> bool {
317        // Memory tables are never shared with another process.
318        // C++ equivalent: MemoryTable::isMultiUsed() returns False.
319        if self.kind == TableKind::Memory {
320            return false;
321        }
322        self.lock_state
323            .as_ref()
324            .map(|s| s.lock_file.is_multi_used())
325            .unwrap_or(false)
326    }

334    #[cfg(unix)]
335    pub(super) fn begin_write_operation(&mut self, operation: &str) -> Result<bool, TableError> {
336        if self.kind == TableKind::Memory {
337            return Ok(false);
338        }
339
340        let Some(state) = self.lock_state.as_mut() else {
341            return Ok(false);
342        };
343
344        match state.options.mode {
345            LockMode::NoLocking => Ok(false),
346            LockMode::UserLocking | LockMode::UserNoReadLocking => {
347                if state.lock_file.has_lock(LockType::Write) {
348                    Ok(false)
349                } else {
350                    Err(TableError::LockFailed {
351                        path: state.path.display().to_string(),
352                        message: format!(
353                            "{operation} requires a write lock when using UserLocking"
354                        ),
355                    })
356                }
357            }
358            LockMode::PermanentLocking | LockMode::PermanentLockingWait => {
359                if state.lock_file.has_lock(LockType::Write) {
360                    Ok(false)
361                } else {
362                    Err(TableError::LockFailed {
363                        path: state.path.display().to_string(),
364                        message: format!(
365                            "{operation} requires the permanent write lock to be held"
366                        ),
367                    })
368                }
369            }
370            LockMode::AutoLocking | LockMode::AutoNoReadLocking | LockMode::DefaultLocking => {
371                if state.lock_file.has_lock(LockType::Write) {
372                    return Ok(false);
373                }
374
375                let acquired = state.lock_file.acquire(LockType::Write, 0).map_err(|e| {
376                    TableError::LockIo {
377                        path: state.path.display().to_string(),
378                        message: e.to_string(),
379                    }
380                })?;
381                if acquired {
382                    Ok(true)
383                } else {
384                    Err(TableError::LockFailed {
385                        path: state.path.display().to_string(),
386                        message: format!("could not acquire temporary write lock for {operation}"),
387                    })
388                }
389            }
390        }
391    }

393    #[cfg(not(unix))]
394    pub(super) fn begin_write_operation(&mut self, _operation: &str) -> Result<bool, TableError> {
395        Ok(false)

398    #[cfg(unix)]
399    pub(super) fn finish_write_operation<R>(
400        &mut self,
401        auto_unlock: bool,
402        result: Result<R, TableError>,
403    ) -> Result<R, TableError> {
404        if !auto_unlock {
405            return result;
406        }
407
408        let unlock_result = self.unlock();
409        match (result, unlock_result) {
410            (Ok(value), Ok(())) => Ok(value),
411            (Ok(_), Err(unlock_err)) => Err(unlock_err),
412            (Err(op_err), Ok(())) => Err(op_err),
413            (Err(op_err), Err(_unlock_err)) => Err(op_err),
414        }
415    }
416
417    #[cfg(not(unix))]
418    pub(super) fn finish_write_operation<R>(
419        &mut self,
420        _auto_unlock: bool,
421        result: Result<R, TableError>,
422    ) -> Result<R, TableError> {
423        result
424    }
