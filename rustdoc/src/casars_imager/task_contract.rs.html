50/// Detail level for imager progress and observability events.
51#[derive(Debug, Clone, Copy, Default, PartialEq, Eq, Serialize, Deserialize, JsonSchema)]
52#[serde(rename_all = "kebab-case")]
53pub enum ImagerProgressDetail {
54    /// Low-rate UI progress with cheap resource snapshots.
55    #[default]
56    Basic,
57    /// Structured performance diagnostics suitable for large-MS debugging.
58    Diagnostic,
59}
60
61impl FromStr for ImagerProgressDetail {
62    type Err = String;
63
64    fn from_str(text: &str) -> Result<Self, Self::Err> {
65        match text.trim().to_ascii_lowercase().as_str() {
66            "basic" => Ok(Self::Basic),
67            "diagnostic" => Ok(Self::Diagnostic),
68            other => Err(format!(
69                "invalid progress detail {other:?}; expected basic or diagnostic"
70            )),
71        }
72    }
73}

104    pub protocol: ImagerProtocolInfo,

106    pub semantic: TaskSemanticContract,
107    /// Shared component schemas reusable across projections.
108    pub components: ProviderComponentSchemas,
109    /// Presentation annotations carried with the canonical bundle.
110    pub annotations: JsonValue,
111    /// Derived projection metadata for UI and CLI consumers.
112    pub projections: ProviderProjectionMetadata,
113    /// JSON schema for [`ImagerTaskRequest`].
114    pub request_schema: RootSchema,
115    /// JSON schema for [`ImagerTaskResult`].
116    pub result_schema: RootSchema,
117    /// JSON schema for [`ImagerProgressEvent`].
118    pub progress_event_schema: RootSchema,
119}
120
121impl ImagerTaskSchemaBundle {
122    /// Build the current request/result schema bundle.
123    pub fn current() -> Self {
124        let request_schema = schema_for!(ImagerTaskRequest);
125        let result_schema = schema_for!(ImagerTaskResult);
126        let progress_event_schema = schema_for!(ImagerProgressEvent);
127        let ui_schema = serde_json::to_value(command_schema("casars-imager"))
128            .expect("serialize imager ui schema projection");
129        Self {
130            protocol: ImagerProtocolInfo::current(),
131            semantic: TaskSemanticContract {
132                request_schema: request_schema.clone(),
133                result_schema: result_schema.clone(),
134                operations: vec![TaskOperationDescriptor {
135                    name: "run".to_string(),
136                    request_kind: "run".to_string(),
137                    result_kind: Some("run".to_string()),
138                }],
139            },
140            components: merged_components([
141                &request_schema,
142                &result_schema,
143                &progress_event_schema,
144            ]),
145            annotations: derived_ui_schema_annotations(),
146            projections: ProviderProjectionMetadata {
147                cli: Some(ProviderCliProjection {
148                    machine_actions: ProviderCliMachineActions {
149                        ui_schema: Some("--ui-schema".to_string()),
150                        json_schema: Some("--json-schema".to_string()),
151                        protocol_info: Some("--protocol-info".to_string()),
152                        json_run: Some("--json-run <SOURCE>".to_string()),
153                        session: None,
154                    },
155                }),
156                ui_schema: Some(ui_schema),
157                python: None,
158            },
159            request_schema,
160            result_schema,
161            progress_event_schema,
162        }
163    }
164
165    /// Return the launcher/TUI compatibility view projected from the bundle.
166    pub fn ui_schema_projection(&self) -> Result<casa_ms::ui_schema::UiCommandSchema, String> {
167        let value = self
168            .projections
169            .ui_schema
170            .clone()
171            .ok_or_else(|| "missing ui_schema projection".to_string())?;
172        serde_json::from_value(value).map_err(|error| format!("parse imager ui schema: {error}"))
173    }
174}

2196    let value = value.trim();

2198        || value.eq_ignore_ascii_case("full")

2204        Ok(parsed) if parsed > 0 => Ok(()),
2205        _ => Err(format!(
2206            "invalid standard_mfs_metal_minor_cycle_chunk {value:?}; expected auto, auto:<positive-ms>, full, or a positive integer"
2207        )),
2208    }

2216        .then_some(target_ms)
