23pub(crate) fn validate_solve_interval(
24    solve_interval: GainSolveInterval,
25) -> Result<(), GainSolveError> {
26    match solve_interval {
27        GainSolveInterval::Infinite | GainSolveInterval::Integration => Ok(()),
28        GainSolveInterval::Seconds(seconds) if seconds > 0.0 => Ok(()),
29        _ => Err(GainSolveError::UnsupportedSolveInterval { solve_interval }),
30    }
31}

33pub(crate) fn validate_smodel(smodel: [f32; 4]) -> Result<(), GainSolveError> {
34    if smodel[0] <= 0.0 || smodel[1] != 0.0 || smodel[2] != 0.0 || smodel[3] != 0.0 {
35        return Err(GainSolveError::UnsupportedSkyModel { smodel });
36    }
37    Ok(())
38}

40pub(crate) fn load_preapplied_rows(
41    ms: &MeasurementSet,
42    selection: &MsSelection,
43    prior_calibration_tables: &[ApplyCalibrationTableSpec],
44    parang: bool,
45) -> Result<Option<HashMap<usize, EvaluatedApplyRow>>, GainSolveError> {
46    if prior_calibration_tables.is_empty() && !parang {
47        return Ok(None);
48    }
49    let plan = plan_apply(
50        ms,
51        &ApplyPlanRequest {
52            selection: selection.clone(),
53            apply_mode: crate::ApplyMode::CalFlag,
54            parang,
55            calibration_tables: prior_calibration_tables.to_vec(),
56        },
57    )
58    .map_err(|source| GainSolveError::PriorCalibrationPlan {
59        source: Box::new(source),
60    })?;
61    evaluate_apply_rows(ms, &plan).map(Some).map_err(|source| {
62        GainSolveError::PriorCalibrationApply {
63            source: Box::new(source),
64        }
65    })
66}

68pub(crate) fn build_solve_groups(
69    ms: &MeasurementSet,
70    rows: &[SelectedSolveRow],
71    preapplied_rows: Option<&HashMap<usize, EvaluatedApplyRow>>,
72    options: SolveGroupOptions,
73) -> Result<BTreeMap<(SolveBaseKey, SolveBucketKey), SolveAccumulator>, GainSolveError> {
74    let mut sorted_rows = rows.to_vec();
75    sorted_rows.sort_by_key(|row| {
76        (
77            row.field_id,
78            row.data_spw_id,
79            row.observation_id,
80            row.scan_number,
81            row.time_seconds.to_bits(),
82            row.interval_seconds.to_bits(),
83            row.row_index,
84        )
85    });
86
87    let mut groups = BTreeMap::<(SolveBaseKey, SolveBucketKey), SolveAccumulator>::new();
88    let mut bucket_starts = BTreeMap::<SolveBaseKey, f64>::new();
89
90    for row in sorted_rows {
91        if row.antenna1 == row.antenna2 {
92            continue;
93        }
94        let base_key = SolveBaseKey {
95            field_id: if options.combine.fields {
96                0
97            } else {
98                row.field_id
99            },
100            spw_id: row.data_spw_id,
101            observation_id: row.observation_id,
102            scan_number: if options.combine.scans {
103                0
104            } else {
105                row.scan_number
106            },
107        };
108        let bucket_key = match options.solve_interval {
109            GainSolveInterval::Infinite => SolveBucketKey::Infinite,
110            GainSolveInterval::Integration => SolveBucketKey::Integration {
111                time_bits: row.time_seconds.to_bits(),
112                interval_bits: row.interval_seconds.to_bits(),
113            },
114            GainSolveInterval::Seconds(seconds) => {
115                let bucket_start = bucket_starts.entry(base_key).or_insert(row.time_seconds);
116                let bucket_index = ((row.time_seconds - *bucket_start) / seconds)
117                    .floor()
118                    .max(0.0);
119                SolveBucketKey::Seconds(bucket_index as u64)
120            }
121        };
122
123        let group = groups.entry((base_key, bucket_key)).or_insert_with(|| {
124            SolveAccumulator::new(row.field_id, row.data_spw_id, row.observation_id)
125        });
126        let preapplied = preapplied_rows.and_then(|rows| rows.get(&row.row_index));
127        group.observe(
128            ms,
129            &row,
130            preapplied,
131            options.gain_type,
132            options.model_source,
133            options.stokes_i,
134        )?;
135    }
136
137    Ok(groups)
138}

140pub(crate) fn all_antenna_ids(ms: &MeasurementSet) -> Result<BTreeSet<i32>, GainSolveError> {
141    let antenna = ms
142        .antenna()
143        .map_err(|source| GainSolveError::OpenMeasurementSet {
144            path: ms
145                .path()
146                .map(|path| path.display().to_string())
147                .unwrap_or_else(|| "<in-memory>".to_string()),
148            source,
149        })?;
150    Ok((0..antenna.row_count())
151        .map(|row| i32::try_from(row).expect("antenna row count should fit in i32"))
152        .collect())
153}

155pub(crate) fn resolve_refant(
156    ms: &MeasurementSet,
157    selector: &RefAntSelector,
158) -> Result<i32, GainSolveError> {
159    match selector {
160        RefAntSelector::AntennaId(id) => Ok(*id),
161        RefAntSelector::AntennaName(name) => {
162            let antenna = ms
163                .antenna()
164                .map_err(|error| GainSolveError::ResolveRefAnt {
165                    selector: name.clone(),
166                    reason: error.to_string(),
167                })?;
168            for row in 0..antenna.row_count() {
169                if antenna
170                    .name(row)
171                    .map_err(|error| GainSolveError::ResolveRefAnt {
172                        selector: name.clone(),
173                        reason: error.to_string(),
174                    })?
175                    == *name
176                {
177                    return Ok(row as i32);
178                }
179            }
180            Err(GainSolveError::ResolveRefAnt {
181                selector: name.clone(),
182                reason: "no ANTENNA.NAME match".to_string(),
183            })
184        }
185    }
186}

188pub(crate) fn collect_selected_rows(
189    ms: &MeasurementSet,
190    selection: &MsSelection,
191) -> Result<Vec<SelectedSolveRow>, GainSolveError> {
192    let selected_rows = resolve_calibration_selection(ms, selection).map_err(|error| {
193        GainSolveError::OpenMeasurementSet {
194            path: ms
195                .path()
196                .map(|path| path.display().to_string())
197                .unwrap_or_else(|| "<in-memory>".to_string()),
198            source: match error {
199                casa_ms::MsSelectionError::Domain(source) => source,
200                other => MsError::InvalidInput(other.to_string()),
201            },
202        }
203    })?;
204    if selected_rows.selected_rows.is_empty() {
205        return Err(GainSolveError::EmptySelection);
206    }
207
208    selected_rows
209        .selected_rows
210        .into_iter()
211        .map(|row| {
212            Ok(SelectedSolveRow {
213                row_index: row.row_index,
214                field_id: row.field_id,
215                observation_id: get_i32(ms.main_table(), row.row_index, "OBSERVATION_ID")?,
216                data_desc_id: row.data_desc_id,
217                data_spw_id: row.spectral_window_id,
218                channel_selection: selection.channel_selection_for_spw(row.spectral_window_id),
219                antenna1: row.antenna1_id,
220                antenna2: row.antenna2_id,
221                time_seconds: row.time_mjd_seconds,
222                time_centroid_seconds: get_f64(ms.main_table(), row.row_index, "TIME_CENTROID")?,
223                interval_seconds: get_f64(ms.main_table(), row.row_index, "INTERVAL")?,
224                scan_number: get_i32(ms.main_table(), row.row_index, "SCAN_NUMBER")?,
225            })
226        })
227        .collect()
228}

230pub(crate) fn correlation_types_for_ddid(
231    ms: &MeasurementSet,
232    data_desc_id: i32,
233) -> Result<Vec<i32>, GainSolveError> {
234    let dd = ms
235        .data_description()
236        .map_err(|source| GainSolveError::OpenMeasurementSet {
237            path: ms
238                .path()
239                .map(|path| path.display().to_string())
240                .unwrap_or_else(|| "<in-memory>".to_string()),
241            source,
242        })?;
243    let pol = ms
244        .polarization()
245        .map_err(|source| GainSolveError::OpenMeasurementSet {
246            path: ms
247                .path()
248                .map(|path| path.display().to_string())
249                .unwrap_or_else(|| "<in-memory>".to_string()),
250            source,
251        })?;
252    let row = usize::try_from(data_desc_id).map_err(|_| {
253        GainSolveError::UnsupportedCorrelationLayout {
254            data_desc_id,
255            correlation_types: Vec::new(),
256        }
257    })?;
258    let pol_id = dd
259        .polarization_id(row)
260        .map_err(|source| GainSolveError::OpenMeasurementSet {
261            path: ms
262                .path()
263                .map(|path| path.display().to_string())
264                .unwrap_or_else(|| "<in-memory>".to_string()),
265            source,
266        })?;
267    pol.corr_type(usize::try_from(pol_id).unwrap_or(usize::MAX))
268        .map_err(|source| GainSolveError::OpenMeasurementSet {
269            path: ms
270                .path()
271                .map(|path| path.display().to_string())
272                .unwrap_or_else(|| "<in-memory>".to_string()),
273            source,
274        })
275}

385    pub(crate) fn observe(
386        &mut self,
387        ms: &MeasurementSet,
388        row: &SelectedSolveRow,
389        preapplied_row: Option<&EvaluatedApplyRow>,
390        gain_type: GainType,
391        model_source: GainSolveModelSource,
392        stokes_i: f32,
393    ) -> Result<(), GainSolveError> {
394        let (data, flags) = match preapplied_row {
395            Some(row) => (&row.corrected_data, &row.flags),
396            None => {
397                let data = ms
398                    .main_table()
399                    .cell_accessor(row.row_index, "DATA")
400                    .and_then(|cell| cell.array())
401                    .map_err(|source| GainSolveError::OpenMeasurementSet {
402                        path: ms
403                            .path()
404                            .map(|path| path.display().to_string())
405                            .unwrap_or_else(|| "<in-memory>".to_string()),
406                        source: MsError::from(source),
407                    })?;
408                let flags = ms
409                    .main_table()
410                    .cell_accessor(row.row_index, "FLAG")
411                    .and_then(|cell| cell.array())
412                    .map_err(|source| GainSolveError::OpenMeasurementSet {
413                        path: ms
414                            .path()
415                            .map(|path| path.display().to_string())
416                            .unwrap_or_else(|| "<in-memory>".to_string()),
417                        source: MsError::from(source),
418                    })?;
419                (data, flags)
420            }
421        };
422        let weights = ms
423            .main_table()
424            .cell_accessor(row.row_index, "WEIGHT")
425            .and_then(|cell| cell.array())
426            .map_err(|source| GainSolveError::OpenMeasurementSet {
427                path: ms
428                    .path()
429                    .map(|path| path.display().to_string())
430                    .unwrap_or_else(|| "<in-memory>".to_string()),
431                source: MsError::from(source),
432            })?;
433        let weight_spectrum = if ms
434            .main_table()
435            .schema()
436            .is_some_and(|schema| schema.contains_column("WEIGHT_SPECTRUM"))
437        {
438            Some(
439                ms.main_table()
440                    .cell_accessor(row.row_index, "WEIGHT_SPECTRUM")
441                    .and_then(|cell| cell.array())
442                    .map_err(|source| GainSolveError::OpenMeasurementSet {
443                        path: ms
444                            .path()
445                            .map(|path| path.display().to_string())
446                            .unwrap_or_else(|| "<in-memory>".to_string()),
447                        source: MsError::from(source),
448                    })?,
449            )
450        } else {
451            None
452        };
453        let model_data = if matches!(model_source, GainSolveModelSource::ModelColumn) {
454            Some(
455                ms.main_table()
456                    .cell_accessor(row.row_index, "MODEL_DATA")
457                    .and_then(|cell| cell.array())
458                    .map_err(|source| GainSolveError::OpenMeasurementSet {
459                        path: ms
460                            .path()
461                            .map(|path| path.display().to_string())
462                            .unwrap_or_else(|| "<in-memory>".to_string()),
463                        source: MsError::from(source),
464                    })?,
465            )
466        } else {
467            None
468        };
469        let correlation_types = correlation_types_for_ddid(ms, row.data_desc_id)?;
470
471        let ArrayValue::Complex32(data) = data else {
472            return Err(GainSolveError::UnsupportedParameterShape {
473                path: "<measurement-set DATA>".to_string(),
474                shape: data.shape().to_vec(),
475            });
476        };
477        let ArrayValue::Bool(flags) = flags else {
478            return Err(GainSolveError::UnsupportedParameterShape {
479                path: "<measurement-set FLAG>".to_string(),
480                shape: flags.shape().to_vec(),
481            });
482        };
483        let ArrayValue::Float32(weights) = weights else {
484            return Err(GainSolveError::UnsupportedParameterShape {
485                path: "<measurement-set WEIGHT>".to_string(),
486                shape: weights.shape().to_vec(),
487            });
488        };
489        let weight_spectrum = match weight_spectrum {
490            Some(ArrayValue::Float32(weight_spectrum)) => Some(weight_spectrum),
491            Some(other) => {
492                return Err(GainSolveError::UnsupportedParameterShape {
493                    path: "<measurement-set WEIGHT_SPECTRUM>".to_string(),
494                    shape: other.shape().to_vec(),
495                });
496            }
497            None => None,
498        };
499        let model_data = match model_data {
500            Some(ArrayValue::Complex32(model_data)) => Some(model_data),
501            Some(other) => {
502                return Err(GainSolveError::UnsupportedParameterShape {
503                    path: "<measurement-set MODEL_DATA>".to_string(),
504                    shape: other.shape().to_vec(),
505                });
506            }
507            None => None,
508        };
509        if data.ndim() != 2 || flags.ndim() != 2 || data.shape() != flags.shape() {
510            return Err(GainSolveError::UnsupportedParameterShape {
511                path: "<measurement-set row>".to_string(),
512                shape: data.shape().to_vec(),
513            });
514        }
515        if let Some(model_data) = model_data.as_ref()
516            && (model_data.ndim() != 2 || model_data.shape() != data.shape())
517        {
518            return Err(GainSolveError::UnsupportedParameterShape {
519                path: "<measurement-set MODEL_DATA>".to_string(),
520                shape: model_data.shape().to_vec(),
521            });
522        }
523        if weights.ndim() != 1 || weights.shape()[0] != data.shape()[0] {
524            return Err(GainSolveError::UnsupportedParameterShape {
525                path: "<measurement-set WEIGHT>".to_string(),
526                shape: weights.shape().to_vec(),
527            });
528        }
529        if let Some(weight_spectrum) = weight_spectrum.as_ref()
530            && (weight_spectrum.ndim() != 2 || weight_spectrum.shape() != data.shape())
531        {
532            return Err(GainSolveError::UnsupportedParameterShape {
533                path: "<measurement-set WEIGHT_SPECTRUM>".to_string(),
534                shape: weight_spectrum.shape().to_vec(),
535            });
536        }
537        let selected_channels = selected_channels(row, data.shape()[1])?;
538
539        let graph_count = match gain_type {
540            GainType::G => 2,
541            GainType::T => 1,
542        };
543        if self.receptor_graphs.len() < graph_count {
544            self.receptor_graphs.resize_with(graph_count, HashMap::new);
545            self.receptor_weights.resize_with(graph_count, HashMap::new);
546            self.receptor_stats.resize_with(graph_count, HashMap::new);
547            self.receptor_accumulations
548                .resize_with(graph_count, HashMap::new);
549        }
550
551        self.min_time = self.min_time.min(row.time_seconds);
552        self.max_time = self.max_time.max(row.time_seconds);
553        if self.unique_time_bits.insert(row.time_seconds.to_bits()) {
554            self.unique_time_sum += row.time_seconds;
555            self.unique_time_count += 1;
556        }
557        self.total_interval += row.interval_seconds;
558        self.sample_rows += 1;
559        self.scan_numbers.insert(row.scan_number);
560        self.antenna_ids.insert(row.antenna1);
561        self.antenna_ids.insert(row.antenna2);
562
563        let mut time_centroid_weight = 0.0_f64;
564        for corr_index in 0..data.shape()[0] {
565            for &chan_index in &selected_channels {
566                let weight = channel_weight(weights, weight_spectrum, corr_index, chan_index);
567                if weight > 0.0 && !solve_sample_flagged(flags, gain_type, corr_index, chan_index) {
568                    time_centroid_weight += f64::from(weight);
569                }
570            }
571        }
572        if time_centroid_weight > 0.0 {
573            self.time_centroid_weighted_sum += row.time_centroid_seconds * time_centroid_weight;
574            self.time_centroid_weight += time_centroid_weight;
575        }
576
577        for corr_index in 0..data.shape()[0] {
578            let Some(receptors) = correlation_receptors(correlation_types[corr_index]) else {
579                return Err(GainSolveError::UnsupportedCorrelationLayout {
580                    data_desc_id: row.data_desc_id,
581                    correlation_types: correlation_types
582                        .iter()
583                        .map(|code| stokes_name(*code).to_string())
584                        .collect(),
585                });
586            };
587            if receptors.0 != receptors.1 {
588                continue;
589            }
590            let graph_index = match gain_type {
591                GainType::G => receptors.0,
592                GainType::T => 0,
593            };
594            for &chan_index in &selected_channels {
595                if solve_sample_flagged(flags, gain_type, corr_index, chan_index) {
596                    continue;
597                }
598                let weight = channel_weight(weights, weight_spectrum, corr_index, chan_index);
599                if weight <= 0.0 {
600                    continue;
601                }
602                let model = match model_data.as_ref() {
603                    Some(model_data) => model_data[[corr_index, chan_index]],
604                    None => Complex32::new(stokes_i, 0.0),
605                };
606                let model_norm_sqr = model.norm_sqr();
607                if model_norm_sqr <= f32::EPSILON {
608                    continue;
609                }
610                let sample = data[[corr_index, chan_index]] / model;
611                if sample.norm() <= f32::EPSILON {
612                    continue;
613                }
614                let effective_weight = weight * model_norm_sqr;
615                accumulate_collapsed_input(
616                    &mut self.receptor_accumulations[graph_index],
617                    SolveAccumulationKey {
618                        baseline: (row.antenna1, row.antenna2),
619                        correlation_index: corr_index,
620                    },
621                    effective_weight,
622                    sample,
623                    f64::from(effective_weight) * f64::from(sample.norm_sqr()),
624                );
625            }
626        }
627
628        Ok(())
629    }
630
631    pub(crate) fn finalize_for_solve(&mut self, solve_mode: GainSolveMode) {
632        for receptor in 0..self.receptor_accumulations.len() {
633            self.receptor_graphs[receptor].clear();
634            self.receptor_weights[receptor].clear();
635            self.receptor_stats[receptor].clear();
636
637            for (&key, accumulation) in &self.receptor_accumulations[receptor] {
638                if accumulation.weight <= f32::EPSILON {
639                    continue;
640                }
641                let averaged = accumulation.weighted_sum / Complex32::new(accumulation.weight, 0.0);
642                let (sample, weight) = match solve_mode {
643                    GainSolveMode::Phase => {
644                        let amplitude = averaged.norm();
645                        if amplitude <= f32::EPSILON {
646                            continue;
647                        }
648                        (
649                            averaged / Complex32::new(amplitude, 0.0),
650                            accumulation.weight * amplitude * amplitude,
651                        )
652                    }
653                    GainSolveMode::AmplitudePhase => (averaged, accumulation.weight),
654                };
655                accumulate_edge_with_stats(
656                    &mut self.receptor_graphs[receptor],
657                    &mut self.receptor_weights[receptor],
658                    &mut self.receptor_stats[receptor],
659                    key.baseline,
660                    weight,
661                    sample * Complex32::new(weight, 0.0),
662                    SolveEdgeStats {
663                        weighted_sample_power: f64::from(weight) * f64::from(sample.norm_sqr()),
664                        raw_weighted_sample_power: accumulation.weighted_sample_power,
665                        sample_count: accumulation.sample_count,
666                        collapsed_count: 1,
667                    },
668                );
669            }
670        }
671    }
672
673    pub(crate) fn aggregate_time(&self) -> f64 {
674        if self.unique_time_count > 0 {
675            self.unique_time_sum / self.unique_time_count as f64
676        } else {
677            (self.min_time + self.max_time) / 2.0
678        }
679    }
680
681    pub(crate) fn aggregate_time_centroid(&self) -> f64 {
682        if self.time_centroid_weight > 0.0 {
683            self.time_centroid_weighted_sum / self.time_centroid_weight
684        } else {
685            self.aggregate_time()
686        }
687    }

690fn selected_channels(
691    row: &SelectedSolveRow,
692    channel_count: usize,
693) -> Result<Vec<usize>, GainSolveError> {
694    match &row.channel_selection {
695        Some(selection) => {
696            selection
697                .indices(channel_count)
698                .map_err(|source| GainSolveError::OpenMeasurementSet {
699                    path: "<measurement-set channel selection>".to_string(),
700                    source,
701                })
702        }
703        None => Ok((0..channel_count).collect()),
704    }
705}

707fn channel_has_any_correlation_flagged(flags: &ArrayD<bool>, chan_index: usize) -> bool {
708    (0..flags.shape()[0]).any(|corr_index| flags[[corr_index, chan_index]])
709}
710
711fn solve_sample_flagged(
712    flags: &ArrayD<bool>,
713    gain_type: GainType,
714    _corr_index: usize,
715    chan_index: usize,
716) -> bool {
717    match gain_type {
718        GainType::G => channel_has_any_correlation_flagged(flags, chan_index),
719        GainType::T => channel_has_any_correlation_flagged(flags, chan_index),
720    }
721}
722
723fn channel_weight(
724    weights: &ArrayD<f32>,
725    weight_spectrum: Option<&ArrayD<f32>>,
726    corr_index: usize,
727    chan_index: usize,
728) -> f32 {
729    weight_spectrum
730        .map(|spectrum| spectrum[[corr_index, chan_index]])
731        .unwrap_or_else(|| weights[[corr_index]])
732}

734fn accumulate_collapsed_input(
735    accumulations: &mut HashMap<SolveAccumulationKey, SolveEdgeAccumulation>,
736    key: SolveAccumulationKey,
737    weight: f32,
738    sample: Complex32,
739    weighted_sample_power: f64,
740) {
741    let entry = accumulations.entry(key).or_default();
742    entry.weighted_sum += sample * Complex32::new(weight, 0.0);
743    entry.weight += weight;
744    entry.weighted_sample_power += weighted_sample_power;
745    entry.sample_count += 1;
746}
