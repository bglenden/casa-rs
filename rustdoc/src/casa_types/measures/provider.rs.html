4use std::fmt::Debug;
5
6/// Interpolated Earth-orientation values at one UTC epoch.
7#[derive(Debug, Clone, Copy, PartialEq)]
8pub struct EopValues {
9    /// UT1 minus UTC, in seconds.
10    pub dut1_seconds: f64,
11    /// Polar-motion X component, in arcseconds.
12    pub x_arcsec: f64,
13    /// Polar-motion Y component, in arcseconds.
14    pub y_arcsec: f64,
15    /// Celestial-pole dX offset, in milliarcseconds.
16    pub dx_mas: f64,
17    /// Celestial-pole dY offset, in milliarcseconds.
18    pub dy_mas: f64,
19    /// Whether either interpolation endpoint is predicted.
20    pub is_predicted: bool,
21}
22
23/// Observatory coordinates returned by a measures-data provider.
24#[derive(Debug, Clone, PartialEq)]
25pub enum ObservatoryPosition {
26    /// Geocentric ITRF coordinates, in metres.
27    Itrf { x_m: f64, y_m: f64, z_m: f64 },
28    /// WGS84 geodetic coordinates.
29    Wgs84 {
30        longitude_rad: f64,
31        latitude_rad: f64,
32        height_m: f64,
33    },
34}
35
36/// Fixed named-source direction returned by a measures-data provider.
37#[derive(Debug, Clone, PartialEq)]
38pub struct NamedSourceDirection {
39    /// Persisted casacore direction-reference name.
40    pub reference: String,
41    /// Longitude in radians.
42    pub longitude_rad: f64,
43    /// Latitude in radians.
44    pub latitude_rad: f64,
45}
46
47/// Pure, immutable input-data boundary used by measure algorithms.
48///
49/// Implementations may load files or cache data, but `casa-types` depends only
50/// on these scientific values and never performs discovery or I/O itself.
51pub trait MeasuresProvider: Debug + Send + Sync {
52    /// Interpolate EOP values at a UTC MJD.
53    fn eop_values(&self, _utc_mjd: f64) -> Result<Option<EopValues>, String> {
54        Ok(None)
55    }
56
57    /// Return TAI minus UTC in seconds at a UTC MJD.
58    fn tai_minus_utc_seconds(&self, _utc_mjd: f64) -> Result<f64, String> {
59        Err("TAI-UTC data is unavailable from this measures provider".to_string())
60    }
61
62    /// Convert a TAI MJD to UTC MJD.
63    fn utc_from_tai_mjd(&self, _tai_mjd: f64) -> Result<f64, String> {
64        Err("TAI-UTC data is unavailable from this measures provider".to_string())
65    }
66
67    /// Return IGRF coefficients and maximum spherical-harmonic degree.
68    fn igrf_coefficients(&self, _decimal_year: f64) -> Result<(Vec<f64>, usize), String> {
69        Err("IGRF data is unavailable from this measures provider".to_string())
70    }
71
72    /// Look up an observatory by case-insensitive name.
73    fn observatory(&self, _name: &str) -> Result<Option<ObservatoryPosition>, String> {
74        Ok(None)
75    }
76
77    /// Look up a fixed source by case-insensitive name.
78    fn source(&self, _name: &str) -> Result<Option<NamedSourceDirection>, String> {
79        Ok(None)
80    }
81
82    /// Look up a rest-line frequency by case-insensitive name.
83    fn spectral_line_hz(&self, _name: &str) -> Result<Option<f64>, String> {
84        Ok(None)
85    }
86}

106        Ok(32.0)
107    }
108
109    fn utc_from_tai_mjd(&self, tai_mjd: f64) -> Result<f64, String> {
110        Ok(tai_mjd - 32.0 / 86_400.0)
111    }
112
113    fn igrf_coefficients(&self, decimal_year: f64) -> Result<(Vec<f64>, usize), String> {
114        if !(1900.0..=2025.999).contains(&decimal_year) {
115            return Err("test IGRF date is outside 1900 through 2025".to_string());
116        }
117        let mut coefficients = vec![0.0; 13 * 15];
118        coefficients[0] = -29_440.0;
119        coefficients[1] = -1_500.0;
120        coefficients[2] = 4_650.0;
121        Ok((coefficients, 13))
122    }
123
124    fn observatory(&self, name: &str) -> Result<Option<ObservatoryPosition>, String> {
125        Ok(match name.to_ascii_uppercase().as_str() {
126            "ALMA" => Some(ObservatoryPosition::Wgs84 {
127                longitude_rad: -67.754_929_f64.to_radians(),
128                latitude_rad: -23.022_886_f64.to_radians(),
129                height_m: 5056.8,
130            }),
131            "VLA" => Some(ObservatoryPosition::Itrf {
132                x_m: -1_601_185.0,
133                y_m: -5_041_977.0,
134                z_m: 3_554_875.0,
135            }),
136            "WSRT" => Some(ObservatoryPosition::Itrf {
137                x_m: 3_826_577.0,
138                y_m: 461_022.0,
139                z_m: 5_064_892.0,
140            }),
141            _ => None,
142        })
143    }
144
145    fn source(&self, name: &str) -> Result<Option<NamedSourceDirection>, String> {
146        Ok(match name.to_ascii_uppercase().as_str() {
147            "CASA" => Some(NamedSourceDirection {
148                reference: "J2000".to_string(),
149                longitude_rad: 6.123_487_680_622_104,
150                latitude_rad: 1.026_515_399_560_464_8,
151            }),
152            "0002-478" => Some(NamedSourceDirection {
153                reference: "ICRS".to_string(),
154                longitude_rad: 0.020_046_3,
155                latitude_rad: -0.830_872,
156            }),
157            _ => None,
158        })
159    }
160
161    fn spectral_line_hz(&self, name: &str) -> Result<Option<f64>, String> {
162        Ok(name.eq_ignore_ascii_case("HI").then_some(1.420_405_752e9))
163    }
164}
