3use serde::Serialize;
4use serde_json::{Value as JsonValue, json};
5
6/// CLI machine-action projections derived from the canonical contract.
7#[derive(Debug, Clone, PartialEq, Eq, Serialize)]
8pub struct ProviderCliMachineActions {
9    /// Legacy compatibility view used by the current launcher/TUI integration.
10    #[serde(skip_serializing_if = "Option::is_none")]
11    pub ui_schema: Option<String>,
12    /// Canonical schema bundle action.
13    #[serde(skip_serializing_if = "Option::is_none")]
14    pub json_schema: Option<String>,
15    /// Protocol descriptor action.
16    #[serde(skip_serializing_if = "Option::is_none")]
17    pub protocol_info: Option<String>,
18    /// One-shot task execution action.
19    #[serde(skip_serializing_if = "Option::is_none")]
20    pub json_run: Option<String>,
21    /// Stateful session action.
22    #[serde(skip_serializing_if = "Option::is_none")]
23    pub session: Option<String>,
24}
25
26/// CLI projection metadata.
27#[derive(Debug, Clone, PartialEq, Eq, Serialize)]
28pub struct ProviderCliProjection {
29    /// Known machine-facing flags derived from the canonical contract.
30    pub machine_actions: ProviderCliMachineActions,
31}
32
33/// Projection metadata for derived consumer views.
34#[derive(Debug, Clone, PartialEq, Serialize)]
35pub struct ProviderProjectionMetadata {
36    /// CLI projection metadata.
37    #[serde(skip_serializing_if = "Option::is_none")]
38    pub cli: Option<ProviderCliProjection>,
39    /// Legacy `--ui-schema` compatibility view when the surface still exposes it.
40    #[serde(skip_serializing_if = "Option::is_none")]
41    pub ui_schema: Option<JsonValue>,
42    /// Python binding projection metadata for direct in-process object surfaces.
43    #[serde(skip_serializing_if = "Option::is_none")]
44    pub python: Option<JsonValue>,
45}
46
47/// Annotations noting that the legacy UI schema is a derived compatibility view.
48pub fn derived_ui_schema_annotations() -> JsonValue {
49    json!({
50        "ui_schema": {
51            "status": "derived_compatibility_view"
52        }
53    })
54}
