2//! Chunk iterators for typed images and image-like views.
3//!
4//! These iterators are compatibility wrappers over the lattice traversal API.
5//! They operate on any [`Lattice`] whose element type satisfies
6//! [`ImagePixel`].
7
8use ndarray::ArrayD;
9
10use casacore_lattices::{Lattice, LatticeMut, TraversalCursorIter, TraversalIter, TraversalSpec};
11
12use crate::error::ImageError;
13use crate::image::ImagePixel;
14
15/// A rectangular chunk of image data together with its origin.
16#[derive(Debug, Clone)]
17pub struct ImageChunk<T: ImagePixel> {
18    /// The pixel data for this chunk.
19    pub data: ArrayD<T>,
20    /// The starting pixel position within the parent image.
21    pub origin: Vec<usize>,
22    /// The shape of this chunk, clipped at image boundaries.
23    pub shape: Vec<usize>,
24}
25
26/// Read-only chunk iterator over any typed image-like lattice.
27pub struct ImageIter<'a, T: ImagePixel, I: Lattice<T>> {
28    inner: TraversalIter<'a, T, I>,
29}
30
31impl<'a, T: ImagePixel, I: Lattice<T>> ImageIter<'a, T, I> {
32    /// Creates a new read-only chunk iterator.
33    pub fn new(image: &'a I, cursor_shape: Vec<usize>) -> Self {
34        Self {
35            inner: TraversalIter::new(image, TraversalSpec::chunks(cursor_shape)),
36        }
37    }
38}
39
40impl<'a, T: ImagePixel, I: Lattice<T>> Iterator for ImageIter<'a, T, I> {
41    type Item = Result<ImageChunk<T>, ImageError>;
42
43    fn next(&mut self) -> Option<Self::Item> {
44        self.inner.next().map(|chunk| {
45            chunk
46                .map(|chunk| ImageChunk {
47                    data: chunk.data,
48                    origin: chunk.cursor.position,
49                    shape: chunk.cursor.shape,
50                })
51                .map_err(ImageError::from)
52        })
53    }
54}
55
56/// Mutable chunk iterator over any writable typed image-like lattice.
57pub struct ImageIterMut<'a, T: ImagePixel, I: LatticeMut<T>> {
58    image: &'a mut I,
59    cursors: TraversalCursorIter,
60    _pixel: std::marker::PhantomData<T>,
61}
62
63impl<'a, T: ImagePixel, I: LatticeMut<T>> ImageIterMut<'a, T, I> {
64    /// Creates a new mutable chunk iterator.
65    pub fn new(image: &'a mut I, cursor_shape: Vec<usize>) -> Self {
66        let cursors = TraversalCursorIter::new(
67            image.shape().to_vec(),
68            image.nice_cursor_shape(),
69            TraversalSpec::chunks(cursor_shape),
70        );
71        Self {
72            image,
73            cursors,
74            _pixel: std::marker::PhantomData,
75        }
76    }
77
78    /// Reads the next chunk and advances the iterator.
79    pub fn next_chunk(&mut self) -> Option<Result<ImageChunk<T>, ImageError>> {
80        self.cursors.next().map(|cursor| {
81            cursor.map_err(ImageError::from).and_then(|cursor| {
82                let stride = vec![1; cursor.position.len()];
83                self.image
84                    .get_slice(&cursor.position, &cursor.shape, &stride)
85                    .map(|data| ImageChunk {
86                        data,
87                        origin: cursor.position,
88                        shape: cursor.shape,
89                    })
90                    .map_err(ImageError::from)
91            })
92        })
93    }
94
95    /// Writes a modified chunk back to the parent image.
96    pub fn flush_chunk(&mut self, chunk: &ImageChunk<T>) -> Result<(), ImageError> {
97        self.image
98            .put_slice(&chunk.data, &chunk.origin)
99            .map_err(ImageError::from)
100    }
101}

103#[cfg(test)]
104mod tests {
105    use super::*;
106    use crate::temp_image::TempImage;
107    use casacore_coordinates::CoordinateSystem;
108    use casacore_types::Complex32;
109    use ndarray::IxDyn;
110
111    fn make_image<T: ImagePixel>(shape: Vec<usize>) -> TempImage<T> {
112        TempImage::new(shape, CoordinateSystem::new()).unwrap()
113    }
114
115    #[test]
116    fn full_scan_sum() {
117        let mut img = make_image::<f32>(vec![8, 6]);
118        img.set(1.0).unwrap();
119
120        let mut total = 0.0f32;
121        for chunk in ImageIter::new(&img, vec![4, 3]) {
122            total += chunk.unwrap().data.sum();
123        }
124        assert_eq!(total, 48.0);
125    }
126
127    #[test]
128    fn chunk_count_exact_division() {
129        let img = make_image::<f32>(vec![8, 6]);
130        assert_eq!(ImageIter::new(&img, vec![4, 3]).count(), 4);
131    }
132
133    #[test]
134    fn chunk_count_non_divisible() {
135        let img = make_image::<f32>(vec![7, 5]);
136        assert_eq!(ImageIter::new(&img, vec![4, 3]).count(), 4);
137    }
138
139    #[test]
140    fn hangover_shape() {
141        let img = make_image::<f32>(vec![7, 5]);
142        let chunks: Vec<_> = ImageIter::new(&img, vec![4, 3])
143            .map(|c| c.unwrap())
144            .collect();
145
146        assert_eq!(chunks[0].shape, vec![4, 3]);
147        assert_eq!(chunks[0].origin, vec![0, 0]);
148        assert_eq!(chunks[3].shape, vec![3, 2]);
149        assert_eq!(chunks[3].origin, vec![4, 3]);
150    }
151
152    #[test]
153    fn covers_all_elements() {
154        let shape = vec![7, 5];
155        let mut img = make_image::<f32>(shape.clone());
156        let data = ArrayD::from_shape_fn(IxDyn(&shape), |idx| (idx[0] * 5 + idx[1]) as f32);
157        img.put_slice(&data, &[0, 0]).unwrap();
158
159        let mut sum = 0.0f32;
160        let mut pixel_count = 0usize;
161        for chunk in ImageIter::new(&img, vec![3, 2]) {
162            let c = chunk.unwrap();
163            sum += c.data.sum();
164            pixel_count += c.data.len();
165        }
166        assert_eq!(pixel_count, 35);
167        assert_eq!(sum, data.sum());
168    }
169
170    #[test]
171    fn mutable_iteration() {
172        let mut img = make_image::<f32>(vec![4, 4]);
173        img.set(1.0).unwrap();
174
175        let mut iter = ImageIterMut::new(&mut img, vec![2, 2]);
176        while let Some(Ok(mut chunk)) = iter.next_chunk() {
177            chunk.data.mapv_inplace(|v| v * 2.0);
178            iter.flush_chunk(&chunk).unwrap();
179        }
180
181        let arr = img.get().unwrap();
182        assert!(arr.iter().all(|&v| v == 2.0));
183    }
184
185    #[test]
186    fn one_d_image() {
187        let mut img = make_image::<f32>(vec![10]);
188        img.set(3.0).unwrap();
189        let mut total = 0.0f32;
190        for chunk in ImageIter::new(&img, vec![4]) {
191            total += chunk.unwrap().data.sum();
192        }
193        assert_eq!(total, 30.0);
194    }
195
196    #[test]
197    fn complex_iteration_preserves_values() {
198        let mut img = make_image::<Complex32>(vec![2, 2]);
199        img.put_at(Complex32::new(1.0, -1.0), &[0, 0]).unwrap();
200        let first = ImageIter::new(&img, vec![1, 2]).next().unwrap().unwrap();
201        assert_eq!(first.data[[0, 0]], Complex32::new(1.0, -1.0));
202    }
203
204    #[test]
205    fn temp_image_iteration() {
206        let mut img = TempImage::<f32>::new(vec![2, 2], CoordinateSystem::new()).unwrap();
207        img.set(1.0).unwrap();
208        assert_eq!(ImageIter::new(&img, vec![1, 1]).count(), 4);
209    }
