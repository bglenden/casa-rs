29/// Byte-ordering format for on-disk table data.
30///
31/// Controls how multi-byte values (integers, floats, complex numbers) are
32/// stored in the table's data files. The choice is recorded in `table.dat`
33/// so that readers can decode values correctly regardless of the host machine's
34/// native byte order.
35///
36/// In C++ casacore this corresponds to `Table::EndianFormat`. The
37/// `AipsrcEndian` variant from C++ is intentionally omitted — Rust callers
38/// should pass an explicit format instead.
39///
40/// # Default
41///
42/// The default is [`LocalEndian`](EndianFormat::LocalEndian), which uses the
43/// byte order of the machine that creates the table. This matches the C++
44/// casacore default.
45#[derive(Debug, Clone, Copy, PartialEq, Eq, Default)]
46pub enum EndianFormat {
47    /// Store data in big-endian (network / "canonical") byte order.
48    ///
49    /// Corresponds to `Table::BigEndian` in C++ casacore. This was the only
50    /// format supported by early casacore versions.
51    BigEndian,
52    /// Store data in little-endian byte order.
53    ///
54    /// Corresponds to `Table::LittleEndian` in C++ casacore.
55    LittleEndian,
56    /// Store data in the byte order of the host machine.
57    ///
58    /// On x86-64 and ARM64 this resolves to little-endian; on SPARC or
59    /// PowerPC (big-endian mode) it resolves to big-endian.
60    /// Corresponds to `Table::LocalEndian` in C++ casacore.
61    #[default]
62    LocalEndian,
63}
64
65impl EndianFormat {
66    /// Resolves this format to a concrete big-endian flag.
67    ///
68    /// [`LocalEndian`](EndianFormat::LocalEndian) queries the host at compile
69    /// time via `cfg!(target_endian = "big")`.
70    pub fn is_big_endian(self) -> bool {
71        match self {
72            Self::BigEndian => true,
73            Self::LittleEndian => false,
74            Self::LocalEndian => cfg!(target_endian = "big"),
75        }
76    }
77}

96/// The kind of table: plain (disk-backed) or memory (transient).
97///
98/// In C++ casacore this corresponds to `Table::TableType`. A
99/// [`Plain`](TableKind::Plain) table can be loaded from and saved to disk
100/// with any [`DataManagerKind`]. A [`Memory`](TableKind::Memory) table
101/// holds all data exclusively in process memory; it is deleted when
102/// dropped.
103///
104/// Memory tables can be materialized to disk via [`Table::save`], which
105/// writes a plain table that is byte-identical to one created directly.
106///
107/// # C++ equivalent
108///
109/// `Table::Plain` and `Table::Memory`.
110#[derive(Debug, Clone, Copy, PartialEq, Eq, Default)]
111pub enum TableKind {
112    /// A regular table, backed by (or destined for) disk storage.
113    #[default]
114    Plain,
115    /// A transient in-memory table.
116    ///
117    /// All data is lost when the table is dropped. Locking is a no-op:
118    /// [`has_lock`](Table::has_lock) always returns `true`, and
119    /// [`lock`](Table::lock) / [`unlock`](Table::unlock) succeed without
120    /// doing any I/O.
121    ///
122    /// C++ equivalent: `Table::Memory` / `MemoryTable`.
123    Memory,
124}

126/// Which data manager to use when writing table data.
127///
128/// This choice is recorded in the table descriptor on disk so that C++ casacore
129/// can select the correct storage-manager plugin when reopening the table.
130/// All variants produce files that are binary-compatible with upstream casacore.
131#[derive(Debug, Clone, Copy, PartialEq, Eq, Default)]
132pub enum DataManagerKind {
133    /// Simple whole-column AipsIO streaming (legacy format).
134    ///
135    /// Each column is stored as a single flat AipsIO stream. This is the
136    /// simplest on-disk layout and is compatible with older versions of
137    /// casacore. It is the default for this crate because it requires no
138    /// extra configuration and produces self-contained files.
139    #[default]
140    StManAipsIO,
141    /// Bucket-based storage (the default in C++ casacore).
142    ///
143    /// Data is partitioned into fixed-size buckets, which allows efficient
144    /// random access and in-place updates. This is the storage manager used
145    /// by default when creating tables with C++ casacore.
146    StandardStMan,
147    /// Delta-compression storage manager (C++ `IncrementalStMan` / `ISMBase`).
148    ///
149    /// Stores column values only when they change from the previous row,
150    /// making it extremely space-efficient for slowly-changing columns
151    /// such as `ANTENNA1`, `FEED_ID`, or `SCAN_NUMBER` in a
152    /// MeasurementSet. Along with `StandardStMan`, ISM is one of the two
153    /// most commonly used storage managers in real radio astronomy data.
154    IncrementalStMan,
155    /// Tiled storage: single hypercube for the entire column.
156    ///
157    /// All rows must have the same array shape. Data is stored in
158    /// rectangular tiles within a single hypercube whose last dimension
159    /// is the row count. This is the standard format for large fixed-shape
160    /// array columns in measurement sets and images.
161    ///
162    /// C++ equivalent: `TiledColumnStMan`.
163    TiledColumnStMan,
164    /// Tiled storage: one hypercube per unique array shape.
165    ///
166    /// Rows with different array shapes are automatically grouped into
167    /// separate hypercubes. An internal row map tracks which cube holds
168    /// each row. This is the standard format for variable-shape array
169    /// columns (e.g. visibilities with varying channel counts).
170    ///
171    /// C++ equivalent: `TiledShapeStMan`.
172    TiledShapeStMan,
173    /// Tiled storage: one hypercube per row.
174    ///
175    /// Each row has its own cube, allowing fully variable shapes per row.
176    /// Most memory-intensive variant; use `TiledShapeStMan` when many rows
177    /// share shapes.
178    ///
179    /// C++ equivalent: `TiledCellStMan`.
180    TiledCellStMan,
181    /// Tiled storage: user-controlled hypercube assignment.
182    ///
183    /// Like `TiledShapeStMan` but with explicit row-to-cube assignment
184    /// rather than automatic shape-based grouping. Found in some older
185    /// datasets.
186    ///
187    /// C++ equivalent: `TiledDataStMan`.
188    TiledDataStMan,
189}

215/// Configuration for opening or saving a [`Table`] to disk.
216///
217/// `TableOptions` bundles the filesystem path with the choice of storage
218/// manager and byte-ordering format. In C++ casacore this information is
219/// passed directly to the `Table(name, option)` constructor. Here it is
220/// factored out into a separate builder so that callers can construct
221/// options independently of the table itself.
222///
223/// # Example
224///
225/// ```rust
226/// use casacore_tables::{TableOptions, DataManagerKind, EndianFormat};
227///
228/// let opts = TableOptions::new("/tmp/my_table")
229///     .with_data_manager(DataManagerKind::StandardStMan)
230///     .with_endian_format(EndianFormat::BigEndian);
231///
232/// // Use IncrementalStMan for slowly-changing columns:
233/// let ism_opts = TableOptions::new("/tmp/my_ism_table")
234///     .with_data_manager(DataManagerKind::IncrementalStMan);
235/// ```
236#[derive(Debug, Clone, PartialEq, Eq)]
237pub struct TableOptions {
238    path: PathBuf,
239    data_manager: DataManagerKind,
240    endian_format: EndianFormat,
241    tile_shape: Option<Vec<usize>>,
242}
243
244impl TableOptions {
245    /// Creates options targeting `path` with the default data manager
246    /// ([`DataManagerKind::StManAipsIO`]) and default endian format
247    /// ([`EndianFormat::LocalEndian`]).
248    pub fn new(path: impl AsRef<Path>) -> Self {
249        Self {
250            path: path.as_ref().to_path_buf(),
251            data_manager: DataManagerKind::default(),
252            endian_format: EndianFormat::default(),
253            tile_shape: None,
254        }
255    }
256
257    /// Overrides the data manager, returning the updated options.
258    pub fn with_data_manager(mut self, kind: DataManagerKind) -> Self {

261    }
262
263    /// Overrides the endian format, returning the updated options.
264    ///
265    /// The endian format controls the byte ordering of multi-byte values
266    /// in the table's data files. It is only used when saving; on open the
267    /// format is detected from the existing `table.dat` marker.
268    pub fn with_endian_format(mut self, format: EndianFormat) -> Self {
269        self.endian_format = format;

271    }

289    }

294    }

297    pub fn endian_format(&self) -> EndianFormat {
298        self.endian_format
299    }
300
301    /// Returns the tile shape, if set.
302    pub fn tile_shape(&self) -> Option<&[usize]> {
303        self.tile_shape.as_deref()
304    }

307/// A contiguous, optionally strided range of row indices.
308///
309/// `RowRange` selects which rows participate in a column read or write
310/// operation. It is the row-axis analogue of C++ casacore's `Slicer`.
311///
312/// # Semantics
313///
314/// The range is half-open: rows are selected from `start` (inclusive) to
315/// `end` (exclusive) stepping by `stride`. For example, `RowRange::with_stride(0, 6, 2)`
316/// selects rows 0, 2, and 4.
317///
318/// # Example
319///
320/// ```rust
321/// use casacore_tables::RowRange;
322///
323/// // Every row in a 10-row table:
324/// let all = RowRange::new(0, 10);
325///
326/// // Every other row:
327/// let evens = RowRange::with_stride(0, 10, 2);
328/// ```
329#[derive(Debug, Clone, Copy, PartialEq, Eq)]
330pub struct RowRange {
331    start: usize,
332    end: usize,
333    stride: usize,
334}
335
336impl RowRange {
337    /// Creates a contiguous range `[start, end)` with stride 1.
338    pub const fn new(start: usize, end: usize) -> Self {
339        Self {
340            start,
341            end,
342            stride: 1,
343        }
344    }
345
346    /// Creates a strided range `[start, end)` stepping by `stride`.
347    pub const fn with_stride(start: usize, end: usize, stride: usize) -> Self {
348        Self { start, end, stride }
349    }
350
351    /// Returns the first row index (inclusive).
352    pub const fn start(&self) -> usize {
353        self.start
354    }
355
356    /// Returns the past-the-end row index (exclusive).
357    pub const fn end(&self) -> usize {
358        self.end
359    }
360
361    /// Returns the step between successive selected rows.
362    pub const fn stride(&self) -> usize {
363        self.stride
364    }
365
366    fn validate(&self, row_count: usize) -> Result<(), TableError> {
367        if self.stride == 0 {
368            return Err(TableError::InvalidRowStride {
369                stride: self.stride,
370            });
371        }
372        if self.start > self.end || self.end > row_count {
373            return Err(TableError::InvalidRowRange {
374                start: self.start,
375                end: self.end,
376                row_count,
377            });
378        }
379        Ok(())
380    }
381
382    fn len(&self) -> usize {
383        if self.start >= self.end || self.stride == 0 {
384            0
385        } else {
386            1 + ((self.end - self.start - 1) / self.stride)
387        }
388    }
389
390    fn iter(&self) -> RowRangeIter {
391        RowRangeIter {
392            next: self.start,
393            end: self.end,
394            stride: self.stride,
395        }
396    }
397}

399/// An n-dimensional sub-region selector for array-valued cells.
400///
401/// Specifies `start`, `end` (exclusive), and `stride` along each dimension.
402/// All three vectors must have equal length matching the array dimensionality.
403///
404/// # C++ equivalent
405///
406/// `casacore::Slicer` — `casa/Arrays/Slicer.h`.
407///
408/// # Example
409///
410/// ```rust
411/// use casacore_tables::Slicer;
412///
413/// // Select rows 0..4 step 2, columns 1..3 step 1 from a 2-D array:
414/// let s = Slicer::new(vec![0, 1], vec![4, 3], vec![2, 1]).unwrap();
415/// assert_eq!(s.ndim(), 2);
416/// ```
417#[derive(Debug, Clone, PartialEq, Eq)]
418pub struct Slicer {
419    start: Vec<usize>,
420    end: Vec<usize>,
421    stride: Vec<usize>,
422}
423
424impl Slicer {
425    /// Creates a new slicer with the given start, end (exclusive), and stride.
426    ///
427    /// Returns an error if the vectors have different lengths or any stride is zero.
428    pub fn new(start: Vec<usize>, end: Vec<usize>, stride: Vec<usize>) -> Result<Self, TableError> {
429        if start.len() != end.len() || start.len() != stride.len() {
430            return Err(TableError::SlicerDimensionMismatch {
431                start_ndim: start.len(),
432                end_ndim: end.len(),
433                stride_ndim: stride.len(),
434            });
435        }
436        for (i, &s) in stride.iter().enumerate() {
437            if s == 0 {
438                return Err(TableError::SlicerZeroStride { axis: i });
439            }
440        }
441        for (i, (&s, &e)) in start.iter().zip(end.iter()).enumerate() {
442            if s > e {
443                return Err(TableError::SlicerInvalidRange {
444                    axis: i,
445                    start: s,
446                    end: e,
447                });
448            }
449        }
450        Ok(Self { start, end, stride })
451    }
452
453    /// Creates a contiguous slicer (stride 1 on all axes).
454    pub fn contiguous(start: Vec<usize>, end: Vec<usize>) -> Result<Self, TableError> {
455        let stride = vec![1; start.len()];
456        Self::new(start, end, stride)
457    }
458
459    /// Number of dimensions.
460    pub fn ndim(&self) -> usize {
461        self.start.len()
462    }
463
464    /// Start indices (inclusive) along each dimension.
465    pub fn start(&self) -> &[usize] {
466        &self.start
467    }
468
469    /// End indices (exclusive) along each dimension.
470    pub fn end(&self) -> &[usize] {
471        &self.end
472    }
473
474    /// Stride along each dimension.
475    pub fn stride(&self) -> &[usize] {
476        &self.stride
477    }
478}

480/// A borrowed reference to a single cell value together with its row index.
481///
482/// `ColumnCellRef` is the item type yielded by [`ColumnCellIter`] and
483/// [`ColumnChunkIter`]. The `value` field is `None` when the cell is absent
484/// from the row (possible for columns that allow undefined cells).
485#[derive(Debug, Clone, Copy, PartialEq)]
486pub struct ColumnCellRef<'a> {
487    /// Zero-based index of the row from which this cell was read.
488    pub row_index: usize,
489    /// The cell value, or `None` if the cell is absent for this row.
490    pub value: Option<&'a Value>,
491}

493/// An owned record cell value together with its row index.
494///
495/// `RecordColumnCell` is the item type yielded by [`RecordColumnIter`]. Unlike
496/// [`ColumnCellRef`], the value is always present: missing record cells are
497/// substituted with an empty [`RecordValue`] when the column schema permits
498/// absent cells.
499#[derive(Debug, Clone, PartialEq)]
500pub struct RecordColumnCell {
501    /// Zero-based index of the row from which this cell was read.
502    pub row_index: usize,
503    /// The record value for this row (never absent; defaults to empty record).
504    pub value: RecordValue,
505}

507/// An iterator over cells in a single column, yielding one [`ColumnCellRef`] per row.
508///
509/// Obtain a `ColumnCellIter` via [`Table::get_column`] or [`Table::get_column_range`].
510/// The iterator borrows the table's row data and does not allocate per cell.
511/// For batch processing, see [`ColumnChunkIter`] via [`Table::iter_column_chunks`].
512pub struct ColumnCellIter<'a> {
513    row_data: &'a [RecordValue],
514    column: &'a str,
515    rows: RowRangeIter,
516}
517
518impl<'a> Iterator for ColumnCellIter<'a> {
519    type Item = ColumnCellRef<'a>;
520
521    fn next(&mut self) -> Option<Self::Item> {
522        self.rows.next().map(|row_index| ColumnCellRef {
523            row_index,
524            value: self.row_data[row_index].get(self.column),
525        })
526    }
527}

529/// An iterator over cells in a record column, yielding one [`RecordColumnCell`] per row.
530///
531/// Obtain a `RecordColumnIter` via [`Table::get_record_column`] or
532/// [`Table::get_record_column_range`]. When a [`TableSchema`] is attached and
533/// the column is typed as [`crate::schema::ColumnType::Record`], rows whose
534/// record cell is absent yield an empty [`RecordValue`] rather than an error.
535pub struct RecordColumnIter<'a> {
536    row_data: &'a [RecordValue],
537    column: &'a str,
538    rows: RowRangeIter,
539    default_missing: bool,
540}
541
542impl<'a> Iterator for RecordColumnIter<'a> {
543    type Item = RecordColumnCell;
544
545    fn next(&mut self) -> Option<Self::Item> {
546        self.rows.next().map(|row_index| {
547            let value = match self.row_data[row_index].get(self.column) {
548                Some(Value::Record(record)) => record.clone(),
549                Some(_) => unreachable!("record iterator was prevalidated"),
550                None => {
551                    if self.default_missing {
552                        RecordValue::default()
553                    } else {
554                        unreachable!("record iterator was prevalidated")
555                    }
556                }
557            };
558            RecordColumnCell { row_index, value }
559        })
560    }
561}

563/// An iterator that yields column cells in fixed-size batches.
564///
565/// Each call to `next` returns a `Vec<ColumnCellRef>` containing up to
566/// `chunk_size` cells. The final chunk may be smaller if the remaining row
567/// count is not a multiple of `chunk_size`. An empty table (or an exhausted
568/// range) causes `next` to return `None` immediately.
569///
570/// Obtain a `ColumnChunkIter` via [`Table::iter_column_chunks`]. This iterator
571/// is useful for processing columns in memory-bounded passes without
572/// materializing the entire column at once.
573pub struct ColumnChunkIter<'a> {
574    inner: ColumnCellIter<'a>,
575    chunk_size: usize,
576}
577
578impl<'a> Iterator for ColumnChunkIter<'a> {
579    type Item = Vec<ColumnCellRef<'a>>;
580
581    fn next(&mut self) -> Option<Self::Item> {
582        let chunk: Vec<_> = self.inner.by_ref().take(self.chunk_size).collect();
583        if chunk.is_empty() { None } else { Some(chunk) }
584    }
585}
586
587#[derive(Debug, Clone, Copy)]
588struct RowRangeIter {
589    next: usize,
590    end: usize,
591    stride: usize,
592}
593
594impl Iterator for RowRangeIter {
595    type Item = usize;
596
597    fn next(&mut self) -> Option<Self::Item> {
598        if self.next >= self.end {
599            return None;
600        }
601        let row_index = self.next;
602        self.next = self.next.saturating_add(self.stride);
603        Some(row_index)
604    }
605}

607/// Result of a TaQL SELECT query via [`Table::query_result()`].
608///
609/// Simple SELECTs (only column references) produce a zero-copy
610/// [`View`](QueryResult::View). SELECTs with computed expressions, GROUP BY,
611/// or aggregate functions produce a [`Materialized`](QueryResult::Materialized)
612/// in-memory table containing the evaluated result rows.
613///
614/// Both variants support reading column names and row data.
615///
616/// C++ equivalent: the result of `tableCommand()` — either a `Table` reference
617/// or a newly materialized `Table` from `makeProjectExprTable()`.
618pub enum QueryResult<'a> {
619    /// A zero-copy view into the source table (row-index mapping).
620    View(crate::RefTable<'a>),
621    /// An owned in-memory table with computed/aggregated values (boxed to
622    /// keep the enum size small).
623    Materialized(Box<Table>),
624}
625
626impl std::fmt::Debug for QueryResult<'_> {
627    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
628        match self {
629            Self::View(v) => write!(f, "QueryResult::View({} rows)", v.row_count()),
630            Self::Materialized(t) => write!(f, "QueryResult::Materialized({} rows)", t.row_count()),
631        }
632    }
633}
634
635impl QueryResult<'_> {
636    /// Returns the number of result rows.
637    pub fn row_count(&self) -> usize {
638        match self {
639            QueryResult::View(v) => v.row_count(),
640            QueryResult::Materialized(t) => t.row_count(),
641        }
642    }
643
644    /// Returns the column names of the result.
645    pub fn column_names(&self) -> Vec<String> {
646        match self {
647            QueryResult::View(v) => v.column_names().to_vec(),
648            QueryResult::Materialized(t) => t
649                .schema()
650                .map(|s| s.columns().iter().map(|c| c.name().to_string()).collect())
651                .unwrap_or_default(),
652        }
653    }
654
655    /// Returns the row at the given index.
656    pub fn row(&self, index: usize) -> Result<&RecordValue, TableError> {
657        match self {
658            QueryResult::View(v) => v.row(index),
659            QueryResult::Materialized(t) => t.row(index),
660        }
661    }
662}

664/// Errors that can occur when reading or writing a [`Table`].
665#[derive(Debug, Error, PartialEq, Eq)]
666pub enum TableError {
667    /// A row index exceeded the number of rows in the table.
668    #[error("row index {row_index} is out of bounds for table with {row_count} rows")]
669    RowOutOfBounds { row_index: usize, row_count: usize },
670    /// A [`RowRange`] had `start > end` or `end > row_count`.
671    #[error("row range [{start}, {end}) is invalid for table with {row_count} rows")]
672    InvalidRowRange {
673        start: usize,
674        end: usize,
675        row_count: usize,
676    },
677    /// A [`RowRange`] stride was zero, which would produce an infinite loop.

885/// A casacore table — the fundamental persistent data container.
886///
887/// A `Table` holds a set of named columns (scalar or array) plus keyword
888/// metadata, backed either by on-disk storage or an in-memory buffer.
889///
890/// # Lifecycle
891///
892/// * **Open** — [`Table::open`] reads an existing table directory.
893/// * **Create** — [`Table::create`] builds a new table from a schema.
894/// * **Save** — [`Table::save`] flushes deferred writes to disk.
895///
896/// # C++ equivalent
897///
898/// `casacore::Table` — the main user-facing class in the Tables module.
899#[derive(Default)]
900pub struct Table {
901    inner: TableImpl,
902    /// Filesystem path this table was last opened from or saved to.
903    source_path: Option<PathBuf>,
904    /// Whether this is a plain (disk-backed) or memory (transient) table.
905    kind: TableKind,
906    /// Names of columns backed by virtual engines (ForwardColumnEngine,
907    /// ScaledArrayEngine, etc.). Empty for tables with no virtual columns.
908    virtual_columns: HashSet<String>,
909    /// Virtual column bindings for save — describes how each virtual column
910    /// maps to its engine and configuration.
911    virtual_bindings: Vec<VirtualColumnBinding>,
912    /// Table metadata (type/subtype) persisted in `table.info`.
913    table_info: TableInfo,
914    /// Data manager info extracted from table.dat (empty for memory tables).
915    dm_info: Vec<crate::storage::DataManagerInfo>,
916    /// Optional external lock synchronization hook.
917    external_sync: Option<Box<dyn crate::lock::ExternalLockSync>>,
918    /// When `true`, the table directory is deleted on [`Drop`].
919    ///
920    /// Set via [`mark_for_delete`](Table::mark_for_delete), cleared via
921    /// [`unmark_for_delete`](Table::unmark_for_delete).
922    marked_for_delete: bool,
923    #[cfg(unix)]
924    lock_state: Option<LockState>,
925}
926
927impl std::fmt::Debug for Table {
928    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
929        let mut s = f.debug_struct("Table");
930        s.field("inner", &self.inner)
931            .field("source_path", &self.source_path)
932            .field("kind", &self.kind)
933            .field("virtual_columns", &self.virtual_columns)
934            .field("table_info", &self.table_info)
935            .field("marked_for_delete", &self.marked_for_delete)
936            .field(
937                "external_sync",
938                &self.external_sync.as_ref().map(|_| "<hook>"),
939            );
940        #[cfg(unix)]
941        s.field("lock_state", &self.lock_state);
942        s.finish()
943    }
944}

1194
1195impl Drop for Table {
1196    fn drop(&mut self) {
1197        #[cfg(unix)]
1198        {
1199            if self.kind != TableKind::Memory {
1200                let had_write_lock = self
1201                    .lock_state
1202                    .as_ref()
1203                    .is_some_and(|state| state.lock_file.has_lock(LockType::Write));
1204
1205                if had_write_lock {
1206                    let save_opts = self.lock_state.as_ref().map(|state| {
1207                        TableOptions::new(&state.path)
1208                            .with_data_manager(state.data_manager)
1209                            .with_endian_format(state.endian_format)
1210                    });
1211
1212                    if let Some(save_opts) = save_opts
1213                        && self.save(save_opts).is_ok()
1214                    {
1215                        let nrrow = self.row_count() as u64;
1216                        let nrcolumn = self.schema().map(|s| s.columns().len() as u32).unwrap_or(0);
1217                        if let Some(state) = self.lock_state.as_mut() {
1218                            state.sync_data.record_write(nrrow, nrcolumn, true, &[true]);
1219                            let _ = state.lock_file.write_sync_data(&state.sync_data);
1220                        }
1221                    }
1222                }
1223            }
1224        }
1225
1226        if self.marked_for_delete {
1227            if let Some(path) = &self.source_path {
1228                let _ = std::fs::remove_dir_all(path);
1229            }
1230        }
1231    }
