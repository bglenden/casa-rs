19/// Structured imaging run report consumed by the `casars` workflow shell.
20#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
21pub struct ManagedImagingOutput {
22    /// High-level request summary used by the shell overview.
23    pub request: ManagedImagingRequest,
24    /// Structured run metrics and per-channel diagnostics.
25    pub run: ManagedImagingRun,
26    /// Expected CASA image products written under the configured prefix.
27    pub artifacts: Vec<ManagedImagingArtifact>,
28}
29
30/// Launcher-facing view of the requested imaging configuration.
31#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
32pub struct ManagedImagingRequest {
33    /// MeasurementSet path supplied for the run.
34    pub measurement_set: String,
35    /// Output CASA image prefix.
36    pub imagename: String,
37    /// Spectral imaging mode.
38    pub spectral_mode: String,
39    /// Requested visibility weighting.
40    pub weighting: String,
41    /// Requested minor-cycle deconvolver.
42    pub deconvolver: String,
43    /// Hogbom minor-cycle iteration accounting policy.
44    pub hogbom_iteration_mode: String,
45    /// Requested `w`-term handling mode.
46    pub w_term_mode: String,
47    /// Optional data-column override.
48    pub data_column: Option<String>,
49    /// Requested model persistence mode.
50    pub save_model: String,
51    /// Image size in pixels.
52    pub imsize: usize,
53    /// Cell size in arcseconds.
54    pub cell_arcsec: f64,
55    /// Whether the run skipped CLEAN.
56    pub dirty_only: bool,
57    /// Whether preview PNG sidecars were requested.
58    pub write_preview_pngs: bool,
59    /// Whether the primary-beam product was requested.
60    pub write_pb: bool,
61    /// Whether per-channel density estimation was requested for cube weighting.
62    pub per_channel_weight_density: bool,
63    /// Requested MTMFS Taylor-term count.
64    pub nterms: usize,
65    /// Output channel count for cube-like runs.
66    pub output_channels: usize,
67    /// Requested raw-correlation or Stokes plane, when explicitly selected.
68    pub correlation: Option<String>,
69    /// Requested restoring-beam mode for restored products.
70    pub restoring_beam_mode: String,
71}

73/// Structured run metrics emitted after one successful imaging run.
74#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
75pub struct ManagedImagingRun {
76    /// Warnings emitted by the imaging run.
77    pub warnings: Vec<String>,
78    /// Number of scalar samples that reached the gridder.
79    pub gridded_samples: usize,
80    /// Total major-cycle count reported by the run.
81    pub major_cycles: usize,
82    /// Total minor-cycle component updates reported by the run.
83    pub minor_iterations: usize,
84    /// Final CLEAN stop reason when deconvolution ran.
85    pub clean_stop_reason: Option<String>,
86    /// Timing breakdown reported by the pure imaging core.
87    pub stage_timings: ManagedImagingStageTimings,
88    /// Timing breakdown for the MeasurementSet-backed frontend.
89    pub frontend_timings: ManagedImagingStageTimings,
90    /// Channel-level diagnostics for cube-like runs.
91    pub channels: Vec<ManagedImagingChannelRun>,
92}

101/// Channel-level convergence summary emitted for cube-like runs.
102#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
103pub struct ManagedImagingChannelRun {
104    /// Zero-based output channel index.
105    pub channel_index: usize,
106    /// Major-cycle count for this plane.
107    pub major_cycles: usize,
108    /// Minor-cycle component updates for this plane.
109    pub minor_iterations: usize,
110    /// Final CLEAN stop reason for this plane.
111    pub clean_stop_reason: Option<String>,
112    /// Peak residual before minor cycles.
113    pub initial_residual_peak_jy_per_beam: f32,
114    /// Peak residual after the final exact refresh.
115    pub final_residual_peak_jy_per_beam: f32,
116    /// Final CASA-style cycle threshold for this plane.
117    pub final_cycle_threshold_jy_per_beam: f32,
118    /// Whether the beam-fit debug summary was available for this plane.
119    pub beam_fit_available: bool,
120}

140    /// Build the structured launcher report from one completed run.
141    pub fn from_run(config: &CliConfig, summary: &RunSummary) -> Self {
142        Self {
143            request: ManagedImagingRequest {
144                measurement_set: config.ms.display().to_string(),
145                imagename: config.imagename.display().to_string(),
146                spectral_mode: canonical_spectral_mode_name(config.spectral_mode).to_string(),
147                weighting: canonical_weighting_name(config.weighting),
148                deconvolver: canonical_deconvolver_name(config.deconvolver).to_string(),
149                hogbom_iteration_mode: canonical_hogbom_iteration_mode_name(
150                    config.hogbom_iteration_mode,
151                )
152                .to_string(),
153                w_term_mode: canonical_w_term_mode_name(config.w_term_mode).to_string(),
154                data_column: config.datacolumn.clone(),
155                save_model: match config.save_model {
156                    crate::SaveModelMode::None => "none",
157                    crate::SaveModelMode::ModelColumn => "modelcolumn",
158                }
159                .to_string(),
160                imsize: config.imsize,
161                cell_arcsec: config.cell_arcsec,
162                dirty_only: config.dirty_only,
163                write_preview_pngs: config.write_preview_pngs,
164                write_pb: config.write_pb,
165                per_channel_weight_density: config.per_channel_weight_density,
166                nterms: config.nterms,
167                output_channels: summary.channel_summaries.len(),
168                correlation: config.correlation.clone(),
169                restoring_beam_mode: canonical_restoring_beam_mode_name(config.restoring_beam_mode)
170                    .to_string(),
171            },
172            run: ManagedImagingRun {
173                warnings: summary.warnings.clone(),
174                gridded_samples: summary.gridded_samples,
175                major_cycles: summary.major_cycles,
176                minor_iterations: summary.minor_iterations,
177                clean_stop_reason: summary
178                    .clean_stop_reason
179                    .map(|reason| format!("{reason:?}")),
180                stage_timings: stage_timings_from_core(summary),
181                frontend_timings: stage_timings_from_frontend(summary.frontend_timings),
182                channels: summary
183                    .channel_summaries
184                    .iter()
185                    .map(channel_run_from_summary)
186                    .collect(),
187            },
188            artifacts: imaging_artifacts(config),
189        }
190    }

499fn imaging_artifacts(config: &CliConfig) -> Vec<ManagedImagingArtifact> {
500    let base = config.imagename.to_string_lossy().to_string();
501    let mut artifacts = Vec::new();
502    match config.spectral_mode {
503        SpectralMode::Mfs
504            if canonical_deconvolver_name(config.deconvolver) == "mtmfs" && config.nterms > 1 =>
505        {
506            for term in 0..config.nterms {
507                for (kind, label) in [
508                    ("psf", "PSF"),
509                    ("residual", "Residual"),
510                    ("model", "Model"),
511                    ("image", "Restored Image"),
512                ] {
513                    let suffix = format!("{kind}.tt{term}");
514                    let path = PathBuf::from(format!("{base}.{suffix}"));
515                    let preview = (term == 0 && config.write_preview_pngs)
516                        .then(|| PathBuf::from(format!("{base}.{suffix}.png")));
517                    artifacts.push(artifact(label_for_term(label, term), kind, path, preview));
518                }
519            }
520            let alpha = PathBuf::from(format!("{base}.alpha"));
521            let alpha_preview = config
522                .write_preview_pngs
523                .then(|| PathBuf::from(format!("{base}.alpha.png")));
524            artifacts.push(artifact(
525                "Spectral Index".to_string(),
526                "alpha",
527                alpha,
528                alpha_preview,
529            ));
530        }
531        _ => {
532            for (kind, label) in [
533                ("psf", "PSF"),
534                ("residual", "Residual"),
535                ("model", "Model"),
536                ("image", "Restored Image"),
537            ] {
538                let path = PathBuf::from(format!("{base}.{kind}"));
539                let preview = config
540                    .write_preview_pngs
541                    .then(|| PathBuf::from(format!("{base}.{kind}.png")));
542                artifacts.push(artifact(label.to_string(), kind, path, preview));
543            }
544            if config.write_pb || config.pbcor {
545                let path = PathBuf::from(format!("{base}.pb"));
546                let preview = config
547                    .write_preview_pngs
548                    .then(|| PathBuf::from(format!("{base}.pb.png")));
549                artifacts.push(artifact("Primary Beam".to_string(), "pb", path, preview));
550            }
551            if config.pbcor {
552                let path = PathBuf::from(format!("{base}.image.pbcor"));
553                let preview = config
554                    .write_preview_pngs
555                    .then(|| PathBuf::from(format!("{base}.image.pbcor.png")));
556                artifacts.push(artifact(
557                    "PB-corrected Image".to_string(),
558                    "image.pbcor",
559                    path,
560                    preview,
561                ));
562            }
563        }
564    }
565    artifacts
566}

568fn label_for_term(base: &str, term: usize) -> String {
569    format!("{base} TT{term}")
570}

572fn artifact(
573    label: String,
574    kind: &str,
575    path: PathBuf,
576    preview: Option<PathBuf>,
577) -> ManagedImagingArtifact {
578    ManagedImagingArtifact {
579        kind: kind.to_string(),
580        label,
581        exists: path.exists(),
582        path: path.display().to_string(),
583        preview_png_exists: preview.as_ref().is_some_and(|path| path.exists()),
584        preview_png_path: preview.map(|path| path.display().to_string()),
585    }
586}

720    fn from_run_reports_mtmfs_artifacts_and_preview_sidecars() {
721        let tempdir = tempdir().unwrap();
722        let imagename = tempdir.path().join("managed-output");
723        fs::write(imagename.with_extension("psf.tt0"), b"psf").unwrap();
724        fs::write(imagename.with_extension("psf.tt0.png"), b"png").unwrap();
725        fs::write(imagename.with_extension("alpha"), b"alpha").unwrap();
726        fs::write(imagename.with_extension("alpha.png"), b"png").unwrap();
727
728        let output =
729            ManagedImagingOutput::from_run(&sample_cli_config(imagename), &sample_run_summary());
730
731        assert_eq!(output.request.spectral_mode, "mfs");
732        assert_eq!(output.request.weighting, "natural");
733        assert_eq!(output.request.deconvolver, "mtmfs");
734        assert_eq!(output.request.w_term_mode, "direct");
735        assert_eq!(output.request.restoring_beam_mode, "common");
736        assert_eq!(output.request.output_channels, 1);
737        assert_eq!(output.request.correlation.as_deref(), Some("XX"));
738
739        assert_eq!(
740            output.run.clean_stop_reason.as_deref(),
741            Some("CycleThresholdReached")
742        );
743        assert_eq!(
744            output.run.stage_timings.values_ns[0],
745            ("controller_total".to_string(), 21)
746        );
747        let stage_timings = output
748            .run
749            .stage_timings
750            .values_ns
751            .iter()
752            .cloned()
753            .collect::<std::collections::BTreeMap<_, _>>();
754        assert_eq!(stage_timings["psf_normalize"], 15);
755        assert_eq!(stage_timings["model_fft"], 16);
756        assert_eq!(stage_timings["minor_cycle"], 18);
757        assert_eq!(stage_timings["minor_cycle_solve"], 19);
758        assert_eq!(stage_timings["major_cycle_refresh"], 20);
759        assert_eq!(stage_timings["beam_fit"], 22);
760        assert_eq!(stage_timings["restore"], 23);
761        assert_eq!(
762            output.run.frontend_timings.values_ns[0],
763            ("open_measurement_set".to_string(), 31)
764        );
765        assert_eq!(output.run.channels.len(), 1);
766        assert_eq!(output.run.channels[0].channel_index, 7);
767        assert_eq!(
768            output.run.channels[0].clean_stop_reason.as_deref(),
769            Some("IterationLimitReached")
770        );
771        assert!(!output.run.channels[0].beam_fit_available);
772
773        assert_eq!(output.artifacts.len(), 9);
774        assert_eq!(output.artifacts[0].label, "PSF TT0");
775        assert_eq!(output.artifacts[0].kind, "psf");
776        assert!(output.artifacts[0].exists);
777        assert!(output.artifacts[0].preview_png_exists);
778        assert!(
779            output.artifacts[0]
780                .preview_png_path
781                .as_deref()
782                .unwrap()
783                .ends_with(".psf.tt0.png")
784        );
785        assert_eq!(output.artifacts[4].label, "PSF TT1");
786        assert_eq!(output.artifacts[4].preview_png_path, None);
787        assert_eq!(output.artifacts[8].label, "Spectral Index");
788        assert_eq!(output.artifacts[8].kind, "alpha");
789        assert!(output.artifacts[8].exists);
790        assert!(output.artifacts[8].preview_png_exists);
791    }

961    #[test]
962    fn artifact_helpers_cover_standard_products_and_preview_flags() {
963        let tempdir = tempdir().unwrap();
964        let imagename = tempdir.path().join("standard-output");
965        fs::write(imagename.with_extension("image"), b"image").unwrap();
966        fs::write(imagename.with_extension("image.png"), b"png").unwrap();
967
968        let mut config = sample_cli_config(imagename.clone());
969        config.deconvolver = Deconvolver::Clark;
970        config.nterms = 1;
971        config.spectral_mode = SpectralMode::Cube;
972        config.write_preview_pngs = true;
973
974        let artifacts = imaging_artifacts(&config);
975        assert_eq!(artifacts.len(), 4);
976        assert_eq!(
977            artifacts
978                .iter()
979                .map(|artifact| artifact.kind.as_str())
980                .collect::<Vec<_>>(),
981            vec!["psf", "residual", "model", "image"]
982        );
983        assert_eq!(artifacts[0].label, "PSF");
984        assert!(!artifacts[0].exists);
985        assert_eq!(
986            artifacts[0].preview_png_path.as_deref(),
987            Some(
988                imagename
989                    .with_extension("psf.png")
990                    .to_string_lossy()
991                    .as_ref()
992            )
993        );
994        assert_eq!(artifacts[3].label, "Restored Image");
995        assert!(artifacts[3].exists);
996        assert!(artifacts[3].preview_png_exists);
997
998        let manual = artifact(
999            label_for_term("Residual", 2),
1000            "residual",
1001            imagename.with_extension("residual"),
1002            None,
1003        );
1004        assert_eq!(manual.label, "Residual TT2");
1005        assert_eq!(manual.kind, "residual");
1006        assert!(!manual.exists);
1007        assert_eq!(manual.preview_png_path, None);
1008        assert!(!manual.preview_png_exists);
1009    }
