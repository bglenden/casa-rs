7/// Shared CASA dataset tier used by test-data discovery and gate preflights.
8#[derive(Debug, Clone, Copy, PartialEq, Eq)]
9pub enum CasaTestDataTier {
10    /// Small shared fixtures that may be used by default local gates.
11    DefaultFixture,
12    /// Tutorial parity data that must be selected explicitly by longer gates.
13    TutorialParity,
14    /// Slow parity or performance data that must be selected explicitly.
15    SlowParity,
16}
17
18impl CasaTestDataTier {
19    /// Stable lower-case name used in diagnostics and scripts.
20    pub fn as_str(self) -> &'static str {
21        match self {
22            Self::DefaultFixture => "default-fixture",
23            Self::TutorialParity => "tutorial-parity",
24            Self::SlowParity => "slow-parity",
25        }
26    }
27}

46    /// Dataset gate tier.
47    pub tier: CasaTestDataTier,
48    /// Path relative to `CASA_RS_TUTORIAL_DATA_ROOT`.

50}

276/// Resolve the shared `casatestdata` checkout used by CASA parity tests.
277///
278/// The default-fixture lookup order is:
279/// 1. `CASA_RS_TESTDATA_ROOT`
280/// 2. `../casatestdata` relative to the repo root
281/// 3. `~/SoftwareProjects/casatestdata`
282pub fn casatestdata_root() -> Option<PathBuf> {
283    casatestdata_root_for_tier(CasaTestDataTier::DefaultFixture)
284}

286/// Resolve the shared `casatestdata` checkout for a gate tier.
287pub fn casatestdata_root_for_tier(tier: CasaTestDataTier) -> Option<PathBuf> {
288    if let Some(root) = std::env::var_os("CASA_RS_TESTDATA_ROOT") {
289        let path = PathBuf::from(root);
290        if path.exists() {
291            return Some(normalize_existing_path(&path));
292        }
293    }
294
295    select_casatestdata_root(
296        casatestdata_root_candidates_for_tier(tier),
297        &[] as &[PathBuf],
298    )
299}

301/// Resolve the shared `casatestdata` checkout for a gate tier, preferring roots
302/// that contain all paths required by that gate.
303pub fn casatestdata_root_for_tier_with_required_paths(
304    tier: CasaTestDataTier,
305    required_paths: &[impl AsRef<Path>],
306) -> Option<PathBuf> {
307    if let Some(root) = std::env::var_os("CASA_RS_TESTDATA_ROOT") {
308        let path = PathBuf::from(root);
309        if path.exists() {
310            return Some(normalize_existing_path(&path));
311        }
312    }
313
314    select_casatestdata_root(casatestdata_root_candidates_for_tier(tier), required_paths)
315}

317/// Resolve a path relative to the shared `casatestdata` checkout.
318pub fn casatestdata_path(relative: impl AsRef<Path>) -> Option<PathBuf> {
319    casatestdata_root().map(|root| root.join(relative.as_ref()))
320}
321
322/// Resolve a path relative to the shared `casatestdata` checkout for a gate tier.
323pub fn casatestdata_path_for_tier(
324    tier: CasaTestDataTier,
325    relative: impl AsRef<Path>,
326) -> Option<PathBuf> {
327    let relative = relative.as_ref();
328    if let Some(root) = std::env::var_os("CASA_RS_TESTDATA_ROOT") {
329        let path = PathBuf::from(root);
330        if path.exists() {
331            return Some(normalize_existing_path(&path).join(relative));
332        }
333    }
334
335    let mut first_existing = None;
336    for candidate in casatestdata_root_candidates_for_tier(tier) {
337        if !candidate.exists() {
338            continue;
339        }
340        let root = normalize_existing_path(&candidate);
341        let path = root.join(relative);
342        if path.exists() {
343            return Some(path);
344        }
345        first_existing.get_or_insert(path);
346    }
347    first_existing
348}

350/// Resolve the local CASA tutorial data mirror used by tutorial-registry keys.
351///
352/// This stays separate from the external C++ `casatestdata` checkout.
353pub fn casa_tutorial_data_root() -> Option<PathBuf> {
354    if let Some(root) = std::env::var_os("CASA_RS_TUTORIAL_DATA_ROOT") {
355        let path = PathBuf::from(root);
356        if path.exists() {
357            return Some(normalize_existing_path(&path));
358        }
359    }
360
361    let home = std::env::var_os("HOME").map(PathBuf::from)?;
362    let candidate = home.join("SoftwareProjects/casa-tutorial-data");
363    if candidate.exists() {
364        Some(normalize_existing_path(&candidate))
365    } else {
366        None
367    }
368}

370/// Resolve a tutorial registry entry by key.
371pub fn tutorial_dataset(key: &str) -> Option<&'static TutorialDataset> {
372    TUTORIAL_DATASETS.iter().find(|dataset| dataset.key == key)
373}
374
375/// Iterate tutorial registry entries that belong to a gate tier.
376pub fn tutorial_datasets_for_tier(
377    tier: CasaTestDataTier,
378) -> impl Iterator<Item = &'static TutorialDataset> {
379    TUTORIAL_DATASETS
380        .iter()
381        .filter(move |dataset| dataset.tier == tier)
382}
383
384/// Resolve a tutorial registry entry to its expected local path.
385pub fn tutorial_dataset_path(key: &str) -> Option<PathBuf> {
386    let dataset = tutorial_dataset(key)?;
387    casa_tutorial_data_root().map(|root| root.join(dataset.relative_path))
388}

430/// Return `git rev-parse HEAD` for a local source tree when available.
431pub fn git_head_commit(path: &Path) -> Option<String> {
432    Command::new("git")
433        .arg("-C")
434        .arg(path)
435        .args(["rev-parse", "HEAD"])
436        .output()
437        .ok()
438        .filter(|output| output.status.success())
439        .and_then(|output| String::from_utf8(output.stdout).ok())
440        .map(|stdout| stdout.trim().to_string())
441        .filter(|value| !value.is_empty())
442}

471pub(crate) fn select_casatestdata_root(
472    candidates: Vec<PathBuf>,
473    required_paths: &[impl AsRef<Path>],
474) -> Option<PathBuf> {
475    let mut first_existing = None;
476    for candidate in candidates {
477        if !candidate.exists() {
478            continue;
479        }
480        let root = normalize_existing_path(&candidate);
481        first_existing.get_or_insert_with(|| root.clone());
482        if required_paths
483            .iter()
484            .all(|relative| root.join(relative.as_ref()).exists())
485        {
486            return Some(root);
487        }
488    }
489    first_existing
490}

492pub(crate) fn normalize_existing_path(path: &Path) -> PathBuf {
493    std::fs::canonicalize(path).unwrap_or_else(|_| path.to_path_buf())
494}

518pub(crate) fn probe_casa_python(program: PathBuf) -> Option<CasaPython> {
519    if !python_can_import(&program, "casatasks") {
520        return None;
521    }
522    Some(CasaPython {
523        plotms_available: python_has_callable(&program, "plotms"),
524        tclean_available: python_has_callable(&program, "tclean"),
525        program,
526    })
527}

529pub(crate) fn python_can_import(program: &Path, module: &str) -> bool {
530    Command::new(program)
531        .arg("-c")
532        .arg(format!("import {module}"))
533        .output()
534        .is_ok_and(|output| output.status.success())
535}
536
537pub(crate) fn python_has_callable(program: &Path, attribute: &str) -> bool {
538    let script = if attribute == "plotms" {
539        "import importlib.util\nimport casatasks\nok = hasattr(casatasks, 'plotms') or importlib.util.find_spec('casaplotms') is not None\nprint('1' if ok else '0')\n".to_string()
540    } else {
541        format!(
542            "import casatasks\nok = hasattr(casatasks, {attribute:?})\nprint('1' if ok else '0')\n"
543        )
544    };
545    Command::new(program)
546        .arg("-c")
547        .arg(script)
548        .output()
549        .ok()
550        .filter(|output| output.status.success())
551        .is_some_and(|output| String::from_utf8_lossy(&output.stdout).trim() == "1")
552}

554pub(crate) fn existing_path_from_candidates<const N: usize>(
555    candidates: [Option<PathBuf>; N],
556) -> Option<PathBuf> {
557    candidates
558        .into_iter()
559        .flatten()
560        .find(|path| path.exists())
561        .map(|path| normalize_existing_path(&path))
562}

564pub(crate) fn dedup_paths(paths: Vec<PathBuf>) -> Vec<PathBuf> {
565    let mut unique = Vec::new();
566    for path in paths {
567        if !unique.iter().any(|candidate| candidate == &path) {
568            unique.push(path);
569        }
570    }
571    unique
572}
