2//! Limited `fluxscale` support for the first bootstrap wave.
3//!
4//! This module intentionally implements a narrow but useful slice:
5//!
6//! - complex antenna-based gain tables (`CPARAM`)
7//! - reference and transfer field resolution by id, exact name, or simple `*` glob
8//! - optional `refspwmap`
9//! - optional `gainthreshold`
10//! - full-table output or incremental correction-factor output
11//!
12//! The acceptance contract is twofold:
13//!
14//! - the output caltable must remain CASA-compatible on disk
15//! - the derived transfer fluxes and scaled gains must agree closely with CASA
16//!   on slow parity fixtures
17

32/// Request for a first-wave `fluxscale` solve.
33#[derive(Debug, Clone, PartialEq, Serialize, Deserialize, JsonSchema)]
34pub struct FluxScaleRequest {
35    /// Input gain table produced by `gaincal`.
36    pub input_table: PathBuf,
37    /// Output flux-scaled gain table.
38    pub output_table: PathBuf,
39    /// Reference field selectors, by id, exact name, or simple `*` glob.
40    pub reference_fields: Vec<String>,
41    /// Transfer field selectors. Empty means all solved non-reference fields.
42    pub transfer_fields: Vec<String>,
43    /// Optional spectral-window remapping from transfer SPW id to reference SPW id.
44    pub refspwmap: Vec<i32>,
45    /// Optional fractional threshold around the median gain amplitude.
46    pub gainthreshold: Option<f64>,
47    /// Emit only multiplicative correction factors instead of a fully scaled table.
48    pub incremental: bool,
49}

51/// Per-SPW flux result for one transfer field.
52#[derive(Debug, Clone, PartialEq, Serialize, Deserialize, JsonSchema)]
53pub struct FluxScaleSpwResult {

55    pub fluxd: [f64; 4],
56    /// Flux-density errors in CASA-style Stokes order.
57    pub fluxd_err: [f64; 4],
58    /// Number of scalar gain samples contributing to the estimate.
59    pub num_sol: [f64; 4],

62/// Per-field `fluxscale` result.
63#[derive(Debug, Clone, PartialEq, Serialize, Deserialize, JsonSchema)]
64pub struct FluxScaleFieldResult {
65    /// Transfer-field name from the FIELD subtable.
66    pub field_name: String,
67    /// Per-SPW transfer flux results keyed by SPW id.
68    pub spw_results: BTreeMap<i32, FluxScaleSpwResult>,
69    /// Fitted reference frequency in Hz, or `0.0` when no single-SPW fit is available.
70    pub fit_ref_frequency_hz: f64,
71    /// Fitted Stokes-I flux density, or `0.0` when spectral fitting is deferred.
72    pub fit_fluxd: f64,
73    /// Fitted Stokes-I flux-density uncertainty.
74    pub fit_fluxd_err: f64,
75    /// Deferred spectral-index coefficients.
76    pub spidx: Vec<f64>,
77    /// Deferred spectral-index uncertainties.
78    pub spidx_err: Vec<f64>,
79    /// Deferred covariance matrix.
80    pub covar_mat: Vec<Vec<f64>>,
81}

98/// Errors returned by the limited `fluxscale` implementation.
99#[derive(Debug, Error)]
100pub enum FluxScaleError {
101    /// Opening or validating the input caltable failed.
102    #[error("failed to summarize input calibration table: {source}")]
103    SummarizeInput {
104        /// Underlying summary error.
105        #[source]
106        source: CalibrationTableError,
107    },
108
109    /// The input table is outside the first supported `fluxscale` surface.
110    #[error("unsupported input calibration table {path}: {reason}")]
111    UnsupportedInput {
112        /// Input path.
113        path: String,
114        /// Error context.
115        reason: String,
116    },
117
118    /// The request did not specify any reference fields.
119    #[error("fluxscale requires at least one reference field selector")]
120    MissingReferenceFields,
121
122    /// A field selector did not match any field rows.
123    #[error("field selector {selector:?} did not match any FIELD rows")]
124    UnresolvedFieldSelector {
125        /// Original selector.
126        selector: String,
127    },
128
129    /// The transfer field selection produced no solved non-reference fields.
130    #[error("fluxscale transfer field selection produced no solved non-reference fields")]
131    EmptyTransferSelection,
132
133    /// No usable gain samples were found for a required `(field, spw)` pair.
134    #[error("no usable gain samples for field={field_id} spw={spw_id}")]
135    MissingGainSamples {
136        /// Field id.
137        field_id: i32,
138        /// SPW id.
139        spw_id: i32,
140    },
141
142    /// The input gain amplitudes did not permit a stable scale factor.
143    #[error("unstable fluxscale factor for field={field_id} spw={spw_id}")]
144    UnstableScaleFactor {
145        /// Field id.
146        field_id: i32,
147        /// SPW id.
148        spw_id: i32,
149    },
150
151    /// An output path operation failed.
152    #[error("failed to prepare output path {path}: {reason}")]
153    PrepareOutput {
154        /// Output path.
155        path: String,
156        /// Error context.
157        reason: String,
158    },
159
160    /// Copying the caltable tree failed.
161    #[error("failed to copy calibration table {source_path} -> {destination_path}: {reason}")]
162    CopyTableTree {
163        /// Source path.
164        source_path: String,
165        /// Destination path.
166        destination_path: String,
167        /// Error context.
168        reason: String,
169    },
170
171    /// Opening or mutating the copied output table failed.
172    #[error("failed to mutate output calibration table {path}: {source}")]
173    MutateOutput {
174        /// Output path.
175        path: String,
176        /// Underlying table error.
177        #[source]
178        source: Box<TableError>,
179    },
180}

210/// Execute a first-wave `fluxscale` request.
211pub fn fluxscale(request: &FluxScaleRequest) -> Result<FluxScaleReport, FluxScaleError> {
212    validate_request(request)?;
213    let summary = summarize_table(&request.input_table)
214        .map_err(|source| FluxScaleError::SummarizeInput { source })?;
215    validate_input_table(&summary)?;
216
217    let field_info = load_field_info(&request.input_table)?;
218    let spw_info = load_spw_info(&request.input_table)?;
219    let rows = load_gain_rows(&request.input_table)?;
220
221    let reference_field_ids = resolve_field_selectors(&request.reference_fields, &field_info)?;
222    let transfer_field_ids = resolve_transfer_fields(
223        &request.transfer_fields,
224        &reference_field_ids,
225        &field_info,
226        &rows,
227    )?;
228
229    let estimates = derive_estimates(
230        &rows,
231        &reference_field_ids,
232        &transfer_field_ids,
233        request.gainthreshold,
234        &request.refspwmap,
235    )?;
236
237    write_scaled_output(
238        &request.input_table,
239        &request.output_table,
240        &rows,
241        &estimates,
242        request.incremental,
243    )?;
244    Ok(build_report(
245        &request.output_table,
246        &transfer_field_ids,
247        &field_info,
248        &spw_info,
249        &estimates,
250    ))
251}

253fn validate_request(request: &FluxScaleRequest) -> Result<(), FluxScaleError> {
254    if request.reference_fields.is_empty() {
255        return Err(FluxScaleError::MissingReferenceFields);
256    }
257    Ok(())
258}

260fn validate_input_table(summary: &crate::CalibrationTableSummary) -> Result<(), FluxScaleError> {
261    if summary.table_type != crate::constants::TABLE_INFO_TYPE {
262        return Err(FluxScaleError::UnsupportedInput {
263            path: summary.path.display().to_string(),
264            reason: format!("table.info type is {:?}", summary.table_type),
265        });
266    }
267    if summary.parameter_family != crate::CalibrationParameterFamily::Complex {
268        return Err(FluxScaleError::UnsupportedInput {
269            path: summary.path.display().to_string(),
270            reason: "fluxscale currently supports only complex CPARAM tables".to_string(),
271        });
272    }
273    match summary.keywords.vis_cal.as_deref() {
274        Some("G Jones") | Some("T Jones") => Ok(()),
275        other => Err(FluxScaleError::UnsupportedInput {
276            path: summary.path.display().to_string(),
277            reason: format!("expected VisCal to be G Jones or T Jones, found {other:?}"),
278        }),
279    }
280}

282fn load_field_info(root: &Path) -> Result<HashMap<i32, FieldInfo>, FluxScaleError> {
283    let table = Table::open(TableOptions::new(root.join("FIELD"))).map_err(|source| {
284        FluxScaleError::MutateOutput {
285            path: root.join("FIELD").display().to_string(),
286            source: Box::new(source),
287        }
288    })?;
289    let mut out = HashMap::new();
290    for row in 0..table.row_count() {
291        let name = get_string(&table, row, "NAME")?;
292        out.insert(row as i32, FieldInfo { name });
293    }
294    Ok(out)
295}

297fn load_spw_info(root: &Path) -> Result<HashMap<i32, SpwInfo>, FluxScaleError> {
298    let table = Table::open(TableOptions::new(root.join("SPECTRAL_WINDOW"))).map_err(|source| {
299        FluxScaleError::MutateOutput {
300            path: root.join("SPECTRAL_WINDOW").display().to_string(),
301            source: Box::new(source),
302        }
303    })?;
304    let mut out = HashMap::new();
305    for row in 0..table.row_count() {
306        let ref_frequency_hz = get_ref_frequency(&table, row)?;
307        let name = get_string(&table, row, "NAME").unwrap_or_default();
308        out.insert(
309            row as i32,
310            SpwInfo {
311                name,
312                ref_frequency_hz,
313            },
314        );
315    }
316    Ok(out)
317}

319fn load_gain_rows(root: &Path) -> Result<Vec<GainSampleRow>, FluxScaleError> {
320    let table =
321        Table::open(TableOptions::new(root)).map_err(|source| FluxScaleError::MutateOutput {
322            path: root.display().to_string(),
323            source: Box::new(source),
324        })?;
325    let mut rows = Vec::new();
326    for row_index in 0..table.row_count() {
327        let field_id = get_i32(&table, row_index, COL_FIELD_ID)?;
328        let spw_id = get_i32(&table, row_index, COL_SPECTRAL_WINDOW_ID)?;
329        let gains = get_complex_array(&table, row_index, COL_CPARAM)?;
330        let flags = get_bool_array(&table, row_index, COL_FLAG)?;
331        rows.push(GainSampleRow {
332            row_index,
333            field_id,
334            spw_id,
335            gains,
336            flags,
337        });
338    }
339    Ok(rows)
340}

342fn resolve_field_selectors(
343    selectors: &[String],
344    field_info: &HashMap<i32, FieldInfo>,
345) -> Result<BTreeSet<i32>, FluxScaleError> {
346    let mut resolved = BTreeSet::new();
347    for selector in selectors {
348        let selector = selector.trim();
349        if selector.is_empty() {
350            continue;
351        }
352        for part in selector
353            .split(',')
354            .map(str::trim)
355            .filter(|part| !part.is_empty())
356        {
357            if let Ok(field_id) = part.parse::<i32>() {
358                if field_info.contains_key(&field_id) {
359                    resolved.insert(field_id);
360                    continue;
361                }
362                return Err(FluxScaleError::UnresolvedFieldSelector {
363                    selector: part.to_string(),
364                });
365            }
366            let mut matched = false;
367            for (&field_id, info) in field_info {
368                if matches_simple_glob(part, &info.name) {
369                    matched = true;
370                    resolved.insert(field_id);
371                }
372            }
373            if !matched {
374                return Err(FluxScaleError::UnresolvedFieldSelector {
375                    selector: part.to_string(),
376                });
377            }
378        }
379    }
380    Ok(resolved)
381}

383fn resolve_transfer_fields(
384    selectors: &[String],
385    reference_field_ids: &BTreeSet<i32>,
386    field_info: &HashMap<i32, FieldInfo>,
387    rows: &[GainSampleRow],
388) -> Result<BTreeSet<i32>, FluxScaleError> {
389    let solved_fields = rows.iter().map(|row| row.field_id).collect::<BTreeSet<_>>();
390    let mut transfer = if selectors.is_empty() {
391        solved_fields
392            .difference(reference_field_ids)
393            .copied()
394            .collect::<BTreeSet<_>>()
395    } else {
396        resolve_field_selectors(selectors, field_info)?
397    };
398    transfer.retain(|field_id| {
399        !reference_field_ids.contains(field_id) && solved_fields.contains(field_id)
400    });
401    if transfer.is_empty() {
402        return Err(FluxScaleError::EmptyTransferSelection);
403    }
404    Ok(transfer)
405}

407fn derive_estimates(
408    rows: &[GainSampleRow],
409    reference_field_ids: &BTreeSet<i32>,
410    transfer_field_ids: &BTreeSet<i32>,
411    gainthreshold: Option<f64>,
412    refspwmap: &[i32],
413) -> Result<HashMap<(i32, i32), FluxScaleEstimate>, FluxScaleError> {
414    let mut amplitudes = HashMap::<(i32, i32), Vec<f64>>::new();
415
416    for row in rows {
417        let key = (row.field_id, row.spw_id);
418        let mut values = row_amplitudes(row);
419        if values.is_empty() {
420            continue;
421        }
422        amplitudes.entry(key).or_default().append(&mut values);
423    }
424
425    let mut filtered = HashMap::<(i32, i32), Vec<f64>>::new();
426    for (key, values) in amplitudes {
427        filtered.insert(key, apply_gainthreshold(values, gainthreshold));
428    }
429
430    let mut reference_means = HashMap::<i32, f64>::new();
431    for field_id in reference_field_ids {
432        for (&(candidate_field_id, spw_id), values) in &filtered {
433            if candidate_field_id != *field_id || values.is_empty() {
434                continue;
435            }
436            reference_means.entry(spw_id).or_default();
437        }
438    }
439    for (spw_id, mean) in reference_means.iter_mut() {
440        let pooled = reference_field_ids
441            .iter()
442            .filter_map(|field_id| filtered.get(&(*field_id, *spw_id)))
443            .flat_map(|values| values.iter().copied())
444            .collect::<Vec<_>>();
445        if pooled.is_empty() {
446            return Err(FluxScaleError::MissingGainSamples {
447                field_id: *reference_field_ids.iter().next().unwrap_or(&-1),
448                spw_id: *spw_id,
449            });
450        }
451        *mean = pooled.iter().sum::<f64>() / pooled.len() as f64;
452    }
453
454    let mut estimates = HashMap::new();
455    for field_id in transfer_field_ids {
456        let field_spws = filtered
457            .keys()
458            .filter_map(|(candidate_field_id, spw_id)| {
459                (*candidate_field_id == *field_id).then_some(*spw_id)
460            })
461            .collect::<BTreeSet<_>>();
462        for spw_id in field_spws {
463            let values =
464                filtered
465                    .get(&(*field_id, spw_id))
466                    .ok_or(FluxScaleError::MissingGainSamples {
467                        field_id: *field_id,
468                        spw_id,
469                    })?;
470            if values.is_empty() {
471                return Err(FluxScaleError::MissingGainSamples {
472                    field_id: *field_id,
473                    spw_id,
474                });
475            }
476            let reference_spw_id = mapped_reference_spw(spw_id, refspwmap);
477            let reference_mean = *reference_means.get(&reference_spw_id).ok_or(
478                FluxScaleError::MissingGainSamples {
479                    field_id: *reference_field_ids.iter().next().unwrap_or(&-1),
480                    spw_id: reference_spw_id,
481                },
482            )?;
483            let transfer_mean = values.iter().sum::<f64>() / values.len() as f64;
484            if !(reference_mean.is_finite()
485                && transfer_mean.is_finite()
486                && reference_mean > 0.0
487                && transfer_mean > 0.0)
488            {
489                return Err(FluxScaleError::UnstableScaleFactor {
490                    field_id: *field_id,
491                    spw_id,
492                });
493            }
494
495            let ratio = transfer_mean / reference_mean;
496            let flux_density_jy = ratio * ratio;
497            let correction_factor = (1.0 / ratio) as f32;
498            let flux_density_error_jy = estimate_flux_error(values, reference_mean);
499            estimates.insert(
500                (*field_id, spw_id),
501                FluxScaleEstimate {
502                    flux_density_jy,
503                    flux_density_error_jy,
504                    correction_factor,
505                    num_sol: values.len(),
506                },
507            );
508        }
509    }
510
511    Ok(estimates)
512}

514fn row_amplitudes(row: &GainSampleRow) -> Vec<f64> {
515    if row.gains.ndim() != row.flags.ndim() {
516        return Vec::new();
517    }
518    row.gains
519        .iter()
520        .zip(row.flags.iter())
521        .filter_map(|(gain, flagged)| {
522            (!*flagged && gain.norm() > f32::EPSILON).then_some(gain.norm() as f64)
523        })
524        .collect()
525}
526
527fn apply_gainthreshold(mut values: Vec<f64>, gainthreshold: Option<f64>) -> Vec<f64> {
528    let Some(gainthreshold) = gainthreshold else {
529        return values;
530    };
531    if gainthreshold < 0.0 || values.is_empty() {
532        return values;
533    }
534    values.sort_by(f64::total_cmp);
535    let median = median(&values);
536    values
537        .into_iter()
538        .filter(|value| (value - median).abs() <= gainthreshold * median)
539        .collect()
540}
541
542fn median(values: &[f64]) -> f64 {
543    if values.is_empty() {
544        return 0.0;
545    }
546    let mid = values.len() / 2;
547    if values.len() % 2 == 0 {
548        (values[mid - 1] + values[mid]) / 2.0
549    } else {
550        values[mid]
551    }
552}

554fn mapped_reference_spw(spw_id: i32, refspwmap: &[i32]) -> i32 {
555    let Ok(index) = usize::try_from(spw_id) else {
556        return spw_id;
557    };
558    match refspwmap.get(index).copied() {
559        Some(mapped) if mapped >= 0 => mapped,
560        _ => spw_id,
561    }
562}
563
564fn estimate_flux_error(transfer_values: &[f64], reference_mean: f64) -> f64 {
565    if transfer_values.len() <= 1 || reference_mean <= 0.0 {
566        return 0.0;
567    }
568    let factors = transfer_values
569        .iter()
570        .map(|value| (value / reference_mean).powi(2))
571        .collect::<Vec<_>>();
572    let mean = factors.iter().sum::<f64>() / factors.len() as f64;
573    let variance = factors
574        .iter()
575        .map(|value| {
576            let delta = *value - mean;
577            delta * delta
578        })
579        .sum::<f64>()
580        / (factors.len() as f64);
581    variance.sqrt() / (factors.len() as f64).sqrt()
582}

584fn write_scaled_output(
585    input_root: &Path,
586    output_root: &Path,
587    rows: &[GainSampleRow],
588    estimates: &HashMap<(i32, i32), FluxScaleEstimate>,
589    incremental: bool,
590) -> Result<(), FluxScaleError> {
591    prepare_output_root(input_root, output_root)?;
592    copy_table_tree(input_root, output_root)?;
593    let mut table = Table::open(TableOptions::new(output_root)).map_err(|source| {
594        FluxScaleError::MutateOutput {
595            path: output_root.display().to_string(),
596            source: Box::new(source),
597        }
598    })?;
599    for row in rows {
600        let Some(estimate) = estimates.get(&(row.field_id, row.spw_id)) else {
601            continue;
602        };
603        let scaled = scale_gains(&row.gains, estimate.correction_factor, incremental);
604        table
605            .set_cell(
606                row.row_index,
607                COL_CPARAM,
608                Value::Array(ArrayValue::Complex32(scaled)),
609            )
610            .map_err(|source| FluxScaleError::MutateOutput {
611                path: output_root.display().to_string(),
612                source: Box::new(source),
613            })?;
614    }
615    table
616        .save(TableOptions::new(output_root))
617        .map_err(|source| FluxScaleError::MutateOutput {
618            path: output_root.display().to_string(),
619            source: Box::new(source),
620        })?;
621    Ok(())
622}
623
624fn scale_gains(
625    values: &ArrayD<Complex32>,
626    correction_factor: f32,
627    incremental: bool,
628) -> ArrayD<Complex32> {
629    if incremental {
630        ArrayD::from_elem(values.raw_dim(), Complex32::new(correction_factor, 0.0))
631    } else {
632        values.mapv(|value| value * correction_factor)
633    }
634}

636fn build_report(
637    output_table: &Path,
638    transfer_field_ids: &BTreeSet<i32>,
639    field_info: &HashMap<i32, FieldInfo>,
640    spw_info: &HashMap<i32, SpwInfo>,
641    estimates: &HashMap<(i32, i32), FluxScaleEstimate>,
642) -> FluxScaleReport {
643    let mut fields = BTreeMap::new();
644    let mut spw_ids = BTreeSet::new();
645
646    for field_id in transfer_field_ids {
647        let mut spw_results = BTreeMap::new();
648        let mut single_spw = None;
649        for (&(candidate_field_id, spw_id), estimate) in estimates {
650            if candidate_field_id != *field_id {
651                continue;
652            }
653            spw_ids.insert(spw_id);
654            spw_results.insert(
655                spw_id,
656                FluxScaleSpwResult {
657                    fluxd: [estimate.flux_density_jy, 0.0, 0.0, 0.0],
658                    fluxd_err: [estimate.flux_density_error_jy, 0.0, 0.0, 0.0],
659                    num_sol: [estimate.num_sol as f64, 0.0, 0.0, 0.0],
660                },
661            );
662            single_spw = Some((spw_id, estimate));
663        }
664
665        let (fit_ref_frequency_hz, fit_fluxd, fit_fluxd_err) = if spw_results.len() == 1 {
666            let (spw_id, estimate) = single_spw.expect("single spw estimate");
667            (
668                spw_info
669                    .get(&spw_id)
670                    .map(|info| info.ref_frequency_hz)
671                    .unwrap_or(0.0),
672                estimate.flux_density_jy,
673                0.0,
674            )
675        } else {
676            (0.0, 0.0, 0.0)
677        };
678
679        fields.insert(
680            *field_id,
681            FluxScaleFieldResult {
682                field_name: field_info
683                    .get(field_id)
684                    .map(|info| info.name.clone())
685                    .unwrap_or_default(),
686                spw_results,
687                fit_ref_frequency_hz,
688                fit_fluxd,
689                fit_fluxd_err,
690                spidx: vec![0.0, 0.0],
691                spidx_err: vec![0.0, 0.0],
692                covar_mat: Vec::new(),
693            },
694        );
695    }
696
697    let sorted_spw_ids = spw_ids.into_iter().collect::<Vec<_>>();
698    let spw_names = sorted_spw_ids
699        .iter()
700        .map(|spw_id| {
701            spw_info
702                .get(spw_id)
703                .map(|info| info.name.clone())
704                .unwrap_or_default()
705        })
706        .collect();
707    let frequencies_hz = sorted_spw_ids
708        .iter()
709        .map(|spw_id| {
710            spw_info
711                .get(spw_id)
712                .map(|info| info.ref_frequency_hz)
713                .unwrap_or(0.0)
714        })
715        .collect();
716
717    FluxScaleReport {
718        output_table: output_table.to_path_buf(),
719        spw_ids: sorted_spw_ids,
720        spw_names,
721        frequencies_hz,
722        fields,
723    }
724}

726fn prepare_output_root(input_root: &Path, output_root: &Path) -> Result<(), FluxScaleError> {
727    if input_root == output_root {
728        return Err(FluxScaleError::PrepareOutput {
729            path: output_root.display().to_string(),
730            reason: "input_table and output_table must differ".to_string(),
731        });
732    }
733    if output_root.exists() {
734        fs::remove_dir_all(output_root)
735            .or_else(|_| fs::remove_file(output_root))
736            .map_err(|error| FluxScaleError::PrepareOutput {
737                path: output_root.display().to_string(),
738                reason: error.to_string(),
739            })?;
740    }
741    if let Some(parent) = output_root.parent() {
742        fs::create_dir_all(parent).map_err(|error| FluxScaleError::PrepareOutput {
743            path: output_root.display().to_string(),
744            reason: error.to_string(),
745        })?;
746    }
747    Ok(())
748}

750fn copy_table_tree(source: &Path, destination: &Path) -> Result<(), FluxScaleError> {
751    if source.is_dir() {
752        fs::create_dir_all(destination).map_err(|error| FluxScaleError::CopyTableTree {
753            source_path: source.display().to_string(),
754            destination_path: destination.display().to_string(),
755            reason: error.to_string(),
756        })?;
757        for entry in fs::read_dir(source).map_err(|error| FluxScaleError::CopyTableTree {
758            source_path: source.display().to_string(),
759            destination_path: destination.display().to_string(),
760            reason: error.to_string(),
761        })? {
762            let entry = entry.map_err(|error| FluxScaleError::CopyTableTree {
763                source_path: source.display().to_string(),
764                destination_path: destination.display().to_string(),
765                reason: error.to_string(),
766            })?;
767            copy_table_tree(&entry.path(), &destination.join(entry.file_name()))?;
768        }
769        return Ok(());
770    }
771    fs::copy(source, destination).map_err(|error| FluxScaleError::CopyTableTree {
772        source_path: source.display().to_string(),
773        destination_path: destination.display().to_string(),
774        reason: error.to_string(),
775    })?;
776    Ok(())
777}

817fn get_ref_frequency(table: &Table, row: usize) -> Result<f64, FluxScaleError> {
818    let chan_freq =
819        table
820            .get_array_cell(row, "CHAN_FREQ")
821            .map_err(|source| FluxScaleError::MutateOutput {
822                path: "CHAN_FREQ".to_string(),
823                source: Box::new(source),
824            })?;
825    match chan_freq {
826        ArrayValue::Float64(values) if !values.is_empty() => {
827            Ok(values.iter().copied().sum::<f64>() / values.len() as f64)
828        }
829        _ => match table.cell(row, "REF_FREQUENCY").map_err(|source| {
830            FluxScaleError::MutateOutput {
831                path: "REF_FREQUENCY".to_string(),
832                source: Box::new(source),
833            }
834        })? {
835            Some(Value::Scalar(ScalarValue::Float64(value))) => Ok(*value),
836            Some(Value::Array(ArrayValue::Float64(values))) if values.len() == 1 => {
837                Ok(values.iter().copied().next().unwrap_or(0.0))
838            }
839            Some(_) | None => Ok(0.0),
840        },
841    }
842}

884fn matches_simple_glob(pattern: &str, candidate: &str) -> bool {
885    if !pattern.contains('*') {
886        return pattern == candidate;
887    }
888    let mut rest = candidate;
889    let parts = pattern.split('*').collect::<Vec<_>>();
890    let anchored_start = !pattern.starts_with('*');
891    let anchored_end = !pattern.ends_with('*');
892
893    for (index, part) in parts.iter().enumerate() {
894        if part.is_empty() {
895            continue;
896        }
897        if index == 0 && anchored_start {
898            if !rest.starts_with(part) {
899                return false;
900            }
901            rest = &rest[part.len()..];
902            continue;
903        }
904        if index == parts.len() - 1 && anchored_end {
905            return rest.ends_with(part);
906        }
907        let Some(position) = rest.find(part) else {
908            return false;
909        };
910        rest = &rest[position + part.len()..];
911    }
912
913    true
914}
