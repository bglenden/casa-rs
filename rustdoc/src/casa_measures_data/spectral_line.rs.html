2//! Spectral-line catalog data loaded from casacore `ephemerides/Lines`.

4use std::collections::HashMap;
5
6/// A single spectral-line row from casacore `ephemerides/Lines`.
7#[derive(Debug, Clone, PartialEq)]
8pub struct SpectralLineEntry {
9    /// Table `MJD` column.
10    pub mjd: f64,
11    /// Table `Name` column.
12    pub name: String,
13    /// Table `Type` column.
14    pub frequency_type: String,
15    /// Table `Freq` column, in GHz.
16    pub frequency_ghz: f64,
17    /// Table `Source` column.
18    pub source: String,
19    /// Table `Comment` column.
20    pub comment: String,
21}
22
23impl SpectralLineEntry {
24    /// Return the rest frequency in Hz.
25    pub fn frequency_hz(&self) -> f64 {
26        self.frequency_ghz * 1.0e9
27    }
28}
29
30/// In-memory catalog of named spectral-line metadata.
31#[derive(Debug, Clone, PartialEq)]
32pub struct SpectralLineCatalog {
33    entries: Vec<SpectralLineEntry>,
34    by_name: HashMap<String, usize>,
35}
36
37impl SpectralLineCatalog {
38    /// Build a catalog from explicitly provided entries.
39    pub fn from_entries(entries: Vec<SpectralLineEntry>) -> Self {
40        let by_name = entries
41            .iter()
42            .enumerate()
43            .fold(HashMap::new(), |mut map, (index, entry)| {
44                map.entry(normalize_name(&entry.name)).or_insert(index);
45                map
46            });
47        Self { entries, by_name }
48    }
49
50    /// Return the packaged/runtime standard catalog.
51    pub fn bundled() -> &'static Self {
52        crate::load_spectral_lines().0
53    }
54
55    /// Iterate over the catalog entries in source order.
56    pub fn iter(&self) -> impl Iterator<Item = &SpectralLineEntry> {
57        self.entries.iter()
58    }
59
60    /// Borrow all entries.
61    pub fn entries(&self) -> &[SpectralLineEntry] {
62        &self.entries
63    }
64
65    /// Look up a spectral line by case-insensitive name.
66    pub fn get(&self, name: &str) -> Option<&SpectralLineEntry> {
67        self.by_name
68            .get(&normalize_name(name))
69            .and_then(|index| self.entries.get(*index))
70    }
71}
72
73fn normalize_name(name: &str) -> String {
74    name.trim().to_ascii_uppercase()
75}
