2//! Quantum column support: unit-tagged table columns.
3//!
4//! This module provides the Rust counterparts of C++ `TableQuantumDesc`,
5//! `ScalarQuantColumn`, and `ArrayQuantColumn` from the casacore
6//! `measures/TableMeasures` subsystem.
7//!
8//! A "quantum column" is a regular numeric column whose values carry physical
9//! units. The unit metadata is stored in the column's keyword record as either:
10//!
11//! - **`QuantumUnits`** — a `Vector<String>` of fixed unit name(s), or
12//! - **`VariableUnits`** — the name of a companion `String` column that holds
13//!   per-row (scalar) or per-element (array) unit strings.
14//!
15//! # Examples
16//!
17//! ```rust
18//! use casacore_tables::{Table, TableSchema, ColumnSchema, table_quantum::TableQuantumDesc};
19//! use casacore_types::*;
20//!
21//! // Create a table with a quantum column
22//! let schema = TableSchema::new(vec![
23//!     ColumnSchema::scalar("flux", PrimitiveType::Float64),
24//! ]).unwrap();
25//! let mut table = Table::with_schema(schema);
26//!
27//! // Attach fixed units
28//! let desc = TableQuantumDesc::with_unit("flux", "Jy");
29//! desc.write(&mut table).unwrap();
30//! assert!(TableQuantumDesc::has_quanta(&table, "flux"));
31//! ```
32
33use crate::schema::ColumnType;
34use crate::table::{Table, TableError};
35use casacore_types::quanta::{Quantity, Unit};
36use casacore_types::{ArrayValue, ScalarValue, Value};

38// ─── Keyword names (matching C++ casacore) ──────────────────────────────────
39
40const QUANTUM_UNITS_KW: &str = "QuantumUnits";
41const VARIABLE_UNITS_KW: &str = "VariableUnits";
42
43// ─── TableQuantumDesc ───────────────────────────────────────────────────────
44
45/// Descriptor for a quantum column's unit metadata.
46///
47/// Corresponds to C++ `casa::TableQuantumDesc`.
48///
49/// A `TableQuantumDesc` records whether a column uses **fixed units**
50/// (stored in the column's keyword record) or **variable units** (stored in
51/// a companion `String` column). It can read, write, and detect quantum
52/// metadata on table columns.
53#[derive(Debug, Clone)]
54pub struct TableQuantumDesc {
55    /// Name of the data column.
56    column_name: String,
57    /// Fixed unit names (non-empty when `units_column_name` is `None`).
58    units: Vec<String>,
59    /// Name of the companion units column (variable-unit mode).
60    units_column_name: Option<String>,
61}
62
63impl TableQuantumDesc {
64    /// Creates a descriptor with a single fixed unit.
65    ///
66    /// All cells in the column share the same unit string.
67    ///
68    /// # Examples
69    ///
70    /// ```
71    /// use casacore_tables::table_quantum::TableQuantumDesc;
72    ///
73    /// let desc = TableQuantumDesc::with_unit("flux", "Jy");
74    /// assert!(!desc.is_unit_variable());
75    /// assert_eq!(desc.units(), &["Jy"]);
76    /// ```
77    pub fn with_unit(column: &str, unit: &str) -> Self {
78        Self {
79            column_name: column.to_owned(),
80            units: vec![unit.to_owned()],
81            units_column_name: None,
82        }
83    }
84
85    /// Creates a descriptor with multiple fixed units.
86    ///
87    /// Typically used for array columns where each axis has a different unit
88    /// (e.g. `["MHz", "GHz"]`).
89    pub fn with_units(column: &str, units: &[&str]) -> Self {
90        Self {
91            column_name: column.to_owned(),
92            units: units.iter().map(|u| (*u).to_owned()).collect(),
93            units_column_name: None,
94        }
95    }
96
97    /// Creates a descriptor with variable units stored in a companion column.
98    ///
99    /// The companion column must be a `String` column. If it is a scalar
100    /// column, units vary per row; if an array column, units vary per element.
101    pub fn with_variable_units(column: &str, units_column: &str) -> Self {
102        Self {
103            column_name: column.to_owned(),
104            units: Vec::new(),
105            units_column_name: Some(units_column.to_owned()),
106        }
107    }
108
109    /// Returns `true` if the column `column` in `table` has quantum keywords.
110    ///
111    /// Checks for the presence of either `QuantumUnits` or `VariableUnits`
112    /// in the column's keyword record.
113    pub fn has_quanta(table: &Table, column: &str) -> bool {
114        if let Some(kw) = table.column_keywords(column) {
115            kw.get(QUANTUM_UNITS_KW).is_some() || kw.get(VARIABLE_UNITS_KW).is_some()
116        } else {
117            false
118        }
119    }
120
121    /// Reconstructs a descriptor from a column's persisted keywords.
122    ///
123    /// Returns `None` if the column has no quantum keywords.
124    pub fn reconstruct(table: &Table, column: &str) -> Option<Self> {
125        let kw = table.column_keywords(column)?;
126
127        // Check for variable units first.
128        if let Some(Value::Scalar(ScalarValue::String(units_col))) = kw.get(VARIABLE_UNITS_KW) {
129            return Some(Self {
130                column_name: column.to_owned(),
131                units: Vec::new(),
132                units_column_name: Some(units_col.clone()),
133            });
134        }
135
136        // Check for fixed units.
137        if let Some(Value::Array(ArrayValue::String(arr))) = kw.get(QUANTUM_UNITS_KW) {
138            let units: Vec<String> = arr.iter().cloned().collect();
139            return Some(Self {
140                column_name: column.to_owned(),
141                units,
142                units_column_name: None,
143            });
144        }
145
146        None
147    }
148
149    /// Writes the quantum keywords to the column's keyword record.
150    ///
151    /// For fixed units, writes `QuantumUnits` as a `Vector<String>`.
152    /// For variable units, writes `VariableUnits` as a `String`.
153    pub fn write(&self, table: &mut Table) -> Result<(), TableError> {
154        let mut kw = table
155            .column_keywords(&self.column_name)
156            .cloned()
157            .unwrap_or_default();
158
159        if self.is_unit_variable() {
160            let col_name = self.units_column_name.as_ref().unwrap();
161            kw.upsert(
162                VARIABLE_UNITS_KW,
163                Value::Scalar(ScalarValue::String(col_name.clone())),
164            );
165            // Remove QuantumUnits if it existed.
166            kw.remove(QUANTUM_UNITS_KW);
167        } else {
168            kw.upsert(
169                QUANTUM_UNITS_KW,
170                Value::Array(ArrayValue::from_string_vec(self.units.clone())),
171            );
172            // Remove VariableUnits if it existed.
173            kw.remove(VARIABLE_UNITS_KW);
174        }
175
176        table.set_column_keywords(&self.column_name, kw);
177        Ok(())
178    }
179
180    /// Returns the data column name.
181    pub fn column_name(&self) -> &str {
182        &self.column_name
183    }
184
185    /// Returns `true` if units vary per row or per element.
186    pub fn is_unit_variable(&self) -> bool {
187        self.units_column_name.is_some()
188    }
189
190    /// Returns the companion units column name, if variable.
191    pub fn unit_column_name(&self) -> Option<&str> {
192        self.units_column_name.as_deref()
193    }
194
195    /// Returns the fixed unit name(s). Empty if variable.
196    pub fn units(&self) -> &[String] {
197        &self.units
198    }
199}

201// ─── ScalarQuantColumn (read-only) ─────────────────────────────────────────
202
203/// Read-only accessor for a scalar column with quantum (unit) metadata.
204///
205/// Corresponds to C++ `casa::ScalarQuantColumn<Double>`.
206///
207/// The unit is resolved once at construction time. Each [`get`](Self::get)
208/// call returns a [`Quantity`] (value + unit) with no additional parsing or
209/// locking overhead.
210pub struct ScalarQuantColumn<'a> {
211    table: &'a Table,
212    column_name: String,
213    desc: TableQuantumDesc,
214    /// Cached unit for fixed-unit columns; `None` for variable-unit.
215    fixed_unit: Option<Unit>,
216    /// Optional on-read conversion target.
217    convert_unit: Option<Unit>,
218}
219
220impl<'a> ScalarQuantColumn<'a> {
221    /// Attaches to a scalar quantum column.
222    ///
223    /// Returns an error if the column has no quantum keywords.
224    pub fn new(table: &'a Table, column: &str) -> Result<Self, TableError> {
225        let desc = TableQuantumDesc::reconstruct(table, column).ok_or_else(|| {
226            TableError::Storage(format!("column '{column}' has no quantum keywords"))
227        })?;
228        let fixed_unit = if !desc.is_unit_variable() && !desc.units.is_empty() {
229            Some(Unit::new(&desc.units[0]).map_err(|e| {
230                TableError::Storage(format!("invalid unit '{}': {e}", &desc.units[0]))
231            })?)
232        } else {
233            None
234        };
235        Ok(Self {
236            table,
237            column_name: column.to_owned(),
238            desc,
239            fixed_unit,
240            convert_unit: None,
241        })
242    }
243
244    /// Attaches to a scalar quantum column with on-read conversion to `unit`.
245    ///
246    /// Every [`get`](Self::get) call converts the stored value to the target
247    /// unit automatically.
248    pub fn with_unit(table: &'a Table, column: &str, unit: &str) -> Result<Self, TableError> {
249        let mut col = Self::new(table, column)?;
250        col.convert_unit = Some(
251            Unit::new(unit)
252                .map_err(|e| TableError::Storage(format!("invalid target unit '{unit}': {e}")))?,
253        );
254        Ok(col)
255    }
256
257    /// Returns `true` if units vary per row.
258    pub fn is_unit_variable(&self) -> bool {
259        self.desc.is_unit_variable()
260    }
261
262    /// Returns the fixed unit name(s). Empty if variable.
263    pub fn units(&self) -> &[String] {
264        self.desc.units()
265    }
266
267    /// Reads the quantity at `row`.
268    ///
269    /// For fixed-unit columns, the unit is the one stored in the keyword.
270    /// For variable-unit columns, the unit is read from the companion column.
271    /// If a conversion unit was set at construction, the value is converted.
272    pub fn get(&self, row: usize) -> Result<Quantity, TableError> {
273        let scalar = self.table.get_scalar_cell(row, &self.column_name)?;
274        let value = scalar_to_f64(scalar, row, &self.column_name)?;
275
276        let unit = if let Some(ref u) = self.fixed_unit {
277            u.clone()
278        } else {
279            self.read_variable_unit(row)?
280        };
281
282        let q = Quantity::with_unit(value, unit);
283
284        if let Some(ref target) = self.convert_unit {
285            q.get_value_in(target)
286                .map(|v| Quantity::with_unit(v, target.clone()))
287                .map_err(|e| TableError::Storage(format!("unit conversion failed: {e}")))
288        } else {
289            Ok(q)
290        }
291    }
292
293    fn read_variable_unit(&self, row: usize) -> Result<Unit, TableError> {
294        let units_col = self.desc.unit_column_name().unwrap();
295        let scalar = self.table.get_scalar_cell(row, units_col)?;
296        match scalar {
297            ScalarValue::String(s) => Unit::new(s)
298                .map_err(|e| TableError::Storage(format!("invalid unit '{s}' at row {row}: {e}"))),
299            _ => Err(TableError::Storage(format!(
300                "units column '{units_col}' at row {row}: expected String"
301            ))),
302        }
303    }
304}

306// ─── ScalarQuantColumnMut (write) ──────────────────────────────────────────
307
308/// Mutable accessor for writing quantum values to a scalar column.
309///
310/// Corresponds to the write path of C++ `casa::ScalarQuantColumn<Double>`.
311pub struct ScalarQuantColumnMut<'a> {
312    table: &'a mut Table,
313    column_name: String,
314    desc: TableQuantumDesc,
315}
316
317impl<'a> ScalarQuantColumnMut<'a> {
318    /// Attaches to a scalar quantum column for writing.
319    pub fn new(table: &'a mut Table, column: &str) -> Result<Self, TableError> {
320        let desc = TableQuantumDesc::reconstruct(table, column).ok_or_else(|| {
321            TableError::Storage(format!("column '{column}' has no quantum keywords"))
322        })?;
323        Ok(Self {
324            table,
325            column_name: column.to_owned(),
326            desc,
327        })
328    }
329
330    /// Writes a quantity at `row`.
331    ///
332    /// For fixed-unit columns, the quantity's value is converted to the
333    /// column's unit before writing. For variable-unit columns, both the
334    /// value and unit string are written to their respective columns.
335    pub fn put(&mut self, row: usize, q: &Quantity) -> Result<(), TableError> {
336        if self.desc.is_unit_variable() {
337            // Write value as-is.
338            self.table.set_cell(
339                row,
340                &self.column_name,
341                Value::Scalar(ScalarValue::Float64(q.value())),
342            )?;
343            // Write unit string.
344            let units_col = self.desc.unit_column_name().unwrap().to_owned();
345            self.table.set_cell(
346                row,
347                &units_col,
348                Value::Scalar(ScalarValue::String(q.unit().name().to_owned())),
349            )?;
350        } else {
351            // Convert to column's fixed unit.
352            let col_unit_str = &self.desc.units[0];
353            let col_unit = Unit::new(col_unit_str).map_err(|e| {
354                TableError::Storage(format!("invalid column unit '{col_unit_str}': {e}"))
355            })?;
356            let converted = q
357                .get_value_in(&col_unit)
358                .map_err(|e| TableError::Storage(format!("unit conversion failed: {e}")))?;
359            self.table.set_cell(
360                row,
361                &self.column_name,
362                Value::Scalar(ScalarValue::Float64(converted)),
363            )?;
364        }
365        Ok(())
366    }
367}

369// ─── ArrayQuantColumn (read-only) ──────────────────────────────────────────
370
371/// Read-only accessor for an array column with quantum (unit) metadata.
372///
373/// Corresponds to C++ `casa::ArrayQuantColumn<Double>`.
374///
375/// Supports three unit modes:
376/// - **Fixed**: all elements share the same unit(s) from the keyword.
377/// - **Variable per-element**: each array element has its own unit, stored in
378///   an array `String` column with the same shape.
379/// - **Variable per-row**: each row has a single unit, stored in a scalar
380///   `String` column.
381pub struct ArrayQuantColumn<'a> {
382    table: &'a Table,
383    column_name: String,
384    desc: TableQuantumDesc,
385    fixed_units: Vec<Unit>,
386    convert_unit: Option<Unit>,
387}
388
389impl<'a> ArrayQuantColumn<'a> {
390    /// Attaches to an array quantum column.
391    pub fn new(table: &'a Table, column: &str) -> Result<Self, TableError> {
392        let desc = TableQuantumDesc::reconstruct(table, column).ok_or_else(|| {
393            TableError::Storage(format!("column '{column}' has no quantum keywords"))
394        })?;
395        let fixed_units = desc
396            .units
397            .iter()
398            .map(|u| {
399                Unit::new(u).map_err(|e| TableError::Storage(format!("invalid unit '{u}': {e}")))
400            })
401            .collect::<Result<Vec<_>, _>>()?;
402        Ok(Self {
403            table,
404            column_name: column.to_owned(),
405            desc,
406            fixed_units,
407            convert_unit: None,
408        })
409    }
410
411    /// Attaches with on-read conversion to `unit`.
412    pub fn with_unit(table: &'a Table, column: &str, unit: &str) -> Result<Self, TableError> {
413        let mut col = Self::new(table, column)?;
414        col.convert_unit = Some(
415            Unit::new(unit)
416                .map_err(|e| TableError::Storage(format!("invalid target unit '{unit}': {e}")))?,
417        );
418        Ok(col)
419    }
420
421    /// Returns `true` if units vary per row or per element.
422    pub fn is_unit_variable(&self) -> bool {
423        self.desc.is_unit_variable()
424    }
425
426    /// Returns the fixed unit name(s). Empty if variable.
427    pub fn units(&self) -> &[String] {
428        self.desc.units()
429    }
430
431    /// Reads all quantities at `row`, returned as a `Vec<Quantity>`.
432    ///
433    /// The array is flattened to a vector of quantities in row-major order.
434    pub fn get(&self, row: usize) -> Result<Vec<Quantity>, TableError> {
435        let arr = self.table.get_array_cell(row, &self.column_name)?;
436        let values = array_to_f64_vec(arr, row, &self.column_name)?;
437
438        if self.desc.is_unit_variable() {
439            self.get_variable(row, &values)
440        } else {
441            self.get_fixed(&values)
442        }
443    }
444
445    fn get_fixed(&self, values: &[f64]) -> Result<Vec<Quantity>, TableError> {
446        let units = if !self.fixed_units.is_empty() {
447            &self.fixed_units
448        } else {
449            return Ok(values.iter().map(|&v| Quantity::dimensionless(v)).collect());
450        };
451        let unit_count = units.len();
452
453        let mut result = Vec::with_capacity(values.len());
454        for (i, &v) in values.iter().enumerate() {
455            let q = Quantity::with_unit(v, units[i % unit_count].clone());
456            if let Some(ref target) = self.convert_unit {
457                let converted = q
458                    .get_value_in(target)
459                    .map_err(|e| TableError::Storage(format!("unit conversion failed: {e}")))?;
460                result.push(Quantity::with_unit(converted, target.clone()));
461            } else {
462                result.push(q);
463            }
464        }
465        Ok(result)
466    }
467
468    fn get_variable(&self, row: usize, values: &[f64]) -> Result<Vec<Quantity>, TableError> {
469        let units_col = self.desc.unit_column_name().unwrap();
470
471        // Try array units column first (per-element), then scalar (per-row).
472        let cell = self.table.cell(row, units_col)?;
473        match cell {
474            Some(Value::Array(ArrayValue::String(arr))) => {
475                let unit_strs: Vec<&str> = arr.iter().map(|s| s.as_str()).collect();
476                if unit_strs.len() != values.len() {
477                    return Err(TableError::Storage(format!(
478                        "unit array length {} != value array length {} at row {row}",
479                        unit_strs.len(),
480                        values.len()
481                    )));
482                }
483                let mut result = Vec::with_capacity(values.len());
484                for (&v, unit_str) in values.iter().zip(unit_strs) {
485                    let unit = Unit::new(unit_str).map_err(|e| {
486                        TableError::Storage(format!("invalid unit '{unit_str}' at row {row}: {e}"))
487                    })?;
488                    let q = Quantity::with_unit(v, unit);
489                    if let Some(ref target) = self.convert_unit {
490                        let converted = q.get_value_in(target).map_err(|e| {
491                            TableError::Storage(format!("unit conversion failed: {e}"))
492                        })?;
493                        result.push(Quantity::with_unit(converted, target.clone()));
494                    } else {
495                        result.push(q);
496                    }
497                }
498                Ok(result)
499            }
500            Some(Value::Scalar(ScalarValue::String(unit_str))) => {
501                let unit = Unit::new(unit_str).map_err(|e| {
502                    TableError::Storage(format!("invalid unit '{unit_str}' at row {row}: {e}"))
503                })?;
504                let mut result = Vec::with_capacity(values.len());
505                for &v in values {
506                    let q = Quantity::with_unit(v, unit.clone());
507                    if let Some(ref target) = self.convert_unit {
508                        let converted = q.get_value_in(target).map_err(|e| {
509                            TableError::Storage(format!("unit conversion failed: {e}"))
510                        })?;
511                        result.push(Quantity::with_unit(converted, target.clone()));
512                    } else {
513                        result.push(q);
514                    }
515                }
516                Ok(result)
517            }
518            _ => Err(TableError::Storage(format!(
519                "units column '{units_col}' has no String value at row {row}"
520            ))),
521        }
522    }
523}

525// ─── ArrayQuantColumnMut (write) ───────────────────────────────────────────
526
527/// Mutable accessor for writing quantum arrays to an array column.
528///
529/// Corresponds to the write path of C++ `casa::ArrayQuantColumn<Double>`.
530pub struct ArrayQuantColumnMut<'a> {
531    table: &'a mut Table,
532    column_name: String,
533    desc: TableQuantumDesc,
534    /// Whether the units companion column is scalar (per-row) vs array (per-element).
535    /// Determined once at construction from the column descriptor, matching C++
536    /// `ArrayQuantColumn::init` which inspects the column desc, not cell contents.
537    units_col_is_scalar: Option<bool>,
538}
539
540impl<'a> ArrayQuantColumnMut<'a> {
541    /// Attaches to an array quantum column for writing.
542    pub fn new(table: &'a mut Table, column: &str) -> Result<Self, TableError> {
543        let desc = TableQuantumDesc::reconstruct(table, column).ok_or_else(|| {
544            TableError::Storage(format!("column '{column}' has no quantum keywords"))
545        })?;
546        let units_col_is_scalar = if let Some(units_col_name) = desc.unit_column_name() {
547            table
548                .schema()
549                .and_then(|s| s.column(units_col_name))
550                .map(|cs| matches!(cs.column_type(), ColumnType::Scalar))
551        } else {
552            None
553        };
554        Ok(Self {
555            table,
556            column_name: column.to_owned(),
557            desc,
558            units_col_is_scalar,
559        })
560    }
561
562    /// Writes an array of quantities at `row`.
563    ///
564    /// For fixed-unit columns, values are converted to the column's unit.
565    /// For variable-unit columns, each quantity's unit string is written to
566    /// the companion column.
567    pub fn put(&mut self, row: usize, quanta: &[Quantity]) -> Result<(), TableError> {
568        if self.desc.is_unit_variable() {
569            let values: Vec<f64> = quanta.iter().map(|q| q.value()).collect();
570            let units: Vec<String> = quanta.iter().map(|q| q.unit().name().to_owned()).collect();
571
572            self.table.set_cell(
573                row,
574                &self.column_name,
575                Value::Array(ArrayValue::from_f64_vec(values)),
576            )?;
577
578            let units_col = self.desc.unit_column_name().unwrap().to_owned();
579            // Determine if units column is scalar or array from schema,
580            // matching C++ ArrayQuantColumn::init which inspects the column
581            // descriptor once at construction (not per-cell contents).
582            let is_scalar = self.units_col_is_scalar.unwrap_or(true);
583
584            if is_scalar && quanta.len() == 1 {
585                self.table.set_cell(
586                    row,
587                    &units_col,
588                    Value::Scalar(ScalarValue::String(units[0].clone())),
589                )?;
590            } else {
591                // Check if the existing cell at this position is scalar
592                // (per-row mode: all elements share one unit).
593                // If so, write the first unit as the per-row unit.
594                if is_scalar {
595                    self.table.set_cell(
596                        row,
597                        &units_col,
598                        Value::Scalar(ScalarValue::String(units[0].clone())),
599                    )?;
600                } else {
601                    self.table.set_cell(
602                        row,
603                        &units_col,
604                        Value::Array(ArrayValue::from_string_vec(units)),
605                    )?;
606                }
607            }
608        } else {
609            let col_units = self
610                .desc
611                .units
612                .iter()
613                .map(|unit| {
614                    Unit::new(unit).map_err(|e| {
615                        TableError::Storage(format!("invalid column unit '{unit}': {e}"))
616                    })
617                })
618                .collect::<Result<Vec<_>, _>>()?;
619            if col_units.is_empty() {
620                self.table.set_cell(
621                    row,
622                    &self.column_name,
623                    Value::Array(ArrayValue::from_f64_vec(
624                        quanta.iter().map(|q| q.value()).collect(),
625                    )),
626                )?;
627                return Ok(());
628            }
629            let unit_count = col_units.len();
630
631            let values: Result<Vec<f64>, _> = quanta
632                .iter()
633                .enumerate()
634                .map(|(i, q)| {
635                    q.get_value_in(&col_units[i % unit_count])
636                        .map_err(|e| TableError::Storage(format!("unit conversion failed: {e}")))
637                })
638                .collect();
639
640            self.table.set_cell(
641                row,
642                &self.column_name,
643                Value::Array(ArrayValue::from_f64_vec(values?)),
644            )?;
645        }
646        Ok(())
647    }
648}

651
652fn scalar_to_f64(sv: &ScalarValue, row: usize, column: &str) -> Result<f64, TableError> {
653    match sv {
654        ScalarValue::Float64(v) => Ok(*v),
655        ScalarValue::Float32(v) => Ok(*v as f64),
656        ScalarValue::Int32(v) => Ok(*v as f64),
657        ScalarValue::Int64(v) => Ok(*v as f64),
658        _ => Err(TableError::Storage(format!(
659            "column '{column}' at row {row}: expected numeric scalar"
660        ))),
661    }
662}
663
664fn array_to_f64_vec(av: &ArrayValue, row: usize, column: &str) -> Result<Vec<f64>, TableError> {
665    match av {
666        ArrayValue::Float64(arr) => Ok(arr.iter().copied().collect()),
667        ArrayValue::Float32(arr) => Ok(arr.iter().map(|&v| v as f64).collect()),
668        ArrayValue::Int32(arr) => Ok(arr.iter().map(|&v| v as f64).collect()),
669        ArrayValue::Int64(arr) => Ok(arr.iter().map(|&v| v as f64).collect()),
670        _ => Err(TableError::Storage(format!(
671            "column '{column}' at row {row}: expected numeric array"
672        ))),
673    }
674}

682    /// Verifies that `ArrayQuantColumnMut::put` writes array units (not scalar)
683    /// on an empty row when the schema says the units column is an array column.
684    #[test]
685    fn array_quant_put_writes_array_units_on_empty_row() {
686        let schema = TableSchema::new(vec![
687            ColumnSchema::array_variable("DATA", PrimitiveType::Float64, Some(1)),
688            ColumnSchema::array_variable("DATA_UNITS", PrimitiveType::String, Some(1)),
689        ])
690        .unwrap();
691        let mut table = crate::Table::with_schema(schema);
692
693        TableQuantumDesc::with_variable_units("DATA", "DATA_UNITS")
694            .write(&mut table)
695            .unwrap();
696
697        // Add an empty row
698        table
699            .add_row(casacore_types::RecordValue::default())
700            .unwrap();
701
702        // Write quanta to the empty row
703        let quanta = vec![
704            Quantity::with_unit(1.0, Unit::new("Hz").unwrap()),
705            Quantity::with_unit(2.0, Unit::new("kHz").unwrap()),
706        ];
707        {
708            let mut col = ArrayQuantColumnMut::new(&mut table, "DATA").unwrap();
709            col.put(0, &quanta).unwrap();
710        }
711
712        // Read back: the units column should be an array, not a scalar
713        let cell = table.cell(0, "DATA_UNITS");
714        assert!(
715            matches!(cell, Ok(Some(Value::Array(ArrayValue::String(_))))),
716            "expected array units but got: {cell:?}"
717        );
718        if let Ok(Some(Value::Array(ArrayValue::String(arr)))) = cell {
719            let units: Vec<&str> = arr.iter().map(|s| s.as_str()).collect();
720            assert_eq!(units, vec!["Hz", "kHz"]);
721        }
722    }
