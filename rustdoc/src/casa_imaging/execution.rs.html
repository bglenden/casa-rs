12/// Internal backend selection for standard MFS execution.
13#[derive(Debug, Clone, Copy, PartialEq, Eq)]
14pub(crate) enum StandardMfsBackend {
15    /// Synchronous CPU execution using the native Rust gridder.
16    Cpu,
17    /// Reserved marker for future backends that must fail before execution.
18    #[allow(dead_code)]
19    Reserved(&'static str),
20}

22/// CPU-only executor for standard MFS imaging.

24    gridder: &'a StandardGridder,
25    plan: StandardMfsVisibilityPlan<'a>,

29/// CPU executor for streaming standard MFS dirty accumulation.
30///
31/// This owns the standard gridder and reusable grids while each call to
32/// `accumulate_batches` builds only a borrowed plan for the current frontend
33/// row block.
34pub(crate) struct StandardMfsDirtyCpuExecutor {
35    gridder: StandardGridder,
36    workspace: StandardMfsWorkspace,
37    normalization_sumwt: f64,
38    reported_sumwt: f64,
39    gridded_samples: usize,
40    skipped_samples: usize,
41    max_abs_w_lambda: f64,
42}

44/// Immutable summary of accumulated streaming standard MFS dirty samples.
45#[derive(Debug, Clone, Copy, PartialEq)]
46pub(crate) struct StandardMfsDirtyAccumulation {
47    /// PSF and residual normalization sum from accepted samples.
48    pub(crate) normalization_sumwt: f64,
49    /// CASA-style reported sumwt from accepted samples.
50    pub(crate) reported_sumwt: f64,
51    /// Number of samples accepted by the standard gridder.
52    pub(crate) gridded_samples: usize,
53    /// Number of samples rejected by flags, weights, or gridder bounds.
54    pub(crate) skipped_samples: usize,
55    /// Maximum absolute `w` coordinate seen in wavelengths.
56    pub(crate) max_abs_w_lambda: f64,
57}

59impl<'a> StandardMfsCpuExecutor<'a> {
60    /// Build a CPU executor over weighted standard-gridder visibility batches.
61    pub(crate) fn new(
62        gridder: &'a StandardGridder,
63        batches: &'a [VisibilityBatch],
64    ) -> Result<Self, ImagingError> {
65        Self::for_backend(StandardMfsBackend::Cpu, gridder, batches)
66    }
67
68    /// Build an executor for a requested internal backend.
69    pub(crate) fn for_backend(
70        backend: StandardMfsBackend,
71        gridder: &'a StandardGridder,
72        batches: &'a [VisibilityBatch],
73    ) -> Result<Self, ImagingError> {
74        match backend {
75            StandardMfsBackend::Cpu => Ok(Self {
76                gridder,
77                plan: StandardMfsVisibilityPlan::new(gridder, batches),
78                workspace: StandardMfsWorkspace::new(gridder),
79            }),
80            StandardMfsBackend::Reserved(name) => Err(ImagingError::Unsupported(format!(
81                "standard MFS backend '{name}' is not implemented"
82            ))),
83        }
84    }
85
86    /// Borrow the executor's gridder, visibility plan, and mutable workspace.
87    pub(crate) fn parts_mut(
88        &mut self,
89    ) -> (
90        &'a StandardGridder,
91        &StandardMfsVisibilityPlan<'a>,
92        &mut StandardMfsWorkspace,
93    ) {
94        (self.gridder, &self.plan, &mut self.workspace)
95    }
96
97    /// Return the prepared visibility plan.
98    #[cfg(test)]
99    pub(crate) fn plan(&self) -> &StandardMfsVisibilityPlan<'a> {
100        &self.plan
101    }
102}

104impl StandardMfsDirtyCpuExecutor {
105    /// Build a dirty accumulator executor for the requested internal backend.
106    pub(crate) fn for_backend(
107        backend: StandardMfsBackend,
108        geometry: ImageGeometry,
109    ) -> Result<Self, ImagingError> {
110        match backend {
111            StandardMfsBackend::Cpu => {
112                let gridder = StandardGridder::new(geometry)?;
113                let workspace = StandardMfsWorkspace::new(&gridder);
114                Ok(Self {
115                    gridder,
116                    workspace,
117                    normalization_sumwt: 0.0,
118                    reported_sumwt: 0.0,
119                    gridded_samples: 0,
120                    skipped_samples: 0,
121                    max_abs_w_lambda: 0.0,
122                })
123            }
124            StandardMfsBackend::Reserved(name) => Err(ImagingError::Unsupported(format!(
125                "standard MFS backend '{name}' is not implemented"
126            ))),
127        }
128    }
129
130    /// Build the current CPU dirty executor.
131    pub(crate) fn new(geometry: ImageGeometry) -> Result<Self, ImagingError> {
132        Self::for_backend(StandardMfsBackend::Cpu, geometry)
133    }
134
135    /// Accumulate a borrowed row-block plan into the reusable dirty grids.
136    pub(crate) fn accumulate_batches(
137        &mut self,
138        batches: &[VisibilityBatch],
139    ) -> Result<(), ImagingError> {
140        for batch in batches {
141            batch.validate()?;
142            self.max_abs_w_lambda = batch
143                .w_lambda
144                .iter()
145                .fold(self.max_abs_w_lambda, |max_abs_w_lambda, &w_lambda| {
146                    max_abs_w_lambda.max(w_lambda.abs())
147                });
148        }
149
150        let plan = StandardMfsVisibilityPlan::new(&self.gridder, batches);
151        let (psf_grid, residual_grid) = self.workspace.dirty_grids_mut();
152        for sample in plan.samples() {
153            let grid_weight = sample.grid_weight_f64();
154            self.gridder.grid_sample_product_planned_f64(
155                psf_grid,
156                &sample.plan.positive,
157                Complex64::new(grid_weight, 0.0),
158            );
159
160            let batch = plan.batch(sample);
161            let observed_visibility = batch.visibility[sample.sample_index];
162            if finite_visibility(observed_visibility) {
163                let residual = Complex64::new(
164                    f64::from(observed_visibility.re) * grid_weight,
165                    f64::from(observed_visibility.im) * grid_weight,
166                );
167                self.gridder.grid_sample_product_planned_f64(
168                    residual_grid,
169                    &sample.plan.positive,
170                    residual,
171                );
172            }
173        }
174
175        self.normalization_sumwt += plan.normalization_sumwt();
176        self.reported_sumwt += plan.reported_sumwt();
177        self.gridded_samples += plan.gridded_samples();
178        self.skipped_samples += plan.skipped_samples();
179        Ok(())
180    }
181
182    /// Return the reusable standard gridder.
183    pub(crate) fn gridder(&self) -> &StandardGridder {
184        &self.gridder
185    }
186
187    /// Return the accumulated dirty PSF and residual grids.
188    pub(crate) fn dirty_grids(&self) -> (&Array2<Complex64>, &Array2<Complex64>) {
189        self.workspace.dirty_grids()
190    }
191
192    /// Return the accumulated sample summary.
193    pub(crate) fn accumulation(&self) -> StandardMfsDirtyAccumulation {
194        StandardMfsDirtyAccumulation {
195            normalization_sumwt: self.normalization_sumwt,
196            reported_sumwt: self.reported_sumwt,
197            gridded_samples: self.gridded_samples,
198            skipped_samples: self.skipped_samples,
199            max_abs_w_lambda: self.max_abs_w_lambda,
200        }
201    }
202}

204/// Borrowed, planned view of weighted standard-gridder visibility samples.
205pub(crate) struct StandardMfsVisibilityPlan<'a> {
206    batches: &'a [VisibilityBatch],
207    samples: Vec<StandardMfsPlannedSample>,
208    normalization_sumwt: f64,
209    reported_sumwt: f64,
210    skipped_samples: usize,
211}
212
213impl<'a> StandardMfsVisibilityPlan<'a> {
214    fn new(gridder: &StandardGridder, batches: &'a [VisibilityBatch]) -> Self {
215        let sample_count = batches.iter().map(VisibilityBatch::len).sum();
216        let mut samples = Vec::with_capacity(sample_count);
217        let mut normalization_sumwt = 0.0f64;
218        let mut reported_sumwt = 0.0f64;
219        let mut skipped_samples = 0usize;
220
221        for (batch_index, batch) in batches.iter().enumerate() {
222            for sample_index in 0..batch.len() {
223                if !batch.gridable[sample_index] {
224                    skipped_samples += 1;
225                    continue;
226                }
227                let weight = batch.weight[sample_index];
228                let sumwt_factor = batch.sumwt_factor[sample_index];
229                if !(weight.is_finite()
230                    && weight > 0.0
231                    && sumwt_factor.is_finite()
232                    && sumwt_factor > 0.0)
233                {
234                    skipped_samples += 1;
235                    continue;
236                }
237                let Some(plan) =
238                    gridder.plan_sample(batch.u_lambda[sample_index], batch.v_lambda[sample_index])
239                else {
240                    skipped_samples += 1;
241                    continue;
242                };
243                let grid_weight = weight * sumwt_factor;
244                let sumwt = f64::from(grid_weight);
245                normalization_sumwt += sumwt;
246                reported_sumwt += sumwt;
247                samples.push(StandardMfsPlannedSample {
248                    batch_index,
249                    sample_index,
250                    grid_weight,
251                    plan,
252                });
253            }
254        }
255
256        Self {
257            batches,
258            samples,
259            normalization_sumwt,
260            reported_sumwt,
261            skipped_samples,
262        }
263    }
264
265    /// Return the planned samples accepted by the standard gridder.
266    pub(crate) fn samples(&self) -> &[StandardMfsPlannedSample] {
267        &self.samples
268    }
269
270    /// Return a planned sample's source batch.
271    pub(crate) fn batch(&self, sample: &StandardMfsPlannedSample) -> &'a VisibilityBatch {
272        &self.batches[sample.batch_index]
273    }
274
275    /// Return the PSF and residual normalization sum of accepted samples.
276    pub(crate) fn normalization_sumwt(&self) -> f64 {
277        self.normalization_sumwt
278    }
279
280    /// Return the CASA-style reported sumwt of accepted samples.
281    pub(crate) fn reported_sumwt(&self) -> f64 {
282        self.reported_sumwt
283    }
284
285    /// Return the number of samples accepted by the gridder plan.
286    pub(crate) fn gridded_samples(&self) -> usize {
287        self.samples.len()
288    }
289
290    /// Return the number of samples rejected while building the plan.
291    pub(crate) fn skipped_samples(&self) -> usize {
292        self.skipped_samples
293    }
294}

296/// One planned standard MFS visibility sample.
297pub(crate) struct StandardMfsPlannedSample {
298    /// Source visibility batch index.
299    pub(crate) batch_index: usize,
300    /// Source scalar sample index inside the batch.
301    pub(crate) sample_index: usize,
302    /// Product of imaging weight and sumwt factor used by standard MFS grids.

315/// Reusable f64 grid workspace for standard MFS CPU execution.
316pub(crate) struct StandardMfsWorkspace {
317    psf_grid: Array2<Complex64>,
318    residual_grid: Array2<Complex64>,
319}
320
321impl StandardMfsWorkspace {
322    fn new(gridder: &StandardGridder) -> Self {
323        let [nx, ny] = gridder.grid_shape();
324        Self {
325            psf_grid: Array2::zeros((nx, ny)),
326            residual_grid: Array2::zeros((nx, ny)),
327        }
328    }
329
330    /// Clear and borrow the PSF grid.
331    pub(crate) fn clear_psf_grid(&mut self) -> &mut Array2<Complex64> {
332        self.psf_grid.fill(Complex64::new(0.0, 0.0));
333        &mut self.psf_grid
334    }
335
336    /// Clear and borrow the residual grid.
337    pub(crate) fn clear_residual_grid(&mut self) -> &mut Array2<Complex64> {
338        self.residual_grid.fill(Complex64::new(0.0, 0.0));
339        &mut self.residual_grid
340    }
341
342    /// Clear and borrow the PSF and residual grids for a combined dirty pass.
343    pub(crate) fn clear_dirty_grids(&mut self) -> (&mut Array2<Complex64>, &mut Array2<Complex64>) {
344        self.psf_grid.fill(Complex64::new(0.0, 0.0));
345        self.residual_grid.fill(Complex64::new(0.0, 0.0));
346        (&mut self.psf_grid, &mut self.residual_grid)
347    }
348
349    /// Borrow the PSF and residual grids without clearing them.
350    pub(crate) fn dirty_grids_mut(&mut self) -> (&mut Array2<Complex64>, &mut Array2<Complex64>) {
351        (&mut self.psf_grid, &mut self.residual_grid)
352    }
353
354    /// Borrow the PSF and residual grids without clearing them.
355    pub(crate) fn dirty_grids(&self) -> (&Array2<Complex64>, &Array2<Complex64>) {
356        (&self.psf_grid, &self.residual_grid)
357    }
358}
359
360/// Return true when a scalar visibility can contribute to a residual grid.
361pub(crate) fn finite_visibility(visibility: Complex32) -> bool {
362    visibility.re.is_finite() && visibility.im.is_finite()
363}

371    #[test]
372    fn standard_mfs_plan_buckets_gridder_accepted_samples() {
373        let geometry = ImageGeometry {
374            image_shape: [32, 32],
375            cell_size_rad: [1.0e-4, 1.0e-4],
376        };
377        let gridder = StandardGridder::new(geometry).unwrap();
378        let batches = vec![VisibilityBatch {
379            u_lambda: vec![0.0, 4.0, 8.0, 12.0],
380            v_lambda: vec![0.0, 1.0, 2.0, 3.0],
381            w_lambda: vec![0.0; 4],
382            weight: vec![1.0, 2.0, 0.0, 3.0],
383            sumwt_factor: vec![1.0, 2.0, 1.0, f32::NAN],
384            gridable: vec![true, true, true, true],
385            visibility: vec![Complex32::new(1.0, 0.0); 4],
386        }];
387
388        let executor = StandardMfsCpuExecutor::new(&gridder, &batches).unwrap();
389        let plan = executor.plan();
390
391        assert_eq!(plan.gridded_samples(), 2);
392        assert_eq!(plan.skipped_samples(), 2);
393        assert!((plan.normalization_sumwt() - 5.0).abs() < 1.0e-6);
394        assert!((plan.reported_sumwt() - 5.0).abs() < 1.0e-6);
395        assert_eq!(plan.samples()[0].batch_index, 0);
396        assert_eq!(plan.samples()[1].sample_index, 1);
397    }

424    #[test]
425    fn streaming_dirty_executor_accumulates_borrowed_row_blocks() {
426        let geometry = ImageGeometry {
427            image_shape: [32, 32],
428            cell_size_rad: [1.0e-4, 1.0e-4],
429        };
430        let left = VisibilityBatch {
431            u_lambda: vec![0.0, 4.0],
432            v_lambda: vec![0.0, 1.0],
433            w_lambda: vec![3.0, -5.0],
434            weight: vec![1.0, 2.0],
435            sumwt_factor: vec![1.0, 2.0],
436            gridable: vec![true, true],
437            visibility: vec![Complex32::new(1.0, 0.0); 2],
438        };
439        let right = VisibilityBatch {
440            u_lambda: vec![8.0, 12.0],
441            v_lambda: vec![2.0, 3.0],
442            w_lambda: vec![7.0, 11.0],
443            weight: vec![0.0, 3.0],
444            sumwt_factor: vec![1.0, f32::NAN],
445            gridable: vec![true, true],
446            visibility: vec![Complex32::new(1.0, 0.0); 2],
447        };
448        let mut executor = StandardMfsDirtyCpuExecutor::new(geometry).unwrap();
449
450        executor.accumulate_batches(&[left]).unwrap();
451        executor.accumulate_batches(&[right]).unwrap();
452        let accumulation = executor.accumulation();
453
454        assert_eq!(accumulation.gridded_samples, 2);
455        assert_eq!(accumulation.skipped_samples, 2);
456        assert!((accumulation.normalization_sumwt - 5.0).abs() < 1.0e-6);
457        assert!((accumulation.reported_sumwt - 5.0).abs() < 1.0e-6);
458        assert!((accumulation.max_abs_w_lambda - 11.0).abs() < 1.0e-6);
459    }
