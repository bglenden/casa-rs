53pub(crate) fn apply_weighting_with_density_source(
54    weighting: WeightingMode,
55    weight_density_mode: WeightDensityMode,
56    uv_taper: Option<GaussianUvTaper>,
57    fractional_bandwidth: f64,
58    target_batches: &[VisibilityBatch],
59    density_batches: &[VisibilityBatch],
60    gridder: &StandardGridder,
61) -> Result<Vec<VisibilityBatch>, crate::ImagingError> {
62    let density_convention = density_cell_convention(weighting, weight_density_mode);
63    match weighting {
64        WeightingMode::Natural => Ok(apply_optional_uv_taper(target_batches.to_vec(), uv_taper)),
65        WeightingMode::Uniform => {
66            let density = build_density_grid(
67                density_batches,
68                gridder,
69                density_includes_conjugates(density_convention),
70                density_convention,
71            );
72            Ok(apply_optional_uv_taper(
73                target_batches
74                    .iter()
75                    .map(|batch| {
76                        reweight_batch(
77                            batch,
78                            gridder,
79                            &density,
80                            density_convention,
81                            |weight, density, _, _| weight / density,
82                        )
83                    })
84                    .collect(),
85                uv_taper,
86            ))
87        }
88        WeightingMode::Briggs { robust } | WeightingMode::BriggsBwTaper { robust } => {
89            let density = build_density_grid(
90                density_batches,
91                gridder,
92                density_includes_conjugates(density_convention),
93                density_convention,
94            );
95            let density_weight_sum = density.iter().map(|value| f64::from(*value)).sum::<f64>();
96            let total_density_weight = density_weight_sum
97                * match density_convention {
98                    DensityCellConvention::VisImagingWeight => 1.0,
99                    DensityCellConvention::CubeBriggsWeightor => 1.0,
100                };
101            let sumlocwt = density
102                .iter()
103                .filter(|value| **value > 0.0)
104                .map(|value| f64::from(*value) * f64::from(*value))
105                .sum::<f64>();
106            let f2 = if sumlocwt > 0.0 && total_density_weight > 0.0 {
107                (5.0f64 * 10f64.powf(-(robust as f64))).powi(2) / (sumlocwt / total_density_weight)
108            } else {
109                0.0
110            } as f32;
111            if std::env::var_os("CASA_RS_TRACE_RUST_WEIGHTING").is_some() {
112                let density_nonzero = density.iter().filter(|value| **value > 0.0).count();
113                let density_max = density
114                    .iter()
115                    .copied()
116                    .fold(0.0f32, |acc, value| acc.max(value));
117                eprintln!(
118                    "CASA_RS_TRACE_RUST_WEIGHTING briggs_density_summary total_density_weight={total_density_weight:.12e} density_sum_sq={sumlocwt:.12e} density_max={density_max:.12e} density_nonzero={density_nonzero} f2={f2:.12e}"
119                );
120            }
121            Ok(apply_optional_uv_taper(
122                target_batches
123                    .iter()
124                    .map(|batch| {
125                        reweight_batch(
126                            batch,
127                            gridder,
128                            &density,
129                            density_convention,
130                            |weight, density, u_lambda, v_lambda| {
131                                let taper_factor = match weighting {
132                                    WeightingMode::BriggsBwTaper { .. } => {
133                                        briggs_bw_taper_uv_distance_factor(
134                                            fractional_bandwidth,
135                                            gridder,
136                                            u_lambda,
137                                            v_lambda,
138                                        ) as f32
139                                    }
140                                    _ => 1.0,
141                                };
142                                weight / ((f2 * density) / taper_factor + 1.0)
143                            },
144                        )
145                    })
146                    .collect(),
147                uv_taper,
148            ))
149        }
150    }
151}

153pub(crate) fn trace_weighting_with_density_source(
154    weighting: WeightingMode,
155    weight_density_mode: WeightDensityMode,
156    uv_taper: Option<GaussianUvTaper>,
157    fractional_bandwidth: f64,
158    target_batches: &[VisibilityBatch],
159    density_batches: &[VisibilityBatch],
160    gridder: &StandardGridder,
161) -> Result<WeightingTraceInternal, crate::ImagingError> {
162    let density_convention = density_cell_convention(weighting, weight_density_mode);
163    let density = match weighting {
164        WeightingMode::Natural => None,
165        WeightingMode::Uniform
166        | WeightingMode::Briggs { .. }
167        | WeightingMode::BriggsBwTaper { .. } => Some(build_density_grid(
168            density_batches,
169            gridder,
170            density_includes_conjugates(density_convention),
171            density_convention,
172        )),
173    };
174    let weighted_batches = apply_weighting_with_density_source(
175        weighting,
176        weight_density_mode,
177        uv_taper,
178        fractional_bandwidth,
179        target_batches,
180        density_batches,
181        gridder,
182    )?;
183    let mut samples = Vec::new();
184    let mut gridded_samples = 0usize;
185    let mut skipped_samples = 0usize;
186    let mut normalization_sumwt = 0.0f32;
187    let mut reported_sumwt = 0.0f32;
188
189    for (batch_index, (input_batch, weighted_batch)) in target_batches
190        .iter()
191        .zip(weighted_batches.iter())
192        .enumerate()
193    {
194        for sample_index in 0..input_batch.len() {
195            let output_weight = weighted_batch.weight[sample_index];
196            let sumwt_factor = weighted_batch.sumwt_factor[sample_index];
197            let gridable = weighted_batch.gridable[sample_index];
198            let contributes = gridable
199                && output_weight.is_finite()
200                && output_weight > 0.0
201                && sumwt_factor.is_finite()
202                && sumwt_factor > 0.0;
203            let reported_contribution = if contributes {
204                output_weight * sumwt_factor
205            } else {
206                0.0
207            };
208            let normalization_contribution = reported_contribution;
209            if contributes {
210                gridded_samples += 1;
211                normalization_sumwt += normalization_contribution;
212                reported_sumwt += reported_contribution;
213            } else {
214                skipped_samples += 1;
215            }
216            samples.push(WeightingSampleTraceInternal {
217                batch_index,
218                sample_index,
219                u_lambda: input_batch.u_lambda[sample_index],
220                v_lambda: input_batch.v_lambda[sample_index],
221                w_lambda: input_batch.w_lambda[sample_index],
222                input_weight: input_batch.weight[sample_index],
223                density_weight: density.as_ref().and_then(|grid| {
224                    gridder.density_at_with_convention(
225                        grid,
226                        input_batch.u_lambda[sample_index],
227                        input_batch.v_lambda[sample_index],
228                        density_convention,
229                    )
230                }),
231                output_weight,
232                sumwt_factor,
233                gridable,
234                normalization_contribution,
235                reported_contribution,
236            });
237        }
238    }
239
240    Ok(WeightingTraceInternal {
241        weighted_batches,
242        samples,
243        gridded_samples,
244        skipped_samples,
245        normalization_sumwt,
246        reported_sumwt,
247    })
248}

250fn density_cell_convention(
251    weighting: WeightingMode,
252    weight_density_mode: WeightDensityMode,
253) -> DensityCellConvention {
254    match (weighting, weight_density_mode) {
255        (
256            WeightingMode::Uniform
257            | WeightingMode::Briggs { .. }
258            | WeightingMode::BriggsBwTaper { .. },
259            WeightDensityMode::PerPlane,
260        ) => DensityCellConvention::CubeBriggsWeightor,
261        _ => DensityCellConvention::VisImagingWeight,
262    }
263}
264
265fn density_includes_conjugates(convention: DensityCellConvention) -> bool {
266    match convention {
267        DensityCellConvention::VisImagingWeight => true,
268        DensityCellConvention::CubeBriggsWeightor => true,
269    }
270}

272fn apply_optional_uv_taper(
273    mut batches: Vec<VisibilityBatch>,
274    taper: Option<GaussianUvTaper>,
275) -> Vec<VisibilityBatch> {
276    let Some(taper) = taper else {
277        return batches;
278    };
279    let (major_coeff, minor_coeff) = taper_coefficients(taper);
280    let cos_pa = taper.position_angle_rad.sin();
281    let sin_pa = taper.position_angle_rad.cos();
282    for batch in &mut batches {
283        for index in 0..batch.len() {
284            let weight = batch.weight[index];
285            if !(weight.is_finite() && weight > 0.0) {
286                batch.weight[index] = 0.0;
287                continue;
288            }
289            let u = batch.u_lambda[index];
290            let v = batch.v_lambda[index];
291            let ru = cos_pa * u + sin_pa * v;
292            let rv = -sin_pa * u + cos_pa * v;
293            let filter = (-major_coeff * ru * ru - minor_coeff * rv * rv).exp() as f32;
294            batch.weight[index] *= filter;
295        }
296    }
297    batches
298}
299
300fn taper_coefficients(taper: GaussianUvTaper) -> (f64, f64) {
301    let image_factor = std::f64::consts::PI * std::f64::consts::PI / (4.0 * std::f64::consts::LN_2);
302    let major = match taper.major {
303        UvTaperSize::ImageFwhmRad(value) => image_factor * value * value,
304        UvTaperSize::BaselineHwhmLambda(value) => std::f64::consts::LN_2 / (value * value),
305    };
306    let minor = match taper.minor {
307        UvTaperSize::ImageFwhmRad(value) => image_factor * value * value,
308        UvTaperSize::BaselineHwhmLambda(value) => std::f64::consts::LN_2 / (value * value),
309    };
310    (major, minor)
311}

313fn build_density_grid(
314    batches: &[VisibilityBatch],
315    gridder: &StandardGridder,
316    mirror_hermitian: bool,
317    convention: DensityCellConvention,
318) -> Array2<f32> {
319    let [nx, ny] = gridder.density_grid_shape();
320    let mut density_grid = Array2::<f32>::zeros((nx, ny));
321    for batch in batches {
322        for index in 0..batch.len() {
323            let weight = batch.weight[index];
324            if !(weight.is_finite() && weight > 0.0) {
325                continue;
326            }
327            let primary = (batch.u_lambda[index], batch.v_lambda[index]);
328            let conjugate = (-batch.u_lambda[index], -batch.v_lambda[index]);
329            let positions = if mirror_hermitian {
330                [Some(primary), Some(conjugate)]
331            } else {
332                [Some(primary), None]
333            };
334            for (u_lambda, v_lambda) in positions.into_iter().flatten() {
335                let Some((x, y)) =
336                    gridder.density_cell_index_with_convention(u_lambda, v_lambda, convention)
337                else {
338                    continue;
339                };
340                density_grid[(x, y)] += weight;
341            }
342        }
343    }
344    density_grid
345}

347fn reweight_batch(
348    batch: &VisibilityBatch,
349    gridder: &StandardGridder,
350    density: &Array2<f32>,
351    convention: DensityCellConvention,
352    transform: impl Fn(f32, f32, f64, f64) -> f32,
353) -> VisibilityBatch {
354    let mut reweighted = batch.clone();
355    for index in 0..batch.len() {
356        let weight = batch.weight[index];
357        let Some(cell_density) = gridder.density_at_with_convention(
358            density,
359            batch.u_lambda[index],
360            batch.v_lambda[index],
361            convention,
362        ) else {
363            reweighted.weight[index] = 0.0;
364            continue;
365        };
366        if !(weight.is_finite() && weight > 0.0 && cell_density.is_finite() && cell_density > 0.0) {
367            reweighted.weight[index] = 0.0;
368            continue;
369        }
370        reweighted.weight[index] = transform(
371            weight,
372            cell_density,
373            batch.u_lambda[index],
374            batch.v_lambda[index],
375        );
376    }
377    reweighted
378}

380pub(crate) fn fractional_bandwidth_from_frequency_range(frequency_range_hz: [f64; 2]) -> f64 {
381    let min_freq = frequency_range_hz[0].abs().min(frequency_range_hz[1].abs());
382    let max_freq = frequency_range_hz[0].abs().max(frequency_range_hz[1].abs());
383    if min_freq > 0.0 && max_freq.is_finite() {
384        2.0 * (max_freq - min_freq) / (max_freq + min_freq)
385    } else {
386        0.0
387    }
388}

390fn briggs_bw_taper_uv_distance_factor(
391    fractional_bandwidth: f64,
392    gridder: &StandardGridder,
393    u_lambda: f64,
394    v_lambda: f64,
395) -> f64 {
396    let n_cells_bw = fractional_bandwidth * gridder.cube_briggs_uv_cell_radius(u_lambda, v_lambda);
397    let mut factor = n_cells_bw + 0.5;
398    if factor < 1.5 {
399        factor = (4.0 - n_cells_bw) / (4.0 - 2.0 * n_cells_bw);
400    }
401    factor.max(f64::MIN_POSITIVE)
402}
