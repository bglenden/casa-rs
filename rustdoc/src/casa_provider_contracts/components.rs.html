3use std::collections::BTreeMap;
4
5use schemars::schema::{RootSchema, Schema};
6
7/// Merged reusable component schemas extracted from one or more root schemas.
8pub type ProviderComponentSchemas = BTreeMap<String, Schema>;
9
10/// Merge reusable component definitions from one or more root schemas.
11pub fn merged_components<'a>(
12    schemas: impl IntoIterator<Item = &'a RootSchema>,
13) -> ProviderComponentSchemas {
14    let mut merged = ProviderComponentSchemas::new();
15    for root in schemas {
16        for (name, schema) in &root.definitions {
17            merged.insert(name.clone(), schema.clone());
18        }
19    }
20    merged
21}
