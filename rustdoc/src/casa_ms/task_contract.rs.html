17/// JSON request accepted by the native `mstransform` provider.
18#[derive(Debug, Clone, Serialize, Deserialize, JsonSchema)]
19pub struct MsTransformTaskRequest {
20    /// Input MeasurementSet.
21    pub vis: PathBuf,
22    /// Output MeasurementSet.
23    pub outputvis: PathBuf,
24    /// CASA spectral-window selector.
25    #[serde(default)]
26    pub spw: String,
27    /// Adjacent channels to average.
28    #[serde(default = "default_width")]
29    pub width: usize,
30    /// CASA field selector.
31    #[serde(default)]
32    pub field: String,
33    /// CASA scan selector.
34    #[serde(default)]
35    pub scan: String,
36    /// CASA antenna selector.
37    #[serde(default)]
38    pub antenna: String,
39    /// MJD-seconds range in `start~end` form.
40    #[serde(default)]
41    pub timerange: String,
42    /// Additional TaQL row selection.
43    #[serde(default)]
44    pub msselect: String,
45    /// Input visibility column.
46    #[serde(default = "default_corrected_data")]
47    pub datacolumn: String,
48    /// Preserve fully flagged rows.
49    #[serde(default = "default_true")]
50    pub keepflags: bool,
51}

198fn one_shot_contract<R: JsonSchema, O: JsonSchema>(
199    protocol_name: &str,
200    surface_ids: &[&str],
201) -> TaskProviderContract {
202    let request_schema = schema_for!(R);
203    let result_schema = schema_for!(O);
204    TaskProviderContract {
205        protocol: ProviderProtocolDescriptor::new(
206            protocol_name,
207            1,
208            ProviderSurfaceKind::Task,
209            env!("CARGO_PKG_VERSION"),
210        ),
211        semantic: TaskSemanticContract {
212            request_schema: request_schema.clone(),
213            result_schema: result_schema.clone(),
214            operations: vec![TaskOperationDescriptor {
215                name: "run".to_string(),
216                request_kind: "run".to_string(),
217                result_kind: Some("run".to_string()),
218            }],
219        },
220        components: merged_components([&request_schema, &result_schema]),
221        annotations: serde_json::json!({}),
222        projections: ProviderProjectionMetadata {
223            cli: Some(ProviderCliProjection {
224                machine_actions: ProviderCliMachineActions {
225                    json_schema: Some("--json-schema".to_string()),
226                    protocol_info: Some("--protocol-info".to_string()),
227                    json_run: Some("--json-run <SOURCE>".to_string()),
228                    session: None,
229                },
230            }),
231            python: None,
232        },
233        parameter_surfaces: surface_ids
234            .iter()
235            .map(|surface| {
236                builtin_surface_bundle(surface)
237                    .unwrap_or_else(|error| panic!("built-in task surface {surface:?}: {error}"))
238            })
239            .collect(),
240        domain_schemas: TaskProviderSchemas {
241            request_schema,
242            result_schema,
243            additional: NoAdditionalProviderSchemas {},
244        },
245    }
246}
