17pub fn read_incremental_stman_file(
18    file_path: &Path,
19    dm_blob: &[u8],
20    col_descs: &[&ColumnDesc],
21    nrrow: usize,
22) -> Result<Vec<(String, ColumnData)>, TableReadError> {
23    let mut file = File::open(file_path)?;
24    let header = parse_ism_header(&mut file)?;
25    let _dm_name = parse_ism_dm_blob(dm_blob)?;
26    let index = parse_ism_index(&mut file, &header)?;
27    let big_endian = header.big_endian;
28
29    let mut builders = col_descs
30        .iter()
31        .map(|desc| ColumnBuilder::new(desc))
32        .collect::<Result<Vec<_>, _>>()?;
33    let mut indirect_arrays = if col_descs.iter().any(|desc| desc.is_indirect_array()) {
34        Some(StManArrayFileReader::open(
35            &append_path_suffix(file_path, "i"),
36            big_endian,
37        )?)
38    } else {
39        None
40    };
41
42    let n_intervals = index.bucket_nrs.len();
43    let mut last_bucket_nr = None;
44    let mut cached_bucket = None;
45
46    for interval in 0..n_intervals {
47        let bucket_start = index.rows[interval] as usize;
48        let bucket_end = index.rows[interval + 1] as usize;
49        let rows_in_bucket = bucket_end - bucket_start;
50        let bucket_nr = index.bucket_nrs[interval];
51
52        if last_bucket_nr != Some(bucket_nr) {
53            let raw = read_ism_bucket(&mut file, &header, bucket_nr)?;
54            cached_bucket = Some(parse_ism_bucket(&raw, col_descs.len(), big_endian)?);
55            last_bucket_nr = Some(bucket_nr);
56        }
57        let bucket = cached_bucket.as_ref().expect("cached bucket");
58
59        for (col_idx, desc) in col_descs.iter().enumerate() {
60            if col_idx >= bucket.col_indices.len() {
61                for _ in 0..rows_in_bucket {
62                    builders[col_idx].push_default()?;
63                }
64                continue;
65            }
66
67            let col_index = &bucket.col_indices[col_idx];
68            for rel_row in 0..rows_in_bucket {
69                let k = get_interval(col_index, rel_row as u32);
70                let data_offset = col_index.offsets[k] as usize;
71                builders[col_idx].push_value(
72                    &bucket.data,
73                    data_offset,
74                    big_endian,
75                    desc,
76                    indirect_arrays.as_mut(),
77                )?;
78            }
79        }
80    }
81
82    let mut result = Vec::with_capacity(col_descs.len());
83    for (desc, builder) in col_descs.iter().zip(builders) {
84        if builder.len() != nrrow {
85            return Err(TableReadError::Format(format!(
86                "column {:?} decoded {} rows but expected {}",
87                desc.col_name,
88                builder.len(),
89                nrrow
90            )));
91        }
92        result.push((desc.col_name.clone(), builder.finish()?));
93    }
94    Ok(result)
95}

97enum ColumnBuilder {
98    Float64(Vec<f64>),
99    String(Vec<String>),
100    ArrayFloat64 {
101        values: Vec<f64>,
102        shape: Option<Vec<i32>>,
103        rows: usize,
104    },
105}
106
107impl ColumnBuilder {
108    fn new(desc: &ColumnDesc) -> Result<Self, TableReadError> {
109        match (desc.data_type, desc.is_array) {
110            (CasacoreDataType::TpDouble, false) => Ok(Self::Float64(Vec::new())),
111            (CasacoreDataType::TpString, false) => Ok(Self::String(Vec::new())),
112            (CasacoreDataType::TpDouble, true) => Ok(Self::ArrayFloat64 {
113                values: Vec::new(),
114                shape: (!desc.shape.is_empty()).then(|| desc.shape.clone()),
115                rows: 0,
116            }),
117            _ => Err(TableReadError::UnsupportedColumn(format!(
118                "unsupported IncrementalStMan column {:?} type {:?} array={}",
119                desc.col_name, desc.data_type, desc.is_array
120            ))),
121        }
122    }
123
124    fn push_default(&mut self) -> Result<(), TableReadError> {
125        match self {
126            Self::Float64(values) => values.push(0.0),
127            Self::String(values) => values.push(String::new()),
128            Self::ArrayFloat64 {
129                values,
130                shape,
131                rows,
132            } => {
133                let nrelem = shape.as_deref().map(shape_nrelem).transpose()?.unwrap_or(0);
134                values.extend(std::iter::repeat_n(0.0, nrelem));
135                *rows += 1;
136            }
137        }
138        Ok(())
139    }
140
141    fn push_value(
142        &mut self,
143        data: &[u8],
144        offset: usize,
145        big_endian: bool,
146        desc: &ColumnDesc,
147        indirect_arrays: Option<&mut StManArrayFileReader>,
148    ) -> Result<(), TableReadError> {
149        match self {
150            Self::Float64(values) => values.push(read_f64_at(data, offset, big_endian)?),
151            Self::String(values) => values.push(read_string_at(data, offset, 1, big_endian)?),
152            Self::ArrayFloat64 {
153                values,
154                shape,
155                rows,
156            } => {
157                let cell = if desc.is_indirect_array() {
158                    let offset = read_i64_at(data, offset, big_endian)?;
159                    let reader = indirect_arrays.ok_or_else(|| {
160                        TableReadError::Format(format!(
161                            "indirect array column {:?} missing array-file reader",
162                            desc.col_name
163                        ))
164                    })?;
165                    let (cell_shape, cell_values) =
166                        reader.read_f64_array_at(offset)?.ok_or_else(|| {
167                            TableReadError::Format(format!(
168                                "indirect array column {:?} has undefined cell",
169                                desc.col_name
170                            ))
171                        })?;
172                    match shape {
173                        Some(existing) if existing != &cell_shape => {
174                            return Err(TableReadError::Format(format!(
175                                "array column {:?} shape mismatch: {:?} vs {:?}",
176                                desc.col_name, existing, cell_shape
177                            )));
178                        }
179                        Some(_) => {}
180                        None => *shape = Some(cell_shape),
181                    }
182                    cell_values
183                } else {
184                    let known_shape = shape.as_deref().ok_or_else(|| {
185                        TableReadError::Format(format!(
186                            "direct array column {:?} is missing shape metadata",
187                            desc.col_name
188                        ))
189                    })?;
190                    let nrelem = shape_nrelem(known_shape)?;
191                    let cell = read_array_f64_at(data, offset, nrelem, big_endian)?;
192                    if cell.len() != nrelem {
193                        return Err(TableReadError::Format(format!(
194                            "array column {:?} expected {nrelem} elements, found {}",
195                            desc.col_name,
196                            cell.len()
197                        )));
198                    }
199                    cell
200                };
201
202                if let Some(known_shape) = shape.as_deref() {
203                    let nrelem = shape_nrelem(known_shape)?;
204                    if cell.len() != nrelem {
205                        return Err(TableReadError::Format(format!(
206                            "array column {:?} expected {nrelem} elements, found {}",
207                            desc.col_name,
208                            cell.len()
209                        )));
210                    }
211                }
212                values.extend(cell);
213                *rows += 1;
214            }
215        }
216        Ok(())
217    }
218
219    fn len(&self) -> usize {
220        match self {
221            Self::Float64(values) => values.len(),
222            Self::String(values) => values.len(),
223            Self::ArrayFloat64 { rows, .. } => *rows,
224        }
225    }
226
227    fn finish(self) -> Result<ColumnData, TableReadError> {
228        match self {
229            Self::Float64(values) => Ok(ColumnData::Float64(values)),
230            Self::String(values) => Ok(ColumnData::String(values)),
231            Self::ArrayFloat64 { values, shape, .. } => {
232                let shape = shape.ok_or_else(|| {
233                    TableReadError::Format("array column is missing shape metadata".to_string())
234                })?;
235                Ok(ColumnData::ArrayFloat64 { values, shape })
236            }
237        }
238    }
239}

272struct IsmAipsIoBuf<'a> {
273    data: &'a [u8],
274    pos: usize,
275    order: ByteOrder,
276    level: usize,
277}
278
279impl<'a> IsmAipsIoBuf<'a> {
280    fn new(data: &'a [u8], order: ByteOrder) -> Self {
281        Self {
282            data,
283            pos: 0,
284            order,
285            level: 0,
286        }
287    }
288
289    fn read_bytes(&mut self, n: usize) -> Result<&'a [u8], TableReadError> {
290        if self.pos + n > self.data.len() {
291            return Err(TableReadError::Format(
292                "ISM AipsIO buffer underrun".to_string(),
293            ));
294        }
295        let slice = &self.data[self.pos..self.pos + n];
296        self.pos += n;
297        Ok(slice)
298    }
299
300    fn read_bool(&mut self) -> Result<bool, TableReadError> {
301        Ok(self.read_bytes(1)?[0] != 0)
302    }
303
304    fn read_u32(&mut self) -> Result<u32, TableReadError> {
305        let b = self.read_bytes(4)?;
306        Ok(match self.order {
307            ByteOrder::BigEndian => u32::from_be_bytes([b[0], b[1], b[2], b[3]]),
308            ByteOrder::LittleEndian => u32::from_le_bytes([b[0], b[1], b[2], b[3]]),
309        })
310    }
311
312    fn read_i32(&mut self) -> Result<i32, TableReadError> {
313        let b = self.read_bytes(4)?;
314        Ok(match self.order {
315            ByteOrder::BigEndian => i32::from_be_bytes([b[0], b[1], b[2], b[3]]),
316            ByteOrder::LittleEndian => i32::from_le_bytes([b[0], b[1], b[2], b[3]]),
317        })
318    }
319
320    fn read_u64(&mut self) -> Result<u64, TableReadError> {
321        let b = self.read_bytes(8)?;
322        Ok(match self.order {
323            ByteOrder::BigEndian => {
324                u64::from_be_bytes([b[0], b[1], b[2], b[3], b[4], b[5], b[6], b[7]])
325            }
326            ByteOrder::LittleEndian => {
327                u64::from_le_bytes([b[0], b[1], b[2], b[3], b[4], b[5], b[6], b[7]])
328            }
329        })
330    }
331
332    fn read_string(&mut self) -> Result<String, TableReadError> {
333        let len = self.read_u32()? as usize;
334        let b = self.read_bytes(len)?;
335        String::from_utf8(b.to_vec())
336            .map_err(|error| TableReadError::Format(format!("invalid UTF-8 in ISM: {error}")))
337    }
338
339    fn getstart(&mut self, expected_type: &str) -> Result<u32, TableReadError> {
340        if self.level == 0 {
341            let magic = self.read_u32()?;
342            if magic != AIPSIO_MAGIC {
343                return Err(TableReadError::Format(format!(
344                    "ISM AipsIO magic mismatch: expected 0x{AIPSIO_MAGIC:08x}, got 0x{magic:08x}"
345                )));
346            }
347        }
348        self.level += 1;
349        let _obj_len = self.read_u32()?;
350        let type_name = self.read_string()?;
351        if type_name != expected_type {
352            return Err(TableReadError::Format(format!(
353                "ISM AipsIO type mismatch: expected {expected_type:?}, got {type_name:?}"
354            )));
355        }
356        self.read_u32()
357    }
358
359    fn getend(&mut self) {
360        if self.level > 0 {
361            self.level -= 1;
362        }
363    }
364}

366impl BlockReaderExt for IsmAipsIoBuf<'_> {
367    fn read_block_u32(&mut self) -> Result<Vec<u32>, TableReadError> {
368        let _version = self.getstart("Block")?;
369        let count = self.read_u32()?;
370        let mut values = Vec::with_capacity(count as usize);
371        for _ in 0..count {
372            values.push(self.read_u32()?);
373        }
374        self.getend();
375        Ok(values)
376    }
377
378    fn read_block_u64(&mut self) -> Result<Vec<u64>, TableReadError> {
379        let _version = self.getstart("Block")?;
380        let count = self.read_u32()?;
381        let mut values = Vec::with_capacity(count as usize);
382        for _ in 0..count {
383            values.push(self.read_u64()?);
384        }
385        self.getend();
386        Ok(values)
387    }
388}

390fn detect_ism_aipsio_byte_order(data: &[u8]) -> Result<ByteOrder, TableReadError> {
391    if data.len() < 8 {
392        return Err(TableReadError::Format(
393            "ISM data too short for byte-order detection".to_string(),
394        ));
395    }
396    let be_len = u32::from_be_bytes([data[4], data[5], data[6], data[7]]);
397    let le_len = u32::from_le_bytes([data[4], data[5], data[6], data[7]]);
398    let be_ok = be_len > 0 && be_len < 4096;
399    let le_ok = le_len > 0 && le_len < 4096;
400    match (be_ok, le_ok) {
401        (true, false) => Ok(ByteOrder::BigEndian),
402        (false, true) => Ok(ByteOrder::LittleEndian),
403        (true, true) => Ok(ByteOrder::BigEndian),
404        (false, false) => Err(TableReadError::Format(format!(
405            "ISM: cannot detect byte order (be_len={be_len}, le_len={le_len})"
406        ))),
407    }
408}

410fn parse_ism_header(file: &mut (impl Read + Seek)) -> Result<IsmHeader, TableReadError> {
411    file.seek(SeekFrom::Start(0))?;
412    let mut header_buf = vec![0u8; ISM_HEADER_SIZE as usize];
413    file.read_exact(&mut header_buf)?;
414
415    let io_order = detect_ism_aipsio_byte_order(&header_buf)?;
416    let mut buf = IsmAipsIoBuf::new(&header_buf, io_order);
417    let version = buf.getstart("IncrementalStMan")?;
418    let big_endian = if version >= 5 { buf.read_bool()? } else { true };
419    let bucket_size = buf.read_u32()?;
420    let nr_buckets = buf.read_u32()?;
421    let _pers_cache_size = buf.read_u32()?;
422    let _uniq_nr = buf.read_u32()?;
423    let _nr_free_buckets = buf.read_u32()?;
424    let _first_free_bucket = buf.read_i32()?;
425    buf.getend();
426    Ok(IsmHeader {
427        bucket_size,
428        nr_buckets,
429        big_endian,
430        io_order,
431    })
432}

434fn parse_ism_dm_blob(data: &[u8]) -> Result<String, TableReadError> {
435    let mut io = AipsIoBuf::with_detected_order(data)?;
436    let _version = io.getstart("ISM")?;
437    let name = io.read_string()?;
438    io.getend()?;
439    Ok(name)
440}

442fn parse_ism_index(file: &mut File, header: &IsmHeader) -> Result<IsmIndex, TableReadError> {
443    let index_offset = ISM_HEADER_SIZE + (header.nr_buckets as u64) * (header.bucket_size as u64);
444    file.seek(SeekFrom::Start(index_offset))?;
445    let mut index_data = Vec::new();
446    file.read_to_end(&mut index_data)?;
447    if index_data.is_empty() {
448        return Ok(IsmIndex {
449            rows: vec![0],
450            bucket_nrs: vec![],
451        });
452    }
453
454    let mut buf = IsmAipsIoBuf::new(&index_data, header.io_order);
455    let version = buf.getstart("ISMIndex")?;
456    let _nused = buf.read_u32()?;
457    let rows = if version == 1 {
458        buf.read_block_u32()?.into_iter().map(u64::from).collect()
459    } else {
460        buf.read_block_u64()?
461    };
462    let bucket_nrs = buf.read_block_u32()?;
463    buf.getend();
464    Ok(IsmIndex { rows, bucket_nrs })
465}

467fn read_ism_bucket(
468    file: &mut File,
469    header: &IsmHeader,
470    bucket_nr: u32,
471) -> Result<Vec<u8>, TableReadError> {
472    let offset = ISM_HEADER_SIZE + (bucket_nr as u64) * (header.bucket_size as u64);
473    file.seek(SeekFrom::Start(offset))?;
474    let mut buf = vec![0u8; header.bucket_size as usize];
475    file.read_exact(&mut buf)?;
476    Ok(buf)
477}

479fn parse_ism_bucket(
480    raw: &[u8],
481    n_cols: usize,
482    big_endian: bool,
483) -> Result<IsmBucket, TableReadError> {
484    if raw.len() < 4 {
485        return Err(TableReadError::Format("ISM bucket too small".to_string()));
486    }
487    let raw_offset = if big_endian {
488        u32::from_be_bytes(raw[0..4].try_into().expect("u32"))
489    } else {
490        u32::from_le_bytes(raw[0..4].try_into().expect("u32"))
491    };
492    if raw_offset & 0x80000000 != 0 {
493        return Err(TableReadError::Format(
494            "ISM 64-bit row numbers are not supported".to_string(),
495        ));
496    }
497    let index_offset = (raw_offset & 0x7fffffff) as usize;
498    let data = raw[4..index_offset.min(raw.len())].to_vec();
499
500    let mut pos = index_offset;
501    let mut col_indices = Vec::with_capacity(n_cols);
502    for _ in 0..n_cols {
503        if pos + 4 > raw.len() {
504            break;
505        }
506        let n_values = read_u32(raw, pos, big_endian)? as usize;
507        pos += 4;
508
509        let mut row_nrs = Vec::with_capacity(n_values);
510        for _ in 0..n_values {
511            row_nrs.push(read_u32(raw, pos, big_endian)?);
512            pos += 4;
513        }
514
515        let mut offsets = Vec::with_capacity(n_values);
516        for _ in 0..n_values {
517            offsets.push(read_u32(raw, pos, big_endian)?);
518            pos += 4;
519        }
520
521        col_indices.push(IsmBucketColIndex { row_nrs, offsets });
522    }
523
524    Ok(IsmBucket { data, col_indices })
525}

527fn get_interval(col_index: &IsmBucketColIndex, rel_row: u32) -> usize {
528    let pos = col_index.row_nrs.partition_point(|&r| r <= rel_row);
529    if pos == 0 { 0 } else { pos - 1 }
530}

538fn shape_nrelem(shape: &[i32]) -> Result<usize, TableReadError> {
539    shape.iter().try_fold(1usize, |acc, &dim| {
540        let dim = usize::try_from(dim)
541            .map_err(|_| TableReadError::Format(format!("negative array dimension {dim}")))?;
542        acc.checked_mul(dim)
543            .ok_or_else(|| TableReadError::Format("array size overflow".to_string()))
544    })
545}

547fn read_f64_at(data: &[u8], offset: usize, big_endian: bool) -> Result<f64, TableReadError> {
548    let bytes = data
549        .get(offset..offset + 8)
550        .ok_or_else(|| TableReadError::Format("ISM scalar overrun".to_string()))?;
551    Ok(if big_endian {
552        f64::from_be_bytes(bytes.try_into().expect("f64"))
553    } else {
554        f64::from_le_bytes(bytes.try_into().expect("f64"))
555    })
556}

558fn read_i64_at(data: &[u8], offset: usize, big_endian: bool) -> Result<i64, TableReadError> {
559    let bytes = data
560        .get(offset..offset + 8)
561        .ok_or_else(|| TableReadError::Format("ISM i64 overrun".to_string()))?;
562    Ok(if big_endian {
563        i64::from_be_bytes(bytes.try_into().expect("i64"))
564    } else {
565        i64::from_le_bytes(bytes.try_into().expect("i64"))
566    })
567}

569fn read_string_at(
570    data: &[u8],
571    offset: usize,
572    nvalues: usize,
573    big_endian: bool,
574) -> Result<String, TableReadError> {
575    let strings = read_ism_string(
576        data.get(offset..)
577            .ok_or_else(|| TableReadError::Format("ISM string overrun".to_string()))?,
578        nvalues,
579        big_endian,
580    )?;
581    Ok(strings.into_iter().next().unwrap_or_default())
582}

584fn read_array_f64_at(
585    data: &[u8],
586    offset: usize,
587    nrelem: usize,
588    big_endian: bool,
589) -> Result<Vec<f64>, TableReadError> {
590    let bytes = data
591        .get(offset..offset + (nrelem * 8))
592        .ok_or_else(|| TableReadError::Format("ISM array overrun".to_string()))?;
593    let mut values = Vec::with_capacity(nrelem);
594    for chunk in bytes.chunks_exact(8) {
595        values.push(if big_endian {
596            f64::from_be_bytes(chunk.try_into().expect("f64"))
597        } else {
598            f64::from_le_bytes(chunk.try_into().expect("f64"))
599        });
600    }
601    Ok(values)
602}

604fn read_ism_string(
605    data: &[u8],
606    nvalues: usize,
607    big_endian: bool,
608) -> Result<Vec<String>, TableReadError> {
609    let total_length = read_u32(data, 0, big_endian)? as usize;
610    let mut pos = 4usize;
611    let mut strings = Vec::with_capacity(nvalues);
612
613    if nvalues == 1 {
614        let str_len = total_length.saturating_sub(4);
615        let bytes = data
616            .get(pos..pos + str_len)
617            .ok_or_else(|| TableReadError::Format("ISM string overrun".to_string()))?;
618        strings.push(String::from_utf8(bytes.to_vec()).map_err(|error| {
619            TableReadError::Format(format!("ISM string is not UTF-8: {error}"))
620        })?);
621        return Ok(strings);
622    }
623
624    for _ in 0..nvalues {
625        let str_len = read_u32(data, pos, big_endian)? as usize;
626        pos += 4;
627        let bytes = data
628            .get(pos..pos + str_len)
629            .ok_or_else(|| TableReadError::Format("ISM string array overrun".to_string()))?;
630        strings.push(String::from_utf8(bytes.to_vec()).map_err(|error| {
631            TableReadError::Format(format!("ISM string is not UTF-8: {error}"))
632        })?);
633        pos += str_len;
634    }
635    Ok(strings)
636}

638fn read_u32(data: &[u8], offset: usize, big_endian: bool) -> Result<u32, TableReadError> {
639    let bytes = data
640        .get(offset..offset + 4)
641        .ok_or_else(|| TableReadError::Format("ISM u32 overrun".to_string()))?;
642    Ok(if big_endian {
643        u32::from_be_bytes(bytes.try_into().expect("u32"))
644    } else {
645        u32::from_le_bytes(bytes.try_into().expect("u32"))
646    })
647}
