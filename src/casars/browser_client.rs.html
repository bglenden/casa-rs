19#[derive(Debug)]
20pub(crate) struct BrowserClient {
21    child: Arc<Mutex<Child>>,
22    stdin: Arc<Mutex<ChildStdin>>,
23    responses: Receiver<BrowserResponseEnvelope>,
24    stderr: Arc<Mutex<String>>,
25}
26
27impl BrowserClient {
28    pub(crate) fn spawn(command: &ResolvedCommand) -> Result<Self, String> {
29        let mut process = command.command();
30        process
31            .arg("--session")
32            .stdin(Stdio::piped())
33            .stdout(Stdio::piped())
34            .stderr(Stdio::piped());
35
36        let mut child = process
37            .spawn()
38            .map_err(|error| format!("spawn tablebrowser session: {error}"))?;
39        let stdin = child
40            .stdin
41            .take()
42            .ok_or_else(|| "tablebrowser session stdin was not captured".to_string())?;
43        let stdout = child
44            .stdout
45            .take()
46            .ok_or_else(|| "tablebrowser session stdout was not captured".to_string())?;
47        let stderr = child
48            .stderr
49            .take()
50            .ok_or_else(|| "tablebrowser session stderr was not captured".to_string())?;
51
52        let child = Arc::new(Mutex::new(child));
53        let stdin = Arc::new(Mutex::new(stdin));
54        let stderr_buffer = Arc::new(Mutex::new(String::new()));
55        let (tx, rx) = mpsc::channel();
56
57        thread::spawn(move || {
58            let reader = BufReader::new(stdout);
59            for line_result in reader.lines() {
60                let Ok(line) = line_result else {
61                    break;
62                };
63                if line.trim().is_empty() {
64                    continue;
65                }
66                let response = serde_json::from_str::<BrowserResponseEnvelope>(&line)
67                    .unwrap_or_else(|error| {
68                        BrowserResponseEnvelope::error(
69                            "invalid_response",
70                            format!("parse browser response: {error}"),
71                        )
72                    });
73                if tx.send(response).is_err() {
74                    break;
75                }
76            }
77        });
78
79        let stderr_target = Arc::clone(&stderr_buffer);
80        thread::spawn(move || {
81            let reader = BufReader::new(stderr);
82            for line_result in reader.lines() {
83                let Ok(line) = line_result else {
84                    break;
85                };
86                if let Ok(mut stderr) = stderr_target.lock() {
87                    stderr.push_str(&line);
88                    stderr.push('\n');
89                }
90            }
91        });
92
93        Ok(Self {
94            child,
95            stdin,
96            responses: rx,
97            stderr: stderr_buffer,
98        })
99    }

101    pub(crate) fn request(&self, command: BrowserCommand) -> Result<BrowserSnapshot, String> {
102        self.request_with_timeout(command, request_timeout())
103    }
104
105    pub(crate) fn request_startup(
106        &self,
107        command: BrowserCommand,
108    ) -> Result<BrowserSnapshot, String> {
109        self.request_with_timeout(command, startup_timeout())
110    }
111
112    fn request_with_timeout(
113        &self,
114        command: BrowserCommand,
115        timeout: Duration,
116    ) -> Result<BrowserSnapshot, String> {
117        let payload = serde_json::to_string(&BrowserRequestEnvelope::new(command))
118            .map_err(|error| format!("serialize browser request: {error}"))?;
119        {
120            let mut stdin = self
121                .stdin
122                .lock()
123                .map_err(|_| "failed to acquire browser stdin lock".to_string())?;
124            stdin
125                .write_all(payload.as_bytes())
126                .and_then(|_| stdin.write_all(b"\n"))
127                .and_then(|_| stdin.flush())
128                .map_err(|error| format!("write browser request: {error}"))?;
129        }
130
131        let response = self
132            .responses
133            .recv_timeout(timeout)
134            .map_err(|error| match error {
135                RecvTimeoutError::Timeout => {
136                    let _ = self.terminate_and_wait();
137                    format_browser_failure(
138                        "timed out waiting for browser response",
139                        self.stderr_text(),
140                        None,
141                    )
142                }
143                RecvTimeoutError::Disconnected => {
144                    let status = self.reap_exit_status();
145                    format_browser_failure(
146                        "tablebrowser session exited",
147                        self.stderr_text(),
148                        status,
149                    )
150                }
151            })?;
152
153        match response.response {
154            BrowserResponse::Snapshot(snapshot) => Ok(*snapshot),
155            BrowserResponse::Error(error) => Err(format!("{}: {}", error.code, error.message)),
156        }
157    }
158
159    pub(crate) fn cancel(&self) -> Result<(), String> {
160        self.terminate_and_wait().map(|_| ())
161    }
162
163    pub(crate) fn stderr_text(&self) -> String {
164        self.stderr
165            .lock()
166            .map(|stderr| stderr.clone())
167            .unwrap_or_default()
168    }
169
170    fn reap_exit_status(&self) -> Option<ExitStatus> {
171        let mut child = self.child.lock().ok()?;
172        child.try_wait().ok().flatten()
173    }
174
175    fn terminate_and_wait(&self) -> Result<Option<ExitStatus>, String> {
176        let mut child = self
177            .child
178            .lock()
179            .map_err(|_| "failed to acquire browser child lock".to_string())?;
180        if let Some(status) = child
181            .try_wait()
182            .map_err(|error| format!("poll browser session: {error}"))?
183        {
184            return Ok(Some(status));
185        }
186        child
187            .kill()
188            .map_err(|error| format!("terminate browser session: {error}"))?;
189        child
190            .wait()
191            .map(Some)
192            .map_err(|error| format!("wait for browser session: {error}"))
193    }
194}
195
196impl Drop for BrowserClient {
197    fn drop(&mut self) {
198        let _ = self.terminate_and_wait();
199    }
200}
201
202fn request_timeout() -> Duration {
203    duration_from_env(
204        "CASARS_BROWSER_REQUEST_TIMEOUT_MS",
205        DEFAULT_REQUEST_TIMEOUT_MS,
206    )
207}
208
209fn startup_timeout() -> Duration {
210    duration_from_env(
211        "CASARS_BROWSER_STARTUP_TIMEOUT_MS",
212        DEFAULT_STARTUP_TIMEOUT_MS,
213    )
214}
215
216fn duration_from_env(name: &str, default_ms: u64) -> Duration {
217    let millis = std::env::var(name)
218        .ok()
219        .and_then(|value| value.parse::<u64>().ok())
220        .filter(|value| *value > 0)
221        .unwrap_or(default_ms);
222    Duration::from_millis(millis)
223}
224
225fn format_browser_failure(prefix: &str, stderr: String, status: Option<ExitStatus>) -> String {
226    let mut message = match status {
227        Some(status) => format!("{prefix} with {status}"),
228        None => prefix.to_string(),
229    };
230    if !stderr.trim().is_empty() {
231        message.push_str(": ");
232        message.push_str(stderr.trim());
233    }
234    message
235}

252    static ENV_LOCK: Mutex<()> = Mutex::new(());

254    #[test]
255    fn timeout_terminates_session_before_a_late_reply_can_be_reused() {
256        let _guard = ENV_LOCK.lock().expect("env lock");
257        let temp = tempdir().expect("tempdir");
258        let script = write_slow_browser_script(temp.path(), 200);
259        unsafe {
260            std::env::set_var("CASARS_BROWSER_REQUEST_TIMEOUT_MS", "50");
261        }
262
263        let client =
264            BrowserClient::spawn(&ResolvedCommand::direct(script)).expect("spawn browser client");
265
266        let error = client
267            .request(BrowserCommand::GetSnapshot { viewport: None })
268            .expect_err("request should time out");
269        assert!(error.contains("timed out waiting for browser response"));
270
271        thread::sleep(Duration::from_millis(250));
272        assert!(
273            client
274                .request(BrowserCommand::GetSnapshot { viewport: None })
275                .is_err(),
276            "late reply should not be reused after timeout"
277        );
278
279        unsafe {
280            std::env::remove_var("CASARS_BROWSER_REQUEST_TIMEOUT_MS");
281        }
282    }

284    fn write_slow_browser_script(root: &Path, delay_ms: u64) -> PathBuf {
285        let path = root.join("slow-browser.sh");
286        let response = serde_json::to_string(&BrowserResponseEnvelope::snapshot(BrowserSnapshot {
287            capabilities: casacore_tablebrowser_protocol::BrowserCapabilities { editable: false },
288            view: casacore_tablebrowser_protocol::BrowserView::Overview,
289            focus: casacore_tablebrowser_protocol::BrowserFocus::Main,
290            table_path: "/tmp/fake.ms".to_string(),
291            breadcrumb: Vec::new(),
292            viewport: BrowserViewport::new(80, 24),
293            status_line: "ok".to_string(),
294            content_lines: vec!["Overview".to_string()],
295            selected_address: None,
296            inspector: None,
297        }))
298        .expect("serialize snapshot");
299        let script = format!(
300            "#!/bin/sh\nwhile IFS= read -r line; do\n  sleep {}\n  printf '%s\\n' '{}'\ndone\n",
301            delay_ms as f64 / 1000.0,
302            response
303        );
304        fs::write(&path, script).expect("write script");
305        let mut permissions = fs::metadata(&path).expect("metadata").permissions();
306        permissions.set_mode(0o755);
307        fs::set_permissions(&path, permissions).expect("chmod");
308        path
309    }
