2use std::io::{BufRead, BufReader, Write};
3use std::process::{Child, ChildStdin, Stdio};
4use std::sync::mpsc::{self, Receiver};
5use std::sync::{Arc, Mutex};
6use std::thread;
7use std::time::Duration;
8
9use casacore_tablebrowser_protocol::{
10    BrowserCommand, BrowserRequestEnvelope, BrowserResponse, BrowserResponseEnvelope,
11    BrowserSnapshot,
12};
13
14use crate::registry::ResolvedCommand;
15
16const REQUEST_TIMEOUT: Duration = Duration::from_secs(2);
17
18#[derive(Debug)]
19pub(crate) struct BrowserClient {
20    child: Arc<Mutex<Child>>,
21    stdin: Arc<Mutex<ChildStdin>>,
22    responses: Receiver<BrowserResponseEnvelope>,
23    stderr: Arc<Mutex<String>>,
24}
25
26impl BrowserClient {
27    pub(crate) fn spawn(command: &ResolvedCommand) -> Result<Self, String> {
28        let mut process = command.command();
29        process
30            .arg("--session")
31            .stdin(Stdio::piped())
32            .stdout(Stdio::piped())
33            .stderr(Stdio::piped());
34
35        let mut child = process
36            .spawn()
37            .map_err(|error| format!("spawn tablebrowser session: {error}"))?;
38        let stdin = child
39            .stdin
40            .take()
41            .ok_or_else(|| "tablebrowser session stdin was not captured".to_string())?;
42        let stdout = child
43            .stdout
44            .take()
45            .ok_or_else(|| "tablebrowser session stdout was not captured".to_string())?;
46        let stderr = child
47            .stderr
48            .take()
49            .ok_or_else(|| "tablebrowser session stderr was not captured".to_string())?;
50
51        let child = Arc::new(Mutex::new(child));
52        let stdin = Arc::new(Mutex::new(stdin));
53        let stderr_buffer = Arc::new(Mutex::new(String::new()));
54        let (tx, rx) = mpsc::channel();
55
56        thread::spawn(move || {
57            let reader = BufReader::new(stdout);
58            for line_result in reader.lines() {
59                let Ok(line) = line_result else {
60                    break;
61                };
62                if line.trim().is_empty() {
63                    continue;
64                }
65                let response = serde_json::from_str::<BrowserResponseEnvelope>(&line)
66                    .unwrap_or_else(|error| {
67                        BrowserResponseEnvelope::error(
68                            "invalid_response",
69                            format!("parse browser response: {error}"),
70                        )
71                    });
72                if tx.send(response).is_err() {
73                    break;
74                }
75            }
76        });
77
78        let stderr_target = Arc::clone(&stderr_buffer);
79        thread::spawn(move || {
80            let reader = BufReader::new(stderr);
81            for line_result in reader.lines() {
82                let Ok(line) = line_result else {
83                    break;
84                };
85                if let Ok(mut stderr) = stderr_target.lock() {
86                    stderr.push_str(&line);
87                    stderr.push('\n');
88                }
89            }
90        });
91
92        Ok(Self {
93            child,
94            stdin,
95            responses: rx,
96            stderr: stderr_buffer,
97        })
98    }
99
100    pub(crate) fn request(&self, command: BrowserCommand) -> Result<BrowserSnapshot, String> {
101        let payload = serde_json::to_string(&BrowserRequestEnvelope::new(command))
102            .map_err(|error| format!("serialize browser request: {error}"))?;
103        {
104            let mut stdin = self
105                .stdin
106                .lock()
107                .map_err(|_| "failed to acquire browser stdin lock".to_string())?;
108            stdin
109                .write_all(payload.as_bytes())
110                .and_then(|_| stdin.write_all(b"\n"))
111                .and_then(|_| stdin.flush())
112                .map_err(|error| format!("write browser request: {error}"))?;
113        }
114
115        let response = self
116            .responses
117            .recv_timeout(REQUEST_TIMEOUT)
118            .map_err(|error| {
119                let stderr = self.stderr_text();
120                let exit = self
121                    .child
122                    .lock()
123                    .ok()
124                    .and_then(|mut child| child.try_wait().ok().flatten());
125                match exit {
126                    Some(status) => format!(
127                        "tablebrowser session exited with {}{}",
128                        status,
129                        if stderr.trim().is_empty() {
130                            String::new()
131                        } else {
132                            format!(": {}", stderr.trim())
133                        }
134                    ),
135                    None => format!(
136                        "timed out waiting for browser response ({error}){}",
137                        if stderr.trim().is_empty() {
138                            String::new()
139                        } else {
140                            format!(": {}", stderr.trim())
141                        }
142                    ),
143                }
144            })?;
145
146        match response.response {
147            BrowserResponse::Snapshot(snapshot) => Ok(*snapshot),
148            BrowserResponse::Error(error) => Err(format!("{}: {}", error.code, error.message)),
149        }
150    }
151
152    pub(crate) fn cancel(&self) -> Result<(), String> {
153        self.child
154            .lock()
155            .map_err(|_| "failed to acquire browser child lock".to_string())?
156            .kill()
157            .map_err(|error| format!("terminate browser session: {error}"))
158    }
159
160    pub(crate) fn stderr_text(&self) -> String {
161        self.stderr
162            .lock()
163            .map(|stderr| stderr.clone())
164            .unwrap_or_default()
165    }
166}
