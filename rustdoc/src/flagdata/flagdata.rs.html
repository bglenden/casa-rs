4use std::env;

14fn main() {
15    if let Err(error) = run() {
16        eprintln!("Error: {error}");
17        process::exit(1);
18    }
19}
20
21fn run() -> Result<(), String> {
22    let args = env::args().skip(1).collect::<Vec<_>>();
23    if args.iter().any(|arg| arg == "--help" || arg == "-h") {
24        println!("{}", usage());
25        return Ok(());
26    }
27    let (vis, request) = parse_args(&args)?;
28    let report = flagdata_path(vis, &request).map_err(|error| error.to_string())?;
29    println!(
30        "{}",
31        serde_json::to_string_pretty(&report).map_err(|error| error.to_string())?
32    );
33    Ok(())
34}

36fn parse_args(args: &[String]) -> Result<(PathBuf, FlagDataRequest), String> {
37    let mut vis = None;
38    let mut request = FlagDataRequest::default();
39    let mut selection = MsSelection::new();
40    let mut index = 0usize;
41    while index < args.len() {
42        match args[index].as_str() {
43            "--vis" | "--ms" => {
44                index += 1;
45                vis = Some(PathBuf::from(args.get(index).ok_or_else(usage)?));
46            }
47            "--mode" => {
48                index += 1;
49                request.mode = parse_mode(args.get(index).ok_or_else(usage)?)?;
50            }
51            "--spw" => {
52                index += 1;
53                let spw = args.get(index).ok_or_else(usage)?.clone();
54                selection = selection
55                    .spw_selector(&spw)
56                    .map_err(|error| error.to_string())?;
57                request.spw = Some(spw);
58            }
59            "--field" => {
60                index += 1;
61                selection = selection.field(
62                    &parse_numeric_id_selector(args.get(index).ok_or_else(usage)?, "field")
63                        .map_err(|error| error.to_string())?,
64                );
65            }
66            "--scan" => {
67                index += 1;
68                selection = selection.scan(
69                    &parse_numeric_id_selector(args.get(index).ok_or_else(usage)?, "scan")
70                        .map_err(|error| error.to_string())?,
71                );
72            }
73            "--antenna" => {
74                index += 1;
75                let value = args.get(index).ok_or_else(usage)?;
76                selection = apply_antenna_selection(selection, value)?;
77            }
78            "--datacolumn" => {
79                index += 1;
80                request.data_column = parse_data_column(args.get(index).ok_or_else(usage)?)?;
81            }
82            "--action" => {
83                index += 1;
84                request.action = parse_action(args.get(index).ok_or_else(usage)?)?;
85            }
86            "--clipzeros" => request.clipzeros = true,
87            "--quackinterval" => {
88                index += 1;
89                request.quackinterval =
90                    parse_f64(args.get(index).ok_or_else(usage)?, "quackinterval")?;
91            }
92            "--quackmode" => {
93                index += 1;
94                request.quackmode = parse_quackmode(args.get(index).ok_or_else(usage)?)?;
95            }
96            "--timecutoff" => {
97                index += 1;
98                request.timecutoff = parse_f64(args.get(index).ok_or_else(usage)?, "timecutoff")?;
99            }
100            "--freqcutoff" => {
101                index += 1;
102                request.freqcutoff = parse_f64(args.get(index).ok_or_else(usage)?, "freqcutoff")?;
103            }
104            "--timedev" => {
105                index += 1;
106                request.timedev = Some(parse_f64(args.get(index).ok_or_else(usage)?, "timedev")?);
107            }
108            "--freqdev" => {
109                index += 1;
110                request.freqdev = Some(parse_f64(args.get(index).ok_or_else(usage)?, "freqdev")?);
111            }
112            "--timedevscale" => {
113                index += 1;
114                request.timedevscale =
115                    parse_f64(args.get(index).ok_or_else(usage)?, "timedevscale")?;
116            }
117            "--freqdevscale" => {
118                index += 1;
119                request.freqdevscale =
120                    parse_f64(args.get(index).ok_or_else(usage)?, "freqdevscale")?;
121            }
122            "--spectralmax" => {
123                index += 1;
124                request.spectralmax = parse_f64(args.get(index).ok_or_else(usage)?, "spectralmax")?;
125            }
126            "--spectralmin" => {
127                index += 1;
128                request.spectralmin = parse_f64(args.get(index).ok_or_else(usage)?, "spectralmin")?;
129            }
130            "--extendflags" => request.extendflags = true,
131            "--no-extendflags" | "--extendflags=false" => request.extendflags = false,
132            "--extendpols" => request.extendpols = true,
133            "--growtime" => {
134                index += 1;
135                request.growtime = parse_f64(args.get(index).ok_or_else(usage)?, "growtime")?;
136            }
137            "--growfreq" => {
138                index += 1;
139                request.growfreq = parse_f64(args.get(index).ok_or_else(usage)?, "growfreq")?;
140            }
141            "--flagbackup" => request.flagbackup = true,
142            "--no-flagbackup" | "--flagbackup=false" => request.flagbackup = false,
143            other => return Err(format!("unknown argument {other:?}\n{}", usage())),
144        }
145        index += 1;
146    }
147    request.selection = selection;
148    Ok((vis.ok_or_else(usage)?, request))
149}

151fn apply_antenna_selection(selection: MsSelection, value: &str) -> Result<MsSelection, String> {
152    let parts = value
153        .split(',')
154        .map(str::trim)
155        .filter(|part| !part.is_empty())
156        .collect::<Vec<_>>();
157    if parts.iter().all(|part| part.parse::<i32>().is_ok()) {
158        Ok(selection.antenna(
159            &parts
160                .iter()
161                .map(|part| part.parse::<i32>().expect("checked int"))
162                .collect::<Vec<_>>(),
163        ))
164    } else {
165        Ok(selection.antenna_name(&parts))
166    }
167}

169fn parse_mode(value: &str) -> Result<FlagDataMode, String> {
170    match value.to_ascii_lowercase().as_str() {
171        "manual" => Ok(FlagDataMode::Manual),
172        "clip" => Ok(FlagDataMode::Clip),
173        "quack" => Ok(FlagDataMode::Quack),
174        "tfcrop" => Ok(FlagDataMode::Tfcrop),
175        "rflag" => Ok(FlagDataMode::Rflag),
176        "extend" => Ok(FlagDataMode::Extend),
177        "summary" => Ok(FlagDataMode::Summary),
178        other => Err(format!("unsupported mode {other:?}")),
179    }
180}
181
182fn parse_action(value: &str) -> Result<FlagDataAction, String> {
183    match value.to_ascii_lowercase().as_str() {
184        "flag" | "apply" => Ok(FlagDataAction::Flag),
185        "unflag" => Ok(FlagDataAction::Unflag),
186        other => Err(format!("unsupported action {other:?}")),
187    }
188}
189
190fn parse_data_column(value: &str) -> Result<FlagDataColumn, String> {
191    match value.to_ascii_lowercase().as_str() {
192        "data" => Ok(FlagDataColumn::Data),
193        "corrected" | "corrected_data" => Ok(FlagDataColumn::CorrectedData),
194        other => Err(format!("unsupported datacolumn {other:?}")),
195    }
196}
197
198fn parse_quackmode(value: &str) -> Result<QuackMode, String> {
199    match value.to_ascii_lowercase().as_str() {
200        "beg" => Ok(QuackMode::Beg),
201        "end" => Ok(QuackMode::End),
202        other => Err(format!("unsupported quackmode {other:?}")),
203    }
204}
205
206fn parse_f64(value: &str, label: &str) -> Result<f64, String> {
207    value
208        .parse::<f64>()
209        .map_err(|error| format!("invalid {label} {value:?}: {error}"))
210}
