48/// Errors arising from table storage operations (I/O, format, schema).
49///
50/// C++ equivalent: various exceptions in `DataManager.h` and `Table.h`.
51#[derive(Debug, Error)]
52pub enum StorageError {
53    /// The table directory does not exist on the filesystem.
54    #[error("table path does not exist: {0}")]
55    MissingPath(PathBuf),
56    /// The `table.dat` control file is absent.
57    #[error("table control file is missing: {0}")]
58    MissingControlFile(PathBuf),
59    /// A required `table.f*` data file is absent.
60    #[error("table data file is missing: {0}")]
61    MissingDataFile(PathBuf),
62    /// On-disk format does not match the expected version or layout.
63    #[error("format mismatch: {0}")]
64    FormatMismatch(String),
65    /// Column schema validation failed.
66    #[error("schema error: {0}")]
67    Schema(String),
68    /// The data manager type is not recognised or not yet implemented.
69    #[error("unsupported data manager: {0}")]
70    UnsupportedDataManager(String),
71    /// An underlying I/O error.
72    #[error("i/o error: {0}")]
73    Io(#[from] std::io::Error),
74    /// An error from the AipsIO serialisation layer.
75    #[error("aipsio error: {0}")]
76    AipsIo(#[from] casacore_aipsio::AipsIoObjectError),
77}
78
79impl From<SchemaError> for StorageError {
80    fn from(value: SchemaError) -> Self {
81        Self::Schema(value.to_string())
82    }
83}

85/// C++ `TableInfo` metadata stored in `table.info`.
86///
87/// Contains the logical table type (e.g. `"MeasurementSet"`) and subtype
88/// (e.g. `"UVFITS"`) written as plain-text key-value pairs.
89///
90/// # C++ equivalent
91///
92/// `casacore::TableInfo` — persisted via `TableInfo::flush()` / `readBack()`.
93#[derive(Debug, Clone, Default, PartialEq, Eq)]
94pub struct TableInfo {
95    /// Logical table type (e.g. `"MeasurementSet"`). Empty if unset.
96    pub table_type: String,
97    /// Table subtype (e.g. `"UVFITS"`). Empty if unset.
98    pub sub_type: String,
99}
100
101impl TableInfo {
102    /// Parse a `table.info` file from its text contents.
103    pub fn parse(contents: &str) -> Self {
104        let mut table_type = String::new();
105        let mut sub_type = String::new();
106        for line in contents.lines() {
107            if let Some(rest) = line.strip_prefix("Type = ") {
108                table_type = rest.to_string();
109            } else if let Some(rest) = line.strip_prefix("SubType = ") {
110                sub_type = rest.to_string();
111            }
112        }
113        Self {
114            table_type,
115            sub_type,
116        }
117    }
118}
119
120impl std::fmt::Display for TableInfo {
121    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
122        write!(
123            f,
124            "Type = {}\nSubType = {}\n",
125            self.table_type, self.sub_type
126        )
127    }
128}

130/// Metadata for one data manager instance.
131///
132/// Each data manager manages one or more columns. The sequence number
133/// is unique within a table and identifies the on-disk data file
134/// (`table.f<seq_nr>`).
135///
136/// # C++ equivalent
137///
138/// `DataManager::dataManagerName()`, `DataManager::dataManagerType()`,
139/// `DataManager::sequenceNr()`.
140#[derive(Debug, Clone, PartialEq, Eq)]
141pub struct DataManagerInfo {
142    /// The data manager type name (e.g. `"StManAipsIO"`, `"StandardStMan"`).
143    pub dm_type: String,
144    /// Unique sequence number within the table.
145    pub seq_nr: u32,
146    /// Column names managed by this data manager.
147    pub columns: Vec<String>,
148}

180fn filter_rows_for_save(
181    rows: &[RecordValue],
182    undefined_cells: &[HashSet<String>],
183) -> Vec<RecordValue> {
184    rows.iter()
185        .enumerate()
186        .map(|(idx, row)| {
187            let undef = undefined_cells.get(idx);
188            let fields: Vec<_> = row
189                .fields()
190                .iter()
191                .filter(|f| undef.is_none_or(|u| !u.contains(&f.name)))
192                .cloned()
193                .collect();
194            RecordValue::new(fields)
195        })
196        .collect()
197}

201/// Reads `table.dat` once, then routes each DM's columns to the appropriate

203#[derive(Debug, Default, Clone, Copy)]

206impl StorageManager for CompositeStorage {

208        self.load_with_row_hint(table_path, None)
209    }

211    fn save(
212        &self,
213        table_path: &Path,
214        snapshot: &StorageSnapshot,
215        dm_kind: crate::table::DataManagerKind,
216        big_endian: bool,
217        tile_shape: Option<&[usize]>,
218    ) -> Result<(), StorageError> {
219        use crate::table::DataManagerKind;
220
221        // Clean up old data files before re-saving to prevent stale data
222        // from a previous save with a different storage manager.
223        if table_path.is_dir() {
224            if let Ok(entries) = fs::read_dir(table_path) {
225                for entry in entries.flatten() {
226                    let name = entry.file_name();
227                    let name_str = name.to_string_lossy();
228                    if name_str.starts_with("table.f") {
229                        let _ = fs::remove_file(entry.path());
230                    }
231                }
232            }
233        }
234
235        fs::create_dir_all(table_path)?;
236
237        let schema = snapshot.schema.as_ref().ok_or_else(|| {
238            StorageError::FormatMismatch("cannot save without schema".to_string())
239        })?;
240
241        let filtered_rows = filter_rows_for_save(&snapshot.rows, &snapshot.undefined_cells);
242        let nrrow = filtered_rows.len() as u64;
243        let data_path = table_path.join(format!("{}0", TABLE_DATA_FILE_PREFIX));
244        let has_virtual = !snapshot.virtual_bindings.is_empty();
245
246        // When saving with virtual bindings, build rows that exclude virtual columns.
247        let stored_rows: Vec<RecordValue>;
248        let rows_for_data = if has_virtual {
249            stored_rows = snapshot
250                .rows
251                .iter()
252                .zip(&snapshot.undefined_cells)
253                .map(|(row, undef)| {
254                    let stored_fields: Vec<_> = row
255                        .fields()
256                        .iter()
257                        .filter(|f| {
258                            !snapshot.virtual_columns.contains(&f.name) && !undef.contains(&f.name)
259                        })
260                        .cloned()
261                        .collect();
262                    RecordValue::new(stored_fields)
263                })
264                .collect();
265            &stored_rows
266        } else {
267            &filtered_rows
268        };
269
270        let dm_type_name;
271        let dm_data;
272        match dm_kind {
273            DataManagerKind::StManAipsIO => {
274                dm_type_name = "StManAipsIO".to_string();
275                dm_data = Vec::new();
276                let table_dat = if has_virtual {
277                    TableDatContents::from_snapshot_with_virtual(
278                        schema,
279                        &snapshot.keywords,
280                        &snapshot.column_keywords,
281                        nrrow,
282                        &dm_type_name,
283                        &dm_data,
284                        big_endian,
285                        &snapshot.virtual_bindings,
286                        table_path,
287                    )
288                } else {
289                    TableDatContents::from_snapshot(
290                        schema,
291                        &snapshot.keywords,
292                        &snapshot.column_keywords,
293                        nrrow,
294                        &dm_type_name,
295                        &dm_data,
296                        big_endian,
297                    )
298                };
299                let control_path = table_path.join(TABLE_CONTROL_FILE);
300                write_table_dat(&control_path, &table_dat)?;
301                // Filter col_descs to stored-only columns for the data file.
302                let stored_col_descs: Vec<_> = table_dat
303                    .table_desc
304                    .columns
305                    .iter()
306                    .filter(|c| !snapshot.virtual_columns.contains(&c.col_name))
307                    .cloned()
308                    .collect();
309                // StManAipsIO always uses canonical (big-endian) AipsIO.
310                write_stman_file(
311                    &data_path,
312                    &stored_col_descs,
313                    rows_for_data,
314                    ByteOrder::BigEndian,
315                )?;
316            }
317            DataManagerKind::StandardStMan => {
318                dm_type_name = "StandardStMan".to_string();
319                // Write SSM data file first (it returns the DM blob).
320                // Use stored-only columns for the data file.
321                let table_dat_tmp = if has_virtual {
322                    TableDatContents::from_snapshot_with_virtual(
323                        schema,
324                        &snapshot.keywords,
325                        &snapshot.column_keywords,
326                        nrrow,
327                        "StandardStMan",
328                        &[],
329                        big_endian,
330                        &snapshot.virtual_bindings,
331                        table_path,
332                    )
333                } else {
334                    TableDatContents::from_snapshot(
335                        schema,
336                        &snapshot.keywords,
337                        &snapshot.column_keywords,
338                        nrrow,
339                        "StandardStMan",
340                        &[],
341                        big_endian,
342                    )
343                };
344                let stored_col_descs: Vec<_> = table_dat_tmp
345                    .table_desc
346                    .columns
347                    .iter()
348                    .filter(|c| !snapshot.virtual_columns.contains(&c.col_name))
349                    .cloned()
350                    .collect();
351                dm_data = write_ssm_file(&data_path, &stored_col_descs, rows_for_data, big_endian)?;
352                // Re-create table_dat with the actual DM blob.
353                let table_dat = if has_virtual {
354                    TableDatContents::from_snapshot_with_virtual(
355                        schema,
356                        &snapshot.keywords,
357                        &snapshot.column_keywords,
358                        nrrow,
359                        &dm_type_name,
360                        &dm_data,
361                        big_endian,
362                        &snapshot.virtual_bindings,
363                        table_path,
364                    )
365                } else {
366                    TableDatContents::from_snapshot(
367                        schema,
368                        &snapshot.keywords,
369                        &snapshot.column_keywords,
370                        nrrow,
371                        &dm_type_name,
372                        &dm_data,
373                        big_endian,
374                    )
375                };
376                let control_path = table_path.join(TABLE_CONTROL_FILE);
377                write_table_dat(&control_path, &table_dat)?;
378            }
379            DataManagerKind::IncrementalStMan => {
380                dm_type_name = "IncrementalStMan".to_string();
381                // Write ISM data file first (it returns the DM blob)
382                let table_dat_tmp = TableDatContents::from_snapshot(
383                    schema,
384                    &snapshot.keywords,
385                    &snapshot.column_keywords,
386                    nrrow,
387                    "IncrementalStMan",
388                    &[],
389                    big_endian,
390                );
391                dm_data = write_ism_file(
392                    &data_path,
393                    &table_dat_tmp.table_desc.columns,
394                    &snapshot.rows,
395                    big_endian,
396                )?;
397                // Re-create table_dat with the actual DM blob
398                let table_dat = TableDatContents::from_snapshot(
399                    schema,
400                    &snapshot.keywords,
401                    &snapshot.column_keywords,
402                    nrrow,
403                    &dm_type_name,
404                    &dm_data,
405                    big_endian,
406                );
407                let control_path = table_path.join(TABLE_CONTROL_FILE);
408                write_table_dat(&control_path, &table_dat)?;
409            }
410            DataManagerKind::TiledColumnStMan
411            | DataManagerKind::TiledShapeStMan
412            | DataManagerKind::TiledCellStMan
413            | DataManagerKind::TiledDataStMan => {
414                dm_type_name = match dm_kind {
415                    DataManagerKind::TiledColumnStMan => "TiledColumnStMan",
416                    DataManagerKind::TiledShapeStMan => "TiledShapeStMan",
417                    DataManagerKind::TiledCellStMan => "TiledCellStMan",
418                    DataManagerKind::TiledDataStMan => "TiledDataStMan",
419                    _ => unreachable!(),
420                }
421                .to_string();
422                dm_data = Vec::new(); // Tiled managers write empty DM blobs.
423                let mut table_dat = TableDatContents::from_snapshot(
424                    schema,
425                    &snapshot.keywords,
426                    &snapshot.column_keywords,
427                    nrrow,
428                    &dm_type_name,
429                    &dm_data,
430                    big_endian,
431                );
432                let dm_group_name = table_dat
433                    .table_desc
434                    .columns
435                    .first()
436                    .map(|c| c.col_name.clone())
437                    .unwrap_or_default();
438                for desc in &mut table_dat.table_desc.columns {
439                    desc.data_manager_type = dm_type_name.clone();
440                    desc.data_manager_group = dm_group_name.clone();
441                }
442                let control_path = table_path.join(TABLE_CONTROL_FILE);
443                write_table_dat(&control_path, &table_dat)?;
444                // Write tiled data to table.f0 header + table.f0_TSM* data files.
445                // Use the first column name as the hypercolumn/DM name.
446                let first_col_name = table_dat
447                    .table_desc
448                    .columns
449                    .first()
450                    .map(|c| c.col_name.as_str())
451                    .unwrap_or("");
452                tiled_stman::save_tiled_columns(
453                    table_path,
454                    0, // dm_seq_nr
455                    &dm_type_name,
456                    &table_dat.table_desc.columns,
457                    &snapshot.rows,
458                    big_endian,
459                    tile_shape,
460                    first_col_name,
461                )?;
462            }
463        }
464
465        let info_path = table_path.join(TABLE_INFO_FILE);
466        fs::write(&info_path, snapshot.table_info.to_string())?;
467
468        Ok(())
469    }
470}

472impl CompositeStorage {
473    pub(crate) fn load_with_row_hint(
474        &self,
475        table_path: &Path,
476        row_hint: Option<u64>,
477    ) -> Result<StorageSnapshot, StorageError> {
478        if !table_path.exists() {
479            return Err(StorageError::MissingPath(table_path.to_path_buf()));
480        }
481
482        let control_path = table_path.join(TABLE_CONTROL_FILE);
483        if !control_path.is_file() {
484            return Err(StorageError::MissingControlFile(control_path));
485        }
486
487        match read_table_dat_dispatch(&control_path)? {
488            TableDatResult::Plain(table_dat) => {
489                self.load_plain_table(table_path, &table_dat, row_hint)
490            }
491            TableDatResult::Ref(ref_dat) => self.load_ref_table(table_path, &ref_dat),
492            TableDatResult::Concat(concat_dat) => self.load_concat_table(table_path, &concat_dat),
493        }
494    }
495
496    pub(crate) fn save_metadata_only(
497        &self,
498        table_path: &Path,
499        snapshot: &StorageSnapshot,
500    ) -> Result<(), StorageError> {
501        let control_path = table_path.join(TABLE_CONTROL_FILE);
502        if !table_path.exists() {
503            return Err(StorageError::MissingPath(table_path.to_path_buf()));
504        }
505        if !control_path.exists() {
506            return Err(StorageError::MissingControlFile(control_path));
507        }
508
509        match read_table_dat_dispatch(&control_path)? {
510            TableDatResult::Plain(mut table_dat) => {
511                table_dat.table_desc.table_keywords = snapshot.keywords.clone();
512                for column in &mut table_dat.table_desc.columns {
513                    column.keywords = snapshot
514                        .column_keywords
515                        .get(&column.col_name)
516                        .cloned()
517                        .unwrap_or_default();
518                }
519                write_table_dat(&control_path, &table_dat)?;
520                let info_path = table_path.join(TABLE_INFO_FILE);
521                fs::write(&info_path, snapshot.table_info.to_string())?;
522                Ok(())
523            }
524            TableDatResult::Ref(_) | TableDatResult::Concat(_) => {
525                Err(StorageError::FormatMismatch(
526                    "metadata-only save is only supported for PlainTable storage".to_string(),
527                ))
528            }
529        }
530    }

532    pub(crate) fn load_metadata_only(
533        &self,
534        table_path: &Path,
535    ) -> Result<StorageSnapshot, StorageError> {
536        let control_path = table_path.join(TABLE_CONTROL_FILE);
537        if !table_path.exists() {
538            return Err(StorageError::MissingPath(table_path.to_path_buf()));
539        }
540        if !control_path.exists() {
541            return Err(StorageError::MissingControlFile(control_path));
542        }
543
544        match read_table_dat_dispatch(&control_path)? {
545            TableDatResult::Plain(table_dat) => {
546                self.load_plain_table_metadata(table_path, &table_dat)
547            }
548            // Metadata-only open is primarily for plain tiled tables. Fall back
549            // to the full loader for more complex table types.
550            TableDatResult::Ref(_) | TableDatResult::Concat(_) => {
551                let mut snapshot = self.load(table_path)?;
552                snapshot.rows.clear();
553                Ok(snapshot)
554            }
555        }
556    }

558    /// Load a PlainTable from table.dat contents and data files.
559    ///
560    /// Uses two-pass loading:
561    /// - Pass 1: Load stored columns from storage managers (StManAipsIO, StandardStMan).
562    /// - Pass 2: Materialize virtual columns from virtual engines.
563    /// - Pass 3: Reject any remaining unknown DM types.
564    fn load_plain_table(
565        &self,
566        table_path: &Path,
567        table_dat: &TableDatContents,
568        row_hint: Option<u64>,
569    ) -> Result<StorageSnapshot, StorageError> {
570        let schema = table_dat.to_table_schema()?;
571        let keywords = table_dat.table_desc.table_keywords.clone();
572        let column_keywords: HashMap<String, RecordValue> = table_dat
573            .table_desc
574            .columns
575            .iter()
576            .filter(|c| !c.keywords.fields().is_empty())
577            .map(|c| (c.col_name.clone(), c.keywords.clone()))
578            .collect();
579        // Real-world casacore tables can carry a stale or zero row count in
580        // the outer "Table" envelope while the nested ColumnSet/storage-
581        // manager state still reflects the actual number of rows. Upstream
582        // C++ lets the ColumnSet/data-manager layer override the caller's
583        // row count during open, so use the larger of the two counts here.
584        let nrrow = table_dat
585            .nrrow
586            .max(table_dat.column_set.nrrow)
587            .max(row_hint.unwrap_or(0)) as usize;
588
589        let mut rows: Vec<RecordValue> = (0..nrrow).map(|_| RecordValue::default()).collect();
590        let mut undefined_cells: Vec<HashSet<String>> =
591            (0..nrrow).map(|_| HashSet::new()).collect();
592        let mut virtual_columns = HashSet::new();
593
594        // Pass 1: Load stored columns from storage managers.
595        for dm in &table_dat.column_set.data_managers {
596            if is_virtual_engine(&dm.type_name) {
597                continue; // Handled in pass 2.
598            }
599
600            let data_path = table_path.join(format!("{}{}", TABLE_DATA_FILE_PREFIX, dm.seq_nr));
601
602            let bound_cols: Vec<(usize, &_)> = table_dat
603                .column_set
604                .columns
605                .iter()
606                .enumerate()
607                .filter(|(_, pc)| pc.dm_seq_nr == dm.seq_nr)
608                .collect();
609
610            match dm.type_name.as_str() {
611                "StManAipsIO" => {
612                    if !data_path.is_file() {
613                        return Err(StorageError::MissingDataFile(data_path));
614                    }
615                    load_stman_aipsio_columns(
616                        &data_path,
617                        &table_dat.table_desc.columns,
618                        &bound_cols,
619                        &mut rows,
620                        &mut undefined_cells,
621                        nrrow,
622                        ByteOrder::BigEndian,
623                    )
624                    .map_err(|err| {
625                        StorageError::FormatMismatch(format!(
626                            "while loading {} seq {} from {}: {err}",
627                            dm.type_name,
628                            dm.seq_nr,
629                            data_path.display()
630                        ))
631                    })?;
632                }
633                "StandardStMan" => {
634                    if !data_path.is_file() {
635                        return Err(StorageError::MissingDataFile(data_path));
636                    }
637                    load_ssm_columns(
638                        &data_path,
639                        &dm.data,
640                        &table_dat.table_desc.columns,
641                        &bound_cols,
642                        &mut rows,
643                        &mut undefined_cells,
644                        nrrow,
645                    )
646                    .map_err(|err| {
647                        StorageError::FormatMismatch(format!(
648                            "while loading {} seq {} from {}: {err}",
649                            dm.type_name,
650                            dm.seq_nr,
651                            data_path.display()
652                        ))
653                    })?;
654                }
655                "IncrementalStMan" => {
656                    if !data_path.is_file() {
657                        return Err(StorageError::MissingDataFile(data_path));
658                    }
659                    load_ism_columns(
660                        &data_path,
661                        &dm.data,
662                        &table_dat.table_desc.columns,
663                        &bound_cols,
664                        &mut rows,
665                        &mut undefined_cells,
666                        nrrow,
667                    )
668                    .map_err(|err| {
669                        StorageError::FormatMismatch(format!(
670                            "while loading {} seq {} from {}: {err}",
671                            dm.type_name,
672                            dm.seq_nr,
673                            data_path.display()
674                        ))
675                    })?;
676                }
677                "TiledColumnStMan" | "TiledShapeStMan" | "TiledCellStMan" | "TiledDataStMan" => {
678                    tiled_stman::load_tiled_columns(
679                        table_path,
680                        dm,
681                        &table_dat.table_desc.columns,
682                        &bound_cols,
683                        &mut rows,
684                        &mut undefined_cells,
685                        nrrow,
686                    )
687                    .map_err(|err| {
688                        StorageError::FormatMismatch(format!(
689                            "while loading {} seq {} from {}: {err}",
690                            dm.type_name,
691                            dm.seq_nr,
692                            data_path.display()
693                        ))
694                    })?;
695                }
696                other => {
697                    return Err(StorageError::UnsupportedDataManager(other.to_string()));
698                }
699            }
700        }
701
702        // Pass 2: Materialize virtual columns.
703        // Only clone rows if there are virtual engines that need the stored
704        // snapshot — avoids an O(rows × cols) clone for tables with no
705        // virtual columns (the common case for MS subtables).
706        let has_virtual = table_dat
707            .column_set
708            .data_managers
709            .iter()
710            .any(|dm| is_virtual_engine(&dm.type_name));
711
712        if has_virtual {
713            let stored_rows = rows.clone();
714            for dm in &table_dat.column_set.data_managers {
715                if !is_virtual_engine(&dm.type_name) {
716                    continue;
717                }
718
719                let engine = lookup_engine(&dm.type_name)
720                    .ok_or_else(|| StorageError::UnsupportedDataManager(dm.type_name.clone()))?;
721
722                let bound_cols: Vec<(usize, &_)> = table_dat
723                    .column_set
724                    .columns
725                    .iter()
726                    .enumerate()
727                    .filter(|(_, pc)| pc.dm_seq_nr == dm.seq_nr)
728                    .collect();
729
730                for &(desc_idx, _) in &bound_cols {
731                    virtual_columns.insert(table_dat.table_desc.columns[desc_idx].col_name.clone());
732                }
733
734                let ctx = VirtualContext {
735                    col_descs: &table_dat.table_desc.columns,
736                    rows: &stored_rows,
737                    table_path,
738                    nrrow,
739                };
740
741                engine.materialize(&ctx, &bound_cols, &mut rows)?;
742            }
743        }
744
745        let table_info = load_table_info(table_path);
746
747        // Build DM info from table.dat entries.
748        let dm_info = extract_dm_info(table_dat);
749
750        Ok(StorageSnapshot {
751            row_count: rows.len(),
752            rows,
753            undefined_cells,
754            keywords,
755            column_keywords,
756            schema: Some(schema),
757            table_info,
758            virtual_columns,
759            virtual_bindings: Vec::new(),
760            dm_info,
761        })
762    }

765        &self,
766        table_path: &Path,
767        table_dat: &TableDatContents,
768    ) -> Result<StorageSnapshot, StorageError> {
769        let schema = table_dat.to_table_schema()?;

772            .table_desc
773            .columns
774            .iter()

789        }
790
791        Ok(StorageSnapshot {
792            row_count: table_dat.nrrow.max(table_dat.column_set.nrrow) as usize,
793            rows: Vec::new(),
794            undefined_cells: Vec::new(),
795            keywords,
796            column_keywords,
797            schema: Some(schema),
798            table_info: load_table_info(table_path),
799            virtual_columns,

805    /// Load a RefTable by opening the parent and extracting referenced rows.
806    fn load_ref_table(
807        &self,
808        table_path: &Path,
809        ref_dat: &RefTableDatContents,
810    ) -> Result<StorageSnapshot, StorageError> {
811        // Resolve parent path using C++ addDirectory convention:
812        //  - Leading "./" means: take parent directory of ref table, then append
813        //  - Leading "././" means: the path is inside the ref table directory
814        //  - Otherwise: use as-is (absolute or plain relative)
815        let parent_path = add_directory(&ref_dat.parent_relative_path, table_path)?;
816
817        // Recursively load the parent table.
818        let parent = self.load(&parent_path)?;
819
820        // Validate that the parent hasn't shrunk.
821        if ref_dat.parent_nrrow > parent.rows.len() as u64 {
822            return Err(StorageError::FormatMismatch(format!(
823                "parent table has {} rows but RefTable expects at least {}",
824                parent.rows.len(),
825                ref_dat.parent_nrrow
826            )));
827        }
828
829        // Extract referenced rows.
830        let mut rows = Vec::with_capacity(ref_dat.row_map.len());
831        let mut undefined_cells = Vec::with_capacity(ref_dat.row_map.len());
832        for &parent_row in &ref_dat.row_map {
833            let idx = parent_row as usize;
834            if idx >= parent.rows.len() {
835                return Err(StorageError::FormatMismatch(format!(
836                    "RefTable references parent row {idx} but parent has {} rows",
837                    parent.rows.len()
838                )));
839            }
840            // Apply column projection if the view uses a subset of columns.
841            if ref_dat.column_names.len() < parent.rows[idx].fields().len() {
842                let mut projected = RecordValue::default();
843                let mut projected_undefined = HashSet::new();
844                for view_col in &ref_dat.column_names {
845                    // Resolve view column name to parent column name.
846                    let parent_col = ref_dat
847                        .column_name_map
848                        .iter()
849                        .find(|(v, _)| v == view_col)
850                        .map(|(_, p)| p.as_str())
851                        .unwrap_or(view_col.as_str());
852                    if let Some(val) = parent.rows[idx].get(parent_col) {
853                        projected.push(RecordField::new(view_col.clone(), val.clone()));
854                    }
855                    if parent.undefined_cells[idx].contains(parent_col) {
856                        projected_undefined.insert(view_col.clone());
857                    }
858                }
859                rows.push(projected);
860                undefined_cells.push(projected_undefined);
861            } else {
862                rows.push(parent.rows[idx].clone());
863                undefined_cells.push(parent.undefined_cells[idx].clone());
864            }
865        }
866
867        // Build projected schema if applicable.
868        let schema = parent.schema.and_then(|s| {
869            if ref_dat.column_names.len() < s.columns().len() {
870                let cols: Vec<_> = ref_dat
871                    .column_names
872                    .iter()
873                    .filter_map(|name| {
874                        let parent_name = ref_dat
875                            .column_name_map
876                            .iter()
877                            .find(|(v, _)| v == name)
878                            .map(|(_, p)| p.as_str())
879                            .unwrap_or(name.as_str());
880                        s.columns()
881                            .iter()
882                            .find(|c| c.name() == parent_name)
883                            .cloned()
884                    })
885                    .collect();
886                crate::schema::TableSchema::new(cols).ok()
887            } else {
888                Some(s)
889            }
890        });
891
892        let table_info = load_table_info(table_path);
893
894        Ok(StorageSnapshot {
895            row_count: rows.len(),
896            rows,
897            undefined_cells,
898            keywords: parent.keywords,
899            column_keywords: parent.column_keywords,
900            schema,
901            table_info,
902            virtual_columns: HashSet::new(),
903            virtual_bindings: Vec::new(),
904            dm_info: vec![],
905        })
906    }

983    /// Save with per-column DM bindings.
984    ///
985    /// Columns listed in `bindings` use their specified DM; all other columns
986    /// use the default DM from `dm_kind`. Each DM group writes its own data
987    /// file (`table.f0`, `table.f1`, ...) and gets a separate entry in `table.dat`.
988    #[allow(clippy::too_many_arguments)]
989    pub(crate) fn save_with_bindings(
990        &self,
991        table_path: &Path,
992        snapshot: &StorageSnapshot,
993        default_dm: crate::table::DataManagerKind,
994        big_endian: bool,
995        default_tile_shape: Option<&[usize]>,
996        bindings: &std::collections::HashMap<String, crate::table::ColumnBinding>,
997    ) -> Result<(), StorageError> {
998        use crate::table::DataManagerKind;
999
1000        fs::create_dir_all(table_path)?;
1001
1002        let schema = snapshot.schema.as_ref().ok_or_else(|| {
1003            StorageError::FormatMismatch("cannot save without schema".to_string())
1004        })?;
1005        let filtered_rows = filter_rows_for_save(&snapshot.rows, &snapshot.undefined_cells);
1006        let nrrow = filtered_rows.len() as u64;
1007        let has_virtual = !snapshot.virtual_bindings.is_empty();
1008
1009        // Group columns by DM kind. Columns in `bindings` get their own DM;
1010        // everything else goes into the default DM.
1011        struct DmGroup {
1012            dm_kind: DataManagerKind,
1013            dm_type_name: String,
1014            seq_nr: u32,
1015            col_names: Vec<String>,
1016            tile_shape: Option<Vec<usize>>,
1017        }
1018
1019        let mut groups: Vec<DmGroup> = Vec::new();
1020
1021        // Default DM group (seq_nr 0).
1022        let default_type_name = match default_dm {
1023            DataManagerKind::StManAipsIO => "StManAipsIO",
1024            DataManagerKind::StandardStMan => "StandardStMan",
1025            DataManagerKind::IncrementalStMan => "IncrementalStMan",
1026            DataManagerKind::TiledColumnStMan => "TiledColumnStMan",
1027            DataManagerKind::TiledShapeStMan => "TiledShapeStMan",
1028            DataManagerKind::TiledCellStMan => "TiledCellStMan",
1029            DataManagerKind::TiledDataStMan => "TiledDataStMan",
1030        };
1031        groups.push(DmGroup {
1032            dm_kind: default_dm,
1033            dm_type_name: default_type_name.to_string(),
1034            seq_nr: 0,
1035            col_names: Vec::new(),
1036            tile_shape: default_tile_shape.map(|s| s.to_vec()),
1037        });
1038
1039        // Collect virtual column names for filtering.
1040        let virtual_col_names: HashSet<&str> = snapshot
1041            .virtual_columns
1042            .iter()
1043            .map(|s| s.as_str())
1044            .collect();
1045
1046        // Build additional DM groups from bindings, then assign columns.
1047        let mut binding_seq_map: std::collections::HashMap<String, usize> =
1048            std::collections::HashMap::new();
1049        let mut next_seq = 1u32;
1050
1051        for col in schema.columns() {
1052            let col_name = col.name();
1053            if virtual_col_names.contains(col_name) {
1054                continue; // Virtual columns don't get stored DM entries.
1055            }
1056            if let Some(binding) = bindings.get(col_name) {
1057                let group_key = format!("{:?}", binding.data_manager);
1058                let group_idx = if let Some(&idx) = binding_seq_map.get(&group_key) {
1059                    idx
1060                } else {
1061                    let dm_type = match binding.data_manager {
1062                        DataManagerKind::StManAipsIO => "StManAipsIO",
1063                        DataManagerKind::StandardStMan => "StandardStMan",
1064                        DataManagerKind::IncrementalStMan => "IncrementalStMan",
1065                        DataManagerKind::TiledColumnStMan => "TiledColumnStMan",
1066                        DataManagerKind::TiledShapeStMan => "TiledShapeStMan",
1067                        DataManagerKind::TiledCellStMan => "TiledCellStMan",
1068                        DataManagerKind::TiledDataStMan => "TiledDataStMan",
1069                    };
1070                    let idx = groups.len();
1071                    groups.push(DmGroup {
1072                        dm_kind: binding.data_manager,
1073                        dm_type_name: dm_type.to_string(),
1074                        seq_nr: next_seq,
1075                        col_names: Vec::new(),
1076                        tile_shape: binding.tile_shape.clone(),
1077                    });
1078                    next_seq += 1;
1079                    binding_seq_map.insert(group_key, idx);
1080                    idx
1081                };
1082                groups[group_idx].col_names.push(col_name.to_string());
1083            } else {
1084                groups[0].col_names.push(col_name.to_string());
1085            }
1086        }
1087
1088        // Build a combined table.dat with all DM entries.
1089        // For now, use from_snapshot for the first DM and add extra DM entries manually.
1090        // This is a simplified approach: we write each group's data separately.
1091
1092        // First, build a complete table_dat with all DMs.
1093        let mut dm_entries = Vec::new();
1094        let mut col_dm_map: std::collections::HashMap<String, u32> =
1095            std::collections::HashMap::new();
1096
1097        for group in &groups {
1098            dm_entries.push(table_control::DataManagerEntry {
1099                type_name: group.dm_type_name.clone(),
1100                seq_nr: group.seq_nr,
1101                data: Vec::new(), // Will be filled in for SSM/ISM.
1102            });
1103            for col_name in &group.col_names {
1104                col_dm_map.insert(col_name.clone(), group.seq_nr);
1105            }
1106        }
1107
1108        // Build initial table_dat (using from_snapshot but overriding DM entries).
1109        let mut table_dat = if has_virtual {
1110            TableDatContents::from_snapshot_with_virtual(
1111                schema,
1112                &snapshot.keywords,
1113                &snapshot.column_keywords,
1114                nrrow,
1115                &groups[0].dm_type_name,
1116                &[],
1117                big_endian,
1118                &snapshot.virtual_bindings,
1119                table_path,
1120            )
1121        } else {
1122            TableDatContents::from_snapshot(
1123                schema,
1124                &snapshot.keywords,
1125                &snapshot.column_keywords,
1126                nrrow,
1127                &groups[0].dm_type_name,
1128                &[],
1129                big_endian,
1130            )
1131        };
1132
1133        // Override DM entries with our multi-DM setup.
1134        // Preserve any virtual DM entries that from_snapshot_with_virtual added.
1135        let virtual_dm_entries: Vec<_> = table_dat
1136            .column_set
1137            .data_managers
1138            .iter()
1139            .filter(|e| e.seq_nr > 0 && is_virtual_engine(&e.type_name))
1140            .cloned()
1141            .collect();
1142
1143        // Update each PlainColumnEntry's dm_seq_nr.
1144        for pc in &mut table_dat.column_set.columns {
1145            if let Some(&seq) = col_dm_map.get(&pc.original_name) {
1146                pc.dm_seq_nr = seq;
1147            }
1148        }
1149
1150        // Override DM entries, filtering out empty groups that have no
1151        // columns and no data file to avoid "missing table.f<N>" errors.
1152        let non_empty_groups: Vec<&DmGroup> =
1153            groups.iter().filter(|g| !g.col_names.is_empty()).collect();
1154        table_dat.column_set.data_managers = non_empty_groups
1155            .iter()
1156            .map(|g| table_control::DataManagerEntry {
1157                type_name: g.dm_type_name.clone(),
1158                seq_nr: g.seq_nr,
1159                data: Vec::new(),
1160            })
1161            .collect();
1162        table_dat
1163            .column_set
1164            .data_managers
1165            .extend(virtual_dm_entries);
1166        table_dat.column_set.seq_count = table_dat
1167            .column_set
1168            .data_managers
1169            .iter()
1170            .map(|e| e.seq_nr + 1)
1171            .max()
1172            .unwrap_or(1);
1173
1174        // Write data files per group.
1175        for group in &groups {
1176            if group.col_names.is_empty() {
1177                continue;
1178            }
1179            let data_path = table_path.join(format!("{}{}", TABLE_DATA_FILE_PREFIX, group.seq_nr));
1180
1181            // Filter rows and col_descs to only this group's columns.
1182            let group_col_set: HashSet<&str> = group.col_names.iter().map(|s| s.as_str()).collect();
1183            let group_rows: Vec<RecordValue> = filtered_rows
1184                .iter()
1185                .map(|row| {
1186                    let fields: Vec<_> = row
1187                        .fields()
1188                        .iter()
1189                        .filter(|f| group_col_set.contains(f.name.as_str()))
1190                        .cloned()
1191                        .collect();
1192                    RecordValue::new(fields)
1193                })
1194                .collect();
1195            let group_col_descs: Vec<_> = table_dat
1196                .table_desc
1197                .columns
1198                .iter()
1199                .filter(|c| group_col_set.contains(c.col_name.as_str()))
1200                .cloned()
1201                .collect();
1202
1203            match group.dm_kind {
1204                DataManagerKind::StManAipsIO => {
1205                    write_stman_file(
1206                        &data_path,
1207                        &group_col_descs,
1208                        &group_rows,
1209                        ByteOrder::BigEndian,
1210                    )?;
1211                }
1212                DataManagerKind::StandardStMan => {
1213                    let dm_data =
1214                        write_ssm_file(&data_path, &group_col_descs, &group_rows, big_endian)?;
1215                    // Update the DM blob in the table_dat.
1216                    if let Some(entry) = table_dat
1217                        .column_set
1218                        .data_managers
1219                        .iter_mut()
1220                        .find(|e| e.seq_nr == group.seq_nr)
1221                    {
1222                        entry.data = dm_data;
1223                    }
1224                }
1225                DataManagerKind::IncrementalStMan => {
1226                    let dm_data =
1227                        write_ism_file(&data_path, &group_col_descs, &group_rows, big_endian)?;
1228                    if let Some(entry) = table_dat
1229                        .column_set
1230                        .data_managers
1231                        .iter_mut()
1232                        .find(|e| e.seq_nr == group.seq_nr)
1233                    {
1234                        entry.data = dm_data;
1235                    }
1236                }
1237                DataManagerKind::TiledColumnStMan
1238                | DataManagerKind::TiledShapeStMan
1239                | DataManagerKind::TiledCellStMan
1240                | DataManagerKind::TiledDataStMan => {
1241                    // Use the first column name as the hypercolumn/DM name.
1242                    let first_col = group.col_names.first().map(|s| s.as_str()).unwrap_or("");
1243                    tiled_stman::save_tiled_columns(
1244                        table_path,
1245                        group.seq_nr,
1246                        &group.dm_type_name,
1247                        &group_col_descs,
1248                        &group_rows,
1249                        big_endian,
1250                        group.tile_shape.as_deref(),
1251                        first_col,
1252                    )?;
1253                }
1254            }
1255        }
1256
1257        let control_path = table_path.join(TABLE_CONTROL_FILE);
1258        write_table_dat(&control_path, &table_dat)?;
1259        let info_path = table_path.join(TABLE_INFO_FILE);
1260        fs::write(&info_path, snapshot.table_info.to_string())?;
1261
1262        Ok(())
1263    }
1264}

1266/// Load columns from a StManAipsIO data file into row records.
1267fn load_stman_aipsio_columns(
1268    data_path: &Path,
1269    all_col_descs: &[table_control::ColumnDescContents],
1270    bound_cols: &[(usize, &table_control::PlainColumnEntry)],
1271    rows: &mut [RecordValue],
1272    undefined_cells: &mut [HashSet<String>],
1273    nrrow: usize,
1274    byte_order: ByteOrder,
1275) -> Result<(), StorageError> {
1276    // Build column info for the StManAipsIO reader using the bound columns
1277    let col_info: Vec<StManColumnInfo> = bound_cols
1278        .iter()
1279        .map(|(desc_idx, _)| {
1280            let c = &all_col_descs[*desc_idx];
1281            let nrelem = if c.is_array && !c.shape.is_empty() {
1282                c.shape.iter().map(|&s| s as usize).product()
1283            } else {
1284                0
1285            };
1286            StManColumnInfo {
1287                is_array: c.is_array,
1288                nrelem,
1289            }
1290        })
1291        .collect();
1292
1293    let stman_data = read_stman_file(data_path, &col_info, byte_order)?;
1294
1295    for (stman_col_idx, (desc_idx, _)) in bound_cols.iter().enumerate() {
1296        if stman_col_idx >= stman_data.columns.len() {
1297            break;
1298        }
1299        let col_desc = &all_col_descs[*desc_idx];
1300        match &stman_data.columns[stman_col_idx] {
1301            StManColumnData::Flat(raw) => {
1302                for (row_idx, row) in rows.iter_mut().enumerate() {
1303                    let value = extract_row_value(raw, col_desc, row_idx, nrrow)?;
1304                    if matches!(&value, Value::Array(_)) {
1305                        undefined_cells[row_idx].insert(col_desc.col_name.clone());
1306                    }
1307                    row.push(RecordField::new(col_desc.col_name.clone(), value));
1308                }
1309            }
1310            StManColumnData::Indirect(per_row) => {
1311                for (row_idx, row) in rows.iter_mut().enumerate() {
1312                    let value = match per_row.get(row_idx) {
1313                        Some(Some(v)) => {
1314                            if col_desc.is_record() {
1315                                // C++ stores records as indirect Vector<uChar>.
1316                                // Deserialize the byte array back to a RecordValue.
1317                                match v {
1318                                    Value::Array(casacore_types::ArrayValue::UInt8(arr)) => {
1319                                        let bytes = arr.as_slice().ok_or_else(|| {
1320                                            StorageError::FormatMismatch(format!(
1321                                                "record column '{}': non-contiguous u8 array",
1322                                                col_desc.col_name
1323                                            ))
1324                                        })?;
1325                                        Value::Record(table_control::deserialize_record_from_uchar(
1326                                            bytes,
1327                                        )?)
1328                                    }
1329                                    Value::Record(_) => v.clone(),
1330                                    _ => {
1331                                        return Err(StorageError::FormatMismatch(format!(
1332                                            "record column '{}' has unexpected value type",
1333                                            col_desc.col_name
1334                                        )));
1335                                    }
1336                                }
1337                            } else {
1338                                v.clone()
1339                            }
1340                        }
1341                        // Undefined cell: empty record or 0-D empty array.
1342                        Some(None) | None => {
1343                            if col_desc.is_record() {
1344                                Value::Record(RecordValue::default())
1345                            } else {
1346                                let dt = CasacoreDataType::from_primitive_type(
1347                                    col_desc.require_primitive_type()?,
1348                                    false,
1349                                );
1350                                undefined_cells[row_idx].insert(col_desc.col_name.clone());
1351                                make_undefined_array(dt, col_desc.nrdim.max(0) as usize)
1352                            }
1353                        }
1354                    };
1355                    row.push(RecordField::new(col_desc.col_name.clone(), value));
1356                }
1357            }
1358        }
1359    }
1360
1361    Ok(())
1362}

1460/// Create a Value representing an undefined variable-shape array cell.
1461///
1462/// For ndim > 0, returns a zero-sized array with `ndim` dimensions (all sizes 0).
1463/// For ndim == 0, returns a 0-D empty array.
1464pub(crate) fn make_undefined_array(dt: CasacoreDataType, ndim: usize) -> casacore_types::Value {
1465    use casacore_types::{ArrayValue, Value};
1466    use ndarray::{ArrayD, IxDyn};
1467
1468    let shape = vec![0usize; ndim];
1469    let s = IxDyn(&shape);
1470    let av = match dt {
1471        CasacoreDataType::TpBool => ArrayValue::Bool(ArrayD::default(s)),
1472        CasacoreDataType::TpUChar => ArrayValue::UInt8(ArrayD::default(s)),
1473        CasacoreDataType::TpShort => ArrayValue::Int16(ArrayD::default(s)),
1474        CasacoreDataType::TpUShort => ArrayValue::UInt16(ArrayD::default(s)),
1475        CasacoreDataType::TpInt => ArrayValue::Int32(ArrayD::default(s)),
1476        CasacoreDataType::TpUInt => ArrayValue::UInt32(ArrayD::default(s)),
1477        CasacoreDataType::TpInt64 => ArrayValue::Int64(ArrayD::default(s)),
1478        CasacoreDataType::TpFloat => ArrayValue::Float32(ArrayD::default(s)),
1479        CasacoreDataType::TpDouble => ArrayValue::Float64(ArrayD::default(s)),
1480        CasacoreDataType::TpComplex => ArrayValue::Complex32(ArrayD::from_elem(
1481            s,
1482            casacore_types::Complex32::new(0.0, 0.0),
1483        )),
1484        CasacoreDataType::TpDComplex => ArrayValue::Complex64(ArrayD::from_elem(
1485            s,
1486            casacore_types::Complex64::new(0.0, 0.0),
1487        )),
1488        CasacoreDataType::TpString => ArrayValue::String(ArrayD::from_elem(s, String::new())),
1489        _ => ArrayValue::Float32(ArrayD::default(s)),
1490    };
1491    Value::Array(av)
1492}

1494/// Compute a relative path using the C++ `Path::stripDirectory` convention.
1495///
1496/// Given `target_path` (the table being referenced) and `from_path` (the table
1497/// that contains the reference), produces a relative path string that
1498/// [`add_directory`] can invert.
1499///
1500/// Convention:
1501/// - If target is a sibling (same parent directory): `"./name"`
1502/// - If target is inside the from directory: `"././name"`
1503/// - Otherwise: absolute path as fallback
1504pub(crate) fn strip_directory(target_path: &Path, from_path: &Path) -> String {
1505    let target_abs = std::path::absolute(target_path).unwrap_or_else(|_| target_path.to_path_buf());
1506    let from_abs = std::path::absolute(from_path).unwrap_or_else(|_| from_path.to_path_buf());
1507
1508    // Check if target is inside the from directory (././ convention).
1509    let from_dir_prefix = format!("{}/", from_abs.display());
1510    let target_str = format!("{}", target_abs.display());
1511    if target_str.starts_with(&from_dir_prefix) {
1512        let remainder = &target_str[from_dir_prefix.len()..];
1513        return format!("././{remainder}");
1514    }
1515
1516    // Check if they share the same parent directory (./ convention).
1517    if let (Some(from_parent), Some(target_parent)) = (from_abs.parent(), target_abs.parent()) {
1518        if from_parent == target_parent {
1519            let name = target_abs.file_name().unwrap_or_default().to_string_lossy();
1520            return format!("./{name}");
1521        }
1522    }
1523
1524    // Fallback: absolute path.
1525    target_abs.to_string_lossy().to_string()
1526}

1528/// Resolve a stored relative path using the C++ `Path::addDirectory` convention.
1529///
1530/// - `"./name"` → `parent_dir(ref_table_path) / name`
1531/// - `"././name"` → `ref_table_path / name`
1532/// - absolute or plain → used as-is
1533pub(crate) fn add_directory(
1534    relative: &str,
1535    ref_table_path: &Path,
1536) -> Result<PathBuf, StorageError> {
1537    let mut name = relative;
1538    let mut stripped_count = 0usize;
1539
1540    // Strip leading "./" segments, counting how many were removed.
1541    while name.len() >= 2 && name.starts_with("./") {
1542        name = &name[2..];
1543        stripped_count += 1;
1544    }
1545
1546    if stripped_count == 0 {
1547        // No "./" prefix — use as-is (absolute or plain relative).
1548        let p = PathBuf::from(relative);
1549        if p.is_absolute() {
1550            return Ok(p);
1551        }
1552        // Plain relative: resolve relative to ref table's parent directory.
1553        return Ok(ref_table_path
1554            .parent()
1555            .unwrap_or(ref_table_path)
1556            .join(relative));
1557    }
1558
1559    if stripped_count == 1 {
1560        // "./" was removed → add parent directory of ref_table_path.
1561        let dir = ref_table_path.parent().ok_or_else(|| {
1562            StorageError::FormatMismatch(format!(
1563                "cannot get parent directory of '{}'",
1564                ref_table_path.display()
1565            ))
1566        })?;
1567        return Ok(dir.join(name));
1568    }
1569
1570    // "././" or more → add the ref table path itself (subtable inside it).
1571    Ok(ref_table_path.join(name))
1572}
