2//! Observatory catalog data loaded from casacore `geodetic/Observatories`.

4use std::collections::HashMap;
5
6/// A single observatory row from casacore `geodetic/Observatories`.
7#[derive(Debug, Clone, PartialEq)]
8pub struct ObservatoryEntry {
9    /// Table `MJD` column.
10    pub mjd: f64,
11    /// Table `Name` column.
12    pub name: String,
13    /// Table `Type` column.
14    pub observatory_type: String,
15    /// Table `Long` column, in degrees.
16    pub longitude_deg: f64,
17    /// Table `Lat` column, in degrees.
18    pub latitude_deg: f64,
19    /// Table `Height` column, in metres.
20    pub height_m: f64,
21    /// Table `X` column, in metres.
22    pub x_m: f64,
23    /// Table `Y` column, in metres.
24    pub y_m: f64,
25    /// Table `Z` column, in metres.
26    pub z_m: f64,
27    /// Table `Source` column.
28    pub source: String,
29    /// Table `Comment` column.
30    pub comment: String,
31    /// Table `AntennaResponses` column.
32    pub antenna_responses: String,
33}
34
35/// In-memory catalog of observatory metadata.
36#[derive(Debug, Clone, PartialEq)]
37pub struct ObservatoryCatalog {
38    entries: Vec<ObservatoryEntry>,
39    by_name: HashMap<String, usize>,
40}
41
42impl ObservatoryCatalog {
43    /// Build a catalog from explicitly provided entries.
44    pub fn from_entries(entries: Vec<ObservatoryEntry>) -> Self {
45        let by_name = entries
46            .iter()
47            .enumerate()
48            .fold(HashMap::new(), |mut map, (index, entry)| {
49                map.entry(normalize_name(&entry.name)).or_insert(index);
50                map
51            });
52        Self { entries, by_name }
53    }
54
55    /// Return the packaged/runtime standard catalog.
56    pub fn bundled() -> &'static Self {
57        crate::load_observatories().0
58    }
59
60    /// Iterate over the catalog entries in source order.
61    pub fn iter(&self) -> impl Iterator<Item = &ObservatoryEntry> {
62        self.entries.iter()
63    }
64
65    /// Borrow all entries.
66    pub fn entries(&self) -> &[ObservatoryEntry] {
67        &self.entries
68    }
69
70    /// Look up an observatory by case-insensitive name.
71    pub fn get(&self, name: &str) -> Option<&ObservatoryEntry> {
72        self.by_name
73            .get(&normalize_name(name))
74            .and_then(|index| self.entries.get(*index))
75    }
76}
77
78fn normalize_name(name: &str) -> String {
79    name.trim().to_ascii_uppercase()
80}
