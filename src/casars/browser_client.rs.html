56#[derive(Debug)]
57struct SessionProcessClient {
58    child: Arc<Mutex<Child>>,
59    stdin: Arc<Mutex<ChildStdin>>,
60    responses: Receiver<String>,
61    stderr: Arc<Mutex<String>>,
62}
63
64impl SessionProcessClient {
65    fn spawn(command: &ResolvedCommand, session_name: &str) -> Result<Self, String> {
66        let mut process = command.command();
67        process
68            .arg("--session")
69            .stdin(Stdio::piped())
70            .stdout(Stdio::piped())
71            .stderr(Stdio::piped());
72
73        let mut child = process
74            .spawn()
75            .map_err(|error| format!("spawn {session_name} session: {error}"))?;
76        let stdin = child
77            .stdin
78            .take()
79            .ok_or_else(|| format!("{session_name} session stdin was not captured"))?;
80        let stdout = child
81            .stdout
82            .take()
83            .ok_or_else(|| format!("{session_name} session stdout was not captured"))?;
84        let stderr = child
85            .stderr
86            .take()
87            .ok_or_else(|| format!("{session_name} session stderr was not captured"))?;
88
89        let child = Arc::new(Mutex::new(child));
90        let stdin = Arc::new(Mutex::new(stdin));
91        let stderr_buffer = Arc::new(Mutex::new(String::new()));
92        let (tx, rx) = mpsc::channel();
93
94        thread::spawn(move || {
95            let reader = BufReader::new(stdout);
96            for line_result in reader.lines() {
97                let Ok(line) = line_result else {
98                    break;
99                };
100                if line.trim().is_empty() {
101                    continue;
102                }
103                if tx.send(line).is_err() {
104                    break;
105                }
106            }
107        });
108
109        let stderr_target = Arc::clone(&stderr_buffer);
110        thread::spawn(move || {
111            let reader = BufReader::new(stderr);
112            for line_result in reader.lines() {
113                let Ok(line) = line_result else {
114                    break;
115                };
116                if let Ok(mut stderr) = stderr_target.lock() {
117                    stderr.push_str(&line);
118                    stderr.push('\n');
119                }
120            }
121        });
122
123        Ok(Self {
124            child,
125            stdin,
126            responses: rx,
127            stderr: stderr_buffer,
128        })
129    }
130
131    fn request_raw(
132        &self,
133        payload: &str,
134        timeout: Duration,
135        session_name: &str,
136    ) -> Result<String, String> {
137        {
138            let mut stdin = self
139                .stdin
140                .lock()
141                .map_err(|_| "failed to acquire browser stdin lock".to_string())?;
142            stdin
143                .write_all(payload.as_bytes())
144                .and_then(|_| stdin.write_all(b"\n"))
145                .and_then(|_| stdin.flush())
146                .map_err(|error| format!("write {session_name} request: {error}"))?;
147        }
148
149        self.responses
150            .recv_timeout(timeout)
151            .map_err(|error| match error {
152                RecvTimeoutError::Timeout => {
153                    let _ = self.terminate_and_wait();
154                    format_browser_failure(
155                        &format!("timed out waiting for {session_name} response"),
156                        self.stderr_text(),
157                        None,
158                    )
159                }
160                RecvTimeoutError::Disconnected => {
161                    let status = self.reap_exit_status();
162                    format_browser_failure(
163                        &format!("{session_name} session exited"),
164                        self.stderr_text(),
165                        status,
166                    )
167                }
168            })
169    }
170
171    pub(crate) fn cancel(&self) -> Result<(), String> {
172        self.terminate_and_wait().map(|_| ())
173    }
174
175    pub(crate) fn stderr_text(&self) -> String {
176        self.stderr
177            .lock()
178            .map(|stderr| stderr.clone())
179            .unwrap_or_default()
180    }
181
182    fn reap_exit_status(&self) -> Option<ExitStatus> {
183        let mut child = self.child.lock().ok()?;
184        child.try_wait().ok().flatten()
185    }
186
187    fn terminate_and_wait(&self) -> Result<Option<ExitStatus>, String> {
188        let mut child = self
189            .child
190            .lock()
191            .map_err(|_| "failed to acquire browser child lock".to_string())?;
192        if let Some(status) = child
193            .try_wait()
194            .map_err(|error| format!("poll browser session: {error}"))?
195        {
196            return Ok(Some(status));
197        }
198        child
199            .kill()
200            .map_err(|error| format!("terminate browser session: {error}"))?;
201        child
202            .wait()
203            .map(Some)
204            .map_err(|error| format!("wait for browser session: {error}"))
205    }
206}

248
249    fn request_with_timeout(
250        &self,
251        command: BrowserCommand,
252        timeout: Duration,
253    ) -> Result<BrowserSnapshot, SessionRequestError> {
254        let payload =
255            serde_json::to_string(&BrowserRequestEnvelope::new(command)).map_err(|error| {
256                SessionRequestError::Transport(format!("serialize tablebrowser request: {error}"))
257            })?;
258        let line = self
259            .process
260            .request_raw(&payload, timeout, "tablebrowser")
261            .map_err(SessionRequestError::Transport)?;
262        let response =
263            serde_json::from_str::<BrowserResponseEnvelope>(&line).unwrap_or_else(|error| {
264                BrowserResponseEnvelope::error(
265                    "invalid_response",
266                    format!("parse browser response: {error}"),
267                )
268            });
269        match response.response {
270            BrowserResponse::Snapshot(snapshot) => Ok(*snapshot),
271            BrowserResponse::Error(error) => Err(SessionRequestError::Backend(format!(
272                "{}: {}",
273                error.code, error.message
274            ))),
275        }
276    }

395fn duration_from_env(name: &str, default_ms: u64) -> Duration {
396    let millis = std::env::var(name)
397        .ok()
398        .and_then(|value| value.parse::<u64>().ok())
399        .filter(|value| *value > 0)
400        .unwrap_or(default_ms);
401    Duration::from_millis(millis)
402}

404fn format_browser_failure(prefix: &str, stderr: String, status: Option<ExitStatus>) -> String {
405    let mut message = match status {
406        Some(status) => format!("{prefix} with {status}"),
407        None => prefix.to_string(),
408    };
409    if !stderr.trim().is_empty() {
410        message.push_str(": ");
411        message.push_str(stderr.trim());
412    }
413    message
414}

500        let response = serde_json::to_string(&BrowserResponseEnvelope::snapshot(BrowserSnapshot {
501            capabilities: casars_tablebrowser_protocol::BrowserCapabilities { editable: false },
502            view: casars_tablebrowser_protocol::BrowserView::Overview,

504            table_path: "/tmp/fake.ms".to_string(),
505            breadcrumb: Vec::new(),

508            content_lines: vec!["Overview".to_string()],

515        let script = format!(

523    fn write_browser_script(root: &Path, script: &str) -> PathBuf {
524        let path = root.join("browser.sh");
525        fs::write(&path, script).expect("write script");
526        let mut permissions = fs::metadata(&path).expect("metadata").permissions();
527        permissions.set_mode(0o755);
528        fs::set_permissions(&path, permissions).expect("chmod");
529        path
530    }
