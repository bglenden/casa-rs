11/// A read-only view of a lattice restricted by a region.
12///
13/// Corresponds to the C++ `SubLattice<T>`. The sub-lattice presents a
14/// view whose coordinate system is relative to the region's bounding
15/// box. Pixels outside the region are returned as the default value
16/// for type `T`.
17///
18/// # Examples
19///
20/// ```rust
21/// use casacore_lattices::*;
22/// use ndarray::{ArrayD, IxDyn};
23///
24/// let data = ArrayD::from_shape_fn(IxDyn(&[10, 10]), |idx| (idx[0] * 10 + idx[1]) as f64);
25/// let lat = ArrayLattice::new(data);
26/// let region = LCBox::new(vec![2, 3], vec![5, 7], vec![10, 10]);
27/// let sub = SubLattice::new(&lat, Box::new(region));
28///
29/// assert_eq!(sub.shape(), &[4, 5]); // bounding box shape
30/// assert_eq!(sub.get_at(&[0, 0]).unwrap(), 23.0); // lat[2, 3]
31/// ```
32pub struct SubLattice<'a, T: LatticeElement, L: Lattice<T>> {
33    parent: &'a L,
34    region: Box<dyn LCRegion>,
35    bb_start: Vec<usize>,
36    bb_shape: Vec<usize>,
37    _phantom: std::marker::PhantomData<T>,
38}
39
40impl<'a, T: LatticeElement, L: Lattice<T>> SubLattice<'a, T, L> {
41    /// Creates a sub-lattice view restricted by the given region.
42    pub fn new(parent: &'a L, region: Box<dyn LCRegion>) -> Self {
43        let bb_start = region.bounding_box_start();
44        let bb_shape = region.bounding_box_shape();
45        Self {
46            parent,
47            region,
48            bb_start,
49            bb_shape,
50            _phantom: std::marker::PhantomData,
51        }
52    }
53
54    /// Returns the region defining this sub-lattice.
55    pub fn region(&self) -> &dyn LCRegion {
56        self.region.as_ref()
57    }
58}
59
60impl<'a, T: LatticeElement, L: Lattice<T>> Lattice<T> for SubLattice<'a, T, L> {
61    fn shape(&self) -> &[usize] {
62        &self.bb_shape
63    }
64
65    fn is_writable(&self) -> bool {
66        false
67    }
68
69    fn get_at(&self, position: &[usize]) -> Result<T, LatticeError> {
70        if position.len() != self.bb_shape.len() {
71            return Err(LatticeError::NdimMismatch {
72                expected: self.bb_shape.len(),
73                got: position.len(),
74            });
75        }
76
77        // Translate to parent coordinates.
78        let parent_pos: Vec<usize> = position
79            .iter()
80            .zip(self.bb_start.iter())
81            .map(|(&p, &s)| p + s)
82            .collect();
83
84        if self.region.contains(&parent_pos) {
85            self.parent.get_at(&parent_pos)
86        } else {
87            Ok(T::default_value())
88        }
89    }
90
91    fn get_slice(
92        &self,
93        start: &[usize],
94        shape: &[usize],
95        stride: &[usize],
96    ) -> Result<ArrayD<T>, LatticeError> {
97        let ndim = self.bb_shape.len();
98        if start.len() != ndim || shape.len() != ndim || stride.len() != ndim {
99            return Err(LatticeError::NdimMismatch {
100                expected: ndim,
101                got: start.len(),
102            });
103        }
104
105        // Build the result array.
106        let mut result = ArrayD::from_elem(IxDyn(shape), T::default_value());
107
108        for (idx, val) in result.indexed_iter_mut() {
109            let sub_pos: Vec<usize> = idx
110                .slice()
111                .iter()
112                .zip(start.iter())
113                .zip(stride.iter())
114                .map(|((&i, &s), &st)| s + i * st)
115                .collect();
116
117            let parent_pos: Vec<usize> = sub_pos
118                .iter()
119                .zip(self.bb_start.iter())
120                .map(|(&p, &s)| p + s)
121                .collect();
122
123            if self.region.contains(&parent_pos) {
124                *val = self.parent.get_at(&parent_pos)?;
125            }
126        }
127
128        Ok(result)
129    }
130}

219impl<'a, T: LatticeElement, L: LatticeMut<T>> LatticeMut<T> for SubLatticeMut<'a, T, L> {
220    fn put_at(&mut self, value: T, position: &[usize]) -> Result<(), LatticeError> {
221        let parent_pos: Vec<usize> = position
222            .iter()
223            .zip(self.bb_start.iter())
224            .map(|(&p, &s)| p + s)
225            .collect();
226        if self.region.contains(&parent_pos) {
227            self.parent.put_at(value, &parent_pos)
228        } else {
229            Ok(()) // silently ignore writes outside region
230        }
231    }
232
233    fn put_slice(&mut self, data: &ArrayD<T>, start: &[usize]) -> Result<(), LatticeError> {
234        // Write each element individually, checking region membership.
235        for (idx, val) in data.indexed_iter() {
236            let sub_pos: Vec<usize> = idx
237                .slice()
238                .iter()
239                .zip(start.iter())
240                .map(|(&i, &s)| i + s)
241                .collect();
242            let parent_pos: Vec<usize> = sub_pos
243                .iter()
244                .zip(self.bb_start.iter())
245                .map(|(&p, &s)| p + s)
246                .collect();
247            if self.region.contains(&parent_pos) {
248                self.parent.put_at(val.clone(), &parent_pos)?;
249            }
250        }
251        Ok(())
252    }
253
254    fn set(&mut self, value: T) -> Result<(), LatticeError> {
255        // Set only pixels within the region.
256        for axis_sizes in IterPositions::new(&self.bb_shape) {
257            let parent_pos: Vec<usize> = axis_sizes
258                .iter()
259                .zip(self.bb_start.iter())
260                .map(|(&p, &s)| p + s)
261                .collect();
262            if self.region.contains(&parent_pos) {
263                self.parent.put_at(value.clone(), &parent_pos)?;
264            }
265        }
266        Ok(())
267    }
268}

270/// Iterator over all positions in a given shape.
271struct IterPositions {
272    shape: Vec<usize>,
273    current: Vec<usize>,
274    done: bool,
275}
276
277impl IterPositions {
278    fn new(shape: &[usize]) -> Self {
279        let done = shape.contains(&0);
280        Self {
281            shape: shape.to_vec(),
282            current: vec![0; shape.len()],
283            done,
284        }
285    }
286}
287
288impl Iterator for IterPositions {
289    type Item = Vec<usize>;
290
291    fn next(&mut self) -> Option<Self::Item> {
292        if self.done {
293            return None;
294        }
295
296        let result = self.current.clone();
297
298        // Advance.
299        for axis in 0..self.shape.len() {
300            self.current[axis] += 1;
301            if self.current[axis] < self.shape[axis] {
302                return Some(result);
303            }
304            self.current[axis] = 0;
305        }
306        self.done = true;
307        Some(result)
308    }
309}

311#[cfg(test)]
312mod tests {
313    use super::*;
314    use crate::{ArrayLattice, LCBox};
315    use ndarray::ArrayD;
316
317    #[test]
318    fn sub_lattice_read() {
319        let data = ArrayD::from_shape_fn(IxDyn(&[10, 10]), |idx| (idx[0] * 10 + idx[1]) as f64);
320        let lat = ArrayLattice::new(data);
321        let region = LCBox::new(vec![2, 3], vec![5, 7], vec![10, 10]);
322        let sub = SubLattice::new(&lat, Box::new(region));
323
324        assert_eq!(sub.shape(), &[4, 5]);
325        assert_eq!(sub.get_at(&[0, 0]).unwrap(), 23.0); // lat[2, 3]
326        assert_eq!(sub.get_at(&[3, 4]).unwrap(), 57.0); // lat[5, 7]
327    }
328
329    #[test]
330    fn sub_lattice_get_slice() {
331        let data = ArrayD::from_shape_fn(IxDyn(&[10, 10]), |idx| (idx[0] * 10 + idx[1]) as f64);
332        let lat = ArrayLattice::new(data);
333        let region = LCBox::new(vec![0, 0], vec![4, 4], vec![10, 10]);
334        let sub = SubLattice::new(&lat, Box::new(region));
335
336        let slice = sub.get_slice(&[1, 1], &[2, 2], &[1, 1]).unwrap();
337        assert_eq!(slice.shape(), &[2, 2]);
338        assert_eq!(slice[IxDyn(&[0, 0])], 11.0); // lat[1, 1]
339    }
340
341    #[test]
342    fn sub_lattice_mut_write() {
343        let mut lat = ArrayLattice::<f64>::zeros(vec![10, 10]);
344        let region = LCBox::new(vec![2, 2], vec![4, 4], vec![10, 10]);
345        {
346            let mut sub = SubLatticeMut::new(&mut lat, Box::new(region));
347            sub.put_at(99.0, &[0, 0]).unwrap(); // writes to lat[2, 2]
348            sub.put_at(88.0, &[2, 2]).unwrap(); // writes to lat[4, 4]
349        }
350        assert_eq!(lat.get_at(&[2, 2]).unwrap(), 99.0);
351        assert_eq!(lat.get_at(&[4, 4]).unwrap(), 88.0);
352        assert_eq!(lat.get_at(&[0, 0]).unwrap(), 0.0); // untouched
353    }
354
355    #[test]
356    fn sub_lattice_mut_set() {
357        let mut lat = ArrayLattice::<f64>::zeros(vec![6, 6]);
358        let region = LCBox::new(vec![1, 1], vec![2, 2], vec![6, 6]);
359        {
360            let mut sub = SubLatticeMut::new(&mut lat, Box::new(region));
361            sub.set(5.0).unwrap();
362        }
363        // Only [1..3, 1..3] should be set.
364        assert_eq!(lat.get_at(&[1, 1]).unwrap(), 5.0);
365        assert_eq!(lat.get_at(&[2, 2]).unwrap(), 5.0);
366        assert_eq!(lat.get_at(&[0, 0]).unwrap(), 0.0);
367        assert_eq!(lat.get_at(&[3, 3]).unwrap(), 0.0);
368    }
