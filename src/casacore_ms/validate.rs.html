2//! Validation logic for MeasurementSet structure.
3//!
4//! Checks that all required subtables are present as table keywords,
5//! all required columns exist with correct data types, that required
6//! `QuantumUnits` / `MEASINFO.type` metadata is present, and that the
7//! MS_VERSION keyword is present.

19/// Validation issue found during MS validation.
20#[derive(Debug, Clone)]
21pub enum ValidationIssue {
22    /// A required subtable keyword is missing.
23    MissingSubtable(SubtableId),
24    /// A required column is missing from a table.
25    MissingColumn {
26        /// The table or subtable name.
27        table_name: String,
28        /// The missing column name.
29        column_name: String,
30    },
31    /// A column exists but has the wrong data type.
32    WrongColumnType {
33        /// The table or subtable name.
34        table_name: String,
35        /// The column name.
36        column_name: String,
37        /// Expected type description.
38        expected: String,
39        /// Actual type description.
40        found: String,
41    },
42    /// A required `QuantumUnits` keyword is missing.
43    MissingQuantumUnits {
44        /// The table or subtable name.
45        table_name: String,
46        /// The column name.
47        column_name: String,
48    },
49    /// A `QuantumUnits` keyword exists but has the wrong value.
50    WrongQuantumUnits {
51        /// The table or subtable name.
52        table_name: String,
53        /// The column name.
54        column_name: String,
55        /// Expected unit vector.
56        expected: Vec<String>,
57        /// Actual unit vector.
58        found: Vec<String>,
59    },
60    /// A required `MEASINFO` keyword is missing.
61    MissingMeasureInfo {
62        /// The table or subtable name.
63        table_name: String,
64        /// The column name.
65        column_name: String,
66    },
67    /// A `MEASINFO.type` value exists but does not match the MS schema.
68    WrongMeasureType {
69        /// The table or subtable name.
70        table_name: String,
71        /// The column name.
72        column_name: String,
73        /// Expected measure type.
74        expected: String,
75        /// Actual measure type.
76        found: String,
77    },
78    /// The MS_VERSION keyword is missing.
79    MissingMsVersion,
80}
81
82impl std::fmt::Display for ValidationIssue {
83    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
84        match self {
85            ValidationIssue::MissingSubtable(id) => {
86                write!(f, "required subtable '{}' missing", id.name())
87            }
88            ValidationIssue::MissingColumn {
89                table_name,
90                column_name,
91            } => write!(f, "required column '{column_name}' missing in {table_name}"),
92            ValidationIssue::WrongColumnType {
93                table_name,
94                column_name,
95                expected,
96                found,
97            } => write!(
98                f,
99                "column '{column_name}' in {table_name}: expected {expected}, found {found}"
100            ),
101            ValidationIssue::MissingQuantumUnits {
102                table_name,
103                column_name,
104            } => write!(
105                f,
106                "column '{column_name}' in {table_name}: missing QuantumUnits"
107            ),
108            ValidationIssue::WrongQuantumUnits {
109                table_name,
110                column_name,
111                expected,
112                found,
113            } => write!(
114                f,
115                "column '{column_name}' in {table_name}: expected QuantumUnits {:?}, found {:?}",
116                expected, found
117            ),
118            ValidationIssue::MissingMeasureInfo {
119                table_name,
120                column_name,
121            } => write!(
122                f,
123                "column '{column_name}' in {table_name}: missing MEASINFO"
124            ),
125            ValidationIssue::WrongMeasureType {
126                table_name,
127                column_name,
128                expected,
129                found,
130            } => write!(
131                f,
132                "column '{column_name}' in {table_name}: expected MEASINFO.type {expected}, found {found}"
133            ),
134            ValidationIssue::MissingMsVersion => write!(f, "MS_VERSION keyword missing"),
135        }
136    }
137}

139/// Validate the main table has all required subtable keywords.
140pub fn validate_subtable_keywords(
141    main_table: &Table,
142    subtables: &HashMap<SubtableId, Table>,
143) -> Vec<ValidationIssue> {
144    let mut issues = Vec::new();
145    for id in SubtableId::ALL_REQUIRED {
146        if !subtables.contains_key(id) {
147            issues.push(ValidationIssue::MissingSubtable(*id));
148        }
149    }
150    // Check MS_VERSION keyword
151    if main_table.keywords().get("MS_VERSION").is_none() {
152        issues.push(ValidationIssue::MissingMsVersion);
153    }
154    issues
155}

157/// Validate that a table has all required columns with correct data types.
158pub fn validate_columns(
159    table: &Table,
160    table_name: &str,
161    required_columns: &[ColumnDef],
162) -> Vec<ValidationIssue> {
163    let mut issues = Vec::new();
164    let schema = match table.schema() {
165        Some(s) => s,
166        None => {
167            // No schema means we can't validate column types — just check by
168            // attempting to access rows (if any exist).
169            return issues;
170        }
171    };
172
173    for col_def in required_columns {
174        match schema.column(col_def.name) {
175            None => {
176                issues.push(ValidationIssue::MissingColumn {
177                    table_name: table_name.to_string(),
178                    column_name: col_def.name.to_string(),
179                });
180            }
181            Some(col_schema) => {
182                // Check data type
183                if let Some(expected_type) = col_schema.data_type() {
184                    if expected_type != col_def.data_type {
185                        issues.push(ValidationIssue::WrongColumnType {
186                            table_name: table_name.to_string(),
187                            column_name: col_def.name.to_string(),
188                            expected: format!("{:?}", col_def.data_type),
189                            found: format!("{expected_type:?}"),
190                        });
191                    }
192                }
193            }
194        }
195    }
196    issues
197}

199/// Validate that required column keywords match casacore's MS expectations.
200pub fn validate_column_metadata(
201    table: &Table,
202    table_name: &str,
203    required_columns: &[ColumnDef],
204) -> Vec<ValidationIssue> {
205    let mut issues = Vec::new();
206
207    for col_def in required_columns {
208        let col_keywords = table.column_keywords(col_def.name);
209
210        if let Some(expected_units) = quantum_units_for(col_def) {
211            match col_keywords.and_then(|keywords| keywords.get("QuantumUnits")) {
212                Some(Value::Array(ArrayValue::String(units))) => {
213                    let found: Vec<String> = units.iter().cloned().collect();
214                    if found != expected_units {
215                        issues.push(ValidationIssue::WrongQuantumUnits {
216                            table_name: table_name.to_string(),
217                            column_name: col_def.name.to_string(),
218                            expected: expected_units,
219                            found,
220                        });
221                    }
222                }
223                _ => issues.push(ValidationIssue::MissingQuantumUnits {
224                    table_name: table_name.to_string(),
225                    column_name: col_def.name.to_string(),
226                }),
227            }
228        }
229
230        if let Some(expected_measure_type) = measure_type_name_for(col_def) {
231            match col_keywords.and_then(|keywords| keywords.get("MEASINFO")) {
232                Some(Value::Record(measinfo)) => match measinfo.get("type") {
233                    Some(Value::Scalar(ScalarValue::String(found_measure_type))) => {
234                        if found_measure_type != &expected_measure_type {
235                            issues.push(ValidationIssue::WrongMeasureType {
236                                table_name: table_name.to_string(),
237                                column_name: col_def.name.to_string(),
238                                expected: expected_measure_type,
239                                found: found_measure_type.clone(),
240                            });
241                        }
242                    }
243                    _ => issues.push(ValidationIssue::MissingMeasureInfo {
244                        table_name: table_name.to_string(),
245                        column_name: col_def.name.to_string(),
246                    }),
247                },
248                _ => issues.push(ValidationIssue::MissingMeasureInfo {
249                    table_name: table_name.to_string(),
250                    column_name: col_def.name.to_string(),
251                }),
252            }
253        }
254    }
255
256    issues
257}

259/// Full validation of an MS: checks main table + all present subtables.
260pub fn validate_ms(
261    main_table: &Table,
262    subtables: &HashMap<SubtableId, Table>,
263) -> MsResult<Vec<ValidationIssue>> {
264    let mut all_issues = Vec::new();
265
266    // Check subtable presence and MS_VERSION
267    all_issues.extend(validate_subtable_keywords(main_table, subtables));
268
269    // Check main table columns
270    all_issues.extend(validate_columns(
271        main_table,
272        "MAIN",
273        schema::main_table::REQUIRED_COLUMNS,
274    ));
275    all_issues.extend(validate_column_metadata(
276        main_table,
277        "MAIN",
278        schema::main_table::REQUIRED_COLUMNS,
279    ));
280
281    // Check each subtable's required columns
282    for (id, table) in subtables {
283        let required = schema::required_columns(*id);
284        all_issues.extend(validate_columns(table, id.name(), required));
285        all_issues.extend(validate_column_metadata(table, id.name(), required));
286    }
287
288    Ok(all_issues)
