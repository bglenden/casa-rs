2//! Limited `bandpass` support for the first `B Jones` solver wave.
3//!
4//! This module intentionally implements only a narrow, testable slice:
5//!
6//! - `bandtype='B'`
7//! - `solint='inf'`
8//! - explicit reference antenna
9//! - prior gain-table preapply
10//! - point-source Stokes-I sky model (`smodel=[I,0,0,0]`)
11//!
12//! The acceptance contract is downstream behavior: the resulting caltable
13//! should be CASA-compatible on disk and should yield corrected visibilities
14//! close to CASA's own `bandpass` when applied after the same prior gains.

256/// Solve a limited `bandpass` request from an on-disk MeasurementSet path.
257pub fn solve_bandpass_from_path(
258    path: impl AsRef<Path>,
259    request: &BandpassSolveRequest,
260) -> Result<BandpassSolveReport, BandpassSolveError> {
261    let path = path.as_ref().to_path_buf();
262    let ms =
263        MeasurementSet::open(&path).map_err(|source| BandpassSolveError::OpenMeasurementSet {
264            path: path.display().to_string(),
265            source,
266        })?;
267    solve_bandpass(&ms, request)
268}

270/// Solve a limited `bandpass` request from an already-open MeasurementSet.
271pub fn solve_bandpass(
272    ms: &MeasurementSet,
273    request: &BandpassSolveRequest,
274) -> Result<BandpassSolveReport, BandpassSolveError> {
275    validate_bandpass_request(request)?;
276    let refant_id = resolve_refant(ms, &request.refant).map_err(map_refant_error)?;
277    let available_antennas = all_antenna_ids(ms)?;
278    let rows = collect_selected_rows(ms, &request.selection).map_err(map_collect_rows_error)?;
279    let preapplied_rows = load_preapplied_rows(ms, request)?;
280    let groups = build_bandpass_groups(
281        ms,
282        &rows,
283        preapplied_rows.as_ref(),
284        request.smodel[0],
285        request.combine,
286    )?;
287
288    if groups.is_empty() {
289        return Err(BandpassSolveError::EmptySelection);
290    }
291
292    let mut solution_rows = Vec::new();
293    for group in groups.into_values() {
294        solution_rows.extend(solve_bandpass_group(
295            group,
296            &available_antennas,
297            refant_id,
298            request.combine,
299            request.normalize_average_amplitude,
300        )?);
301    }
302    match request.band_type {
303        BandpassType::B => write_bandpass_caltable(ms, request, refant_id, &solution_rows),
304        BandpassType::BPoly => write_bpoly_caltable(ms, request, refant_id, &solution_rows),
305    }
306}

308fn validate_bandpass_request(request: &BandpassSolveRequest) -> Result<(), BandpassSolveError> {
309    validate_smodel(request.smodel).map_err(map_validate_smodel_error)?;
310    if matches!(request.band_type, BandpassType::BPoly)
311        && (request.amplitude_degree > 31 || request.phase_degree > 31)
312    {
313        return Err(BandpassSolveError::UnsupportedConfiguration {
314            reason: "BPOLY degree must be <= 31".to_string(),
315        });
316    }
317    Ok(())
318}

320fn load_preapplied_rows(
321    ms: &MeasurementSet,
322    request: &BandpassSolveRequest,
323) -> Result<Option<HashMap<usize, EvaluatedApplyRow>>, BandpassSolveError> {
324    if request.prior_calibration_tables.is_empty() && !request.parang {
325        return Ok(None);
326    }
327    let plan = plan_apply(
328        ms,
329        &ApplyPlanRequest {
330            selection: request.selection.clone(),
331            apply_mode: crate::ApplyMode::CalFlag,
332            parang: request.parang,
333            calibration_tables: request.prior_calibration_tables.clone(),
334        },
335    )
336    .map_err(|source| BandpassSolveError::PriorCalibrationPlan {
337        source: Box::new(source),
338    })?;
339    evaluate_apply_rows(ms, &plan).map(Some).map_err(|source| {
340        BandpassSolveError::PriorCalibrationApply {
341            source: Box::new(source),
342        }
343    })
344}

354#[derive(Debug, Clone)]
355struct BandpassAccumulator {
356    field_id: i32,
357    spw_id: i32,
358    observation_id: i32,
359    min_time: f64,
360    max_time: f64,
361    total_interval: f64,
362    sample_rows: usize,
363    field_ids: BTreeSet<i32>,
364    scan_numbers: BTreeSet<i32>,
365    antenna_ids: BTreeSet<i32>,
366    channel_count: usize,
367    receptor_graphs: Vec<Vec<HashMap<(i32, i32), Complex32>>>,
368    receptor_weights: Vec<Vec<HashMap<(i32, i32), f32>>>,
369}
370
371impl BandpassAccumulator {
372    fn new(field_id: i32, spw_id: i32, observation_id: i32, channel_count: usize) -> Self {
373        let mut receptor_graphs = Vec::new();
374        let mut receptor_weights = Vec::new();
375        for _ in 0..2 {
376            let mut per_channel_graphs = Vec::new();
377            let mut per_channel_weights = Vec::new();
378            for _ in 0..channel_count {
379                per_channel_graphs.push(HashMap::new());
380                per_channel_weights.push(HashMap::new());
381            }
382            receptor_graphs.push(per_channel_graphs);
383            receptor_weights.push(per_channel_weights);
384        }
385
386        Self {
387            field_id,
388            spw_id,
389            observation_id,
390            min_time: f64::INFINITY,
391            max_time: f64::NEG_INFINITY,
392            total_interval: 0.0,
393            sample_rows: 0,
394            field_ids: BTreeSet::new(),
395            scan_numbers: BTreeSet::new(),
396            antenna_ids: BTreeSet::new(),
397            channel_count,
398            receptor_graphs,
399            receptor_weights,
400        }
401    }
402
403    fn observe(
404        &mut self,
405        ms: &MeasurementSet,
406        row: &SelectedSolveRow,
407        preapplied_row: Option<&EvaluatedApplyRow>,
408        stokes_i: f32,
409    ) -> Result<(), BandpassSolveError> {
410        let (data, flags) = match preapplied_row {
411            Some(row) => (&row.corrected_data, &row.flags),
412            None => {
413                let data = ms
414                    .main_table()
415                    .cell_accessor(row.row_index, "DATA")
416                    .and_then(|cell| cell.array())
417                    .map_err(|source| BandpassSolveError::OpenMeasurementSet {
418                        path: ms
419                            .path()
420                            .map(|path| path.display().to_string())
421                            .unwrap_or_else(|| "<in-memory>".to_string()),
422                        source: MsError::from(source),
423                    })?;
424                let flags = ms
425                    .main_table()
426                    .cell_accessor(row.row_index, "FLAG")
427                    .and_then(|cell| cell.array())
428                    .map_err(|source| BandpassSolveError::OpenMeasurementSet {
429                        path: ms
430                            .path()
431                            .map(|path| path.display().to_string())
432                            .unwrap_or_else(|| "<in-memory>".to_string()),
433                        source: MsError::from(source),
434                    })?;
435                (data, flags)
436            }
437        };
438        let weights = ms
439            .main_table()
440            .cell_accessor(row.row_index, "WEIGHT")
441            .and_then(|cell| cell.array())
442            .map_err(|source| BandpassSolveError::OpenMeasurementSet {
443                path: ms
444                    .path()
445                    .map(|path| path.display().to_string())
446                    .unwrap_or_else(|| "<in-memory>".to_string()),
447                source: MsError::from(source),
448            })?;
449        let correlation_types =
450            correlation_types_for_ddid(ms, row.data_desc_id).map_err(map_correlation_error)?;
451
452        let ArrayValue::Complex32(data) = data else {
453            return Err(BandpassSolveError::UnsupportedParameterShape {
454                path: "<measurement-set DATA>".to_string(),
455                shape: data.shape().to_vec(),
456            });
457        };
458        let ArrayValue::Bool(flags) = flags else {
459            return Err(BandpassSolveError::UnsupportedParameterShape {
460                path: "<measurement-set FLAG>".to_string(),
461                shape: flags.shape().to_vec(),
462            });
463        };
464        let ArrayValue::Float32(weights) = weights else {
465            return Err(BandpassSolveError::UnsupportedParameterShape {
466                path: "<measurement-set WEIGHT>".to_string(),
467                shape: weights.shape().to_vec(),
468            });
469        };
470        if data.ndim() != 2 || flags.ndim() != 2 || data.shape() != flags.shape() {
471            return Err(BandpassSolveError::UnsupportedParameterShape {
472                path: "<measurement-set row>".to_string(),
473                shape: data.shape().to_vec(),
474            });
475        }
476        if self.channel_count != data.shape()[1] {
477            return Err(BandpassSolveError::UnsupportedParameterShape {
478                path: "<measurement-set DATA>".to_string(),
479                shape: data.shape().to_vec(),
480            });
481        }
482        if weights.ndim() != 1 || weights.shape()[0] != data.shape()[0] {
483            return Err(BandpassSolveError::UnsupportedParameterShape {
484                path: "<measurement-set WEIGHT>".to_string(),
485                shape: weights.shape().to_vec(),
486            });
487        }
488
489        self.min_time = self.min_time.min(row.time_seconds);
490        self.max_time = self.max_time.max(row.time_seconds);
491        self.total_interval += row.interval_seconds;
492        self.sample_rows += 1;
493        self.field_ids.insert(row.field_id);
494        self.scan_numbers.insert(row.scan_number);
495        self.antenna_ids.insert(row.antenna1);
496        self.antenna_ids.insert(row.antenna2);
497
498        for corr_index in 0..data.shape()[0] {
499            let Some(receptors) = correlation_receptors(correlation_types[corr_index]) else {
500                return Err(BandpassSolveError::UnsupportedCorrelationLayout {
501                    data_desc_id: row.data_desc_id,
502                    correlation_types: correlation_types
503                        .iter()
504                        .map(|code| stokes_name(*code).to_string())
505                        .collect(),
506                });
507            };
508            if receptors.0 != receptors.1 {
509                continue;
510            }
511            let receptor = receptors.0;
512            let weight = weights[[corr_index]];
513            if weight <= 0.0 {
514                continue;
515            }
516            for chan_index in 0..self.channel_count {
517                if flags[[corr_index, chan_index]] {
518                    continue;
519                }
520                let sample = data[[corr_index, chan_index]] / Complex32::new(stokes_i, 0.0);
521                if sample.norm() <= f32::EPSILON {
522                    continue;
523                }
524                crate::solve::kernel::accumulate_edge(
525                    &mut self.receptor_graphs[receptor][chan_index],
526                    &mut self.receptor_weights[receptor][chan_index],
527                    row.antenna1,
528                    row.antenna2,
529                    weight,
530                    sample * Complex32::new(weight, 0.0),
531                );
532            }
533        }
534
535        Ok(())
536    }
537}

568fn build_bandpass_groups(
569    ms: &MeasurementSet,
570    rows: &[SelectedSolveRow],
571    preapplied_rows: Option<&HashMap<usize, EvaluatedApplyRow>>,
572    stokes_i: f32,
573    combine: BandpassSolveCombine,
574) -> Result<BTreeMap<BandpassGroupKey, BandpassAccumulator>, BandpassSolveError> {
575    let mut sorted_rows = rows.to_vec();
576    sorted_rows.sort_by_key(|row| {
577        (
578            row.field_id,
579            row.data_spw_id,
580            row.observation_id,
581            row.time_seconds.to_bits(),
582            row.row_index,
583        )
584    });
585
586    let mut groups = BTreeMap::<BandpassGroupKey, BandpassAccumulator>::new();
587    for row in sorted_rows {
588        if row.antenna1 == row.antenna2 {
589            continue;
590        }
591        let data = match preapplied_rows.and_then(|rows| rows.get(&row.row_index)) {
592            Some(preapplied) => &preapplied.corrected_data,
593            None => ms
594                .main_table()
595                .cell_accessor(row.row_index, "DATA")
596                .and_then(|cell| cell.array())
597                .map_err(|source| BandpassSolveError::OpenMeasurementSet {
598                    path: ms
599                        .path()
600                        .map(|path| path.display().to_string())
601                        .unwrap_or_else(|| "<in-memory>".to_string()),
602                    source: MsError::from(source),
603                })?,
604        };
605        let channel_count = data.shape().get(1).copied().ok_or_else(|| {
606            BandpassSolveError::UnsupportedParameterShape {
607                path: "<measurement-set DATA>".to_string(),
608                shape: data.shape().to_vec(),
609            }
610        })?;
611        let key = BandpassGroupKey {
612            field_id: if combine.fields { 0 } else { row.field_id },
613            spw_id: row.data_spw_id,
614            observation_id: row.observation_id,
615            scan_number: if combine.scans { 0 } else { row.scan_number },
616        };
617        let group = groups.entry(key).or_insert_with(|| {
618            BandpassAccumulator::new(
619                row.field_id,
620                row.data_spw_id,
621                row.observation_id,
622                channel_count,
623            )
624        });
625        let preapplied = preapplied_rows.and_then(|rows| rows.get(&row.row_index));
626        group.observe(ms, &row, preapplied, stokes_i)?;
627    }
628    Ok(groups)
629}

631fn solve_bandpass_group(
632    group: BandpassAccumulator,
633    available_antennas: &BTreeSet<i32>,
634    refant_id: i32,
635    combine: BandpassSolveCombine,
636    normalize_average_amplitude: bool,
637) -> Result<Vec<BandpassSolutionRow>, BandpassSolveError> {
638    let averaged_time = (group.min_time + group.max_time) / 2.0;
639    let averaged_interval = group.total_interval / group.sample_rows.max(1) as f64;
640    let scan_number = *group.scan_numbers.iter().next().unwrap_or(&0);
641
642    let mut solved_by_receptor = Vec::new();
643    for receptor in 0..2 {
644        let mut channel_solutions = Vec::new();
645        for chan_index in 0..group.channel_count {
646            channel_solutions.push(
647                solve_graph(
648                    &group.receptor_graphs[receptor][chan_index],
649                    &group.receptor_weights[receptor][chan_index],
650                    &HashMap::new(),
651                    GainSolveMode::AmplitudePhase,
652                    SolveGraphOptions {
653                        refant_id,
654                        min_baselines_per_antenna: 0,
655                    },
656                )
657                .map_err(map_solve_graph_error)?,
658            );
659        }
660        solved_by_receptor.push(channel_solutions);
661    }
662
663    let mut antenna_ids = available_antennas.clone();
664    antenna_ids.extend(group.antenna_ids);
665    antenna_ids.insert(refant_id);
666
667    let output_field_ids = if combine.fields && !combine.scans {
668        let mut fields = group.field_ids.iter().copied().collect::<Vec<_>>();
669        fields.sort_unstable();
670        fields
671    } else {
672        vec![group.field_id]
673    };
674
675    let mut solution_rows = Vec::new();
676    for field_id in output_field_ids {
677        for antenna_id in antenna_ids.iter().copied() {
678            let mut gains = Vec::with_capacity(2 * group.channel_count);
679            let mut flags = Vec::with_capacity(2 * group.channel_count);
680            for channel_solutions in solved_by_receptor[0]
681                .iter()
682                .zip(solved_by_receptor[1].iter())
683            {
684                for solved in [channel_solutions.0, channel_solutions.1] {
685                    gains.push(
686                        *solved
687                            .gains
688                            .get(&antenna_id)
689                            .unwrap_or(&Complex32::new(1.0, 0.0)),
690                    );
691                    flags.push(antenna_id != refant_id && !solved.reachable.contains(&antenna_id));
692                }
693            }
694            solution_rows.push(BandpassSolutionRow {
695                time_seconds: averaged_time,
696                interval_seconds: averaged_interval,
697                field_id,
698                spw_id: group.spw_id,
699                antenna_id,
700                scan_number,
701                observation_id: group.observation_id,
702                channel_count: group.channel_count,
703                gains,
704                flags,
705            });
706        }
707    }
708
709    if normalize_average_amplitude {
710        Ok(normalize_bandpass_rows(solution_rows, group.channel_count))
711    } else {
712        Ok(solution_rows)
713    }
714}

716fn normalize_bandpass_rows(
717    mut rows: Vec<BandpassSolutionRow>,
718    channel_count: usize,
719) -> Vec<BandpassSolutionRow> {
720    for row in &mut rows {
721        normalize_bandpass_row(row, channel_count);
722    }
723    rows

726fn normalize_bandpass_row(row: &mut BandpassSolutionRow, channel_count: usize) {
727    if channel_count == 0 {
728        return;
729    }
730    for receptor in 0..2 {
731        let mut coherent_sum = Complex32::new(0.0, 0.0);
732        let mut power_sum = 0.0_f32;
733        let mut good_count = 0usize;
734        for channel in 0..channel_count {
735            let index = receptor + 2 * channel;
736            if row.flags.get(index).copied().unwrap_or(false) {
737                continue;
738            }
739            let gain = row.gains[index];
740            let amplitude = gain.norm();
741            if amplitude <= f32::EPSILON {
742                continue;
743            }
744            coherent_sum += gain / Complex32::new(amplitude, 0.0);
745            power_sum += amplitude * amplitude;
746            good_count += 1;
747        }
748        if good_count <= 1 {
749            continue;
750        }
751        let amplitude_factor = (power_sum / good_count as f32).sqrt();
752        if amplitude_factor <= f32::EPSILON {
753            continue;
754        }
755        let phase_factor = if coherent_sum.norm() > f32::EPSILON {
756            coherent_sum / Complex32::new(coherent_sum.norm(), 0.0)
757        } else {
758            Complex32::new(1.0, 0.0)
759        };
760        let normalization = phase_factor * Complex32::new(amplitude_factor, 0.0);
761        for channel in 0..channel_count {
762            let index = receptor + 2 * channel;
763            row.gains[index] /= normalization;
764        }
765    }
766}

1408fn fit_bpoly_row(
1409    row: &BandpassSolutionRow,
1410    channel_frequencies_hz: &[f64],
1411    amplitude_degree: usize,
1412    phase_degree: usize,
1413) -> Result<BPolySolutionRow, BandpassSolveError> {
1414    if channel_frequencies_hz.len() != row.channel_count || row.channel_count == 0 {
1415        return Err(BandpassSolveError::UnsupportedParameterShape {
1416            path: "BPOLY fit channel grid".to_string(),
1417            shape: vec![channel_frequencies_hz.len(), row.channel_count],
1418        });
1419    }
1420    let valid_domain_hz = [
1421        channel_frequencies_hz.first().copied().unwrap_or(0.0),
1422        channel_frequencies_hz.last().copied().unwrap_or(0.0),
1423    ];
1424    let receptor_count = 2;
1425    let mut amp_coefficients = Vec::with_capacity(receptor_count);
1426    let mut phase_coefficients = Vec::with_capacity(receptor_count);
1427    for receptor in 0..receptor_count {
1428        let mut log_amplitudes = Vec::with_capacity(row.channel_count);
1429        let mut phases = Vec::with_capacity(row.channel_count);
1430        for channel in 0..row.channel_count {
1431            let gain = row.gains[receptor + receptor_count * channel];
1432            log_amplitudes.push(gain.norm().max(f32::EPSILON).ln() as f64);
1433            phases.push((gain.im as f64).atan2(gain.re as f64));
1434        }
1435        let unwrapped_phases = unwrap_phases(&phases);
1436        amp_coefficients.push(fit_legacy_bpoly_coefficients(
1437            channel_frequencies_hz,
1438            &log_amplitudes,
1439            valid_domain_hz,
1440            amplitude_degree,
1441            row,
1442        )?);
1443        phase_coefficients.push(fit_legacy_bpoly_coefficients(
1444            channel_frequencies_hz,
1445            &unwrapped_phases,
1446            valid_domain_hz,
1447            phase_degree,
1448            row,
1449        )?);
1450    }
1451    Ok(BPolySolutionRow {
1452        time_seconds: row.time_seconds,
1453        interval_seconds: row.interval_seconds,
1454        field_id: row.field_id,
1455        spw_id: row.spw_id,
1456        antenna_id: row.antenna_id,
1457        valid_domain_hz,
1458        scan_number: row.scan_number,
1459        observation_id: row.observation_id,
1460        scale_factor: Complex32::new(1.0, 0.0),
1461        amp_coefficients,
1462        phase_coefficients,
1463    })
1464}

1466fn fit_legacy_bpoly_coefficients(
1467    channel_frequencies_hz: &[f64],
1468    values: &[f64],
1469    valid_domain_hz: [f64; 2],
1470    degree: usize,
1471    row: &BandpassSolutionRow,
1472) -> Result<Vec<f64>, BandpassSolveError> {
1473    let coefficient_count = degree
1474        .saturating_add(1)
1475        .min(channel_frequencies_hz.len().max(1));
1476    if coefficient_count == 0 {
1477        return Ok(Vec::new());
1478    }
1479
1480    let mut basis = vec![0.0_f64; coefficient_count];
1481    let mut rows = Vec::with_capacity(channel_frequencies_hz.len());
1482    for (frequency_hz, value) in channel_frequencies_hz
1483        .iter()
1484        .copied()
1485        .zip(values.iter().copied())
1486    {
1487        legacy_bpoly_basis(
1488            valid_domain_hz[0],
1489            valid_domain_hz[1],
1490            frequency_hz,
1491            &mut basis,
1492        );
1493        rows.push((basis.clone(), value, 1.0));
1494    }
1495    solve_weighted_least_squares(&rows, coefficient_count).ok_or_else(|| {
1496        BandpassSolveError::FitPolynomial {
1497            antenna_id: row.antenna_id,
1498            field_id: row.field_id,
1499            spw_id: row.spw_id,
1500            reason: "least-squares system was singular".to_string(),
1501        }
1502    })
1503}

1505fn legacy_bpoly_basis(x_start: f64, x_end: f64, x: f64, basis: &mut [f64]) {
1506    if basis.is_empty() {
1507        return;
1508    }
1509    basis.fill(0.0);
1510    basis[0] = 0.5_f64;
1511    if basis.len() == 1 || (x_end - x_start).abs() <= f64::EPSILON {
1512        return;
1513    }
1514    let xcap = ((x - x_start) - (x_end - x)) / (x_end - x_start);
1515    let mut t_prev = 1.0_f64;
1516    let mut t_curr = xcap;
1517    basis[1] = t_curr;
1518    for slot in basis.iter_mut().skip(2) {
1519        let t_next = 2.0_f64 * xcap * t_curr - t_prev;
1520        *slot = t_next;
1521        t_prev = t_curr;
1522        t_curr = t_next;
1523    }
1524}

1526fn unwrap_phases(phases: &[f64]) -> Vec<f64> {
1527    let mut unwrapped = Vec::with_capacity(phases.len());
1528    let mut offset = 0.0_f64;
1529    let mut previous = None;
1530    for phase in phases.iter().copied() {
1531        if let Some(previous_phase) = previous {
1532            let delta = phase - previous_phase;
1533            if delta > std::f64::consts::PI {
1534                offset -= 2.0_f64 * std::f64::consts::PI;
1535            } else if delta < -std::f64::consts::PI {
1536                offset += 2.0_f64 * std::f64::consts::PI;
1537            }
1538        }
1539        let adjusted = phase + offset;
1540        unwrapped.push(adjusted);
1541        previous = Some(phase);
1542    }
1543    unwrapped
1544}
