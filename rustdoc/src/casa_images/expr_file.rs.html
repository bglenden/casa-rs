2//! Persistence format for casacore-compatible expression image files.
3//!
4//! An expression image (`.imgexpr`) is a **directory** containing a single
5//! `imageexpr.json` manifest file.  This format is defined by C++ casacore
6//! (class `ImageExpr<T>`, `ImageOpener`).
7//!
8//! ## On-disk layout
9//!
10//! ```text
11//! my_expr.imgexpr/
12//! └── imageexpr.json
13//! ```
14//!
15//! ## JSON schema
16//!
17//! ```json
18//! {
19//!   "Version": 1,
20//!   "DataType": "float",
21//!   "ImageExpr": "'a.image' + 'b.image' * 2.0",
22//!   "MiscInfo": {}
23//! }
24//! ```
25//!
26//! | Field       | Type   | Description                                         |
27//! |-------------|--------|-----------------------------------------------------|
28//! | `Version`   | int    | Format version, currently `1`                       |
29//! | `DataType`  | string | Pixel type: `"float"`, `"double"`, `"complex"`, `"dcomplex"` |
30//! | `ImageExpr` | string | LEL expression string                               |
31//! | `MiscInfo`  | object | Miscellaneous metadata record (may be empty)        |
32//!
33//! ## Reopen semantics
34//!
35//! When an expression file is reopened, the expression string is parsed and
36//! compiled once; all source images referenced in it must still exist at the
37//! embedded paths. Repeated reads reuse that owned graph. Coordinates and
38//! shape are derived from the source images, not stored in the JSON.
39//!
40//! ## Cross-language interop
41//!
42//! Files written by [`save`] are directly openable by C++ casacore via
43//! `ImageOpener::openImageExpr()`, and files written by C++ are openable
44//! by [`open`].

64/// Returns the C++ casacore type string for a given `PrimitiveType`.
65///
66/// Matches `ValType::getTypeStr()` in C++.
67fn data_type_str(pt: PrimitiveType) -> &'static str {
68    match pt {
69        PrimitiveType::Float32 => "float",
70        PrimitiveType::Float64 => "double",
71        PrimitiveType::Complex32 => "complex",
72        PrimitiveType::Complex64 => "dcomplex",
73        _ => "float", // fallback, shouldn't happen for image types
74    }
75}
76
77/// Parses a C++ casacore type string into a `PrimitiveType`.
78fn parse_data_type(s: &str) -> Result<PrimitiveType, ImageError> {
79    match s.trim().to_lowercase().as_str() {
80        "float" => Ok(PrimitiveType::Float32),
81        "double" => Ok(PrimitiveType::Float64),
82        "complex" => Ok(PrimitiveType::Complex32),
83        "dcomplex" => Ok(PrimitiveType::Complex64),
84        other => Err(ImageError::InvalidMetadata(format!(
85            "unknown DataType in imageexpr.json: {other:?}"
86        ))),
87    }
88}

94/// A source image that may be native, pixel-type-converted, or a nested
95/// expression image.
96///
97/// Used internally by [`open`] to hold heterogeneous sources while the
98/// canonical graph captures them.
99enum SourceImage<T: ImageExprValue + PartialOrd + ExprValueConvert> {
100    /// Source image with matching pixel type `T`.
101    Native(PagedImage<T>),
102    /// Source image stored as f32 on disk, pixels converted to `T` on read.
103    ConvertedF32(PagedImage<f32>),
104    /// Source image stored as f64 on disk, pixels converted to `T` on read.
105    ConvertedF64(PagedImage<f64>),
106    /// A nested expression image (another `.imgexpr` file).
107    Expr(Box<ImageExpression<T>>),
108    /// Nested `.imgexpr` with f32 pixels converted to `T` on read.
109    ConvertedExprF32(Box<ImageExpression<f32>>),
110    /// Nested `.imgexpr` with f64 pixels converted to `T` on read.
111    ConvertedExprF64(Box<ImageExpression<f64>>),
112}

114impl<T: ImageExprValue + PartialOrd + ExprValueConvert> Lattice<T> for SourceImage<T> {
115    fn shape(&self) -> &[usize] {
116        match self {
117            Self::Native(img) => img.shape(),
118            Self::ConvertedF32(img) => img.shape(),
119            Self::ConvertedF64(img) => img.shape(),
120            Self::Expr(expr) => expr.shape(),
121            Self::ConvertedExprF32(expr) => expr.shape(),
122            Self::ConvertedExprF64(expr) => expr.shape(),
123        }
124    }
125
126    fn is_writable(&self) -> bool {
127        false
128    }
129
130    fn get_at(&self, position: &[usize]) -> Result<T, LatticeError> {
131        match self {
132            Self::Native(img) => Lattice::get_at(img, position),
133            Self::ConvertedF32(img) => {
134                let val = Lattice::get_at(img, position)?;
135                Ok(T::from_f64(val.to_f64()))
136            }
137            Self::ConvertedF64(img) => {
138                let val = Lattice::get_at(img, position)?;
139                Ok(T::from_f64(val.to_f64()))
140            }
141            Self::Expr(expr) => Lattice::get_at(&**expr, position),
142            Self::ConvertedExprF32(expr) => {
143                let val = expr
144                    .get_at(position)
145                    .map_err(|e| LatticeError::Table(e.to_string()))?;
146                Ok(T::from_f64(val.to_f64()))
147            }
148            Self::ConvertedExprF64(expr) => {
149                let val = expr
150                    .get_at(position)
151                    .map_err(|e| LatticeError::Table(e.to_string()))?;
152                Ok(T::from_f64(val.to_f64()))
153            }
154        }
155    }
156
157    fn get_slice(
158        &self,
159        start: &[usize],
160        shape: &[usize],
161        stride: &[usize],
162    ) -> Result<ArrayD<T>, LatticeError> {
163        match self {
164            Self::Native(img) => Lattice::get_slice(img, start, shape, stride),
165            Self::ConvertedF32(img) => {
166                let data = Lattice::get_slice(img, start, shape, stride)?;
167                Ok(data.mapv(|v| T::from_f64(v.to_f64())))
168            }
169            Self::ConvertedF64(img) => {
170                let data = Lattice::get_slice(img, start, shape, stride)?;
171                Ok(data.mapv(|v| T::from_f64(v.to_f64())))
172            }
173            Self::Expr(expr) => Lattice::get_slice(&**expr, start, shape, stride),
174            Self::ConvertedExprF32(expr) => {
175                let data = Lattice::get_slice(&**expr, start, shape, stride)?;
176                Ok(data.mapv(|v| T::from_f64(v.to_f64())))
177            }
178            Self::ConvertedExprF64(expr) => {
179                let data = Lattice::get_slice(&**expr, start, shape, stride)?;
180                Ok(data.mapv(|v| T::from_f64(v.to_f64())))
181            }
182        }
183    }
184}

247/// Opens a single source dependency, handling type mismatches and nested
248/// `.imgexpr` directories.
249fn open_source<T>(img_path: &Path) -> Result<SourceImage<T>, ImageError>
250where
251    T: ImageExprValue + PartialOrd + ExprValueConvert,
252{
253    // Nested .imgexpr directory — open recursively.
254    if is_image_expr(img_path) {
255        let inner_info = read_info(img_path)?;
256        return match inner_info.data_type {
257            pt if pt == T::PRIMITIVE_TYPE => Ok(SourceImage::Expr(Box::new(open::<T>(img_path)?))),
258            PrimitiveType::Float32 => Ok(SourceImage::ConvertedExprF32(Box::new(open::<f32>(
259                img_path,
260            )?))),
261            PrimitiveType::Float64 => Ok(SourceImage::ConvertedExprF64(Box::new(open::<f64>(
262                img_path,
263            )?))),
264            other => Err(ImageError::InvalidMetadata(format!(
265                "cannot convert nested .imgexpr {:?} to {:?}",
266                other,
267                T::PRIMITIVE_TYPE,
268            ))),
269        };
270    }
271
272    // Regular image — try matching type first.
273    match PagedImage::<T>::open(img_path) {
274        Ok(img) => return Ok(SourceImage::Native(img)),
275        Err(ImageError::InvalidMetadata(ref msg)) if msg.contains("pixel type mismatch") => {}
276        Err(e) => return Err(e),
277    }
278
279    // Type mismatch — try f32↔f64 conversion.
280    let actual = image_pixel_type(img_path)?;
281    match actual {
282        ImagePixelType::Float32 => Ok(SourceImage::ConvertedF32(PagedImage::<f32>::open(
283            img_path,
284        )?)),
285        ImagePixelType::Float64 => Ok(SourceImage::ConvertedF64(PagedImage::<f64>::open(
286            img_path,
287        )?)),
288        _ => Err(ImageError::InvalidMetadata(format!(
289            "cannot convert {:?} source image to {:?}",
290            actual,
291            T::PRIMITIVE_TYPE,
292        ))),
293    }
294}

300fn misc_info_to_json(record: &RecordValue) -> serde_json::Value {
301    let mut map = serde_json::Map::new();
302    for field in record.fields() {
303        let val = match &field.value {
304            Value::Scalar(s) => match s {
305                ScalarValue::Bool(v) => serde_json::Value::Bool(*v),
306                ScalarValue::Int32(v) => serde_json::json!(*v),
307                ScalarValue::Int64(v) => serde_json::json!(*v),
308                ScalarValue::Float32(v) => serde_json::json!(*v),
309                ScalarValue::Float64(v) => serde_json::json!(*v),
310                ScalarValue::String(v) => serde_json::Value::String(v.clone()),
311                _ => continue, // skip unsupported scalar types
312            },
313            Value::Record(sub) => misc_info_to_json(sub),
314            _ => continue, // skip arrays, table refs
315        };
316        map.insert(field.name.clone(), val);
317    }
318    serde_json::Value::Object(map)
319}
320
321fn json_to_misc_info(val: &serde_json::Value) -> RecordValue {
322    let mut record = RecordValue::default();
323    if let Some(obj) = val.as_object() {
324        for (key, v) in obj {
325            let field_val = match v {
326                serde_json::Value::Bool(b) => Value::Scalar(ScalarValue::Bool(*b)),
327                serde_json::Value::Number(n) => {
328                    if let Some(i) = n.as_i64() {
329                        if i >= i32::MIN as i64 && i <= i32::MAX as i64 {
330                            Value::Scalar(ScalarValue::Int32(i as i32))
331                        } else {
332                            Value::Scalar(ScalarValue::Int64(i))
333                        }
334                    } else if let Some(f) = n.as_f64() {
335                        Value::Scalar(ScalarValue::Float64(f))
336                    } else {
337                        continue;
338                    }
339                }
340                serde_json::Value::String(s) => Value::Scalar(ScalarValue::String(s.clone())),
341                serde_json::Value::Object(_) => Value::Record(json_to_misc_info(v)),
342                _ => continue, // skip arrays, null
343            };
344            record.push(RecordField::new(key, field_val));
345        }
346    }
347    record
348}

354/// Saves an expression image in casacore-compatible `.imgexpr` format.
355///
356/// Creates a directory at `path` containing `imageexpr.json`.
357///
358/// Corresponds to C++ `ImageExpr<T>::save()`.
359pub fn save(
360    path: impl AsRef<Path>,
361    expr_string: &str,
362    pixel_type: PrimitiveType,
363    misc_info: &RecordValue,
364) -> Result<(), ImageError> {
365    let dir = path.as_ref();
366    fs::create_dir_all(dir).map_err(|e| {
367        ImageError::InvalidMetadata(format!("cannot create .imgexpr directory {dir:?}: {e}"))
368    })?;
369
370    let json = serde_json::json!({
371        "Version": 1,
372        "DataType": data_type_str(pixel_type),
373        "ImageExpr": expr_string,
374        "MiscInfo": misc_info_to_json(misc_info),
375    });
376
377    let json_path = dir.join("imageexpr.json");
378    let content = serde_json::to_string_pretty(&json).map_err(|e| {
379        ImageError::InvalidMetadata(format!("cannot serialize imageexpr.json: {e}"))
380    })?;
381    fs::write(&json_path, content)
382        .map_err(|e| ImageError::InvalidMetadata(format!("cannot write {json_path:?}: {e}")))?;
383
384    Ok(())
385}

403/// Metadata read from an `imageexpr.json` file.
404#[derive(Debug, Clone)]
405pub struct ExprFileInfo {
406    /// Format version (expected: 1).
407    pub version: u32,
408    /// Pixel data type string (e.g. `"float"`).
409    pub data_type: PrimitiveType,
410    /// The LEL expression string.
411    pub expr_string: String,
412    /// Miscellaneous metadata.
413    pub misc_info: RecordValue,
414}
415
416/// Reads and parses the `imageexpr.json` from an expression image directory.
417pub fn read_info(path: impl AsRef<Path>) -> Result<ExprFileInfo, ImageError> {
418    let json_path = path.as_ref().join("imageexpr.json");
419    let content = fs::read_to_string(&json_path)
420        .map_err(|e| ImageError::InvalidMetadata(format!("cannot read {json_path:?}: {e}")))?;
421    let json: serde_json::Value = serde_json::from_str(&content)
422        .map_err(|e| ImageError::InvalidMetadata(format!("invalid JSON in {json_path:?}: {e}")))?;
423
424    let version = json.get("Version").and_then(|v| v.as_u64()).unwrap_or(1) as u32;
425
426    let data_type_s = json
427        .get("DataType")
428        .and_then(|v| v.as_str())
429        .ok_or_else(|| ImageError::InvalidMetadata("missing DataType in imageexpr.json".into()))?;
430    let data_type = parse_data_type(data_type_s)?;
431
432    let expr_string = json
433        .get("ImageExpr")
434        .and_then(|v| v.as_str())
435        .ok_or_else(|| ImageError::InvalidMetadata("missing ImageExpr in imageexpr.json".into()))?
436        .to_string();
437
438    let misc_info = json
439        .get("MiscInfo")
440        .map(json_to_misc_info)
441        .unwrap_or_default();
442
443    Ok(ExprFileInfo {
444        version,
445        data_type,
446        expr_string,
447        misc_info,
448    })
449}

455/// Opens an expression image from an `.imgexpr` directory.
456///
457/// Reads `imageexpr.json`, opens all source images referenced in the
458/// expression string, and returns the canonical owned [`ImageExpression`].
459///
460/// Source images may be:
461/// - `PagedImage<T>` with matching pixel type,
462/// - a pixel-type-converting wrapper (e.g. f32 source in an f64 expression),
463/// - a nested `.imgexpr` directory (recursively opened).
464///
465/// Image names in the expression are resolved relative to the parent
466/// directory of the `.imgexpr` path, matching C++ casacore behavior.
467///
468/// # Errors
469///
470/// Returns an error if the JSON is malformed, the pixel type doesn't match
471/// `T`, any referenced source image cannot be opened, or the expression
472/// string fails to parse.
473pub fn open<T>(path: impl AsRef<Path>) -> Result<ImageExpression<T>, ImageError>
474where
475    T: ImageExprValue + PartialOrd + ExprValueConvert,
476{
477    let path = path.as_ref();
478    let info = read_info(path)?;
479
480    if info.data_type != T::PRIMITIVE_TYPE {
481        return Err(ImageError::InvalidMetadata(format!(
482            "expression file has DataType {:?} but opened as {:?}",
483            info.data_type,
484            T::PRIMITIVE_TYPE,
485        )));
486    }
487
488    // Determine the base directory for resolving relative image paths.
489    let base_dir = path.parent().unwrap_or(Path::new("."));
490
491    // Extract image names from the expression.
492    let names = extract_image_names(&info.expr_string);
493
494    // Open each source image (native, converted, or nested .imgexpr).
495    let mut sources = Vec::new();
496    let mut source_names = Vec::new();
497    for name in &names {
498        let img_path = if Path::new(name).is_absolute() {
499            std::path::PathBuf::from(name)
500        } else {
501            base_dir.join(name)
502        };
503        let src = open_source::<T>(&img_path).map_err(|e| {
504            ImageError::InvalidMetadata(format!(
505                "cannot open source image {name:?} (resolved to {img_path:?}): {e}"
506            ))
507        })?;
508        sources.push(src);
509        source_names.push(name.clone());
510    }
511
512    // Derive shape and metadata from the first source image.
513    if sources.is_empty() {
514        return Err(ImageError::InvalidMetadata(
515            "expression references no images; cannot determine shape".into(),
516        ));
517    }
518    let first: &dyn ImageInterface<T> = &sources[0];
519    let shape = first.shape().to_vec();
520    let mut compiled = {
521        let mut images: HashMap<String, &dyn ImageInterface<T>> = HashMap::new();
522        for (name, source) in source_names.iter().zip(sources.iter()) {
523            images.insert(name.clone(), source);
524        }
525        let resolver = HashMapResolver(images);
526        expr_parser::parse_image_expr(&info.expr_string, &resolver)?
527    };
528    debug_assert_eq!(compiled.shape(), shape);
529    compiled.set_persisted_metadata(path.to_path_buf(), info.misc_info);
530    Ok(compiled)
531}

533/// Extracts image name tokens from a LEL expression string.
534///
535/// Scans for quoted strings (`'...'`, `"..."`) and bare identifiers that are
536/// not function calls.
537fn extract_image_names(expr: &str) -> Vec<String> {
538    let mut names = Vec::new();
539    let mut seen = std::collections::HashSet::new();
540    let chars: Vec<char> = expr.chars().collect();
541    let mut i = 0;
542
543    while i < chars.len() {
544        let ch = chars[i];
545        // Quoted path
546        if ch == '\'' || ch == '"' {
547            let quote = ch;
548            i += 1;
549            let start = i;
550            while i < chars.len() && chars[i] != quote {
551                i += 1;
552            }
553            let name: String = chars[start..i].iter().collect();
554            if !name.is_empty() && seen.insert(name.clone()) {
555                names.push(name);
556            }
557            if i < chars.len() {
558                i += 1; // skip closing quote
559            }
560        }
561        // Bare identifier that might be a filename.
562        // Skip 'e'/'E' that form part of scientific notation (e.g. 3.5e10).
563        else if (ch.is_ascii_alphabetic() || ch == '_')
564            && !((ch == 'e' || ch == 'E') && i > 0 && chars[i - 1].is_ascii_digit())
565        {
566            let start = i;
567            while i < chars.len()
568                && (chars[i].is_alphanumeric()
569                    || chars[i] == '_'
570                    || chars[i] == '.'
571                    || chars[i] == '/'
572                    || chars[i] == '~')
573            {
574                i += 1;
575            }
576            let token: String = chars[start..i].iter().collect();
577            // The parser treats bare identifiers as image names unless they are
578            // followed by `(`, in which case they are function calls.
579            if !is_function_call(&chars, i)
580                && is_image_reference(&token)
581                && seen.insert(token.clone())
582            {
583                names.push(token);
584            }
585        } else {
586            i += 1;
587        }
588    }
589    names
590}
591
592fn is_function_call(chars: &[char], mut idx: usize) -> bool {
593    while idx < chars.len() && chars[idx].is_ascii_whitespace() {
594        idx += 1;
595    }
596    idx < chars.len() && chars[idx] == '('
597}
598
599/// Returns `true` if a bare token should be treated as an image reference.
600///
601/// This mirrors the parser's bare-identifier handling: any non-numeric token
602/// is an image reference unless it is immediately followed by `(`.
603fn is_image_reference(token: &str) -> bool {
604    // Skip numeric literals
605    if token.parse::<f64>().is_ok() {
606        return false;
607    }
608    true
609}

817    #[test]
818    fn builder_parser_and_persisted_paths_share_one_evaluator_contract() {
819        use std::collections::HashMap;
820
821        use casa_coordinates::CoordinateSystem;
822        use ndarray::{ArrayD, IxDyn};
823
824        use crate::ImageExprBuilder;
825        use crate::expr_parser::{HashMapResolver, parse_image_expr};
826        use crate::image::ImageInterface;
827
828        let dir = tempfile::tempdir().unwrap();
829        let image_path = dir.path().join("src.image");
830        let mut image =
831            PagedImage::<f32>::create(vec![2, 3], CoordinateSystem::new(), &image_path).unwrap();
832        image
833            .put_slice(
834                &ArrayD::from_shape_vec(IxDyn(&[2, 3]), vec![-4.0, 1.0, 2.0, 3.0, 9.0, 16.0])
835                    .unwrap(),
836                &[0, 0],
837            )
838            .unwrap();
839        image.save().unwrap();
840
841        let condition = ImageExprBuilder::from_image(&image).unwrap().gt_scalar(2.0);
842        let if_true = ImageExprBuilder::from_image(&image)
843            .unwrap()
844            .abs()
845            .sqrt()
846            .add_scalar(3.0);
847        let if_false = ImageExprBuilder::from_image(&image)
848            .unwrap()
849            .multiply_scalar(2.0);
850        let built = ImageExprBuilder::iif(condition, if_true, if_false)
851            .unwrap()
852            .compile()
853            .unwrap();
854
855        let expression = "iif('src.image' > 2.0, sqrt(abs('src.image')) + 3.0, 'src.image' * 2.0)";
856        let mut sources = HashMap::new();
857        sources.insert("src.image".to_string(), &image as &dyn ImageInterface<f32>);
858        let parsed = parse_image_expr(expression, &HashMapResolver(sources)).unwrap();
859        let descriptor = dir.path().join("parity.imgexpr");
860        parsed.save_expr(&descriptor).unwrap();
861        let reopened = open::<f32>(&descriptor).unwrap();
862
863        let expected = built.get().unwrap();
864        assert_eq!(parsed.get().unwrap(), expected);
865        assert_eq!(reopened.get().unwrap(), expected);
866    }
