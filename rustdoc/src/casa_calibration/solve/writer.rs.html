27pub(crate) fn write_gain_caltable(
28    ms: &MeasurementSet,
29    request: &GainSolveRequest,
30    refant_id: i32,
31    rows: &[SolutionRow],
32) -> Result<GainSolveReport, GainSolveError> {
33    prepare_output_root(&request.output_table)?;
34
35    let schema = TableSchema::new(vec![
36        ColumnSchema::scalar(COL_TIME, casa_types::PrimitiveType::Float64),
37        ColumnSchema::scalar(COL_FIELD_ID, casa_types::PrimitiveType::Int32),
38        ColumnSchema::scalar(COL_SPECTRAL_WINDOW_ID, casa_types::PrimitiveType::Int32),
39        ColumnSchema::scalar(COL_ANTENNA1, casa_types::PrimitiveType::Int32),
40        ColumnSchema::scalar(COL_ANTENNA2, casa_types::PrimitiveType::Int32),
41        ColumnSchema::scalar(COL_INTERVAL, casa_types::PrimitiveType::Float64),
42        ColumnSchema::scalar(COL_SCAN_NUMBER, casa_types::PrimitiveType::Int32),
43        ColumnSchema::scalar(COL_OBSERVATION_ID, casa_types::PrimitiveType::Int32),
44        ColumnSchema::scalar(COL_ARRAY_ID, casa_types::PrimitiveType::Int32),
45        ColumnSchema::scalar(COL_TIME_EXTRA_PREC, casa_types::PrimitiveType::Float64),
46        ColumnSchema::array_variable(COL_CPARAM, casa_types::PrimitiveType::Complex32, Some(2)),
47        ColumnSchema::array_variable(COL_PARAMERR, casa_types::PrimitiveType::Float32, Some(2)),
48        ColumnSchema::array_variable(COL_FLAG, casa_types::PrimitiveType::Bool, Some(2)),
49        ColumnSchema::array_variable(COL_SNR, casa_types::PrimitiveType::Float32, Some(2)),
50        ColumnSchema::array_variable(COL_WEIGHT, casa_types::PrimitiveType::Float32, Some(2)),
51    ])
52    .expect("valid gain caltable schema");
53    let mut table = Table::with_schema(schema);
54    table.set_info(TableInfo {
55        table_type: TABLE_INFO_TYPE.to_string(),
56        sub_type: request.gain_type.vis_cal().to_string(),
57    });
58    table.keywords_mut().upsert(
59        KEY_PAR_TYPE,
60        Value::Scalar(ScalarValue::String("Complex".to_string())),
61    );
62    table.keywords_mut().upsert(
63        KEY_VIS_CAL,
64        Value::Scalar(ScalarValue::String(request.gain_type.vis_cal().to_string())),
65    );
66    table.keywords_mut().upsert(
67        KEY_MS_NAME,
68        Value::Scalar(ScalarValue::String(
69            ms.path()
70                .map(|path| path.display().to_string())
71                .unwrap_or_else(|| "<in-memory>".to_string()),
72        )),
73    );
74    table.keywords_mut().upsert(
75        KEY_POL_BASIS,
76        Value::Scalar(ScalarValue::String("unknown".to_string())),
77    );
78    table.keywords_mut().upsert(
79        KEY_CASA_VERSION,
80        Value::Scalar(ScalarValue::String("casa-rs".to_string())),
81    );
82    set_fixed_unit_keyword(&mut table, COL_TIME, &["s"]);
83    set_measinfo_keyword(&mut table, COL_TIME, "epoch", Some("UTC"));
84    set_fixed_unit_keyword(&mut table, COL_INTERVAL, &["s"]);
85    set_fixed_unit_keyword(&mut table, COL_TIME_EXTRA_PREC, &["s"]);
86    for name in STANDARD_SUBTABLE_KEYWORDS {
87        table.keywords_mut().upsert(
88            *name,
89            Value::table_ref(subtable_keyword_value(
90                &request.output_table,
91                &request.output_table.join(name),
92            )),
93        );
94    }
95
96    for row in rows {
97        table
98            .add_row(RecordValue::new(vec![
99                RecordField::new(
100                    COL_TIME,
101                    Value::Scalar(ScalarValue::Float64(row.time_seconds)),
102                ),
103                RecordField::new(
104                    COL_FIELD_ID,
105                    Value::Scalar(ScalarValue::Int32(row.field_id)),
106                ),
107                RecordField::new(
108                    COL_SPECTRAL_WINDOW_ID,
109                    Value::Scalar(ScalarValue::Int32(row.spw_id)),
110                ),
111                RecordField::new(
112                    COL_ANTENNA1,
113                    Value::Scalar(ScalarValue::Int32(row.antenna_id)),
114                ),
115                RecordField::new(
116                    COL_ANTENNA2,
117                    Value::Scalar(ScalarValue::Int32(row.refant_id)),
118                ),
119                RecordField::new(
120                    COL_INTERVAL,
121                    Value::Scalar(ScalarValue::Float64(row.interval_seconds)),
122                ),
123                RecordField::new(
124                    COL_SCAN_NUMBER,
125                    Value::Scalar(ScalarValue::Int32(row.scan_number)),
126                ),
127                RecordField::new(
128                    COL_OBSERVATION_ID,
129                    Value::Scalar(ScalarValue::Int32(row.observation_id)),
130                ),
131                RecordField::new(COL_ARRAY_ID, Value::Scalar(ScalarValue::Int32(0))),
132                RecordField::new(
133                    COL_TIME_EXTRA_PREC,
134                    Value::Scalar(ScalarValue::Float64(0.0)),
135                ),
136                RecordField::new(
137                    COL_CPARAM,
138                    Value::Array(ArrayValue::Complex32(
139                        ArrayD::from_shape_vec(IxDyn(&[row.gains.len(), 1]).f(), row.gains.clone())
140                            .expect("gain vector should reshape to receptor x channel"),
141                    )),
142                ),
143                RecordField::new(
144                    COL_PARAMERR,
145                    Value::Array(ArrayValue::Float32(
146                        ArrayD::from_shape_vec(
147                            IxDyn(&[row.gains.len(), 1]).f(),
148                            row.param_errors.clone(),
149                        )
150                        .expect("paramerr vector should reshape to receptor x channel"),
151                    )),
152                ),
153                RecordField::new(
154                    COL_FLAG,
155                    Value::Array(ArrayValue::Bool(
156                        ArrayD::from_shape_vec(IxDyn(&[row.flags.len(), 1]).f(), row.flags.clone())
157                            .expect("flag vector should reshape to receptor x channel"),
158                    )),
159                ),
160                RecordField::new(
161                    COL_SNR,
162                    Value::Array(ArrayValue::Float32(
163                        ArrayD::from_shape_vec(IxDyn(&[row.gains.len(), 1]).f(), row.snrs.clone())
164                            .expect("snr vector should reshape to receptor x channel"),
165                    )),
166                ),
167                RecordField::new(
168                    COL_WEIGHT,
169                    Value::Array(ArrayValue::Float32(
170                        ArrayD::from_shape_vec(
171                            IxDyn(&[row.gains.len(), 1]).f(),
172                            row.weights.clone(),
173                        )
174                        .expect("weight vector should reshape to receptor x channel"),
175                    )),
176                ),
177            ]))
178            .expect("insert gain solution row");
179    }
180
181    table
182        .save(
183            TableOptions::new(&request.output_table)
184                .with_data_manager(DataManagerKind::StandardStMan),
185        )
186        .map_err(|source| GainSolveError::SaveCalibrationTable {
187            path: request.output_table.display().to_string(),
188            source: Box::new(source),
189        })?;
190
191    for (id, name) in [
192        (SubtableId::Observation, "OBSERVATION"),
193        (SubtableId::Antenna, "ANTENNA"),
194        (SubtableId::Field, "FIELD"),
195        (SubtableId::History, "HISTORY"),
196    ] {
197        ms.subtable(id)
198            .expect("required subtable available")
199            .save(TableOptions::new(request.output_table.join(name)))
200            .map_err(|source| GainSolveError::CopySubtable {
201                subtable: name.to_string(),
202                path: request.output_table.display().to_string(),
203                source: Box::new(source),
204            })?;
205    }
206    write_gain_spectral_window_subtable(ms, &request.output_table, rows)?;
207
208    let field_ids = rows.iter().map(|row| row.field_id).collect::<BTreeSet<_>>();
209    let spw_ids = rows.iter().map(|row| row.spw_id).collect::<BTreeSet<_>>();
210    Ok(GainSolveReport {
211        output_table: request.output_table.clone(),
212        gain_type: request.gain_type,
213        refant_antenna_id: refant_id,
214        field_ids: field_ids.iter().copied().collect(),
215        spectral_window_ids: spw_ids.iter().copied().collect(),
216        solution_row_count: rows.len(),
217    })
218}

220pub(crate) fn prepare_output_root(path: &Path) -> Result<(), GainSolveError> {
221    if path.exists() {
222        fs::remove_dir_all(path)
223            .or_else(|_| fs::remove_file(path))
224            .map_err(|error| GainSolveError::PrepareOutput {
225                path: path.display().to_string(),
226                reason: error.to_string(),
227            })?;
228    }
229    if let Some(parent) = path.parent() {
230        fs::create_dir_all(parent).map_err(|error| GainSolveError::PrepareOutput {
231            path: path.display().to_string(),
232            reason: error.to_string(),
233        })?;
234    }
235    Ok(())
236}

238pub(crate) fn set_fixed_unit_keyword(table: &mut Table, column: &str, units: &[&str]) {
239    let mut keywords = table.column_keywords(column).cloned().unwrap_or_default();
240    keywords.upsert(
241        "QuantumUnits",
242        Value::Array(ArrayValue::from_string_vec(
243            units.iter().map(|unit| (*unit).to_string()).collect(),
244        )),
245    );
246    table.set_column_keywords(column, keywords);
247}

249pub(crate) fn set_measinfo_keyword(
250    table: &mut Table,
251    column: &str,
252    measure_type: &str,
253    measure_ref: Option<&str>,
254) {
255    let mut keywords = table.column_keywords(column).cloned().unwrap_or_default();
256    let mut fields = vec![RecordField::new(
257        "type",
258        Value::Scalar(ScalarValue::String(measure_type.to_string())),
259    )];
260    if let Some(measure_ref) = measure_ref {
261        fields.push(RecordField::new(
262            "Ref",
263            Value::Scalar(ScalarValue::String(measure_ref.to_string())),
264        ));
265    }
266    keywords.upsert("MEASINFO", Value::Record(RecordValue::new(fields)));
267    table.set_column_keywords(column, keywords);
268}

270fn write_gain_spectral_window_subtable(
271    ms: &MeasurementSet,
272    output_root: &Path,
273    rows: &[SolutionRow],
274) -> Result<(), GainSolveError> {
275    let destination = output_root.join("SPECTRAL_WINDOW");
276    ms.subtable(SubtableId::SpectralWindow)
277        .expect("required subtable available")
278        .save(TableOptions::new(&destination))
279        .map_err(|source| GainSolveError::CopySubtable {
280            subtable: "SPECTRAL_WINDOW".to_string(),
281            path: output_root.display().to_string(),
282            source: Box::new(source),
283        })?;
284
285    let solved_spw_ids = rows.iter().map(|row| row.spw_id).collect::<BTreeSet<_>>();
286    let mut table = Table::open(TableOptions::new(&destination)).map_err(|source| {
287        GainSolveError::CopySubtable {
288            subtable: "SPECTRAL_WINDOW".to_string(),
289            path: output_root.display().to_string(),
290            source: Box::new(source),
291        }
292    })?;
293
294    for spw_id in solved_spw_ids {
295        let row =
296            usize::try_from(spw_id).map_err(|_| GainSolveError::UnsupportedParameterShape {
297                path: "SPECTRAL_WINDOW row id".to_string(),
298                shape: vec![spw_id as usize],
299            })?;
300        let chan_freq = get_f64_array(&table, row, "CHAN_FREQ")?;
301        let chan_width = get_f64_array(&table, row, "CHAN_WIDTH")?;
302        let effective_bw = get_f64_array(&table, row, "EFFECTIVE_BW")?;
303        let resolution = get_f64_array(&table, row, "RESOLUTION")?;
304
305        if chan_freq.len() <= 1 {
306            continue;
307        }
308
309        let center_index = chan_freq.len() / 2;
310        table
311            .cell_accessor_mut(row, "NUM_CHAN")
312            .and_then(|mut cell| cell.set(Value::Scalar(ScalarValue::Int32(1))))
313            .map_err(|source| GainSolveError::CopySubtable {
314                subtable: "SPECTRAL_WINDOW".to_string(),
315                path: output_root.display().to_string(),
316                source: Box::new(source),
317            })?;
318        table
319            .cell_accessor_mut(row, "CHAN_FREQ")
320            .and_then(|mut cell| cell.set(Value::Array(f64_array(&[chan_freq[center_index]]))))
321            .map_err(|source| GainSolveError::CopySubtable {
322                subtable: "SPECTRAL_WINDOW".to_string(),
323                path: output_root.display().to_string(),
324                source: Box::new(source),
325            })?;
326        table
327            .cell_accessor_mut(row, "CHAN_WIDTH")
328            .and_then(|mut cell| {
329                cell.set(Value::Array(f64_array(&[chan_width.iter().copied().sum()])))
330            })
331            .map_err(|source| GainSolveError::CopySubtable {
332                subtable: "SPECTRAL_WINDOW".to_string(),
333                path: output_root.display().to_string(),
334                source: Box::new(source),
335            })?;
336        table
337            .cell_accessor_mut(row, "EFFECTIVE_BW")
338            .and_then(|mut cell| {
339                cell.set(Value::Array(f64_array(&[effective_bw
340                    .iter()
341                    .copied()
342                    .sum()])))
343            })
344            .map_err(|source| GainSolveError::CopySubtable {
345                subtable: "SPECTRAL_WINDOW".to_string(),
346                path: output_root.display().to_string(),
347                source: Box::new(source),
348            })?;
349        table
350            .cell_accessor_mut(row, "RESOLUTION")
351            .and_then(|mut cell| {
352                cell.set(Value::Array(f64_array(&[resolution.iter().copied().sum()])))
353            })
354            .map_err(|source| GainSolveError::CopySubtable {
355                subtable: "SPECTRAL_WINDOW".to_string(),
356                path: output_root.display().to_string(),
357                source: Box::new(source),
358            })?;
359    }
360
361    table
362        .save(TableOptions::new(&destination))
363        .map_err(|source| GainSolveError::CopySubtable {
364            subtable: "SPECTRAL_WINDOW".to_string(),
365            path: output_root.display().to_string(),
366            source: Box::new(source),
367        })?;
368
369    Ok(())
370}

372fn get_f64_array(table: &Table, row: usize, column: &str) -> Result<Vec<f64>, GainSolveError> {
373    match table
374        .cell_accessor(row, column)
375        .and_then(|cell| cell.array())
376        .map_err(MsError::from)
377        .map_err(|source| GainSolveError::OpenMeasurementSet {
378            path: column.to_string(),
379            source,
380        })? {
381        ArrayValue::Float64(values) => Ok(values.iter().copied().collect()),
382        other => Err(GainSolveError::UnsupportedParameterShape {
383            path: format!("{column} array"),
384            shape: other.shape().to_vec(),
385        }),
386    }
387}
388
389fn f64_array(values: &[f64]) -> ArrayValue {
390    ArrayValue::Float64(
391        ArrayD::from_shape_vec(IxDyn(&[values.len()]).f(), values.to_vec())
392            .expect("f64 vector should reshape to 1-D array"),
393    )
394}

396pub(crate) fn subtable_keyword_value(base_path: &Path, subtable_path: &Path) -> String {
397    if let Ok(relative) = subtable_path.strip_prefix(base_path) {
398        let rel = relative.to_string_lossy();
399        return format!("././{rel}");
400    }
401    if let Some(parent) = base_path.parent()
402        && let Ok(relative) = subtable_path.strip_prefix(parent)
403    {
404        let rel = relative.to_string_lossy();
405        return format!("./{rel}");
406    }
407    if subtable_path.is_relative() {
408        let rel = subtable_path.to_string_lossy();
409        return format!("././{}", rel.trim_start_matches("./"));
410    }
411    subtable_path.to_string_lossy().to_string()
412}
