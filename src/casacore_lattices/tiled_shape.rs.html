2//! Lattice shape with optional tiling specification.
3
4use crate::LatticeError;
5
6/// A lattice shape paired with an optional tile shape.
7///
8/// Corresponds to the C++ `TiledShape` class, which bundles the overall
9/// lattice shape with a tile (chunk) shape for storage managers like
10/// `TiledCellStMan` and `TiledShapeStMan`.
11///
12/// If no tile shape is provided, [`TiledShape::tile_shape`] returns a
13/// default computed by [`TiledShape::default_tile_shape`], which targets
14/// tiles of roughly 32 KiB — the same heuristic used by C++ casacore.
15///
16/// # Examples
17///
18/// ```rust
19/// use casacore_lattices::TiledShape;
20///
21/// // Default tiling for a 256×256×64 cube:
22/// let ts = TiledShape::new(vec![256, 256, 64]);
23/// assert_eq!(ts.shape(), &[256, 256, 64]);
24/// assert_eq!(ts.tile_shape().len(), 3);
25///
26/// // Explicit tile shape:
27/// let ts = TiledShape::with_tile_shape(vec![100, 100], vec![32, 32]).unwrap();
28/// assert_eq!(ts.tile_shape(), &[32, 32]);
29/// ```
30#[derive(Debug, Clone)]
31pub struct TiledShape {
32    shape: Vec<usize>,
33    tile_shape: Option<Vec<usize>>,
34}
35
36impl TiledShape {
37    /// Creates a `TiledShape` with default tiling.
38    ///
39    /// The tile shape is computed lazily by [`tile_shape`](Self::tile_shape)
40    /// using [`default_tile_shape`](Self::default_tile_shape).
41    pub fn new(shape: Vec<usize>) -> Self {
42        Self {
43            shape,
44            tile_shape: None,
45        }
46    }
47
48    /// Creates a `TiledShape` with an explicit tile shape.
49    ///
50    /// Returns an error if the tile shape has a different number of
51    /// dimensions than the lattice shape, or if any tile dimension exceeds
52    /// the corresponding lattice dimension.
53    pub fn with_tile_shape(
54        shape: Vec<usize>,
55        tile_shape: Vec<usize>,
56    ) -> Result<Self, LatticeError> {
57        if shape.len() != tile_shape.len() {
58            return Err(LatticeError::NdimMismatch {
59                expected: shape.len(),
60                got: tile_shape.len(),
61            });
62        }
63        for (axis, (&ts, &ls)) in tile_shape.iter().zip(shape.iter()).enumerate() {
64            if ts > ls {
65                return Err(LatticeError::TileMismatch {
66                    tile_shape: tile_shape.clone(),
67                    lattice_shape: shape.clone(),
68                    axis,
69                });
70            }
71        }
72        Ok(Self {
73            shape,
74            tile_shape: Some(tile_shape),
75        })
76    }
77
78    /// Returns the lattice shape.
79    pub fn shape(&self) -> &[usize] {
80        &self.shape
81    }
82
83    /// Returns the tile shape, computing a default if none was specified.
84    ///
85    /// The default targets tiles of roughly 32 KiB (4096 elements of 8 bytes
86    /// each), following the C++ casacore `TiledStMan` heuristic.
87    pub fn tile_shape(&self) -> Vec<usize> {
88        match &self.tile_shape {
89            Some(ts) => ts.clone(),
90            None => Self::default_tile_shape(&self.shape),
91        }
92    }
93
94    /// Computes a default tile shape for the given lattice shape.
95    ///
96    /// The heuristic targets tiles of roughly 4096 elements (32 KiB at 8
97    /// bytes/element). Each axis is clamped to the lattice extent, and the
98    /// product is kept near the target. This mirrors the C++ casacore
99    /// `TiledStMan::makeTileShape` logic.
100    pub fn default_tile_shape(shape: &[usize]) -> Vec<usize> {
101        if shape.is_empty() {
102            return vec![];
103        }
104
105        let target_elements: usize = 4096; // ~32 KiB at 8 bytes/element
106        let ndim = shape.len();
107
108        // Start with cube root (or nth root) of target, clamped to shape.
109        let root = (target_elements as f64).powf(1.0 / ndim as f64).ceil() as usize;
110        let mut tile: Vec<usize> = shape.iter().map(|&s| root.min(s).max(1)).collect();
111
112        // Adjust: if product is too small, grow the first axis that has room.
113        let product = |t: &[usize]| -> usize { t.iter().product() };
114        for axis in 0..ndim {
115            if product(&tile) >= target_elements {
116                break;
117            }
118            tile[axis] = shape[axis].min(target_elements / (product(&tile) / tile[axis]));
119        }
120
121        // Clamp if product greatly exceeds target (can happen with many axes).
122        while product(&tile) > target_elements * 4 && tile.iter().any(|&t| t > 1) {
123            // Shrink the largest tile axis.
124            if let Some(axis) = tile
125                .iter()
126                .enumerate()
127                .filter(|&(_, t)| *t > 1)
128                .max_by_key(|&(_, t)| *t)
129                .map(|(i, _)| i)
130            {
131                tile[axis] = (tile[axis] / 2).max(1);
132            }
133        }
134
135        tile
136    }
137}

139#[cfg(test)]
140mod tests {
141    use super::*;
142
143    #[test]
144    fn default_tile_shape_3d() {
145        let tile = TiledShape::default_tile_shape(&[256, 256, 64]);
146        assert_eq!(tile.len(), 3);
147        let product: usize = tile.iter().product();
148        // Should be in a reasonable range around 4096.
149        assert!((512..=32768).contains(&product), "product = {product}");
150        // Each tile axis should not exceed the lattice axis.
151        for (i, &t) in tile.iter().enumerate() {
152            assert!(t <= [256, 256, 64][i]);
153        }
154    }
155
156    #[test]
157    fn default_tile_shape_small() {
158        let tile = TiledShape::default_tile_shape(&[4, 4]);
159        assert_eq!(tile, vec![4, 4]);
160    }
161
162    #[test]
163    fn default_tile_shape_empty() {
164        assert!(TiledShape::default_tile_shape(&[]).is_empty());
165    }
166
167    #[test]
168    fn explicit_tile_shape_ok() {
169        let ts = TiledShape::with_tile_shape(vec![100, 100], vec![32, 32]).unwrap();
170        assert_eq!(ts.tile_shape(), vec![32, 32]);
171    }
172
173    #[test]
174    fn explicit_tile_shape_exceeds() {
175        let err = TiledShape::with_tile_shape(vec![10, 10], vec![32, 32]).unwrap_err();
176        assert!(matches!(err, LatticeError::TileMismatch { axis: 0, .. }));
177    }
178
179    #[test]
180    fn explicit_tile_ndim_mismatch() {
181        let err = TiledShape::with_tile_shape(vec![10, 10], vec![5]).unwrap_err();
182        assert!(matches!(err, LatticeError::NdimMismatch { .. }));
183    }
184
185    #[test]
186    fn new_uses_default_tiling() {
187        let ts = TiledShape::new(vec![64, 64]);
188        let tile = ts.tile_shape();
189        assert_eq!(tile.len(), 2);
190        assert!(tile[0] <= 64);
191        assert!(tile[1] <= 64);
192    }
193}
