43pub(crate) fn shadowed_antennas_from_projected_baselines<I>(
44    antenna_count: usize,
45    baselines: I,
46) -> Vec<bool>
47where
48    I: IntoIterator<Item = (usize, usize, [f64; 3], f64, f64)>,
49{
50    let mut shadowed = vec![false; antenna_count];
51    for (antenna1, antenna2, uvw_m, diameter1_m, diameter2_m) in baselines {
52        if antenna1 == antenna2 || antenna1 >= antenna_count || antenna2 >= antenna_count {
53            continue;
54        }
55        let (fraction1, fraction2) =
56            casa_projected_blockage_fractions(uvw_m, diameter1_m, diameter2_m);
57        if fraction1 > CASA_SIMOBSERVE_SHADOW_FRACTION_LIMIT {
58            shadowed[antenna1] = true;
59        }
60        if fraction2 > CASA_SIMOBSERVE_SHADOW_FRACTION_LIMIT {
61            shadowed[antenna2] = true;
62        }
63    }
64    shadowed
65}

67fn casa_projected_blockage_fractions(
68    uvw_m: [f64; 3],
69    diameter1_m: f64,
70    diameter2_m: f64,
71) -> (f64, f64) {
72    let separation = (uvw_m[0] * uvw_m[0] + uvw_m[1] * uvw_m[1]).sqrt();
73    let diameter1 = diameter1_m.abs();
74    let diameter2 = diameter2_m.abs();
75    let rmin = 0.5 * diameter1.min(diameter2);
76    let rmax = 0.5 * diameter1.max(diameter2);
77    let (mut fraction1, mut fraction2) = if separation >= rmin + rmax {
78        (0.0, 0.0)
79    } else if separation + rmin <= rmax {
80        (
81            1.0_f64.min((diameter2 / diameter1).powi(2)),
82            1.0_f64.min((diameter1 / diameter2).powi(2)),
83        )
84    } else {
85        let c = separation / (0.5 * diameter1);
86        let s = diameter2 / diameter1;
87        let sinb = (2.0 * ((c * s).powi(2) + c.powi(2) + s.powi(2)) - c.powi(4) - s.powi(4) - 1.0)
88            .sqrt()
89            / (2.0 * c);
90        let sinb = sinb.min(1.0);
91        let sina = (sinb / s).min(1.0);
92        let b = sinb.asin();
93        let a = sina.asin();
94        let area = (s.powi(2) * a + b) - (s.powi(2) * sina * a.cos() + sinb * b.cos());
95        let fraction1 = area / std::f64::consts::PI;
96        let fraction2 = fraction1 / s.powi(2);
97        (fraction1, fraction2)
98    };
99
100    if uvw_m[2] > 0.0 {
101        fraction2 = 0.0;
102    }
103    if uvw_m[2] < 0.0 {
104        fraction1 = 0.0;
105    }
106    (fraction1, fraction2)
107}

342/// Run native `flagdata` on an opened MeasurementSet.
343pub fn flagdata(
344    ms: &mut MeasurementSet,
345    request: &FlagDataRequest,
346) -> Result<FlagDataReport, FlaggingError> {
347    let ms_path = display_ms_path(ms);
348    tracing::info!(
349        path = %ms_path,
350        mode = ?request.mode,
351        action = ?request.action,
352        "flagdata started"
353    );
354    let selected_rows = selected_rows(ms, request)?;
355    if selected_rows.is_empty() {
356        tracing::warn!(path = %ms_path, mode = ?request.mode, "flagdata selected no rows");
357    }
358    let channel_selections = request
359        .spw
360        .as_deref()
361        .map(parse_channel_selections)
362        .transpose()
363        .map_err(|source| FlaggingError::MutateFlags {
364            path: ms_path.clone(),
365            reason: source.to_string(),
366        })?;
367
368    if request.mode == FlagDataMode::Summary {
369        let report = report_after(
370            ms,
371            request,
372            selected_rows.len(),
373            ChangeSet::default(),
374            None,
375            None,
376        )?;
377        tracing::info!(
378            path = %ms_path,
379            selected_rows = report.selected_rows,
380            flagged_samples = report.flagged_samples,
381            total_samples = report.total_samples,
382            "flagdata summary completed"
383        );
384        return Ok(report);
385    }
386
387    let backup_version = if request.flagbackup && !selected_rows.is_empty() {
388        Some(save_next_flagdata_backup(ms)?)
389    } else {
390        None
391    };
392
393    let changes = match request.mode {
394        FlagDataMode::Manual => apply_manual_flags(
395            ms,
396            &selected_rows,
397            channel_selections.as_ref(),
398            request.action,
399        )?,
400        FlagDataMode::Clip => {
401            apply_clip_flags(ms, &selected_rows, channel_selections.as_ref(), request)?
402        }
403        FlagDataMode::Quack => {
404            apply_quack_flags(ms, &selected_rows, channel_selections.as_ref(), request)?
405        }
406        FlagDataMode::Tfcrop => {
407            apply_tfcrop_flags(ms, &selected_rows, channel_selections.as_ref(), request)?
408        }
409        FlagDataMode::Rflag => {
410            apply_rflag_flags(ms, &selected_rows, channel_selections.as_ref(), request)?
411        }
412        FlagDataMode::Extend => {
413            apply_extend_flags(ms, &selected_rows, channel_selections.as_ref(), request)?
414        }
415        FlagDataMode::Summary => ChangeSet::default(),
416    };
417
418    if !changes.changed_row_indices.is_empty() {
419        if changes.changed_flag_row_rows == 0 {
420            ms.main_table_mut()
421                .save_selected_rows_in_place_assuming_valid(
422                    &[FLAG_COLUMN],
423                    &changes.changed_row_indices,
424                )
425                .map_err(|source| FlaggingError::MutateFlags {
426                    path: ms_path.clone(),
427                    reason: format!("save changed MAIN flag rows: {source}"),
428                })?;
429        } else {
430            ms.main_table_mut()
431                .save_selected_rows_in_place_assuming_valid(
432                    &[FLAG_COLUMN, FLAG_ROW_COLUMN],
433                    &changes.changed_row_indices,
434                )
435                .map_err(|source| FlaggingError::MutateFlags {
436                    path: ms_path.clone(),
437                    reason: format!("save changed MAIN flag rows: {source}"),
438                })?;
439        }
440    }
441    let thresholds = changes.thresholds;
442    let report = report_after(
443        ms,
444        request,
445        selected_rows.len(),
446        changes,
447        backup_version,
448        thresholds,
449    )?;
450    tracing::info!(
451        path = %ms_path,
452        mode = ?report.mode,
453        selected_rows = report.selected_rows,
454        changed_rows = report.changed_rows,
455        changed_samples = report.changed_samples,
456        "flagdata completed"
457    );
458    Ok(report)
459}

461/// Open an MS, run native `flagdata`, and save the changed MAIN table.
462pub fn flagdata_path(
463    ms_path: impl AsRef<Path>,
464    request: &FlagDataRequest,
465) -> Result<FlagDataReport, FlaggingError> {
466    let path = ms_path.as_ref();
467    tracing::info!(path = %path.display(), "opening MeasurementSet for flagdata");
468    let mut ms =
469        MeasurementSet::open(path).map_err(|source| FlaggingError::OpenMeasurementSet {
470            path: path.display().to_string(),
471            source,
472        })?;
473    flagdata(&mut ms, request)
474}

476fn report_after(
477    ms: &MeasurementSet,
478    request: &FlagDataRequest,
479    selected_row_count: usize,
480    changes: ChangeSet,
481    backup_version: Option<String>,
482    thresholds: Option<(f64, f64)>,
483) -> Result<FlagDataReport, FlaggingError> {
484    let summary = match changes.summary {
485        Some(summary) => summary,
486        None => {
487            let summary_rows = selected_rows(ms, request)?;
488            summarize_flags(ms, &summary_rows)?
489        }
490    };
491    Ok(FlagDataReport {
492        mode: request.mode,
493        selected_rows: selected_row_count,
494        changed_rows: changes.changed_rows,
495        changed_samples: changes.changed_samples,
496        flagged_samples: summary.flagged_samples,
497        total_samples: summary.total_samples,
498        backup_version,
499        timedev: thresholds.map(|pair| pair.0),
500        freqdev: thresholds.map(|pair| pair.1),
501    })
502}

517fn apply_manual_flags(
518    ms: &mut MeasurementSet,
519    selected_rows: &[usize],
520    channel_selections: Option<&BTreeMap<i32, ChannelSelection>>,

533fn apply_clip_flags(
534    ms: &mut MeasurementSet,
535    selected_rows: &[usize],
536    channel_selections: Option<&BTreeMap<i32, ChannelSelection>>,
537    request: &FlagDataRequest,
538) -> Result<ChangeSet, FlaggingError> {
539    if !request.clipzeros {
540        return Ok(ChangeSet::default());
541    }
542    let path = display_ms_path(ms);
543    let ddid_to_spw = data_description_spw_map(ms)?;
544    let mut updates = BTreeMap::<usize, ArrayD<bool>>::new();
545    let mut touched_rows = std::collections::BTreeSet::<usize>::new();
546    let mut changed = ChangeSet::default();
547    let mut flagged_samples = 0usize;
548    let mut total_samples = 0usize;
549    let table = ms.main_table();
550    let mut row_changes = Vec::<(usize, usize)>::new();
551    for rows in selected_rows.chunks(CLIP_SCAN_CHUNK_ROWS) {
552        let data_cells = table
553            .column_accessor(request.data_column.name())
554            .and_then(|column| column.array_cells_owned(rows))
555            .map_err(|source| FlaggingError::MutateFlags {
556                path: path.clone(),
557                reason: format!(
558                    "read selected {} rows: {source}",
559                    request.data_column.name()
560                ),
561            })?;
562        let flag_cells = table
563            .column_accessor(FLAG_COLUMN)
564            .and_then(|column| column.array_cells_owned(rows))
565            .map_err(|source| FlaggingError::MutateFlags {
566                path: path.clone(),
567                reason: format!("read selected FLAG rows: {source}"),
568            })?;
569        let ddids = if channel_selections.is_some() {
570            Some(load_i32_cells(table, rows, "DATA_DESC_ID", &path)?)
571        } else {
572            None
573        };
574        for (row_slot, &row) in rows.iter().enumerate() {
575            let flags = match flag_cells.get(row_slot).and_then(Option::as_ref) {
576                Some(ArrayValue::Bool(values)) => values,
577                Some(other) => {
578                    return Err(FlaggingError::MutateFlags {
579                        path: path.clone(),
580                        reason: format!("FLAG row {row} must be bool array, found {other:?}"),
581                    });
582                }
583                None => {
584                    return Err(FlaggingError::MutateFlags {
585                        path: path.clone(),
586                        reason: format!("FLAG row {row} is undefined"),
587                    });
588                }
589            };
590            let data_value = data_cells
591                .get(row_slot)
592                .and_then(Option::as_ref)
593                .ok_or_else(|| FlaggingError::MutateFlags {
594                    path: path.clone(),
595                    reason: format!("{} row {row} is undefined", request.data_column.name()),
596                })?;
597            let data_shape = numeric_array_value_shape(data_value).ok_or_else(|| {
598                FlaggingError::MutateFlags {
599                    path: path.clone(),
600                    reason: format!(
601                        "{} row {row} must be numeric array, found {data_value:?}",
602                        request.data_column.name()
603                    ),
604                }
605            })?;
606            if data_shape != flags.shape() {
607                return Err(FlaggingError::MutateFlags {
608                    path: path.clone(),
609                    reason: format!(
610                        "{} and FLAG shapes differ on row {row}: {:?} vs {:?}",
611                        request.data_column.name(),
612                        data_shape,
613                        flags.shape()
614                    ),
615                });
616            }
617            if data_shape.len() != 2 {
618                return Err(FlaggingError::MutateFlags {
619                    path: path.clone(),
620                    reason: format!("{} row {row} is not rank-2", request.data_column.name()),
621                });
622            }
623            total_samples += flags.len();
624            flagged_samples += flags.iter().filter(|flag| **flag).count();
625            let channels = if let Some(ddids) = ddids.as_ref() {
626                let ddid = ddids[row_slot];
627                let spw =
628                    ddid_to_spw
629                        .get(&ddid)
630                        .copied()
631                        .ok_or_else(|| FlaggingError::MutateFlags {
632                            path: path.clone(),
633                            reason: format!("DATA_DESC_ID {ddid} has no SPW mapping"),
634                        })?;
635                Some(
636                    match channel_selections.and_then(|selectors| selectors.get(&spw)) {
637                        Some(selection) => selection.indices(data_shape[1]).map_err(|source| {
638                            FlaggingError::MutateFlags {
639                                path: path.clone(),
640                                reason: format!("resolve SPW {spw} channels: {source}"),
641                            }
642                        })?,
643                        None => (0..data_shape[1]).collect(),
644                    },
645                )
646            } else {
647                None
648            };
649            row_changes.clear();
650            apply_clip_zero_array(
651                data_value,
652                flags,
653                channels.as_deref(),
654                &path,
655                row,
656                &mut row_changes,
657            )?;
658            let row_changed = row_changes.len();
659            if row_changed > 0 {
660                let mut updated_flags = flags.clone();
661                {
662                    let mut updated_flags = updated_flags
663                        .view_mut()
664                        .into_dimensionality::<Ix2>()
665                        .map_err(|source| FlaggingError::MutateFlags {
666                        path: path.clone(),
667                        reason: format!("FLAG row {row} must be rank-2: {source}"),
668                    })?;
669                    for &(corr, chan) in &row_changes {
670                        updated_flags[(corr, chan)] = true;
671                    }
672                }
673                changed.changed_samples += row_changed;
674                flagged_samples += row_changed;
675                updates.insert(row, updated_flags);
676                touched_rows.insert(row);
677            }

688fn apply_quack_flags(
689    ms: &mut MeasurementSet,
690    selected_rows: &[usize],
691    channel_selections: Option<&BTreeMap<i32, ChannelSelection>>,
692    request: &FlagDataRequest,
693) -> Result<ChangeSet, FlaggingError> {
694    let table = ms.main_table();
695    let mut by_scan = BTreeMap::<i32, Vec<(usize, f64)>>::new();
696    for &row in selected_rows {
697        let scan = table_i32(table, row, "SCAN_NUMBER", ms.path())?;
698        let time = table_f64(table, row, "TIME", ms.path())?;
699        by_scan.entry(scan).or_default().push((row, time));
700    }
701    let mut quack_rows = Vec::new();
702    for rows in by_scan.values_mut() {
703        rows.sort_by(|left, right| left.1.total_cmp(&right.1));
704        let Some(edge_time) = (match request.quackmode {
705            QuackMode::Beg => rows.first().map(|row| row.1),
706            QuackMode::End => rows.last().map(|row| row.1),
707        }) else {
708            continue;
709        };
710        for &(row, time) in rows.iter() {
711            let in_edge = match request.quackmode {
712                QuackMode::Beg => time - edge_time < request.quackinterval,
713                QuackMode::End => edge_time - time < request.quackinterval,
714            };
715            if in_edge {
716                quack_rows.push(row);
717            }
718        }
719    }
720    apply_manual_flags(ms, &quack_rows, channel_selections, FlagDataAction::Flag)
721}

899fn extend_autoflag_grouped(
900    groups: &BTreeMap<GroupKey, Vec<Sample>>,
901    samples: &[Sample],
902    modified: &mut [bool],
903) {
904    let mut grown = modified.to_vec();
905    for group in groups.values() {
906        for by_corr in samples_by_corr(group) {
907            for by_row in sample_row_runs(&by_corr) {
908                if percent_modified(by_row, modified) >= 80.0 {
909                    for sample in by_row {
910                        grown[sample.index] = true;
911                    }
912                }
913            }
914            for by_chan in samples_by_chan_dense(&by_corr) {
915                if percent_modified(&by_chan, modified) >= 50.0 {
916                    for sample in &by_chan {
917                        grown[sample.index] = true;
918                    }
919                }
920            }
921        }
922    }
923    modified.copy_from_slice(&grown);
924    for group in samples_by_row_chan(samples).values() {
925        if group
926            .iter()
927            .any(|sample| sample.flag || modified[sample.index])
928        {
929            for sample in group {
930                modified[sample.index] = true;
931            }
932        }
933    }
934}

1069fn write_touched_flag_updates(
1070    ms: &mut MeasurementSet,
1071    mut updates: std::collections::BTreeMap<usize, ArrayD<bool>>,
1072    touched_rows: std::collections::BTreeSet<usize>,
1073    changed: &mut ChangeSet,
1074) -> Result<(), FlaggingError> {
1075    let path = display_ms_path(ms);
1076    let mut prepared = Vec::with_capacity(touched_rows.len());
1077    for row in touched_rows {
1078        let Some(flags) = updates.remove(&row) else {
1079            continue;
1080        };
1081        let flag_row = flags.iter().all(|flag| *flag);
1082        prepared.push((row, flags, flag_row));
1083    }
1084    let current_flag_rows = load_bool_cells(
1085        ms.main_table(),
1086        &prepared.iter().map(|(row, _, _)| *row).collect::<Vec<_>>(),
1087        FLAG_ROW_COLUMN,
1088        &path,
1089    )?;
1090    ms.main_table_mut()
1091        .reserve_array_cell_updates(FLAG_COLUMN, prepared.len());
1092    {
1093        let mut flag_column = ms
1094            .main_table_mut()
1095            .column_accessor_mut(FLAG_COLUMN)
1096            .map_err(|source| FlaggingError::MutateFlags {
1097                path: path.clone(),
1098                reason: format!("open FLAG column: {source}"),
1099            })?;
1100        for (row, flags, _) in &prepared {
1101            flag_column
1102                .set_array_assuming_valid(*row, ArrayValue::Bool(flags.clone()))
1103                .map_err(|source| FlaggingError::MutateFlags {
1104                    path: path.clone(),
1105                    reason: format!("write FLAG row {row}: {source}"),
1106                })?;
1107        }
1108    }
1109    let flag_row_updates = prepared
1110        .iter()
1111        .zip(current_flag_rows.iter())
1112        .filter_map(|((row, _, flag_row), current)| {
1113            (*current != *flag_row).then_some((*row, *flag_row))
1114        })
1115        .collect::<Vec<_>>();
1116    if !flag_row_updates.is_empty() {
1117        let mut flag_row_column = ms
1118            .main_table_mut()
1119            .column_accessor_mut(FLAG_ROW_COLUMN)
1120            .map_err(|source| FlaggingError::MutateFlags {
1121                path: path.clone(),
1122                reason: format!("open FLAG_ROW column: {source}"),
1123            })?;
1124        for (row, flag_row) in &flag_row_updates {
1125            flag_row_column
1126                .set_scalar_assuming_valid(*row, ScalarValue::Bool(*flag_row))
1127                .map_err(|source| FlaggingError::MutateFlags {
1128                    path: path.clone(),
1129                    reason: format!("write FLAG_ROW row {row}: {source}"),
1130                })?;
1131        }
1132    }
1133    changed.changed_flag_row_rows += flag_row_updates.len();
1134    for (row, _, _) in prepared {
1135        changed.changed_rows += 1;
1136        changed.changed_row_indices.push(row);
1137    }
1138    Ok(())
1139}

1205fn load_samples(
1206    ms: &MeasurementSet,
1207    selected_rows: &[usize],
1208    channel_selections: Option<&BTreeMap<i32, ChannelSelection>>,
1209    data_column: FlagDataColumn,
1210) -> Result<LoadedSamples, FlaggingError> {
1211    let path = display_ms_path(ms);
1212    let ddid_to_spw = data_description_spw_map(ms)?;
1213    let table = ms.main_table();
1214    let data_cells = table
1215        .column_accessor(data_column.name())
1216        .and_then(|column| column.array_cells_owned(selected_rows))
1217        .map_err(|source| FlaggingError::MutateFlags {
1218            path: path.clone(),
1219            reason: format!("read selected {} rows: {source}", data_column.name()),
1220        })?;
1221    let flag_cells = table
1222        .column_accessor(FLAG_COLUMN)
1223        .and_then(|column| column.array_cells_owned(selected_rows))
1224        .map_err(|source| FlaggingError::MutateFlags {
1225            path: path.clone(),
1226            reason: format!("read selected FLAG rows: {source}"),
1227        })?;
1228    let ddids = load_i32_cells(table, selected_rows, "DATA_DESC_ID", &path)?;
1229    let fields = load_i32_cells(table, selected_rows, "FIELD_ID", &path)?;
1230    let scans = load_i32_cells(table, selected_rows, "SCAN_NUMBER", &path)?;
1231    let ant1s = load_i32_cells(table, selected_rows, "ANTENNA1", &path)?;
1232    let ant2s = load_i32_cells(table, selected_rows, "ANTENNA2", &path)?;
1233    let mut samples = Vec::new();
1234    let mut flags_by_row = BTreeMap::new();
1235    for (row_slot, &row) in selected_rows.iter().enumerate() {
1236        let ddid = ddids[row_slot];
1237        let spw = ddid_to_spw
1238            .get(&ddid)
1239            .copied()
1240            .ok_or_else(|| FlaggingError::MutateFlags {
1241                path: path.clone(),
1242                reason: format!("DATA_DESC_ID {ddid} has no SPW mapping"),
1243            })?;
1244        let flags = match flag_cells.get(row_slot).and_then(Option::as_ref) {
1245            Some(ArrayValue::Bool(values)) => values.clone(),
1246            Some(other) => {
1247                return Err(FlaggingError::MutateFlags {
1248                    path: path.clone(),
1249                    reason: format!("FLAG row {row} must be bool array, found {other:?}"),
1250                });
1251            }
1252            None => {
1253                return Err(FlaggingError::MutateFlags {
1254                    path: path.clone(),
1255                    reason: format!("FLAG row {row} is undefined"),
1256                });
1257            }
1258        };
1259        let data_value = data_cells
1260            .get(row_slot)
1261            .and_then(Option::as_ref)
1262            .ok_or_else(|| FlaggingError::MutateFlags {
1263                path: path.clone(),
1264                reason: format!("{} row {row} is undefined", data_column.name()),
1265            })?;
1266        let data_shape =
1267            numeric_array_value_shape(data_value).ok_or_else(|| FlaggingError::MutateFlags {
1268                path: path.clone(),
1269                reason: format!(
1270                    "{} row {row} must be numeric array, found {data_value:?}",
1271                    data_column.name()
1272                ),
1273            })?;
1274        if data_shape != flags.shape() {
1275            return Err(FlaggingError::MutateFlags {
1276                path: path.clone(),
1277                reason: format!(
1278                    "{} and FLAG shapes differ on row {row}: {:?} vs {:?}",
1279                    data_column.name(),
1280                    data_shape,
1281                    flags.shape()
1282                ),
1283            });
1284        }
1285        if data_shape.len() != 2 {
1286            return Err(FlaggingError::MutateFlags {
1287                path: path.clone(),
1288                reason: format!("{} row {row} is not rank-2", data_column.name()),
1289            });
1290        }
1291        let channels =
1292            match channel_selections.and_then(|selectors| selectors.get(&spw)) {
1293                Some(selection) => selection.indices(data_shape[1]).map_err(|source| {
1294                    FlaggingError::MutateFlags {
1295                        path: path.clone(),
1296                        reason: format!("resolve SPW {spw} channels: {source}"),
1297                    }
1298                })?,
1299                None => (0..data_shape[1]).collect(),
1300            };
1301        let field = fields[row_slot];
1302        let scan = scans[row_slot];
1303        let ant1 = ant1s[row_slot];
1304        let ant2 = ant2s[row_slot];
1305        push_row_samples(
1306            data_value,
1307            &flags,
1308            &channels,
1309            RowSampleMetadata {
1310                row,
1311                field,
1312                spw,
1313                scan,
1314                ant1,
1315                ant2,
1316            },
1317            &path,
1318            data_column.name(),
1319            &mut samples,
1320        )?;
1321        flags_by_row.insert(row, flags);
1322    }
1323    Ok(LoadedSamples {
1324        samples,
1325        flags_by_row,
1326    })
1327}

1682fn tfcrop_fit_base_and_flag(
1683    samples: &[Sample],
1684    direction: TfcropDirection,
1685    fit: TfcropFit,
1686    cutoff: f64,
1687    modified: &mut [bool],
1688    direction_flags: &mut [bool],
1689) {
1690    if samples.is_empty() {
1691        return;
1692    }
1693    let rows = unique_sample_rows(samples);
1694    let chans = unique_sample_chans(samples);
1695    let (axis0, axis1) = match direction {
1696        TfcropDirection::Freq => (chans, rows),
1697        TfcropDirection::Time => (rows, chans),
1698    };
1699    if axis0.is_empty() || axis1.is_empty() {
1700        return;
1701    }
1702
1703    let mut by_row_chan = HashMap::<(usize, usize), Sample>::with_capacity(samples.len());
1704    for sample in samples {
1705        by_row_chan.insert((sample.row, sample.chan), *sample);
1706    }
1707
1708    let mut avg_dat = vec![0.0f32; axis0.len()];
1709    let mut avg_flag = vec![false; axis0.len()];
1710    let mut all_zeros = true;
1711    for (i0, key0) in axis0.iter().enumerate() {
1712        let mut sum = 0.0f32;
1713        let mut count = 0usize;
1714        for key1 in &axis1 {
1715            let (row, chan) = tfcrop_row_chan(direction, *key0, *key1);
1716            if let Some(sample) = by_row_chan.get(&(row, chan)) {
1717                if !sample_is_modified(sample, modified) {
1718                    sum += tfcrop_value(sample);
1719                    count += 1;
1720                }
1721            }
1722        }
1723        if count > 0 {
1724            avg_dat[i0] = sum / count as f32;
1725            all_zeros = false;
1726        } else {
1727            avg_flag[i0] = true;
1728        }
1729    }
1730    if all_zeros {
1731        return;
1732    }
1733
1734    let avg_fit = match fit {
1735        TfcropFit::Line => tfcrop_fit_piecewise_poly(&avg_dat, &mut avg_flag, 1, 1),
1736        TfcropFit::Poly => tfcrop_fit_piecewise_poly(&avg_dat, &mut avg_flag, 7, 4),
1737    };
1738
1739    for key1 in &axis1 {
1740        let mut normalized = vec![0.0f32; axis0.len()];
1741        let mut flags = vec![false; axis0.len()];
1742        for (i0, key0) in axis0.iter().enumerate() {
1743            let (row, chan) = tfcrop_row_chan(direction, *key0, *key1);
1744            if let Some(sample) = by_row_chan.get(&(row, chan)) {
1745                flags[i0] = sample_is_modified(sample, modified);
1746                if !flags[i0] && avg_fit[i0].abs() > f32::EPSILON {
1747                    normalized[i0] = tfcrop_value(sample) / avg_fit[i0];
1748                }
1749            } else {
1750                flags[i0] = true;
1751            }
1752        }
1753
1754        let mut previous_sd = 0.0f32;
1755        for _ in 0..5 {
1756            let sd = tfcrop_std_about_mean(&normalized, &flags, 1.0);
1757            if !sd.is_finite() || sd <= f32::EPSILON {
1758                break;
1759            }
1760            for i0 in 0..axis0.len() {
1761                if !flags[i0] && (normalized[i0] - 1.0).abs() > cutoff as f32 * sd {
1762                    flags[i0] = true;
1763                }
1764            }
1765            if (previous_sd - sd).abs() < 0.1 {
1766                break;
1767            }
1768            previous_sd = sd;
1769        }
1770
1771        for (i0, key0) in axis0.iter().enumerate() {
1772            if !flags[i0] {
1773                continue;
1774            }
1775            let (row, chan) = tfcrop_row_chan(direction, *key0, *key1);
1776            if let Some(sample) = by_row_chan.get(&(row, chan)) {
1777                if !modified[sample.index] {
1778                    modified[sample.index] = true;
1779                    direction_flags[sample.index] = true;
1780                }
1781            }
1782        }
1783    }
1784}

1801fn tfcrop_fit_piecewise_poly(
1802    data: &[f32],
1803    flag: &mut [bool],
1804    max_pieces: usize,
1805    max_degree: usize,
1806) -> Vec<f32> {
1807    let mut tdata = data.to_vec();
1808    for i in 0..tdata.len() {
1809        if tdata[i] != 0.0 {
1810            continue;
1811        }
1812        if i == 0 {
1813            if let Some((index, _)) = tdata
1814                .iter()
1815                .enumerate()
1816                .skip(1)
1817                .find(|(_, value)| **value != 0.0)
1818            {
1819                tdata[i] = tdata[index];
1820            }
1821        } else {
1822            let left = (0..=i).rev().find(|index| tdata[*index] != 0.0);
1823            let right = (i + 1..tdata.len()).find(|index| tdata[*index] != 0.0);
1824            match (left, right) {
1825                (Some(left), Some(right)) => tdata[i] = (tdata[left] + tdata[right]) / 2.0,
1826                (Some(left), None) => tdata[i] = tdata[left],
1827                (None, Some(right)) => tdata[i] = tdata[right],
1828                (None, None) => {}
1829            }
1830        }
1831    }
1832    for (index, value) in tdata.iter().enumerate() {
1833        if *value == 0.0 {
1834            flag[index] = true;
1835        }
1836    }
1837
1838    let mut fitted = tdata.clone();
1839    for iteration in 0..5usize {
1840        let npieces = (2 * iteration + 1).min(max_pieces);
1841        let mut degree = 1usize;
1842        if iteration > 1 {
1843            degree = 2;
1844        }
1845        if iteration > 2 {
1846            degree = 3;
1847        }
1848        degree = degree.min(max_degree);
1849        let piece_size = tdata.len() / npieces;
1850        let leftover = tdata.len() % npieces;
1851        let leftover_front = leftover / 2;
1852        for piece in 0..npieces {
1853            let (left, right) = if npieces > 1 {
1854                let mut left = leftover_front + piece * piece_size;
1855                let mut right = left + piece_size;
1856                if piece == 0 {
1857                    left = 0;
1858                    right = leftover_front + piece_size;
1859                }
1860                if piece == npieces - 1 {
1861                    right = tdata.len() - 1;
1862                }
1863                (left, right)
1864            } else {
1865                (0, tdata.len() - 1)
1866            };
1867            if degree == 1 {
1868                tfcrop_line_fit(&tdata, flag, &mut fitted, left, right);
1869            } else {
1870                tfcrop_poly_fit(&tdata, flag, &mut fitted, left, right, degree);
1871            }
1872        }
1873
1874        for i in 2..tdata.len().saturating_sub(2) {
1875            let win_start = i - 2;
1876            let win_end = (i + 2).min(tdata.len() - 1);
1877            if win_end <= win_start {
1878                break;
1879            }
1880            let sum = (win_start..=win_end)
1881                .map(|index| fitted[index])
1882                .sum::<f32>();
1883            fitted[i] = sum / (win_end - win_start + 1) as f32;
1884        }
1885
1886        let sd = tfcrop_std_about_fit(&tdata, flag, &fitted);
1887        if !sd.is_finite() {
1888            continue;
1889        }
1890        let tolerance = if iteration >= 2 { 2.0 } else { 3.0 };
1891        for i in 0..tdata.len() {
1892            if tdata[i] - fitted[i] > tolerance * sd {
1893                flag[i] = true;
1894            }
1895        }
1896    }
1897    fitted
1898}

1900fn tfcrop_line_fit(data: &[f32], flag: &[bool], fit: &mut [f32], left: usize, right: usize) {
1901    let mean = tfcrop_mean(data, flag);
1902    let sd = tfcrop_std_about_mean(data, flag, mean);
1903    if !sd.is_finite() || sd <= f32::EPSILON {
1904        for fitted in fit.iter_mut().take(right + 1).skip(left) {
1905            *fitted = mean;
1906        }
1907        return;
1908    }
1909    let mut sum_w = 0.0f32;
1910    let mut sum_x = 0.0f32;
1911    let mut sum_y = 0.0f32;
1912    let mut sum_xx = 0.0f32;
1913    let mut sum_xy = 0.0f32;
1914    let weight = 1.0f32 / (sd * sd);
1915    for i in left..=right {
1916        if !flag[i] {
1917            let x = i as f32;
1918            sum_w += weight;
1919            sum_x += x * weight;
1920            sum_y += data[i] * weight;
1921            sum_xx += x * x * weight;
1922            sum_xy += x * data[i] * weight;
1923        }
1924    }
1925    let denom = sum_w * sum_xx - sum_x * sum_x;
1926    if denom.abs() <= f32::EPSILON {
1927        for fitted in fit.iter_mut().take(right + 1).skip(left) {
1928            *fitted = mean;
1929        }
1930        return;
1931    }
1932    let intercept = (sum_xx * sum_y - sum_x * sum_xy) / denom;
1933    let slope = (sum_w * sum_xy - sum_x * sum_y) / denom;
1934    for (i, fitted) in fit.iter_mut().enumerate().take(right + 1).skip(left) {
1935        *fitted = intercept + slope * i as f32;
1936    }
1937}

1939fn tfcrop_poly_fit(
1940    data: &[f32],
1941    flag: &[bool],
1942    fit: &mut [f32],
1943    left: usize,
1944    right: usize,
1945    degree: usize,
1946) {
1947    let rows = (left..=right)
1948        .filter(|index| !flag[*index])
1949        .map(|index| {
1950            let x = index as f64 + 1.0;
1951            let basis = (0..=degree)
1952                .map(|power| x.powi(power as i32))
1953                .collect::<Vec<_>>();
1954            (basis, f64::from(data[index]), 1.0)
1955        })
1956        .collect::<Vec<_>>();
1957    if rows.len() <= degree {
1958        tfcrop_line_fit(data, flag, fit, left, right);
1959        return;
1960    }
1961    let Some(solution) = solve_weighted_least_squares(&rows, degree + 1) else {
1962        tfcrop_line_fit(data, flag, fit, left, right);
1963        return;
1964    };
1965    for (i, fitted) in fit.iter_mut().enumerate().take(right + 1).skip(left) {
1966        let x = i as f64 + 1.0;
1967        *fitted = solution
1968            .iter()
1969            .enumerate()
1970            .map(|(power, coefficient)| coefficient * x.powi(power as i32))
1971            .sum::<f64>() as f32;
1972    }
1973}

2839    match merge {
2840        FlagMerge::Replace => source,

2847    source: &ArrayD<bool>,
2848    dest: &ArrayD<bool>,
2849    merge: FlagMerge,
2850) -> Result<ArrayD<bool>, FlaggingError> {

2863        .zip(dest.iter())

2869        .collect::<Vec<_>>();
