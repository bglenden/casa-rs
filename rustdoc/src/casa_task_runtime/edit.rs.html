9/// Stable category for a rejected parameter edit.
10#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
11#[serde(rename_all = "snake_case")]
12pub enum ParameterEditDiagnosticCode {
13    UnknownParameter,
14    InvalidText,
15    InvalidValue,
16    InvalidSelectorSyntax,
17    UnsupportedSelectorCapability,
18    UnknownSelectorValue,
19    SelectorValueOutOfRange,
20    DatasetUnavailable,
21}
22
23/// One parameter-scoped edit diagnostic shared by every surface.
24#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
25pub struct ParameterEditDiagnostic {
26    pub code: ParameterEditDiagnosticCode,
27    pub parameter: String,
28    pub message: String,
29}
30
31impl ParameterEditDiagnostic {
32    pub fn new(
33        code: ParameterEditDiagnosticCode,
34        parameter: impl Into<String>,
35        message: impl Into<String>,
36    ) -> Self {
37        Self {
38            code,
39            parameter: parameter.into(),
40            message: message.into(),
41        }
42    }
43}
44
45/// One presentation-safe value suggested for an edit.
46#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
47pub struct ParameterEditSuggestion {
48    pub label: String,
49    pub value: ParameterValue,
50}
51
52/// Typed result of validating one parameter text edit.
53#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
54pub struct ParameterEditResult {
55    pub parameter: String,
56    pub normalized_value: Option<ParameterValue>,
57    pub diagnostics: Vec<ParameterEditDiagnostic>,
58    pub supported_capabilities: Vec<String>,
59    pub suggestions: Vec<ParameterEditSuggestion>,
60}
61
62impl ParameterEditResult {
63    pub fn is_valid(&self) -> bool {
64        self.diagnostics.is_empty()
65    }
66
67    pub fn reject(&mut self, code: ParameterEditDiagnosticCode, message: impl Into<String>) {
68        self.normalized_value = None;
69        self.diagnostics.push(ParameterEditDiagnostic::new(
70            code,
71            self.parameter.clone(),
72            message,
73        ));
74    }
75}

77/// Parse, normalize, and validate one edit through its declared contract.
78///
79/// An empty edit represents an absent/reset value. Domain crates may extend
80/// this result with syntax or dataset-aware diagnostics after this generic
81/// contract pass.
82pub fn validate_parameter_edit(
83    bundle: &SurfaceContractBundle,
84    parameter: &str,
85    text: &str,
86    suggestions: impl IntoIterator<Item = ParameterEditSuggestion>,
87) -> ParameterEditResult {
88    let mut result = ParameterEditResult {
89        parameter: parameter.to_string(),
90        normalized_value: None,
91        diagnostics: Vec::new(),
92        supported_capabilities: Vec::new(),
93        suggestions: Vec::new(),
94    };
95    let Some(binding) = bundle
96        .surface
97        .bindings()
98        .iter()
99        .find(|binding| binding.name == parameter)
100    else {
101        result.reject(
102            ParameterEditDiagnosticCode::UnknownParameter,
103            format!("unknown {} parameter {parameter:?}", bundle.surface.id()),
104        );
105        return result;
106    };
107    let Some(concept) = bundle.catalog.concept(&binding.concept) else {
108        result.reject(
109            ParameterEditDiagnosticCode::UnknownParameter,
110            format!("missing parameter concept for {parameter:?}"),
111        );
112        return result;
113    };
114    collect_selector_capabilities(&concept.normalization, &mut result.supported_capabilities);
115
116    if text.trim().is_empty() {
117        return result;
118    }
119    let parsed = match parse_parameter_text(text, &concept.value_domain) {
120        Ok(value) => value,
121        Err(error) => {
122            result.reject(ParameterEditDiagnosticCode::InvalidText, error.to_string());
123            return result;
124        }
125    };
126    let normalized = match normalize_value(&parsed, concept) {
127        Ok(value) => value,
128        Err(error) => {
129            let code = if error.to_string().contains("unsupported") {
130                ParameterEditDiagnosticCode::UnsupportedSelectorCapability
131            } else {
132                ParameterEditDiagnosticCode::InvalidValue
133            };
134            result.reject(code, error.to_string());
135            return result;
136        }
137    };
138    if let Err(error) = validate_value(&normalized, concept, &binding.refinements) {
139        result.reject(ParameterEditDiagnosticCode::InvalidValue, error.to_string());
140        return result;
141    }
142    result.normalized_value = Some(normalized);
143
144    result.suggestions = suggestions
145        .into_iter()
146        .filter_map(|suggestion| {
147            let normalized = normalize_value(&suggestion.value, concept).ok()?;
148            validate_value(&normalized, concept, &binding.refinements).ok()?;
149            Some(ParameterEditSuggestion {
150                label: suggestion.label,
151                value: normalized,
152            })
153        })
154        .collect();
155    result
156}
157
158fn collect_selector_capabilities(rule: &NormalizationRule, output: &mut Vec<String>) {
159    match rule {
160        NormalizationRule::CasaSelector { capabilities, .. } => {
161            for capability in capabilities {
162                if !output.contains(capability) {
163                    output.push(capability.clone());
164                }
165            }
166        }
167        NormalizationRule::Sequence { rules } => {
168            for rule in rules {
169                collect_selector_capabilities(rule, output);
170            }
171        }
172        NormalizationRule::Identity
173        | NormalizationRule::Trim
174        | NormalizationRule::Lowercase
175        | NormalizationRule::Path
176        | NormalizationRule::Quantity { .. } => {}
177    }
178}

180#[cfg(test)]
181mod tests {
182    use casa_provider_contracts::{ParameterValue, builtin_surface_bundle};
183
184    use super::*;
185
186    #[test]
187    fn edit_validation_normalizes_and_enforces_surface_capabilities() {
188        let bundle = builtin_surface_bundle("msexplore").unwrap();
189        let valid = validate_parameter_edit(&bundle, "field", " 0, 2 ", []);
190        assert!(valid.is_valid(), "{:?}", valid.diagnostics);
191        assert_eq!(
192            valid.normalized_value,
193            Some(ParameterValue::String("0,2".to_string()))
194        );
195
196        let unknown = validate_parameter_edit(&bundle, "not_a_parameter", "0", []);
197        assert_eq!(
198            unknown.diagnostics[0].code,
199            ParameterEditDiagnosticCode::UnknownParameter
200        );
201    }
