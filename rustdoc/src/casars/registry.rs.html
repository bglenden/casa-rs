54fn registry_app_from_catalog(entry: &ApplicationDefinition) -> Option<RegistryApp> {
55    if !entry.show_in_tui || entry.kind != ApplicationKind::Task {
56        return None;
57    }
58    let surface = entry.surface_bundle().ok()??;
59    let shell_kind = match entry.shell_kind {
60        ApplicationShellKind::Inspect => AppShellKind::Inspect,
61        ApplicationShellKind::Browser => AppShellKind::Browser,
62        ApplicationShellKind::Workflow => AppShellKind::Workflow,
63        ApplicationShellKind::Launcher => return None,
64    };
65    let interaction = match entry.interaction {
66        ApplicationInteraction::OneShot => AppInteraction::OneShot,
67        ApplicationInteraction::BrowserSession => {
68            let browser_kind = match entry.browser_kind {
69                Some(ApplicationBrowserKind::Table) => BrowserAppKind::Table,
70                Some(ApplicationBrowserKind::Image) => BrowserAppKind::Image,
71                _ => return None,
72            };
73            AppInteraction::BrowserSession(browser_kind)
74        }
75        ApplicationInteraction::Launcher => return None,
76    };
77    Some(RegistryApp {
78        id: entry.id.clone(),
79        category: surface.surface.category().to_string(),
80        display_name: surface.surface.display_name().to_string(),
81        shell_kind,
82        kind: RegistryAppKind::Subprocess {
83            binary_name: entry.launch.executable.clone(),
84            cargo_package: entry.launch.cargo_package.clone(),
85            override_env: entry.launch.override_env.clone(),
86            interaction,
87        },
88    })
89}

91fn configured_launch_mode() -> Result<ApplicationLaunchMode, String> {
92    match env::var("CASARS_LAUNCH_MODE") {
93        Ok(value) if value == "installed_suite" => Ok(ApplicationLaunchMode::InstalledSuite),
94        Ok(value) if value == "development_workspace" => {
95            Ok(ApplicationLaunchMode::DevelopmentWorkspace)
96        }
97        Ok(value) => Err(format!(
98            "invalid CASARS_LAUNCH_MODE {value:?}; expected installed_suite or development_workspace"
99        )),
100        Err(env::VarError::NotPresent) if cfg!(debug_assertions) => {
101            Ok(ApplicationLaunchMode::DevelopmentWorkspace)
102        }
103        Err(env::VarError::NotPresent) => Ok(ApplicationLaunchMode::InstalledSuite),
104        Err(error) => Err(format!("read CASARS_LAUNCH_MODE: {error}")),
105    }
106}

108#[derive(Debug, Clone)]
109pub(crate) struct ResolvedCommand {
110    program: OsString,
111    prefix_args: Vec<OsString>,
112}
113
114impl ResolvedCommand {
115    pub(crate) fn command(&self) -> Command {
116        let mut command = Command::new(&self.program);
117        command.args(&self.prefix_args);
118        command
119    }
120
121    #[cfg(test)]
122    pub(crate) fn direct(program: impl Into<OsString>) -> Self {
123        Self {
124            program: program.into(),
125            prefix_args: Vec::new(),
126        }
127    }
128}

130impl RegistryApp {
131    /// Derive the form schema from the canonical built-in surface definition.
132    /// Provider subprocesses no longer supply a fallback schema or alias table.
133    pub(crate) fn load_schema(&self) -> Result<UiCommandSchema, String> {
134        let bundle = builtin_surface_bundle(&self.id)?;
135        serde_json::from_value(casa_provider_contracts::project_ui_form(&bundle))
136            .map_err(|error| format!("project {} parameter definition: {error}", self.id))
137    }
138
139    pub(crate) fn resolve_command(&self) -> Result<ResolvedCommand, String> {
140        let RegistryAppKind::Subprocess {
141            binary_name,
142            cargo_package,
143            override_env,
144            ..
145        } = &self.kind;
146
147        if let Some(path) = env::var_os(override_env) {
148            return Ok(ResolvedCommand {
149                program: path,
150                prefix_args: Vec::new(),
151            });
152        }
153        match configured_launch_mode()? {
154            ApplicationLaunchMode::InstalledSuite => {
155                let path = sibling_binary(binary_name).ok_or_else(|| {
156                    format!(
157                        "installed-suite executable {binary_name:?} is missing beside the casars launcher; set {override_env} to its installed path"
158                    )
159                })?;
160                Ok(ResolvedCommand {
161                    program: path.into_os_string(),
162                    prefix_args: Vec::new(),
163                })
164            }
165            ApplicationLaunchMode::DevelopmentWorkspace => {
166                let cargo = env::var_os("CARGO").unwrap_or_else(|| OsString::from("cargo"));
167                Ok(ResolvedCommand {
168                    program: cargo,
169                    prefix_args: vec![
170                        OsString::from("run"),
171                        OsString::from("--manifest-path"),
172                        workspace_manifest_path().into_os_string(),
173                        OsString::from("-q"),
174                        OsString::from("-p"),
175                        OsString::from(cargo_package),
176                        OsString::from("--bin"),
177                        OsString::from(binary_name),
178                        OsString::from("--"),
179                    ],
180                })
181            }
182        }
183    }
184
185    pub(crate) fn is_browser_session(&self) -> bool {
186        matches!(
187            self.kind,
188            RegistryAppKind::Subprocess {
189                interaction: AppInteraction::BrowserSession(_),
190                ..
191            }
192        )
193    }
194
195    pub(crate) fn shell_kind(&self) -> AppShellKind {
196        self.shell_kind
197    }
198
199    pub(crate) fn browser_kind(&self) -> Option<BrowserAppKind> {
200        match self.kind {
201            RegistryAppKind::Subprocess {
202                interaction: AppInteraction::BrowserSession(kind),
203                ..
204            } => Some(kind),
205            RegistryAppKind::Subprocess {
206                interaction: AppInteraction::OneShot,
207                ..
208            } => None,
209        }
210    }
211
212    pub(crate) fn browser_path_field_id(&self) -> Option<&'static str> {
213        match self.browser_kind()? {
214            BrowserAppKind::Table => Some("table"),
215            BrowserAppKind::Image => Some("image"),
216        }
217    }
218
219    pub(crate) fn ready_status_line(&self) -> &'static str {
220        match (self.shell_kind, &self.kind) {
221            (
222                AppShellKind::Browser,
223                RegistryAppKind::Subprocess {
224                    interaction: AppInteraction::BrowserSession(_),
225                    ..
226                },
227            ) => "Ready. Press r to open the browser session.",
228            (AppShellKind::Inspect, _) => {
229                "Ready. Press r to run and refresh the current inspection view."
230            }
231            (AppShellKind::Workflow, _) => "Ready. Press r to run the selected workflow stage.",
232            (_, _) => "Ready. Press r to run the selected command.",
233        }
234    }
235}

237pub(crate) fn resolve_app(id: Option<&str>) -> Result<RegistryApp, String> {
238    let requested = id.unwrap_or("msexplore");
239    registered_apps()
240        .into_iter()
241        .find(|app| app.id == requested)
242        .ok_or_else(|| {
243            let expected = registered_apps()
244                .into_iter()
245                .map(|app| app.id)
246                .collect::<Vec<_>>()
247                .join(", ");
248            format!("unknown casars app {requested:?}; expected one of: {expected}")
249        })
250}
251
252pub(crate) fn registered_apps() -> Vec<RegistryApp> {
253    builtin_application_catalog()
254        .expect("built-in application catalog should validate")
255        .applications
256        .iter()
257        .filter_map(registry_app_from_catalog)
258        .collect()
259}

291fn sibling_binary(binary_name: &str) -> Option<PathBuf> {
292    let mut path = env::current_exe().ok()?;
293    path.pop();
294    path.push(binary_name);
295    path.set_extension(env::consts::EXE_EXTENSION);
296    if path.exists() { Some(path) } else { None }
297}
298
299fn workspace_manifest_path() -> PathBuf {
300    let manifest_dir = PathBuf::from(env!("CARGO_MANIFEST_DIR"));
301    manifest_dir
302        .parent()
303        .and_then(|path| path.parent())
304        .unwrap_or_else(|| {
305            panic!("casars manifest dir should live under <workspace>/crates/casars")
306        })
307        .join("Cargo.toml")
308}

324    #[test]
325    fn schema_projection_uses_casa_names_without_alias_enrichment() {
326        let imager = imager_app().load_schema().unwrap();
327        assert!(imager.argument("vis").is_some());
328        assert!(imager.argument("cell").is_some());
329        assert!(imager.argument("ms").is_none());
330        assert!(imager.argument("cell_arcsec").is_none());
331
332        // These catalog surfaces intentionally share executable routes without
333        // a provider surface selector, so frontend forms are projected directly
334        // from each canonical surface rather than inferred from a family binary.
335        let split_app = resolve_app(Some("split")).unwrap();
336        let mstransform_app = resolve_app(Some("mstransform")).unwrap();
337        let RegistryAppKind::Subprocess {
338            binary_name: split_binary,
339            ..
340        } = &split_app.kind;
341        let RegistryAppKind::Subprocess {
342            binary_name: mstransform_binary,
343            ..
344        } = &mstransform_app.kind;
345        assert_eq!(split_binary, mstransform_binary);
346        let split = split_app.load_schema().unwrap();
347        let mstransform = mstransform_app.load_schema().unwrap();
348        assert_eq!(split.command_id, "split");
349        assert_eq!(mstransform.command_id, "mstransform");
350        assert!(split.argument("vis").is_some());
351
352        let plotms_app = resolve_app(Some("plotms")).unwrap();
353        let msexplore_app = resolve_app(Some("msexplore")).unwrap();
354        let RegistryAppKind::Subprocess {
355            binary_name: plotms_binary,
356            ..
357        } = &plotms_app.kind;
358        let RegistryAppKind::Subprocess {
359            binary_name: msexplore_binary,
360            ..
361        } = &msexplore_app.kind;
362        assert_eq!(plotms_binary, msexplore_binary);
363        let plotms = plotms_app.load_schema().unwrap();
364        let msexplore = msexplore_app.load_schema().unwrap();
365        assert_eq!(plotms.command_id, "plotms");
366        assert_eq!(msexplore.command_id, "msexplore");
367    }
