36/// Supported native `gencal` families for the VLA IRC+10216 prior-cal slice.
37#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize, JsonSchema)]
38#[serde(rename_all = "snake_case")]
39pub enum GencalType {
40    /// Antenna-position delay correction table (`KAntPos Jones`).
41    Antpos,
42    /// VLA gain-curve plus antenna-efficiency table (`EGainCurve`).
43    Gceff,
44    /// Zenith-opacity table (`TOpac`).
45    Opac,
46}
47
48impl GencalType {
49    fn table_subtype(self) -> &'static str {
50        match self {
51            Self::Antpos => "KAntPos Jones",
52            Self::Gceff => "EGainCurve",
53            Self::Opac => "TOpac",
54        }
55    }
56}
57
58impl std::str::FromStr for GencalType {
59    type Err = String;
60
61    fn from_str(value: &str) -> Result<Self, Self::Err> {
62        match value.to_ascii_lowercase().as_str() {
63            "antpos" => Ok(Self::Antpos),
64            "gceff" | "gc" => Ok(Self::Gceff),
65            "opac" | "opacity" => Ok(Self::Opac),
66            other => Err(format!(
67                "unsupported gencal caltype {other:?}; expected antpos, gceff, or opac"
68            )),
69        }
70    }
71}

73/// Request for writing one native prior-cal table.
74#[derive(Debug, Clone, PartialEq, Serialize, Deserialize, JsonSchema)]
75pub struct GencalRequest {
76    /// Input MeasurementSet root path.
77    pub measurement_set: PathBuf,
78    /// Output calibration-table path.
79    pub output_table: PathBuf,
80    /// Prior calibration family to generate.
81    pub caltype: GencalType,
82    /// CASA-style antenna selector for `antpos`.
83    #[serde(default)]
84    pub antenna: String,
85    /// CASA-style SPW selector for `opac`.
86    #[serde(default)]
87    pub spw: String,
88    /// Numeric parameters. `antpos` expects triples per selected antenna; `opac`
89    /// expects one opacity per selected SPW.
90    #[serde(default)]
91    pub parameter: Vec<f64>,
92    /// Optional explicit VLA GainCurves table path for `gceff`.
93    #[serde(default)]
94    pub gaincurve_table: Option<PathBuf>,
95}

114/// Errors from native prior-cal table generation.
115#[derive(Debug, Error)]
116pub enum GencalError {
117    /// Opening the input MS failed.
118    #[error("failed to open measurement set {path}: {source}")]
119    OpenMeasurementSet {
120        /// Path that failed.
121        path: String,
122        /// Source error.
123        source: Box<casa_ms::MsError>,
124    },
125    /// Opening the VLA GainCurves table failed.
126    #[error("failed to open VLA GainCurves table {path}: {source}")]
127    OpenGainCurves {
128        /// Path that failed.
129        path: String,
130        /// Source error.
131        source: Box<casa_tables::TableError>,
132    },
133    /// Output table preparation failed.
134    #[error("failed to prepare output caltable {path}: {reason}")]
135    PrepareOutput {
136        /// Path that failed.
137        path: String,
138        /// Reason.
139        reason: String,
140    },
141    /// Saving a table failed.
142    #[error("failed to save caltable {path}: {source}")]
143    Save {
144        /// Path that failed.
145        path: String,
146        /// Source error.
147        source: Box<casa_tables::TableError>,
148    },
149    /// Copying an MS subtable failed.
150    #[error("failed to copy {subtable} into caltable {path}: {source}")]
151    CopySubtable {
152        /// Subtable name.
153        subtable: String,
154        /// Caltable root path.
155        path: String,
156        /// Source error.
157        source: Box<casa_tables::TableError>,
158    },
159    /// The request does not match the supported native surface.
160    #[error("{0}")]
161    InvalidRequest(String),
162    /// A needed value in a CASA table had an unsupported type or shape.
163    #[error("{0}")]
164    UnsupportedTableValue(String),
165}

168struct PriorCalRow {
169    field_id: i32,
170    spw_id: i32,
171    antenna_id: i32,
172    observation_id: i32,
173    fparam: Vec<f32>,
174    snr_value: f32,

177/// Generate one native prior calibration table.
178pub fn gencal(request: &GencalRequest) -> Result<GencalReport, GencalError> {
179    let ms = MeasurementSet::open(&request.measurement_set).map_err(|source| {
180        GencalError::OpenMeasurementSet {
181            path: request.measurement_set.display().to_string(),
182            source: Box::new(source),
183        }
184    })?;
185    let rows = match request.caltype {
186        GencalType::Antpos => antpos_rows(&ms, request)?,
187        GencalType::Gceff => gaincurve_rows(&ms, request)?,
188        GencalType::Opac => opacity_rows(&ms, request)?,
189    };
190    write_float_caltable(&ms, request, &rows)
191}

193fn antpos_rows(
194    ms: &MeasurementSet,
195    request: &GencalRequest,
196) -> Result<Vec<PriorCalRow>, GencalError> {
197    let antenna = ms
198        .antenna()
199        .map_err(|source| GencalError::OpenMeasurementSet {
200            path: "ANTENNA".to_string(),
201            source: Box::new(source),
202        })?;
203    let selected = parse_antenna_offsets(ms, &request.antenna, &request.parameter)?;
204    let mut rows = Vec::with_capacity(antenna.row_count());
205    for antenna_id in 0..antenna.row_count() {
206        let offsets = selected
207            .get(&(antenna_id as i32))
208            .copied()
209            .unwrap_or([0.0, 0.0, 0.0]);
210        rows.push(PriorCalRow {
211            field_id: -1,
212            spw_id: 0,
213            antenna_id: antenna_id as i32,
214            observation_id: 0,
215            fparam: offsets.iter().map(|value| *value as f32).collect(),
216            snr_value: 1.0,
217        });
218    }
219    Ok(rows)
220}

222fn gaincurve_rows(
223    ms: &MeasurementSet,
224    request: &GencalRequest,
225) -> Result<Vec<PriorCalRow>, GencalError> {
226    let source_path = request
227        .gaincurve_table
228        .clone()
229        .or_else(default_vla_gaincurve_table)
230        .ok_or_else(|| {
231            GencalError::InvalidRequest(
232                "gceff requires --gaincurve-table or a CASA data tree at ~/.casa/data/nrao/VLA/GainCurves".to_string(),
233            )
234        })?;
235    let gaincurves = Table::open(TableOptions::new(&source_path)).map_err(|source| {
236        GencalError::OpenGainCurves {
237            path: source_path.display().to_string(),
238            source: Box::new(source),
239        }
240    })?;
241    let antenna_count = ms
242        .antenna()
243        .map_err(|source| GencalError::OpenMeasurementSet {
244            path: "ANTENNA".to_string(),
245            source: Box::new(source),
246        })?
247        .row_count();
248    let antenna_keys = vla_gaincurve_antenna_keys(ms)?;
249    let spw = ms
250        .spectral_window()
251        .map_err(|source| GencalError::OpenMeasurementSet {
252            path: "SPECTRAL_WINDOW".to_string(),
253            source: Box::new(source),
254        })?;
255    let obs_time = first_observation_time(ms).or_else(|_| first_main_time(ms))?;
256    let mut rows = Vec::with_capacity(antenna_count * spw.row_count());
257    for spw_id in 0..spw.row_count() {
258        let center_hz = spw_center_frequency_hz(&spw, spw_id)?;
259        let band = vla_spw_band(&spw, spw_id)?;
260        let efficiency = vla_efficiency(center_hz / 1.0e9) as f32;
261        for (antenna_id, antenna_key) in antenna_keys.iter().enumerate().take(antenna_count) {
262            let mut fparam = matching_gaincurve(
263                &gaincurves,
264                obs_time,
265                center_hz,
266                band.as_deref(),
267                antenna_key,
268            )?;
269            for value in &mut fparam {
270                *value *= efficiency;
271            }
272            rows.push(PriorCalRow {
273                field_id: -1,
274                spw_id: spw_id as i32,
275                antenna_id: antenna_id as i32,
276                observation_id: -1,
277                fparam,
278                snr_value: 0.0,
279            });
280        }
281    }
282    Ok(rows)
283}

285fn opacity_rows(
286    ms: &MeasurementSet,
287    request: &GencalRequest,
288) -> Result<Vec<PriorCalRow>, GencalError> {
289    let antenna_count = ms
290        .antenna()
291        .map_err(|source| GencalError::OpenMeasurementSet {
292            path: "ANTENNA".to_string(),
293            source: Box::new(source),
294        })?
295        .row_count();
296    let selected_spws = parse_spw_selector(ms, &request.spw)?;
297    if request.parameter.len() != selected_spws.len() {
298        return Err(GencalError::InvalidRequest(format!(
299            "opac requires one parameter per selected SPW (got {} parameters for {} SPWs)",
300            request.parameter.len(),
301            selected_spws.len()
302        )));
303    }
304    let mut rows = Vec::with_capacity(antenna_count * selected_spws.len());
305    for (index, spw_id) in selected_spws.iter().copied().enumerate() {
306        for antenna_id in 0..antenna_count {
307            rows.push(PriorCalRow {
308                field_id: -1,
309                spw_id,
310                antenna_id: antenna_id as i32,
311                observation_id: 0,
312                fparam: vec![request.parameter[index] as f32],
313                snr_value: 1.0,
314            });
315        }
316    }
317    Ok(rows)
318}

320fn write_float_caltable(
321    ms: &MeasurementSet,
322    request: &GencalRequest,
323    rows: &[PriorCalRow],
324) -> Result<GencalReport, GencalError> {
325    prepare_output_root(&request.output_table)?;
326    let schema = TableSchema::new(vec![
327        ColumnSchema::scalar(COL_TIME, casa_types::PrimitiveType::Float64),
328        ColumnSchema::scalar(COL_FIELD_ID, casa_types::PrimitiveType::Int32),
329        ColumnSchema::scalar(COL_SPECTRAL_WINDOW_ID, casa_types::PrimitiveType::Int32),
330        ColumnSchema::scalar(COL_ANTENNA1, casa_types::PrimitiveType::Int32),
331        ColumnSchema::scalar(COL_ANTENNA2, casa_types::PrimitiveType::Int32),
332        ColumnSchema::scalar(COL_INTERVAL, casa_types::PrimitiveType::Float64),
333        ColumnSchema::scalar(COL_SCAN_NUMBER, casa_types::PrimitiveType::Int32),
334        ColumnSchema::scalar(COL_OBSERVATION_ID, casa_types::PrimitiveType::Int32),
335        ColumnSchema::array_variable(COL_FPARAM, casa_types::PrimitiveType::Float32, Some(2)),
336        ColumnSchema::array_variable(COL_PARAMERR, casa_types::PrimitiveType::Float32, Some(2)),
337        ColumnSchema::array_variable(COL_FLAG, casa_types::PrimitiveType::Bool, Some(2)),
338        ColumnSchema::array_variable(COL_SNR, casa_types::PrimitiveType::Float32, Some(2)),
339        ColumnSchema::array_variable(COL_WEIGHT, casa_types::PrimitiveType::Float32, Some(2)),
340    ])
341    .expect("valid gencal caltable schema");
342    let subtype = request.caltype.table_subtype();
343    let mut table = Table::with_schema(schema);
344    table.set_info(TableInfo {
345        table_type: TABLE_INFO_TYPE.to_string(),
346        sub_type: subtype.to_string(),
347        readme: Vec::new(),
348    });
349    table.keywords_mut().upsert(
350        KEY_PAR_TYPE,
351        Value::Scalar(ScalarValue::String("Float".to_string())),
352    );
353    table.keywords_mut().upsert(
354        KEY_VIS_CAL,
355        Value::Scalar(ScalarValue::String(subtype.to_string())),
356    );
357    table.keywords_mut().upsert(
358        KEY_MS_NAME,
359        Value::Scalar(ScalarValue::String(
360            ms.path()
361                .and_then(Path::file_name)
362                .map(|name| name.to_string_lossy().to_string())
363                .unwrap_or_else(|| "<in-memory>".to_string()),
364        )),
365    );
366    table.keywords_mut().upsert(
367        KEY_POL_BASIS,
368        Value::Scalar(ScalarValue::String("unknown".to_string())),
369    );
370    table.keywords_mut().upsert(
371        KEY_CASA_VERSION,
372        Value::Scalar(ScalarValue::String("casa-rs".to_string())),
373    );
374    set_fixed_unit_keyword(&mut table, COL_TIME, &["s"]);
375    set_measinfo_keyword(&mut table, COL_TIME, "epoch", Some("UTC"));
376    set_fixed_unit_keyword(&mut table, COL_INTERVAL, &["s"]);
377    for name in STANDARD_SUBTABLE_KEYWORDS {
378        table.keywords_mut().upsert(
379            *name,
380            Value::table_ref(subtable_keyword_value(
381                &request.output_table,
382                &request.output_table.join(name),
383            )),
384        );
385    }
386
387    for row in rows {
388        let len = row.fparam.len();
389        let zeros = vec![0.0_f32; len];
390        let flags = vec![false; len];
391        let snr = vec![row.snr_value; len];
392        table
393            .add_row(RecordValue::new(vec![
394                RecordField::new(COL_TIME, Value::Scalar(ScalarValue::Float64(0.0))),
395                RecordField::new(
396                    COL_FIELD_ID,
397                    Value::Scalar(ScalarValue::Int32(row.field_id)),
398                ),
399                RecordField::new(
400                    COL_SPECTRAL_WINDOW_ID,
401                    Value::Scalar(ScalarValue::Int32(row.spw_id)),
402                ),
403                RecordField::new(
404                    COL_ANTENNA1,
405                    Value::Scalar(ScalarValue::Int32(row.antenna_id)),
406                ),
407                RecordField::new(COL_ANTENNA2, Value::Scalar(ScalarValue::Int32(-1))),
408                RecordField::new(COL_INTERVAL, Value::Scalar(ScalarValue::Float64(0.0))),
409                RecordField::new(COL_SCAN_NUMBER, Value::Scalar(ScalarValue::Int32(-1))),
410                RecordField::new(
411                    COL_OBSERVATION_ID,
412                    Value::Scalar(ScalarValue::Int32(row.observation_id)),
413                ),
414                RecordField::new(COL_FPARAM, Value::Array(f32_column(&row.fparam))),
415                RecordField::new(COL_PARAMERR, Value::Array(f32_column(&zeros))),
416                RecordField::new(COL_FLAG, Value::Array(bool_column(&flags))),
417                RecordField::new(COL_SNR, Value::Array(f32_column(&snr))),
418                RecordField::new(COL_WEIGHT, Value::Array(f32_column(&zeros))),
419            ]))
420            .map_err(|source| GencalError::Save {
421                path: request.output_table.display().to_string(),
422                source: Box::new(source),
423            })?;
424    }
425
426    table
427        .save(
428            TableOptions::new(&request.output_table)
429                .with_data_manager(DataManagerKind::StandardStMan),
430        )
431        .map_err(|source| GencalError::Save {
432            path: request.output_table.display().to_string(),
433            source: Box::new(source),
434        })?;
435    copy_standard_subtables(ms, &request.output_table)?;
436
437    let spectral_window_ids = rows
438        .iter()
439        .map(|row| row.spw_id)
440        .collect::<BTreeSet<_>>()
441        .into_iter()
442        .collect();
443    let antenna_ids = rows
444        .iter()
445        .map(|row| row.antenna_id)
446        .collect::<BTreeSet<_>>()
447        .into_iter()
448        .collect();
449    Ok(GencalReport {
450        output_table: request.output_table.clone(),
451        caltype: request.caltype,
452        table_subtype: subtype.to_string(),
453        row_count: rows.len(),
454        spectral_window_ids,
455        antenna_ids,
456    })
457}

459fn parse_antenna_offsets(
460    ms: &MeasurementSet,
461    antenna_selector: &str,
462    parameters: &[f64],
463) -> Result<BTreeMap<i32, [f64; 3]>, GencalError> {
464    if antenna_selector.trim().is_empty() {
465        return Err(GencalError::InvalidRequest(
466            "antpos currently requires explicit --antenna and --parameter offsets; automatic VLA baseline lookup is not implemented".to_string(),
467        ));
468    }
469    let antenna = ms
470        .antenna()
471        .map_err(|source| GencalError::OpenMeasurementSet {
472            path: "ANTENNA".to_string(),
473            source: Box::new(source),
474        })?;
475    let mut name_to_id = BTreeMap::new();
476    for row in 0..antenna.row_count() {
477        let name = antenna.name(row).map_err(|source| {
478            GencalError::InvalidRequest(format!("failed to read ANTENNA.NAME row {row}: {source}"))
479        })?;
480        name_to_id.insert(name.to_ascii_lowercase(), row as i32);
481    }
482    let selected = antenna_selector
483        .split(',')
484        .map(str::trim)
485        .filter(|part| !part.is_empty())
486        .map(|part| {
487            if let Ok(index) = part.parse::<i32>() {
488                return Ok(index);
489            }
490            name_to_id
491                .get(&part.to_ascii_lowercase())
492                .copied()
493                .ok_or_else(|| GencalError::InvalidRequest(format!("unknown antenna {part:?}")))
494        })
495        .collect::<Result<Vec<_>, _>>()?;
496    if parameters.len() != selected.len() * 3 {
497        return Err(GencalError::InvalidRequest(format!(
498            "antpos requires three parameters per selected antenna (got {} parameters for {} antennas)",
499            parameters.len(),
500            selected.len()
501        )));
502    }
503    let mut offsets = BTreeMap::new();
504    for (index, antenna_id) in selected.iter().copied().enumerate() {
505        offsets.insert(
506            antenna_id,
507            [
508                parameters[index * 3],
509                parameters[index * 3 + 1],
510                parameters[index * 3 + 2],
511            ],
512        );
513    }
514    Ok(offsets)
515}

517fn parse_spw_selector(ms: &MeasurementSet, selector: &str) -> Result<Vec<i32>, GencalError> {
518    let count = ms
519        .spectral_window()
520        .map_err(|source| GencalError::OpenMeasurementSet {
521            path: "SPECTRAL_WINDOW".to_string(),
522            source: Box::new(source),
523        })?
524        .row_count() as i32;
525    if selector.trim().is_empty() {
526        return Ok((0..count).collect());
527    }
528    let mut ids = BTreeSet::new();
529    for part in selector.split(',').map(str::trim).filter(|p| !p.is_empty()) {
530        if let Some((start, end)) = part.split_once('~') {
531            let start = start.parse::<i32>().map_err(|_| {
532                GencalError::InvalidRequest(format!("invalid SPW selector {part:?}"))
533            })?;
534            let end = end.parse::<i32>().map_err(|_| {
535                GencalError::InvalidRequest(format!("invalid SPW selector {part:?}"))
536            })?;
537            for id in start.min(end)..=start.max(end) {
538                ids.insert(id);
539            }
540        } else {
541            ids.insert(part.parse::<i32>().map_err(|_| {
542                GencalError::InvalidRequest(format!("invalid SPW selector {part:?}"))
543            })?);
544        }
545    }
546    if ids.iter().any(|id| *id < 0 || *id >= count) {
547        return Err(GencalError::InvalidRequest(format!(
548            "SPW selector {selector:?} is outside 0..{}",
549            count.saturating_sub(1)
550        )));
551    }
552    Ok(ids.into_iter().collect())
553}

555fn matching_gaincurve(
556    table: &Table,
557    time_seconds: f64,
558    frequency_hz: f64,
559    band: Option<&str>,
560    antenna_key: &str,
561) -> Result<Vec<f32>, GencalError> {
562    if let Some(band) = band {
563        if let Some(gain) = matching_gaincurve_from_rows(
564            table,
565            time_seconds,
566            |row| {
567                get_string(table, row, COL_BANDNAME)
568                    .map(|row_band| row_band == band)
569                    .unwrap_or(false)
570            },
571            antenna_key,
572        )? {
573            return Ok(gain);
574        }
575    }
576    matching_gaincurve_from_rows(
577        table,
578        time_seconds,
579        |row| {
580            let Ok(bfreq) = get_f64(table, row, COL_BFREQ) else {
581                return false;
582            };
583            let Ok(efreq) = get_f64(table, row, COL_EFREQ) else {
584                return false;
585            };
586            bfreq <= frequency_hz && efreq > frequency_hz
587        },
588        antenna_key,
589    )?
590    .ok_or_else(|| {
591        GencalError::InvalidRequest(format!(
592            "no VLA GainCurves row covers antenna {antenna_key}, time {time_seconds}, frequency {frequency_hz}"
593        ))
594    })
595}
596
597fn matching_gaincurve_from_rows(
598    table: &Table,
599    time_seconds: f64,
600    mut row_matches: impl FnMut(usize) -> bool,
601    antenna_key: &str,
602) -> Result<Option<Vec<f32>>, GencalError> {
603    let nominal_key = "0";
604    let mut nominal = None;
605    for row in 0..table.row_count() {
606        if get_f64(table, row, COL_BTIME)? > time_seconds
607            || get_f64(table, row, COL_ETIME)? <= time_seconds
608            || !row_matches(row)
609        {
610            continue;
611        }
612        let antenna = get_string(table, row, COL_ANTENNA)?;
613        let gain = get_f32_array_fortran(table, row, COL_GAIN)?;
614        if antenna == antenna_key {
615            return Ok(Some(gain));
616        }
617        if antenna == nominal_key {
618            nominal = Some(gain);
619        }
620    }
621    Ok(nominal)
622}

624fn spw_center_frequency_hz(
625    spw: &casa_ms::subtables::MsSpectralWindow<'_>,
626    row: usize,
627) -> Result<f64, GencalError> {
628    let freqs = spw.chan_freq(row).map_err(|source| {
629        GencalError::InvalidRequest(format!(
630            "failed to read SPECTRAL_WINDOW row {row}: {source}"
631        ))
632    })?;
633    casa_middle_channel_frequency_hz(&freqs, row)
634}

636fn casa_middle_channel_frequency_hz(freqs: &[f64], row: usize) -> Result<f64, GencalError> {
637    if freqs.is_empty() {
638        return Err(GencalError::InvalidRequest(format!(
639            "SPECTRAL_WINDOW row {row} has no channel frequencies"
640        )));
641    }
642    // CASA EGainCurve uses chanfreqs(chanfreqs.nelements()/2), not the
643    // arithmetic center, for the VLA efficiency lookup.
644    Ok(freqs[freqs.len() / 2])
645}

661fn vla_gaincurve_antenna_keys(ms: &MeasurementSet) -> Result<Vec<String>, GencalError> {
662    let table = ms
663        .subtable(SubtableId::Antenna)
664        .ok_or_else(|| GencalError::InvalidRequest("missing ANTENNA subtable".to_string()))?;
665    let mut keys = Vec::with_capacity(table.row_count());
666    for row in 0..table.row_count() {
667        let name = get_string(table, row, COL_NAME)?;
668        let digits = name
669            .chars()
670            .skip_while(|ch| !ch.is_ascii_digit())
671            .collect::<String>();
672        let stripped = digits.trim_start_matches('0');
673        keys.push(if stripped.is_empty() {
674            row.to_string()
675        } else {
676            stripped.to_string()
677        });
678    }
679    Ok(keys)
680}

682fn vla_efficiency(frequency_ghz: f64) -> f64 {
683    const FREQ: [f64; 42] = [
684        1.0,
685        1.1,
686        1.2,
687        1.3,
688        1.4,
689        1.5,
690        1.6,
691        1.7,
692        1.8,
693        1.9,
694        2.0,
695        2.0 + 1e-9,
696        2.3,
697        2.7,
698        3.0,
699        3.5,
700        3.7,
701        4.0,
702        4.0 + 1e-9,
703        5.0,
704        6.0,
705        7.0,
706        8.0,
707        8.0 + 1e-9,
708        12.0,
709        12.0 + 1e-9,
710        13.0,
711        14.0,
712        15.0,
713        16.0,
714        17.0,
715        18.0,
716        19.0,
717        24.0,
718        26.0,
719        26.5,
720        28.0,
721        33.0,
722        38.0,
723        40.0,
724        43.0,
725        48.0,
726    ];
727    const EFF: [f64; 42] = [
728        0.45, 0.48, 0.48, 0.45, 0.46, 0.45, 0.43, 0.44, 0.44, 0.49, 0.48, 0.52, 0.52, 0.51, 0.53,
729        0.55, 0.53, 0.54, 0.55, 0.54, 0.56, 0.62, 0.64, 0.60, 0.60, 0.65, 0.65, 0.62, 0.58, 0.59,
730        0.60, 0.60, 0.57, 0.52, 0.48, 0.50, 0.49, 0.42, 0.35, 0.29, 0.28, 0.26,
731    ];
732    let scaled = EFF.map(|value| (value / 5.622).sqrt());
733    if frequency_ghz <= FREQ[0] {
734        return scaled[0];
735    }
736    for index in 1..FREQ.len() {
737        if frequency_ghz <= FREQ[index] {
738            let fraction = (frequency_ghz - FREQ[index - 1]) / (FREQ[index] - FREQ[index - 1]);
739            return scaled[index - 1] + (scaled[index] - scaled[index - 1]) * fraction;
740        }
741    }
742    scaled[scaled.len() - 1]
743}

745fn first_observation_time(ms: &MeasurementSet) -> Result<f64, GencalError> {
746    let table = ms
747        .subtable(SubtableId::Observation)
748        .ok_or_else(|| GencalError::InvalidRequest("missing OBSERVATION subtable".to_string()))?;
749    match table
750        .cell_accessor(0, COL_TIME_RANGE)
751        .and_then(|cell| cell.array())
752        .map_err(|error| GencalError::UnsupportedTableValue(error.to_string()))?
753    {
754        ArrayValue::Float64(values) => values.iter().next().copied().ok_or_else(|| {
755            GencalError::UnsupportedTableValue("OBSERVATION.TIME_RANGE is empty".to_string())
756        }),
757        other => Err(GencalError::UnsupportedTableValue(format!(
758            "OBSERVATION.TIME_RANGE has shape {:?}, expected Float64 array",
759            other.shape()
760        ))),
761    }
762}

771fn get_string(table: &Table, row: usize, column: &str) -> Result<String, GencalError> {
772    match table
773        .cell_accessor(row, column)
774        .and_then(|cell| cell.scalar())
775        .map_err(|error| GencalError::UnsupportedTableValue(error.to_string()))?
776    {
777        ScalarValue::String(value) => Ok(value.clone()),
778        other => Err(GencalError::UnsupportedTableValue(format!(
779            "{column} row {row} is {other:?}, expected string"
780        ))),
781    }
782}

784fn get_f64(table: &Table, row: usize, column: &str) -> Result<f64, GencalError> {
785    match table
786        .cell_accessor(row, column)
787        .and_then(|cell| cell.scalar())
788        .map_err(|error| GencalError::UnsupportedTableValue(error.to_string()))?
789    {
790        ScalarValue::Float64(value) => Ok(*value),
791        ScalarValue::Float32(value) => Ok(f64::from(*value)),
792        other => Err(GencalError::UnsupportedTableValue(format!(
793            "{column} row {row} is {other:?}, expected float"
794        ))),
795    }
796}

798fn get_f32_array_fortran(table: &Table, row: usize, column: &str) -> Result<Vec<f32>, GencalError> {
799    match table
800        .cell_accessor(row, column)
801        .and_then(|cell| cell.array())
802        .map_err(|error| GencalError::UnsupportedTableValue(error.to_string()))?
803    {
804        ArrayValue::Float32(values) => {
805            let shape = values.shape();
806            if shape.len() != 2 {
807                return Ok(values.iter().copied().collect());
808            }
809            let mut out = Vec::with_capacity(shape[0] * shape[1]);
810            for j in 0..shape[1] {
811                for i in 0..shape[0] {
812                    out.push(values[[i, j]]);
813                }
814            }
815            Ok(out)
816        }
817        other => Err(GencalError::UnsupportedTableValue(format!(
818            "{column} row {row} has shape {:?}, expected Float32 array",
819            other.shape()
820        ))),
821    }
822}

838fn prepare_output_root(path: &Path) -> Result<(), GencalError> {
839    if path.exists() {
840        fs::remove_dir_all(path)
841            .or_else(|_| fs::remove_file(path))
842            .map_err(|error| GencalError::PrepareOutput {
843                path: path.display().to_string(),
844                reason: error.to_string(),
845            })?;
846    }
847    if let Some(parent) = path.parent() {
848        fs::create_dir_all(parent).map_err(|error| GencalError::PrepareOutput {
849            path: path.display().to_string(),
850            reason: error.to_string(),
851        })?;
852    }
853    Ok(())
854}

856fn copy_standard_subtables(ms: &MeasurementSet, output_root: &Path) -> Result<(), GencalError> {
857    for (id, name) in [
858        (SubtableId::Observation, "OBSERVATION"),
859        (SubtableId::Antenna, "ANTENNA"),
860        (SubtableId::Field, "FIELD"),
861        (SubtableId::SpectralWindow, "SPECTRAL_WINDOW"),
862        (SubtableId::History, "HISTORY"),
863    ] {
864        if let Some(table) = ms.subtable(id) {
865            table
866                .save(TableOptions::new(output_root.join(name)))
867                .map_err(|source| GencalError::CopySubtable {
868                    subtable: name.to_string(),
869                    path: output_root.display().to_string(),
870                    source: Box::new(source),
871                })?;
872        }
873    }
874    Ok(())
875}
