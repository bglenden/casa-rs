126    pub fn open(project_root: impl AsRef<Path>) -> Result<Self, CorpusError> {
127        let project_root = project_root.as_ref();
128        if !project_root.is_absolute() {
129            return Err(CorpusError::ProjectRootMustBeAbsolute {
130                path: project_root.to_owned(),
131            });
132        }
133        let metadata = fs::metadata(project_root).map_err(|source| CorpusError::Io {
134            action: "inspect corpus project root",
135            path: project_root.to_owned(),
136            source,
137        })?;
138        if !metadata.is_dir() {
139            return Err(CorpusError::ProjectRootNotDirectory {
140                path: project_root.to_owned(),
141            });
142        }
143        let managed_dir = project_root.join(".casa-rs/corpus");
144        fs::create_dir_all(&managed_dir).map_err(|source| CorpusError::Io {
145            action: "create corpus directory",
146            path: managed_dir.clone(),
147            source,
148        })?;
149        let index = Self {
150            project_root: project_root.to_owned(),
151            managed_dir,
152        };
153        index.initialize()?;
154        Ok(index)
155    }

170    /// Compare a complete metadata-only project-document inventory with the
171    /// last successfully indexed source state.
172    pub fn plan_project_sources(
173        &self,
174        sources: &[ProjectCorpusSource],
175    ) -> Result<ProjectCorpusPlan, CorpusError> {
176        let sources = validated_project_sources(sources)?;
177        let connection = self.connection()?;
178        let existing = load_project_source_fingerprints(&connection)?;
179        let mut extract_paths = Vec::new();
180        let mut unchanged_paths = Vec::new();
181        for (path, source) in &sources {
182            let fingerprint = project_source_fingerprint(source)?;
183            if existing.get(path) == Some(&fingerprint) {
184                unchanged_paths.push(path.clone());
185            } else {
186                extract_paths.push(path.clone());
187            }
188        }
189        let removed_paths = existing
190            .keys()
191            .filter(|path| !sources.contains_key(*path))
192            .cloned()
193            .collect();
194        Ok(ProjectCorpusPlan {
195            schema_version: CORPUS_SCHEMA_VERSION,
196            extract_paths,
197            unchanged_paths,
198            removed_paths,
199        })
200    }

202    /// Atomically reconcile a complete project-source inventory while only
203    /// replacing documents extracted from changed sources. Failed sources keep
204    /// their last usable indexed content and fingerprint so a later refresh
205    /// retries them.
206    pub fn index_documents_with_project_sources(
207        &self,
208        documents: &[CorpusDocumentInput],
209        remove_missing_layers: &BTreeSet<CorpusLayer>,
210        project_sources: &[ProjectCorpusSource],
211        failed_project_sources: &BTreeSet<PathBuf>,
212    ) -> Result<CorpusIndexReport, CorpusError> {
213        self.index_documents_internal(
214            documents,
215            remove_missing_layers,
216            Some(project_sources),
217            failed_project_sources,
218        )
219    }

221    fn index_documents_internal(
222        &self,
223        documents: &[CorpusDocumentInput],
224        remove_missing_layers: &BTreeSet<CorpusLayer>,
225        project_sources: Option<&[ProjectCorpusSource]>,
226        failed_project_sources: &BTreeSet<PathBuf>,
227    ) -> Result<CorpusIndexReport, CorpusError> {
228        let project_sources = project_sources.map(validated_project_sources).transpose()?;
229        if project_sources.is_some()
230            && remove_missing_layers.contains(&CorpusLayer::ProjectDocument)
231        {
232            return Err(CorpusError::InvalidDocument(
233                "project_document removal must use the project-source snapshot".to_owned(),
234            ));
235        }
236        for path in failed_project_sources {
237            validate_project_source_path(path)?;
238            if project_sources
239                .as_ref()
240                .is_some_and(|sources| !sources.contains_key(path))
241            {
242                return Err(CorpusError::InvalidDocument(format!(
243                    "failed project source {} is absent from the source snapshot",
244                    path.display()
245                )));
246            }
247        }
248        let mut connection = self.connection()?;
249        let transaction = connection.transaction()?;
250        let mut indexed_documents = 0;
251        let mut unchanged_documents = 0;
252        let mut seen = BTreeSet::new();
253
254        let mut successful_project_paths = BTreeSet::new();
255        let mut project_document_ids: BTreeMap<PathBuf, BTreeSet<String>> = BTreeMap::new();
256        for input in documents {
257            if input.layer != CorpusLayer::ProjectDocument {
258                continue;
259            }
260            let path = input.citation.source_path.clone().ok_or_else(|| {
261                CorpusError::InvalidDocument(format!(
262                    "project document {} has no source path",
263                    input.id
264                ))
265            })?;
266            validate_project_source_path(&path)?;
267            if let Some(sources) = &project_sources
268                && !sources.contains_key(&path)
269            {
270                return Err(CorpusError::InvalidDocument(format!(
271                    "project document {} source {} is absent from the source snapshot",
272                    input.id,
273                    path.display()
274                )));
275            }
276            successful_project_paths.insert(path.clone());
277            project_document_ids
278                .entry(path)
279                .or_default()
280                .insert(input.id.clone());
281        }
282        if successful_project_paths
283            .iter()
284            .any(|path| failed_project_sources.contains(path))
285        {
286            return Err(CorpusError::InvalidDocument(
287                "a project source cannot be both successfully extracted and failed".to_owned(),
288            ));
289        }
290
291        let mut old_project_document_ids = BTreeMap::new();
292        if project_sources.is_some() {
293            for path in &successful_project_paths {
294                let ids = project_document_ids_for_source(&transaction, path)?;
295                for id in &ids {
296                    delete_document(&transaction, id)?;
297                }
298                old_project_document_ids.insert(path.clone(), ids);
299            }
300        }
301
302        for input in documents {
303            validate_input(input)?;
304            if !seen.insert(input.id.clone()) {
305                return Err(CorpusError::DuplicateDocumentId(input.id.clone()));
306            }
307            let content_sha256 = sha256(input.content.as_bytes());
308            let chunks = chunk_content(&input.content);
309            let first_citation_json = chunk_citation_json(&input.citation, &chunks[0])?;
310            let expected_fingerprint = index_fingerprint(
311                layer_name(input.layer),
312                &input.title,
313                &input.source_identity,
314                &content_sha256,
315                input.redistribution_cleared,
316                &first_citation_json,
317            )?;
318            let current_fingerprint: Option<String> = transaction
319                .query_row(
320                    "SELECT d.layer, d.title, d.source_identity, d.content_sha256,
321                            d.redistribution_cleared, c.citation_json
322                     FROM documents d
323                     JOIN chunks c ON c.document_id = d.id AND c.ordinal = 0
324                     WHERE d.id = ?1",
325                    [&input.id],
326                    |row| {
327                        let layer = row.get::<_, String>(0)?;
328                        let title = row.get::<_, String>(1)?;
329                        let source_identity = row.get::<_, String>(2)?;
330                        let content_sha256 = row.get::<_, String>(3)?;
331                        let redistribution_cleared = row.get::<_, bool>(4)?;
332                        let citation_json = row.get::<_, String>(5)?;
333                        index_fingerprint(
334                            &layer,
335                            &title,
336                            &source_identity,
337                            &content_sha256,
338                            redistribution_cleared,
339                            &citation_json,
340                        )
341                        .map_err(|error| {
342                            rusqlite::Error::FromSqlConversionFailure(
343                                5,
344                                rusqlite::types::Type::Text,
345                                Box::new(error),
346                            )
347                        })
348                    },
349                )
350                .optional()?;
351            if current_fingerprint.as_deref() == Some(expected_fingerprint.as_str()) {
352                unchanged_documents += 1;
353                continue;
354            }
355
356            delete_document(&transaction, &input.id)?;
357            let project_source_path = (input.layer == CorpusLayer::ProjectDocument)
358                .then(|| {
359                    input
360                        .citation
361                        .source_path
362                        .as_ref()
363                        .map(|path| path.to_string_lossy().into_owned())
364                })
365                .flatten();
366            transaction.execute(
367                "INSERT INTO documents
368                 (id, layer, title, source_identity, content_sha256, redistribution_cleared,
369                  project_source_path)
370                 VALUES (?1, ?2, ?3, ?4, ?5, ?6, ?7)",
371                params![
372                    input.id,
373                    layer_name(input.layer),
374                    input.title,
375                    input.source_identity,
376                    content_sha256,
377                    input.redistribution_cleared,
378                    project_source_path,
379                ],
380            )?;
381            for (ordinal, chunk) in chunks.into_iter().enumerate() {
382                let chunk_id = format!("{}:{ordinal}", input.id);
383                let citation_json = chunk_citation_json(&input.citation, &chunk)?;
384                transaction.execute(
385                    "INSERT INTO chunks
386                     (id, document_id, ordinal, text, citation_json)
387                     VALUES (?1, ?2, ?3, ?4, ?5)",
388                    params![
389                        chunk_id,
390                        input.id,
391                        ordinal as u64,
392                        chunk.text,
393                        citation_json
394                    ],
395                )?;
396                transaction.execute(
397                    "INSERT INTO chunks_fts (chunk_id, text) VALUES (?1, ?2)",
398                    params![chunk_id, chunk.text],
399                )?;
400            }
401            indexed_documents += 1;
402        }
403
404        let mut removed_documents = old_project_document_ids
405            .iter()
406            .map(|(path, old_ids)| {
407                let new_ids = project_document_ids.get(path).cloned().unwrap_or_default();
408                old_ids.difference(&new_ids).count()
409            })
410            .sum();
411        for layer in remove_missing_layers {
412            let mut statement = transaction.prepare("SELECT id FROM documents WHERE layer = ?1")?;
413            let existing: Vec<String> = statement
414                .query_map([layer_name(*layer)], |row| row.get(0))?
415                .collect::<Result<_, _>>()?;
416            drop(statement);
417            for id in existing {
418                if !seen.contains(&id) {
419                    delete_document(&transaction, &id)?;
420                    removed_documents += 1;
421                }
422            }
423        }
424
425        if let Some(project_sources) = &project_sources {
426            let existing = load_project_source_fingerprints(&transaction)?;
427            for path in existing
428                .keys()
429                .filter(|path| !project_sources.contains_key(*path))
430            {
431                let ids = project_document_ids_for_source(&transaction, path)?;
432                for id in &ids {
433                    delete_document(&transaction, id)?;
434                }
435                removed_documents += ids.len();
436                transaction.execute(
437                    "DELETE FROM project_sources WHERE relative_path = ?1",
438                    [path.to_string_lossy().as_ref()],
439                )?;
440            }
441            for (path, source) in project_sources {
442                if failed_project_sources.contains(path) {
443                    continue;
444                }
445                let fingerprint = project_source_fingerprint(source)?;
446                let existing_fingerprint = existing.get(path);
447                if successful_project_paths.contains(path) {
448                    transaction.execute(
449                        "INSERT OR REPLACE INTO project_sources
450                         (relative_path, fingerprint_json) VALUES (?1, ?2)",
451                        params![path.to_string_lossy(), fingerprint],
452                    )?;
453                } else if existing_fingerprint == Some(&fingerprint) {
454                    unchanged_documents +=
455                        project_document_ids_for_source(&transaction, path)?.len();
456                } else {
457                    return Err(CorpusError::InvalidDocument(format!(
458                        "changed project source {} supplied neither extracted documents nor a failure marker",
459                        path.display()
460                    )));
461                }
462            }
463        }
464        let chunk_count = transaction.query_row("SELECT COUNT(*) FROM chunks", [], |row| {
465            row.get::<_, u64>(0)
466        })? as usize;
467        transaction.commit()?;
468        Ok(CorpusIndexReport {
469            schema_version: CORPUS_SCHEMA_VERSION,
470            retrieval_engine: "sqlite_fts5_unicode61".to_owned(),
471            indexed_documents,
472            unchanged_documents,
473            removed_documents,
474            chunk_count,
475        })
476    }

478    pub fn documents(&self) -> Result<Vec<CorpusDocument>, CorpusError> {
479        let connection = self.connection()?;
480        let mut statement = connection.prepare(
481            "SELECT d.id, d.layer, d.title, d.source_identity, d.content_sha256,
482                    d.redistribution_cleared, COUNT(c.id)
483             FROM documents d LEFT JOIN chunks c ON c.document_id = d.id
484             GROUP BY d.id ORDER BY d.layer, d.title, d.id",
485        )?;
486        statement
487            .query_map([], |row| {
488                Ok(CorpusDocument {
489                    id: row.get(0)?,
490                    layer: parse_layer(&row.get::<_, String>(1)?).map_err(|error| {
491                        rusqlite::Error::FromSqlConversionFailure(
492                            1,
493                            rusqlite::types::Type::Text,
494                            Box::new(error),
495                        )
496                    })?,
497                    title: row.get(2)?,
498                    source_identity: row.get(3)?,
499                    content_sha256: row.get(4)?,
500                    redistribution_cleared: row.get(5)?,
501                    chunk_count: row.get::<_, u64>(6)? as usize,
502                })
503            })?
504            .collect::<Result<Vec<_>, _>>()
505            .map_err(CorpusError::Sqlite)
506    }

512    pub fn search_layers(
513        &self,
514        query: &str,
515        limit: usize,
516        layers: &BTreeSet<CorpusLayer>,
517    ) -> Result<Vec<CorpusSearchHit>, CorpusError> {
518        let query = query.trim();
519        if query.is_empty() || limit == 0 {
520            return Ok(Vec::new());
521        }
522        let connection = self.connection()?;
523        let expression = fts_expression(query);
524        if expression.is_empty() {
525            return Ok(Vec::new());
526        }
527        let mut statement = connection.prepare(
528            "SELECT c.id, c.document_id, d.layer, d.title, c.text, c.citation_json,
529                    bm25(chunks_fts)
530             FROM chunks_fts
531             JOIN chunks c ON c.id = chunks_fts.chunk_id
532             JOIN documents d ON d.id = c.document_id
533             WHERE chunks_fts MATCH ?1
534               AND (?3 = '' OR instr(',' || ?3 || ',', ',' || d.layer || ',') > 0)
535             ORDER BY bm25(chunks_fts), c.id
536             LIMIT ?2",
537        )?;
538        let layer_filter = layers
539            .iter()
540            .map(|layer| layer_name(*layer))
541            .collect::<Vec<_>>()
542            .join(",");
543        let rows = statement.query_map(params![expression, limit as u64, layer_filter], |row| {
544            Ok((
545                row.get::<_, String>(0)?,
546                row.get::<_, String>(1)?,
547                row.get::<_, String>(2)?,
548                row.get::<_, String>(3)?,
549                row.get::<_, String>(4)?,
550                row.get::<_, String>(5)?,
551                row.get::<_, f64>(6)?,
552            ))
553        })?;
554        let mut hits = Vec::new();
555        for row in rows {
556            let (chunk_id, document_id, layer, title, text, citation_json, rank) = row?;
557            hits.push(CorpusSearchHit {
558                chunk_id,
559                document_id,
560                layer: parse_layer(&layer)?,
561                title,
562                text,
563                score: (1.0 / (1.0 + rank.abs())) as f32,
564                citation: serde_json::from_str(&citation_json)?,
565                untrusted_evidence: true,
566            });
567        }
568        Ok(hits)
569    }

571    fn initialize(&self) -> Result<(), CorpusError> {
572        let connection = self.connection()?;
573        connection.execute_batch(
574            "PRAGMA foreign_keys = ON;
575             CREATE TABLE IF NOT EXISTS corpus_meta (
576                 key TEXT PRIMARY KEY,
577                 value TEXT NOT NULL
578             );
579             CREATE TABLE IF NOT EXISTS documents (
580                 id TEXT PRIMARY KEY,
581                 layer TEXT NOT NULL,
582                 title TEXT NOT NULL,
583                 source_identity TEXT NOT NULL,
584                 content_sha256 TEXT NOT NULL,
585                 redistribution_cleared INTEGER NOT NULL,
586                 project_source_path TEXT
587             );
588             CREATE TABLE IF NOT EXISTS chunks (
589                 id TEXT PRIMARY KEY,
590                 document_id TEXT NOT NULL REFERENCES documents(id) ON DELETE CASCADE,
591                 ordinal INTEGER NOT NULL,
592                 text TEXT NOT NULL,
593                 citation_json TEXT NOT NULL,
594                 UNIQUE(document_id, ordinal)
595             );
596             CREATE VIRTUAL TABLE IF NOT EXISTS chunks_fts USING fts5(
597                 chunk_id UNINDEXED,
598                 text,
599                 tokenize = 'unicode61'
600             );
601             CREATE TABLE IF NOT EXISTS project_sources (
602                 relative_path TEXT PRIMARY KEY,
603                 fingerprint_json TEXT NOT NULL
604             );",
605        )?;
606        let stored_version: Option<String> = connection
607            .query_row(
608                "SELECT value FROM corpus_meta WHERE key = 'schema_version'",
609                [],
610                |row| row.get(0),
611            )
612            .optional()?;
613        match stored_version.as_deref() {
614            None | Some("3") => {}
615            Some("2") => {
616                if !table_has_column(&connection, "documents", "project_source_path")? {
617                    connection.execute(
618                        "ALTER TABLE documents ADD COLUMN project_source_path TEXT",
619                        [],
620                    )?;
621                }
622            }
623            Some(version) => {
624                return Err(CorpusError::UnsupportedSchemaVersion(version.to_owned()));
625            }
626        }
627        // Schema v2 project documents predate the explicit source-path column.
628        // Backfill it from the durable per-chunk citation so the first v3
629        // incremental replacement can remove every old page from that source.
630        connection.execute(
631            "UPDATE documents
632             SET project_source_path = (
633                 SELECT json_extract(c.citation_json, '$.source_path')
634                 FROM chunks c
635                 WHERE c.document_id = documents.id
636                 ORDER BY c.ordinal
637                 LIMIT 1
638             )
639             WHERE layer = 'project_document' AND project_source_path IS NULL",
640            [],
641        )?;
642        connection.execute(
643            "INSERT OR REPLACE INTO corpus_meta (key, value) VALUES ('schema_version', ?1)",
644            [CORPUS_SCHEMA_VERSION.to_string()],
645        )?;
646        Ok(())
647    }

656#[derive(Debug)]
657struct TextChunk {
658    text: String,
659    line_start: u32,
660    line_end: u32,
661}
662
663fn chunk_content(content: &str) -> Vec<TextChunk> {
664    let mut chunks = Vec::new();
665    let mut current = String::new();
666    let mut line_start = 1_u32;
667    let mut line_end = 1_u32;
668    for (index, line) in content.lines().enumerate() {
669        let line_number = index as u32 + 1;
670        let addition = line.len() + usize::from(!current.is_empty());
671        if !current.is_empty() && current.len().saturating_add(addition) > MAX_CHUNK_BYTES {
672            chunks.push(TextChunk {
673                text: std::mem::take(&mut current),
674                line_start,
675                line_end,
676            });
677            line_start = line_number;
678        }
679        if !current.is_empty() {
680            current.push('\n');
681        }
682        current.push_str(line);
683        line_end = line_number;
684    }
685    if !current.is_empty() {
686        chunks.push(TextChunk {
687            text: current,
688            line_start,
689            line_end,
690        });
691    }
692    if chunks.is_empty() && !content.is_empty() {
693        chunks.push(TextChunk {
694            text: content.to_owned(),
695            line_start: 1,
696            line_end: 1,
697        });
698    }
699    chunks
700}

702fn chunk_citation_json(
703    base: &CorpusCitation,
704    chunk: &TextChunk,
705) -> Result<String, serde_json::Error> {
706    let mut citation = base.clone();
707    citation.line_start.get_or_insert(chunk.line_start);
708    citation.line_end.get_or_insert(chunk.line_end);
709    if citation.section.is_none() {
710        citation.section = section_or_symbol_hint(&chunk.text);
711    }
712    serde_json::to_string(&citation)
713}

744fn validated_project_sources(
745    sources: &[ProjectCorpusSource],
746) -> Result<BTreeMap<PathBuf, ProjectCorpusSource>, CorpusError> {
747    let mut validated = BTreeMap::new();
748    for source in sources {
749        validate_project_source_path(&source.relative_path)?;
750        if source.file_type.trim().is_empty() || source.file_identity.trim().is_empty() {
751            return Err(CorpusError::InvalidDocument(format!(
752                "project source {} has incomplete file identity metadata",
753                source.relative_path.display()
754            )));
755        }
756        let extension = source
757            .relative_path
758            .extension()
759            .and_then(|value| value.to_str())
760            .unwrap_or_default();
761        if !extension.eq_ignore_ascii_case(&source.file_type) {
762            return Err(CorpusError::InvalidDocument(format!(
763                "project source {} type does not match its extension",
764                source.relative_path.display()
765            )));
766        }
767        if validated
768            .insert(source.relative_path.clone(), source.clone())
769            .is_some()
770        {
771            return Err(CorpusError::InvalidDocument(format!(
772                "duplicate project source {}",
773                source.relative_path.display()
774            )));
775        }
776    }
777    Ok(validated)
778}

780fn validate_project_source_path(path: &Path) -> Result<(), CorpusError> {
781    let mut components = path.components();
782    if !matches!(components.next(), Some(Component::Normal(value)) if value == "documents")
783        || components.next().is_none()
784        || path.components().any(|component| {
785            matches!(
786                component,
787                Component::ParentDir
788                    | Component::RootDir
789                    | Component::Prefix(_)
790                    | Component::CurDir
791            )
792        })
793    {
794        return Err(CorpusError::InvalidDocument(format!(
795            "project source path must be a relative file under documents/: {}",
796            path.display()
797        )));
798    }
799    Ok(())
800}

844fn section_or_symbol_hint(text: &str) -> Option<String> {
845    const DECLARATIONS: [&str; 14] = [
846        "pub fn ",
847        "fn ",
848        "pub struct ",
849        "struct ",
850        "pub enum ",
851        "enum ",
852        "pub trait ",
853        "trait ",
854        "impl ",
855        "package func ",
856        "public func ",
857        "private func ",
858        "func ",
859        "class ",
860    ];
861    for line in text.lines() {
862        let trimmed = line.trim();
863        if let Some(heading) = trimmed.strip_prefix('#') {
864            let heading = heading.trim_start_matches('#').trim();
865            if !heading.is_empty() {
866                return Some(heading.chars().take(160).collect());
867            }
868        }
869        if DECLARATIONS
870            .iter()
871            .any(|prefix| trimmed.starts_with(prefix))
872        {
873            return Some(trimmed.chars().take(160).collect());
874        }
875    }
876    None
877}

879fn fts_expression(query: &str) -> String {
880    tokens(query)
881        .into_iter()
882        .take(16)
883        .map(|term| format!("\"{}\"", term.replace('"', "\"\"")))
884        .collect::<Vec<_>>()
885        .join(" OR ")
886}
887
888fn tokens(text: &str) -> Vec<String> {
889    text.split(|character: char| !character.is_alphanumeric() && character != '_')
890        .filter(|term| term.len() > 1)
891        .map(str::to_lowercase)
892        .collect()
893}

905fn validate_input(input: &CorpusDocumentInput) -> Result<(), CorpusError> {
906    if input.id.trim().is_empty() || input.title.trim().is_empty() {
907        return Err(CorpusError::InvalidDocument(
908            "document id and title must not be empty".to_owned(),
909        ));
910    }
911    if input.content.trim().is_empty() {
912        return Err(CorpusError::InvalidDocument(format!(
913            "document {} has no extracted content",
914            input.id
915        )));
916    }
917    if let Some(path) = &input.citation.source_path
918        && (path.is_absolute()
919            || path
920                .components()
921                .any(|component| matches!(component, Component::ParentDir | Component::RootDir)))
922    {
923        return Err(CorpusError::InvalidDocument(format!(
924            "document {} citation path must be project/release relative",
925            input.id
926        )));
927    }
928    if input.layer == CorpusLayer::Baseline && !input.redistribution_cleared {
929        return Err(CorpusError::BaselineRedistributionNotCleared(
930            input.id.clone(),
931        ));
932    }
933    Ok(())
934}

936fn layer_name(layer: CorpusLayer) -> &'static str {
937    match layer {
938        CorpusLayer::Baseline => "baseline",
939        CorpusLayer::ProjectDocument => "project_document",
940        CorpusLayer::ReleaseSource => "release_source",
941        CorpusLayer::LiveSource => "live_source",
942    }

945fn parse_layer(value: &str) -> Result<CorpusLayer, CorpusError> {
946    match value {
947        "baseline" => Ok(CorpusLayer::Baseline),
948        "project_document" => Ok(CorpusLayer::ProjectDocument),
949        "release_source" => Ok(CorpusLayer::ReleaseSource),
950        "live_source" => Ok(CorpusLayer::LiveSource),
951        other => Err(CorpusError::InvalidLayer(other.to_owned())),
952    }
