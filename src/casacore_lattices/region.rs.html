2//! Lattice region types for spatial selection.
3
4use ndarray::ArrayD;
5
6/// A region within a lattice defining a subset of pixels.
7///
8/// Corresponds to the C++ `LCRegion` abstract base class. Regions describe
9/// which pixels of a lattice are "inside" and which are "outside" via a
10/// boolean mask. The region is bounded by a rectangular bounding box.
11///
12/// Concrete implementations include [`LCBox`](crate::LCBox),
13/// [`LCEllipsoid`](crate::LCEllipsoid), and set-algebra combinators.
14pub trait LCRegion: std::fmt::Debug + Send + Sync {
15    /// Returns the shape of the parent lattice.
16    fn lattice_shape(&self) -> Vec<usize>;
17
18    /// Returns the start position of the bounding box.
19    fn bounding_box_start(&self) -> Vec<usize>;
20
21    /// Returns the shape of the bounding box.
22    fn bounding_box_shape(&self) -> Vec<usize>;
23
24    /// Returns the boolean mask within the bounding box.
25    ///
26    /// The returned array has the same shape as [`bounding_box_shape`](Self::bounding_box_shape).
27    /// `true` means the pixel is inside the region.
28    fn get_mask(&self) -> ArrayD<bool>;
29
30    /// Tests whether a single position is inside the region.
31    fn contains(&self, position: &[usize]) -> bool;
32}
