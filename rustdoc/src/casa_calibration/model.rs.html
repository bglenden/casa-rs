9/// Parameter family carried by the calibration table payload.
10///
11/// CASA calibration tables use either complex parameters (`CPARAM`) or float
12/// parameters (`FPARAM`) depending on the table family. The first-wave reader
13/// accepts both on disk but only marks the complex family as supported for the
14/// upcoming `applycal` v1 work.
15#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize, JsonSchema)]
16pub enum CalibrationParameterFamily {
17    /// `ParType=Complex` / `CPARAM`.
18    Complex,
19    /// `ParType=Float` / `FPARAM`.
20    Float,
21    /// Unable to infer a supported family from the current table.
22    Unknown,
23}

25/// Severity assigned to a validation issue discovered while summarizing a
26/// calibration table.
27#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize, JsonSchema)]
28pub enum CalibrationIssueSeverity {
29    /// The issue places the table outside the supported v1 apply surface.
30    Error,
31    /// The issue is tolerated during read and should be normalized later.
32    Warning,
33}

35/// A single validation issue surfaced during summary generation.
36#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize, JsonSchema)]
37pub struct CalibrationValidationIssue {
38    /// Stable issue code for tests and future UI consumers.
39    pub code: String,
40    /// Human-readable severity.
41    pub severity: CalibrationIssueSeverity,
42    /// Human-readable explanation of the issue.
43    pub message: String,
44}

61/// Summary of a standard keyword-linked subtable.
62#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize, JsonSchema)]
63pub struct CalibrationSubtableSummary {
64    /// Keyword name, for example `FIELD`.
65    pub name: String,
66    /// Stored table-reference string from the keyword record.
67    pub stored_reference: Option<String>,
68    /// Resolved absolute path when the link is present.
69    pub resolved_path: Option<PathBuf>,
70    /// Whether the resolved subtable exists on disk.
71    pub exists: bool,
72    /// Row count if the subtable could be opened.
73    pub row_count: Option<usize>,
74    /// Open error if the path existed but the subtable could not be opened.
75    pub open_error: Option<String>,
76}

102/// Machine-readable summary for one calibration table.
103#[derive(Debug, Clone, PartialEq, Serialize, Deserialize, JsonSchema)]
104pub struct CalibrationTableSummary {
105    /// Table root path that was opened.
106    pub path: PathBuf,
107    /// `table.info` logical type.
108    pub table_type: String,
109    /// `table.info` subtype.
110    pub table_subtype: String,
111    /// Total MAIN row count.
112    pub row_count: usize,
113    /// Column names present in the MAIN table.
114    pub columns: Vec<String>,
115    /// Normalized scalar keywords.
116    pub keywords: CalibrationKeywordSummary,
117    /// Keyword-linked subtables of interest.
118    pub subtables: Vec<CalibrationSubtableSummary>,
119    /// Derived parameter family.
120    pub parameter_family: CalibrationParameterFamily,
121    /// Payload-column summary.
122    pub parameter_column: CalibrationColumnSummary,
123    /// Sorted unique field identifiers observed in MAIN rows.
124    pub field_ids: Vec<i32>,
125    /// Sorted unique spectral-window identifiers observed in MAIN rows.
126    pub spectral_window_ids: Vec<i32>,
127    /// Sorted unique `ANTENNA1` identifiers observed in MAIN rows.
128    pub antenna1_ids: Vec<i32>,
129    /// Sorted unique `ANTENNA2` identifiers observed in MAIN rows.
130    pub antenna2_ids: Vec<i32>,
131    /// Sorted unique observation identifiers observed in MAIN rows.
132    pub observation_ids: Vec<i32>,
133    /// Time-domain coverage if `TIME` cells were readable.
134    pub time_coverage: Option<TimeCoverageSummary>,
135    /// Validation issues discovered while summarizing the table.
136    pub issues: Vec<CalibrationValidationIssue>,
137}

139impl CalibrationTableSummary {
140    /// Returns `true` when the table lies inside the currently implemented
141    /// apply surface:
142    ///
143    /// - complex `CPARAM` antenna-based tables, or
144    /// - float `FPARAM` support for `K Jones` and the VLA prior families
145    ///   `KAntPos Jones`, `EGainCurve`, and `TOpac`
146    /// - legacy `BPOLY` bandpass tables
147    ///
148    /// and carries no error-level validation issues.
149    pub fn supported_for_v1_apply(&self) -> bool {
150        self.table_type == crate::constants::TABLE_INFO_TYPE
151            && (matches!(
152                (&self.parameter_family, self.table_subtype.as_str()),
153                (CalibrationParameterFamily::Complex, _)
154                    | (CalibrationParameterFamily::Float, "K Jones")
155                    | (CalibrationParameterFamily::Float, "KAntPos Jones")
156                    | (CalibrationParameterFamily::Float, "EGainCurve")
157                    | (CalibrationParameterFamily::Float, "TOpac")
158            ) || self.table_subtype == "BPOLY")
159            && self
160                .issues
161                .iter()
162                .all(|issue| issue.severity != CalibrationIssueSeverity::Error)
163    }
164}
