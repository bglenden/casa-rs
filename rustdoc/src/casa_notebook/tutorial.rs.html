387    /// Load and validate an immutable portable template folder.
388    pub fn load_template(path: impl AsRef<Path>) -> Result<TutorialTemplate, TutorialError> {
389        let root = fs::canonicalize(path.as_ref()).map_err(|source| TutorialError::Io {
390            action: "resolve tutorial template",
391            path: path.as_ref().to_owned(),
392            source,
393        })?;
394        if !root.is_dir() {
395            return Err(TutorialError::TemplateNotDirectory { path: root });
396        }
397        let manifest_path = root.join(TUTORIAL_MANIFEST);
398        let markdown_path = root.join(TUTORIAL_MARKDOWN);
399        let manifest_source =
400            fs::read_to_string(&manifest_path).map_err(|source| TutorialError::Io {
401                action: "read tutorial manifest",
402                path: manifest_path.clone(),
403                source,
404            })?;
405        let manifest: TutorialManifest =
406            toml::from_str(&manifest_source).map_err(|source| TutorialError::Manifest {
407                path: manifest_path.clone(),
408                detail: source.to_string(),
409            })?;
410        validate_manifest(&root, &manifest)?;
411        let markdown = fs::read_to_string(&markdown_path).map_err(|source| TutorialError::Io {
412            action: "read tutorial Markdown",
413            path: markdown_path,
414            source,
415        })?;
416        validate_template_cells(&manifest, &markdown)?;
417        let content_sha256 = tutorial_template_hash(&manifest_source, &markdown, &root)?;
418        Ok(TutorialTemplate {
419            root,
420            manifest,
421            markdown,
422            content_sha256,
423        })
424    }

426    /// Convert one legacy `tutorial-pack.v0` folder into a new immutable v1 template.
427    ///
428    /// This is a one-shot conversion: the destination must not already exist and
429    /// no runtime compatibility reader is retained after conversion.
430    pub fn migrate_v0_template(
431        pack_path: impl AsRef<Path>,
432        destination: impl AsRef<Path>,
433    ) -> Result<TutorialTemplate, TutorialError> {
434        let pack_path = pack_path.as_ref();
435        let manifest_path = if pack_path.is_dir() {
436            pack_path.join("pack.json")
437        } else {
438            pack_path.to_owned()
439        };
440        let pack_root =
441            manifest_path
442                .parent()
443                .ok_or_else(|| TutorialError::UnsafeTemplatePath {
444                    path: manifest_path.clone(),
445                })?;
446        let destination = destination.as_ref();
447        if destination.exists() {
448            return Err(TutorialError::MigrationDestinationExists {
449                path: destination.to_owned(),
450            });
451        }
452        let source = fs::read_to_string(&manifest_path).map_err(|source| TutorialError::Io {
453            action: "read tutorial-pack v0 manifest",
454            path: manifest_path.clone(),
455            source,
456        })?;
457        let pack: TutorialPackV0 =
458            serde_json::from_str(&source).map_err(|source| TutorialError::Manifest {
459                path: manifest_path.clone(),
460                detail: source.to_string(),
461            })?;
462        if pack.schema_version != "tutorial-pack.v0" {
463            return Err(TutorialError::UnsupportedV0Schema {
464                version: pack.schema_version,
465            });
466        }
467        let parent = destination
468            .parent()
469            .ok_or_else(|| TutorialError::UnsafeTemplatePath {
470                path: destination.to_owned(),
471            })?;
472        fs::create_dir_all(parent).map_err(|source| TutorialError::Io {
473            action: "create migrated tutorial parent",
474            path: parent.to_owned(),
475            source,
476        })?;
477        let temporary = tempfile::Builder::new()
478            .prefix(".tutorial-v1-")
479            .tempdir_in(parent)
480            .map_err(|source| TutorialError::Io {
481                action: "create migrated tutorial staging directory",
482                path: parent.to_owned(),
483                source,
484            })?;
485        let staging = temporary.path();
486
487        let docs_path = resolve_pack_relative(pack_root, &pack.learner.docs_index)?;
488        let mut markdown = fs::read_to_string(&docs_path).map_err(|source| TutorialError::Io {
489            action: "read tutorial-pack learner prose",
490            path: docs_path,
491            source,
492        })?;
493        let mut sections = Vec::with_capacity(pack.sections.len());
494        for section in &pack.sections {
495            let mut cell_ids = Vec::new();
496            for step in section
497                .steps
498                .iter()
499                .filter(|step| step.surface == "gui" && step.provider_kind == "native-rust")
500            {
501                let cell_id = CellId::new();
502                let intent = TaskCellIntent {
503                    format: 1,
504                    surface: step.task_id.clone(),
505                    kind: "task".to_owned(),
506                    contract: 1,
507                    parameters: step
508                        .parameters
509                        .iter()
510                        .map(|(name, value)| {
511                            json_to_toml(value)
512                                .map(|value| (name.clone(), value))
513                                .ok_or_else(|| TutorialError::V0Parameter {
514                                    section_id: section.id.clone(),
515                                    parameter: name.clone(),
516                                })
517                        })
518                        .collect::<Result<_, _>>()?,
519                };
520                markdown.push_str(&format!(
521                    "\n\n<!-- casa-rs-cell:v1 id={cell_id} kind=task -->\n{}<!-- /casa-rs-cell -->\n",
522                    intent.to_markdown().map_err(TutorialError::Notebook)?
523                ));
524                cell_ids.push(cell_id);
525            }
526            sections.push(TutorialSection {
527                id: section.id.clone(),
528                title: section.title.clone(),
529                dataset_ids: section.input_refs.clone(),
530                cell_ids,
531            });
532        }
533        fs::write(staging.join(TUTORIAL_MARKDOWN), markdown).map_err(|source| {
534            TutorialError::Io {
535                action: "write migrated tutorial Markdown",
536                path: staging.join(TUTORIAL_MARKDOWN),
537                source,
538            }
539        })?;
540
541        let legacy_assets = pack_root.join("assets");
542        if legacy_assets.is_dir() {
543            copy_tree(&legacy_assets, &staging.join("assets"))?;
544        }
545        let regression_candidate = pack_root.join(&pack.regression.evidence_path);
546        let regression = if regression_candidate.exists() {
547            let regression_source =
548                resolve_pack_relative(pack_root, &pack.regression.evidence_path)?;
549            let regression_path = PathBuf::from("regression");
550            if regression_source.is_dir() {
551                copy_tree(&regression_source, &staging.join(&regression_path))?;
552            } else {
553                fs::create_dir_all(staging.join(&regression_path)).map_err(|source| {
554                    TutorialError::Io {
555                        action: "create migrated regression directory",
556                        path: staging.join(&regression_path),
557                        source,
558                    }
559                })?;
560                fs::copy(
561                    &regression_source,
562                    staging.join(&regression_path).join("evidence"),
563                )
564                .map_err(|source| TutorialError::Io {
565                    action: "copy migrated regression evidence",
566                    path: regression_source,
567                    source,
568                })?;
569            }
570            Some(TutorialRegressionOverlay {
571                path: regression_path,
572            })
573        } else {
574            None
575        };
576        let manifest = TutorialManifest {
577            schema_version: TUTORIAL_MANIFEST_SCHEMA_VERSION,
578            tutorial_id: pack.tutorial_id,
579            title: pack.title,
580            datasets: pack
581                .inputs
582                .into_iter()
583                .map(|input| {
584                    let archive_format = if input.filename.ends_with(".tar.gz")
585                        || input.filename.ends_with(".tgz")
586                    {
587                        Some(TutorialArchiveFormat::TarGz)
588                    } else if input.filename.ends_with(".tar") {
589                        Some(TutorialArchiveFormat::Tar)
590                    } else {
591                        None
592                    };
593                    let unpack = archive_format.map(|format| TutorialUnpackPlan {
594                        format,
595                        archive_root: input.pack_path.file_name().map(PathBuf::from),
596                        max_entries: 100_000,
597                        max_expanded_bytes: input
598                            .size_bytes
599                            .unwrap_or(1024 * 1024 * 1024)
600                            .saturating_mul(10),
601                    });
602                    TutorialDataset {
603                        id: input.id,
604                        display_name: input.display_name,
605                        uri: input.source_artifact_url,
606                        destination: input.pack_path,
607                        expected_size_bytes: input.size_bytes,
608                        sha256: checksum_policy_sha256(&input.checksum_policy),
609                        unpack,
610                        checks: Vec::new(),
611                    }
612                })
613                .collect(),
614            sections,
615            regression,
616        };
617        fs::write(
618            staging.join(TUTORIAL_MANIFEST),
619            toml::to_string_pretty(&manifest)
620                .map_err(|source| TutorialError::Serialize(source.to_string()))?,
621        )
622        .map_err(|source| TutorialError::Io {
623            action: "write migrated tutorial manifest",
624            path: staging.join(TUTORIAL_MANIFEST),
625            source,
626        })?;
627        let staging_path = temporary.keep();
628        fs::rename(&staging_path, destination).map_err(|source| TutorialError::Io {
629            action: "publish migrated tutorial template",
630            path: destination.to_owned(),
631            source,
632        })?;
633        Self::load_template(destination)
634    }

636    /// Fork one template into an editable notebook and managed tutorial lock.
637    pub fn fork_template(
638        &self,
639        template: &TutorialTemplate,
640        filename: &str,
641    ) -> Result<TutorialForkResult, TutorialError> {
642        let store = NotebookStore::open(&self.project_root).map_err(TutorialError::Store)?;
643        let created = store
644            .create_named(filename, &template.manifest.title)
645            .map_err(TutorialError::Store)?;
646        let notebook_id = created.entry.id;
647        let result = self.finish_fork(&store, template, created);
648        if result.is_err() {
649            let _ = fs::remove_file(self.project_root.join("notebooks").join(filename));
650            let _ = fs::remove_dir_all(
651                self.project_root
652                    .join("notebooks/assets")
653                    .join(notebook_id.to_string()),
654            );
655            let _ = fs::remove_dir_all(self.tutorial_directory(notebook_id));
656        }
657        result
658    }
659
660    fn finish_fork(
661        &self,
662        store: &NotebookStore,
663        template: &TutorialTemplate,
664        created: NotebookSnapshot,
665    ) -> Result<TutorialForkResult, TutorialError> {
666        let notebook_id = created.entry.id;
667        let assets_destination = self
668            .project_root
669            .join("notebooks/assets")
670            .join(notebook_id.to_string());
671        let lock_directory = self.tutorial_directory(notebook_id);
672        let mut source = strip_notebook_marker(&template.markdown);
673        source = rewrite_template_asset_links(&source, notebook_id);
674        let source = format!(
675            "<!-- casa-rs-notebook:v1 id={notebook_id} -->\n\n{}",
676            source.trim_start()
677        );
678        NotebookDocument::parse(source.clone()).map_err(TutorialError::Notebook)?;
679
680        if template.root.join("assets").exists() {
681            copy_tree(&template.root.join("assets"), &assets_destination)?;
682        }

686            source,
687        })?;
688        atomic_write(
689            &lock_directory.join(TUTORIAL_MANIFEST),

702            }
703        };
704        let lock = TutorialLock {
705            schema_version: TUTORIAL_LOCK_SCHEMA_VERSION,

713            sections: template.manifest.sections.clone(),
714            datasets: template
715                .manifest
716                .datasets
717                .iter()
718                .cloned()
719                .map(|dataset| TutorialDatasetLock {
720                    dataset,
721                    phase: TutorialAcquisitionPhase::Missing,
722                    staged: false,
723                    current_generation: 0,
724                    pinned_sha256: None,

890    fn begin_attempt(
891        &self,
892        plan: &TutorialAcquisitionPlan,
893        approval: TutorialAcquisitionApproval,
894        kind: TutorialAttemptKind,
895        resume: bool,
896        validate_plan_hash: bool,
897    ) -> Result<TutorialDatasetLock, TutorialError> {
898        let _guard = self.lock_project()?;
899        let mut lock = self.load_lock(plan.notebook_id)?;
900        let dataset = dataset_mut(&mut lock, &plan.dataset_id)?;
901        validate_plan_dataset(plan, dataset)?;
902        if plan.registry_version != self.registry.version()
903            || (validate_plan_hash && plan.approval_sha256 != acquisition_plan_hash(plan)?)
904            || approval.approval_sha256 != plan.approval_sha256
905        {
906            return Err(TutorialError::ApprovalMismatch);
907        }
908        if dataset.phase == TutorialAcquisitionPhase::Ready
909            || (kind == TutorialAttemptKind::Initial && dataset.phase.is_running())
910        {
911            return Err(TutorialError::InvalidTransition {
912                dataset_id: plan.dataset_id.clone(),
913                phase: dataset.phase,
914                action: "begin",
915            });
916        }
917        if plan.missing_digest && !approval.allow_missing_digest {
918            return Err(TutorialError::MissingDigestApprovalRequired);
919        }
920        let available_disk_bytes =
921            fs2::available_space(&self.project_root).map_err(|source| TutorialError::Io {
922                action: "recheck tutorial project free space",
923                path: self.project_root.clone(),
924                source,
925            })?;
926        if !plan.has_enough_disk() || available_disk_bytes < plan.required_disk_bytes {
927            return Err(TutorialError::InsufficientDisk {
928                required: plan.required_disk_bytes,
929                available: available_disk_bytes.min(plan.available_disk_bytes),
930            });
931        }
932        let known_checks: BTreeSet<&str> = dataset
933            .dataset
934            .checks
935            .iter()
936            .map(|check| check.id.as_str())
937            .collect();
938        if let Some(unknown) = approval
939            .skipped_check_ids
940            .iter()
941            .find(|id| !known_checks.contains(id.as_str()))
942        {
943            return Err(TutorialError::UnknownCheck {
944                check_id: unknown.clone(),
945            });
946        }
947        let generation = dataset.current_generation.saturating_add(1);
948        let attempt_directory = self.attempt_directory(plan.notebook_id, &plan.dataset_id);
949        fs::create_dir_all(&attempt_directory).map_err(|source| TutorialError::Io {
950            action: "create tutorial attempt directory",
951            path: attempt_directory.clone(),
952            source,
953        })?;
954        let part_path = attempt_directory.join(ATTEMPT_ARCHIVE);
955        if !resume {
956            remove_path_if_exists(&part_path)?;
957            remove_path_if_exists(&attempt_directory.join(MATERIALIZED))?;
958        }
959        let downloaded_bytes = fs::metadata(&part_path)
960            .map(|value| value.len())
961            .unwrap_or(0);
962        if dataset.phase.is_running()
963            && let Ok(previous) = current_attempt_mut(dataset)
964        {
965            previous.phase = TutorialAcquisitionPhase::Cancelled;
966            previous.error = Some("Superseded by a newer acquisition generation".to_owned());
967            previous.finished_at = Some(Timestamp::now());
968            self.finalize_current_receipt(
969                dataset,
970                ExecutionStatus::Cancelled,
971                vec!["Superseded by a newer acquisition generation".to_owned()],
972            )?;
973        }
974        let receipt_handle = NotebookStore::open(&self.project_root)
975            .map_err(TutorialError::Store)?
976            .begin_attempt(RecordingRequest {
977                initiating_surface: "tutorial".to_owned(),
978                operation_id: format!("tutorial.acquire.{}", plan.dataset_id),
979                notebook_id: Some(plan.notebook_id),
980                cell_id: None,
981                task_intent: None,
982                execution_input: None,
983                provider_contract_version: plan.registry_version,
984                resolved_parameters: BTreeMap::from([
985                    (
986                        "requested_uri".to_owned(),
987                        plan.requested_uri.clone().into(),
988                    ),
989                    ("resolved_uri".to_owned(), plan.resolved_uri.clone().into()),
990                    (
991                        "destination".to_owned(),
992                        plan.destination.to_string_lossy().into_owned().into(),
993                    ),
994                    (
995                        "attempt_kind".to_owned(),
996                        format!("{kind:?}").to_lowercase().into(),
997                    ),
998                ]),
999                run_safety: RunSafetyRecord {
1000                    classification: "project_local_dataset_acquisition".to_owned(),
1001                    affected_paths: vec![plan.destination.clone()],
1002                },
1003                approvals: vec![ApprovalRecord {
1004                    kind: "tutorial_dataset_acquisition".to_owned(),
1005                    actor: "user".to_owned(),
1006                    timestamp: Timestamp::now(),
1007                    content_hash: Some(plan.approval_sha256.clone()),
1008                }],
1009            })
1010            .map_err(TutorialError::Store)?;
1011        let attempt = TutorialDatasetAttempt {
1012            generation,
1013            kind,
1014            phase: TutorialAcquisitionPhase::Downloading,
1015            requested_uri: plan.requested_uri.clone(),
1016            resolved_uri: plan.resolved_uri.clone(),
1017            redirects: plan.redirects.clone(),
1018            expected_size_bytes: plan.expected_size_bytes.or(plan.resolved_size_bytes),
1019            expected_sha256: plan.expected_sha256.clone(),
1020            approval_sha256: plan.approval_sha256.clone(),
1021            approved_missing_digest: approval.allow_missing_digest,
1022            skipped_check_ids: approval.skipped_check_ids,
1023            downloaded_bytes,
1024            computed_sha256: None,
1025            checks: Vec::new(),
1026            error: None,
1027            started_at: Timestamp::now(),
1028            finished_at: None,
1029            receipt_handle: Some(receipt_handle),
1030        };
1031        dataset.current_generation = generation;
1032        dataset.phase = TutorialAcquisitionPhase::Downloading;
1033        dataset.staged = false;
1034        dataset.attempts.push(attempt);
1035        let result = dataset.clone();
1036        self.write_lock(&lock)?;
1037        Ok(result)
1038    }

1040    fn resume_or_restart(
1041        &self,
1042        notebook_id: NotebookId,
1043        dataset_id: &str,
1044        kind: TutorialAttemptKind,
1045        resume: bool,
1046    ) -> Result<TutorialDatasetLock, TutorialError> {
1047        let lock = self.load_lock(notebook_id)?;
1048        let dataset = lock
1049            .datasets
1050            .iter()
1051            .find(|entry| entry.dataset.id == dataset_id)
1052            .ok_or_else(|| TutorialError::DatasetNotFound {
1053                dataset_id: dataset_id.to_owned(),
1054            })?;
1055        if kind == TutorialAttemptKind::Resume
1056            && !matches!(
1057                dataset.phase,
1058                TutorialAcquisitionPhase::Cancelled | TutorialAcquisitionPhase::NetworkFailed
1059            )
1060        {
1061            return Err(TutorialError::InvalidTransition {
1062                dataset_id: dataset_id.to_owned(),
1063                phase: dataset.phase,
1064                action: "resume",
1065            });
1066        }
1067        if dataset.phase == TutorialAcquisitionPhase::Ready {
1068            return Err(TutorialError::InvalidTransition {
1069                dataset_id: dataset_id.to_owned(),
1070                phase: dataset.phase,
1071                action: "restart",
1072            });
1073        }
1074        let previous = dataset
1075            .attempts
1076            .last()
1077            .ok_or_else(|| TutorialError::NoAttempt {
1078                dataset_id: dataset_id.to_owned(),
1079            })?;
1080        let expected_sha256 = dataset
1081            .pinned_sha256
1082            .clone()
1083            .or_else(|| previous.expected_sha256.clone());
1084        let plan = TutorialAcquisitionPlan {
1085            approval_sha256: previous.approval_sha256.clone(),
1086            registry_version: lock.registry_version,
1087            notebook_id,
1088            dataset_id: dataset_id.to_owned(),
1089            scheme: uri_scheme(&previous.requested_uri)?,
1090            requested_uri: previous.requested_uri.clone(),
1091            resolved_uri: previous.resolved_uri.clone(),
1092            redirects: previous.redirects.clone(),
1093            expected_size_bytes: previous.expected_size_bytes,
1094            resolved_size_bytes: previous.expected_size_bytes,
1095            destination: dataset.dataset.destination.clone(),
1096            expected_sha256: expected_sha256.clone(),
1097            required_disk_bytes: 0,
1098            available_disk_bytes: u64::MAX,
1099            unpack: dataset.dataset.unpack.clone(),
1100            checks: dataset.dataset.checks.clone(),
1101            missing_digest: expected_sha256.is_none(),
1102        };
1103        let approval = TutorialAcquisitionApproval {
1104            approval_sha256: previous.approval_sha256.clone(),
1105            allow_missing_digest: previous.approved_missing_digest,
1106            skipped_check_ids: previous.skipped_check_ids.clone(),
1107        };
1108        self.begin_attempt(&plan, approval, kind, resume, false)
1109    }

1207    fn advance_dataset(
1208        &self,
1209        notebook_id: NotebookId,
1210        dataset: &mut TutorialDatasetLock,
1211        max_download_bytes: u64,
1212    ) -> Result<(), TutorialError> {
1213        let attempt_directory = self.attempt_directory(notebook_id, &dataset.dataset.id);
1214        let part_path = attempt_directory.join(ATTEMPT_ARCHIVE);
1215        let phase = dataset.phase;
1216        match phase {
1217            TutorialAcquisitionPhase::Downloading => {
1218                let limit = max_download_bytes.clamp(1, DOWNLOAD_CHUNK_LIMIT);
1219                let attempt = current_attempt_mut(dataset)?;
1220                let handler = self.registry.handler(&attempt.requested_uri)?;
1221                let chunk =
1222                    handler.read_chunk(&attempt.resolved_uri, attempt.downloaded_bytes, limit)?;
1223                if chunk.bytes.is_empty() && !chunk.complete {
1224                    return Err(TutorialError::UnexpectedEndOfSource {
1225                        uri: attempt.resolved_uri.clone(),
1226                    });
1227                }
1228                let mut file = OpenOptions::new()
1229                    .create(true)
1230                    .append(true)
1231                    .open(&part_path)
1232                    .map_err(|source| TutorialError::Io {
1233                        action: "append tutorial download",
1234                        path: part_path.clone(),
1235                        source,
1236                    })?;
1237                file.write_all(&chunk.bytes)
1238                    .and_then(|()| file.sync_all())
1239                    .map_err(|source| TutorialError::Io {
1240                        action: "persist tutorial download chunk",
1241                        path: part_path.clone(),
1242                        source,
1243                    })?;
1244                attempt.downloaded_bytes = attempt
1245                    .downloaded_bytes
1246                    .saturating_add(u64::try_from(chunk.bytes.len()).unwrap_or(u64::MAX));
1247                if let Some(expected) = attempt.expected_size_bytes {
1248                    if attempt.downloaded_bytes > expected {
1249                        return Err(TutorialError::SizeMismatch {
1250                            expected,
1251                            actual: attempt.downloaded_bytes,
1252                        });
1253                    }
1254                    if attempt.downloaded_bytes == expected {
1255                        attempt.phase = TutorialAcquisitionPhase::Verifying;
1256                        dataset.phase = TutorialAcquisitionPhase::Verifying;
1257                    } else if chunk.complete {
1258                        return Err(TutorialError::SizeMismatch {
1259                            expected,
1260                            actual: attempt.downloaded_bytes,
1261                        });
1262                    }
1263                } else if chunk.complete {
1264                    attempt.phase = TutorialAcquisitionPhase::Verifying;
1265                    dataset.phase = TutorialAcquisitionPhase::Verifying;
1266                }
1267            }
1268            TutorialAcquisitionPhase::Verifying => {
1269                let size = fs::metadata(&part_path)
1270                    .map_err(|source| TutorialError::Io {
1271                        action: "inspect downloaded tutorial source",
1272                        path: part_path.clone(),
1273                        source,
1274                    })?
1275                    .len();
1276                let has_unpack = dataset.dataset.unpack.is_some();
1277                let expected_size = current_attempt_mut(dataset)?.expected_size_bytes;
1278                if let Some(expected) = expected_size
1279                    && size != expected
1280                {
1281                    return Err(TutorialError::SizeMismatch {
1282                        expected,
1283                        actual: size,
1284                    });
1285                }
1286                let computed = file_sha256(&part_path)?;
1287                let expected_sha256 = current_attempt_mut(dataset)?.expected_sha256.clone();
1288                if let Some(expected) = &expected_sha256
1289                    && !expected.eq_ignore_ascii_case(&computed)
1290                {
1291                    return Err(TutorialError::ChecksumMismatch {
1292                        expected: expected.clone(),
1293                        actual: computed,
1294                    });
1295                }
1296                let approved_missing_digest = current_attempt_mut(dataset)?.approved_missing_digest;
1297                if expected_sha256.is_none() && !approved_missing_digest {
1298                    return Err(TutorialError::MissingDigestApprovalRequired);
1299                }
1300                if expected_sha256.is_none() {
1301                    dataset.pinned_sha256 = Some(computed.clone());
1302                }
1303                let next = if has_unpack {
1304                    TutorialAcquisitionPhase::Unpacking
1305                } else {
1306                    TutorialAcquisitionPhase::Checking
1307                };
1308                let attempt = current_attempt_mut(dataset)?;
1309                attempt.computed_sha256 = Some(computed);
1310                attempt.phase = next;
1311                dataset.phase = next;
1312            }
1313            TutorialAcquisitionPhase::Unpacking => {
1314                let materialized = attempt_directory.join(MATERIALIZED);
1315                remove_path_if_exists(&materialized)?;
1316                let plan = dataset
1317                    .dataset
1318                    .unpack
1319                    .as_ref()
1320                    .ok_or(TutorialError::MissingUnpackPlan)?;
1321                safely_extract_archive(&part_path, &materialized, plan)?;
1322                current_attempt_mut(dataset)?.phase = TutorialAcquisitionPhase::Checking;
1323                dataset.phase = TutorialAcquisitionPhase::Checking;
1324            }
1325            TutorialAcquisitionPhase::Checking => {
1326                if dataset.dataset.unpack.is_none() {
1327                    let materialized = attempt_directory.join(MATERIALIZED);
1328                    remove_path_if_exists(&materialized)?;
1329                    fs::copy(&part_path, &materialized).map_err(|source| TutorialError::Io {
1330                        action: "stage verified tutorial file",
1331                        path: materialized,
1332                        source,
1333                    })?;
1334                }
1335                let check_root = materialized_source(&attempt_directory, &dataset.dataset)?;
1336                let skipped: BTreeSet<String> = current_attempt_mut(dataset)?
1337                    .skipped_check_ids
1338                    .iter()
1339                    .cloned()
1340                    .collect();
1341                let outcomes = dataset
1342                    .dataset
1343                    .checks
1344                    .iter()
1345                    .map(|check| run_optional_check(check, &check_root, &skipped))
1346                    .collect();
1347                let attempt = current_attempt_mut(dataset)?;
1348                attempt.checks = outcomes;
1349                attempt.phase = TutorialAcquisitionPhase::Materializing;
1350                dataset.phase = TutorialAcquisitionPhase::Materializing;
1351            }
1352            TutorialAcquisitionPhase::Materializing => {
1353                let source = materialized_source(&attempt_directory, &dataset.dataset)?;
1354                let destination =
1355                    prepare_project_destination(&self.project_root, &dataset.dataset.destination)?;
1356                fs::rename(&source, &destination).map_err(|source| TutorialError::Io {
1357                    action: "publish verified tutorial dataset",
1358                    path: destination,
1359                    source,
1360                })?;
1361                let attempt = current_attempt_mut(dataset)?;
1362                attempt.phase = TutorialAcquisitionPhase::Ready;
1363                attempt.finished_at = Some(Timestamp::now());
1364                dataset.phase = TutorialAcquisitionPhase::Ready;
1365                dataset.staged = true;
1366            }
1367            _ => {
1368                return Err(TutorialError::InvalidTransition {
1369                    dataset_id: dataset.dataset.id.clone(),
1370                    phase,
1371                    action: "advance",
1372                });
1373            }
1374        }
1375        Ok(())
1376    }

1435fn validate_manifest(root: &Path, manifest: &TutorialManifest) -> Result<(), TutorialError> {
1436    if manifest.schema_version != TUTORIAL_MANIFEST_SCHEMA_VERSION {
1437        return Err(TutorialError::UnsupportedManifestVersion {
1438            version: manifest.schema_version,
1439        });
1440    }
1441    validate_identifier("tutorial", &manifest.tutorial_id)?;
1442    let mut dataset_ids = BTreeSet::new();
1443    for dataset in &manifest.datasets {
1444        validate_identifier("dataset", &dataset.id)?;
1445        if !dataset_ids.insert(dataset.id.as_str()) {
1446            return Err(TutorialError::DuplicateIdentifier {
1447                kind: "dataset",
1448                id: dataset.id.clone(),
1449            });
1450        }
1451        validate_project_path(&dataset.destination)?;
1452        let scheme = uri_scheme(&dataset.uri)?;
1453        if scheme == "file" {
1454            let _ = file_uri_path(&dataset.uri)?;
1455        }
1456        if let Some(digest) = &dataset.sha256
1457            && (digest.len() != 64 || !digest.bytes().all(|value| value.is_ascii_hexdigit()))
1458        {
1459            return Err(TutorialError::InvalidDigest {
1460                dataset_id: dataset.id.clone(),
1461            });
1462        }
1463        if let Some(unpack) = &dataset.unpack {
1464            if unpack.max_entries == 0 || unpack.max_expanded_bytes == 0 {
1465                return Err(TutorialError::InvalidArchiveBounds {
1466                    dataset_id: dataset.id.clone(),
1467                });
1468            }
1469            if let Some(archive_root) = &unpack.archive_root {
1470                validate_relative_path(archive_root, false)?;
1471            }
1472        }
1473        let mut check_ids = BTreeSet::new();
1474        for check in &dataset.checks {
1475            validate_identifier("check", &check.id)?;
1476            if !check_ids.insert(check.id.as_str()) {
1477                return Err(TutorialError::DuplicateIdentifier {
1478                    kind: "check",
1479                    id: check.id.clone(),
1480                });
1481            }
1482            validate_relative_path(&check.path, true)?;
1483        }
1484    }
1485    let mut section_ids = BTreeSet::new();
1486    for section in &manifest.sections {
1487        validate_identifier("section", &section.id)?;
1488        if !section_ids.insert(section.id.as_str()) {
1489            return Err(TutorialError::DuplicateIdentifier {
1490                kind: "section",
1491                id: section.id.clone(),
1492            });
1493        }
1494        if let Some(missing) = section
1495            .dataset_ids
1496            .iter()
1497            .find(|id| !dataset_ids.contains(id.as_str()))
1498        {
1499            return Err(TutorialError::UnknownSectionDataset {
1500                section_id: section.id.clone(),
1501                dataset_id: missing.clone(),
1502            });
1503        }
1504    }
1505    if let Some(regression) = &manifest.regression {
1506        validate_relative_path(&regression.path, false)?;
1507        let path = root.join(&regression.path);
1508        let resolved = fs::canonicalize(&path).map_err(|source| TutorialError::Io {
1509            action: "resolve tutorial regression overlay",
1510            path,
1511            source,
1512        })?;
1513        if !resolved.starts_with(root) {
1514            return Err(TutorialError::UnsafeTemplatePath { path: resolved });
1515        }
1516    }
1517    Ok(())
1518}

1580fn tutorial_template_hash(
1581    manifest_source: &str,
1582    markdown: &str,
1583    root: &Path,
1584) -> Result<String, TutorialError> {
1585    let mut digest = Sha256::new();
1586    hash_field(&mut digest, manifest_source.as_bytes());
1587    hash_field(&mut digest, markdown.as_bytes());
1588    let assets = root.join("assets");
1589    if assets.exists() {
1590        hash_tree(&assets, &assets, &mut digest)?;
1591    }
1592    let manifest: TutorialManifest =
1593        toml::from_str(manifest_source).map_err(|source| TutorialError::Manifest {
1594            path: root.join(TUTORIAL_MANIFEST),
1595            detail: source.to_string(),
1596        })?;
1597    if let Some(regression) = manifest.regression {
1598        hash_field(&mut digest, regression.path.as_os_str().as_encoded_bytes());
1599        let evidence = root.join(&regression.path);
1600        if evidence.is_dir() {
1601            hash_tree(&evidence, &evidence, &mut digest)?;
1602        } else {
1603            hash_field(
1604                &mut digest,
1605                &fs::read(&evidence).map_err(|source| TutorialError::Io {
1606                    action: "read tutorial regression evidence",
1607                    path: evidence,
1608                    source,
1609                })?,
1610            );
1611        }
1612    }
1613    Ok(format!("{:x}", digest.finalize()))
1614}

1616fn hash_tree(root: &Path, path: &Path, digest: &mut Sha256) -> Result<(), TutorialError> {
1617    let mut entries = fs::read_dir(path)
1618        .map_err(|source| TutorialError::Io {
1619            action: "read tutorial asset directory",
1620            path: path.to_owned(),
1621            source,
1622        })?
1623        .collect::<Result<Vec<_>, _>>()
1624        .map_err(|source| TutorialError::Io {
1625            action: "read tutorial asset entry",
1626            path: path.to_owned(),
1627            source,
1628        })?;
1629    entries.sort_by_key(std::fs::DirEntry::file_name);
1630    for entry in entries {
1631        let entry_path = entry.path();
1632        let metadata = fs::symlink_metadata(&entry_path).map_err(|source| TutorialError::Io {
1633            action: "inspect tutorial asset",
1634            path: entry_path.clone(),
1635            source,
1636        })?;
1637        if metadata.file_type().is_symlink() {
1638            return Err(TutorialError::UnsafeTemplatePath { path: entry_path });
1639        }
1640        let relative = entry_path
1641            .strip_prefix(root)
1642            .expect("tutorial asset is below root");
1643        hash_field(digest, relative.as_os_str().as_encoded_bytes());
1644        if metadata.is_dir() {
1645            hash_tree(root, &entry_path, digest)?;
1646        } else if metadata.is_file() {
1647            hash_field(
1648                digest,
1649                &fs::read(&entry_path).map_err(|source| TutorialError::Io {
1650                    action: "read tutorial asset",
1651                    path: entry_path,
1652                    source,
1653                })?,
1654            );
1655        } else {
1656            return Err(TutorialError::UnsafeTemplatePath { path: entry_path });
1657        }
1658    }
1659    Ok(())
1660}

1662fn hash_field(digest: &mut Sha256, value: &[u8]) {
1663    digest.update(u64::try_from(value.len()).unwrap_or(u64::MAX).to_le_bytes());
1664    digest.update(value);
1665}

1694fn json_to_toml(value: &serde_json::Value) -> Option<toml::Value> {
1695    match value {
1696        serde_json::Value::Bool(value) => Some(toml::Value::Boolean(*value)),
1697        serde_json::Value::Number(value) => value
1698            .as_i64()
1699            .map(toml::Value::Integer)
1700            .or_else(|| value.as_f64().map(toml::Value::Float)),
1701        serde_json::Value::String(value) => Some(toml::Value::String(value.clone())),
1702        serde_json::Value::Array(values) => values
1703            .iter()
1704            .map(json_to_toml)
1705            .collect::<Option<Vec<_>>>()
1706            .map(toml::Value::Array),
1707        serde_json::Value::Object(values) => values
1708            .iter()
1709            .map(|(key, value)| json_to_toml(value).map(|value| (key.clone(), value)))
1710            .collect::<Option<toml::map::Map<_, _>>>()
1711            .map(toml::Value::Table),
1712        serde_json::Value::Null => None,
1713    }
1714}

1833fn prepare_project_destination(
1834    project_root: &Path,
1835    relative_destination: &Path,
1836) -> Result<PathBuf, TutorialError> {
1837    validate_project_path(relative_destination)?;
1838    let canonical_root = fs::canonicalize(project_root).map_err(|source| TutorialError::Io {
1839        action: "resolve tutorial project root",
1840        path: project_root.to_owned(),
1841        source,
1842    })?;
1843    let relative_parent =
1844        relative_destination
1845            .parent()
1846            .ok_or_else(|| TutorialError::UnsafeProjectPath {
1847                path: relative_destination.to_owned(),
1848            })?;
1849    let mut parent = project_root.to_owned();
1850    for component in relative_parent.components() {
1851        let Component::Normal(component) = component else {
1852            return Err(TutorialError::UnsafeProjectPath {
1853                path: relative_destination.to_owned(),
1854            });
1855        };
1856        parent.push(component);
1857        match fs::symlink_metadata(&parent) {
1858            Ok(metadata) => {
1859                if metadata.file_type().is_symlink() || !metadata.is_dir() {
1860                    return Err(TutorialError::UnsafeProjectPath {
1861                        path: relative_destination.to_owned(),
1862                    });
1863                }
1864            }
1865            Err(source) if source.kind() == std::io::ErrorKind::NotFound => {
1866                fs::create_dir(&parent).map_err(|source| TutorialError::Io {
1867                    action: "create tutorial dataset destination parent",
1868                    path: parent.clone(),
1869                    source,
1870                })?;
1871            }
1872            Err(source) => {
1873                return Err(TutorialError::Io {
1874                    action: "inspect tutorial dataset destination parent",
1875                    path: parent,
1876                    source,
1877                });
1878            }
1879        }
1880        let resolved_parent = fs::canonicalize(&parent).map_err(|source| TutorialError::Io {
1881            action: "resolve tutorial dataset destination parent",
1882            path: parent.clone(),
1883            source,
1884        })?;
1885        if !resolved_parent.starts_with(&canonical_root) {
1886            return Err(TutorialError::UnsafeProjectPath {
1887                path: relative_destination.to_owned(),
1888            });
1889        }
1890    }
1891
1892    let resolved_parent = fs::canonicalize(&parent).map_err(|source| TutorialError::Io {
1893        action: "resolve tutorial dataset destination parent",
1894        path: parent.clone(),
1895        source,
1896    })?;
1897    if !resolved_parent.starts_with(&canonical_root) {
1898        return Err(TutorialError::UnsafeProjectPath {
1899            path: relative_destination.to_owned(),
1900        });
1901    }
1902
1903    let destination = project_root.join(relative_destination);
1904    match fs::symlink_metadata(&destination) {

1969fn safely_extract_archive(
1970    archive_path: &Path,
1971    destination: &Path,
1972    plan: &TutorialUnpackPlan,
1973) -> Result<(), TutorialError> {
1974    fs::create_dir_all(destination).map_err(|source| TutorialError::Io {
1975        action: "create tutorial extraction staging",
1976        path: destination.to_owned(),
1977        source,
1978    })?;
1979    let file = File::open(archive_path).map_err(|source| TutorialError::Io {
1980        action: "open tutorial archive",
1981        path: archive_path.to_owned(),
1982        source,
1983    })?;
1984    let reader: Box<dyn Read> = match plan.format {
1985        TutorialArchiveFormat::Tar => Box::new(file),
1986        TutorialArchiveFormat::TarGz => Box::new(GzDecoder::new(file)),
1987    };
1988    let mut archive = Archive::new(reader);
1989    let mut entries = archive
1990        .entries()
1991        .map_err(|source| TutorialError::UnsafeArchive {
1992            detail: source.to_string(),
1993        })?;
1994    let mut count = 0_u64;
1995    let mut expanded = 0_u64;
1996    let mut paths = BTreeSet::new();
1997    for entry in &mut entries {
1998        let mut entry = entry.map_err(|source| TutorialError::UnsafeArchive {
1999            detail: source.to_string(),
2000        })?;
2001        count = count.saturating_add(1);
2002        if count > plan.max_entries {
2003            return Err(TutorialError::ArchiveLimit {
2004                detail: format!("entry count exceeds {}", plan.max_entries),
2005            });
2006        }
2007        let path = entry
2008            .path()
2009            .map_err(|source| TutorialError::UnsafeArchive {
2010                detail: source.to_string(),
2011            })?
2012            .into_owned();
2013        validate_relative_path(&path, false).map_err(|_| TutorialError::UnsafeArchive {
2014            detail: format!("unsafe archive path {}", path.display()),
2015        })?;
2016        if !paths.insert(path.to_string_lossy().to_lowercase()) {
2017            return Err(TutorialError::ArchiveCollision { path });
2018        }
2019        let entry_type = entry.header().entry_type();
2020        if !(entry_type.is_file() || entry_type.is_dir()) {
2021            return Err(TutorialError::UnsafeArchive {
2022                detail: format!("unsupported archive entry type for {}", path.display()),
2023            });
2024        }
2025        expanded = expanded.saturating_add(entry.header().size().unwrap_or(u64::MAX));
2026        if expanded > plan.max_expanded_bytes {
2027            return Err(TutorialError::ArchiveLimit {
2028                detail: format!("expanded bytes exceed {}", plan.max_expanded_bytes),
2029            });
2030        }
2031        let unpacked =
2032            entry
2033                .unpack_in(destination)
2034                .map_err(|source| TutorialError::UnsafeArchive {
2035                    detail: source.to_string(),
2036                })?;
2037        if !unpacked {
2038            return Err(TutorialError::UnsafeArchive {
2039                detail: format!("entry escaped extraction root: {}", path.display()),
2040            });
2041        }
2042    }
2043    if let Some(root) = &plan.archive_root
2044        && !destination.join(root).exists()
2045    {
2046        return Err(TutorialError::MaterializedRootMissing {
2047            path: destination.join(root),
2048        });
2049    }
2050    Ok(())
2051}

2053fn copy_tree(source: &Path, destination: &Path) -> Result<(), TutorialError> {
2054    let metadata = fs::symlink_metadata(source).map_err(|source_error| TutorialError::Io {
2055        action: "inspect tutorial asset source",
2056        path: source.to_owned(),
2057        source: source_error,
2058    })?;
2059    if metadata.file_type().is_symlink() {
2060        return Err(TutorialError::UnsafeTemplatePath {
2061            path: source.to_owned(),
2062        });
2063    }
2064    if metadata.is_file() {
2065        if let Some(parent) = destination.parent() {
2066            fs::create_dir_all(parent).map_err(|source| TutorialError::Io {
2067                action: "create tutorial asset parent",
2068                path: parent.to_owned(),
2069                source,
2070            })?;
2071        }
2072        fs::copy(source, destination).map_err(|source_error| TutorialError::Io {
2073            action: "copy tutorial asset",
2074            path: destination.to_owned(),
2075            source: source_error,
2076        })?;
2077        return Ok(());
2078    }
2079    if !metadata.is_dir() {
2080        return Err(TutorialError::UnsafeTemplatePath {
2081            path: source.to_owned(),
2082        });
2083    }
2084    fs::create_dir_all(destination).map_err(|source| TutorialError::Io {
2085        action: "create tutorial asset directory",
2086        path: destination.to_owned(),
2087        source,
2088    })?;
2089    for entry in fs::read_dir(source).map_err(|source_error| TutorialError::Io {
2090        action: "read tutorial asset directory",
2091        path: source.to_owned(),
2092        source: source_error,
2093    })? {
2094        let entry = entry.map_err(|source_error| TutorialError::Io {
2095            action: "read tutorial asset entry",
2096            path: source.to_owned(),
2097            source: source_error,
2098        })?;
2099        copy_tree(&entry.path(), &destination.join(entry.file_name()))?;
2100    }
2101    Ok(())
2102}

2127fn file_sha256(path: &Path) -> Result<String, TutorialError> {
2128    let mut file = File::open(path).map_err(|source| TutorialError::Io {
2129        action: "open tutorial source for SHA-256",

2131        source,
2132    })?;

2134    let mut buffer = [0_u8; 64 * 1024];
2135    loop {
2136        let read = file.read(&mut buffer).map_err(|source| TutorialError::Io {
2137            action: "read tutorial source for SHA-256",
2138            path: path.to_owned(),
2139            source,
2140        })?;
2141        if read == 0 {
2142            break;
2143        }
2144        digest.update(&buffer[..read]);
2145    }
2146    Ok(format!("{:x}", digest.finalize()))
