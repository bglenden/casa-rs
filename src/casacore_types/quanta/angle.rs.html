2//! Angle formatting helpers for radian values.
3//!
4//! This module provides [`MvAngle`], a lightweight Rust counterpart to C++
5//! `casacore::MVAngle`. Like `MVAngle`, it stores an angle as radians and
6//! supports normalization plus sexagesimal formatting. The current Rust
7//! implementation covers the subset needed by CASA-style summary/reporting
8//! code: fixed-width angle formatting, two-digit declination-style formatting,
9//! and time-style formatting for right ascension.
10
11/// An angle value stored in radians.
12///
13/// This is the Rust counterpart to C++ `casacore::MVAngle`. It is intended for
14/// angle normalization and human-readable formatting. For directional
15/// reference-frame conversion use the measure-layer types such as
16/// [`crate::measures::direction::MDirection`].
17#[derive(Debug, Clone, Copy, PartialEq)]
18pub struct MvAngle {
19    radians: f64,
20}
21
22impl MvAngle {
23    /// Creates an `MvAngle` from radians.
24    pub const fn from_radians(radians: f64) -> Self {
25        Self { radians }
26    }
27
28    /// Returns the stored angle in radians.
29    pub const fn as_radians(self) -> f64 {
30        self.radians
31    }
32
33    /// Returns the stored angle in degrees.
34    pub fn as_degrees(self) -> f64 {
35        self.radians.to_degrees()
36    }
37
38    /// Normalizes the angle into the interval `[lower_turns, lower_turns + 1)`.
39    ///
40    /// This mirrors the common C++ `MVAngle::operator(Double)` usage where the
41    /// bound is expressed in turns rather than radians. For example,
42    /// `normalized(0.0)` yields the `0..360 deg` range typically used for right
43    /// ascension display.
44    pub fn normalized(self, lower_turns: f64) -> Self {
45        let lower = lower_turns * std::f64::consts::TAU;
46        let radians = (self.radians - lower).rem_euclid(std::f64::consts::TAU) + lower;
47        Self { radians }
48    }
49
50    /// Formats the angle as `+ddd.mm.ss[.f...]`.
51    pub fn format_angle(self, second_decimals: usize) -> String {
52        format_sexagesimal(
53            self.as_degrees(),
54            3,
55            '.',
56            '.',
57            second_decimals,
58            SignStyle::Always,
59        )
60    }
61
62    /// Formats the angle as `+dd.mm.ss[.f...]`.
63    ///
64    /// This matches the common `MVAngle::DIG2` usage for declination- and
65    /// latitude-style values.
66    pub fn format_angle_dig2(self, second_decimals: usize) -> String {
67        format_sexagesimal(
68            self.as_degrees(),
69            2,
70            '.',
71            '.',
72            second_decimals,
73            SignStyle::Always,
74        )
75    }
76
77    /// Formats the angle as `hh:mm:ss[.f...]`.
78    ///
79    /// Positive values omit the sign, matching the common `MVAngle::TIME`
80    /// display used for right ascension. Negative values are prefixed with `-`.
81    pub fn format_time(self, second_decimals: usize) -> String {
82        format_sexagesimal(
83            self.radians * (24.0 / std::f64::consts::TAU),
84            2,
85            ':',
86            '.',
87            second_decimals,
88            SignStyle::NegativeOnly,
89        )
90    }
91}
92
93#[derive(Debug, Clone, Copy, PartialEq, Eq)]
94enum SignStyle {
95    Always,
96    NegativeOnly,
97}
98
99fn format_sexagesimal(
100    primary_units: f64,
101    primary_width: usize,
102    field_sep: char,
103    fraction_sep: char,
104    second_decimals: usize,
105    sign_style: SignStyle,
106) -> String {
107    let sign_char = if primary_units.is_sign_negative() {
108        '-'
109    } else {
110        '+'
111    };
112    let abs_primary = primary_units.abs();
113    let scale = 10_i64.pow(second_decimals as u32);
114    let total_scaled_seconds = (abs_primary * 3_600.0 * scale as f64).round() as i64;
115    let seconds_per_primary = 3_600_i64 * scale;
116    let primary = total_scaled_seconds / seconds_per_primary;
117    let minutes = (total_scaled_seconds / (60 * scale)) % 60;
118    let seconds = (total_scaled_seconds / scale) % 60;
119    let fraction = total_scaled_seconds % scale;
120
121    let sign = match sign_style {
122        SignStyle::Always => sign_char.to_string(),
123        SignStyle::NegativeOnly if sign_char == '-' => "-".to_string(),
124        SignStyle::NegativeOnly => String::new(),
125    };
126
127    if second_decimals == 0 {
128        return format!(
129            "{sign}{primary:0primary_width$}{field_sep}{minutes:02}{field_sep}{seconds:02}"
130        );
131    }
132
133    format!(
134        "{sign}{primary:0primary_width$}{field_sep}{minutes:02}{field_sep}{seconds:02}{fraction_sep}{fraction:0second_decimals$}"
135    )
136}

138#[cfg(test)]
139mod tests {
140    use super::MvAngle;
141
142    #[test]
143    fn right_ascension_time_wraps_to_zero_to_twenty_four_hours() {
144        let ra = MvAngle::from_radians(-0.25 * std::f64::consts::TAU).normalized(0.0);
145        assert_eq!(ra.format_time(6), "18:00:00.000000");
146    }
147
148    #[test]
149    fn declination_uses_two_digit_signed_format() {
150        let dec = MvAngle::from_radians((-12.5_f64).to_radians());
151        assert_eq!(dec.format_angle_dig2(5), "-12.30.00.00000");
152    }
153
154    #[test]
155    fn longitude_uses_three_digit_signed_format() {
156        let lon = MvAngle::from_radians(123.5_f64.to_radians());
157        assert_eq!(lon.format_angle(1), "+123.30.00.0");
158    }
159}
