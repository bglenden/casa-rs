5    /// Returns the attached schema, if any.
6    ///
7    /// When a schema is present, all row and cell mutations are validated
8    /// against it. Returns `None` for schema-free tables.
9    pub fn schema(&self) -> Option<&TableSchema> {
10        self.inner.schema()
11    }

13    /// Attaches or replaces the schema, validating all existing rows.
14    ///
15    /// If the new schema is incompatible with any existing row the schema is
16    /// not updated and the original is restored. Returns the first
17    /// [`TableError`] encountered during validation.
18    pub fn set_schema(&mut self, schema: TableSchema) -> Result<(), TableError> {
19        let auto_unlock = self.begin_write_operation("set_schema")?;
20        let result = (|| {
21            let previous = self.inner.schema().cloned();
22            self.inner.set_schema(Some(schema));
23            if let Err(err) = self.validate() {
24                self.inner.set_schema(previous);
25                return Err(err);
26            }
27            Ok(())
28        })();
29        self.finish_write_operation(auto_unlock, result)
30    }

39    /// Validates all rows against the attached schema.
40    ///
41    /// Returns `Ok(())` immediately when no schema is attached. Otherwise,
42    /// checks every cell in every row and returns the first [`TableError`]
43    /// encountered. This is called automatically by [`save`][Table::save],
44    /// [`open`][Table::open], and schema-setting methods.
45    pub fn validate(&self) -> Result<(), TableError> {
46        let Some(schema) = self.schema() else {
47            return Ok(());
48        };
49
50        for (row_index, row) in self.rows()?.iter().enumerate() {
51            validate_row_against_schema(row_index, row, schema)?;
52        }
53        Ok(())
54    }

71    /// Appends a row to the table.
72    ///
73    /// If a schema is attached, the row is validated before insertion.
74    /// Returns [`TableError`] if the row violates the schema; the table is
75    /// left unchanged in that case.
76    pub fn add_row(&mut self, row: RecordValue) -> Result<(), TableError> {
77        let auto_unlock = self.begin_write_operation("add_row")?;
78        let result = (|| {
79            if let Some(schema) = self.schema() {
80                validate_row_against_schema(self.row_count(), &row, schema)?;
81            }
82            self.inner.add_row(row)?;
83            Ok(())
84        })();
85        self.finish_write_operation(auto_unlock, result)
86    }

151    /// Returns the record value at `(row_index, column)`.
152    ///
153    /// When a schema is attached and the column is declared as a record column,
154    /// a missing cell is treated as an empty [`RecordValue`]. Without a schema,
155    /// an absent cell returns [`TableError::ColumnNotFound`].
156    ///
157    /// Returns [`TableError::SchemaColumnNotRecord`] if the schema declares the
158    /// column with a non-record type.
159    pub fn record_cell(&self, row_index: usize, column: &str) -> Result<RecordValue, TableError> {
160        self.require_row(row_index)?;
161        if let Some(schema) = self.schema() {
162            let column_schema =
163                schema
164                    .column(column)
165                    .ok_or_else(|| TableError::SchemaColumnUnknown {
166                        column: column.to_string(),
167                    })?;
168            if !matches!(column_schema.column_type(), ColumnType::Record) {
169                return Err(TableError::SchemaColumnNotRecord {
170                    column: column.to_string(),
171                });
172            }
173            return match self.cell(row_index, column)? {
174                Some(Value::Record(record)) => Ok(record.clone()),
175                Some(value) => Err(TableError::ColumnTypeMismatch {
176                    row_index,
177                    column: column.to_string(),
178                    expected: "record",
179                    found: value.kind(),
180                }),
181                None => Ok(RecordValue::default()),
182            };
183        }
184
185        match self.cell(row_index, column)? {
186            Some(Value::Record(record)) => Ok(record.clone()),
187            Some(value) => Err(TableError::ColumnTypeMismatch {
188                row_index,
189                column: column.to_string(),
190                expected: "record",
191                found: value.kind(),
192            }),
193            None => Err(TableError::ColumnNotFound {
194                row_index,
195                column: column.to_string(),
196            }),
197        }
198    }

223    /// Returns an iterator over the record cells in `column` within `row_range`.
224    ///
225    /// Each item is a [`RecordColumnCell`] whose `value` is always populated:
226    /// absent cells are defaulted to an empty [`RecordValue`] when the schema
227    /// permits it. Returns [`TableError`] if the column is unknown, not a
228    /// record column, or a cell has the wrong type.
229    pub fn get_record_column_range<'a>(
230        &'a self,
231        column: &'a str,
232        row_range: RowRange,
233    ) -> Result<RecordColumnIter<'a>, TableError> {
234        self.require_column(column)?;
235        row_range.validate(self.row_count())?;
236
237        let default_missing = if let Some(schema) = self.schema() {
238            let column_schema =
239                schema
240                    .column(column)
241                    .ok_or_else(|| TableError::SchemaColumnUnknown {
242                        column: column.to_string(),
243                    })?;
244            if !matches!(column_schema.column_type(), ColumnType::Record) {
245                return Err(TableError::SchemaColumnNotRecord {
246                    column: column.to_string(),
247                });
248            }
249            true
250        } else {
251            false
252        };
253
254        let row_data = self.rows()?;
255        for row_index in row_range.iter() {
256            match row_data[row_index].get(column) {
257                Some(Value::Record(_)) => {}
258                Some(value) => {
259                    return Err(TableError::ColumnTypeMismatch {
260                        row_index,
261                        column: column.to_string(),
262                        expected: "record",
263                        found: value.kind(),
264                    });
265                }
266                None => {
267                    if !default_missing {
268                        return Err(TableError::ColumnNotFound {
269                            row_index,
270                            column: column.to_string(),
271                        });
272                    }
273                }
274            }
275        }
276
277        Ok(RecordColumnIter {
278            row_data,
279            column,
280            rows: row_range.iter(),
281            default_missing,
282        })
283    }

285    /// Writes values from an iterator into `column` for all rows.
286    ///
287    /// Shorthand for `put_column_range(column, RowRange::new(0, row_count()), values)`.
288    /// Returns the number of cells written, or [`TableError`] if the value
289    /// count does not match the row count.

291    where
292        I: IntoIterator<Item = Value>,
293    {

295    }

297    /// Writes values from an iterator into `column` for the rows in `row_range`.
298    ///
299    /// The iterator must produce exactly as many values as there are rows in
300    /// `row_range`; otherwise [`TableError::ColumnWriteTooFewValues`] or
301    /// [`TableError::ColumnWriteTooManyValues`] is returned. Returns the
302    /// number of cells written on success.
303    pub fn put_column_range<I>(
304        &mut self,
305        column: &str,
306        row_range: RowRange,
307        values: I,
308    ) -> Result<usize, TableError>
309    where
310        I: IntoIterator<Item = Value>,
311    {
312        let auto_unlock = self.begin_write_operation("put_column_range")?;
313        let result = (|| {
314            self.require_column(column)?;
315            row_range.validate(self.row_count())?;
316
317            let expected = row_range.len();
318            let row_iter = row_range.iter();
319            let mut value_iter = values.into_iter();
320            let mut provided = 0usize;
321            for row_index in row_iter {
322                let Some(value) = value_iter.next() else {
323                    return Err(TableError::ColumnWriteTooFewValues { expected, provided });
324                };
325                self.set_cell_impl(row_index, column, value)?;
326                provided += 1;
327            }
328            if value_iter.next().is_some() {
329                return Err(TableError::ColumnWriteTooManyValues { expected });
330            }
331            Ok(provided)
332        })();
333        self.finish_write_operation(auto_unlock, result)
334    }

336    /// Returns `true` if the cell at `(row_index, column)` is considered defined.
337    ///
338    /// A cell is defined if a value is present in the row. For record columns
339    /// with a schema, a missing cell is still considered defined because it
340    /// defaults to an empty record. Returns [`TableError`] if `row_index` is
341    /// out of bounds or the column is unknown per the schema.
342    pub fn is_cell_defined(&self, row_index: usize, column: &str) -> Result<bool, TableError> {
343        self.require_row(row_index)?;
344        self.require_column(column)?;
345        if let Some(undefined) = self
346            .inner
347            .undefined_cells()?
348            .get(row_index)
349            .map(|set| set.contains(column))
350        {
351            if undefined {
352                return Ok(false);
353            }
354        }
355        if self.cell(row_index, column)?.is_some() {
356            return Ok(true);
357        }
358        if let Some(schema) = self.schema()
359            && matches!(
360                schema.column(column).map(ColumnSchema::column_type),
361                Some(ColumnType::Record)
362            )
363        {
364            return Ok(true);
365        }
366        Ok(false)
367    }

392    /// Writes `value` to the cell at `(row_index, column)`.
393    ///
394    /// When a schema is attached the value is validated against the column
395    /// schema before writing; new column names are allowed only if they are
396    /// declared in the schema. Without a schema the column must already exist
397    /// in the row (use [`add_row`][Table::add_row] to populate new rows first).
398    ///
399    /// Returns [`TableError`] if the row is out of bounds, the column is
400    /// unknown per the schema, or the value violates the column type/shape.
401    pub fn set_cell(
402        &mut self,
403        row_index: usize,
404        column: &str,
405        value: Value,
406    ) -> Result<(), TableError> {
407        let auto_unlock = self.begin_write_operation("set_cell")?;
408        let result = self.set_cell_impl(row_index, column, value);
409        self.finish_write_operation(auto_unlock, result)
410    }
411
412    fn set_cell_impl(
413        &mut self,
414        row_index: usize,
415        column: &str,
416        value: Value,
417    ) -> Result<(), TableError> {
418        let schema_column = if let Some(schema) = self.schema() {
419            Some(
420                schema
421                    .column(column)
422                    .ok_or_else(|| TableError::SchemaColumnUnknown {
423                        column: column.to_string(),
424                    })?
425                    .clone(),
426            )
427        } else {
428            None
429        };
430        if let Some(column_schema) = &schema_column {
431            validate_cell_against_schema_column(row_index, column_schema, Some(&value))?;
432        }
433
434        {
435            if let Some(set) = self.inner.undefined_for_row_mut(row_index)? {
436                set.remove(column);
437            }
438        }
439
440        let row = self.row_mut(row_index)?;
441
442        if schema_column.is_some() {
443            row.upsert(column.to_string(), value);
444            return Ok(());
445        }
446
447        let target = row
448            .get_mut(column)
449            .ok_or_else(|| TableError::ColumnNotFound {
450                row_index,
451                column: column.to_string(),
452            })?;
453        *target = value;
454        Ok(())
455    }

594
595pub(super) fn validate_row_against_schema(
596    row_index: usize,
597    row: &RecordValue,
598    schema: &TableSchema,
599) -> Result<(), TableError> {
600    for column in schema.columns() {
601        validate_cell_against_schema_column(row_index, column, row.get(column.name()))?;
602    }
603    for field in row.fields() {
604        if !schema.contains_column(&field.name) {
605            return Err(TableError::RowContainsUnknownColumn {
606                row_index,
607                column: field.name.clone(),
608            });
609        }
610    }
611    Ok(())
612}
613
614pub(super) fn validate_cell_against_schema_column(
615    row_index: usize,
616    column: &ColumnSchema,
617    value: Option<&Value>,
618) -> Result<(), TableError> {
619    match (column.column_type(), value) {
620        (ColumnType::Scalar, Some(Value::Scalar(_))) => Ok(()),
621        (ColumnType::Scalar, Some(value)) => Err(TableError::ColumnTypeMismatch {
622            row_index,
623            column: column.name().to_string(),
624            expected: "scalar",
625            found: value.kind(),
626        }),
627        (ColumnType::Scalar, None) => {
628            if column.options().undefined {
629                Ok(())
630            } else {
631                Err(TableError::SchemaColumnMissing {
632                    row_index,
633                    column: column.name().to_string(),
634                })
635            }
636        }
637        (ColumnType::Record, Some(Value::Record(_))) => Ok(()),
638        (ColumnType::Record, Some(value)) => Err(TableError::ColumnTypeMismatch {
639            row_index,
640            column: column.name().to_string(),
641            expected: "record",
642            found: value.kind(),
643        }),
644        (ColumnType::Record, None) => Ok(()),
645        (ColumnType::Array(contract), Some(Value::Array(array))) => {
646            validate_array_contract(row_index, column.name(), contract, array)
647        }
648        (ColumnType::Array(_), Some(value)) => Err(TableError::ColumnTypeMismatch {
649            row_index,
650            column: column.name().to_string(),
651            expected: "array",
652            found: value.kind(),
653        }),
654        (ColumnType::Array(ArrayShapeContract::Fixed { .. }), None) => {
655            Err(TableError::SchemaColumnMissing {
656                row_index,
657                column: column.name().to_string(),
658            })
659        }
660        (ColumnType::Array(ArrayShapeContract::Variable { .. }), None) => Ok(()),
661    }
662}
663
664fn validate_array_contract(
665    row_index: usize,
666    column_name: &str,
667    contract: &ArrayShapeContract,
668    array: &ArrayValue,
669) -> Result<(), TableError> {
670    match contract {
671        ArrayShapeContract::Fixed { shape } => {
672            let found = array.shape().to_vec();
673            if found == *shape {
674                Ok(())
675            } else {
676                Err(TableError::ArrayShapeMismatch {
677                    row_index,
678                    column: column_name.to_string(),
679                    expected: shape.clone(),
680                    found,
681                })
682            }
683        }
684        ArrayShapeContract::Variable {
685            ndim: Some(expected),
686        } => {
687            let found = array.ndim();
688            if found == *expected {
689                Ok(())
690            } else {
691                Err(TableError::ArrayNdimMismatch {
692                    row_index,
693                    column: column_name.to_string(),
694                    expected: *expected,
695                    found,
696                })
697            }
698        }
699        ArrayShapeContract::Variable { ndim: None } => Ok(()),
700    }
701}
