40        SubtableId::DataDescription => data_description::REQUIRED_COLUMNS,
41        SubtableId::Feed => feed::REQUIRED_COLUMNS,
42        SubtableId::Field => field::REQUIRED_COLUMNS,
43        SubtableId::FlagCmd => flag_cmd::REQUIRED_COLUMNS,
44        SubtableId::History => history::REQUIRED_COLUMNS,
45        SubtableId::Observation => observation::REQUIRED_COLUMNS,
46        SubtableId::Pointing => pointing::REQUIRED_COLUMNS,
47        SubtableId::Polarization => polarization::REQUIRED_COLUMNS,
48        SubtableId::Processor => processor::REQUIRED_COLUMNS,
49        SubtableId::SpectralWindow => spectral_window::REQUIRED_COLUMNS,
50        SubtableId::State => state::REQUIRED_COLUMNS,
51        SubtableId::Doppler => doppler::REQUIRED_COLUMNS,
52        SubtableId::FreqOffset => freq_offset::REQUIRED_COLUMNS,
53        SubtableId::Source => source::REQUIRED_COLUMNS,
54        SubtableId::SysCal => syscal::REQUIRED_COLUMNS,
55        SubtableId::Weather => weather::REQUIRED_COLUMNS,

57}

82#[cfg(test)]
83mod tests {
84    use super::*;
85    use crate::column_def::build_table_schema;
86    use std::collections::HashSet;
87
88    #[test]
89    fn all_subtable_schemas_build() {
90        let all_ids: Vec<SubtableId> = SubtableId::ALL_REQUIRED
91            .iter()
92            .chain(SubtableId::ALL_OPTIONAL.iter())
93            .copied()
94            .collect();
95
96        for id in &all_ids {
97            build_table_schema(required_columns(*id))
98                .unwrap_or_else(|e| panic!("{}: {e}", id.name()));
99        }
100    }
101
102    #[test]
103    fn no_duplicate_column_names_within_subtables() {
104        let all_ids: Vec<SubtableId> = SubtableId::ALL_REQUIRED
105            .iter()
106            .chain(SubtableId::ALL_OPTIONAL.iter())
107            .copied()
108            .collect();
109
110        for id in &all_ids {
111            let mut names = HashSet::new();
112            for col in required_columns(*id)
113                .iter()
114                .chain(optional_columns(*id).iter())
115            {
116                assert!(
117                    names.insert(col.name),
118                    "{}: duplicate column {}",
119                    id.name(),
120                    col.name
121                );
122            }
123        }
124    }
