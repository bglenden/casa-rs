9pub(super) fn draw_browser_mode_picker(
10    frame: &mut Frame<'_>,
11    app: &AppState,
12    layout: &UiLayout,
13    palette: Theme,
14) {
15    let area = browser_mode_picker_area(layout.browser_mode_selector, layout.form_block);
16    let list_area = browser_mode_picker_list_area(area);
17    frame.render_widget(Clear, area);
18    let block = Block::default()
19        .title("Choose Left Pane View")
20        .title_style(
21            Style::default()
22                .fg(palette.header_fg)
23                .add_modifier(Modifier::BOLD),
24        )
25        .borders(Borders::ALL)
26        .border_set(palette.border_set)
27        .border_style(Style::default().fg(palette.active_pane_border_fg));
28    let inner = block.inner(area);
29    frame.render_widget(block, area);
30    let selected = app.browser_mode_picker_selection();
31    let items = crate::app::ImageBrowserLeftPaneMode::all()
32        .into_iter()
33        .map(|mode| {
34            let style = if Some(mode) == selected {
35                Style::default()
36                    .fg(palette.field_selected_fg)
37                    .bg(palette.field_selected_bg)
38            } else {
39                Style::default().fg(palette.footer_fg)
40            };
41            ListItem::new(Line::from(format!("  {}", mode.label()))).style(style)
42        })
43        .collect::<Vec<_>>();
44    frame.render_widget(List::new(items), list_area);
45    frame.render_widget(
46        Paragraph::new("Enter/click choose  Esc cancel  Arrows/jk move")
47            .style(Style::default().fg(palette.footer_fg)),
48        Rect {
49            x: inner.x,
50            y: inner.y + inner.height.saturating_sub(1),
51            width: inner.width,
52            height: 1,
53        },
54    );
55}

57pub(super) fn visible_browser_manager_hits(
58    form_inner: Rect,
59    form_row_count: usize,
60    selector: Option<Rect>,
61    app: &AppState,
62) -> Vec<FormRowHit> {
63    let Some(selector) = selector else {
64        return Vec::new();
65    };
66    let rows = app.browser_manager_rows();
67    if rows.is_empty() {
68        return Vec::new();
69    }
70    let form_rows_height = form_row_count.min(form_inner.height as usize) as u16;
71    let gap = u16::from(form_rows_height > 0);
72    let summary_area = Rect {
73        x: form_inner.x,
74        y: form_inner
75            .y
76            .saturating_add(form_rows_height)
77            .saturating_add(gap),
78        width: form_inner.width,
79        height: form_inner
80            .height
81            .saturating_sub(form_rows_height.saturating_add(gap)),
82    };
83    let start_y = selector.y.saturating_add(selector.height).saturating_add(1);
84    let available_height = summary_area
85        .y
86        .saturating_add(summary_area.height)
87        .saturating_sub(start_y);
88    rows.into_iter()
89        .take(available_height as usize)
90        .enumerate()
91        .map(|(offset, row)| FormRowHit {
92            target: FormSelection::BrowserPane(row.target),
93            rect: Rect {
94                x: summary_area.x,
95                y: start_y + offset as u16,
96                width: summary_area.width,
97                height: 1,
98            },
99        })
100        .collect()
101}

115pub(super) fn browser_mode_selector_rect(
116    form_inner: Rect,
117    form_rows_len: usize,
118    app: &AppState,
119) -> Option<Rect> {
120    if !(app.browser_is_active() && app.browser_uses_parameter_pane()) {
121        return None;
122    }
123    let rows_height = form_rows_len.min(form_inner.height as usize) as u16;
124    let gap = u16::from(rows_height > 0);
125    let summary_area = Rect {
126        x: form_inner.x,
127        y: form_inner.y.saturating_add(rows_height).saturating_add(gap),
128        width: form_inner.width,
129        height: form_inner
130            .height
131            .saturating_sub(rows_height.saturating_add(gap)),
132    };
133    if summary_area.height == 0 {
134        return None;
135    }
136    Some(Rect {
137        x: summary_area.x,
138        y: summary_area.y,
139        width: summary_area.width,
140        height: 1,
141    })
142}

144fn selector_popup_area(anchor: Option<Rect>, bounds: Rect, item_count: usize) -> Rect {
145    let width = bounds.width.clamp(1, 24);
146    let content_height = item_count.min(8) as u16;
147    let height = content_height
148        .saturating_add(3)
149        .clamp(5, bounds.height.max(1));
150    let Some(anchor) = anchor else {
151        return centered_rect(width, height, bounds);
152    };
153    let mut x = anchor.x;
154    if x.saturating_add(width) > bounds.x.saturating_add(bounds.width) {
155        x = bounds.x.saturating_add(bounds.width.saturating_sub(width));
156    }
157    let below_y = anchor.y.saturating_add(anchor.height);
158    let above_y = anchor.y.saturating_sub(height);
159    let y = if below_y.saturating_add(height) <= bounds.y.saturating_add(bounds.height) {
160        below_y
161    } else if anchor.y >= bounds.y.saturating_add(height) {
162        above_y
163    } else {
164        bounds.y
165    };
166    Rect {
167        x,
168        y,
169        width,
170        height,
171    }
172}

174fn selector_popup_list_area(area: Rect) -> Rect {
175    Rect {
176        x: area.x.saturating_add(1),
177        y: area.y.saturating_add(1),
178        width: area.width.saturating_sub(2),
179        height: area.height.saturating_sub(3),
180    }
181}

183fn centered_rect(width: u16, height: u16, area: Rect) -> Rect {
184    let popup_width = width.min(area.width);
185    let popup_height = height.min(area.height);
186    Rect {
187        x: area.x + area.width.saturating_sub(popup_width) / 2,
188        y: area.y + area.height.saturating_sub(popup_height) / 2,
189        width: popup_width,
190        height: popup_height,
191    }
192}

194#[cfg(test)]
195mod tests {
196    use super::*;
197
198    #[test]
199    fn selector_popup_area_anchors_below_when_there_is_room() {
200        let bounds = Rect::new(0, 0, 40, 20);
201        let anchor = Rect::new(5, 3, 12, 1);
202        let area = selector_popup_area(Some(anchor), bounds, 3);
203        assert_eq!(area.x, anchor.x);
204        assert_eq!(area.y, anchor.y + anchor.height);
205        assert_eq!(area.width, 24);
206        assert_eq!(area.height, 6);
207    }
208
209    #[test]
210    fn selector_popup_area_clamps_to_bounds_and_prefers_above_when_needed() {
211        let bounds = Rect::new(0, 0, 10, 6);
212        let anchor = Rect::new(8, 5, 2, 1);
213        let area = selector_popup_area(Some(anchor), bounds, 8);
214        assert_eq!(area.width, 10);
215        assert_eq!(area.height, 6);
216        assert_eq!(area.x, 0);
217        assert_eq!(area.y, 0);
218    }
