12pub(crate) fn apply_weighting(
13    request: &ImagingRequest,
14    gridder: &StandardGridder,
15) -> Result<Vec<VisibilityBatch>, crate::ImagingError> {
16    apply_weighting_with_density_source(
17        request.weighting,
18        WeightDensityMode::Combined,
19        None,
20        &request.visibility_batches,
21        &request.visibility_batches,
22        gridder,
23    )
24}

26pub(crate) fn apply_weighting_with_density_source(
27    weighting: WeightingMode,
28    weight_density_mode: WeightDensityMode,
29    uv_taper: Option<GaussianUvTaper>,
30    target_batches: &[VisibilityBatch],
31    density_batches: &[VisibilityBatch],
32    gridder: &StandardGridder,
33) -> Result<Vec<VisibilityBatch>, crate::ImagingError> {
34    let density_convention = density_cell_convention(weighting, weight_density_mode);
35    match weighting {
36        WeightingMode::Natural => Ok(apply_optional_uv_taper(target_batches.to_vec(), uv_taper)),
37        WeightingMode::Uniform => {
38            let density = build_density_grid(density_batches, gridder, true, density_convention);
39            Ok(apply_optional_uv_taper(
40                target_batches
41                    .iter()
42                    .map(|batch| {
43                        reweight_batch(
44                            batch,
45                            gridder,
46                            &density,
47                            density_convention,
48                            |weight, density| weight / density,
49                        )
50                    })
51                    .collect(),
52                uv_taper,
53            ))
54        }
55        WeightingMode::Briggs { robust } => {
56            let density = build_density_grid(density_batches, gridder, true, density_convention);
57            let total_density_weight = density.iter().map(|value| f64::from(*value)).sum::<f64>();
58            let sumlocwt = density
59                .iter()
60                .filter(|value| **value > 0.0)
61                .map(|value| f64::from(*value) * f64::from(*value))
62                .sum::<f64>();
63            let f2 = if sumlocwt > 0.0 && total_density_weight > 0.0 {
64                (5.0f64 * 10f64.powf(-(robust as f64))).powi(2) / (sumlocwt / total_density_weight)
65            } else {
66                0.0
67            } as f32;
68            Ok(apply_optional_uv_taper(
69                target_batches
70                    .iter()
71                    .map(|batch| {
72                        reweight_batch(
73                            batch,
74                            gridder,
75                            &density,
76                            density_convention,
77                            |weight, density| weight / (f2 * density + 1.0),
78                        )
79                    })
80                    .collect(),
81                uv_taper,
82            ))
83        }
84    }
85}

87fn density_cell_convention(
88    weighting: WeightingMode,
89    weight_density_mode: WeightDensityMode,
90) -> DensityCellConvention {
91    match (weighting, weight_density_mode) {
92        (WeightingMode::Uniform | WeightingMode::Briggs { .. }, WeightDensityMode::PerPlane) => {
93            DensityCellConvention::CubeBriggsWeightor
94        }
95        _ => DensityCellConvention::VisImagingWeight,
96    }
97}

99fn apply_optional_uv_taper(
100    mut batches: Vec<VisibilityBatch>,
101    taper: Option<GaussianUvTaper>,
102) -> Vec<VisibilityBatch> {
103    let Some(taper) = taper else {
104        return batches;
105    };
106    let (major_coeff, minor_coeff) = taper_coefficients(taper);
107    let cos_pa = taper.position_angle_rad.sin();
108    let sin_pa = taper.position_angle_rad.cos();
109    for batch in &mut batches {
110        for index in 0..batch.len() {
111            let weight = batch.weight[index];
112            if !(weight.is_finite() && weight > 0.0) {
113                batch.weight[index] = 0.0;
114                continue;
115            }
116            let u = batch.u_lambda[index];
117            let v = batch.v_lambda[index];
118            let ru = cos_pa * u + sin_pa * v;
119            let rv = -sin_pa * u + cos_pa * v;
120            let filter = (-major_coeff * ru * ru - minor_coeff * rv * rv).exp() as f32;
121            batch.weight[index] *= filter;
122        }
123    }
124    batches
125}
126
127fn taper_coefficients(taper: GaussianUvTaper) -> (f64, f64) {
128    let image_factor = std::f64::consts::PI * std::f64::consts::PI / (4.0 * std::f64::consts::LN_2);
129    let major = match taper.major {
130        UvTaperSize::ImageFwhmRad(value) => image_factor * value * value,
131        UvTaperSize::BaselineHwhmLambda(value) => std::f64::consts::LN_2 / (value * value),
132    };
133    let minor = match taper.minor {
134        UvTaperSize::ImageFwhmRad(value) => image_factor * value * value,
135        UvTaperSize::BaselineHwhmLambda(value) => std::f64::consts::LN_2 / (value * value),
136    };
137    (major, minor)
138}

140fn build_density_grid(
141    batches: &[VisibilityBatch],
142    gridder: &StandardGridder,
143    mirror_hermitian: bool,
144    convention: DensityCellConvention,
145) -> Array2<f32> {
146    let [nx, ny] = gridder.density_grid_shape();
147    let mut density_grid = Array2::<f32>::zeros((nx, ny));
148    for batch in batches {
149        for index in 0..batch.len() {
150            let weight = batch.weight[index];
151            if !(weight.is_finite() && weight > 0.0) {
152                continue;
153            }
154            let primary = (batch.u_lambda[index], batch.v_lambda[index]);
155            let conjugate = (-batch.u_lambda[index], -batch.v_lambda[index]);
156            let positions = if mirror_hermitian {
157                [Some(primary), Some(conjugate)]
158            } else {
159                [Some(primary), None]
160            };
161            for (u_lambda, v_lambda) in positions.into_iter().flatten() {
162                let Some((x, y)) =
163                    gridder.density_cell_index_with_convention(u_lambda, v_lambda, convention)
164                else {
165                    continue;
166                };
167                density_grid[(x, y)] += weight;
168            }
169        }
170    }
171    density_grid
172}

174fn reweight_batch(
175    batch: &VisibilityBatch,
176    gridder: &StandardGridder,
177    density: &Array2<f32>,
178    convention: DensityCellConvention,
179    transform: impl Fn(f32, f32) -> f32,
180) -> VisibilityBatch {
181    let mut reweighted = batch.clone();
182    for index in 0..batch.len() {
183        let weight = batch.weight[index];
184        let Some(cell_density) = gridder.density_at_with_convention(
185            density,
186            batch.u_lambda[index],
187            batch.v_lambda[index],
188            convention,
189        ) else {
190            reweighted.weight[index] = 0.0;
191            continue;
192        };
193        if !(weight.is_finite() && weight > 0.0 && cell_density.is_finite() && cell_density > 0.0) {
194            reweighted.weight[index] = 0.0;
195            continue;
196        }
197        reweighted.weight[index] = transform(weight, cell_density);
198    }
199    reweighted
200}

240    #[test]
241    fn uniform_weighting_downweights_dense_uv_regions() {
242        let request = request_for(WeightingMode::Uniform);
243        let gridder = StandardGridder::new(request.geometry).unwrap();
244        let weighted = apply_weighting(&request, &gridder).unwrap();
245        let dense_weight = weighted[0].weight[0];
246        let sparse_weight = weighted[0].weight[4];
247        assert!(dense_weight < sparse_weight);
248    }

250    #[test]
251    fn briggs_extremes_interpolate_between_natural_and_uniform() {
252        let geometry = request_for(WeightingMode::Natural).geometry;
253        let gridder = StandardGridder::new(geometry).unwrap();
254        let natural = apply_weighting(&request_for(WeightingMode::Natural), &gridder).unwrap();
255        let uniform = apply_weighting(&request_for(WeightingMode::Uniform), &gridder).unwrap();
256        let briggs_naturalish = apply_weighting(
257            &request_for(WeightingMode::Briggs { robust: 2.0 }),
258            &gridder,
259        )
260        .unwrap();
261        let briggs_uniformish = apply_weighting(
262            &request_for(WeightingMode::Briggs { robust: -2.0 }),
263            &gridder,
264        )
265        .unwrap();
266
267        let dense_index = 0usize;
268        let sparse_index = 4usize;
269        let natural_ratio = natural[0].weight[dense_index] / natural[0].weight[sparse_index];
270        let uniform_ratio = uniform[0].weight[dense_index] / uniform[0].weight[sparse_index];
271        let briggs_naturalish_ratio =
272            briggs_naturalish[0].weight[dense_index] / briggs_naturalish[0].weight[sparse_index];
273        let briggs_uniformish_ratio =
274            briggs_uniformish[0].weight[dense_index] / briggs_uniformish[0].weight[sparse_index];
275
276        assert!(
277            (briggs_naturalish_ratio - natural_ratio).abs()
278                < (briggs_uniformish_ratio - natural_ratio).abs()
279        );
280        assert!(
281            (briggs_uniformish_ratio - uniform_ratio).abs()
282                < (briggs_naturalish_ratio - uniform_ratio).abs()
283        );
284    }

286    #[test]
287    fn density_grid_accumulates_conjugate_samples_for_uniform_weighting() {
288        let geometry = ImageGeometry {
289            image_shape: [32, 32],
290            cell_size_rad: [1.0e-4, 1.0e-4],
291        };
292        let gridder = StandardGridder::new(geometry).unwrap();
293        let du = 1.0 / (geometry.image_shape[0] as f64 * geometry.cell_size_rad[0]);
294        let batch = VisibilityBatch {
295            u_lambda: vec![1.2 * du],
296            v_lambda: vec![-1.2 * du],
297            w_lambda: vec![0.0],
298            weight: vec![2.0],
299            sumwt_factor: vec![1.0],
300            gridable: vec![true],
301            visibility: vec![Complex32::new(1.0, 0.0)],
302        };
303        let density = build_density_grid(
304            &[batch],
305            &gridder,
306            true,
307            DensityCellConvention::VisImagingWeight,
308        );
309        assert_eq!(density[(17, 17)], 2.0);
310        assert_eq!(density[(14, 14)], 2.0);
311    }
