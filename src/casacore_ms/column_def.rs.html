2//! Compile-time column and keyword metadata for MeasurementSet tables.
3//!
4//! Each MS table (main and subtables) is defined by arrays of [`ColumnDef`]
5//! constants that carry the column name, data type, shape contract, units,
6//! measure metadata, and documentation comment — everything needed to construct
7//! a [`casacore_tables::TableSchema`] and attach MEASINFO / QuantumUnits
8//! keywords for C++ interoperability.
9//!
10//! The column names and types match the C++ casacore `MS*Enums.h` headers
11//! exactly, ensuring binary compatibility with C++-created MeasurementSets.
12
13use casacore_tables::table_measures::MeasureType;
14use casacore_tables::{ColumnSchema, SchemaError, TableSchema};
15use casacore_types::PrimitiveType;
16
17/// Describes the shape contract for a column in the MS schema.
18///
19/// This is the MS-level analog of [`casacore_tables::ColumnType`]. It carries
20/// shape information as `&'static` data so that column definitions can be
21/// `const`.
22///
23/// Cf. C++ `MSTableImpl::addColumnToDesc` which switches on `TpArray*` vs
24/// `Tp*` types and optionally attaches an `IPosition` shape.
25#[derive(Debug, Clone, Copy, PartialEq, Eq)]
26pub enum ColumnKind {
27    /// A single scalar value per row.
28    Scalar,
29    /// A fixed-shape array (all rows share this shape).
30    FixedArray {
31        /// The fixed shape shared by all rows.
32        shape: &'static [usize],
33    },
34    /// A variable-shape array with a known number of dimensions.
35    VariableArray {
36        /// Number of dimensions (rank) of the array.
37        ndim: usize,
38    },
39}
40
41/// Compile-time metadata for a single column in an MS table.
42///
43/// Each MS table module exports `REQUIRED_COLUMNS` and `OPTIONAL_COLUMNS`
44/// arrays of these. The metadata is sufficient to:
45/// 1. Build a [`TableSchema`] via [`build_table_schema`].
46/// 2. Attach MEASINFO keywords via [`casacore_tables::table_measures::TableMeasDesc`].
47/// 3. Attach QuantumUnits keywords via [`casacore_tables::table_quantum::TableQuantumDesc`].
48///
49/// Cf. C++ `MSTableImpl::colMapDef` which populates `SimpleOrderedMap`
50/// entries with the same information.
51#[derive(Debug, Clone, Copy)]
52pub struct ColumnDef {
53    /// Column name, matching C++ `MS*Enums::columnName()` exactly.
54    pub name: &'static str,
55    /// Element data type (e.g. `Float64` for `TpDouble`).
56    pub data_type: PrimitiveType,
57    /// Scalar vs. fixed-shape array vs. variable-shape array.
58    pub column_kind: ColumnKind,
59    /// Physical unit string (e.g. `"s"`, `"m"`, `"Hz"`, `"rad"`), or empty if dimensionless.
60    pub unit: &'static str,
61    /// Measure type for MEASINFO attachment, if any.
62    pub measure_type: Option<MeasureType>,
63    /// Default measure reference frame string (e.g. `"UTC"`, `"J2000"`, `"ITRF"`).
64    pub measure_ref: &'static str,
65    /// Documentation comment, matching the C++ column description.
66    pub comment: &'static str,
67}
68
69/// Compile-time metadata for a table-level keyword.
70///
71/// Used for the MS main table keywords (subtable references, MS_VERSION, etc.)
72/// and any subtable-level keywords.
73#[derive(Debug, Clone, Copy)]
74pub struct KeywordDef {
75    /// Keyword name.
76    pub name: &'static str,
77    /// The expected value type.
78    pub value_type: KeywordValueType,
79    /// Whether this keyword must be present.
80    pub required: bool,
81    /// Documentation comment.
82    pub comment: &'static str,
83}
84
85/// The type of value a keyword holds.
86#[derive(Debug, Clone, Copy, PartialEq, Eq)]
87pub enum KeywordValueType {
88    /// A table reference (path to subtable directory).
89    Table,
90    /// A floating-point version number.
91    Float,
92    /// A string value.
93    String,
94}
95
96/// Convert an array of [`ColumnDef`] constants into a [`TableSchema`].
97///
98/// Each `ColumnDef` is mapped to a [`ColumnSchema`] with the appropriate
99/// scalar/array type. This does not attach MEASINFO or QuantumUnits keywords;
100/// those must be added separately after the table is created.
101///
102/// # Errors
103///
104/// Returns [`SchemaError::DuplicateColumn`] if any two columns share the
105/// same name.
106pub fn build_table_schema(columns: &[ColumnDef]) -> Result<TableSchema, SchemaError> {
107    let col_schemas: Vec<ColumnSchema> = columns
108        .iter()
109        .map(|def| match def.column_kind {
110            ColumnKind::Scalar => ColumnSchema::scalar(def.name, def.data_type),
111            ColumnKind::FixedArray { shape } => {
112                ColumnSchema::array_fixed(def.name, def.data_type, shape.to_vec())
113            }
114            ColumnKind::VariableArray { ndim } => {
115                ColumnSchema::array_variable(def.name, def.data_type, Some(ndim))
116            }
117        })
118        .collect();
119    TableSchema::new(col_schemas)
120}

122#[cfg(test)]
123mod tests {
124    use super::*;
125
126    const TEST_COLS: &[ColumnDef] = &[
127        ColumnDef {
128            name: "COL_A",
129            data_type: PrimitiveType::Int32,
130            column_kind: ColumnKind::Scalar,
131            unit: "",
132            measure_type: None,
133            measure_ref: "",
134            comment: "test scalar",
135        },
136        ColumnDef {
137            name: "COL_B",
138            data_type: PrimitiveType::Float64,
139            column_kind: ColumnKind::FixedArray { shape: &[3] },
140            unit: "m",
141            measure_type: Some(MeasureType::Position),
142            measure_ref: "ITRF",
143            comment: "test fixed array",
144        },
145        ColumnDef {
146            name: "COL_C",
147            data_type: PrimitiveType::Complex32,
148            column_kind: ColumnKind::VariableArray { ndim: 2 },
149            unit: "",
150            measure_type: None,
151            measure_ref: "",
152            comment: "test variable array",
153        },
154    ];
155
156    #[test]
157    fn build_schema_from_column_defs() {
158        let schema = build_table_schema(TEST_COLS).expect("valid schema");
159        assert_eq!(schema.columns().len(), 3);
160        assert!(schema.contains_column("COL_A"));
161        assert!(schema.contains_column("COL_B"));
162        assert!(schema.contains_column("COL_C"));
163    }
164
165    #[test]
166    fn build_schema_rejects_duplicates() {
167        let dups = &[
168            ColumnDef {
169                name: "DUP",
170                data_type: PrimitiveType::Int32,
171                column_kind: ColumnKind::Scalar,
172                unit: "",
173                measure_type: None,
174                measure_ref: "",
175                comment: "",
176            },
177            ColumnDef {
178                name: "DUP",
179                data_type: PrimitiveType::Float64,
180                column_kind: ColumnKind::Scalar,
181                unit: "",
182                measure_type: None,
183                measure_ref: "",
184                comment: "",
185            },
186        ];
187        assert!(build_table_schema(dups).is_err());
188    }
189}
