2//! Error types for the TaQL parser and evaluator.
3//!
4//! All errors produced by lexing, parsing, evaluating, or executing TaQL
5//! queries are represented by the [`TaqlError`] enum. Each variant carries
6//! enough context to produce a useful diagnostic message.
7
8use std::fmt;
9
10/// Position in the TaQL source string (1-based line and column).
11#[derive(Debug, Clone, Copy, PartialEq, Eq)]
12pub struct SourcePos {
13    /// 1-based line number.
14    pub line: usize,
15    /// 1-based column number (byte offset within the line).
16    pub col: usize,
17}
18
19impl fmt::Display for SourcePos {
20    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
21        write!(f, "{}:{}", self.line, self.col)
22    }
23}
24
25/// All errors produced by the TaQL subsystem.
26///
27/// C++ casacore throws `TableError` or `TableGramError` for query failures.
28/// This enum provides structured variants with source-position context.
29#[derive(Debug, thiserror::Error)]
30pub enum TaqlError {
31    /// The lexer encountered an invalid token.
32    #[error("lexer error at {pos}: {message}")]
33    LexError { pos: SourcePos, message: String },
34
35    /// The parser encountered unexpected input.
36    #[error("parse error at {pos}: {message}")]
37    ParseError { pos: SourcePos, message: String },
38
39    /// The parser reached end of input unexpectedly.
40    #[error("unexpected end of query: {message}")]
41    UnexpectedEnd { message: String },
42
43    /// An expression evaluated to an unexpected type.
44    #[error("type error: {message}")]
45    TypeError { message: String },
46
47    /// A column referenced in a query does not exist.
48    #[error("column not found: \"{name}\"")]
49    ColumnNotFound { name: String },
50
51    /// A function referenced in a query does not exist.
52    #[error("unknown function: \"{name}\"")]
53    UnknownFunction { name: String },
54
55    /// A function was called with the wrong number of arguments.
56    #[error("function \"{name}\" expects {expected} argument(s), got {got}")]
57    ArgumentCount {
58        name: String,
59        expected: String,
60        got: usize,
61    },
62
63    /// A feature is parsed but not yet implemented.
64    #[error("unsupported: {message}")]
65    Unsupported { message: String },
66
67    /// An error from the underlying table operations.
68    #[error("table error: {0}")]
69    Table(String),
70
71    /// Division by zero in expression evaluation.
72    #[error("division by zero")]
73    DivisionByZero,
74
75    /// INSERT column count does not match value count.
76    #[error("INSERT column count ({columns}) does not match value count ({values})")]
77    InsertColumnMismatch { columns: usize, values: usize },
78
79    /// A JOIN table could not be opened.
80    #[error("cannot open JOIN table \"{name}\": {message}")]
81    JoinTableOpen { name: String, message: String },
82}
83
84impl TaqlError {
85    /// Convenience constructor for a parse error at a position.
86    pub fn parse(pos: SourcePos, msg: impl Into<String>) -> Self {
87        Self::ParseError {
88            pos,
89            message: msg.into(),
90        }
91    }
92
93    /// Convenience constructor for an unexpected-end error.
94    pub fn unexpected_end(msg: impl Into<String>) -> Self {
95        Self::UnexpectedEnd {
96            message: msg.into(),
97        }
98    }
99}
