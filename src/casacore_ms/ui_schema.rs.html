8/// Machine-readable UI schema emitted by `--ui-schema`.
9#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
10pub struct UiCommandSchema {
11    /// Schema version for the `--ui-schema` payload itself.
12    pub schema_version: u32,
13    /// Stable command identifier shared across aliases.
14    pub command_id: String,
15    /// Invocation name used for this schema instance, such as `msexplore`.
16    pub invocation_name: String,
17    /// Human-friendly command name for launcher UIs.
18    pub display_name: String,
19    /// Category label for grouping related apps.
20    pub category: String,
21    /// One-line command summary.
22    pub summary: String,
23    /// Usage line rendered in `--help`.
24    pub usage: String,
25    /// Ordered argument definitions for parsing, help text, and TUI forms.
26    pub arguments: Vec<UiArgumentSchema>,
27    /// Structured-output contract used by rich UI renderers.
28    pub managed_output: Option<UiManagedOutputSchema>,
29}
30
31impl UiCommandSchema {
32    /// Render the command's human-readable help text from the schema.
33    pub fn render_help(&self) -> String {
34        let mut out = String::new();
35        let positionals = self
36            .arguments
37            .iter()
38            .filter(|argument| matches!(argument.parser, UiArgumentParser::Positional { .. }))
39            .collect::<Vec<_>>();
40        let options = self
41            .arguments
42            .iter()
43            .filter(|argument| !matches!(argument.parser, UiArgumentParser::Positional { .. }))
44            .collect::<Vec<_>>();
45
46        let _ = writeln!(out, "{} - {}", self.invocation_name, self.summary);
47        let _ = writeln!(out);
48        let _ = writeln!(out, "Usage:");
49        let _ = writeln!(out, "  {}", self.usage);
50        if !positionals.is_empty() {
51            let _ = writeln!(out);
52            let _ = writeln!(out, "Arguments:");
53            write_help_section(&mut out, &positionals);
54        }
55        if !options.is_empty() {
56            let _ = writeln!(out);
57            let _ = writeln!(out, "Options:");
58            write_help_section(&mut out, &options);
59        }
60        out
61    }
62
63    /// Serialize the schema as pretty-printed JSON.
64    pub fn render_json_pretty(&self) -> Result<String, serde_json::Error> {
65        serde_json::to_string_pretty(self)
66    }
67
68    /// Return an argument definition by stable identifier.
69    pub fn argument(&self, id: &str) -> Option<&UiArgumentSchema> {
70        self.arguments.iter().find(|argument| argument.id == id)
71    }
72}

74/// Argument definition within the `--ui-schema` payload.
75#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
76pub struct UiArgumentSchema {
77    /// Stable argument identifier used by launchers and tests.
78    pub id: String,
79    /// Human-friendly label for form rendering.
80    pub label: String,
81    /// Zero-based presentation and parsing order.
82    pub order: usize,
83    /// Parsing model for this argument.
84    pub parser: UiArgumentParser,
85    /// Value type expected by the argument.
86    pub value_kind: UiValueKind,
87    /// Whether the argument must be provided explicitly.
88    pub required: bool,
89    /// Default value encoded as a string when applicable.
90    pub default: Option<String>,
91    /// Help text used by `--help` and TUI affordances.
92    pub help: String,
93    /// Logical group name for TUI sections.
94    pub group: String,
95    /// Whether the argument should be treated as advanced.
96    pub advanced: bool,
97    /// Whether the argument should be hidden from the TUI form.
98    pub hidden_in_tui: bool,
99}
100
101impl UiArgumentSchema {
102    fn help_spec(&self) -> String {
103        match &self.parser {
104            UiArgumentParser::Positional { metavar } => format!("<{metavar}>"),
105            UiArgumentParser::Option { flags, metavar, .. } => {
106                format!("{} <{metavar}>", flags.join(", "))
107            }
108            UiArgumentParser::Toggle {
109                true_flags,
110                false_flags,
111            } => true_flags
112                .iter()
113                .chain(false_flags.iter())
114                .cloned()
115                .collect::<Vec<_>>()
116                .join(", "),
117            UiArgumentParser::Action { flags, .. } => flags.join(", "),
118        }
119    }
120
121    /// Return the boolean default value when this argument is a toggle.
122    pub fn default_bool(&self) -> Option<bool> {
123        self.default.as_deref().and_then(|value| match value {
124            "true" => Some(true),
125            "false" => Some(false),
126            _ => None,
127        })
128    }
129}

131/// Parsing mode for one command argument.
132#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
133#[serde(tag = "kind", rename_all = "snake_case")]
134pub enum UiArgumentParser {
135    /// A positional argument consumed in declaration order.
136    Positional {
137        /// Placeholder name shown in help output.
138        metavar: String,
139    },
140    /// An option taking an explicit value.
141    Option {
142        /// Accepted short and long flags.
143        flags: Vec<String>,
144        /// Placeholder name shown in help output.
145        metavar: String,
146        /// Allowed values for choice-style options.
147        choices: Vec<String>,
148    },
149    /// A boolean toggle with explicit enable and disable flags.
150    Toggle {
151        /// Flags that set the toggle to `true`.
152        true_flags: Vec<String>,
153        /// Flags that set the toggle to `false`.
154        false_flags: Vec<String>,
155    },
156    /// A meta action such as `--help`.
157    Action {
158        /// Accepted flags that trigger the action.
159        flags: Vec<String>,
160        /// Action semantics for the flag.
161        action: UiActionKind,
162    },
163}

165/// High-level value type used by the TUI form renderer.
166#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
167#[serde(rename_all = "snake_case")]
168pub enum UiValueKind {
169    /// No associated value.
170    None,
171    /// Filesystem path.
172    Path,
173    /// Free-form string.
174    String,
175    /// Boolean toggle.
176    Bool,
177    /// Enumerated choice.
178    Choice,
179    /// Floating-point number.
180    Float,
181}

183/// Meta action exposed as a flag in the command schema.
184#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
185#[serde(rename_all = "snake_case")]
186pub enum UiActionKind {
187    /// Render human-readable help.
188    Help,
189    /// Emit the machine-readable UI schema.
190    UiSchema,
191}

193/// Structured-output contract for rich UI renderers.
194#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
195pub struct UiManagedOutputSchema {
196    /// Renderer identifier understood by the launcher.
197    pub renderer: String,
198    /// Expected stdout encoding for the structured renderer.
199    pub stdout_format: String,
200    /// Arguments the launcher should inject when it wants structured output.
201    pub inject_arguments: Vec<UiInjectedArgument>,
202    /// Whether raw stdout inspection should remain available.
203    pub raw_stdout_available: bool,
204    /// Whether raw stderr inspection should remain available.
205    pub raw_stderr_available: bool,
206}

208/// One launcher-managed argument injection for structured output.
209#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
210pub struct UiInjectedArgument {
211    /// Flag name to inject into the subprocess argv.
212    pub flag: String,
213    /// Value paired with the injected flag.
214    pub value: String,
215}

217fn write_help_section(out: &mut String, arguments: &[&UiArgumentSchema]) {
218    let width = arguments
219        .iter()
220        .map(|argument| argument.help_spec().len())
221        .max()
222        .unwrap_or(0);
223    for argument in arguments {
224        let spec = argument.help_spec();
225        let _ = writeln!(out, "  {spec:<width$}  {}", argument.help);
226    }
