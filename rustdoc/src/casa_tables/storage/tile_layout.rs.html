2//! Checked byte-aware physical tile layout planning.
3
4/// Default physical I/O target for newly created tiled arrays and columns.
5///
6/// Four MiB is large enough to sustain sequential local-disk throughput while
7/// keeping a small fixed number of tiles comfortably bounded in memory. The
8/// value controls new-file layout only; readers always honor the tile shape
9/// persisted in an existing casacore table.
10pub const DEFAULT_TILE_IO_BYTES: usize = 4 * 1024 * 1024;
11
12/// Errors produced while planning a physical tile layout.
13#[derive(Debug, Clone, PartialEq, Eq, thiserror::Error)]
14pub enum TileLayoutError {
15    #[error("tile layout requires at least one logical cell axis")]
16    EmptyShape,
17    #[error("tile layout element size must be positive")]
18    ZeroElementBytes,
19    #[error("tile layout I/O target must be positive")]
20    ZeroTargetBytes,
21    #[error("storage axis order {order:?} is not a permutation of 0..{rank}")]
22    InvalidAxisOrder { order: Vec<usize>, rank: usize },
23    #[error("explicit tile rank mismatch: expected {expected}, got {got}")]
24    ExplicitRank { expected: usize, got: usize },
25    #[error("explicit tile extent on axis {axis} must be positive")]
26    ZeroTileExtent { axis: usize },
27    #[error("tile layout size overflow while computing {what}")]
28    Overflow { what: &'static str },
29}
30
31/// Deterministic physical layout chosen for one tiled cell or array.
32#[derive(Debug, Clone, PartialEq, Eq)]
33pub struct TileLayoutPlan {
34    cell_tile_shape: Vec<usize>,
35    row_extent: Option<usize>,
36    tile_shape: Vec<usize>,
37    tile_elements: usize,
38    tile_bytes: usize,
39    target_io_bytes: usize,
40}

68/// Pure byte-aware planner shared by tiled tables, lattices, and images.
69#[derive(Debug, Clone, Copy, PartialEq, Eq)]
70pub struct TileLayoutPlanner {
71    target_io_bytes: usize,
72}
73
74impl TileLayoutPlanner {
75    pub fn new(target_io_bytes: usize) -> Result<Self, TileLayoutError> {
76        if target_io_bytes == 0 {
77            return Err(TileLayoutError::ZeroTargetBytes);
78        }
79        Ok(Self { target_io_bytes })
80    }
81
82    pub const fn repository_default() -> Self {
83        Self {
84            target_io_bytes: DEFAULT_TILE_IO_BYTES,
85        }
86    }
87
88    pub const fn target_io_bytes(self) -> usize {
89        self.target_io_bytes
90    }
91
92    pub fn plan_array(
93        self,
94        logical_shape: &[usize],
95        element_bytes: usize,
96    ) -> Result<TileLayoutPlan, TileLayoutError> {
97        let order: Vec<usize> = (0..logical_shape.len()).collect();
98        self.plan(logical_shape, None, element_bytes, &order, None)
99    }
100
101    pub fn plan_explicit_array(
102        self,
103        logical_shape: &[usize],
104        element_bytes: usize,
105        tile_shape: &[usize],
106    ) -> Result<TileLayoutPlan, TileLayoutError> {
107        let order: Vec<usize> = (0..logical_shape.len()).collect();
108        self.plan(logical_shape, None, element_bytes, &order, Some(tile_shape))
109    }
110
111    pub fn plan_column(
112        self,
113        logical_cell_shape: &[usize],
114        row_count: usize,
115        element_bytes: usize,
116    ) -> Result<TileLayoutPlan, TileLayoutError> {
117        let order: Vec<usize> = (0..logical_cell_shape.len()).collect();
118        self.plan(
119            logical_cell_shape,
120            Some(row_count),
121            element_bytes,
122            &order,
123            None,
124        )
125    }
126
127    /// Plans a layout in caller-supplied storage-axis order.
128    ///
129    /// A zero-row table uses physical row extent one because casacore tile
130    /// shapes must be positive; its logical row count remains zero.
131    pub fn plan(
132        self,
133        logical_cell_shape: &[usize],
134        row_count: Option<usize>,
135        element_bytes: usize,
136        storage_axis_order: &[usize],
137        explicit_tile_shape: Option<&[usize]>,
138    ) -> Result<TileLayoutPlan, TileLayoutError> {
139        validate_inputs(
140            logical_cell_shape,
141            element_bytes,
142            self.target_io_bytes,
143            storage_axis_order,
144        )?;
145
146        let expected_rank = logical_cell_shape.len() + usize::from(row_count.is_some());
147        let (cell_tile_shape, row_extent) = if let Some(explicit) = explicit_tile_shape {
148            if explicit.len() != expected_rank {
149                return Err(TileLayoutError::ExplicitRank {
150                    expected: expected_rank,
151                    got: explicit.len(),
152                });
153            }
154            for (axis, &extent) in explicit.iter().enumerate() {
155                if extent == 0 {
156                    return Err(TileLayoutError::ZeroTileExtent { axis });
157                }
158            }
159            (
160                explicit[..logical_cell_shape.len()].to_vec(),
161                row_count.map(|_| explicit[logical_cell_shape.len()]),
162            )
163        } else {
164            let target_elements = (self.target_io_bytes / element_bytes).max(1);
165            let mut remaining = target_elements;
166            let mut cell_tile_shape = vec![1usize; logical_cell_shape.len()];
167            for &axis in storage_axis_order {
168                let extent = logical_cell_shape[axis].max(1).min(remaining.max(1));
169                cell_tile_shape[axis] = extent;
170                remaining = remaining.div_ceil(extent);
171            }
172            let cell_elements = checked_product(&cell_tile_shape, "cell tile elements")?;
173            let row_extent = row_count.map(|rows| {
174                let capacity = (target_elements / cell_elements).max(1);
175                rows.max(1).min(capacity)
176            });
177            (cell_tile_shape, row_extent)
178        };
179
180        let mut tile_shape = cell_tile_shape.clone();
181        if let Some(rows) = row_extent {
182            tile_shape.push(rows);
183        }
184        let tile_elements = checked_product(&tile_shape, "tile elements")?;
185        let tile_bytes = tile_elements
186            .checked_mul(element_bytes)
187            .ok_or(TileLayoutError::Overflow { what: "tile bytes" })?;
188
189        Ok(TileLayoutPlan {
190            cell_tile_shape,
191            row_extent,
192            tile_shape,
193            tile_elements,
194            tile_bytes,
195            target_io_bytes: self.target_io_bytes,
196        })
197    }
198}
199
200fn validate_inputs(
201    logical_shape: &[usize],
202    element_bytes: usize,
203    target_io_bytes: usize,
204    storage_axis_order: &[usize],
205) -> Result<(), TileLayoutError> {
206    if logical_shape.is_empty() {
207        return Err(TileLayoutError::EmptyShape);
208    }
209    if element_bytes == 0 {
210        return Err(TileLayoutError::ZeroElementBytes);
211    }
212    if target_io_bytes == 0 {
213        return Err(TileLayoutError::ZeroTargetBytes);
214    }
215    let mut sorted = storage_axis_order.to_vec();
216    sorted.sort_unstable();
217    if sorted != (0..logical_shape.len()).collect::<Vec<_>>() {
218        return Err(TileLayoutError::InvalidAxisOrder {
219            order: storage_axis_order.to_vec(),
220            rank: logical_shape.len(),
221        });
222    }
223    checked_product(logical_shape, "logical cell elements")?;
224    Ok(())
225}
226
227fn checked_product(values: &[usize], what: &'static str) -> Result<usize, TileLayoutError> {
228    values.iter().try_fold(1usize, |product, &value| {
229        product
230            .checked_mul(value)
231            .ok_or(TileLayoutError::Overflow { what })
232    })
233}

235#[cfg(test)]
236mod tests {
237    use super::*;
238
239    #[test]
240    fn plane_oriented_cube_plan_is_byte_bounded() {
241        let plan = TileLayoutPlanner::repository_default()
242            .plan_array(&[4096, 4096, 768], 4)
243            .unwrap();
244        assert_eq!(plan.tile_shape(), &[4096, 256, 1]);
245        assert_eq!(plan.tile_bytes(), 4 * 1024 * 1024);
246    }
247
248    #[test]
249    fn equal_byte_targets_respect_element_size() {
250        let planner = TileLayoutPlanner::new(64).unwrap();
251        assert_eq!(planner.plan_array(&[100], 4).unwrap().tile_shape(), &[16]);
252        assert_eq!(planner.plan_array(&[100], 8).unwrap().tile_shape(), &[8]);
253        assert_eq!(planner.plan_array(&[100], 16).unwrap().tile_shape(), &[4]);
254    }
255
256    #[test]
257    fn column_row_extent_consumes_remaining_budget() {
258        let plan = TileLayoutPlanner::new(1024)
259            .unwrap()
260            .plan_column(&[2, 8], 1000, 4)
261            .unwrap();
262        assert_eq!(plan.tile_shape(), &[2, 8, 16]);
263        assert_eq!(plan.tile_bytes(), 1024);
264    }
265
266    #[test]
267    fn zero_row_column_has_positive_physical_extent() {
268        let plan = TileLayoutPlanner::new(64)
269            .unwrap()
270            .plan_column(&[4], 0, 4)
271            .unwrap();
272        assert_eq!(plan.row_extent(), Some(1));
273        assert_eq!(plan.tile_shape(), &[4, 1]);
274    }
275
276    #[test]
277    fn explicit_shape_is_strictly_validated() {
278        let planner = TileLayoutPlanner::new(64).unwrap();
279        assert!(matches!(
280            planner.plan_explicit_array(&[4, 4], 4, &[4, 0]),
281            Err(TileLayoutError::ZeroTileExtent { axis: 1 })
282        ));
283        let padded = planner.plan_explicit_array(&[4, 4], 4, &[5, 1]).unwrap();
284        assert_eq!(padded.tile_shape(), &[5, 1]);
285        assert!(matches!(
286            planner.plan_explicit_array(&[4, 4], 4, &[4]),
287            Err(TileLayoutError::ExplicitRank { .. })
288        ));
289    }
290
291    #[test]
292    fn invalid_inputs_and_overflow_fail_before_io() {
293        let planner = TileLayoutPlanner::new(64).unwrap();
294        assert_eq!(
295            planner.plan_array(&[4, 0], 4).unwrap().tile_shape(),
296            &[4, 1]
297        );
298        assert!(matches!(
299            planner.plan(&[4, 4], None, 4, &[0, 0], None),
300            Err(TileLayoutError::InvalidAxisOrder { .. })
301        ));
302        assert!(matches!(
303            planner.plan_array(&[usize::MAX, 2], 4),
304            Err(TileLayoutError::Overflow { .. })
305        ));
306    }
