81    pub(crate) fn to_primitive_type(self) -> Option<PrimitiveType> {
82        match self {

84            Self::TpUChar | Self::TpArrayUChar => Some(PrimitiveType::UInt8),
85            Self::TpShort | Self::TpArrayShort => Some(PrimitiveType::Int16),
86            Self::TpUShort | Self::TpArrayUShort => Some(PrimitiveType::UInt16),
87            Self::TpInt | Self::TpArrayInt => Some(PrimitiveType::Int32),

99    pub(crate) fn is_scalar(self) -> bool {
100        (self as i32) <= Self::TpString as i32 && self != Self::TpTable || self == Self::TpInt64
101    }

104        if is_array {

114                PrimitiveType::Complex32 => Self::TpArrayComplex,
115                PrimitiveType::Complex64 => Self::TpArrayDComplex,
116                PrimitiveType::String => Self::TpArrayString,
117                PrimitiveType::Int64 => Self::TpArrayInt64,
118            }
119        } else {
120            match pt {
121                PrimitiveType::Bool => Self::TpBool,
122                PrimitiveType::UInt8 => Self::TpUChar,
123                PrimitiveType::Int16 => Self::TpShort,
124                PrimitiveType::UInt16 => Self::TpUShort,

127                PrimitiveType::Float32 => Self::TpFloat,

130                PrimitiveType::Complex64 => Self::TpDComplex,
131                PrimitiveType::String => Self::TpString,
132                PrimitiveType::Int64 => Self::TpInt64,
133            }
134        }
135    }

138/// Return the casacore className for a scalar column descriptor.
139///
140/// C++ casacore format: `"ScalarColumnDesc<" + getTypeStr(T)` where getTypeStr
141/// returns an 8-char padded type name (e.g. `"Bool    "`, `"float   "`).
142/// No closing `>`. See `casacore/casa/Utilities/ValType.h`.
143pub(crate) fn scalar_column_class_name(pt: PrimitiveType) -> &'static str {
144    match pt {
145        PrimitiveType::Bool => "ScalarColumnDesc<Bool    ",
146        PrimitiveType::UInt8 => "ScalarColumnDesc<uChar   ",
147        PrimitiveType::Int16 => "ScalarColumnDesc<Short   ",
148        PrimitiveType::UInt16 => "ScalarColumnDesc<uShort  ",
149        PrimitiveType::Int32 => "ScalarColumnDesc<Int     ",
150        PrimitiveType::UInt32 => "ScalarColumnDesc<uInt    ",
151        PrimitiveType::Float32 => "ScalarColumnDesc<float   ",
152        PrimitiveType::Float64 => "ScalarColumnDesc<double  ",
153        PrimitiveType::Complex32 => "ScalarColumnDesc<Complex ",
154        PrimitiveType::Complex64 => "ScalarColumnDesc<DComplex",
155        PrimitiveType::String => "ScalarColumnDesc<String  ",
156        PrimitiveType::Int64 => "ScalarColumnDesc<Int64   ",
157    }
158}

180/// Result of parsing a casacore column descriptor class name.
181///
182/// Distinguishes between typed columns (scalar/array with a [`PrimitiveType`])
183/// and record columns (`ScalarRecordColumnDesc` / `TpRecord`).
184#[derive(Debug, Clone, Copy, PartialEq, Eq)]
185pub(crate) enum ColumnClassInfo {
186    /// A scalar or array column with a known element type.
187    Typed {
188        primitive_type: PrimitiveType,
189        is_array: bool,
190    },
191    /// A record column (`ScalarRecordColumnDesc`).
192    Record,
193}
194
195/// The C++ class name written for record column descriptors.
196///
197/// Corresponds to C++ `ScalarRecordColumnDesc`.
198pub(crate) const RECORD_COLUMN_CLASS_NAME: &str = "ScalarRecordColumnDesc";
199
200/// Parse a casacore className string into a [`ColumnClassInfo`].
201///
202/// Handles `ScalarColumnDesc<T>`, `ArrayColumnDesc<T>` (both the C++ on-disk
203/// format with 8-char padded type names and the legacy Rust format with closing
204/// `>`), and `ScalarRecordColumnDesc`.
205pub(crate) fn parse_column_class_name(class_name: &str) -> Option<ColumnClassInfo> {
206    if class_name == RECORD_COLUMN_CLASS_NAME {
207        Some(ColumnClassInfo::Record)
208    } else if let Some(inner) = class_name.strip_prefix("ScalarColumnDesc<") {
209        // Strip optional closing '>' (legacy Rust format) then trim spaces
210        let inner = inner.strip_suffix('>').unwrap_or(inner).trim();
211        casacore_type_name_to_primitive(inner).map(|pt| ColumnClassInfo::Typed {
212            primitive_type: pt,
213            is_array: false,
214        })
215    } else if let Some(inner) = class_name.strip_prefix("ArrayColumnDesc<") {
216        let inner = inner.strip_suffix('>').unwrap_or(inner).trim();
217        casacore_type_name_to_primitive(inner).map(|pt| ColumnClassInfo::Typed {
218            primitive_type: pt,
219            is_array: true,
220        })
221    } else {
222        None
223    }
224}

226/// Map a casacore type name to PrimitiveType.
227///
228/// Accepts both C++ canonical names (`"float"`, `"double"`) and capitalized
229/// forms (`"Float"`, `"Double"`) for robustness.
230fn casacore_type_name_to_primitive(name: &str) -> Option<PrimitiveType> {
231    match name {
232        "Bool" => Some(PrimitiveType::Bool),
233        "uChar" => Some(PrimitiveType::UInt8),
234        "Short" => Some(PrimitiveType::Int16),
235        "uShort" => Some(PrimitiveType::UInt16),
236        "Int" => Some(PrimitiveType::Int32),
237        "uInt" => Some(PrimitiveType::UInt32),
238        "float" | "Float" => Some(PrimitiveType::Float32),
239        "double" | "Double" => Some(PrimitiveType::Float64),
240        "Complex" => Some(PrimitiveType::Complex32),
241        "DComplex" => Some(PrimitiveType::Complex64),
242        "String" => Some(PrimitiveType::String),
243        "Int64" => Some(PrimitiveType::Int64),
244        _ => None,
245    }
