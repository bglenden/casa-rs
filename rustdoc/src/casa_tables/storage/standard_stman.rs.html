2//! StandardStMan (SSM) bucket-based storage manager reader/writer.
3//!
4//! On-disk format: `table.fN` file with 512-byte AipsIO header at offset 0,
5//! followed by fixed-size buckets at offset 512+.
6//!
7//! The SSM file header and index data use in-memory AipsIO whose byte order
8//! matches the table's endian setting (BE for BE tables, LE for LE tables).
9//! The DM data blob from `table.dat` always uses big-endian AipsIO
10//! (file-based). Bucket column data byte order is indicated by the
11//! `big_endian` flag in the SSM header. String bucket metadata (freeLink,
12//! usedLength, nDeleted, nextBucket) and index bucket chain pointers always
13//! use canonical (big-endian) encoding regardless of the table's endian
14//! setting.
15

67/// Minimal reader for AipsIO-framed data in either byte order.
68struct AipsIoBuf<'a> {
69    data: &'a [u8],
70    pos: usize,
71    order: ByteOrder,
72    level: usize,
73}
74
75impl<'a> AipsIoBuf<'a> {
76    fn new(data: &'a [u8], order: ByteOrder) -> Self {
77        Self {
78            data,
79            pos: 0,
80            order,
81            level: 0,
82        }
83    }
84
85    fn read_bytes(&mut self, n: usize) -> Result<&'a [u8], StorageError> {
86        if self.pos + n > self.data.len() {
87            return Err(StorageError::FormatMismatch(
88                "SSM AipsIO buffer underrun".to_string(),
89            ));
90        }
91        let slice = &self.data[self.pos..self.pos + n];
92        self.pos += n;
93        Ok(slice)
94    }
95
96    fn read_u32(&mut self) -> Result<u32, StorageError> {
97        let b = self.read_bytes(4)?;
98        Ok(match self.order {
99            ByteOrder::BigEndian => u32::from_be_bytes([b[0], b[1], b[2], b[3]]),
100            ByteOrder::LittleEndian => u32::from_le_bytes([b[0], b[1], b[2], b[3]]),
101        })
102    }
103
104    fn read_i32(&mut self) -> Result<i32, StorageError> {
105        let b = self.read_bytes(4)?;
106        Ok(match self.order {
107            ByteOrder::BigEndian => i32::from_be_bytes([b[0], b[1], b[2], b[3]]),
108            ByteOrder::LittleEndian => i32::from_le_bytes([b[0], b[1], b[2], b[3]]),
109        })
110    }
111
112    fn read_u64(&mut self) -> Result<u64, StorageError> {
113        let b = self.read_bytes(8)?;
114        Ok(match self.order {
115            ByteOrder::BigEndian => {
116                u64::from_be_bytes([b[0], b[1], b[2], b[3], b[4], b[5], b[6], b[7]])
117            }
118            ByteOrder::LittleEndian => {
119                u64::from_le_bytes([b[0], b[1], b[2], b[3], b[4], b[5], b[6], b[7]])
120            }
121        })
122    }
123
124    fn read_bool(&mut self) -> Result<bool, StorageError> {
125        let b = self.read_bytes(1)?;
126        Ok(b[0] != 0)
127    }
128
129    fn read_string(&mut self) -> Result<String, StorageError> {
130        let len = self.read_u32()? as usize;
131        let b = self.read_bytes(len)?;
132        String::from_utf8(b.to_vec())
133            .map_err(|e| StorageError::FormatMismatch(format!("invalid UTF-8 in SSM: {e}")))
134    }
135
136    /// Read AipsIO getstart: magic at level 0, then obj_len + type + version.
137    fn getstart(&mut self, expected_type: &str) -> Result<u32, StorageError> {
138        if self.level == 0 {
139            let magic = self.read_u32()?;
140            if magic != AIPSIO_MAGIC {
141                return Err(StorageError::FormatMismatch(format!(
142                    "SSM AipsIO magic mismatch: expected 0x{AIPSIO_MAGIC:08x}, got 0x{magic:08x}"
143                )));
144            }
145        }
146        self.level += 1;
147        let _obj_len = self.read_u32()?;
148        let type_name = self.read_string()?;
149        if type_name != expected_type {
150            return Err(StorageError::FormatMismatch(format!(
151                "SSM AipsIO type mismatch: expected '{expected_type}', got '{type_name}'"
152            )));
153        }
154        self.read_u32()
155    }
156
157    /// Finish reading an object (decrements level, no validation).
158    fn getend(&mut self) {
159        if self.level > 0 {
160            self.level -= 1;
161        }
162    }
163
164    /// Read a Block<uInt> (nested AipsIO "Block" object containing u32 values).
165    fn read_block_u32(&mut self) -> Result<Vec<u32>, StorageError> {
166        let _version = self.getstart("Block")?;
167        let count = self.read_u32()?;
168        let mut values = Vec::with_capacity(count as usize);
169        for _ in 0..count {
170            values.push(self.read_u32()?);
171        }
172        self.getend();
173        Ok(values)
174    }
175
176    /// Read a Block<Int64> (nested AipsIO "Block" object containing u64 values).
177    fn read_block_u64(&mut self) -> Result<Vec<u64>, StorageError> {
178        let _version = self.getstart("Block")?;
179        let count = self.read_u32()?;
180        let mut values = Vec::with_capacity(count as usize);
181        for _ in 0..count {
182            values.push(self.read_u64()?);
183        }
184        self.getend();
185        Ok(values)
186    }
187}

189/// Detect the byte order of in-memory AipsIO data.
190/// Returns the byte order by checking the object length field after the magic.
191fn detect_aipsio_byte_order(data: &[u8]) -> Result<ByteOrder, StorageError> {
192    if data.len() < 8 {
193        return Err(StorageError::FormatMismatch(
194            "SSM data too short for byte order detection".to_string(),
195        ));
196    }
197    // Bytes 0-3: magic (same in both byte orders for 0xbebebebe)
198    // Bytes 4-7: object length
199    let be_len = u32::from_be_bytes([data[4], data[5], data[6], data[7]]);
200    let le_len = u32::from_le_bytes([data[4], data[5], data[6], data[7]]);
201
202    // A valid object length is small (< 4096 for any reasonable SSM header/index)
203    let be_ok = be_len > 0 && be_len < 4096;
204    let le_ok = le_len > 0 && le_len < 4096;
205
206    match (be_ok, le_ok) {
207        (true, false) => Ok(ByteOrder::BigEndian),
208        (false, true) => Ok(ByteOrder::LittleEndian),
209        (true, true) => {
210            // Both look valid — prefer canonical (big-endian) when ambiguous
211            Ok(ByteOrder::BigEndian)
212        }
213        (false, false) => Err(StorageError::FormatMismatch(format!(
214            "SSM: cannot detect byte order (be_len={be_len}, le_len={le_len})"
215        ))),
216    }
217}

266fn parse_ssm_header(file: &mut File) -> Result<SsmHeader, StorageError> {
267    file.seek(SeekFrom::Start(0))?;
268    let mut header_buf = vec![0u8; SSM_HEADER_SIZE as usize];
269    file.read_exact(&mut header_buf)?;
270
271    let io_order = detect_aipsio_byte_order(&header_buf)?;
272    let mut buf = AipsIoBuf::new(&header_buf, io_order);
273    let version = buf.getstart("StandardStMan")?;
274
275    let big_endian = if version >= 3 {
276        buf.read_bool()?
277    } else {
278        true // versions 1-2 are always big-endian
279    };
280
281    let bucket_size = buf.read_u32()?;
282    let nr_buckets = buf.read_u32()?;
283    let pers_cache_size = buf.read_u32()?;
284    let nr_free_buckets = buf.read_u32()?;
285    let first_free_bucket = buf.read_i32()?;
286    let nr_idx_buckets = buf.read_u32()?;
287    let first_idx_bucket = buf.read_i32()?;
288
289    let idx_bucket_offset = if version >= 2 { buf.read_u32()? } else { 0 };
290
291    let last_string_bucket = buf.read_i32()?;
292    let index_length = buf.read_u32()?;
293    let nr_indices = buf.read_u32()?;
294
295    Ok(SsmHeader {
296        bucket_size,
297        nr_buckets,
298        pers_cache_size,
299        nr_free_buckets,
300        first_free_bucket,
301        nr_idx_buckets,
302        first_idx_bucket,
303        idx_bucket_offset,
304        last_string_bucket,
305        index_length,
306        nr_indices,
307        big_endian,
308        io_order,
309    })
310}

316/// Parse the DM data blob from table.dat. This blob is written within the
317/// file-based AipsIO stream (always big-endian canonical encoding).
318///
319/// C++ casacore serialises the arrays via `putBlock`/`getBlock` which wraps
320/// each array in a nested AipsIO `"Block"` object.
321fn parse_ssm_dm_blob(data: &[u8]) -> Result<SsmDmInfo, StorageError> {
322    let cursor = std::io::Cursor::new(data.to_vec());
323    let mut io = AipsIo::new_read_only(cursor);
324    let _version = io.getstart("SSM")?;
325
326    let name = io.get_string()?;
327
328    let column_offsets = read_block_u32(&mut io)?;
329    let col_index_map = read_block_u32(&mut io)?;
330
331    io.getend()?;
332
333    Ok(SsmDmInfo {
334        name,
335        column_offsets,
336        col_index_map,
337    })
338}

340/// Read a `Block<uInt>` as serialised by C++ casacore's `putBlock`.
341fn read_block_u32(io: &mut AipsIo) -> Result<Vec<u32>, StorageError> {
342    let _version = io.getstart("Block")?;
343    let count = io.get_u32()?;
344    let mut values = vec![0u32; count as usize];
345    for v in &mut values {
346        *v = io.get_u32()?;
347    }
348    io.getend()?;
349    Ok(values)
350}

356fn read_bucket(
357    file: &mut File,
358    header: &SsmHeader,
359    bucket_nr: u32,
360) -> Result<Vec<u8>, StorageError> {
361    let offset = SSM_HEADER_SIZE + (bucket_nr as u64) * (header.bucket_size as u64);
362    let file_len = file.metadata()?.len();
363    if offset + header.bucket_size as u64 > file_len {
364        return Err(StorageError::FormatMismatch(format!(
365            "SSM bucket {bucket_nr} out of range: offset={offset}, bucket_size={}, file_len={file_len}",
366            header.bucket_size
367        )));
368    }
369    file.seek(SeekFrom::Start(offset))?;
370    let mut buf = vec![0u8; header.bucket_size as usize];
371    file.read_exact(&mut buf)?;
372    Ok(buf)
373}

375fn parse_ssm_indices(file: &mut File, header: &SsmHeader) -> Result<Vec<SsmIndex>, StorageError> {
376    if header.nr_indices == 0 || header.first_idx_bucket < 0 {
377        return Ok(Vec::new());
378    }
379
380    // Assemble index data from index bucket(s)
381    let mut index_data = Vec::with_capacity(header.index_length as usize);
382
383    if header.nr_idx_buckets <= 1 && header.idx_bucket_offset > 0 {
384        // Index fits in part of one bucket
385        let bucket = read_bucket(file, header, header.first_idx_bucket as u32)?;
386        let start = header.idx_bucket_offset as usize;
387        let end = start + header.index_length as usize;
388        if end <= bucket.len() {
389            index_data.extend_from_slice(&bucket[start..end]);
390        } else {
391            return Err(StorageError::FormatMismatch(
392                "SSM index data exceeds bucket bounds".to_string(),
393            ));
394        }
395    } else {
396        // Index spans one or more full buckets.
397        // Each index bucket has an 8-byte header: 2 × canonical Int
398        //   bytes 0-3: check number (canonical/big-endian)
399        //   bytes 4-7: next bucket number (canonical/big-endian)
400        // Note: chain pointers use CanonicalConversion (always big-endian),
401        // independent of the SSM's big_endian flag.
402        let chain_overhead = 8usize;
403        let data_per_bucket = header.bucket_size as usize - chain_overhead;
404        let mut bucket_nr = header.first_idx_bucket as u32;
405        let mut remaining = header.index_length as usize;
406
407        for _ in 0..header.nr_idx_buckets {
408            if remaining == 0 {
409                break;
410            }
411            let bucket = read_bucket(file, header, bucket_nr)?;
412            let chunk = remaining.min(data_per_bucket);
413            index_data.extend_from_slice(&bucket[chain_overhead..chain_overhead + chunk]);
414            remaining = remaining.saturating_sub(data_per_bucket);
415
416            if remaining > 0 {
417                // Next bucket from chain pointer (always canonical/big-endian)
418                bucket_nr = read_i32_be(&bucket[4..8]) as u32;
419            }
420        }
421
422        if remaining > 0 {
423            return Err(StorageError::FormatMismatch(
424                "SSM index data incomplete after following bucket chain".to_string(),
425            ));
426        }
427    }
428
429    // Parse index data using the same byte order as the header (in-memory AipsIO).
430    // Each SSMIndex is a top-level AipsIO object (level 0 → prefixed with magic).
431    let mut buf = AipsIoBuf::new(&index_data, header.io_order);
432
433    let mut indices = Vec::with_capacity(header.nr_indices as usize);
434    for _ in 0..header.nr_indices {
435        let version = buf.getstart("SSMIndex")?;
436
437        let n_used = buf.read_u32()?;
438        let rows_per_bucket = buf.read_u32()?;
439        let _nr_columns = buf.read_i32()?;
440
441        // Free space map: nested "SimpleOrderedMap" AipsIO object
442        {
443            let _map_version = buf.getstart("SimpleOrderedMap")?;
444            let _default_val = buf.read_i32()?;
445            let nr = buf.read_u32()?;
446            let _incr = buf.read_u32()?;
447            for _ in 0..nr {
448                let _key = buf.read_i32()?;
449                let _val = buf.read_i32()?;
450            }
451            buf.getend();
452        }
453
454        // Last row numbers: Block<uInt> (v1) or Block<Int64> (v2)
455        let last_row = if version == 1 {
456            buf.read_block_u32()?
457                .into_iter()
458                .map(|v| v as u64)
459                .collect()
460        } else {
461            buf.read_block_u64()?
462        };
463
464        // Bucket numbers: Block<uInt>
465        let bucket_number = buf.read_block_u32()?;
466
467        buf.getend();
468
469        indices.push(SsmIndex {
470            n_used,
471            rows_per_bucket,
472            last_row,
473            bucket_number,
474        });
475    }
476
477    Ok(indices)
478}

480impl SsmIndex {
481    /// Find the bucket containing `row_nr`.
482    /// Returns (bucket_nr, start_row, end_row) where the data bucket contains
483    /// rows [start_row..=end_row].
484    fn find_bucket(&self, row_nr: u64) -> Option<(u32, u64, u64)> {
485        let pos = self.last_row.partition_point(|&lr| lr < row_nr);
486        if pos >= self.last_row.len() {
487            return None;
488        }
489        let bucket_nr = self.bucket_number[pos];
490        let start_row = if pos == 0 {
491            0
492        } else {
493            self.last_row[pos - 1] + 1
494        };
495        let end_row = self.last_row[pos];
496        Some((bucket_nr, start_row, end_row))
497    }
498}

512fn extract_scalar_from_ssm_bytes(
513    bytes: &[u8],
514    data_type: CasacoreDataType,
515    big_endian: bool,
516) -> Result<Value, StorageError> {
517    let value = match data_type {
518        CasacoreDataType::TpUChar => Value::Scalar(ScalarValue::UInt8(bytes[0])),
519        CasacoreDataType::TpShort => Value::Scalar(ScalarValue::Int16(if big_endian {
520            i16::from_be_bytes(bytes.try_into().map_err(|_| {
521                StorageError::FormatMismatch("invalid SSM i16 scalar width".to_string())
522            })?)
523        } else {
524            i16::from_le_bytes(bytes.try_into().map_err(|_| {
525                StorageError::FormatMismatch("invalid SSM i16 scalar width".to_string())
526            })?)
527        })),
528        CasacoreDataType::TpUShort => Value::Scalar(ScalarValue::UInt16(if big_endian {
529            u16::from_be_bytes(bytes.try_into().map_err(|_| {
530                StorageError::FormatMismatch("invalid SSM u16 scalar width".to_string())
531            })?)
532        } else {
533            u16::from_le_bytes(bytes.try_into().map_err(|_| {
534                StorageError::FormatMismatch("invalid SSM u16 scalar width".to_string())
535            })?)
536        })),
537        CasacoreDataType::TpInt => Value::Scalar(ScalarValue::Int32(if big_endian {
538            read_i32_be(bytes)
539        } else {
540            i32::from_le_bytes(bytes.try_into().map_err(|_| {
541                StorageError::FormatMismatch("invalid SSM i32 scalar width".to_string())
542            })?)
543        })),
544        CasacoreDataType::TpUInt => Value::Scalar(ScalarValue::UInt32(if big_endian {
545            u32::from_be_bytes(bytes.try_into().map_err(|_| {
546                StorageError::FormatMismatch("invalid SSM u32 scalar width".to_string())
547            })?)
548        } else {
549            u32::from_le_bytes(bytes.try_into().map_err(|_| {
550                StorageError::FormatMismatch("invalid SSM u32 scalar width".to_string())
551            })?)
552        })),
553        CasacoreDataType::TpFloat => Value::Scalar(ScalarValue::Float32(if big_endian {
554            read_f32_be(bytes)
555        } else {
556            read_f32_le(bytes)
557        })),
558        CasacoreDataType::TpDouble => Value::Scalar(ScalarValue::Float64(if big_endian {
559            read_f64_be(bytes)
560        } else {
561            read_f64_le(bytes)
562        })),
563        CasacoreDataType::TpInt64 => Value::Scalar(ScalarValue::Int64(if big_endian {
564            i64::from_be_bytes(bytes.try_into().map_err(|_| {
565                StorageError::FormatMismatch("invalid SSM i64 scalar width".to_string())
566            })?)
567        } else {
568            i64::from_le_bytes(bytes.try_into().map_err(|_| {
569                StorageError::FormatMismatch("invalid SSM i64 scalar width".to_string())
570            })?)
571        })),
572        CasacoreDataType::TpComplex => {
573            if bytes.len() != 8 {
574                return Err(StorageError::FormatMismatch(
575                    "invalid SSM complex32 scalar width".to_string(),
576                ));
577            }
578            let re = if big_endian {
579                read_f32_be(&bytes[..4])
580            } else {
581                read_f32_le(&bytes[..4])
582            };
583            let im = if big_endian {
584                read_f32_be(&bytes[4..8])
585            } else {
586                read_f32_le(&bytes[4..8])
587            };
588            Value::Scalar(ScalarValue::Complex32(casa_types::Complex32::new(re, im)))
589        }
590        CasacoreDataType::TpDComplex => {
591            if bytes.len() != 16 {
592                return Err(StorageError::FormatMismatch(
593                    "invalid SSM complex64 scalar width".to_string(),
594                ));
595            }
596            let re = if big_endian {
597                read_f64_be(&bytes[..8])
598            } else {
599                read_f64_le(&bytes[..8])
600            };
601            let im = if big_endian {
602                read_f64_be(&bytes[8..16])
603            } else {
604                read_f64_le(&bytes[8..16])
605            };
606            Value::Scalar(ScalarValue::Complex64(casa_types::Complex64::new(re, im)))
607        }
608        other => {
609            return Err(StorageError::FormatMismatch(format!(
610                "unsupported SSM scalar type {other:?}"
611            )));
612        }
613    };
614    Ok(value)
615}

618// String bucket reader
619// ---------------------------------------------------------------------------
620
621const STRING_BUCKET_HEADER: usize = 16;
622
623fn read_ssm_string(
624    file: &mut File,
625    header: &SsmHeader,
626    bucket_nr: i32,
627    offset: i32,
628    length: i32,
629) -> Result<String, StorageError> {
630    if length == 0 {
631        return Ok(String::new());
632    }
633    if bucket_nr < 0 || offset < 0 || length < 0 {
634        return Err(StorageError::FormatMismatch(format!(
635            "invalid SSM string locator bucket={} offset={} length={}",
636            bucket_nr, offset, length
637        )));
638    }
639
640    let mut result = Vec::with_capacity(length as usize);
641    let mut remaining = length as usize;
642    let mut cur_bucket = bucket_nr;
643    let mut cur_offset = offset as usize;
644
645    while remaining > 0 {
646        let bucket = read_bucket(file, header, cur_bucket as u32)?;
647        let data_start = STRING_BUCKET_HEADER + cur_offset;
648        let available = bucket.len() - data_start;
649        let chunk = remaining.min(available);
650        result.extend_from_slice(&bucket[data_start..data_start + chunk]);
651        remaining -= chunk;
652
653        if remaining > 0 {
654            // String bucket metadata (nextBucket) is always canonical (BE)
655            // per C++ CanonicalConversion in SSMStringHandler.
656            let next_bucket = read_i32_be(&bucket[12..16]);
657            if next_bucket < 0 {
658                return Err(StorageError::FormatMismatch(
659                    "SSM string chain ended prematurely".to_string(),
660                ));
661            }
662            cur_bucket = next_bucket;
663            cur_offset = 0;
664        }
665    }
666
667    String::from_utf8(result)
668        .map_err(|e| StorageError::FormatMismatch(format!("SSM string is not valid UTF-8: {e}")))
669}

672fn read_ssm_string_bytes(
673    file: &mut File,
674    header: &SsmHeader,
675    bucket_nr: i32,
676    offset: i32,
677    length: i32,
678) -> Result<Vec<u8>, StorageError> {
679    if length == 0 {
680        return Ok(Vec::new());
681    }
682    if bucket_nr < 0 || offset < 0 || length < 0 {
683        return Err(StorageError::FormatMismatch(format!(
684            "invalid SSM string locator bucket={} offset={} length={}",
685            bucket_nr, offset, length
686        )));
687    }
688
689    let mut result = Vec::with_capacity(length as usize);
690    let mut remaining = length as usize;
691    let mut cur_bucket = bucket_nr;
692    let mut cur_offset = offset as usize;
693
694    while remaining > 0 {
695        let bucket = read_bucket(file, header, cur_bucket as u32)?;
696        let data_start = STRING_BUCKET_HEADER + cur_offset;
697        let available = bucket.len() - data_start;
698        let chunk = remaining.min(available);
699        result.extend_from_slice(&bucket[data_start..data_start + chunk]);
700        remaining -= chunk;
701
702        if remaining > 0 {

704            if next_bucket < 0 {

715}

717fn read_ssm_string_array_value(
718    file: &mut File,
719    header: &SsmHeader,
720    bucket_nr: i32,
721    offset: i32,
722    length: i32,
723    fixed_shape: Option<&[i32]>,
724) -> Result<Option<Value>, StorageError> {
725    if length == 0 {
726        return Ok(None);
727    }
728
729    let block = read_ssm_string_bytes(file, header, bucket_nr, offset, length)?;
730    let (shape, mut pos, filled_flag) = if let Some(shape) = fixed_shape {
731        (
732            shape.iter().map(|&dim| dim as usize).collect::<Vec<_>>(),
733            0usize,
734            true,
735        )
736    } else {
737        if block.len() < 8 {
738            return Err(StorageError::FormatMismatch(
739                "SSM string array block too short for shape header".to_string(),
740            ));
741        }
742        let ndim = u32::from_be_bytes([block[0], block[1], block[2], block[3]]) as usize;
743        let shape_bytes = 4 + ndim * 4 + 4;
744        if block.len() < shape_bytes {
745            return Err(StorageError::FormatMismatch(
746                "SSM string array block too short for declared shape".to_string(),
747            ));
748        }
749
750        let mut shape = Vec::with_capacity(ndim);
751        let mut pos = 4usize;
752        for _ in 0..ndim {
753            let dim =
754                i32::from_be_bytes([block[pos], block[pos + 1], block[pos + 2], block[pos + 3]]);
755            if dim < 0 {
756                return Err(StorageError::FormatMismatch(
757                    "SSM string array shape contains negative extent".to_string(),
758                ));
759            }
760            shape.push(dim as usize);
761            pos += 4;
762        }
763        let filled_flag =
764            i32::from_be_bytes([block[pos], block[pos + 1], block[pos + 2], block[pos + 3]]) != 0;
765        (shape, pos + 4, filled_flag)
766    };
767
768    let nrelem = shape.iter().try_fold(1usize, |acc, &dim| {
769        acc.checked_mul(dim).ok_or_else(|| {
770            StorageError::FormatMismatch(format!(
771                "SSM string array shape product overflow for shape {:?}",
772                shape
773            ))
774        })
775    })?;
776    let mut values = Vec::with_capacity(nrelem);
777    if filled_flag {
778        for _ in 0..nrelem {
779            if pos + 4 > block.len() {
780                return Err(StorageError::FormatMismatch(
781                    "SSM string array: block too short for element length prefix".to_string(),
782                ));
783            }
784            let slen =
785                u32::from_be_bytes([block[pos], block[pos + 1], block[pos + 2], block[pos + 3]])
786                    as usize;
787            pos += 4;
788            if pos + slen > block.len() {
789                return Err(StorageError::FormatMismatch(
790                    "SSM string array: block too short for element payload".to_string(),
791                ));
792            }
793            let s = String::from_utf8(block[pos..pos + slen].to_vec())
794                .map_err(|e| StorageError::FormatMismatch(format!("invalid UTF-8: {e}")))?;
795            pos += slen;
796            values.push(s);
797        }
798    } else {
799        values.resize(nrelem, String::new());
800    }
801
802    let arr = ndarray::ArrayD::from_shape_vec(ndarray::IxDyn(&shape).f(), values)
803        .map_err(|e| StorageError::FormatMismatch(format!("array shape: {e}")))?;
804    Ok(Some(Value::Array(casa_types::ArrayValue::String(arr))))
805}

811pub(crate) fn read_ssm_file(
812    file_path: &Path,
813    dm_blob: &[u8],
814    col_descs: &[&ColumnDescContents],
815    nrrow: usize,
816) -> Result<Vec<(String, SsmColumnResult)>, StorageError> {
817    let mut file = File::open(file_path)?;
818    let header = parse_ssm_header(&mut file)?;
819    let dm_info = parse_ssm_dm_blob(dm_blob)?;
820    let indices = parse_ssm_indices(&mut file, &header)?;
821
822    // Check if any array columns are stored indirectly via the shared array file.
823    let has_indirect = col_descs.iter().copied().any(is_ssm_indirect);
824
825    // Lazily open the shared array file for indirect columns.
826    let mut array_reader: Option<StManArrayFileReader> = if has_indirect {
827        let mut arr_path = file_path.as_os_str().to_os_string();
828        arr_path.push("i");
829        let arr_pathbuf = std::path::PathBuf::from(arr_path);
830        if arr_pathbuf.exists() {
831            Some(StManArrayFileReader::open(&arr_pathbuf, header.big_endian)?)
832        } else {
833            None
834        }
835    } else {
836        None
837    };
838
839    let mut result = Vec::with_capacity(col_descs.len());
840
841    for (col_idx, col_desc) in col_descs.iter().enumerate() {
842        if col_idx >= dm_info.column_offsets.len() {
843            return Err(StorageError::FormatMismatch(format!(
844                "SSM column index {col_idx} out of range for columnOffsets"
845            )));
846        }
847
848        let column_offset = dm_info.column_offsets[col_idx] as usize;
849        let index_nr = dm_info.col_index_map[col_idx] as usize;
850
851        if index_nr >= indices.len() {
852            return Err(StorageError::FormatMismatch(format!(
853                "SSM column {} references index {index_nr} but only {} indices exist",
854                col_desc.col_name,
855                indices.len()
856            )));
857        }
858        let index = &indices[index_nr];
859
860        let is_indirect = is_ssm_indirect(col_desc);
861        let is_string_array = col_desc.is_array && is_ssm_variable_string(col_desc);
862
863        if is_string_array {
864            let refs = read_column_from_buckets(
865                &mut file,
866                &header,
867                index,
868                column_offset,
869                CasacoreDataType::TpInt,
870                3,
871                nrrow,
872            )?;
873
874            let ref_values = match refs {
875                ColumnRawData::Int32(v) => v,
876                _ => {
877                    return Err(StorageError::FormatMismatch(
878                        "SSM string array: expected Int32 bucket/offset/length references"
879                            .to_string(),
880                    ));
881                }
882            };
883
884            let mut per_row = Vec::with_capacity(nrrow);
885            for row_idx in 0..nrrow {
886                let base = row_idx * 3;
887                let bucket_nr = ref_values[base];
888                let str_offset = ref_values[base + 1];
889                let str_length = ref_values[base + 2];
890                let value = read_ssm_string_array_value(
891                    &mut file,
892                    &header,
893                    bucket_nr,
894                    str_offset,
895                    str_length,
896                    if col_desc.shape.is_empty() {
897                        None
898                    } else {
899                        Some(&col_desc.shape)
900                    },
901                )
902                .map_err(|err| {
903                    StorageError::FormatMismatch(format!(
904                        "SSM string array column '{}' row {} locator ({}, {}, {}): {}",
905                        col_desc.col_name, row_idx, bucket_nr, str_offset, str_length, err
906                    ))
907                })?;
908                per_row.push(value);
909            }
910
911            result.push((
912                col_desc.col_name.clone(),
913                SsmColumnResult::Indirect(per_row),
914            ));
915        } else if is_indirect {
916            // Read i64 offsets from bucket data (8 bytes per row).
917            let offsets = read_column_from_buckets(
918                &mut file,
919                &header,
920                index,
921                column_offset,
922                CasacoreDataType::TpInt64,
923                1,
924                nrrow,
925            )?;
926
927            let offset_values = match offsets {
928                ColumnRawData::Int64(v) => v,
929                _ => {
930                    return Err(StorageError::FormatMismatch(
931                        "SSM indirect: expected Int64 offsets".to_string(),
932                    ));
933                }
934            };
935
936            let dt = if col_desc.is_record() {
937                CasacoreDataType::TpUChar
938            } else {
939                CasacoreDataType::from_primitive_type(col_desc.require_primitive_type()?, false)
940            };
941            let mut per_row = Vec::with_capacity(nrrow);
942            if let Some(reader) = array_reader.as_mut() {
943                for (row_idx, &off) in offset_values.iter().enumerate() {
944                    let value = reader.read_array_at(off, dt).map_err(|err| {
945                        StorageError::FormatMismatch(format!(
946                            "SSM indirect column '{}' row {} offset {} type {:?}: {}",
947                            col_desc.col_name, row_idx, off, dt, err
948                        ))
949                    })?;
950                    let value = if col_desc.is_record() {
951                        match value {
952                            Some(Value::Array(ArrayValue::UInt8(bytes))) => {
953                                if bytes.ndim() != 1 {
954                                    return Err(StorageError::FormatMismatch(format!(
955                                        "SSM record column '{}' row {} expected 1-D u8 payload, got shape {:?}",
956                                        col_desc.col_name,
957                                        row_idx,
958                                        bytes.shape()
959                                    )));
960                                }
961                                let raw_bytes: Vec<u8> = bytes.iter().copied().collect();
962                                Some(Value::Record(
963                                    super::table_control::deserialize_record_from_uchar(
964                                        &raw_bytes,
965                                    )?,
966                                ))
967                            }
968                            Some(other) => {
969                                return Err(StorageError::FormatMismatch(format!(
970                                    "SSM record column '{}' row {} expected u8 payload, got {other:?}",
971                                    col_desc.col_name, row_idx
972                                )));
973                            }
974                            None => None,
975                        }
976                    } else {
977                        value
978                    };
979                    per_row.push(value);
980                }
981            } else if nrrow == 0 || offset_values.iter().all(|&off| off == 0) {
982                per_row.resize(nrrow, None);
983            } else {
984                return Err(StorageError::FormatMismatch(
985                    "SSM indirect column but no array file found".to_string(),
986                ));
987            }
988
989            result.push((
990                col_desc.col_name.clone(),
991                SsmColumnResult::Indirect(per_row),
992            ));
993        } else {
994            let nrelem = if col_desc.is_array && !col_desc.shape.is_empty() {
995                col_desc.shape.iter().map(|&s| s as usize).product()
996            } else {
997                1usize
998            };
999
1000            let raw = read_column_from_buckets(
1001                &mut file,
1002                &header,
1003                index,
1004                column_offset,
1005                col_desc.data_type,
1006                nrelem,
1007                nrrow,
1008            )?;
1009
1010            result.push((col_desc.col_name.clone(), SsmColumnResult::Flat(raw)));
1011        }
1012    }
1013
1014    Ok(result)
1015}

1017pub(crate) fn read_ssm_scalar_column_rows(
1018    file_path: &Path,
1019    dm_blob: &[u8],
1020    col_descs: &[&ColumnDescContents],
1021    target_col_idx: usize,
1022    selected_rows: &[usize],
1023) -> Result<Option<Vec<Option<ScalarValue>>>, StorageError> {
1024    if selected_rows.is_empty() {
1025        return Ok(Some(Vec::new()));
1026    }
1027
1028    let mut file = File::open(file_path)?;
1029    let header = parse_ssm_header(&mut file)?;
1030    let dm_info = parse_ssm_dm_blob(dm_blob)?;
1031    let indices = parse_ssm_indices(&mut file, &header)?;
1032
1033    if target_col_idx >= col_descs.len() {
1034        return Err(StorageError::FormatMismatch(format!(
1035            "SSM scalar column index {target_col_idx} is out of range"
1036        )));
1037    }
1038    if target_col_idx >= dm_info.column_offsets.len()
1039        || target_col_idx >= dm_info.col_index_map.len()
1040    {
1041        return Err(StorageError::FormatMismatch(format!(
1042            "SSM scalar column index {target_col_idx} missing dm blob metadata"
1043        )));
1044    }
1045
1046    let col_desc = col_descs[target_col_idx];
1047    if col_desc.is_array || col_desc.is_record() {
1048        return Ok(None);
1049    }
1050
1051    let column_offset = dm_info.column_offsets[target_col_idx] as usize;
1052    let index_nr = dm_info.col_index_map[target_col_idx] as usize;
1053    if index_nr >= indices.len() {
1054        return Err(StorageError::FormatMismatch(format!(
1055            "SSM column '{}' references index {index_nr} but only {} indices exist",
1056            col_desc.col_name,
1057            indices.len()
1058        )));
1059    }
1060    let index = &indices[index_nr];
1061    let be = header.big_endian;
1062
1063    let mut requests: Vec<(usize, usize)> = selected_rows
1064        .iter()
1065        .copied()
1066        .enumerate()
1067        .map(|(out_idx, row_index)| (row_index, out_idx))
1068        .collect();
1069    requests.sort_unstable_by_key(|&(row_index, _)| row_index);
1070
1071    let mut cached_bucket_nr = None;
1072    let mut cached_bucket = Vec::new();
1073    let mut values = vec![None; selected_rows.len()];
1074    let (elem_bytes, _) = canonical_element_size(col_desc.data_type);
1075
1076    for (row_index, out_idx) in requests {
1077        let (bucket_nr, start_row, _end_row) =
1078            index.find_bucket(row_index as u64).ok_or_else(|| {
1079                StorageError::FormatMismatch(format!("SSM index has no bucket for row {row_index}"))
1080            })?;
1081        if cached_bucket_nr != Some(bucket_nr) {
1082            cached_bucket = read_bucket(&mut file, &header, bucket_nr)?;
1083            cached_bucket_nr = Some(bucket_nr);
1084        }
1085        let row_in_bucket = (row_index as u64 - start_row) as usize;
1086        let value = match col_desc.data_type {
1087            CasacoreDataType::TpBool => {
1088                let byte_offset = column_offset + row_in_bucket / 8;
1089                let bit_offset = row_in_bucket % 8;
1090                Value::Scalar(ScalarValue::Bool(
1091                    ((cached_bucket[byte_offset] >> bit_offset) & 1) != 0,
1092                ))
1093            }
1094            CasacoreDataType::TpString => {
1095                let ref_offset = column_offset + row_in_bucket * 12;
1096                let str_bucket = read_i32_canonical(&cached_bucket[ref_offset..], be);
1097                let str_offset = read_i32_canonical(&cached_bucket[ref_offset + 4..], be);
1098                let str_length = read_i32_canonical(&cached_bucket[ref_offset + 8..], be);
1099                let string = if str_length <= 8 {
1100                    String::from_utf8(
1101                        cached_bucket[ref_offset..ref_offset + str_length as usize].to_vec(),
1102                    )
1103                    .map_err(|err| StorageError::FormatMismatch(format!("invalid UTF-8: {err}")))?
1104                } else {
1105                    read_ssm_string(&mut file, &header, str_bucket, str_offset, str_length)?
1106                };
1107                Value::Scalar(ScalarValue::String(string))
1108            }
1109            _ => {
1110                let data_start = column_offset + row_in_bucket * elem_bytes;
1111                let data_end = data_start + elem_bytes;
1112                extract_scalar_from_ssm_bytes(
1113                    &cached_bucket[data_start..data_end],
1114                    col_desc.data_type,
1115                    be,
1116                )?
1117            }
1118        };
1119        values[out_idx] = Some(match value {
1120            Value::Scalar(scalar) => scalar,
1121            other => {
1122                return Err(StorageError::FormatMismatch(format!(
1123                    "SSM column '{}' expected scalar value, found {:?}",
1124                    col_desc.col_name,
1125                    other.kind()
1126                )));
1127            }
1128        });
1129    }
1130
1131    Ok(Some(values))
1132}

1134pub(crate) fn read_ssm_array_column_rows(

1137    col_descs: &[&ColumnDescContents],

1142        return Ok(Some(Vec::new()));

1148    let indices = parse_ssm_indices(&mut file, &header)?;
1149
1150    if target_col_idx >= col_descs.len() {
1151        return Err(StorageError::FormatMismatch(format!(
1152            "SSM array column index {target_col_idx} is out of range"
1153        )));
1154    }
1155    if target_col_idx >= dm_info.column_offsets.len()
1156        || target_col_idx >= dm_info.col_index_map.len()
1157    {
1158        return Err(StorageError::FormatMismatch(format!(
1159            "SSM array column index {target_col_idx} missing dm blob metadata"
1160        )));
1161    }
1162
1163    let col_desc = col_descs[target_col_idx];
1164    if !col_desc.is_array || col_desc.is_record() || is_ssm_variable_string(col_desc) {
1165        return Ok(None);
1166    }
1167
1168    let column_offset = dm_info.column_offsets[target_col_idx] as usize;
1169    let index_nr = dm_info.col_index_map[target_col_idx] as usize;
1170    if index_nr >= indices.len() {
1171        return Err(StorageError::FormatMismatch(format!(
1172            "SSM column '{}' references index {index_nr} but only {} indices exist",
1173            col_desc.col_name,
1174            indices.len()
1175        )));
1176    }
1177    let index = &indices[index_nr];
1178    let rows_to_read = selected_rows
1179        .iter()
1180        .copied()
1181        .max()
1182        .map(|row| row + 1)
1183        .unwrap_or(0);
1184
1185    if is_ssm_array_file_indirect(col_desc) {
1186        let offsets = read_column_from_buckets(
1187            &mut file,
1188            &header,
1189            index,
1190            column_offset,
1191            CasacoreDataType::TpInt64,
1192            1,
1193            rows_to_read,
1194        )?;
1195        let offset_values = match offsets {
1196            ColumnRawData::Int64(values) => values,
1197            _ => {
1198                return Err(StorageError::FormatMismatch(
1199                    "SSM indirect array: expected Int64 offsets".to_string(),
1200                ));
1201            }
1202        };
1203
1204        let dt = CasacoreDataType::from_primitive_type(col_desc.require_primitive_type()?, false);
1205        let mut array_path = file_path.as_os_str().to_os_string();
1206        array_path.push("i");
1207        let array_path = std::path::PathBuf::from(array_path);
1208        if !array_path.exists() {
1209            if selected_rows
1210                .iter()
1211                .all(|&row| offset_values.get(row).copied().unwrap_or(0) == 0)
1212            {
1213                return Ok(Some(vec![None; selected_rows.len()]));
1214            }
1215            return Err(StorageError::FormatMismatch(
1216                "SSM indirect array column but no array file found".to_string(),
1217            ));
1218        }
1219
1220        let mut reader = StManArrayFileReader::open(&array_path, header.big_endian)?;
1221        let mut values = Vec::with_capacity(selected_rows.len());
1222        for &row in selected_rows {
1223            let offset = *offset_values.get(row).ok_or_else(|| {
1224                StorageError::FormatMismatch(format!(
1225                    "SSM indirect array column '{}' missing offset for row {row}",
1226                    col_desc.col_name
1227                ))
1228            })?;
1229            let value = reader.read_array_at(offset, dt).map_err(|err| {
1230                StorageError::FormatMismatch(format!(
1231                    "SSM indirect array column '{}' row {row} offset {offset} type {:?}: {err}",
1232                    col_desc.col_name, dt
1233                ))
1234            })?;
1235            values.push(match value {
1236                Some(Value::Array(array)) => Some(array),
1237                Some(other) => {
1238                    return Err(StorageError::FormatMismatch(format!(
1239                        "SSM indirect array column '{}' row {row} expected array value, found {:?}",
1240                        col_desc.col_name,
1241                        other.kind()
1242                    )));
1243                }
1244                None => None,
1245            });
1246        }
1247        return Ok(Some(values));
1248    }
1249
1250    let nrelem = if col_desc.nrdim > 0 && !col_desc.shape.is_empty() {
1251        col_desc.shape.iter().map(|&s| s as usize).product()
1252    } else {
1253        return Ok(None);
1254    };
1255    let raw = read_column_from_buckets(
1256        &mut file,
1257        &header,
1258        index,
1259        column_offset,
1260        col_desc.data_type,
1261        nrelem,
1262        rows_to_read,
1263    )?;
1264    let mut values = Vec::with_capacity(selected_rows.len());
1265    for &row in selected_rows {
1266        let value = extract_row_value(&raw, col_desc, row, rows_to_read)?;
1267        values.push(match value {
1268            Value::Array(array) => Some(array),
1269            other => {
1270                return Err(StorageError::FormatMismatch(format!(
1271                    "SSM direct array column '{}' row {row} expected array value, found {:?}",
1272                    col_desc.col_name,
1273                    other.kind()
1274                )));
1275            }
1276        });
1277    }
1278    Ok(Some(values))
1279}

1281fn read_column_from_buckets(
1282    file: &mut File,
1283    header: &SsmHeader,
1284    index: &SsmIndex,
1285    column_offset: usize,
1286    data_type: CasacoreDataType,
1287    nrelem: usize,
1288    nrrow: usize,
1289) -> Result<ColumnRawData, StorageError> {
1290    let (elem_bytes, _) = canonical_element_size(data_type);
1291    let be = header.big_endian;
1292
1293    match data_type {
1294        CasacoreDataType::TpBool => {
1295            let mut values = Vec::with_capacity(nrrow * nrelem);
1296            let mut row = 0usize;
1297            while row < nrrow {
1298                let (bucket_nr, start_row, end_row) =
1299                    index.find_bucket(row as u64).ok_or_else(|| {
1300                        StorageError::FormatMismatch(format!(
1301                            "SSM index has no bucket for row {row}"
1302                        ))
1303                    })?;
1304                let bucket = read_bucket(file, header, bucket_nr)?;
1305                let row_in_bucket = (row as u64 - start_row) as usize;
1306                let rows_in_chunk =
1307                    ((end_row - start_row + 1) as usize - row_in_bucket).min(nrrow - row);
1308                let bit_offset = row_in_bucket * nrelem;
1309                let byte_offset = column_offset + bit_offset / 8;
1310                let sub_bit = bit_offset % 8;
1311                let bools = read_bool_bits(&bucket[byte_offset..], sub_bit, rows_in_chunk * nrelem);
1312                values.extend_from_slice(&bools);
1313                row += rows_in_chunk;
1314            }
1315            Ok(ColumnRawData::Bool(values))
1316        }
1317        CasacoreDataType::TpUChar => {
1318            let mut values = Vec::with_capacity(nrrow * nrelem);
1319            read_typed_column(
1320                file,
1321                header,
1322                index,
1323                column_offset,
1324                nrrow,
1325                nrelem,
1326                elem_bytes,
1327                |data, count| {
1328                    values.extend_from_slice(&data[..count]);
1329                    Ok(())
1330                },
1331            )?;
1332            Ok(ColumnRawData::UInt8(values))
1333        }
1334        CasacoreDataType::TpShort => {
1335            let mut values = Vec::with_capacity(nrrow * nrelem);
1336            read_typed_column(
1337                file,
1338                header,
1339                index,
1340                column_offset,
1341                nrrow,
1342                nrelem,
1343                elem_bytes,
1344                |data, count| {
1345                    values.extend(if be {
1346                        read_i16_slice_be(data, count)
1347                    } else {
1348                        read_i16_slice_le(data, count)
1349                    });
1350                    Ok(())
1351                },
1352            )?;
1353            Ok(ColumnRawData::Int16(values))
1354        }
1355        CasacoreDataType::TpUShort => {
1356            let mut values = Vec::with_capacity(nrrow * nrelem);
1357            read_typed_column(
1358                file,
1359                header,
1360                index,
1361                column_offset,
1362                nrrow,
1363                nrelem,
1364                elem_bytes,
1365                |data, count| {
1366                    values.extend(if be {
1367                        read_u16_slice_be(data, count)
1368                    } else {
1369                        read_u16_slice_le(data, count)
1370                    });
1371                    Ok(())
1372                },
1373            )?;
1374            Ok(ColumnRawData::UInt16(values))
1375        }
1376        CasacoreDataType::TpInt => {
1377            let mut values = Vec::with_capacity(nrrow * nrelem);
1378            read_typed_column(
1379                file,
1380                header,
1381                index,
1382                column_offset,
1383                nrrow,
1384                nrelem,
1385                elem_bytes,
1386                |data, count| {
1387                    values.extend(if be {
1388                        read_i32_slice_be(data, count)
1389                    } else {
1390                        read_i32_slice_le(data, count)
1391                    });
1392                    Ok(())
1393                },
1394            )?;
1395            Ok(ColumnRawData::Int32(values))
1396        }
1397        CasacoreDataType::TpUInt => {
1398            let mut values = Vec::with_capacity(nrrow * nrelem);
1399            read_typed_column(
1400                file,
1401                header,
1402                index,
1403                column_offset,
1404                nrrow,
1405                nrelem,
1406                elem_bytes,
1407                |data, count| {
1408                    values.extend(if be {
1409                        read_u32_slice_be(data, count)
1410                    } else {
1411                        read_u32_slice_le(data, count)
1412                    });
1413                    Ok(())
1414                },
1415            )?;
1416            Ok(ColumnRawData::UInt32(values))
1417        }
1418        CasacoreDataType::TpFloat => {
1419            let mut values = Vec::with_capacity(nrrow * nrelem);
1420            read_typed_column(
1421                file,
1422                header,
1423                index,
1424                column_offset,
1425                nrrow,
1426                nrelem,
1427                elem_bytes,
1428                |data, count| {
1429                    values.extend(if be {
1430                        read_f32_slice_be(data, count)
1431                    } else {
1432                        read_f32_slice_le(data, count)
1433                    });
1434                    Ok(())
1435                },
1436            )?;
1437            Ok(ColumnRawData::Float32(values))
1438        }
1439        CasacoreDataType::TpDouble => {
1440            let mut values = Vec::with_capacity(nrrow * nrelem);
1441            read_typed_column(
1442                file,
1443                header,
1444                index,
1445                column_offset,
1446                nrrow,
1447                nrelem,
1448                elem_bytes,
1449                |data, count| {
1450                    values.extend(if be {
1451                        read_f64_slice_be(data, count)
1452                    } else {
1453                        read_f64_slice_le(data, count)
1454                    });
1455                    Ok(())
1456                },
1457            )?;
1458            Ok(ColumnRawData::Float64(values))
1459        }
1460        CasacoreDataType::TpInt64 => {
1461            let mut values = Vec::with_capacity(nrrow * nrelem);
1462            read_typed_column(
1463                file,
1464                header,
1465                index,
1466                column_offset,
1467                nrrow,
1468                nrelem,
1469                elem_bytes,
1470                |data, count| {
1471                    values.extend(if be {
1472                        read_i64_slice_be(data, count)
1473                    } else {
1474                        read_i64_slice_le(data, count)
1475                    });
1476                    Ok(())
1477                },
1478            )?;
1479            Ok(ColumnRawData::Int64(values))
1480        }
1481        CasacoreDataType::TpComplex => {
1482            let mut values = Vec::with_capacity(nrrow * nrelem);
1483            read_typed_column(
1484                file,
1485                header,
1486                index,
1487                column_offset,
1488                nrrow,
1489                nrelem,
1490                elem_bytes,
1491                |data, count| {
1492                    for i in 0..count {
1493                        let (re, im) = if be {
1494                            (read_f32_be(&data[i * 8..]), read_f32_be(&data[i * 8 + 4..]))
1495                        } else {
1496                            (read_f32_le(&data[i * 8..]), read_f32_le(&data[i * 8 + 4..]))
1497                        };
1498                        values.push(casa_types::Complex32::new(re, im));
1499                    }
1500                    Ok(())
1501                },
1502            )?;
1503            Ok(ColumnRawData::Complex32(values))
1504        }
1505        CasacoreDataType::TpDComplex => {
1506            let mut values = Vec::with_capacity(nrrow * nrelem);
1507            read_typed_column(
1508                file,
1509                header,
1510                index,
1511                column_offset,
1512                nrrow,
1513                nrelem,
1514                elem_bytes,
1515                |data, count| {
1516                    for i in 0..count {
1517                        let (re, im) = if be {
1518                            (
1519                                read_f64_be(&data[i * 16..]),
1520                                read_f64_be(&data[i * 16 + 8..]),
1521                            )
1522                        } else {
1523                            (
1524                                read_f64_le(&data[i * 16..]),
1525                                read_f64_le(&data[i * 16 + 8..]),
1526                            )
1527                        };
1528                        values.push(casa_types::Complex64::new(re, im));
1529                    }
1530                    Ok(())
1531                },
1532            )?;
1533            Ok(ColumnRawData::Complex64(values))
1534        }
1535        CasacoreDataType::TpString => {
1536            // C++ SSM stores ONE 12-byte reference per row for strings.
1537            // Scalars (nrelem=1): short strings (<=8 bytes) inline, long in bucket.
1538            // Arrays (nrelem>1): all elements serialized as [BE_i32_len][data]
1539            // pairs in a string bucket, one reference per row.
1540            let mut values = Vec::with_capacity(nrrow * nrelem);
1541            for row in 0..nrrow {
1542                let (bucket_nr, start_row, _end_row) =
1543                    index.find_bucket(row as u64).ok_or_else(|| {
1544                        StorageError::FormatMismatch(format!(
1545                            "SSM index has no bucket for row {row}"
1546                        ))
1547                    })?;
1548                let bucket = read_bucket(file, header, bucket_nr)?;
1549                let row_in_bucket = (row as u64 - start_row) as usize;
1550                let ref_offset = column_offset + row_in_bucket * 12;
1551                let str_bucket = read_i32_canonical(&bucket[ref_offset..], be);
1552                let str_offset = read_i32_canonical(&bucket[ref_offset + 4..], be);
1553                let str_length = read_i32_canonical(&bucket[ref_offset + 8..], be);
1554
1555                if nrelem == 1 {
1556                    let s = if str_length <= 8 {
1557                        String::from_utf8(
1558                            bucket[ref_offset..ref_offset + str_length as usize].to_vec(),
1559                        )
1560                        .map_err(|e| StorageError::FormatMismatch(format!("invalid UTF-8: {e}")))?
1561                    } else {
1562                        read_ssm_string(file, header, str_bucket, str_offset, str_length)?
1563                    };
1564                    values.push(s);
1565                } else if str_length == 0 {
1566                    for _ in 0..nrelem {
1567                        values.push(String::new());
1568                    }
1569                } else {
1570                    let block =
1571                        read_ssm_string_bytes(file, header, str_bucket, str_offset, str_length)?;
1572                    let mut pos = 0usize;
1573                    for _ in 0..nrelem {
1574                        if pos + 4 > block.len() {
1575                            return Err(StorageError::FormatMismatch(
1576                                "SSM string array: block too short for length prefix".to_string(),
1577                            ));
1578                        }
1579                        let slen = u32::from_be_bytes([
1580                            block[pos],
1581                            block[pos + 1],
1582                            block[pos + 2],
1583                            block[pos + 3],
1584                        ]) as usize;
1585                        pos += 4;
1586                        if pos + slen > block.len() {
1587                            return Err(StorageError::FormatMismatch(
1588                                "SSM string array: block too short for string data".to_string(),
1589                            ));
1590                        }
1591                        let s =
1592                            String::from_utf8(block[pos..pos + slen].to_vec()).map_err(|e| {
1593                                StorageError::FormatMismatch(format!("invalid UTF-8: {e}"))
1594                            })?;
1595                        pos += slen;
1596                        values.push(s);
1597                    }
1598                }
1599            }
1600            Ok(ColumnRawData::String(values))
1601        }
1602        _ => Err(StorageError::FormatMismatch(format!(
1603            "unsupported SSM data type: {data_type:?}"
1604        ))),
1605    }
1606}

1608/// Iterate over data buckets and process column data for non-Bool, non-String types.
1609#[allow(clippy::too_many_arguments)]
1610fn read_typed_column<F>(
1611    file: &mut File,
1612    header: &SsmHeader,
1613    index: &SsmIndex,
1614    column_offset: usize,
1615    nrrow: usize,
1616    nrelem: usize,
1617    elem_bytes: usize,
1618    mut process: F,
1619) -> Result<(), StorageError>
1620where
1621    F: FnMut(&[u8], usize) -> Result<(), StorageError>,
1622{
1623    let bytes_per_row = elem_bytes * nrelem;
1624    let mut row = 0usize;
1625
1626    while row < nrrow {
1627        let (bucket_nr, start_row, end_row) = index.find_bucket(row as u64).ok_or_else(|| {
1628            StorageError::FormatMismatch(format!("SSM index has no bucket for row {row}"))
1629        })?;
1630        let bucket = read_bucket(file, header, bucket_nr)?;
1631        let row_in_bucket = (row as u64 - start_row) as usize;
1632        let rows_in_chunk = ((end_row - start_row + 1) as usize - row_in_bucket).min(nrrow - row);
1633        let data_start = column_offset + row_in_bucket * bytes_per_row;
1634        process(&bucket[data_start..], rows_in_chunk * nrelem)?;
1635        row += rows_in_chunk;
1636    }
1637
1638    Ok(())
1639}

1714fn write_ssm_file_impl(
1715    file_path: &Path,
1716    col_descs: &[ColumnDescContents],
1717    rows: &[casa_types::RecordValue],
1718    field_indices: Option<&[usize]>,
1719    scalar_columns: Option<&[&[Option<ScalarValue>]]>,
1720    big_endian: bool,
1721) -> Result<Vec<u8>, StorageError> {
1722    let nrrow = scalar_columns
1723        .and_then(|columns| columns.first().map(|values| values.len()))
1724        .unwrap_or(rows.len());
1725    let ncol = col_descs.len();
1726
1727    // Check if any array columns are stored indirectly.
1728    let has_indirect = col_descs.iter().any(is_ssm_indirect);
1729
1730    // Create shared array file for indirect columns (SSM uses version 0, no refcount).
1731    let mut array_writer = if has_indirect {
1732        let mut arr_path = file_path.as_os_str().to_os_string();
1733        arr_path.push("i");
1734        Some(StManArrayFileWriter::create(
1735            &std::path::PathBuf::from(arr_path),
1736            big_endian,
1737            0,
1738        )?)
1739    } else {
1740        None
1741    };
1742
1743    // 1. Compute column sizes in bits (for Bool bit-packing)
1744    // For indirect columns: 8 bytes (i64 offset) per row.
1745    // String columns always use 96 bits (12 bytes) per row for a 3-int reference,
1746    // regardless of array element count — all string data goes to string buckets.
1747    let col_sizes_bits: Vec<usize> = col_descs
1748        .iter()
1749        .map(|c| {
1750            if is_ssm_variable_string(c) {
1751                96 // 12 bytes = 3 ints (bucketNr, offset, length) per row
1752            } else if is_ssm_indirect(c) {
1753                64 // i64 offset = 8 bytes = 64 bits
1754            } else {
1755                let nrelem = if c.is_array && !c.shape.is_empty() {
1756                    c.shape.iter().map(|&s| s as usize).product()
1757                } else {
1758                    1
1759                };
1760                let (_, bits) = canonical_element_size(c.data_type);
1761                nrelem * bits
1762            }
1763        })
1764        .collect();
1765
1766    // 2. Compute rows_per_bucket and bucket_size
1767    let total_bits_per_row: usize = col_sizes_bits.iter().sum();
1768    let total_bytes_per_row = total_bits_per_row.div_ceil(8);
1769    let rows_per_bucket = if total_bytes_per_row == 0 {
1770        32u32
1771    } else {
1772        // Start with 32 rows/bucket (SSM default), adjust to fit
1773        let mut rpb = 32u32;
1774        loop {
1775            let size: usize = col_sizes_bits
1776                .iter()
1777                .map(|&bits| (rpb as usize * bits).div_ceil(8))
1778                .sum();
1779            let next_size: usize = col_sizes_bits
1780                .iter()
1781                .map(|&bits| ((rpb as usize + 1) * bits).div_ceil(8))
1782                .sum();
1783            if next_size > 128.max(size) {
1784                break;
1785            }
1786            rpb += 1;
1787        }
1788        rpb
1789    };
1790
1791    let bucket_size: u32 = {
1792        let data_size: usize = col_sizes_bits
1793            .iter()
1794            .map(|&bits| (rows_per_bucket as usize * bits).div_ceil(8))
1795            .sum();
1796        data_size.max(128) as u32
1797    };
1798
1799    // 3. Compute column offsets within a data bucket
1800    let mut column_offsets = vec![0u32; ncol];
1801    let mut offset = 0u32;
1802    for (i, &bits) in col_sizes_bits.iter().enumerate() {
1803        column_offsets[i] = offset;
1804        offset += (rows_per_bucket as usize * bits).div_ceil(8) as u32;
1805    }
1806
1807    // 4. Build data buckets
1808    let nr_data_buckets = if nrrow == 0 {
1809        0
1810    } else {
1811        nrrow.div_ceil(rows_per_bucket as usize)
1812    };
1813
1814    let mut buckets: Vec<Vec<u8>> = (0..nr_data_buckets)
1815        .map(|_| vec![0u8; bucket_size as usize])
1816        .collect();
1817
1818    // String handling: collect long strings into string buckets
1819    let mut string_buckets: Vec<Vec<u8>> = Vec::new();
1820
1821    for (col_idx, col_desc) in col_descs.iter().enumerate() {
1822        let is_indirect = is_ssm_indirect(col_desc);
1823        let nrelem = if is_indirect {
1824            1usize
1825        } else if col_desc.is_array && !col_desc.shape.is_empty() {
1826            col_desc.shape.iter().map(|&s| s as usize).product()
1827        } else {
1828            1usize
1829        };
1830        let col_off = column_offsets[col_idx] as usize;
1831
1832        if is_indirect {
1833            // Write array data to shared file, store i64 offsets in buckets.
1834            let dt = if col_desc.is_record() {
1835                CasacoreDataType::TpUChar
1836            } else {
1837                CasacoreDataType::from_primitive_type(col_desc.require_primitive_type()?, false)
1838            };
1839            let writer = array_writer.as_mut().ok_or_else(|| {
1840                StorageError::FormatMismatch(
1841                    "no array file writer for indirect SSM column".to_string(),
1842                )
1843            })?;
1844
1845            for row in 0..nrrow {
1846                let bucket_idx = row / rows_per_bucket as usize;
1847                let row_in_bucket = row % rows_per_bucket as usize;
1848                let bucket = &mut buckets[bucket_idx];
1849                let byte_off = col_off + row_in_bucket * 8;
1850
1851                let scalar_value;
1852                let value = if let Some(columns) = scalar_columns {
1853                    scalar_value = columns[col_idx][row].as_ref().cloned().map(Value::Scalar);
1854                    scalar_value.as_ref()
1855                } else {
1856                    let row_record = &rows[row];
1857                    row_record
1858                        .fields()
1859                        .get(
1860                            field_indices
1861                                .and_then(|indices| indices.get(col_idx))
1862                                .copied()
1863                                .unwrap_or(usize::MAX),
1864                        )
1865                        .filter(|field| field.name == col_desc.col_name)
1866                        .map(|field| &field.value)
1867                        .or_else(|| {
1868                            row_record
1869                                .fields()
1870                                .iter()
1871                                .find(|f| f.name == col_desc.col_name)
1872                                .map(|f| &f.value)
1873                        })
1874                };
1875
1876                let offset: i64 =
1877                    match value {
1878                        Some(Value::Record(record)) if col_desc.is_record() => {
1879                            let bytes = super::table_control::serialize_record_to_uchar(record)?;
1880                            let payload = Value::Array(ArrayValue::UInt8(
1881                            ArrayD::from_shape_vec(IxDyn(&[bytes.len()]), bytes).map_err(|err| {
1882                                StorageError::FormatMismatch(format!(
1883                                    "SSM record column '{}': invalid byte payload shape: {err}",
1884                                    col_desc.col_name
1885                                ))
1886                            })?,
1887                        ));
1888                            writer.write_array(&payload, dt)?
1889                        }
1890                        Some(val @ Value::Array(_)) if !col_desc.is_record() => {
1891                            writer.write_array(val, dt)?
1892                        }
1893                        Some(Value::Array(_)) | Some(Value::Record(_)) => {
1894                            return Err(StorageError::FormatMismatch(format!(
1895                                "SSM indirect column '{}' received incompatible value type",
1896                                col_desc.col_name
1897                            )));
1898                        }
1899                        _ => 0i64, // undefined cell
1900                    };
1901
1902                // Write i64 offset in canonical byte order.
1903                if big_endian {
1904                    write_i64_be(&mut bucket[byte_off..], offset);
1905                } else {
1906                    write_i64_le(&mut bucket[byte_off..], offset);
1907                }
1908            }
1909        } else {
1910            for row in 0..nrrow {
1911                let bucket_idx = row / rows_per_bucket as usize;
1912                let row_in_bucket = row % rows_per_bucket as usize;
1913                let bucket = &mut buckets[bucket_idx];
1914
1915                let scalar_value;
1916                let value = if let Some(columns) = scalar_columns {
1917                    scalar_value = columns[col_idx][row].as_ref().cloned().map(Value::Scalar);
1918                    scalar_value.as_ref()
1919                } else {
1920                    let row_record = &rows[row];
1921                    row_record
1922                        .fields()
1923                        .get(
1924                            field_indices
1925                                .and_then(|indices| indices.get(col_idx))
1926                                .copied()
1927                                .unwrap_or(usize::MAX),
1928                        )
1929                        .filter(|field| field.name == col_desc.col_name)
1930                        .map(|field| &field.value)
1931                        .or_else(|| {
1932                            row_record
1933                                .fields()
1934                                .iter()
1935                                .find(|f| f.name == col_desc.col_name)
1936                                .map(|f| &f.value)
1937                        })
1938                };
1939
1940                write_cell_to_bucket(
1941                    bucket,
1942                    col_off,
1943                    row_in_bucket,
1944                    col_desc,
1945                    col_desc.data_type,
1946                    nrelem,
1947                    value,
1948                    &mut string_buckets,
1949                    nr_data_buckets,
1950                    bucket_size as usize,
1951                    big_endian,
1952                );
1953            }
1954        }
1955    }
1956
1957    // Flush the array file header and buffer if we wrote indirect arrays.
1958    if let Some(ref mut w) = array_writer {
1959        w.finish()?;
1960    }
1961
1962    // 5. Build the SSMIndex
1963    let mut last_row = Vec::with_capacity(nr_data_buckets);
1964    let mut bucket_number = Vec::with_capacity(nr_data_buckets);
1965    for i in 0..nr_data_buckets {
1966        let end_row = ((i + 1) * rows_per_bucket as usize - 1).min(nrrow - 1);
1967        last_row.push(end_row as u64);
1968        bucket_number.push(i as u32);
1969    }
1970
1971    // 6. Serialize the SSMIndex to bytes (byte order matches table endian)
1972    let index_data = serialize_ssm_index(
1973        nr_data_buckets as u32,
1974        rows_per_bucket,
1975        ncol as i32,
1976        &last_row,
1977        &bucket_number,
1978        big_endian,
1979    );
1980
1981    // 7. Write the index into index buckets
1982    let chain_overhead = 8usize; // check number + next bucket pointer
1983    let data_per_index_bucket = bucket_size as usize - chain_overhead;
1984    let nr_idx_buckets = if index_data.is_empty() {
1985        0
1986    } else {
1987        index_data.len().div_ceil(data_per_index_bucket)
1988    };
1989
1990    // String bucket count (each goes after data buckets, before index buckets)
1991    let nr_string_buckets = string_buckets.len();
1992    let first_string_bucket = if nr_string_buckets > 0 {
1993        nr_data_buckets as i32
1994    } else {
1995        -1i32
1996    };
1997
1998    let first_idx_bucket = (nr_data_buckets + nr_string_buckets) as i32;
1999    let nr_buckets = nr_data_buckets + nr_string_buckets + nr_idx_buckets;
2000
2001    // 8. Write the 512-byte header (BE AipsIO framing, v3)
2002    let header_buf = serialize_ssm_header(
2003        bucket_size,
2004        nr_buckets as u32,
2005        0,  // nr_free_buckets
2006        -1, // first_free_bucket
2007        nr_idx_buckets as u32,
2008        first_idx_bucket,
2009        0, // idx_bucket_offset
2010        first_string_bucket,
2011        index_data.len() as u32,
2012        1, // nr_indices (single index for all columns)
2013        big_endian,
2014    );
2015
2016    // 9. Assemble the file
2017    let mut file = File::create(file_path)?;
2018    file.write_all(&header_buf)?;
2019
2020    // Data buckets
2021    for bucket in &buckets {
2022        file.write_all(bucket)?;
2023    }
2024
2025    // String buckets
2026    for sb in &string_buckets {
2027        file.write_all(sb)?;
2028    }
2029
2030    // Index buckets
2031    let mut idx_remaining = &index_data[..];
2032    for i in 0..nr_idx_buckets {
2033        let mut idx_bucket = vec![0u8; bucket_size as usize];
2034        // Check number at offset 0 (canonical/big-endian)
2035        idx_bucket[0..4].copy_from_slice(&0i32.to_be_bytes());
2036        // Next bucket pointer at offset 4 (canonical/big-endian)
2037        let next = if i + 1 < nr_idx_buckets {
2038            (first_idx_bucket as usize + i + 1) as i32
2039        } else {
2040            -1i32
2041        };
2042        idx_bucket[4..8].copy_from_slice(&next.to_be_bytes());
2043        // Copy index data
2044        let chunk = idx_remaining.len().min(data_per_index_bucket);
2045        idx_bucket[chain_overhead..chain_overhead + chunk].copy_from_slice(&idx_remaining[..chunk]);
2046        idx_remaining = &idx_remaining[chunk..];
2047        file.write_all(&idx_bucket)?;
2048    }
2049
2050    // 10. Generate the DM data blob (always big-endian, file-based AipsIO)
2051    let dm_blob = serialize_ssm_dm_blob("SSM", &column_offsets, ncol)?;
2052
2053    Ok(dm_blob)
2054}

2056/// Write a single cell value into a data bucket using the specified byte order.
2057#[allow(clippy::too_many_arguments)]
2058fn write_cell_to_bucket(
2059    bucket: &mut [u8],
2060    col_offset: usize,
2061    row_in_bucket: usize,
2062    col_desc: &ColumnDescContents,
2063    data_type: CasacoreDataType,
2064    nrelem: usize,
2065    value: Option<&casa_types::Value>,
2066    string_buckets: &mut Vec<Vec<u8>>,
2067    string_bucket_base: usize,
2068    bucket_size: usize,
2069    big_endian: bool,
2070) {
2071    use casa_types::{ArrayValue, ScalarValue, Value};
2072
2073    match data_type {
2074        CasacoreDataType::TpBool => {
2075            let bit_offset = row_in_bucket * nrelem;
2076            let byte_offset = col_offset + bit_offset / 8;
2077            let sub_bit = bit_offset % 8;
2078            let bools: Vec<bool> = match value {
2079                Some(Value::Scalar(ScalarValue::Bool(b))) => vec![*b],
2080                _ => vec![false; nrelem],
2081            };
2082            write_bool_bits(&mut bucket[byte_offset..], sub_bit, &bools);
2083        }
2084        CasacoreDataType::TpString => {
2085            // C++ SSM stores ONE 12-byte reference (bucketNr, offset, length)
2086            // per row, regardless of nrelem. For scalars (nrelem=1), short
2087            // strings (<=8 bytes) are inlined in the reference slot. For arrays
2088            // or long scalars, all element data goes to a string bucket as
2089            // [BE_i32_len][string_bytes] pairs. String bucket metadata is BE.
2090            let ref_offset = col_offset + row_in_bucket * 12;
2091
2092            if !col_desc.is_array {
2093                // Scalar string
2094                let s = match value {
2095                    Some(Value::Scalar(ScalarValue::String(s))) => s.as_str(),
2096                    _ => "",
2097                };
2098                let len = s.len() as i32;
2099                if len <= 8 {
2100                    bucket[ref_offset..ref_offset + s.len()].copy_from_slice(s.as_bytes());
2101                    if big_endian {
2102                        write_i32_be(&mut bucket[ref_offset + 8..], len);
2103                    } else {
2104                        write_i32_le(&mut bucket[ref_offset + 8..], len);
2105                    }
2106                } else {
2107                    let (sb_nr, sb_offset) =
2108                        allocate_string_in_bucket(string_buckets, s.as_bytes(), bucket_size);
2109                    let abs_bucket = (string_bucket_base + sb_nr) as i32;
2110                    if big_endian {
2111                        write_i32_be(&mut bucket[ref_offset..], abs_bucket);
2112                        write_i32_be(&mut bucket[ref_offset + 4..], sb_offset as i32);
2113                        write_i32_be(&mut bucket[ref_offset + 8..], len);
2114                    } else {
2115                        write_i32_le(&mut bucket[ref_offset..], abs_bucket);
2116                        write_i32_le(&mut bucket[ref_offset + 4..], sb_offset as i32);
2117                        write_i32_le(&mut bucket[ref_offset + 8..], len);
2118                    }
2119                }
2120            } else {
2121                // String arrays also use one 12-byte locator per row.
2122                // Fixed-shape arrays store only the element data; variable-shape
2123                // arrays prefix ndim, shape, and a filled flag.
2124                let (shape, strings) = match value {
2125                    Some(Value::Array(ArrayValue::String(arr))) => (
2126                        arr.shape().to_vec(),
2127                        arr.t().iter().cloned().collect::<Vec<_>>(),
2128                    ),
2129                    _ => {
2130                        let shape = if col_desc.shape.is_empty() {
2131                            Vec::new()
2132                        } else {
2133                            col_desc.shape.iter().map(|&dim| dim as usize).collect()
2134                        };
2135                        let count = if shape.is_empty() {
2136                            0
2137                        } else {
2138                            shape.iter().product()
2139                        };
2140                        (shape, vec![String::new(); count])
2141                    }
2142                };
2143
2144                let mut block = Vec::new();
2145                if col_desc.shape.is_empty() {
2146                    block.extend_from_slice(&(shape.len() as u32).to_be_bytes());
2147                    for &dim in &shape {
2148                        block.extend_from_slice(&(dim as i32).to_be_bytes());
2149                    }
2150                    let filled = if strings.iter().any(|s| !s.is_empty()) {
2151                        1u32
2152                    } else {
2153                        0u32
2154                    };
2155                    block.extend_from_slice(&filled.to_be_bytes());
2156                }
2157                for s in &strings {
2158                    block.extend_from_slice(&(s.len() as u32).to_be_bytes());
2159                    block.extend_from_slice(s.as_bytes());
2160                }
2161
2162                if block.is_empty() {
2163                    // All empty strings — leave reference as zeroes
2164                } else {
2165                    let (sb_nr, sb_offset) =
2166                        allocate_string_in_bucket(string_buckets, &block, bucket_size);
2167                    let abs_bucket = (string_bucket_base + sb_nr) as i32;
2168                    let total = block.len() as i32;
2169                    if big_endian {
2170                        write_i32_be(&mut bucket[ref_offset..], abs_bucket);
2171                        write_i32_be(&mut bucket[ref_offset + 4..], sb_offset as i32);
2172                        write_i32_be(&mut bucket[ref_offset + 8..], total);
2173                    } else {
2174                        write_i32_le(&mut bucket[ref_offset..], abs_bucket);
2175                        write_i32_le(&mut bucket[ref_offset + 4..], sb_offset as i32);
2176                        write_i32_le(&mut bucket[ref_offset + 8..], total);
2177                    }
2178                }
2179            }
2180        }
2181        _ => {
2182            // Fixed-size types
2183            let (elem_bytes, _) = canonical_element_size(data_type);
2184            let bytes_per_row = elem_bytes * nrelem;
2185            let data_start = col_offset + row_in_bucket * bytes_per_row;
2186            write_value_canonical(
2187                &mut bucket[data_start..],
2188                data_type,
2189                nrelem,
2190                value,
2191                big_endian,
2192            );
2193        }
2194    }

2412/// Allocate space for a string in string buckets. Returns (bucket_nr, offset).
2413fn allocate_string_in_bucket(
2414    string_buckets: &mut Vec<Vec<u8>>,
2415    data: &[u8],
2416    bucket_size: usize,
2417) -> (usize, usize) {
2418    // String bucket layout: 4 canonical (BE) ints (freeLink, usedLength, nDeleted, nextBucket) + data
2419    let header_size = 16;
2420    let data_capacity = bucket_size - header_size;
2421
2422    // Try to fit in the last bucket
2423    if let Some(last) = string_buckets.last_mut() {
2424        let used_len = i32::from_be_bytes([last[4], last[5], last[6], last[7]]) as usize;
2425        if used_len + data.len() <= data_capacity {
2426            let offset = used_len;
2427            last[header_size + offset..header_size + offset + data.len()].copy_from_slice(data);
2428            let new_used = (used_len + data.len()) as i32;
2429            last[4..8].copy_from_slice(&new_used.to_be_bytes());
2430            return (string_buckets.len() - 1, offset);
2431        }
2432    }
2433
2434    // Allocate a new string bucket
2435    let mut bucket = vec![0u8; bucket_size];
2436    // freeLink = -1
2437    bucket[0..4].copy_from_slice(&(-1i32).to_be_bytes());
2438    // usedLength = data.len()
2439    bucket[4..8].copy_from_slice(&(data.len() as i32).to_be_bytes());
2440    // nDeleted = 0
2441    bucket[8..12].copy_from_slice(&0i32.to_be_bytes());
2442    // nextBucket = -1 (will be updated if chaining needed)
2443    bucket[12..16].copy_from_slice(&(-1i32).to_be_bytes());
2444    // Write data
2445    bucket[header_size..header_size + data.len()].copy_from_slice(data);
2446
2447    let bucket_nr = string_buckets.len();
2448    string_buckets.push(bucket);
2449    (bucket_nr, 0)
2450}

2452/// Serialize the SSM header into a 512-byte buffer.
2453///
2454/// The AipsIO framing byte order matches the table's endian setting:
2455/// `CanonicalIO` (big-endian) for BE tables, `LECanonicalIO` (little-endian)
2456/// for LE tables. C++ casacore creates the corresponding TypeIO based on
2457/// `asBigEndian()` when reading this header. The `asBigEndian` flag inside
2458/// the header controls bucket data byte order.
2459#[allow(clippy::too_many_arguments)]
2460fn serialize_ssm_header(
2461    bucket_size: u32,
2462    nr_buckets: u32,
2463    nr_free_buckets: u32,
2464    first_free_bucket: i32,
2465    nr_idx_buckets: u32,
2466    first_idx_bucket: i32,
2467    idx_bucket_offset: u32,
2468    last_string_bucket: i32,
2469    index_length: u32,
2470    nr_indices: u32,
2471    big_endian: bool,
2472) -> Vec<u8> {
2473    // AipsIO framing uses the table's byte order (C++ creates CanonicalIO or
2474    // LECanonicalIO based on asBigEndian()).
2475    let io_order = if big_endian {
2476        ByteOrder::BigEndian
2477    } else {
2478        ByteOrder::LittleEndian
2479    };
2480    let mut buf = AipsIoWriteBuf::new(io_order);
2481    buf.putstart("StandardStMan", 3);
2482    buf.put_bool(big_endian);
2483    buf.put_u32(bucket_size);
2484    buf.put_u32(nr_buckets);
2485    buf.put_u32(0); // persCacheSize
2486    buf.put_u32(nr_free_buckets);
2487    buf.put_i32(first_free_bucket);
2488    buf.put_u32(nr_idx_buckets);
2489    buf.put_i32(first_idx_bucket);
2490    buf.put_u32(idx_bucket_offset);
2491    buf.put_i32(last_string_bucket);
2492    buf.put_u32(index_length);
2493    buf.put_u32(nr_indices);
2494    buf.putend();
2495
2496    // Pad to 512 bytes
2497    let mut header = buf.into_bytes();
2498    header.resize(SSM_HEADER_SIZE as usize, 0);
2499    header
2500}

2502/// Serialize SSMIndex to bytes using the table's byte order for AipsIO framing.
2503fn serialize_ssm_index(
2504    n_used: u32,
2505    rows_per_bucket: u32,
2506    nr_columns: i32,
2507    last_row: &[u64],
2508    bucket_number: &[u32],
2509    big_endian: bool,
2510) -> Vec<u8> {
2511    let io_order = if big_endian {
2512        ByteOrder::BigEndian
2513    } else {
2514        ByteOrder::LittleEndian
2515    };
2516    let mut buf = AipsIoWriteBuf::new(io_order);
2517    // SSMIndex is a top-level object (level 0 → has magic)
2518    buf.putstart("SSMIndex", 2);
2519    buf.put_u32(n_used);
2520    buf.put_u32(rows_per_bucket);
2521    buf.put_i32(nr_columns);
2522    // freeSpace map: empty SimpleOrderedMap
2523    buf.putstart_nested("SimpleOrderedMap", 1);
2524    buf.put_i32(0); // default_value
2525    buf.put_u32(0); // nr entries
2526    buf.put_u32(0); // incr
2527    buf.putend_nested();
2528    // lastRow: Block<Int64> (version 2)
2529    buf.putstart_nested("Block", 1);
2530    buf.put_u32(last_row.len() as u32);
2531    for &v in last_row {
2532        buf.put_u64(v);
2533    }
2534    buf.putend_nested();
2535    // bucketNumber: Block<uInt>
2536    buf.putstart_nested("Block", 1);
2537    buf.put_u32(bucket_number.len() as u32);
2538    for &v in bucket_number {
2539        buf.put_u32(v);
2540    }
2541    buf.putend_nested();
2542    buf.putend();
2543    buf.into_bytes()
2544}

2546/// Serialize the DM data blob ("SSM" v2) using file-based AipsIO (always BE).
2547fn serialize_ssm_dm_blob(
2548    name: &str,
2549    column_offsets: &[u32],
2550    ncol: usize,
2551) -> Result<Vec<u8>, StorageError> {
2552    let mut io = AipsIo::new_write_only(std::io::Cursor::new(Vec::new()));
2553    io.putstart("SSM", 2)?;
2554    io.put_string(name)?;
2555    // putBlock for column_offsets
2556    write_block_u32(&mut io, column_offsets)?;
2557    // putBlock for col_index_map (all zeros → single index for all columns)
2558    let col_index_map = vec![0u32; ncol];
2559    write_block_u32(&mut io, &col_index_map)?;
2560    io.putend()?;
2561    let cursor: std::io::Cursor<Vec<u8>> = io.into_inner_typed()?;
2562    Ok(cursor.into_inner())
2563}

2580/// Minimal writer for in-memory AipsIO in the given byte order.
2581struct AipsIoWriteBuf {
2582    data: Vec<u8>,
2583    order: ByteOrder,
2584    /// Stack of obj_len placeholder positions for putend backpatching.
2585    len_positions: Vec<usize>,
2586    level: usize,
2587}
2588
2589impl AipsIoWriteBuf {
2590    fn new(order: ByteOrder) -> Self {
2591        Self {
2592            data: Vec::new(),
2593            order,
2594            len_positions: Vec::new(),
2595            level: 0,
2596        }
2597    }
2598
2599    fn into_bytes(self) -> Vec<u8> {
2600        self.data
2601    }
2602
2603    fn put_u8(&mut self, val: u8) {
2604        self.data.push(val);
2605    }
2606
2607    fn put_u32(&mut self, val: u32) {
2608        match self.order {
2609            ByteOrder::BigEndian => self.data.extend_from_slice(&val.to_be_bytes()),
2610            ByteOrder::LittleEndian => self.data.extend_from_slice(&val.to_le_bytes()),
2611        }
2612    }
2613
2614    fn put_i32(&mut self, val: i32) {
2615        match self.order {
2616            ByteOrder::BigEndian => self.data.extend_from_slice(&val.to_be_bytes()),
2617            ByteOrder::LittleEndian => self.data.extend_from_slice(&val.to_le_bytes()),
2618        }
2619    }
2620
2621    fn put_u64(&mut self, val: u64) {
2622        match self.order {
2623            ByteOrder::BigEndian => self.data.extend_from_slice(&val.to_be_bytes()),
2624            ByteOrder::LittleEndian => self.data.extend_from_slice(&val.to_le_bytes()),
2625        }
2626    }
2627
2628    fn put_bool(&mut self, val: bool) {
2629        self.put_u8(if val { 1 } else { 0 });
2630    }
2631
2632    fn put_string(&mut self, s: &str) {
2633        self.put_u32(s.len() as u32);
2634        self.data.extend_from_slice(s.as_bytes());
2635    }
2636
2637    /// Begin a top-level AipsIO object (writes magic at level 0).
2638    fn putstart(&mut self, type_name: &str, version: u32) {
2639        // Magic only at level 0
2640        self.put_u32(AIPSIO_MAGIC);
2641        // Placeholder for obj_len (will be backpatched in putend)
2642        let pos = self.data.len();
2643        self.put_u32(0); // placeholder
2644        self.len_positions.push(pos);
2645        self.put_string(type_name);
2646        self.put_u32(version);
2647        self.level += 1;
2648    }
2649
2650    /// Begin a nested AipsIO object (no magic).
2651    fn putstart_nested(&mut self, type_name: &str, version: u32) {
2652        let pos = self.data.len();
2653        self.put_u32(0); // placeholder for obj_len
2654        self.len_positions.push(pos);
2655        self.put_string(type_name);
2656        self.put_u32(version);
2657        self.level += 1;
2658    }
2659
2660    /// End the current object, backpatch obj_len.
2661    fn putend(&mut self) {
2662        if let Some(pos) = self.len_positions.pop() {
2663            // obj_len includes everything from the obj_len field to here
2664            let obj_len = (self.data.len() - pos) as u32;
2665            let bytes = match self.order {
2666                ByteOrder::BigEndian => obj_len.to_be_bytes(),
2667                ByteOrder::LittleEndian => obj_len.to_le_bytes(),
2668            };
2669            self.data[pos..pos + 4].copy_from_slice(&bytes);
2670        }
2671        if self.level > 0 {
2672            self.level -= 1;
2673        }
2674    }
2675
2676    /// End a nested object (same as putend, alias for clarity).
2677    fn putend_nested(&mut self) {
2678        self.putend();
2679    }
2680}
