5fn public_weighting_diagnostics(
6    weighting: WeightingMode,
7    weight_density_mode: WeightDensityMode,
8    uv_taper: Option<GaussianUvTaper>,
9    trace: weighting::WeightingTraceInternal,

12        weighting,
13        weight_density_mode,

19                batch_index: sample.batch_index,
20                sample_index: sample.sample_index,
21                u_lambda: sample.u_lambda,
22                v_lambda: sample.v_lambda,
23                w_lambda: sample.w_lambda,
24                input_weight: sample.input_weight,
25                density_weight: sample.density_weight,
26                output_weight: sample.output_weight,
27                sumwt_factor: sample.sumwt_factor,
28                gridable: sample.gridable,
29                normalization_contribution: sample.normalization_contribution,
30                reported_contribution: sample.reported_contribution,
31            })
32            .collect(),
33        gridded_samples: trace.gridded_samples,
34        skipped_samples: trace.skipped_samples,
35        normalization_sumwt: trace.normalization_sumwt,
36        reported_sumwt: trace.reported_sumwt,

103/// Trace the explicit weighting seam for one MFS imaging request.
104///
105/// This exposes the final imaging weight assigned to each scalar sample, along
106/// with the separate normalization and persisted-`sumwt` accumulators that the
107/// CASA-compatible dirty path uses downstream.
108pub fn trace_weighting(request: &ImagingRequest) -> Result<WeightingDiagnostics, ImagingError> {
109    request.validate()?;
110    let gridder = StandardGridder::new(request.geometry)?;
111    let trace = trace_weighting_with_density_source(
112        request.weighting,
113        WeightDensityMode::Combined,
114        None,
115        weighting::fractional_bandwidth_from_frequency_range(request.selected_frequency_range_hz),
116        &request.visibility_batches,
117        &request.visibility_batches,
118        &gridder,
119    )?;
120    Ok(public_weighting_diagnostics(
121        request.weighting,
122        WeightDensityMode::Combined,
123        None,
124        trace,
125    ))
126}

128fn public_w_project_diagnostics(prepared: WProjectPreparedData) -> WProjectDiagnostics {
129    let kernels = (0..prepared.projector.plane_count())
130        .map(|plane_index| WProjectKernelDiagnostics {
131            plane_index,
132            w_lambda: prepared.projector.kernel_w_lambda(plane_index),
133            support: prepared.projector.kernel_support(plane_index),
134            kernel_integral: prepared.projector.kernel_integral(plane_index),
135        })
136        .collect();
137    let samples = prepared
138        .samples
139        .into_iter()
140        .map(|sample| WProjectSamplePlanDiagnostics {
141            batch_index: sample.batch_index,
142            sample_index: sample.sample_index,
143            u_lambda: sample.u_lambda,
144            v_lambda: sample.v_lambda,
145            w_lambda: sample.w_lambda,
146            weight: sample.weight,
147            sumwt_factor: sample.sumwt_factor,
148            plane_index: sample.positive_plan.plane_index,
149            loc_x: sample.positive_plan.loc_x,
150            loc_y: sample.positive_plan.loc_y,
151            off_x: sample.positive_plan.off_x,
152            off_y: sample.positive_plan.off_y,
153            conjugate_kernel: sample.positive_plan.conjugate_kernel,
154            normalization: sample.positive_plan.normalization,
155            support: prepared
156                .projector
157                .kernel_support(sample.positive_plan.plane_index),
158        })
159        .collect();
160    let skipped_samples = prepared
161        .skipped_samples
162        .into_iter()
163        .map(|sample| WProjectSkippedSampleDiagnostics {
164            batch_index: sample.batch_index,
165            sample_index: sample.sample_index,
166            u_lambda: sample.u_lambda,
167            v_lambda: sample.v_lambda,
168            w_lambda: sample.w_lambda,
169            weight: sample.weight,
170            sumwt_factor: sample.sumwt_factor,
171            reason: sample.reason,
172        })
173        .collect();
174    WProjectDiagnostics {
175        requested_plane_count: prepared.requested_plane_count,
176        plane_count: prepared.projector.plane_count(),
177        sampling: prepared.projector.sampling(),
178        w_scale: prepared.projector.w_scale(),
179        max_abs_w_lambda: prepared.max_abs_w_lambda,
180        kernels,
181        samples,
182        skipped_samples,
183        normalization_sumwt: prepared.normalization_sumwt,
184        reported_sumwt: prepared.reported_sumwt,
185        gridded_samples: prepared.gridded_samples,
186    }
187}

189/// Trace the explicit `wproject` CF/grid-planning seam for one imaging plane.
190pub fn trace_w_project_plan(request: &ImagingRequest) -> Result<WProjectDiagnostics, ImagingError> {
191    request.validate()?;
192    if request.w_term_mode != WTermMode::WProject {
193        return Err(ImagingError::InvalidRequest(
194            "trace_w_project_plan requires w_term_mode='wproject'".to_string(),
195        ));
196    }
197    let gridder = StandardGridder::new(request.geometry)?;
198    let weighted = apply_weighting(request, &gridder, 1)?;
199    let prepared = prepare_w_project_data(
200        request.geometry,
201        &weighted,
202        &gridder,
203        request.w_project_planes,
204        1,
205    )?;
206    Ok(public_w_project_diagnostics(prepared))
207}

209/// Trace the standard major-cycle residual-refresh seam for one imaging plane.
210///
211/// This applies the normal weighting/PSF path and then exposes the predicted
212/// visibilities, residual visibilities, and refreshed residual image for the
213/// supplied model. The current trace surface is limited to standard 2-D
214/// imaging (`w_term_mode = None`).
215pub fn trace_residual_refresh(
216    request: &ImagingRequest,
217    model: &Array2<f32>,
218    execution: &ImagingExecutionPlan,
219) -> Result<ResidualRefreshDiagnostics, ImagingError> {
220    request.validate()?;
221    if model.dim()
222        != (
223            request.geometry.image_shape[0],
224            request.geometry.image_shape[1],
225        )
226    {
227        return Err(ImagingError::InvalidRequest(format!(
228            "residual-refresh trace model shape {:?} does not match image geometry {:?}",
229            model.dim(),
230            request.geometry.image_shape,
231        )));
232    }
233    if request.w_term_mode != WTermMode::None {
234        return Err(ImagingError::Unsupported(
235            "trace_residual_refresh currently supports only standard 2-D imaging".to_string(),
236        ));
237    }
238    let gridder = StandardGridder::new(request.geometry)?;
239    let mut stage_timings = ImagingStageTimings::default();
240    let weighting_started = Instant::now();
241    let weighted_batches = apply_weighting(request, &gridder, 1)?;
242    stage_timings.weighting += weighting_started.elapsed();
243    let psf_state = compute_psf(
244        request,
245        &weighted_batches,
246        &gridder,
247        execution,
248        &mut stage_timings,
249    )?;
250    let trace = compute_residual_trace_standard(
251        request.geometry,
252        &weighted_batches,
253        &gridder,
254        model,
255        &psf_state,
256        false,
257        &mut stage_timings,
258    )?;
259    Ok(public_residual_refresh_diagnostics(trace))
260}

262/// Trace the standard major-cycle residual-refresh seam for one cube plane.
263///
264/// Unlike [`trace_residual_refresh`], this uses explicit per-sample cube
265/// interpolation state so predicted visibilities can draw from neighboring
266/// model planes in the same way CASA's cube major cycle does. The trace surface
267/// is currently limited to standard 2-D imaging (`w_term_mode = None`).
268pub fn trace_cube_channel_residual_refresh(
269    request: &ImagingRequest,
270    model_interpolation_batches: &[CubeModelInterpolationBatch],
271    identity_model_channel_index: usize,
272    model_planes: &[Array2<f32>],
273    model_channel_frequencies_hz: &[f64],
274    execution: &ImagingExecutionPlan,
275) -> Result<ResidualRefreshDiagnostics, ImagingError> {
276    trace_cube_channel_residual_refresh_with_mode(
277        request,
278        model_interpolation_batches,
279        identity_model_channel_index,
280        model_planes,
281        model_channel_frequencies_hz,
282        execution,
283        CubePredictionLambdaMode::OutputChannel,
284    )
285}

311fn trace_cube_channel_residual_refresh_with_mode(
312    request: &ImagingRequest,
313    model_interpolation_batches: &[CubeModelInterpolationBatch],
314    identity_model_channel_index: usize,
315    model_planes: &[Array2<f32>],
316    model_channel_frequencies_hz: &[f64],
317    execution: &ImagingExecutionPlan,
318    prediction_lambda_mode: CubePredictionLambdaMode,
319) -> Result<ResidualRefreshDiagnostics, ImagingError> {
320    request.validate()?;
321    if request.w_term_mode != WTermMode::None {
322        return Err(ImagingError::Unsupported(
323            "trace_cube_channel_residual_refresh currently supports only standard 2-D imaging"
324                .to_string(),
325        ));
326    }
327    if model_planes.len() != model_channel_frequencies_hz.len() {
328        return Err(ImagingError::InvalidRequest(format!(
329            "cube residual-refresh trace model plane count {} does not match model frequency count {}",
330            model_planes.len(),
331            model_channel_frequencies_hz.len()
332        )));
333    }
334    let expected_shape = (
335        request.geometry.image_shape[0],
336        request.geometry.image_shape[1],
337    );
338    for (model_channel_index, model_plane) in model_planes.iter().enumerate() {
339        if model_plane.dim() != expected_shape {
340            return Err(ImagingError::InvalidRequest(format!(
341                "cube residual-refresh trace model plane {model_channel_index} shape {:?} does not match image geometry {:?}",
342                model_plane.dim(),
343                request.geometry.image_shape
344            )));
345        }
346    }
347    if identity_model_channel_index >= model_planes.len() {
348        return Err(ImagingError::InvalidRequest(format!(
349            "cube residual-refresh trace identity channel index {identity_model_channel_index} is out of range for {} model planes",
350            model_planes.len()
351        )));
352    }
353    let gridder = StandardGridder::new(request.geometry)?;
354    let mut stage_timings = ImagingStageTimings::default();
355    let weighting_started = Instant::now();
356    let weighted_batches = apply_weighting(request, &gridder, 1)?;
357    stage_timings.weighting += weighting_started.elapsed();
358    let psf_state = compute_psf(
359        request,
360        &weighted_batches,
361        &gridder,
362        execution,
363        &mut stage_timings,
364    )?;
365    let trace = compute_residual_trace_cube_standard(
366        &weighted_batches,
367        model_interpolation_batches,
368        &gridder,
369        model_planes,
370        identity_model_channel_index,
371        request.reffreq_hz,
372        model_channel_frequencies_hz,
373        prediction_lambda_mode,
374        &psf_state,
375        &mut stage_timings,
376    )?;
377    Ok(public_residual_refresh_diagnostics(trace))
