3//! Project-scoped MCP server used by external coding-agent backends.
4//!
5//! The server deliberately exposes typed CASA projections instead of a second
6//! shell or filesystem API. Each call must carry the per-session nonce supplied
7//! by the host when this process is launched.

30#[derive(Debug, Clone, PartialEq, Eq)]
31struct ProjectToolError {
32    code: i64,
33    message: String,
34}
35
36impl ProjectToolError {
37    fn new(code: i64, message: impl Into<String>) -> Self {
38        Self {
39            code,
40            message: message.into(),
41        }
42    }
43}
44
45type ProjectToolResult<T> = Result<T, ProjectToolError>;
46
47fn main() {
48    if let Err(error) = run() {
49        eprintln!("casars-project-mcp: {error}");
50        std::process::exit(1);
51    }
52}

54fn run() -> Result<(), String> {
55    let (project_root, nonce) = parse_args()?;
56    let stdin = io::stdin();
57    let mut stdout = io::stdout().lock();
58    for line in stdin.lock().lines() {
59        let line = line.map_err(|error| format!("read request: {error}"))?;
60        if line.trim().is_empty() {
61            continue;
62        }
63        let request: Value =
64            serde_json::from_str(&line).map_err(|error| format!("parse request: {error}"))?;
65        if let Some(response) = handle(&project_root, &nonce, &request) {
66            serde_json::to_writer(&mut stdout, &response)
67                .map_err(|error| format!("serialize response: {error}"))?;
68            stdout
69                .write_all(b"\n")
70                .and_then(|()| stdout.flush())
71                .map_err(|error| format!("write response: {error}"))?;
72        }
73    }
74    Ok(())
75}

77fn parse_args() -> Result<(PathBuf, String), String> {
78    let mut project_root = None;
79    let mut nonce = None;
80    let mut arguments = env::args().skip(1);
81    while let Some(argument) = arguments.next() {
82        match argument.as_str() {
83            "--project-root" => project_root = arguments.next().map(PathBuf::from),
84            "--nonce" => nonce = arguments.next(),
85            other => return Err(format!("unrecognized argument {other}")),
86        }
87    }
88    let project_root = project_root.ok_or("missing --project-root")?;
89    if !project_root.is_absolute() || !project_root.is_dir() {
90        return Err("--project-root must be an existing absolute directory".to_owned());
91    }
92    let nonce = nonce.ok_or("missing --nonce")?;
93    if nonce.len() < 24 {
94        return Err("--nonce must contain at least 24 characters".to_owned());
95    }
96    Ok((project_root, nonce))
97}

99fn handle(project_root: &Path, nonce: &str, request: &Value) -> Option<Value> {
100    let id = request.get("id")?.clone();
101    let method = request.get("method").and_then(Value::as_str).unwrap_or("");
102    let result = match method {
103        "initialize" => Ok(json!({
104            "protocolVersion": PROTOCOL_VERSION,
105            "capabilities": {"tools": {"listChanged": false}},
106            "serverInfo": {"name": "casa-rs-project", "version": env!("CARGO_PKG_VERSION")}
107        })),
108        "ping" => Ok(json!({})),
109        "tools/list" => Ok(json!({"tools": tool_definitions()})),
110        "tools/call" => call_tool(project_root, nonce, request),
111        _ => Err(ProjectToolError::new(
112            -32601,
113            format!("unsupported MCP method {method}"),
114        )),
115    };
116    Some(match result {
117        Ok(result) => json!({"jsonrpc": "2.0", "id": id, "result": result}),
118        Err(error) => {
119            json!({"jsonrpc": "2.0", "id": id, "error": {"code": error.code, "message": error.message}})
120        }
121    })
122}

130#[derive(Debug, Clone, PartialEq, Eq)]
131enum ProjectToolArguments {
132    Empty,
133    Search(SearchArguments),

135    TaskSuggestion {

141type ProjectToolDecoder = fn(&Map<String, Value>) -> ProjectToolResult<ProjectToolArguments>;
142type ProjectToolHandler = fn(&Path, ProjectToolArguments) -> ProjectToolResult<Value>;
143
144struct ProjectToolSpec {
145    name: &'static str,
146    description: &'static str,
147    properties: fn() -> Map<String, Value>,
148    required: &'static [&'static str],
149    requires_context: bool,
150    decode: ProjectToolDecoder,
151    handler: ProjectToolHandler,
152}
153
154impl ProjectToolSpec {
155    fn definition(&self) -> Value {
156        let mut properties = (self.properties)();
157        properties.insert(
158            "nonce".to_owned(),
159            json!({
160                "type": "string",
161                "description": "Exact per-session CASA project nonce supplied in the agent instructions"
162            }),
163        );
164        let required = self
165            .required
166            .iter()
167            .copied()
168            .chain(std::iter::once("nonce"))
169            .collect::<Vec<_>>();
170        json!({
171            "name": self.name,
172            "description": self.description,
173            "inputSchema": {
174                "type": "object",
175                "properties": properties,
176                "required": required,
177                "additionalProperties": false
178            }
179        })
180    }
181}

183fn decode_empty(_: &Map<String, Value>) -> ProjectToolResult<ProjectToolArguments> {
184    Ok(ProjectToolArguments::Empty)
185}

187fn decode_search(arguments: &Map<String, Value>) -> ProjectToolResult<ProjectToolArguments> {
188    Ok(ProjectToolArguments::Search(SearchArguments {
189        query: required_string(arguments, "query")?.to_owned(),
190        limit: arguments
191            .get("limit")
192            .and_then(Value::as_u64)
193            .ok_or_else(|| ProjectToolError::new(-32602, "search requires integer limit"))?,
194    }))
195}
196
197fn decode_task_id(arguments: &Map<String, Value>) -> ProjectToolResult<ProjectToolArguments> {
198    Ok(ProjectToolArguments::TaskId(
199        required_string(arguments, "task_id")?.to_owned(),
200    ))
201}
202
203fn decode_task_suggestion(
204    arguments: &Map<String, Value>,
205) -> ProjectToolResult<ProjectToolArguments> {
206    let task_id = required_string(arguments, "task_id")?.to_owned();
207    let parameters = arguments
208        .get("parameters")
209        .and_then(Value::as_object)
210        .ok_or_else(|| ProjectToolError::new(-32602, "task.suggest requires string parameters"))?
211        .iter()
212        .map(|(name, value)| {
213            value
214                .as_str()
215                .map(|value| (name.clone(), value.to_owned()))
216                .ok_or_else(|| {
217                    ProjectToolError::new(-32602, "task.suggest parameter values must be strings")
218                })
219        })
220        .collect::<ProjectToolResult<std::collections::BTreeMap<_, _>>>()?;
221    Ok(ProjectToolArguments::TaskSuggestion {
222        task_id,
223        parameters,
224    })
225}

351fn call_tool(project_root: &Path, nonce: &str, request: &Value) -> ProjectToolResult<Value> {
352    let params = request
353        .get("params")
354        .and_then(Value::as_object)
355        .ok_or_else(|| ProjectToolError::new(-32602, "tools/call requires object params"))?;
356    let name = params.get("name").and_then(Value::as_str).unwrap_or("");
357    let arguments = params
358        .get("arguments")
359        .and_then(Value::as_object)
360        .ok_or_else(|| ProjectToolError::new(-32602, "tools/call requires object arguments"))?;
361    if arguments.get("nonce").and_then(Value::as_str) != Some(nonce) {
362        return Err(ProjectToolError::new(-32001, "CASA project nonce mismatch"));
363    }
364    let tool = project_tools()
365        .into_iter()
366        .find(|tool| tool.name == name)
367        .ok_or_else(|| {
368            ProjectToolError::new(-32602, format!("unknown CASA project tool {name}"))
369        })?;
370    if tool.requires_context {
371        validate_context_projection(project_root, nonce)?;
372    }
373    let arguments = (tool.decode)(arguments)?;
374    let output = (tool.handler)(project_root, arguments)?;
375    let text = serde_json::to_string(&output).map_err(|error| {
376        ProjectToolError::new(-32000, format!("serialize tool result: {error}"))
377    })?;
378    Ok(json!({"content": [{"type": "text", "text": text}]}))
379}

391fn planned_search(
392    project_root: &Path,
393    arguments: SearchArguments,
394    layers: Vec<String>,
395) -> ProjectToolResult<Vec<casars_frontend_services::AssistantCorpusSearchHitState>> {
396    let SearchArguments { query, limit } = arguments;
397    let plan = read_resource_plan(project_root)?;
398    let chunk_target = u64::try_from(CORPUS_INDEX_CHUNK_TARGET_BYTES)
399        .map_err(|_| ProjectToolError::new(-32003, "corpus chunk target is not representable"))?;
400    let planned_hit_limit = plan.corpus_text_units / chunk_target;
401    if limit > planned_hit_limit {
402        return Err(ProjectToolError::new(
403            -32602,
404            format!(
405                "search limit {limit} exceeds the current resource-plan limit {planned_hit_limit}"
406            ),
407        ));
408    }
409    let hits = assistant_corpus_search(AssistantCorpusSearchRequest {
410        project_root: project_root.display().to_string(),
411        query,
412        limit,
413        layers,
414    })
415    .map_err(frontend_error)?;
416    fit_search_hits_to_budget(hits, plan.corpus_text_units)
417}

419fn fit_search_hits_to_budget(
420    hits: Vec<casars_frontend_services::AssistantCorpusSearchHitState>,
421    budget: u64,
422) -> ProjectToolResult<Vec<casars_frontend_services::AssistantCorpusSearchHitState>> {
423    let budget = usize::try_from(budget)
424        .map_err(|_| ProjectToolError::new(-32003, "corpus result budget is not representable"))?;
425    let mut fitted = Vec::new();
426    for hit in hits {
427        let mut candidate = fitted.clone();
428        candidate.push(hit.clone());
429        if measured_search_result_units(&candidate)? <= budget {
430            fitted.push(hit);
431            continue;
432        }
433
434        let boundaries = std::iter::once(0)
435            .chain(hit.text.char_indices().skip(1).map(|(index, _)| index))
436            .chain(std::iter::once(hit.text.len()))
437            .collect::<Vec<_>>();
438        let mut low = 0_usize;
439        let mut high = boundaries.len();
440        while low < high {
441            let middle = low + (high - low) / 2;
442            let mut truncated = hit.clone();
443            truncated.text = hit.text[..boundaries[middle]].to_owned();
444            let mut measured = fitted.clone();
445            measured.push(truncated);
446            if measured_search_result_units(&measured)? <= budget {
447                low = middle + 1;
448            } else {
449                high = middle;
450            }
451        }
452        if low > 0 {
453            let mut truncated = hit;
454            truncated.text = truncated.text[..boundaries[low - 1]].to_owned();
455            fitted.push(truncated);
456        }
457        // A truncated final hit consumes the remaining result envelope.
458        if !fitted.is_empty() {
459            break;
460        }
461    }
462    if fitted.is_empty() {
463        return Err(ProjectToolError::new(
464            -32003,
465            "the resource plan cannot fit one corpus citation and its metadata",
466        ));
467    }
468    Ok(fitted)
469}

471fn measured_search_result_units(
472    hits: &[casars_frontend_services::AssistantCorpusSearchHitState],
473) -> ProjectToolResult<usize> {
474    serde_json::to_vec(hits)

593fn read_context_projection(
594    project_root: &Path,
595) -> ProjectToolResult<AssistantContextProjectionState> {
596    let path = project_root.join(".casa-rs/assistant-context.json");
597    let bytes = fs::read(&path).map_err(|error| {
598        ProjectToolError::new(-32002, format!("read {}: {error}", path.display()))
599    })?;
600    let projection: AssistantContextProjectionState =
601        serde_json::from_slice(&bytes).map_err(|error| {
602            ProjectToolError::new(-32002, format!("parse {}: {error}", path.display()))
603        })?;
604    if projection.schema_version != 1 {
605        return Err(ProjectToolError::new(
606            -32002,
607            format!(
608                "unsupported context projection schema {}",
609                projection.schema_version
610            ),
611        ));
612    }
613    Ok(projection)
614}

616fn read_resource_plan(
617    project_root: &Path,
618) -> ProjectToolResult<AssistantContextResourcePlanProjection> {
619    let plan = read_context_projection(project_root)?.resource_plan;
620    if plan.schema_version != 1 {
621        return Err(ProjectToolError::new(
622            -32002,
623            format!(
624                "unsupported context resource-plan schema {}",
625                plan.schema_version
626            ),
627        ));
628    }
629    if plan.corpus_text_units == 0 {
630        let detail = if plan.diagnostics.is_empty() {
631            "the planner allocated no corpus capacity".to_owned()
632        } else {
633            plan.diagnostics.join("; ")
634        };
635        return Err(ProjectToolError::new(-32002, detail));
636    }
637    Ok(plan)
638}

640fn validate_context_projection(project_root: &Path, nonce: &str) -> ProjectToolResult<()> {
641    let projection = read_context_projection(project_root)?;
642    if projection.session_nonce != nonce {
643        return Err(ProjectToolError::new(
644            -32002,
645            "context projection is stale for the current CASA session nonce",
646        ));
647    }
648    Ok(())
649}
