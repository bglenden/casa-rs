11const STANDARD_MFS_PROFILE_BLOCK_DETAIL_ENV: &str = "CASA_RS_STANDARD_MFS_PROFILE_BLOCK_DETAIL";

13/// Return true when detailed standard-MFS profiling lines should be emitted.
14pub(crate) fn standard_mfs_profile_detail_enabled() -> bool {
15    static ENABLED: LazyLock<bool> =
16        LazyLock::new(|| env_flag_enabled(STANDARD_MFS_PROFILE_DETAIL_ENV));
17    *ENABLED
18}
19
20/// Return true when row-block level standard-MFS profiling lines should be emitted.
21pub(crate) fn standard_mfs_profile_block_detail_enabled() -> bool {
22    static ENABLED: LazyLock<bool> = LazyLock::new(|| {
23        standard_mfs_profile_detail_enabled()
24            && env_flag_enabled(STANDARD_MFS_PROFILE_BLOCK_DETAIL_ENV)
25    });
26    *ENABLED
27}
28
29fn env_flag_enabled(name: &str) -> bool {
30    env::var(name)
31        .map(|value| {
32            !matches!(
33                value.trim().to_ascii_lowercase().as_str(),
34                "0" | "false" | "no" | "off"
35            )
36        })
37        .unwrap_or(false)
38}

40/// Return a timestamp only when detailed profiling is enabled.
41pub(crate) fn maybe_profile_now() -> Option<Instant> {
42    standard_mfs_profile_detail_enabled().then(Instant::now)
43}
44
45/// Return elapsed time for an optional profiling timestamp.
46pub(crate) fn elapsed_since(started_at: Option<Instant>) -> Duration {
47    started_at.map_or(Duration::ZERO, |started_at| started_at.elapsed())
48}

50/// Return elapsed time in milliseconds.
51pub(crate) fn millis(duration: Duration) -> f64 {
52    duration.as_secs_f64() * 1_000.0
53}

55/// Return min/p50/max for a possibly empty `usize` sample.
56pub(crate) fn min_p50_max_usize(values: &[usize]) -> (usize, usize, usize) {
57    if values.is_empty() {
58        return (0, 0, 0);
59    }
60    let mut values = values.to_vec();
61    values.sort_unstable();
62    let mid = values.len() / 2;
63    (values[0], values[mid], values[values.len() - 1])
64}
65
66/// Return min/p50/max for a possibly empty duration sample.
67pub(crate) fn min_p50_max_duration(values: &[Duration]) -> (Duration, Duration, Duration) {
68    if values.is_empty() {
69        return (Duration::ZERO, Duration::ZERO, Duration::ZERO);
70    }
71    let mut values = values.to_vec();
72    values.sort_unstable();
73    let mid = values.len() / 2;
74    (values[0], values[mid], values[values.len() - 1])
75}

77/// Profile summary for one parallel standard-MFS stage.
78pub(crate) struct ParallelStageProfile<'a> {
79    /// Stage name.
80    pub(crate) stage: &'a str,
81    /// Requested worker count before batch/sample and hardware caps.
82    pub(crate) requested_threads: usize,
83    /// Actual worker count used.
84    pub(crate) actual_threads: usize,
85    /// Chunking unit, for example `batch` or `planned_sample`.
86    pub(crate) chunking: &'a str,
87    /// Chunk length in chunking units.
88    pub(crate) chunk_len: usize,
89    /// Total sample count represented by all chunks.
90    pub(crate) samples_total: usize,
91    /// Samples assigned to each worker.
92    pub(crate) samples_per_worker: Vec<usize>,
93    /// Bytes of local grid storage per worker.
94    pub(crate) local_grid_bytes_per_worker: usize,
95    /// Number of local grids allocated per worker.
96    pub(crate) local_grid_count: usize,
97    /// Per-worker local allocation and zero-fill duration.
98    pub(crate) local_alloc_zero_by_worker: Vec<Duration>,
99    /// Per-worker compute duration.
100    pub(crate) worker_compute_by_worker: Vec<Duration>,
101    /// Time spent joining workers.
102    pub(crate) join_duration: Duration,
103    /// Time spent merging local grids or worker outputs.
104    pub(crate) merge_duration: Duration,
105    /// Total stage duration.
106    pub(crate) stage_duration: Duration,
107}
108
109/// Emit one structured parallel-stage profiling line when enabled.
110pub(crate) fn log_parallel_stage(profile: ParallelStageProfile<'_>) {
111    if !standard_mfs_profile_block_detail_enabled() {
112        return;
113    }
114    let (samples_min, samples_p50, samples_max) = min_p50_max_usize(&profile.samples_per_worker);
115    let (alloc_min, alloc_p50, alloc_max) =
116        min_p50_max_duration(&profile.local_alloc_zero_by_worker);
117    let (compute_min, compute_p50, compute_max) =
118        min_p50_max_duration(&profile.worker_compute_by_worker);
119    let alloc_total = profile
120        .local_alloc_zero_by_worker
121        .iter()
122        .copied()
123        .sum::<Duration>();
124    eprintln!(
125        "standard_mfs_parallel_stage stage={} requested_threads={} actual_threads={} chunking={} chunk_len={} samples_total={} samples_per_worker_min={} samples_per_worker_p50={} samples_per_worker_max={} local_grid_bytes_per_worker={} local_grid_count={} local_alloc_zero_total_ms={:.3} local_alloc_zero_min_ms={:.3} local_alloc_zero_p50_ms={:.3} local_alloc_zero_max_ms={:.3} worker_compute_min_ms={:.3} worker_compute_p50_ms={:.3} worker_compute_max_ms={:.3} join_ms={:.3} merge_ms={:.3} stage_total_ms={:.3}",
126        profile.stage,
127        profile.requested_threads,
128        profile.actual_threads,
129        profile.chunking,
130        profile.chunk_len,
131        profile.samples_total,
132        samples_min,
133        samples_p50,
134        samples_max,
135        profile.local_grid_bytes_per_worker,
136        profile.local_grid_count,
137        millis(alloc_total),
138        millis(alloc_min),
139        millis(alloc_p50),
140        millis(alloc_max),
141        millis(compute_min),
142        millis(compute_p50),
143        millis(compute_max),
144        millis(profile.join_duration),
145        millis(profile.merge_duration),
146        millis(profile.stage_duration),
147    );
148}

150/// Profile summary for one worker inside a parallel standard-MFS stage.
151pub(crate) struct ParallelWorkerProfile<'a> {
152    /// Stage name.
153    pub(crate) stage: &'a str,
154    /// Stable worker index in chunk order.
155    pub(crate) worker_index: usize,
156    /// Raw samples assigned to this worker.
157    pub(crate) samples: usize,
158    /// Samples accepted for the stage's main useful work.
159    pub(crate) accepted_samples: usize,
160    /// Samples with finite visibilities.
161    pub(crate) finite_visibility_samples: usize,
162    /// Samples skipped because the visibility was not finite.
163    pub(crate) nonfinite_visibility_samples: usize,
164    /// Samples skipped because the row/channel was not gridable.
165    pub(crate) skipped_not_gridable: usize,
166    /// Samples skipped because the input weight was unusable.
167    pub(crate) skipped_invalid_weight: usize,
168    /// Samples skipped because the sum-weight factor was unusable.
169    pub(crate) skipped_invalid_sumwt: usize,
170    /// Samples skipped because the density value was unusable.
171    pub(crate) skipped_invalid_density: usize,
172    /// Samples skipped because the grid or density lookup was out of bounds.
173    pub(crate) skipped_out_of_grid: usize,
174    /// Estimated degrid tap visits.
175    pub(crate) degrid_tap_visits: usize,
176    /// Estimated grid tap visits.
177    pub(crate) grid_tap_visits: usize,
178    /// Density-grid cell updates performed by weighting stages.
179    pub(crate) density_cell_hits: usize,
180    /// Per-worker local allocation and zero-fill duration.
181    pub(crate) local_alloc_zero: Duration,
182    /// Per-worker compute duration.
183    pub(crate) worker_compute: Duration,
184}

186/// Emit one structured per-worker profiling line when enabled.

193        profile.stage,
194        profile.worker_index,
195        profile.samples,
196        profile.accepted_samples,
197        profile.finite_visibility_samples,
198        profile.nonfinite_visibility_samples,
199        profile.skipped_not_gridable,
200        profile.skipped_invalid_weight,
201        profile.skipped_invalid_sumwt,
202        profile.skipped_invalid_density,
203        profile.skipped_out_of_grid,
204        profile.degrid_tap_visits,
205        profile.grid_tap_visits,
206        profile.density_cell_hits,
207        millis(profile.local_alloc_zero),
208        millis(profile.worker_compute),
209    );
210}
