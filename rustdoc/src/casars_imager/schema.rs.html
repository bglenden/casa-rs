863struct OptionArgumentConfig<'a> {
864    id: &'a str,
865    label: &'a str,
866    order: usize,
867    flags: &'a [&'a str],
868    metavar: &'a str,
869    value_kind: UiValueKind,
870    default: Option<&'a str>,
871    choices: &'a [&'a str],
872    help: &'a str,
873    group: &'a str,
874    required: bool,
875    advanced: bool,
876}
877
878struct ToggleArgumentConfig<'a> {
879    id: &'a str,
880    label: &'a str,
881    order: usize,
882    true_flags: &'a [&'a str],
883    false_flags: &'a [&'a str],
884    default: Option<&'a str>,
885    help: &'a str,
886    group: &'a str,
887    advanced: bool,
888}
889
890fn option_argument(config: OptionArgumentConfig<'_>) -> UiArgumentSchema {
891    UiArgumentSchema {
892        id: config.id.to_string(),
893        label: config.label.to_string(),
894        order: config.order,
895        parser: UiArgumentParser::Option {
896            flags: config
897                .flags
898                .iter()
899                .map(|flag| (*flag).to_string())
900                .collect(),
901            metavar: config.metavar.to_string(),
902            choices: config
903                .choices
904                .iter()
905                .map(|choice| (*choice).to_string())
906                .collect(),
907        },
908        value_kind: config.value_kind,
909        required: config.required,
910        default: config.default.map(ToString::to_string),
911        help: config.help.to_string(),
912        group: config.group.to_string(),
913        advanced: config.advanced,
914        hidden_in_tui: false,
915    }
916}
917
918fn toggle_argument(config: ToggleArgumentConfig<'_>) -> UiArgumentSchema {
919    UiArgumentSchema {
920        id: config.id.to_string(),
921        label: config.label.to_string(),
922        order: config.order,
923        parser: UiArgumentParser::Toggle {
924            true_flags: config
925                .true_flags
926                .iter()
927                .map(|flag| (*flag).to_string())
928                .collect(),
929            false_flags: config
930                .false_flags
931                .iter()
932                .map(|flag| (*flag).to_string())
933                .collect(),
934        },
935        value_kind: UiValueKind::Bool,
936        required: false,
937        default: config.default.map(ToString::to_string),
938        help: config.help.to_string(),
939        group: config.group.to_string(),
940        advanced: config.advanced,
941        hidden_in_tui: false,
942    }
943}

945fn action_argument(
946    order: usize,
947    id: &str,
948    flags: &[&str],
949    action: UiActionKind,
950) -> UiArgumentSchema {
951    UiArgumentSchema {
952        id: id.to_string(),
953        label: id.to_string(),
954        order,
955        parser: UiArgumentParser::Action {
956            flags: flags.iter().map(|flag| (*flag).to_string()).collect(),
957            action,
958        },
959        value_kind: UiValueKind::None,
960        required: false,
961        default: None,
962        help: match action {
963            UiActionKind::Help => "Print this help message".to_string(),
964            UiActionKind::UiSchema => "Print the machine-readable UI schema".to_string(),
965        },
966        group: "Meta".to_string(),
967        advanced: false,
968        hidden_in_tui: true,
969    }
