2//! Canonical calibration-table keyword and column names.
3//!
4//! These constants mirror the modern CASA `NewCalTable` family closely enough
5//! for the first compatibility wave. The reader remains permissive, but later
6//! writer work will use these names to emit canonical on-disk layout.
7
8/// Expected `table.info` type for CASA calibration tables.
9pub const TABLE_INFO_TYPE: &str = "Calibration";
10
11/// Table keyword naming the parameter family.
12pub const KEY_PAR_TYPE: &str = "ParType";
13/// Table keyword naming the source MeasurementSet.
14pub const KEY_MS_NAME: &str = "MSName";
15/// Table keyword naming the calibration subtype (for example `G Jones`).
16pub const KEY_VIS_CAL: &str = "VisCal";
17/// Table keyword naming the polarization basis.
18pub const KEY_POL_BASIS: &str = "PolBasis";
19/// Table keyword naming the CASA writer version.
