2//! Record serialization for measures, compatible with C++ `MeasureHolder`.
3//!
4//! C++ casacore stores measures in `TableRecord` values with a specific field
5//! layout. This module provides `to_record` / `from_record` functions that
6//! produce and consume [`RecordValue`] in the same format, ensuring
7//! interoperability with C++ tables.
8//!
9// TODO: The record functions depend on `RecordValue` / `Value` / `ScalarValue` from
10// the tables layer. When `casacore-tables` is split into its own crate these types
11// will need a shared dependency (e.g. a `casacore-record` crate) so that
12// `casacore-types` does not pull in table-storage details.
13//!
14//! ## MEpoch record format
15//!
16//! ```text
17//! type:  "epoch"
18//! refer: "UTC"          (EpochRef::as_str())
19//! m0:    { value: <f64>, unit: "d" }
20//! ```
21//!
22//! ## MPosition record format
23//!
24//! ```text
25//! type:  "position"
26//! refer: "ITRF"
27//! m0:    { value: <lon_rad>, unit: "rad" }
28//! m1:    { value: <lat_rad>, unit: "rad" }
29//! m2:    { value: <radius_m>, unit: "m" }
30//! ```
31//!
32//! Note: C++ always serializes MPosition in spherical form (lon, lat, radius)
33//! regardless of reference type.
34

45/// Creates a quantity-like sub-record `{ value: <v>, unit: <u> }`.
46fn quantity_record(value: f64, unit: &str) -> RecordValue {
47    RecordValue::new(vec![
48        RecordField::new("value", Value::Scalar(ScalarValue::Float64(value))),
49        RecordField::new("unit", Value::Scalar(ScalarValue::String(unit.to_owned()))),
50    ])
51}
52
53/// Extracts an f64 from a quantity sub-record.
54fn extract_quantity_value(rec: &RecordValue) -> Result<f64, MeasureError> {
55    match rec.get("value") {
56        Some(Value::Scalar(ScalarValue::Float64(v))) => Ok(*v),
57        _ => Err(MeasureError::InvalidRecord {
58            reason: "quantity sub-record missing 'value' field".to_owned(),
59        }),
60    }
61}
62
63/// Extracts a string field from a record.
64fn extract_string(rec: &RecordValue, field: &str) -> Result<String, MeasureError> {
65    match rec.get(field) {
66        Some(Value::Scalar(ScalarValue::String(s))) => Ok(s.clone()),
67        _ => Err(MeasureError::InvalidRecord {
68            reason: format!("missing or non-string field '{field}'"),
69        }),
70    }
71}
72
73/// Extracts a sub-record from a record field.
74fn extract_sub_record<'a>(
75    rec: &'a RecordValue,
76    field: &str,
77) -> Result<&'a RecordValue, MeasureError> {
78    match rec.get(field) {
79        Some(Value::Record(r)) => Ok(r),
80        _ => Err(MeasureError::InvalidRecord {
81            reason: format!("missing or non-record field '{field}'"),
82        }),
83    }
84}

90/// Serializes an [`MEpoch`] to a [`RecordValue`] compatible with C++ `MeasureHolder`.
91pub fn epoch_to_record(epoch: &MEpoch) -> RecordValue {
92    RecordValue::new(vec![
93        RecordField::new(
94            "type",
95            Value::Scalar(ScalarValue::String("epoch".to_owned())),
96        ),
97        RecordField::new(
98            "refer",
99            Value::Scalar(ScalarValue::String(epoch.refer().as_str().to_owned())),
100        ),
101        RecordField::new(
102            "m0",
103            Value::Record(quantity_record(epoch.value().as_mjd(), "d")),
104        ),
105    ])
106}
107
108/// Deserializes an [`MEpoch`] from a [`RecordValue`] in C++ `MeasureHolder` format.
109pub fn epoch_from_record(rec: &RecordValue) -> Result<MEpoch, MeasureError> {
110    let type_str = extract_string(rec, "type")?;
111    if type_str != "epoch" {
112        return Err(MeasureError::InvalidRecord {
113            reason: format!("expected type 'epoch', got '{type_str}'"),
114        });
115    }
116
117    let refer_str = extract_string(rec, "refer")?;
118    let refer: EpochRef = refer_str.parse()?;
119
120    let m0 = extract_sub_record(rec, "m0")?;
121    let mjd = extract_quantity_value(m0)?;
122
123    Ok(MEpoch::new(MjdHighPrec::from_mjd(mjd), refer))
124}

130/// Serializes an [`MPosition`] to a [`RecordValue`] compatible with C++ `MeasureHolder`.
131///
132/// C++ `MVPosition::getRecordValue()` always stores positions in geocentric
133/// spherical form (lon, lat, radius) for ITRF. For WGS84, we store the
134/// geodetic coordinates (lon, lat, height) as-is.
135pub fn position_to_record(pos: &MPosition) -> RecordValue {
136    let (v0, v1, v2, u0, u1, u2) = match pos.refer() {
137        PositionRef::ITRF => {
138            let (lon, lat, r) = pos.as_spherical();
139            (lon, lat, r, "rad", "rad", "m")
140        }
141        PositionRef::WGS84 => {
142            let vals = pos.values();
143            (vals[0], vals[1], vals[2], "rad", "rad", "m")
144        }
145    };
146
147    RecordValue::new(vec![
148        RecordField::new(
149            "type",
150            Value::Scalar(ScalarValue::String("position".to_owned())),
151        ),
152        RecordField::new(
153            "refer",
154            Value::Scalar(ScalarValue::String(pos.refer().as_str().to_owned())),
155        ),
156        RecordField::new("m0", Value::Record(quantity_record(v0, u0))),
157        RecordField::new("m1", Value::Record(quantity_record(v1, u1))),
158        RecordField::new("m2", Value::Record(quantity_record(v2, u2))),
159    ])
160}
161
162/// Deserializes an [`MPosition`] from a [`RecordValue`] in C++ `MeasureHolder` format.
163///
164/// The record is expected to contain spherical coordinates (lon, lat, radius).
165/// The result is returned in ITRF (converted from spherical to Cartesian) if
166/// the refer is ITRF, or as WGS84 if refer is WGS84.
167pub fn position_from_record(rec: &RecordValue) -> Result<MPosition, MeasureError> {
168    let type_str = extract_string(rec, "type")?;
169    if type_str != "position" {
170        return Err(MeasureError::InvalidRecord {
171            reason: format!("expected type 'position', got '{type_str}'"),
172        });
173    }
174
175    let refer_str = extract_string(rec, "refer")?;
176    let refer: PositionRef = refer_str.parse()?;
177
178    let m0 = extract_sub_record(rec, "m0")?;
179    let lon = extract_quantity_value(m0)?;
180    let m1 = extract_sub_record(rec, "m1")?;
181    let lat = extract_quantity_value(m1)?;
182    let m2 = extract_sub_record(rec, "m2")?;
183    let radius = extract_quantity_value(m2)?;
184
185    match refer {
186        PositionRef::ITRF => {
187            // Convert spherical (lon, lat, radius) back to Cartesian.
188            // The radius in the record is the geocentric radius, but C++
189            // MeasureHolder actually stores geodetic height for ITRF.
190            // For maximum compatibility, use gd2gc with height derived from radius.
191            // However, the C++ record format stores (lon, lat, radius) where
192            // radius is the geocentric distance. We reconstruct Cartesian directly.
193            let x = radius * lat.cos() * lon.cos();
194            let y = radius * lat.cos() * lon.sin();
195            let z = radius * lat.sin();
196            Ok(MPosition::new_itrf(x, y, z))
197        }
198        PositionRef::WGS84 => {
199            // For WGS84, the record stores geodetic (lon, lat, height).
200            // But the m2 field is labeled "m" (radius in m). In practice
201            // C++ MVPosition stores the height for WGS84.
202            Ok(MPosition::new_wgs84(lon, lat, radius))
203        }
204    }
205}

211/// Serializes an [`MDirection`] to a [`RecordValue`] compatible with C++ `MeasureHolder`.
212///
213/// Format: `{ type: "direction", refer: "J2000", m0: { value: lon_rad, unit: "rad" }, m1: { value: lat_rad, unit: "rad" } }`
214pub fn direction_to_record(dir: &MDirection) -> RecordValue {
215    let (lon, lat) = dir.as_angles();
216    RecordValue::new(vec![
217        RecordField::new(
218            "type",
219            Value::Scalar(ScalarValue::String("direction".to_owned())),
220        ),
221        RecordField::new(
222            "refer",
223            Value::Scalar(ScalarValue::String(dir.refer().as_str().to_owned())),
224        ),
225        RecordField::new("m0", Value::Record(quantity_record(lon, "rad"))),
226        RecordField::new("m1", Value::Record(quantity_record(lat, "rad"))),
227    ])
228}
229
230/// Deserializes an [`MDirection`] from a [`RecordValue`] in C++ `MeasureHolder` format.
231pub fn direction_from_record(rec: &RecordValue) -> Result<MDirection, MeasureError> {
232    let type_str = extract_string(rec, "type")?;
233    if type_str != "direction" {
234        return Err(MeasureError::InvalidRecord {
235            reason: format!("expected type 'direction', got '{type_str}'"),
236        });
237    }
238
239    let refer_str = extract_string(rec, "refer")?;
240    let refer: DirectionRef = refer_str.parse()?;
241
242    let m0 = extract_sub_record(rec, "m0")?;
243    let lon = extract_quantity_value(m0)?;
244    let m1 = extract_sub_record(rec, "m1")?;
245    let lat = extract_quantity_value(m1)?;
246
247    Ok(MDirection::from_angles(lon, lat, refer))
248}

367#[cfg(test)]
368mod tests {
369    use super::*;
370
371    #[test]
372    fn epoch_record_roundtrip() {
373        let epoch = MEpoch::from_mjd(51544.5, EpochRef::UTC);
374        let rec = epoch_to_record(&epoch);
375        let decoded = epoch_from_record(&rec).unwrap();
376
377        assert_eq!(decoded.refer(), EpochRef::UTC);
378        assert!((decoded.value().as_mjd() - 51544.5).abs() < 1e-12);
379    }
380
381    #[test]
382    fn epoch_record_field_names() {
383        let epoch = MEpoch::from_mjd(51544.5, EpochRef::TAI);
384        let rec = epoch_to_record(&epoch);
385
386        assert_eq!(
387            rec.get("type"),
388            Some(&Value::Scalar(ScalarValue::String("epoch".to_owned())))
389        );
390        assert_eq!(
391            rec.get("refer"),
392            Some(&Value::Scalar(ScalarValue::String("TAI".to_owned())))
393        );
394        assert!(rec.get("m0").is_some());
395    }
396
397    #[test]
398    fn position_record_roundtrip() {
399        let pos = MPosition::new_itrf(-1601185.4, -5041977.5, 3554875.9);
400        let rec = position_to_record(&pos);
401        let decoded = position_from_record(&rec).unwrap();
402
403        assert_eq!(decoded.refer(), PositionRef::ITRF);
404        let orig = pos.values();
405        let dec = decoded.values();
406        for i in 0..3 {
407            assert!(
408                (orig[i] - dec[i]).abs() < 1.0,
409                "coordinate {i}: {:.3} vs {:.3}",
410                orig[i],
411                dec[i]
412            );
413        }
414    }
415
416    #[test]
417    fn position_record_field_names() {
418        let pos = MPosition::new_itrf(1.0, 2.0, 3.0);
419        let rec = position_to_record(&pos);
420
421        assert_eq!(
422            rec.get("type"),
423            Some(&Value::Scalar(ScalarValue::String("position".to_owned())))
424        );
425        assert_eq!(
426            rec.get("refer"),
427            Some(&Value::Scalar(ScalarValue::String("ITRF".to_owned())))
428        );
429        assert!(rec.get("m0").is_some());
430        assert!(rec.get("m1").is_some());
431        assert!(rec.get("m2").is_some());
432    }
433
434    #[test]
435    fn invalid_record_type() {
436        let rec = RecordValue::new(vec![RecordField::new(
437            "type",
438            Value::Scalar(ScalarValue::String("direction".to_owned())),
439        )]);
440        assert!(epoch_from_record(&rec).is_err());
441    }
