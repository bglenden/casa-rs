32fn run(args: Vec<OsString>) -> Result<(), String> {
33    let task = extract_task(&args)?;
34    let bundle = adapter_surface(task)?;
35
36    if has_flag(&args, "-h") || has_flag(&args, "--help") {
37        print!("{}", command_schema(&bundle).render_help());
38        return Ok(());
39    }
40    if has_flag(&args, "--ui-schema") {
41        print!(
42            "{}",
43            command_schema(&bundle)
44                .render_json_pretty()
45                .map_err(|error| error.to_string())?
46        );
47        return Ok(());
48    }
49    if has_flag(&args, "--json-schema") {
50        print!("{}", schema_bundle_json(&bundle)?);
51        return Ok(());
52    }
53    if has_flag(&args, "--protocol-info") {
54        print!(
55            "{}",
56            serde_json::to_string_pretty(&json!({
57                "protocol_name": "casars_casa_task_adapter",
58                "protocol_version": 1,
59                "surface_kind": "task",
60                "backend": if task == "plotcal" { "casa-rs" } else { "casatasks" }
61            }))
62            .map_err(|error| error.to_string())?
63        );
64        return Ok(());
65    }
66
67    let values = parse_values(&bundle, &args)?;
68    if task == "plotcal" {
69        run_plotcal(values)
70    } else {
71        run_casatask(task, &bundle, values)
72    }
73}

75fn adapter_surface(task: &str) -> Result<SurfaceContractBundle, String> {
76    let bundle = builtin_surface_bundle(task)
77        .map_err(|error| format!("unknown CASA-backed task {task:?}: {error}"))?;
78    let execution = bundle.surface.execution();
79    let routed_here = execution.invocation_name == "casars-casa-task"
80        && execution
81            .fixed_args
82            .windows(2)
83            .any(|args| args[0] == "--task" && args[1] == task);
84    if !routed_here {
85        return Err(format!(
86            "task {task:?} is not routed through casars-casa-task"
87        ));
88    }
89    Ok(bundle)
90}

92fn extract_task(args: &[OsString]) -> Result<&str, String> {
93    for (index, arg) in args.iter().enumerate() {
94        if arg == "--task" {
95            return args
96                .get(index + 1)
97                .and_then(|value| value.to_str())
98                .ok_or_else(|| "--task requires a value".to_string());
99        }
100        if let Some(value) = arg.to_str().and_then(|value| value.strip_prefix("--task=")) {
101            return Ok(value);
102        }
103    }
104    Err("--task is required".to_string())
105}

111fn command_schema(bundle: &SurfaceContractBundle) -> UiCommandSchema {
112    let mut schema: UiCommandSchema = serde_json::from_value(project_ui_schema(bundle))
113        .expect("canonical adapter UI projection must match UiCommandSchema");
114    schema.usage = format!(
115        "{} {} [parameters]",
116        schema.invocation_name,
117        bundle.surface.execution().fixed_args.join(" ")
118    );
119    schema
120}

146fn parse_values(
147    bundle: &SurfaceContractBundle,
148    args: &[OsString],
149) -> Result<BTreeMap<String, String>, String> {
150    let mut values = BTreeMap::new();
151    let mut positionals = bundle
152        .surface
153        .bindings()
154        .iter()
155        .filter_map(|binding| {
156            binding
157                .projections
158                .cli
159                .as_ref()
160                .and_then(|projection| projection.positional)
161                .map(|position| (position, binding))
162        })
163        .collect::<BTreeMap<_, _>>();
164    let mut positional_index = 0usize;
165    let mut index = 0usize;
166    while index < args.len() {
167        let raw = args[index]
168            .to_str()
169            .ok_or_else(|| format!("argument {index} is not valid UTF-8"))?;
170        if raw == "--task" {
171            index += 2;
172            continue;
173        }
174        if raw.starts_with("--task=")
175            || matches!(
176                raw,
177                "--ui-schema" | "--json-schema" | "--protocol-info" | "-h" | "--help"
178            )
179        {
180            index += 1;
181            continue;
182        }
183        if raw.starts_with("--") {
184            let binding = bundle
185                .surface
186                .bindings()
187                .iter()
188                .find(|binding| {
189                    binding.projections.cli.as_ref().is_some_and(|projection| {
190                        projection.flags.iter().any(|flag| flag == raw)
191                            || projection.false_flags.iter().any(|flag| flag == raw)
192                    })
193                })
194                .ok_or_else(|| format!("{} does not accept option {raw}", bundle.surface.id()))?;
195            let projection = binding
196                .projections
197                .cli
198                .as_ref()
199                .expect("matched CLI projection");
200            let name = binding
201                .projections
202                .python
203                .as_ref()
204                .map_or(binding.name.as_str(), |projection| projection.name.as_str());
205            if is_bool_domain(
206                &bundle
207                    .catalog
208                    .concept(&binding.concept)
209                    .expect("validated adapter concept")
210                    .value_domain,
211            ) {
212                let enabled = !projection.false_flags.iter().any(|flag| flag == raw);
213                values.insert(name.to_string(), enabled.to_string());
214                index += 1;
215                continue;
216            }
217            let value = args
218                .get(index + 1)
219                .and_then(|value| value.to_str())
220                .ok_or_else(|| format!("{raw} requires a value"))?;
221            values.insert(name.to_string(), value.to_string());
222            index += 2;
223            continue;
224        }
225        let binding = positionals
226            .remove(&positional_index)
227            .ok_or_else(|| format!("unexpected positional argument {raw:?}"))?;
228        let name = binding
229            .projections
230            .python
231            .as_ref()
232            .map_or(binding.name.as_str(), |projection| projection.name.as_str());
233        values.insert(name.to_string(), raw.to_string());
234        positional_index += 1;
235        index += 1;
236    }
237
238    for binding in bundle.surface.bindings() {
239        if matches!(binding.default, DefaultSpec::Required)
240            && matches!(binding.required_when, Predicate::Always)
241        {
242            let name = binding
243                .projections
244                .python
245                .as_ref()
246                .map_or(binding.name.as_str(), |projection| projection.name.as_str());
247            if values.get(name).is_none_or(|value| value.trim().is_empty()) {
248                return Err(format!(
249                    "{} requires --{}",
250                    bundle.surface.id(),
251                    binding.name.replace('_', "-")
252                ));
253            }
254        }
255    }
256    Ok(values)
257}

259fn is_bool_domain(domain: &ParameterType) -> bool {
260    match domain {
261        ParameterType::Bool => true,
262        ParameterType::Optional { value, .. } => is_bool_domain(value),
263        _ => false,
264    }
265}

267fn run_casatask(
268    task: &str,
269    bundle: &SurfaceContractBundle,
270    values: BTreeMap<String, String>,
271) -> Result<(), String> {
272    let python = env::var_os("CASA_RS_CASATASKS_PYTHON")
273        .unwrap_or_else(|| OsString::from(DEFAULT_CASA_TASKS_PYTHON));
274    let payload = serde_json::to_string(&json!({
275        "task": task,
276        "values": values,
277        "param_types": bundle.surface.bindings().iter().map(|binding| {
278            let name = binding.projections.python.as_ref()
279                .map_or(binding.name.as_str(), |projection| projection.name.as_str());
280            let domain = &bundle.catalog.concept(&binding.concept)
281                .expect("validated adapter concept").value_domain;
282            (name, value_type_name(domain))
283        }).collect::<BTreeMap<_, _>>()
284    }))
285    .map_err(|error| error.to_string())?;
286
287    let script = r#"
288import ast
289import contextlib
290import io
291import inspect
292import json
293import os
294import sys
295
296request = json.loads(sys.stdin.read())
297task_name = request["task"]
298values = request["values"]
299param_types = request["param_types"]
300
301os.environ.setdefault("MPLCONFIGDIR", "/private/tmp/casa-rs-mpl")
302with contextlib.redirect_stdout(io.StringIO()) as captured:
303    import casatasks
304
305task = getattr(casatasks, task_name)
306signature = inspect.signature(task)
307
308def convert(name, value):
309    kind = param_types.get(name, "string")
310    if kind == "bool":
311        return str(value).lower() == "true"
312    if kind == "integer":
313        return int(value)
314    if kind == "float":
315        return float(value)
316    if kind == "array":
317        text = str(value)
318        try:
319            parsed = ast.literal_eval(text)
320            return list(parsed) if isinstance(parsed, (list, tuple)) else [parsed]
321        except Exception:
322            return [part.strip() for part in text.split(",") if part.strip()]
323    text = str(value)
324    if text == "":
325        return text
326    default = signature.parameters.get(name).default if name in signature.parameters else inspect._empty
327    if isinstance(default, bool):
328        return text.lower() == "true"
329    if isinstance(default, int) and not isinstance(default, bool):
330        return int(text)
331    if isinstance(default, float):
332        return float(text)
333    if isinstance(default, (list, tuple, dict)):
334        try:
335            return ast.literal_eval(text)
336        except Exception:
337            return text
338    if text[0:1] in "[{(":
339        try:
340            return ast.literal_eval(text)
341        except Exception:
342            return text
343    return text
344
345kwargs = {name: convert(name, value) for name, value in values.items()}
346result = task(**kwargs)
347print(json.dumps({"task": task_name, "kwargs": kwargs, "result": result}, default=str, indent=2))
348"#;
349    let mut child = Command::new(&python)
350        .arg("-c")
351        .arg(script)
352        .stdin(Stdio::piped())
353        .stdout(Stdio::piped())
354        .stderr(Stdio::piped())
355        .spawn()
356        .map_err(|error| {
357            format!(
358                "spawn CASA Python {}: {error}",
359                PathBuf::from(&python).display()
360            )
361        })?;
362    child
363        .stdin
364        .as_mut()
365        .ok_or_else(|| "failed to open CASA Python stdin".to_string())?
366        .write_all(payload.as_bytes())
367        .map_err(|error| format!("write CASA task payload: {error}"))?;
368    let output = child
369        .wait_with_output()
370        .map_err(|error| format!("wait for CASA Python: {error}"))?;
371    if !output.status.success() {
372        return Err(format!(
373            "CASA task {} exited with {}: {}",
374            task,
375            output.status,
376            String::from_utf8_lossy(&output.stderr).trim()
377        ));
378    }
379    let stderr = String::from_utf8_lossy(&output.stderr);
380    if !stderr.trim().is_empty() {
381        eprintln!("{}", stderr.trim());
382    }
383    print!("{}", String::from_utf8_lossy(&output.stdout));
384    Ok(())
385}

387fn value_type_name(domain: &ParameterType) -> &'static str {
388    match domain {
389        ParameterType::Bool => "bool",
390        ParameterType::Integer => "integer",
391        ParameterType::Float => "float",
392        ParameterType::Array { .. } => "array",
393        ParameterType::Table { .. } => "table",
394        ParameterType::Optional { value, .. } => value_type_name(value),
395        ParameterType::String
396        | ParameterType::Path { .. }
397        | ParameterType::Choice { .. }
398        | ParameterType::Quantity { .. } => "string",
399    }
400}

402fn run_plotcal(values: BTreeMap<String, String>) -> Result<(), String> {
403    let preset = values
404        .get("preset")
405        .map(String::as_str)
406        .unwrap_or("gain_phase_vs_time");
407    let preset = parse_plotcal_preset(preset)?;
408    let request = CalibrationPlotRequest {
409        measurement_set_path: optional_path(&values, "vis"),
410        calibration_table_path: optional_path(&values, "caltable"),
411        selection: MsSelectionSpec {
412            selectdata: true,
413            field: optional_string(&values, "field"),
414            spw: optional_string(&values, "spw"),
415            timerange: optional_string(&values, "timerange"),
416            uvrange: optional_string(&values, "uvrange"),
417            antenna: optional_string(&values, "antenna"),
418            scan: optional_string(&values, "scan"),
419            correlation: optional_string(&values, "correlation"),
420            observation: optional_string(&values, "observation"),
421            array: optional_string(&values, "array"),
422            intent: optional_string(&values, "intent"),
423            feed: optional_string(&values, "feed"),
424            msselect: optional_string(&values, "msselect"),
425        },
426    };
427    let payload = build_calibration_plot_payload(&request, preset)
428        .map_err(|error| format!("plotcal failed: {error}"))?;
429    print!(
430        "{}",
431        serde_json::to_string_pretty(&json!({
432            "task": "plotcal",
433            "preset": format!("{preset:?}"),
434            "payload_debug": format!("{payload:#?}"),
435        }))
436        .map_err(|error| error.to_string())?
437    );
438    Ok(())
439}

441fn optional_path(values: &BTreeMap<String, String>, key: &str) -> Option<PathBuf> {
442    optional_string(values, key).map(PathBuf::from)
443}
444
445fn optional_string(values: &BTreeMap<String, String>, key: &str) -> Option<String> {
446    values
447        .get(key)
448        .map(|value| value.trim())
449        .filter(|value| !value.is_empty())
450        .map(str::to_string)
451}
452
453fn parse_plotcal_preset(value: &str) -> Result<CalibrationPlotPreset, String> {
454    match value {
455        "gain_phase_vs_time" => Ok(CalibrationPlotPreset::GainPhaseVsTime),
456        "gain_amplitude_vs_time" => Ok(CalibrationPlotPreset::GainAmplitudeVsTime),
457        "bandpass_amplitude_vs_frequency" => {
458            Ok(CalibrationPlotPreset::BandpassAmplitudeVsFrequency)
459        }
460        "bandpass_phase_vs_frequency" => Ok(CalibrationPlotPreset::BandpassPhaseVsFrequency),
461        "corrected_amplitude_vs_time" => Ok(CalibrationPlotPreset::CorrectedAmplitudeVsTime),
462        "corrected_phase_vs_time" => Ok(CalibrationPlotPreset::CorrectedPhaseVsTime),
463        "corrected_amplitude_vs_frequency" => {
464            Ok(CalibrationPlotPreset::CorrectedAmplitudeVsFrequency)
465        }
466        "corrected_phase_vs_frequency" => Ok(CalibrationPlotPreset::CorrectedPhaseVsFrequency),
467        other => Err(format!("unknown plotcal preset {other:?}")),
468    }

477    #[test]
478    fn schema_bundles_embed_each_current_adapter_parameter_contract() {
479        let aggregate = builtin_surface_catalog().expect("built-in parameter catalog");
480        let adapter_surfaces = aggregate
481            .surfaces
482            .iter()
483            .filter(|surface| surface.execution().invocation_name == "casars-casa-task")
484            .collect::<Vec<_>>();
485        assert!(!adapter_surfaces.is_empty());
486        for surface in adapter_surfaces {
487            let expected_id = surface.id();
488            let contract = adapter_surface(expected_id).expect("current adapter surface");
489            let bundle = serde_json::from_str::<serde_json::Value>(
490                &schema_bundle_json(&contract).expect("serialize adapter schema bundle"),
491            )
492            .expect("parse adapter schema bundle");
493
494            assert_eq!(
495                bundle["protocol"]["protocol_name"],
496                "casars_casa_task_adapter"
497            );
498            assert!(bundle["request_schema"].is_object());
499            assert!(bundle["result_schema"].is_object());
500
501            let surfaces = serde_json::from_value::<Vec<SurfaceContractBundle>>(
502                bundle["parameter_surfaces"].clone(),
503            )
504            .expect("serialized adapter parameter surface");
505            assert_eq!(surfaces.len(), 1);
506            assert_eq!(surfaces[0].surface.id(), expected_id);
507            surfaces[0]
508                .validate()
509                .expect("embedded adapter parameter surface");
510            assert_eq!(
511                bundle["projections"]["ui_schema"],
512                project_ui_schema(&contract)
