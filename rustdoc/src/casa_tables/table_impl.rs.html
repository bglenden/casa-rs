50impl PendingArrayColumn {
51    fn insert(&mut self, row_index: usize, value: ArrayValue) {
52        if let Some((last_row_index, last_value)) = self.rows.last_mut()
53            && *last_row_index == row_index
54        {
55            *last_value = value;
56            return;
57        }
58
59        let keeps_sorted = self
60            .rows
61            .last()
62            .is_none_or(|(last_row_index, _)| *last_row_index <= row_index);
63        self.sorted &= keeps_sorted;
64        self.rows.push((row_index, value));
65    }
66
67    fn get(&self, row_index: usize) -> Option<&ArrayValue> {
68        if self.sorted {
69            let found = self
70                .rows
71                .partition_point(|(candidate, _)| *candidate <= row_index);
72            if found == 0 {
73                return None;
74            }
75            let (candidate, value) = &self.rows[found - 1];
76            return (*candidate == row_index).then_some(value);
77        }
78
79        self.rows
80            .iter()
81            .rev()
82            .find(|(candidate, _)| *candidate == row_index)
83            .map(|(_, value)| value)
84    }
85
86    fn reserve(&mut self, additional: usize) {
87        self.rows.reserve(additional);
88    }
89}

108fn lazy_array_column_store(
109    schema: Option<&TableSchema>,
110) -> HashMap<String, OnceCell<Vec<Option<ArrayValue>>>> {
111    schema
112        .into_iter()
113        .flat_map(|schema| schema.columns())
114        .filter(|column| matches!(column.column_type(), ColumnType::Array(_)))
115        .map(|column| (column.name().to_string(), OnceCell::new()))
116        .collect()
117}

119fn eager_loaded_rows(
120    rows: Vec<RecordValue>,
121    mut undefined_cells: Vec<HashSet<String>>,
122) -> LoadedRows {
123    if undefined_cells.len() != rows.len() {
124        undefined_cells = (0..rows.len()).map(|_| HashSet::new()).collect();
125    }
126    LoadedRows {
127        rows,
128        undefined_cells,
129    }
130}

239    fn load_rows_now(source: &LazyRowsSource) -> Result<LoadedRows, TableError> {
240        let mut profiler = StorageProfiler::start(format!(
241            "table_impl::load_rows_now path={}",
242            source.path.display()
243        ));
244        let storage = CompositeStorage;
245        let snapshot = storage
246            .load_with_row_hint(&source.path, Some(source.row_count_hint as u64))
247            .map_err(|err| {
248                TableError::Storage(format!(
249                    "failed to materialize rows for table {}: {err}",
250                    source.path.display()
251                ))
252            })?;
253        if let Some(profiler) = profiler.as_mut() {
254            profiler.mark_with_detail(
255                "storage_load_complete",
256                Some(format!(
257                    "row_count_hint={} rows={}",
258                    source.row_count_hint, snapshot.row_count
259                )),
260            );
261        }
262        let loaded = eager_loaded_rows(snapshot.rows, snapshot.undefined_cells);
263        if let Some(profiler) = profiler.as_mut() {
264            profiler.mark_with_detail(
265                "eager_loaded_rows",
266                Some(format!(
267                    "rows={} undefined_rows={}",
268                    loaded.rows.len(),
269                    loaded.undefined_cells.len()
270                )),
271            );
272        }
273        Ok(loaded)
274    }

376    fn ensure_loaded(&self) -> Result<&LoadedRows, TableError> {
377        if let Some(loaded) = self.loaded_rows.get() {
378            return Ok(loaded);
379        }
380        let loaded = match &self.lazy_rows {
381            Some(source) => Self::load_rows_now(source)?,
382            None => LoadedRows {
383                rows: Vec::new(),
384                undefined_cells: Vec::new(),
385            },
386        };
387        self.loaded_rows
388            .set(loaded)
389            .expect("initialize immutable row store");
390        Ok(self
391            .loaded_rows
392            .get()
393            .expect("row store initialized before shared access"))
394    }
395
396    fn ensure_loaded_mut(&mut self) -> Result<&mut LoadedRows, TableError> {
397        if self.loaded_rows.get().is_none() {
398            let loaded = match &self.lazy_rows {
399                Some(source) => Self::load_rows_now(source)?,
400                None => LoadedRows {
401                    rows: Vec::new(),
402                    undefined_cells: Vec::new(),
403                },
404            };
405            self.loaded_rows
406                .set(loaded)
407                .expect("initialize mutable row store");
408        }
409        if !self.pending_array_cells.by_column.is_empty() {
410            let pending = std::mem::take(&mut self.pending_array_cells.by_column);
411            if let Some(loaded) = self.loaded_rows.get_mut() {
412                for (column, rows) in pending {
413                    for (row_index, value) in rows.rows {
414                        if let Some(row) = loaded.rows.get_mut(row_index) {
415                            row.upsert(column.clone(), Value::Array(value));
416                        }
417                    }
418                }
419            }
420        }
421        if !self.pending_scalar_cells.by_column.is_empty() {
422            let pending = std::mem::take(&mut self.pending_scalar_cells.by_column);
423            if let Some(loaded) = self.loaded_rows.get_mut() {
424                for (column, rows) in pending {
425                    for (row_index, value) in rows {
426                        if let Some(row) = loaded.rows.get_mut(row_index) {
427                            row.upsert(column.clone(), Value::Scalar(value));
428                        }
429                    }
430                }
431            }
432        }
433        Ok(self
434            .loaded_rows
435            .get_mut()
436            .expect("row store initialized before mutable access"))
437    }

444        self.lazy_rows = None;
445        Ok(())

470    pub(crate) fn scalar_cell(
471        &self,
472        row_index: usize,
473        column: &str,
474    ) -> Result<LazyScalarLookup<'_>, TableError> {
475        if let Some(loaded) = self.loaded_rows.get() {
476            let Some(row) = loaded.rows.get(row_index) else {
477                return Ok(LazyScalarLookup::Missing);
478            };
479            return Ok(match row.get(column) {
480                Some(Value::Scalar(scalar)) => LazyScalarLookup::Hit(scalar),
481                Some(_) => LazyScalarLookup::Unknown,
482                None => LazyScalarLookup::Missing,
483            });
484        }
485
486        let Some(source) = &self.lazy_rows else {
487            return Ok(LazyScalarLookup::Unknown);
488        };
489
490        if let Some(overrides) = self.pending_scalar_cells.by_column.get(column)
491            && let Some(value) = overrides.get(&row_index)
492        {
493            return Ok(LazyScalarLookup::Hit(value));
494        }
495
496        if self.loaded_scalar_columns.get().is_none() {
497            let loaded = Self::load_scalar_columns_now(source)?;
498            let _ = self.loaded_scalar_columns.set(loaded);
499        }
500        let columns = self
501            .loaded_scalar_columns
502            .get()
503            .expect("scalar columns initialized before access");
504        let Some(values) = columns.columns.get(column) else {
505            return Ok(LazyScalarLookup::Unknown);
506        };
507        match values.get(row_index) {
508            Some(Some(value)) => Ok(LazyScalarLookup::Hit(value)),
509            Some(None) | None => Ok(LazyScalarLookup::Missing),
510        }
511    }

536    pub(crate) fn scalar_cells_owned(
537        &self,
538        column: &str,
539    ) -> Result<Option<Vec<Option<ScalarValue>>>, TableError> {
540        if let Some(loaded) = self.loaded_rows.get() {
541            let values = loaded
542                .rows
543                .iter()
544                .map(|row| match row.get(column) {
545                    Some(Value::Scalar(scalar)) => Some(scalar.clone()),
546                    _ => None,
547                })
548                .collect();
549            return Ok(Some(values));
550        }
551
552        if let Some(columns) = self.loaded_scalar_columns.get()
553            && let Some(values) = columns.columns.get(column)
554        {
555            let mut values = values.clone();
556            if let Some(overrides) = self.pending_scalar_cells.by_column.get(column) {
557                for (&row_index, value) in overrides {
558                    if let Some(cell) = values.get_mut(row_index) {
559                        *cell = Some(value.clone());
560                    }
561                }
562            }
563            return Ok(Some(values));
564        }
565
566        let Some(source) = &self.lazy_rows else {
567            return Ok(None);
568        };
569
570        let mut profiler = StorageProfiler::start(format!(
571            "table_impl::scalar_cells_owned path={} column={column}",
572            source.path.display()
573        ));
574        let storage = CompositeStorage;
575        let mut values = storage
576            .load_scalar_column_with_row_hint(
577                &source.path,
578                column,
579                Some(source.row_count_hint as u64),
580            )
581            .map_err(|err| {
582                TableError::Storage(format!(
583                    "failed to load scalar column '{column}' for table {}: {err}",
584                    source.path.display()
585                ))
586            })?;
587        if let Some(profiler) = profiler.as_mut() {
588            profiler.mark_with_detail(
589                "storage_load_complete",
590                Some(format!(
591                    "row_count_hint={} values={}",
592                    source.row_count_hint,
593                    values.len()
594                )),
595            );
596        }
597        if let Some(overrides) = self.pending_scalar_cells.by_column.get(column) {
598            for (&row_index, value) in overrides {
599                if let Some(cell) = values.get_mut(row_index) {
600                    *cell = Some(value.clone());
601                }
602            }
603        }
604        Ok(Some(values))
605    }

607    pub(crate) fn array_cell(
608        &self,
609        row_index: usize,
610        column: &str,
611    ) -> Result<LazyArrayLookup<'_>, TableError> {
612        if let Some(loaded) = self.loaded_rows.get() {
613            let Some(row) = loaded.rows.get(row_index) else {
614                return Ok(LazyArrayLookup::Missing);
615            };
616            return Ok(match row.get(column) {
617                Some(Value::Array(array)) => LazyArrayLookup::Hit(array),
618                Some(_) => LazyArrayLookup::Unknown,
619                None => LazyArrayLookup::Missing,
620            });
621        }
622
623        let Some(source) = &self.lazy_rows else {
624            return Ok(LazyArrayLookup::Unknown);
625        };
626        if let Some(overrides) = self.pending_array_cells.by_column.get(column)
627            && let Some(value) = overrides.get(row_index)
628        {
629            return Ok(LazyArrayLookup::Hit(value));
630        }
631        let Some(cached_column) = self.loaded_array_columns.get(column) else {
632            return Ok(LazyArrayLookup::Unknown);
633        };
634        if cached_column.get().is_none() {
635            let loaded = Self::load_array_column_now(source, column)?;
636            let _ = cached_column.set(loaded);
637        }
638        let values = cached_column
639            .get()
640            .expect("array column initialized before access");
641        match values.get(row_index) {
642            Some(Some(value)) => Ok(LazyArrayLookup::Hit(value)),
643            Some(None) | None => Ok(LazyArrayLookup::Missing),
644        }
645    }

647    pub(crate) fn array_cells_owned(
648        &self,
649        row_indices: &[usize],
650        column: &str,
651    ) -> Result<Option<Vec<Option<ArrayValue>>>, TableError> {
652        if let Some(loaded) = self.loaded_rows.get() {
653            let values = row_indices
654                .iter()
655                .map(|&row_index| {
656                    loaded
657                        .rows
658                        .get(row_index)
659                        .and_then(|row| match row.get(column) {
660                            Some(Value::Array(array)) => Some(array.clone()),
661                            _ => None,
662                        })
663                })
664                .collect();
665            return Ok(Some(values));
666        }
667
668        let Some(source) = &self.lazy_rows else {
669            return Ok(None);
670        };
671        let mut values = if let Some(cached_column) = self.loaded_array_columns.get(column)
672            && let Some(cached_values) = cached_column.get()
673        {
674            row_indices
675                .iter()
676                .map(|&row_index| cached_values.get(row_index).cloned().unwrap_or(None))
677                .collect()
678        } else {
679            Self::load_array_column_rows_now(source, column, row_indices)?
680        };
681        if let Some(overrides) = self.pending_array_cells.by_column.get(column) {
682            for (out_idx, &row_index) in row_indices.iter().enumerate() {
683                if let Some(value) = overrides.get(row_index) {
684                    values[out_idx] = Some(value.clone());
685                }
686            }
687        }
688
689        Ok(Some(values))
690    }

709        if let Some(loaded_rows) = self.loaded_rows.get_mut() {
710            let Some(row) = loaded_rows.rows.get_mut(row_index) else {
711                return Ok(Some(value));
712            };
713            row.upsert(column.to_string(), Value::Scalar(value));
714            return Ok(None);
715        }
716
717        if row_index >= self.row_count() {
718            return Ok(Some(value));
719        }
720
721        if let Some(loaded) = self.loaded_scalar_columns.get_mut()
722            && let Some(column_values) = loaded.columns.get_mut(column)
723        {
724            if let Some(cell) = column_values.get_mut(row_index) {
725                *cell = Some(value);
726                return Ok(None);
727            }
728            return Ok(Some(value));
729        }
730
731        self.pending_scalar_cells
732            .by_column
733            .entry(column.to_string())
734            .or_default()
735            .insert(row_index, value);
736        Ok(None)
737    }

739    pub(crate) fn set_cached_array_cell(
740        &mut self,
741        row_index: usize,
742        column: &str,
743        value: ArrayValue,
744    ) -> Result<Option<ArrayValue>, TableError> {
745        if let Some(loaded_rows) = self.loaded_rows.get_mut() {
746            let Some(row) = loaded_rows.rows.get_mut(row_index) else {
747                return Ok(Some(value));
748            };
749            row.upsert(column.to_string(), Value::Array(value));
750            return Ok(None);
751        }
752
753        if row_index >= self.row_count() {
754            return Ok(Some(value));
755        }
756
757        if let Some(cached_column) = self.loaded_array_columns.get_mut(column) {
758            if cached_column.get().is_some() {
759                let column_values = cached_column
760                    .get_mut()
761                    .expect("array column initialized before mutable access");
762                if let Some(cell) = column_values.get_mut(row_index) {
763                    *cell = Some(value);
764                    return Ok(None);
765                }
766                return Ok(Some(value));
767            }
768        }
769        self.pending_array_cells
770            .by_column
771            .entry(column.to_string())
772            .or_default()
773            .insert(row_index, value);
774        Ok(None)
775    }

808        self.lazy_rows = None;
809        Ok(removed)
810    }

815        loaded.undefined_cells.insert(index, HashSet::new());

818        Ok(())

859    /// Replace all inner state from a storage snapshot.
860    ///
861    /// Used by `Table::lock()` when sync data indicates another process
862    /// modified the table and a full reload is needed.
863    pub(crate) fn replace_from_snapshot(
864        &mut self,
865        rows: Vec<RecordValue>,
866        undefined_cells: Vec<HashSet<String>>,
867        keywords: RecordValue,
868        column_keywords: HashMap<String, RecordValue>,
869        schema: Option<TableSchema>,
870    ) {
871        self.loaded_rows = {
872            let loaded_rows = OnceCell::new();
873            loaded_rows
874                .set(eager_loaded_rows(rows, undefined_cells))
875                .expect("replace eager row store");
876            loaded_rows
877        };
878        self.loaded_scalar_columns = OnceCell::new();
879        self.pending_scalar_cells = PendingScalarCells::default();
880        self.loaded_array_columns = lazy_array_column_store(schema.as_ref());
881        self.pending_array_cells = PendingArrayCells::default();
882        self.persisted_row_count = self.loaded_rows.get().map_or(0, |loaded| loaded.rows.len());
883        self.lazy_rows = None;
884        self.keywords = keywords;
885        self.column_keywords = column_keywords;
886        self.schema = schema;
887    }
