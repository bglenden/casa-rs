2use std::env;

9#[derive(Debug, Clone)]
10pub(crate) struct RegistryApp {
11    pub id: &'static str,
12    pub category: &'static str,
13    pub display_name: &'static str,
14    binary_name: &'static str,
15    cargo_package: &'static str,
16}
17
18#[derive(Debug, Clone)]
19pub(crate) struct ResolvedCommand {
20    program: OsString,
21    prefix_args: Vec<OsString>,
22}
23
24impl ResolvedCommand {
25    pub(crate) fn command(&self) -> Command {
26        let mut command = Command::new(&self.program);
27        command.args(&self.prefix_args);
28        command
29    }
30}
31
32impl RegistryApp {
33    pub(crate) fn load_schema(&self) -> Result<UiCommandSchema, String> {
34        let resolved = self.resolve_command()?;
35        let output = resolved
36            .command()
37            .arg("--ui-schema")
38            .output()
39            .map_err(|error| format!("spawn {} --ui-schema: {error}", self.binary_name))?;
40        if !output.status.success() {
41            let stderr = String::from_utf8_lossy(&output.stderr);
42            return Err(format!(
43                "{} --ui-schema exited with {}: {}",
44                self.binary_name,
45                output.status,
46                stderr.trim()
47            ));
48        }
49        serde_json::from_slice(&output.stdout)
50            .map_err(|error| format!("parse {} --ui-schema output: {error}", self.binary_name))
51    }
52
53    pub(crate) fn resolve_command(&self) -> Result<ResolvedCommand, String> {
54        if let Some(path) = env::var_os("CASARS_LISTOBS_BIN") {
55            return Ok(ResolvedCommand {
56                program: path,
57                prefix_args: Vec::new(),
58            });
59        }
60
61        if let Some(path) = env::var_os(format!("CARGO_BIN_EXE_{}", self.binary_name)) {
62            return Ok(ResolvedCommand {
63                program: path,
64                prefix_args: Vec::new(),
65            });
66        }
67
68        if let Some(path) = sibling_binary(self.binary_name) {
69            return Ok(ResolvedCommand {
70                program: path.into_os_string(),
71                prefix_args: Vec::new(),
72            });
73        }
74
75        let cargo = env::var_os("CARGO").unwrap_or_else(|| OsString::from("cargo"));
76        Ok(ResolvedCommand {
77            program: cargo,
78            prefix_args: vec![
79                OsString::from("run"),
80                OsString::from("-q"),
81                OsString::from("-p"),
82                OsString::from(self.cargo_package),
83                OsString::from("--bin"),
84                OsString::from(self.binary_name),
85                OsString::from("--"),
86            ],
87        })
88    }
89}

91pub(crate) fn listobs_app() -> RegistryApp {
92    RegistryApp {
93        id: "listobs",
94        category: "MeasurementSet",
95        display_name: "ListObs",
96        binary_name: "listobs",
97        cargo_package: "casacore-ms",
98    }

101fn sibling_binary(binary_name: &str) -> Option<PathBuf> {
102    let mut path = env::current_exe().ok()?;
103    path.pop();
104    path.push(binary_name);
105    path.set_extension(env::consts::EXE_EXTENSION);
106    if path.exists() { Some(path) } else { None }
107}
