14/// Structured imaging run report consumed by the `casars` workflow shell.
15#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
16pub struct ManagedImagingOutput {
17    /// High-level request summary used by the shell overview.
18    pub request: ManagedImagingRequest,
19    /// Structured run metrics and per-channel diagnostics.
20    pub run: ManagedImagingRun,
21    /// Expected CASA image products written under the configured prefix.
22    pub artifacts: Vec<ManagedImagingArtifact>,
23}
24
25/// Launcher-facing view of the requested imaging configuration.
26#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
27pub struct ManagedImagingRequest {
28    /// MeasurementSet path supplied for the run.
29    pub measurement_set: String,
30    /// Output CASA image prefix.
31    pub imagename: String,
32    /// Spectral imaging mode.
33    pub spectral_mode: String,
34    /// Requested visibility weighting.
35    pub weighting: String,
36    /// Requested minor-cycle deconvolver.
37    pub deconvolver: String,
38    /// Requested `w`-term handling mode.
39    pub w_term_mode: String,
40    /// Optional data-column override.
41    pub data_column: Option<String>,
42    /// Image size in pixels.
43    pub imsize: usize,
44    /// Cell size in arcseconds.
45    pub cell_arcsec: f64,
46    /// Whether the run skipped CLEAN.
47    pub dirty_only: bool,
48    /// Whether preview PNG sidecars were requested.
49    pub write_preview_pngs: bool,
50    /// Whether per-channel density estimation was requested for cube weighting.
51    pub per_channel_weight_density: bool,
52    /// Requested MTMFS Taylor-term count.
53    pub nterms: usize,
54    /// Output channel count for cube-like runs.
55    pub output_channels: usize,
56    /// Requested raw-correlation or Stokes plane, when explicitly selected.
57    pub correlation: Option<String>,
58    /// Requested restoring-beam mode for restored products.
59    pub restoring_beam_mode: String,
60}
61
62/// Structured run metrics emitted after one successful imaging run.
63#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
64pub struct ManagedImagingRun {
65    /// Warnings emitted by the imaging run.
66    pub warnings: Vec<String>,
67    /// Number of scalar samples that reached the gridder.
68    pub gridded_samples: usize,
69    /// Total major-cycle count reported by the run.
70    pub major_cycles: usize,
71    /// Total minor-cycle component updates reported by the run.
72    pub minor_iterations: usize,
73    /// Final CLEAN stop reason when deconvolution ran.
74    pub clean_stop_reason: Option<String>,
75    /// Timing breakdown reported by the pure imaging core.
76    pub stage_timings: ManagedImagingStageTimings,
77    /// Timing breakdown for the MeasurementSet-backed frontend.
78    pub frontend_timings: ManagedImagingStageTimings,
79    /// Channel-level diagnostics for cube-like runs.
80    pub channels: Vec<ManagedImagingChannelRun>,
81}

90/// Channel-level convergence summary emitted for cube-like runs.
91#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
92pub struct ManagedImagingChannelRun {
93    /// Zero-based output channel index.
94    pub channel_index: usize,
95    /// Major-cycle count for this plane.
96    pub major_cycles: usize,
97    /// Minor-cycle component updates for this plane.
98    pub minor_iterations: usize,
99    /// Final CLEAN stop reason for this plane.
100    pub clean_stop_reason: Option<String>,
101    /// Peak residual before minor cycles.
102    pub initial_residual_peak_jy_per_beam: f32,
103    /// Peak residual after the final exact refresh.
104    pub final_residual_peak_jy_per_beam: f32,
105    /// Final CASA-style cycle threshold for this plane.
106    pub final_cycle_threshold_jy_per_beam: f32,
107    /// Whether the beam-fit debug summary was available for this plane.
108    pub beam_fit_available: bool,
109}

111/// One expected output artifact written by the imaging run.
112#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
113pub struct ManagedImagingArtifact {
114    /// Stable artifact kind identifier such as `psf`, `image`, or `alpha`.
115    pub kind: String,
116    /// Human-readable artifact label.
117    pub label: String,
118    /// On-disk path for the CASA image product.
119    pub path: String,
120    /// Whether that product exists after the run.
121    pub exists: bool,
122    /// Optional preview sidecar path.
123    pub preview_png_path: Option<String>,
124    /// Whether the preview sidecar exists after the run.
125    pub preview_png_exists: bool,
126}

128impl ManagedImagingOutput {
129    /// Build the structured launcher report from one completed run.
130    pub fn from_run(config: &CliConfig, summary: &RunSummary) -> Self {
131        Self {
132            request: ManagedImagingRequest {
133                measurement_set: config.ms.display().to_string(),
134                imagename: config.imagename.display().to_string(),
135                spectral_mode: canonical_spectral_mode_name(config.spectral_mode).to_string(),
136                weighting: canonical_weighting_name(config.weighting),
137                deconvolver: canonical_deconvolver_name(config.deconvolver).to_string(),
138                w_term_mode: canonical_w_term_mode_name(config.w_term_mode).to_string(),
139                data_column: config.datacolumn.clone(),
140                imsize: config.imsize,
141                cell_arcsec: config.cell_arcsec,
142                dirty_only: config.dirty_only,
143                write_preview_pngs: config.write_preview_pngs,
144                per_channel_weight_density: config.per_channel_weight_density,
145                nterms: config.nterms,
146                output_channels: summary.channel_summaries.len(),
147                correlation: config.correlation.clone(),
148                restoring_beam_mode: canonical_restoring_beam_mode_name(config.restoring_beam_mode)
149                    .to_string(),
150            },
151            run: ManagedImagingRun {
152                warnings: summary.warnings.clone(),
153                gridded_samples: summary.gridded_samples,
154                major_cycles: summary.major_cycles,
155                minor_iterations: summary.minor_iterations,
156                clean_stop_reason: summary
157                    .clean_stop_reason
158                    .map(|reason| format!("{reason:?}")),
159                stage_timings: stage_timings_from_core(summary),
160                frontend_timings: stage_timings_from_frontend(summary.frontend_timings),
161                channels: summary
162                    .channel_summaries
163                    .iter()
164                    .map(channel_run_from_summary)
165                    .collect(),
166            },
167            artifacts: imaging_artifacts(config),
168        }
169    }
170}

270fn imaging_artifacts(config: &CliConfig) -> Vec<ManagedImagingArtifact> {
271    let base = config.imagename.to_string_lossy().to_string();
272    let mut artifacts = Vec::new();
273    match config.spectral_mode {
274        SpectralMode::Mfs
275            if canonical_deconvolver_name(config.deconvolver) == "mtmfs" && config.nterms > 1 =>
276        {
277            for term in 0..config.nterms {
278                for (kind, label) in [
279                    ("psf", "PSF"),
280                    ("residual", "Residual"),
281                    ("model", "Model"),
282                    ("image", "Restored Image"),
283                ] {
284                    let suffix = format!("{kind}.tt{term}");
285                    let path = PathBuf::from(format!("{base}.{suffix}"));
286                    let preview = (term == 0 && config.write_preview_pngs)
287                        .then(|| PathBuf::from(format!("{base}.{suffix}.png")));
288                    artifacts.push(artifact(label_for_term(label, term), kind, path, preview));
289                }
290            }
291            let alpha = PathBuf::from(format!("{base}.alpha"));
292            let alpha_preview = config
293                .write_preview_pngs
294                .then(|| PathBuf::from(format!("{base}.alpha.png")));
295            artifacts.push(artifact(
296                "Spectral Index".to_string(),
297                "alpha",
298                alpha,
299                alpha_preview,
300            ));
301        }
302        _ => {
303            for (kind, label) in [
304                ("psf", "PSF"),
305                ("residual", "Residual"),
306                ("model", "Model"),
307                ("image", "Restored Image"),
308            ] {
309                let path = PathBuf::from(format!("{base}.{kind}"));
310                let preview = config
311                    .write_preview_pngs
312                    .then(|| PathBuf::from(format!("{base}.{kind}.png")));
313                artifacts.push(artifact(label.to_string(), kind, path, preview));
314            }
315        }
316    }
317    artifacts
318}
319
320fn label_for_term(base: &str, term: usize) -> String {
321    format!("{base} TT{term}")
322}

324fn artifact(
325    label: String,
326    kind: &str,
327    path: PathBuf,
328    preview: Option<PathBuf>,
329) -> ManagedImagingArtifact {
330    ManagedImagingArtifact {
331        kind: kind.to_string(),
332        label,
333        exists: path.exists(),
334        path: path.display().to_string(),
335        preview_png_exists: preview.as_ref().is_some_and(|path| path.exists()),
336        preview_png_path: preview.map(|path| path.display().to_string()),
337    }
