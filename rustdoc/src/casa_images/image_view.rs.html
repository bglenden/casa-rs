159impl ImageRegion {
160    /// Returns `true` when any polygon is still open for editing.
161    pub fn has_open_shape(&self) -> bool {
162        self.shapes.iter().any(|shape| !shape.closed)
163    }
164
165    /// Returns `true` if the region has no shapes.
166    pub fn is_empty(&self) -> bool {
167        self.shapes.is_empty()
168    }
169
170    /// Starts a new polygon shape.
171    pub fn start_shape(&mut self) -> Result<(), ImageError> {
172        if self.has_open_shape() {
173            return Err(ImageError::InvalidMetadata(
174                "close or cancel the current polygon before starting another".into(),
175            ));
176        }
177        self.shapes.push(ImageRegionShape {
178            vertices: Vec::new(),
179            closed: false,
180        });
181        Ok(())
182    }
183
184    /// Appends a new vertex to the active open polygon.
185    pub fn append_vertex(&mut self, vertex: ImageRegionVertex) -> Result<bool, ImageError> {
186        let Some(shape) = self.shapes.iter_mut().rev().find(|shape| !shape.closed) else {
187            return Err(ImageError::InvalidMetadata(
188                "start a region with R before adding vertices".into(),
189            ));
190        };
191        if shape
192            .vertices
193            .last()
194            .is_some_and(|last| last.world == vertex.world)
195        {
196            return Ok(false);
197        }
198        shape.vertices.push(vertex);
199        Ok(true)
200    }
201
202    /// Closes the active open polygon after validating its geometry.
203    pub fn close_active_shape(&mut self) -> Result<(), ImageError> {
204        let Some(shape) = self.shapes.iter_mut().rev().find(|shape| !shape.closed) else {
205            return Err(ImageError::InvalidMetadata(
206                "no open polygon to close".into(),
207            ));
208        };
209        if shape.vertices.len() < 3 {
210            return Err(ImageError::InvalidMetadata(
211                "polygon regions require at least 3 vertices".into(),
212            ));
213        }
214        if polygon_self_intersects(&shape.vertices) {
215            return Err(ImageError::InvalidMetadata(
216                "self-intersecting polygons are not supported".into(),
217            ));
218        }
219        shape.closed = true;
220        Ok(())
221    }
222
223    /// Removes the last vertex from the active open polygon.
224    pub fn undo_active_vertex(&mut self) -> Result<bool, ImageError> {
225        let Some(shape_index) = self.shapes.iter().rposition(|shape| !shape.closed) else {
226            return Err(ImageError::InvalidMetadata(
227                "no open polygon to edit".into(),
228            ));
229        };
230        let shape = &mut self.shapes[shape_index];
231        shape.vertices.pop();
232        if shape.vertices.is_empty() {
233            self.shapes.remove(shape_index);
234        }
235        Ok(!self.shapes.is_empty())
236    }
237
238    /// Cancels the active open polygon, if any.
239    pub fn cancel_active_shape(&mut self) -> bool {
240        let Some(shape_index) = self.shapes.iter().rposition(|shape| !shape.closed) else {
241            return false;
242        };
243        self.shapes.remove(shape_index);
244        true
245    }
246
247    /// Removes all shapes.
248    pub fn clear(&mut self) {
249        self.shapes.clear();
250    }
251}

322    pub fn format_inc(&self) -> String {
323        format_axis_list(&self.inc)
324    }
325
326    pub(crate) fn sampled_axis_len(&self, axis: usize) -> usize {
327        sampled_axis_len(self.blc[axis], self.trc[axis], self.inc[axis])
328    }
329
330    pub(crate) fn sampled_axis_value(&self, axis: usize, sample_index: usize) -> Option<usize> {
331        (sample_index < self.sampled_axis_len(axis))
332            .then_some(self.blc[axis] + sample_index * self.inc[axis])
333    }
334
335    pub(crate) fn sampled_axis_values(&self, axis: usize) -> Vec<usize> {
336        (0..self.sampled_axis_len(axis))
337            .map(|index| self.blc[axis] + index * self.inc[axis])
338            .collect()
339    }
340
341    pub(crate) fn nearest_sample_index(&self, axis: usize, pixel: usize) -> usize {
342        let start = self.blc[axis];
343        let end = self.trc[axis];
344        if pixel <= start {
345            return 0;
346        }
347        if pixel >= end {
348            return self.sampled_axis_len(axis).saturating_sub(1);
349        }
350        ((pixel - start) / self.inc[axis]).min(self.sampled_axis_len(axis).saturating_sub(1))
351    }

369    pub(crate) fn open_with_cache_bytes(
370        path: impl AsRef<Path>,
371        cache_bytes: usize,
372    ) -> Result<Self, ImageError> {
373        let path = path.as_ref().to_path_buf();
374        let image = open_view_image_with_cache(&path, cache_bytes)?;
375        let axis_model = determine_axis_model(&image);
376        let world_coords_available =
377            coordinates_cover_image_axes(image_coordinates(&image), image.shape())
378                && coordinate_system_pixel_to_world(
379                    image_coordinates(&image),
380                    &vec![0usize; image.shape().len()],
381                )
382                .is_ok();
383        let capabilities = ImageViewCapabilities {
384            renderable_plane: matches!(
385                image.pixel_type(),
386                ImagePixelType::Float32 | ImagePixelType::Float64
387            ) && axis_model.display_axes.is_some(),
388            world_coords_available,
389            pixel_only_mode: !world_coords_available,
390            non_display_axis_selectors: !axis_model.non_display_axes.is_empty(),
391            mask_present: image_has_pixel_mask(&image),
392            complex_unsupported: matches!(
393                image.pixel_type(),
394                ImagePixelType::Complex32 | ImagePixelType::Complex64
395            ),
396        };
397
398        Ok(Self {
399            path,
400            image,
401            capabilities,
402            axis_model,
403        })
404    }

486    /// Returns metadata/header/coordinate text sections for the inspector.
487    pub fn metadata_sections(&self) -> Result<Vec<ImageMetadataSection>, ImageError> {
488        let axis_descriptors = build_axis_descriptors(image_coordinates(&self.image), self.shape());
489        let default_mask_name =
490            image_default_mask_name(&self.image).unwrap_or_else(|| "none".into());
491        let history = image_history(&self.image)?;
492        let image_info = image_image_info(&self.image)?;
493        let beam_lines = format_beam_summary_lines(&image_info.beam_set)?;
494
495        let mut sections = Vec::new();
496        let mut summary_lines = vec![
497            format!("path: {}", self.path.display()),
498            format!("pixel type: {:?}", self.pixel_type()),
499            format!("shape: {:?}", self.shape()),
500            format!("units: {}", image_units(&self.image)),
501            format!("default mask: {default_mask_name}"),
502            format!("history entries: {}", history.len()),
503            format!("image type: {}", image_info.image_type),
504            format!(
505                "object name: {}",
506                if image_info.object_name.is_empty() {
507                    "<none>"
508                } else {
509                    &image_info.object_name
510                }
511            ),
512        ];
513        summary_lines.extend(beam_lines);
514        summary_lines.push(format!("status: {}", self.status_line()));
515        sections.push(ImageMetadataSection {
516            title: "Summary".into(),
517            lines: summary_lines,
518        });
519        sections.push(ImageMetadataSection {
520            title: "Observation".into(),
521            lines: build_observation_lines(image_coordinates(&self.image).obs_info()),
522        });
523        sections.push(ImageMetadataSection {
524            title: "Axes".into(),
525            lines: axis_descriptors
526                .iter()
527                .enumerate()
528                .map(|(index, axis)| {
529                    format!(
530                        "axis {index}: name={} type={} unit={} len={} ref_pix={} ref_val={} incr={}",
531                        axis.name,
532                        axis.coord_type,
533                        if axis.unit.is_empty() { "<none>" } else { &axis.unit },
534                        self.shape().get(index).copied().unwrap_or_default(),
535                        format_reference_pixel_for_display(axis.reference_pixel),
536                        format_axis_value_for_display(
537                            axis.reference_value,
538                            &axis.name,
539                            &axis.unit,
540                        ),
541                        format_axis_increment_for_display(axis.increment, &axis.name, &axis.unit),
542                    )
543                })
544                .collect(),
545        });
546
547        let mut coordinate_lines = Vec::new();
548        if self.capabilities.pixel_only_mode {
549            coordinate_lines.push("pixel-only mode: coordinate reconstruction unavailable".into());
550        }
551        if let Some(display_axes) = self.axis_model.display_axes {
552            coordinate_lines.push(format!(
553                "display axes: {} ({}) and {} ({})",
554                axis_descriptors[display_axes[0]].name,
555                display_axes[0],
556                axis_descriptors[display_axes[1]].name,
557                display_axes[1],
558            ));
559        }
560        for &axis in &self.axis_model.non_display_axes {
561            coordinate_lines.push(format!(
562                "non-display axis: {} ({axis}), index range 0..{}",
563                axis_names(&self.image)
564                    .get(axis)
565                    .map(String::as_str)
566                    .unwrap_or("Axis"),
567                self.shape()
568                    .get(axis)
569                    .copied()
570                    .unwrap_or_default()
571                    .saturating_sub(1),
572            ));
573        }
574        coordinate_lines.extend(build_coordinate_summary_lines(
575            image_coordinates(&self.image),
576            self.shape(),
577        ));
578        sections.push(ImageMetadataSection {
579            title: "Coordinates".into(),
580            lines: coordinate_lines,
581        });
582
583        let misc = image_misc_info(&self.image);
584        let misc_lines = if misc.is_empty() {
585            vec!["<none>".into()]
586        } else {
587            misc.fields()
588                .iter()
589                .map(|field| format!("{}: {}", field.name, format_value(&field.value)))
590                .collect()
591        };
592        sections.push(ImageMetadataSection {
593            title: "Misc".into(),
594            lines: misc_lines,
595        });
596
597        Ok(sections)
598    }

664    /// Renders the current 2D plane and returns a timing breakdown for the work.
665    pub(crate) fn render_plane_with_window_and_axes_and_stretch_timed(
666        &self,
667        viewport: (usize, usize),
668        window: &ImageViewWindow,
669        non_display_indices: &[usize],
670        stretch: &PlaneStretchSettings,
671        clip_override: Option<(f64, f64)>,
672        region: Option<&ImageRegion>,
673    ) -> Result<(PlaneRaster, PlaneRenderTelemetry), ImageError> {
674        let total_started_at = Instant::now();
675        if !self.capabilities.renderable_plane {
676            return Err(ImageError::InvalidMetadata(self.status_line()));
677        }
678        let extract_started_at = Instant::now();
679        let (plane, mask) = self.read_plane(window, non_display_indices)?;
680        let plane_extract_ns = duration_ns(extract_started_at.elapsed());
681        let stats_started_at = Instant::now();
682        let plane_stats = if let Some(region) =
683            region.filter(|region| region_has_closed_shapes(region))
684        {
685            let region_mask = self.region_dense_plane_mask(region, window, non_display_indices)?;
686            let (dense_plane, dense_mask) = self.read_dense_plane(window, non_display_indices)?;
687            let combined_mask = combine_masks(dense_mask.as_ref(), Some(&region_mask));
688            collect_plane_stats(&dense_plane, combined_mask.as_ref())
689        } else {
690            collect_plane_stats(&plane, mask.as_ref())
691        };
692        let stat_collection_ns = duration_ns(stats_started_at.elapsed());
693        let width = viewport.0.max(1).min(plane.shape()[0].max(1));
694        let height = viewport.1.max(1).min(plane.shape()[1].max(1));
695        let (clip_min, clip_max) = clip_override
696            .or_else(|| plane_stats.clip_bounds_for(stretch))
697            .unwrap_or((0.0, 0.0));
698
699        let rasterize_started_at = Instant::now();
700        let mut pixels_u8 = Vec::with_capacity(width * height);
701        for y in 0..height {
702            let y0 = y * plane.shape()[1] / height;
703            let y1 = ((y + 1) * plane.shape()[1]).div_ceil(height);
704            for x in 0..width {
705                let x0 = x * plane.shape()[0] / width;
706                let x1 = ((x + 1) * plane.shape()[0]).div_ceil(width);
707                let mut sum = 0.0;
708                let mut count = 0usize;
709                for src_x in x0..x1.max(x0 + 1) {
710                    for src_y in y0..y1.max(y0 + 1) {
711                        if !plane_stats.is_valid(mask.as_ref(), src_x, src_y) {
712                            continue;
713                        }
714                        sum += plane[[src_x, src_y]];
715                        count += 1;
716                    }
717                }
718                let sample = if count == 0 || plane_stats.no_finite_values {
719                    0
720                } else if (clip_max - clip_min).abs() < f64::EPSILON {
721                    128
722                } else {
723                    let scaled = ((sum / count as f64) - clip_min) / (clip_max - clip_min);
724                    (scaled.clamp(0.0, 1.0) * 255.0).round() as u8
725                };
726                pixels_u8.push(sample);
727            }
728        }
729        let rasterize_ns = duration_ns(rasterize_started_at.elapsed());
730        let histogram_started_at = Instant::now();
731        let histogram_bins = plane_stats.histogram_bins(48);
732        let histogram_ns = duration_ns(histogram_started_at.elapsed());
733
734        Ok((
735            PlaneRaster {
736                width,
737                height,
738                pixels_u8,
739                clip_min,
740                clip_max,
741                data_min: plane_stats.data_min.unwrap_or(0.0),
742                data_max: plane_stats.data_max.unwrap_or(0.0),
743                value_unit: image_units(&self.image).to_string(),
744                histogram_bins,
745                masked_or_non_finite_count: plane_stats.masked_or_non_finite_count,
746                no_finite_values: plane_stats.no_finite_values,
747            },
748            PlaneRenderTelemetry {
749                plane_extract_ns,
750                stat_collection_ns,
751                histogram_ns,
752                rasterize_ns,
753                total_plane_ns: duration_ns(total_started_at.elapsed()),
754            },
755        ))
756    }

881    /// Returns the current cursor probe for an explicit view window and
882    /// non-display-axis sample selections.
883    pub fn probe_with_window_and_axes(
884        &self,
885        pixel_xy: (usize, usize),
886        window: &ImageViewWindow,
887        non_display_indices: &[usize],
888    ) -> Result<ImageProbe, ImageError> {
889        let display_axes = self
890            .axis_model
891            .display_axes
892            .ok_or_else(|| ImageError::InvalidMetadata(self.status_line()))?;
893        let sampled_width = window.sampled_axis_len(display_axes[0]);
894        let sampled_height = window.sampled_axis_len(display_axes[1]);
895        if pixel_xy.0 >= sampled_width || pixel_xy.1 >= sampled_height {
896            return Err(ImageError::ShapeMismatch {
897                expected: vec![sampled_width, sampled_height],
898                got: vec![pixel_xy.0, pixel_xy.1],
899            });
900        }
901        let mut pixel_indices = window.blc.clone();
902        pixel_indices[display_axes[0]] = window
903            .sampled_axis_value(display_axes[0], pixel_xy.0)
904            .ok_or_else(|| ImageError::ShapeMismatch {
905                expected: vec![sampled_width],
906                got: vec![pixel_xy.0],
907            })?;
908        pixel_indices[display_axes[1]] = window
909            .sampled_axis_value(display_axes[1], pixel_xy.1)
910            .ok_or_else(|| ImageError::ShapeMismatch {
911                expected: vec![sampled_height],
912                got: vec![pixel_xy.1],
913            })?;
914        apply_non_display_axis_selections(
915            &mut pixel_indices,
916            window,
917            &self.axis_model.non_display_axes,
918            non_display_indices,
919        )?;
920        for (axis, &pixel) in pixel_indices.iter().enumerate() {
921            if pixel >= self.shape()[axis] {
922                return Err(ImageError::ShapeMismatch {
923                    expected: self.shape().to_vec(),
924                    got: pixel_indices.clone(),
925                });
926            }
927        }
928
929        let value = image_real_get_at(&self.image, &pixel_indices)?;
930        let masked = image_get_mask_value(&self.image, &pixel_indices)?.is_some_and(|value| !value);
931        let axis_descriptors = build_axis_descriptors(image_coordinates(&self.image), self.shape());
932        let pixel_axes = pixel_indices
933            .iter()
934            .enumerate()
935            .map(|(index, &pixel)| ImageAxisValue {
936                name: axis_descriptors[index].name.clone(),
937                unit: "px".into(),
938                value: pixel as f64,
939            })
940            .collect();
941        let world_axes = if self.capabilities.world_coords_available {
942            let world = self.pixel_indices_to_world(&pixel_indices)?;
943            axis_descriptors
944                .iter()
945                .enumerate()
946                .map(|(index, axis)| ImageAxisValue {
947                    name: axis.name.clone(),
948                    unit: axis.unit.clone(),
949                    value: world[index],
950                })
951                .collect()
952        } else {
953            Vec::new()
954        };
955
956        Ok(ImageProbe {
957            pixel_indices,
958            pixel_axes,
959            value,
960            masked,
961            finite: value.is_finite(),
962            world_axes,
963        })
964    }

1150    /// Computes active-region statistics for the current plane window.
1151    pub fn region_stats_with_window_and_axes(
1152        &self,
1153        region: &ImageRegion,
1154        window: &ImageViewWindow,
1155        non_display_indices: &[usize],
1156    ) -> Result<Option<ImageRegionStats>, ImageError> {
1157        let display_axes = self
1158            .axis_model
1159            .display_axes
1160            .ok_or_else(|| ImageError::InvalidMetadata(self.status_line()))?;
1161        validate_region_axes(region, display_axes)?;
1162        if !region_has_closed_shapes(region) {
1163            return Ok(None);
1164        }
1165        let dense_mask = self.region_dense_plane_mask(region, window, non_display_indices)?;
1166        let (plane, mask) = self.read_dense_plane(window, non_display_indices)?;
1167        let stats = collect_plane_stats(
1168            &plane,
1169            combine_masks(mask.as_ref(), Some(&dense_mask)).as_ref(),
1170        );
1171        let values = stats.finite_values;
1172        if values.is_empty() {
1173            return Ok(None);
1174        }
1175        let pixel_count = values.len();
1176        let sum = values.iter().copied().sum::<f64>();
1177        let sum_sq = values.iter().map(|value| value * value).sum::<f64>();
1178        let mean = sum / pixel_count as f64;
1179        let variance = values
1180            .iter()
1181            .map(|value| {
1182                let delta = *value - mean;
1183                delta * delta
1184            })
1185            .sum::<f64>()
1186            / pixel_count as f64;
1187        let sigma = variance.sqrt();
1188        let rms = (sum_sq / pixel_count as f64).sqrt();
1189        let median = region_stat_median(&values).unwrap_or(f64::NAN);
1190        Ok(Some(ImageRegionStats {
1191            pixel_count,
1192            median,
1193            min: values
1194                .iter()
1195                .copied()
1196                .fold(f64::INFINITY, |current, value| current.min(value)),
1197            max: values
1198                .iter()
1199                .copied()
1200                .fold(f64::NEG_INFINITY, |current, value| current.max(value)),
1201            mean,
1202            sigma,
1203            rms,
1204            sum,
1205            value_unit: image_units(&self.image).to_string(),
1206        }))
1207    }

1280                "a saved region named '{trimmed}' already exists"
1281            )));
1282        }
1283        let record = if self.path.as_os_str().is_empty() {
1284            self.image.get_region_record(old_name)?
1285        } else {
1286            AnyPagedImage::open(&self.path)?.get_region_record(old_name)?
1287        };
1288        let renamed = rename_native_region_record_name(&record, trimmed);
1289        save_native_region_record(&self.path, self.pixel_type(), trimmed, &renamed)?;
1290        remove_native_region_record(&self.path, self.pixel_type(), old_name)?;
1291        Ok(trimmed.to_string())
1292    }

1412        let display_axes = self
1413            .axis_model
1414            .display_axes

1418            .blc
1419            .iter()
1420            .zip(window.trc.iter())
1421            .map(|(blc, trc)| trc - blc + 1)
1422            .collect::<Vec<_>>();

1424            window,
1425            &self.axis_model.non_display_axes,
1426            non_display_indices,
1427        )? {
1428            start[axis] = pixel;
1429            shape[axis] = 1;
1430        }

1437            image_real_get_slice(&self.image, &start, &shape)?,

1439        );

1471        validate_region_axes(region, display_axes)?;

1473        let polygons = region

1476            .filter(|shape| shape.closed && shape.vertices.len() >= 3)

1502                .iter()

1505    }

1564    /// Extracts a 1D spectrum/profile along the requested non-display axis.
1565    pub fn profile_with_window_and_axes(
1566        &self,
1567        pixel_xy: (usize, usize),
1568        window: &ImageViewWindow,
1569        non_display_indices: &[usize],
1570        profile_axis: usize,
1571    ) -> Result<ImageProfile, ImageError> {
1572        let display_axes = self
1573            .axis_model
1574            .display_axes
1575            .ok_or_else(|| ImageError::InvalidMetadata(self.status_line()))?;
1576        let sampled_width = window.sampled_axis_len(display_axes[0]);
1577        let sampled_height = window.sampled_axis_len(display_axes[1]);
1578        if pixel_xy.0 >= sampled_width || pixel_xy.1 >= sampled_height {
1579            return Err(ImageError::ShapeMismatch {
1580                expected: vec![sampled_width, sampled_height],
1581                got: vec![pixel_xy.0, pixel_xy.1],
1582            });
1583        }
1584        let profile_position = self
1585            .axis_model
1586            .non_display_axes
1587            .iter()
1588            .position(|axis| *axis == profile_axis)
1589            .ok_or_else(|| {
1590                ImageError::InvalidMetadata(format!(
1591                    "axis {profile_axis} is not a non-display axis in this view"
1592                ))
1593            })?;
1594        let selections = selected_non_display_axes(
1595            window,
1596            &self.axis_model.non_display_axes,
1597            non_display_indices,
1598        )?;
1599        let descriptors = build_axis_descriptors(image_coordinates(&self.image), self.shape());
1600        let profile_descriptor =
1601            descriptors
1602                .get(profile_axis)
1603                .cloned()
1604                .unwrap_or_else(|| AxisDescriptor {
1605                    coord_type: CoordinateType::Linear,
1606                    name: format!("Axis{profile_axis}"),
1607                    unit: String::new(),
1608                    reference_value: None,
1609                    reference_pixel: None,
1610                    increment: None,
1611                });
1612        let selected_sample_index = selections
1613            .get(profile_position)
1614            .map(|(_, index, _)| *index)
1615            .unwrap_or_default();
1616
1617        let mut fixed_pixel_indices = window.blc.clone();
1618        fixed_pixel_indices[display_axes[0]] = window
1619            .sampled_axis_value(display_axes[0], pixel_xy.0)
1620            .ok_or_else(|| ImageError::ShapeMismatch {
1621                expected: vec![sampled_width],
1622                got: vec![pixel_xy.0],
1623            })?;
1624        fixed_pixel_indices[display_axes[1]] = window
1625            .sampled_axis_value(display_axes[1], pixel_xy.1)
1626            .ok_or_else(|| ImageError::ShapeMismatch {
1627                expected: vec![sampled_height],
1628                got: vec![pixel_xy.1],
1629            })?;
1630        for &(axis, _index, pixel) in &selections {
1631            fixed_pixel_indices[axis] = pixel;
1632        }
1633
1634        let axis_len = window.sampled_axis_len(profile_axis);
1635        let mut mask_start = fixed_pixel_indices.clone();
1636        let mut mask_shape = vec![1; self.shape().len()];
1637        let mut mask_stride = vec![1; self.shape().len()];
1638        mask_start[profile_axis] = window.blc[profile_axis];
1639        mask_shape[profile_axis] = axis_len;
1640        mask_stride[profile_axis] = window.inc[profile_axis];
1641        let mask = image_get_mask_slice(&self.image, &mask_start, &mask_shape, &mask_stride)?
1642            .map(|mask| {
1643                let squeeze_axes = (0..mask.ndim())
1644                    .filter(|&axis| axis != profile_axis)
1645                    .collect::<Vec<_>>();
1646                squeeze_plane_axes(mask, &squeeze_axes)
1647                    .into_dimensionality::<Ix1>()
1648                    .map_err(|_| {
1649                        ImageError::InvalidMetadata(format!(
1650                            "expected 1D mask profile for axis {profile_axis}"
1651                        ))
1652                    })
1653            })
1654            .transpose()?;
1655        let samples = (0..axis_len)
1656            .map(|sample_index| {
1657                let pixel = window
1658                    .sampled_axis_value(profile_axis, sample_index)
1659                    .ok_or_else(|| ImageError::ShapeMismatch {
1660                        expected: vec![axis_len],
1661                        got: vec![sample_index],
1662                    })?;
1663                let mut pixel_indices = fixed_pixel_indices.clone();
1664                pixel_indices[profile_axis] = pixel;
1665                let value = image_real_get_at(&self.image, &pixel_indices)?;
1666                let masked = mask.as_ref().is_some_and(|data| !data[sample_index]);
1667                let world_axis = if self.capabilities.world_coords_available {
1668                    let world = self.pixel_indices_to_world(&pixel_indices)?;
1669                    Some(ImageAxisValue {
1670                        name: profile_descriptor.name.clone(),
1671                        unit: profile_descriptor.unit.clone(),
1672                        value: world[profile_axis],
1673                    })
1674                } else {
1675                    None
1676                };
1677                Ok(ImageProfileSample {
1678                    sample_index,
1679                    pixel_index: pixel,
1680                    value,
1681                    masked,
1682                    finite: value.is_finite(),
1683                    world_axis,
1684                })
1685            })
1686            .collect::<Result<Vec<_>, ImageError>>()?;
1687
1688        Ok(ImageProfile {
1689            axis: profile_axis,
1690            axis_name: profile_descriptor.name,
1691            axis_unit: profile_descriptor.unit,
1692            value_unit: image_units(&self.image).to_string(),
1693            coord_type: profile_descriptor.coord_type,
1694            selected_sample_index,
1695            samples,
1696        })
1697    }

1991fn validate_window(
1992    shape: &[usize],
1993    blc: &[usize],
1994    trc: &[usize],
1995    inc: &[usize],
1996) -> Result<(), ImageError> {
1997    if blc.len() != shape.len() || trc.len() != shape.len() || inc.len() != shape.len() {
1998        return Err(ImageError::InvalidMetadata(
1999            "BLC/TRC/INC dimensionality does not match image shape".into(),
2000        ));
2001    }
2002    for axis in 0..shape.len() {
2003        if shape[axis] == 0 {
2004            return Err(ImageError::InvalidMetadata(format!(
2005                "axis {axis} has zero length"
2006            )));
2007        }
2008        if inc[axis] == 0 {
2009            return Err(ImageError::InvalidMetadata(format!(
2010                "INC axis {axis} must be >= 1"
2011            )));
2012        }
2013        if blc[axis] >= shape[axis] {
2014            return Err(ImageError::InvalidMetadata(format!(
2015                "BLC axis {axis}={} is outside image length {}",
2016                blc[axis], shape[axis]
2017            )));
2018        }
2019        if trc[axis] >= shape[axis] {
2020            return Err(ImageError::InvalidMetadata(format!(
2021                "TRC axis {axis}={} is outside image length {}",
2022                trc[axis], shape[axis]
2023            )));
2024        }
2025        if blc[axis] > trc[axis] {
2026            return Err(ImageError::InvalidMetadata(format!(
2027                "BLC axis {axis}={} must be <= TRC {}",
2028                blc[axis], trc[axis]
2029            )));
2030        }
2031    }
2032    Ok(())
2033}

2508fn format_frequency_quantity_auto(value: f64, unit: &str) -> Option<String> {
2509    let quantity = Quantity::new(value, unit).ok()?;
2510    let hz = Unit::new("Hz").ok()?;
2511    if !quantity.unit().conformant(&hz) {
2512        return None;
2513    }
2514    let abs_hz = quantity.get_value_in(&hz).ok()?.abs();
2515    let display_unit = if abs_hz >= 1e9 {
2516        "GHz"
2517    } else if abs_hz >= 1e6 {
2518        "MHz"
2519    } else if abs_hz >= 1e3 {
2520        "kHz"
2521    } else {
2522        "Hz"
2523    };
2524    let converted = quantity.get_value_in(&Unit::new(display_unit).ok()?).ok()?;
2525    Some(format!(
2526        "{} {display_unit}",
2527        trim_float_text(format!("{converted:.6}"))
2528    ))
2529}

2651fn coordinate_header_details(coord_type: CoordinateType, record: &RecordValue) -> String {
2652    match coord_type {
2653        CoordinateType::Direction => {
2654            let frame = record_string(record, "direction_ref").unwrap_or("unknown");
2655            let projection = record_string(record, "projection").unwrap_or("unknown");
2656            format!("frame={frame} projection={projection}")
2657        }
2658        CoordinateType::Spectral => {
2659            let native_frame = record_string(record, "frequency_ref").unwrap_or("unknown");
2660            let frame = record_subrecord(record, "conversion")
2661                .and_then(|conversion| record_string(conversion, "system"))
2662                .unwrap_or(native_frame);
2663            let restfreq = record_f64(record, "restfreq")
2664                .map(|value| format!(" restfreq={}", format_numeric_value_with_unit(value, "Hz")))
2665                .unwrap_or_default();
2666            if frame == native_frame {
2667                format!("frame={frame}{restfreq}")
2668            } else {
2669                format!("frame={frame} native={native_frame}{restfreq}")
2670            }
2671        }
2672        CoordinateType::Stokes => {
2673            let stokes = record_stokes_values(record);
2674            if stokes.is_empty() {
2675                "values=<unknown>".into()
2676            } else {
2677                format!("values={}", stokes.join(","))
2678            }
2679        }
2680        CoordinateType::Linear => "linear mapping".into(),
2681        CoordinateType::Tabular => {
2682            let name = record_string(record, "name").unwrap_or("lookup");
2683            format!("lookup={name}")
2684        }
2685    }
2686}

2691        _ => None,

2698        _ => None,
2699    }

2703    match record.get(key) {
2704        Some(Value::Scalar(ScalarValue::Float64(value))) => Some(*value),
2705        Some(Value::Scalar(ScalarValue::Float32(value))) => Some(f64::from(*value)),
2706        _ => None,

2710fn record_stokes_values(record: &RecordValue) -> Vec<String> {
2711    match record.get("stokes") {

2717                    .unwrap_or_else(|| value.to_string())
2718            })
2719            .collect(),
2720        Some(Value::Array(ArrayValue::String(values))) => values.iter().cloned().collect(),
2721        _ => Vec::new(),
2722    }
2723}

2726    match value {
2727        Value::Scalar(ScalarValue::Bool(value)) => value.to_string(),
2728        Value::Scalar(ScalarValue::UInt8(value)) => value.to_string(),
2729        Value::Scalar(ScalarValue::UInt16(value)) => value.to_string(),
2730        Value::Scalar(ScalarValue::UInt32(value)) => value.to_string(),
2731        Value::Scalar(ScalarValue::Int16(value)) => value.to_string(),
2732        Value::Scalar(ScalarValue::Int32(value)) => value.to_string(),
2733        Value::Scalar(ScalarValue::Int64(value)) => value.to_string(),
2734        Value::Scalar(ScalarValue::Float32(value)) => value.to_string(),
2735        Value::Scalar(ScalarValue::Float64(value)) => value.to_string(),
2736        Value::Scalar(ScalarValue::Complex32(value)) => format!("{value:?}"),

2738        Value::Scalar(ScalarValue::String(value)) => value.clone(),
2739        Value::Array(array) => format!("{:?} array {:?}", array.primitive_type(), array.shape()),
2740        Value::Record(record) => format!("record({} fields)", record.len()),
