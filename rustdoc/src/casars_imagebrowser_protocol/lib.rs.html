14/// Stable protocol name advertised by `imexplore --protocol-info`.
15pub const IMAGEBROWSER_SESSION_PROTOCOL_NAME: &str = "casa_imagebrowser_session";

17/// Current JSON protocol version.

20/// Version/compatibility information for the imagebrowser session protocol.
21#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize, JsonSchema)]
22pub struct ImageBrowserProtocolInfo {
23    /// Stable protocol identifier.
24    pub protocol_name: String,
25    /// Monotonic protocol version for compatibility checks.
26    pub protocol_version: u32,
27    /// Provider surface kind defined by the shared architecture contract.
28    pub surface_kind: ProviderSurfaceKind,
29    /// Binary version implementing the protocol.
30    pub binary_version: String,
31}
32
33impl ImageBrowserProtocolInfo {
34    /// Build the current imagebrowser session protocol descriptor.
35    pub fn current() -> Self {
36        Self {
37            protocol_name: IMAGEBROWSER_SESSION_PROTOCOL_NAME.to_string(),
38            protocol_version: PROTOCOL_VERSION,
39            surface_kind: ProviderSurfaceKind::Session,
40            binary_version: env!("CARGO_PKG_VERSION").to_string(),
41        }
42    }
43}

45/// Canonical JSON-schema bundle for the public imagebrowser session protocol.
46#[derive(Debug, Clone, Serialize)]
47pub struct ImageBrowserSessionSchemaBundle {
48    /// Compatibility descriptor for the session protocol.
49    pub protocol: ImageBrowserProtocolInfo,
50    /// Canonical semantic session contract.
51    pub semantic: SessionSemanticContract,
52    /// Shared component schemas reusable across projections.
53    pub components: ProviderComponentSchemas,
54    /// Presentation annotations carried with the canonical bundle.
55    pub annotations: JsonValue,
56    /// Derived projection metadata for UI and CLI consumers.
57    pub projections: ProviderProjectionMetadata,
58    /// Canonical durable parameter contract for the session.
59    pub parameter_surfaces: Vec<SurfaceContractBundle>,
60    /// JSON schema for [`ImageBrowserRequestEnvelope`].
61    pub request_schema: schemars::schema::RootSchema,
62    /// JSON schema for [`ImageBrowserResponseEnvelope`].
63    pub response_schema: schemars::schema::RootSchema,
64}
65
66impl ImageBrowserSessionSchemaBundle {
67    /// Build the current imagebrowser schema bundle.
68    pub fn current() -> Self {
69        let request_schema = schema_for!(ImageBrowserRequestEnvelope);
70        let response_schema = schema_for!(ImageBrowserResponseEnvelope);
71        let canonical_surface = builtin_surface_bundle("imexplore")
72            .expect("built-in imexplore parameter surface must remain valid");
73        let ui_schema = project_ui_schema(&canonical_surface);
74        Self {
75            protocol: ImageBrowserProtocolInfo::current(),
76            semantic: SessionSemanticContract {
77                transport: "jsonl_stdio".to_string(),
78                request_schema: request_schema.clone(),
79                response_schema: response_schema.clone(),
80            },
81            components: merged_components([&request_schema, &response_schema]),
82            annotations: derived_ui_schema_annotations(),
83            projections: ProviderProjectionMetadata {
84                cli: Some(ProviderCliProjection {
85                    machine_actions: ProviderCliMachineActions {
86                        ui_schema: Some("--ui-schema".to_string()),
87                        json_schema: Some("--json-schema".to_string()),
88                        protocol_info: Some("--protocol-info".to_string()),
89                        json_run: None,
90                        session: Some("--session".to_string()),
91                    },
92                }),
93                ui_schema: Some(ui_schema),
94                python: None,
95            },
96            parameter_surfaces: vec![canonical_surface],
97            request_schema,
98            response_schema,
99        }
100    }
101
102    /// Return the launcher/TUI compatibility view projected from the bundle.
103    pub fn ui_schema_projection(&self) -> Result<JsonValue, String> {
104        self.projections
105            .ui_schema
106            .clone()
107            .ok_or_else(|| "missing ui_schema projection".to_string())
108    }
109}

295/// JSON Lines response envelope sent from the backend to `casars`.

302impl ImageBrowserResponseEnvelope {
303    pub fn snapshot(snapshot: ImageBrowserSnapshot) -> Self {
304        Self {
305            version: PROTOCOL_VERSION,
306            response: ImageBrowserResponse::Snapshot(Box::new(snapshot)),
307        }
308    }
309
310    pub fn preview(payload: ImageBrowserPreviewPayload) -> Self {
311        Self {
312            version: PROTOCOL_VERSION,
313            response: ImageBrowserResponse::Preview(Box::new(payload)),
314        }
315    }
316
317    pub fn error(code: impl Into<String>, message: impl Into<String>) -> Self {
318        Self {
319            version: PROTOCOL_VERSION,
320            response: ImageBrowserResponse::Error(ImageBrowserErrorPayload {
321                code: code.into(),
322                message: message.into(),
323            }),
324        }
325    }

885            IMAGEBROWSER_SESSION_PROTOCOL_NAME

890        assert!(bundle.components.contains_key("ImageBrowserCommand"));
891        assert!(bundle.components.contains_key("ImageBrowserResponse"));

894        bundle.parameter_surfaces[0]
895            .validate()
896            .expect("embedded imexplore parameter surface");

898            serde_json::to_value(&bundle).unwrap()["parameter_surfaces"]
899                .as_array()
900                .unwrap()
901                .len(),
902            1
903        );
