13/// Stable protocol name advertised by `imexplore --protocol-info`.
14pub const IMAGEBROWSER_SESSION_PROTOCOL_NAME: &str = "casa_imagebrowser_session";

16/// Current JSON protocol version.
17pub const PROTOCOL_VERSION: u32 = 1;

19/// Version/compatibility information for the imagebrowser session protocol.
20#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize, JsonSchema)]
21pub struct ImageBrowserProtocolInfo {
22    /// Stable protocol identifier.
23    pub protocol_name: String,
24    /// Monotonic protocol version for compatibility checks.
25    pub protocol_version: u32,
26    /// Provider surface kind defined by the shared architecture contract.
27    pub surface_kind: ProviderSurfaceKind,
28    /// Binary version implementing the protocol.
29    pub binary_version: String,
30}
31
32impl ImageBrowserProtocolInfo {
33    /// Build the current imagebrowser session protocol descriptor.
34    pub fn current() -> Self {
35        Self {
36            protocol_name: IMAGEBROWSER_SESSION_PROTOCOL_NAME.to_string(),
37            protocol_version: PROTOCOL_VERSION,
38            surface_kind: ProviderSurfaceKind::Session,
39            binary_version: env!("CARGO_PKG_VERSION").to_string(),
40        }
41    }
42}

44/// Canonical JSON-schema bundle for the public imagebrowser session protocol.
45#[derive(Debug, Clone, Serialize)]
46pub struct ImageBrowserSessionSchemaBundle {
47    /// Compatibility descriptor for the session protocol.
48    pub protocol: ImageBrowserProtocolInfo,
49    /// Canonical semantic session contract.
50    pub semantic: SessionSemanticContract,
51    /// Shared component schemas reusable across projections.
52    pub components: ProviderComponentSchemas,
53    /// Presentation annotations carried with the canonical bundle.
54    pub annotations: JsonValue,
55    /// Derived projection metadata for UI and CLI consumers.
56    pub projections: ProviderProjectionMetadata,
57    /// JSON schema for [`ImageBrowserRequestEnvelope`].
58    pub request_schema: schemars::schema::RootSchema,
59    /// JSON schema for [`ImageBrowserResponseEnvelope`].
60    pub response_schema: schemars::schema::RootSchema,
61}
62
63impl ImageBrowserSessionSchemaBundle {
64    /// Build the current imagebrowser schema bundle.
65    pub fn current(ui_schema: JsonValue) -> Self {
66        let request_schema = schema_for!(ImageBrowserRequestEnvelope);
67        let response_schema = schema_for!(ImageBrowserResponseEnvelope);
68        Self {
69            protocol: ImageBrowserProtocolInfo::current(),
70            semantic: SessionSemanticContract {
71                transport: "jsonl_stdio".to_string(),
72                request_schema: request_schema.clone(),
73                response_schema: response_schema.clone(),
74            },
75            components: merged_components([&request_schema, &response_schema]),
76            annotations: derived_ui_schema_annotations(),
77            projections: ProviderProjectionMetadata {
78                cli: Some(ProviderCliProjection {
79                    machine_actions: ProviderCliMachineActions {
80                        ui_schema: Some("--ui-schema".to_string()),
81                        json_schema: Some("--json-schema".to_string()),
82                        protocol_info: Some("--protocol-info".to_string()),
83                        json_run: None,
84                        session: Some("--session".to_string()),
85                    },
86                }),
87                ui_schema: Some(ui_schema),
88                python: None,
89            },
90            request_schema,
91            response_schema,
92        }
93    }
94
95    /// Return the launcher/TUI compatibility view projected from the bundle.
96    pub fn ui_schema_projection(&self) -> Result<JsonValue, String> {
97        self.projections
98            .ui_schema
99            .clone()
100            .ok_or_else(|| "missing ui_schema projection".to_string())
101    }
102}

273/// JSON Lines response envelope sent from the backend to `casars`.
274#[derive(Debug, Clone, Serialize, Deserialize, JsonSchema, PartialEq)]
275pub struct ImageBrowserResponseEnvelope {
276    pub version: u32,
277    pub response: ImageBrowserResponse,
278}
279
280impl ImageBrowserResponseEnvelope {
281    pub fn snapshot(snapshot: ImageBrowserSnapshot) -> Self {
282        Self {
283            version: PROTOCOL_VERSION,
284            response: ImageBrowserResponse::Snapshot(Box::new(snapshot)),
285        }
286    }
287
288    pub fn preview(payload: ImageBrowserPreviewPayload) -> Self {
289        Self {
290            version: PROTOCOL_VERSION,
291            response: ImageBrowserResponse::Preview(Box::new(payload)),
292        }
293    }
294
295    pub fn error(code: impl Into<String>, message: impl Into<String>) -> Self {
296        Self {
297            version: PROTOCOL_VERSION,
298            response: ImageBrowserResponse::Error(ImageBrowserErrorPayload {
299                code: code.into(),
300                message: message.into(),
301            }),
302        }
303    }
