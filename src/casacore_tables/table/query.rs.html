67    /// Returns row indices present in **either** `a` or `b` (union).
68    ///
69    /// The result is sorted and deduplicated. If both inputs are already
70    /// sorted, this runs in O(n) via a merge; otherwise it falls back to
71    /// sort + dedup.
72    ///
73    /// # C++ equivalent
74    ///
75    /// `TableExprNode::operator|` (set union on row numbers).
76    pub fn row_union(a: &[usize], b: &[usize]) -> Vec<usize> {
77        if a.is_sorted() && b.is_sorted() {
78            let mut result = Vec::with_capacity(a.len() + b.len());
79            let (mut i, mut j) = (0, 0);
80            while i < a.len() && j < b.len() {
81                match a[i].cmp(&b[j]) {
82                    std::cmp::Ordering::Less => {
83                        result.push(a[i]);
84                        i += 1;
85                    }
86                    std::cmp::Ordering::Greater => {
87                        result.push(b[j]);
88                        j += 1;
89                    }
90                    std::cmp::Ordering::Equal => {
91                        result.push(a[i]);
92                        i += 1;
93                        j += 1;
94                    }
95                }
96            }
97            result.extend_from_slice(&a[i..]);
98            result.extend_from_slice(&b[j..]);
99            result
100        } else {
101            let mut set: Vec<usize> = a.iter().chain(b.iter()).copied().collect();
102            set.sort_unstable();
103            set.dedup();
104            set
105        }
106    }

108    /// Returns row indices present in **both** `a` and `b` (intersection).
109    ///
110    /// The result is sorted. If both inputs are already sorted, this runs
111    /// in O(n) via a merge; otherwise it falls back to hash + sort.
112    ///
113    /// # C++ equivalent
114    ///
115    /// `TableExprNode::operator&` (set intersection on row numbers).
116    pub fn row_intersection(a: &[usize], b: &[usize]) -> Vec<usize> {
117        if a.is_sorted() && b.is_sorted() {
118            let mut result = Vec::new();
119            let (mut i, mut j) = (0, 0);
120            while i < a.len() && j < b.len() {
121                match a[i].cmp(&b[j]) {
122                    std::cmp::Ordering::Less => i += 1,
123                    std::cmp::Ordering::Greater => j += 1,
124                    std::cmp::Ordering::Equal => {
125                        result.push(a[i]);
126                        i += 1;
127                        j += 1;
128                    }
129                }
130            }
131            result
132        } else {
133            let set_b: std::collections::HashSet<usize> = b.iter().copied().collect();
134            let mut result: Vec<usize> = a.iter().copied().filter(|x| set_b.contains(x)).collect();
135            result.sort_unstable();
136            result.dedup();
137            result
138        }
139    }

142    ///
143    /// The result is sorted. If both inputs are already sorted, this runs
144    /// in O(n) via a merge; otherwise it falls back to hash + sort.
145    ///
146    /// # C++ equivalent
147    ///
148    /// `TableExprNode::operator-` (set difference on row numbers).
149    pub fn row_difference(a: &[usize], b: &[usize]) -> Vec<usize> {
150        if a.is_sorted() && b.is_sorted() {
151            let mut result = Vec::new();
152            let (mut i, mut j) = (0, 0);
153            while i < a.len() && j < b.len() {
154                match a[i].cmp(&b[j]) {
155                    std::cmp::Ordering::Less => {
156                        result.push(a[i]);
157                        i += 1;
158                    }
159                    std::cmp::Ordering::Greater => j += 1,
160                    std::cmp::Ordering::Equal => {
161                        i += 1;
162                        j += 1;
163                    }
164                }
165            }
166            result.extend_from_slice(&a[i..]);
167            result
168        } else {
169            let set_b: std::collections::HashSet<usize> = b.iter().copied().collect();
170            let mut result: Vec<usize> = a.iter().copied().filter(|x| !set_b.contains(x)).collect();
171            result.sort_unstable();

173            result

181    /// Reads a sub-region of an array cell.
182    ///
183    /// Returns a new `Value::Array` containing only the elements selected by
184    /// the [`Slicer`]. The cell must be an array-valued cell; returns
185    /// [`TableError::CellNotArray`] otherwise.
186    ///
187    /// # C++ equivalent
188    ///
189    /// `ArrayColumn<T>::getSlice(rownr, slicer)`.
190    pub fn get_cell_slice(
191        &self,
192        column: &str,
193        row: usize,
194        slicer: &Slicer,
195    ) -> Result<Value, TableError> {
196        let cell = self
197            .cell(row, column)?
198            .ok_or_else(|| TableError::ColumnNotFound {
199                row_index: row,
200                column: column.to_string(),
201            })?;
202        match cell {
203            Value::Array(av) => {
204                let shape = av.shape();
205                validate_slicer_bounds(slicer, shape, row, column)?;
206                Ok(Value::Array(slice_array_value(av, slicer)))
207            }
208            _ => Err(TableError::CellNotArray {
209                row,
210                column: column.to_string(),
211            }),
212        }
213    }

215    /// Writes a sub-region of an array cell.
216    ///
217    /// Loads the full cell, replaces the slice region with `data`, and writes
218    /// the updated array back. Both the existing cell and `data` must be
219    /// arrays.
220    ///
221    /// # C++ equivalent
222    ///
223    /// `ArrayColumn<T>::putSlice(rownr, slicer, array)`.
224    pub fn put_cell_slice(
225        &mut self,
226        column: &str,
227        row: usize,
228        slicer: &Slicer,
229        data: &ArrayValue,
230    ) -> Result<(), TableError> {
231        let _ = self.row(row)?;
232        let cell = self
233            .inner
234            .row_mut(row)?
235            .expect("validated row must exist")
236            .get_mut(column)
237            .ok_or_else(|| TableError::ColumnNotFound {
238                row_index: row,
239                column: column.to_string(),
240            })?;
241        match cell {
242            Value::Array(av) => {
243                let shape = av.shape();
244                validate_slicer_bounds(slicer, shape, row, column)?;
245                put_slice_array_value(av, slicer, data);
246                Ok(())
247            }
248            _ => Err(TableError::CellNotArray {
249                row,
250                column: column.to_string(),
251            }),
252        }
253    }

255    /// Reads a sub-region of an array cell for each row in `row_range`,
256    /// returning one sliced value per row.
257    ///
258    /// Combines row selection ([`RowRange`]) with array slicing ([`Slicer`]).
259    /// Each returned element is the slice of the array cell for that row.
260    ///
261    /// # Errors
262    ///
263    /// - [`TableError::CellNotArray`] if a cell in the column is scalar
264    /// - [`TableError::SlicerDimensionMismatch`] if slicer ndim != array ndim
265    /// - [`TableError::SlicerOutOfBounds`] if slicer exceeds array shape
266    /// - [`TableError::ColumnNotFound`] if `column` does not exist
267    ///
268    /// # C++ equivalent
269    ///
270    /// `ArrayColumn<T>::getColumnRange(slicer, rowRange)`.
271    pub fn get_column_slice(
272        &self,
273        column: &str,
274        row_range: RowRange,
275        slicer: &Slicer,
276    ) -> Result<Vec<Value>, TableError> {
277        row_range.validate(self.row_count())?;
278        let mut results = Vec::new();
279        for row in row_range.iter() {
280            results.push(self.get_cell_slice(column, row, slicer)?);
281        }
282        Ok(results)
283    }

285    /// Writes a sub-region of an array cell for each row in `row_range`.
286    ///
287    /// `data` must have one element per selected row. Each element replaces
288    /// the corresponding slice region in that row's array cell.
289    ///
290    /// # Errors
291    ///
292    /// - [`TableError::CellNotArray`] if a cell in the column is scalar
293    /// - [`TableError::SlicerDimensionMismatch`] if slicer ndim != array ndim
294    /// - [`TableError::SlicerOutOfBounds`] if slicer exceeds array shape
295    /// - [`TableError::ColumnNotFound`] if `column` does not exist
296    /// - [`TableError::ColumnSliceLengthMismatch`] if `data` does not contain
297    ///   one slice per selected in-bounds row
298    ///
299    /// # C++ equivalent
300    ///
301    /// `ArrayColumn<T>::putColumnRange(slicer, rowRange, data)`.
302    pub fn put_column_slice(
303        &mut self,
304        column: &str,
305        row_range: RowRange,
306        slicer: &Slicer,
307        data: &[ArrayValue],
308    ) -> Result<(), TableError> {
309        row_range.validate(self.row_count())?;
310        let rows: Vec<usize> = row_range.iter().collect();
311        if rows.len() != data.len() {
312            return Err(TableError::ColumnSliceLengthMismatch {
313                rows: rows.len(),
314                data_len: data.len(),
315            });
316        }
317        for (row, patch) in rows.into_iter().zip(data.iter()) {
318            self.put_cell_slice(column, row, slicer, patch)?;
319        }
320        Ok(())
321    }

449    /// Executes a TaQL SELECT query and returns a [`QueryResult`].
450    ///
451    /// For simple SELECTs (only column references, no computed expressions),
452    /// returns `QueryResult::View` with a zero-copy [`RefTable`](crate::RefTable). For SELECTs
453    /// with computed columns, GROUP BY, or aggregate functions, returns
454    /// `QueryResult::Materialized` with an owned in-memory [`Table`] containing
455    /// the evaluated result rows.
456    ///
457    /// # Errors
458    ///
459    /// Returns [`TableError::Taql`] if the query is invalid or execution fails.
460    ///
461    /// # Examples
462    ///
463    /// ```rust
464    /// # use casacore_tables::{Table, TableSchema, ColumnSchema, QueryResult};
465    /// # use casacore_types::*;
466    /// # let schema = TableSchema::new(vec![
467    /// #     ColumnSchema::scalar("id", PrimitiveType::Int32),
468    /// #     ColumnSchema::scalar("flux", PrimitiveType::Float64),
469    /// # ]).unwrap();
470    /// # let mut table = Table::with_schema(schema);
471    /// # for i in 0..5 {
472    /// #     table.add_row(RecordValue::new(vec![
473    /// #         RecordField::new("id", Value::Scalar(ScalarValue::Int32(i))),
474    /// #         RecordField::new("flux", Value::Scalar(ScalarValue::Float64(i as f64))),
475    /// #     ])).unwrap();
476    /// # }
477    /// match table.query_result("SELECT * WHERE flux > 2.0").unwrap() {
478    ///     QueryResult::View(view) => assert_eq!(view.row_count(), 2),
479    ///     QueryResult::Materialized(mat) => {
480    ///         // Computed columns produce a materialized table
481    ///         let _ = mat.row_count();
482    ///     }
483    /// }
484    /// ```
485    ///
486    /// C++ equivalent: `tableCommand()` with a SELECT query.
487    pub fn query_result(&self, taql: &str) -> Result<QueryResult<'_>, TableError> {
488        let stmt = crate::taql::parse(taql).map_err(|e| TableError::Taql(e.to_string()))?;
489        let result = crate::taql::exec::execute_materializing_readonly(&stmt, self)
490            .map_err(|e| TableError::Taql(e.to_string()))?;
491        match result {
492            crate::taql::TaqlResult::Select {
493                row_indices,
494                columns,
495            } => {
496                let view = if columns.is_empty() {
497                    crate::RefTable::from_rows(self, row_indices)?
498                } else {
499                    crate::RefTable::from_rows_and_columns(self, row_indices, &columns)?
500                };
501                Ok(QueryResult::View(view))
502            }
503            crate::taql::TaqlResult::Materialized { table } => {
504                Ok(QueryResult::Materialized(table))
505            }
506            _ => Err(TableError::Taql(
507                "Table::query_result() only supports SELECT statements; use execute_taql() for mutations"
508                    .to_string(),
509            )),
510        }
511    }

604fn sort_indices_by_column<F>(
605    table: &Table,
606    column: &str,
607    compare: &F,
608) -> Result<Vec<usize>, TableError>
609where
610    F: Fn(&Value, &Value) -> std::cmp::Ordering,
611{
612    let n = table.row_count();
613    let mut indices: Vec<usize> = (0..n).collect();
614    let error = std::cell::RefCell::new(None);
615
616    indices.sort_by(|&a, &b| {
617        if error.borrow().is_some() {
618            return std::cmp::Ordering::Equal;
619        }
620
621        let va = match table.cell(a, column) {
622            Ok(value) => value,
623            Err(err) => {
624                *error.borrow_mut() = Some(err);
625                return std::cmp::Ordering::Equal;
626            }
627        };
628        let vb = match table.cell(b, column) {
629            Ok(value) => value,
630            Err(err) => {
631                *error.borrow_mut() = Some(err);
632                return std::cmp::Ordering::Equal;
633            }
634        };
635
636        match (va, vb) {
637            (Some(a), Some(b)) => compare(a, b),
638            (Some(_), None) => std::cmp::Ordering::Less,
639            (None, Some(_)) => std::cmp::Ordering::Greater,
640            (None, None) => std::cmp::Ordering::Equal,
641        }
642    });
643
644    if let Some(err) = error.into_inner() {
645        return Err(err);
646    }
647
648    Ok(indices)
649}

653fn validate_slicer_bounds(
654    slicer: &Slicer,
655    shape: &[usize],
656    row: usize,
657    column: &str,
658) -> Result<(), TableError> {
659    if slicer.ndim() != shape.len() {
660        return Err(TableError::SlicerDimensionMismatch {
661            start_ndim: slicer.ndim(),
662            end_ndim: shape.len(),
663            stride_ndim: slicer.ndim(),
664        });
665    }
666    for (axis, ((&s, &e), &ext)) in slicer
667        .start()
668        .iter()
669        .zip(slicer.end().iter())
670        .zip(shape.iter())
671        .enumerate()
672    {
673        if e > ext {
674            return Err(TableError::SlicerOutOfBounds {
675                axis,
676                index: e,
677                extent: ext,
678            });
679        }
680        let _ = (s, row, column); // suppress unused warnings
681    }
682    Ok(())
683}

685/// Build ndarray `SliceInfoElem` vector from a `Slicer`.
686fn slicer_to_slice_elems(slicer: &Slicer) -> Vec<ndarray::SliceInfoElem> {
687    slicer
688        .start()
689        .iter()
690        .zip(slicer.end().iter())
691        .zip(slicer.stride().iter())
692        .map(|((&s, &e), &st)| ndarray::SliceInfoElem::Slice {
693            start: s as isize,
694            end: Some(e as isize),
695            step: st as isize,
696        })
697        .collect()
698}

700/// Extract a sub-array from `av` using `slicer`.
701fn slice_array_value(av: &ArrayValue, slicer: &Slicer) -> ArrayValue {
702    use ndarray::SliceInfoElem;
703    let elems = slicer_to_slice_elems(slicer);
704    let si: Vec<SliceInfoElem> = elems;
705
706    macro_rules! do_slice {
707        ($arr:expr) => {{
708            let view = $arr.slice_each_axis(|ax| match si[ax.axis.index()] {
709                SliceInfoElem::Slice { start, end, step } => ndarray::Slice { start, end, step },
710                _ => unreachable!(),
711            });
712            view.to_owned()
713        }};
714    }
715
716    match av {
717        ArrayValue::Bool(a) => ArrayValue::Bool(do_slice!(a)),
718        ArrayValue::UInt8(a) => ArrayValue::UInt8(do_slice!(a)),
719        ArrayValue::UInt16(a) => ArrayValue::UInt16(do_slice!(a)),
720        ArrayValue::UInt32(a) => ArrayValue::UInt32(do_slice!(a)),
721        ArrayValue::Int16(a) => ArrayValue::Int16(do_slice!(a)),
722        ArrayValue::Int32(a) => ArrayValue::Int32(do_slice!(a)),
723        ArrayValue::Int64(a) => ArrayValue::Int64(do_slice!(a)),
724        ArrayValue::Float32(a) => ArrayValue::Float32(do_slice!(a)),
725        ArrayValue::Float64(a) => ArrayValue::Float64(do_slice!(a)),
726        ArrayValue::Complex32(a) => ArrayValue::Complex32(do_slice!(a)),
727        ArrayValue::Complex64(a) => ArrayValue::Complex64(do_slice!(a)),
728        ArrayValue::String(a) => ArrayValue::String(do_slice!(a)),
729    }
730}

732/// Write `data` into a sub-region of `target` specified by `slicer`.
733fn put_slice_array_value(target: &mut ArrayValue, slicer: &Slicer, data: &ArrayValue) {
734    use ndarray::SliceInfoElem;
735    let elems = slicer_to_slice_elems(slicer);
736    let si: Vec<SliceInfoElem> = elems;
737
738    macro_rules! do_put {
739        ($dst:expr, $src:expr) => {{
740            let mut view = $dst.slice_each_axis_mut(|ax| match si[ax.axis.index()] {
741                SliceInfoElem::Slice { start, end, step } => ndarray::Slice { start, end, step },
742                _ => unreachable!(),
743            });
744            view.assign($src);
745        }};
746    }
747
748    match (target, data) {
749        (ArrayValue::Bool(t), ArrayValue::Bool(s)) => do_put!(t, s),
750        (ArrayValue::UInt8(t), ArrayValue::UInt8(s)) => do_put!(t, s),
751        (ArrayValue::UInt16(t), ArrayValue::UInt16(s)) => do_put!(t, s),
752        (ArrayValue::UInt32(t), ArrayValue::UInt32(s)) => do_put!(t, s),
753        (ArrayValue::Int16(t), ArrayValue::Int16(s)) => do_put!(t, s),
754        (ArrayValue::Int32(t), ArrayValue::Int32(s)) => do_put!(t, s),
755        (ArrayValue::Int64(t), ArrayValue::Int64(s)) => do_put!(t, s),
756        (ArrayValue::Float32(t), ArrayValue::Float32(s)) => do_put!(t, s),
757        (ArrayValue::Float64(t), ArrayValue::Float64(s)) => do_put!(t, s),
758        (ArrayValue::Complex32(t), ArrayValue::Complex32(s)) => do_put!(t, s),
759        (ArrayValue::Complex64(t), ArrayValue::Complex64(s)) => do_put!(t, s),
760        (ArrayValue::String(t), ArrayValue::String(s)) => do_put!(t, s),
761        _ => {} // type mismatch silently ignored (validated at higher level)
762    }
763}
