62/// JSON-schema bundle for the public imager task protocol.
63#[derive(Debug, Clone, Serialize)]
64pub struct ImagerTaskSchemaBundle {
65    /// Compatibility descriptor for the request/result schemas.
66    pub protocol: ImagerProtocolInfo,
67    /// Canonical semantic task contract.
68    pub semantic: TaskSemanticContract,
69    /// Shared component schemas reusable across projections.
70    pub components: ProviderComponentSchemas,
71    /// Presentation annotations carried with the canonical bundle.
72    pub annotations: JsonValue,
73    /// Derived projection metadata for UI and CLI consumers.
74    pub projections: ProviderProjectionMetadata,
75    /// JSON schema for [`ImagerTaskRequest`].
76    pub request_schema: RootSchema,
77    /// JSON schema for [`ImagerTaskResult`].
78    pub result_schema: RootSchema,
79}
80
81impl ImagerTaskSchemaBundle {
82    /// Build the current request/result schema bundle.
83    pub fn current() -> Self {
84        let request_schema = schema_for!(ImagerTaskRequest);
85        let result_schema = schema_for!(ImagerTaskResult);
86        let ui_schema = serde_json::to_value(command_schema("casars-imager"))
87            .expect("serialize imager ui schema projection");
88        Self {
89            protocol: ImagerProtocolInfo::current(),
90            semantic: TaskSemanticContract {
91                request_schema: request_schema.clone(),
92                result_schema: result_schema.clone(),
93                operations: vec![TaskOperationDescriptor {
94                    name: "run".to_string(),
95                    request_kind: "run".to_string(),
96                    result_kind: Some("run".to_string()),
97                }],
98            },
99            components: merged_components([&request_schema, &result_schema]),
100            annotations: derived_ui_schema_annotations(),
101            projections: ProviderProjectionMetadata {
102                cli: Some(ProviderCliProjection {
103                    machine_actions: ProviderCliMachineActions {
104                        ui_schema: Some("--ui-schema".to_string()),
105                        json_schema: Some("--json-schema".to_string()),
106                        protocol_info: Some("--protocol-info".to_string()),
107                        json_run: Some("--json-run <SOURCE>".to_string()),
108                        session: None,
109                    },
110                }),
111                ui_schema: Some(ui_schema),
112                python: None,
113            },
114            request_schema,
115            result_schema,
116        }
117    }
118
119    /// Return the launcher/TUI compatibility view projected from the bundle.
120    pub fn ui_schema_projection(&self) -> Result<casa_ms::ui_schema::UiCommandSchema, String> {
121        let value = self
122            .projections
123            .ui_schema
124            .clone()
125            .ok_or_else(|| "missing ui_schema projection".to_string())?;
126        serde_json::from_value(value).map_err(|error| format!("parse imager ui schema: {error}"))
127    }
128}

572impl ImagerCubeAxisConfig {
573    fn into_runtime(self, spectral_mode: SpectralMode) -> Result<CubeAxisConfig, String> {
574        let veltype = parse_doppler_ref(&self.veltype)?;
575        if let Some(rest_frequency_hz) = self.rest_frequency_hz {
576            let text = format!("{rest_frequency_hz}Hz");
577            parse_ms_rest_frequency_hz(&text).map_err(|error| error.to_string())?;
578        }
579        Ok(CubeAxisConfig {
580            specmode: spectral_mode.cube_specmode(),
581            outframe: parse_frequency_ref(&self.outframe)?,
582            veltype,
583            interpolation: self.interpolation.into(),
584            rest_frequency_hz: self.rest_frequency_hz,
585            start: self.start.map(|value| value.into_runtime()).transpose()?,
586            width: self.width.map(|value| value.into_runtime()).transpose()?,
587        })
588    }

1283fn build_artifacts(request: &ImagerRunTaskRequest) -> Vec<ImagerArtifact> {
1284    let base = request.image_name.to_string_lossy().to_string();
1285    let mut artifacts = Vec::new();
1286    match request.spectral_mode {
1287        ImagerSpectralMode::Mfs
1288            if request.deconvolver == ImagerDeconvolver::Mtmfs && request.nterms > 1 =>
1289        {
1290            for term in 0..request.nterms {
1291                for (kind, label) in [
1292                    (ImagerArtifactKind::Psf, "PSF"),
1293                    (ImagerArtifactKind::Residual, "Residual"),
1294                    (ImagerArtifactKind::Model, "Model"),
1295                    (ImagerArtifactKind::Image, "Restored Image"),
1296                ] {
1297                    let suffix = format!("{}.tt{term}", kind.as_suffix());
1298                    let preview = (term == 0 && request.write_preview_pngs)
1299                        .then(|| PathBuf::from(format!("{base}.{suffix}.png")));
1300                    artifacts.push(artifact(
1301                        kind,
1302                        format!("{label} tt{term}"),
1303                        PathBuf::from(format!("{base}.{suffix}")),
1304                        preview,
1305                    ));
1306                }
1307            }
1308            let alpha_preview = request
1309                .write_preview_pngs
1310                .then(|| PathBuf::from(format!("{base}.alpha.png")));
1311            artifacts.push(artifact(
1312                ImagerArtifactKind::Alpha,
1313                "Spectral Index".to_string(),
1314                PathBuf::from(format!("{base}.alpha")),
1315                alpha_preview,
1316            ));
1317        }
1318        _ => {
1319            for (kind, label) in [
1320                (ImagerArtifactKind::Psf, "PSF"),
1321                (ImagerArtifactKind::Residual, "Residual"),
1322                (ImagerArtifactKind::Model, "Model"),
1323                (ImagerArtifactKind::Image, "Restored Image"),
1324            ] {
1325                let suffix = kind.as_suffix();
1326                let preview = request
1327                    .write_preview_pngs
1328                    .then(|| PathBuf::from(format!("{base}.{suffix}.png")));
1329                artifacts.push(artifact(
1330                    kind,
1331                    label.to_string(),
1332                    PathBuf::from(format!("{base}.{suffix}")),
1333                    preview,
1334                ));
1335            }
1336        }
1337    }
1338    artifacts
1339}
