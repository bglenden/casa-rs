32/// Parse environment arguments, run `importvla`, and return a process exit code.
33pub fn run_env(program_name: &str) -> i32 {
34    let (logging_guard, args) =
35        match casa_logging::init_global_from_env_and_args(std::env::args_os().skip(1)) {
36            Ok((guard, args)) => (guard, args),
37            Err(error) => {
38                eprintln!("Error: failed to initialize logging: {error}");
39                return 1;
40            }
41        };
42    tracing::info!("importvla started");
43    let code = run_with_cli_args(program_name, args);
44    if code == 0 {
45        tracing::info!("importvla completed");
46    } else {
47        tracing::error!(
48            casa.priority = "SEVERE",
49            exit_code = code,
50            "importvla failed"
51        );
52    }
53    if let Err(error) = logging_guard.flush() {
54        eprintln!("Error: failed to flush logging: {error}");
55        return 1;
56    }
57    code
58}

60/// Run `importvla` with already-filtered CLI arguments.
61pub fn run_with_cli_args(program_name: &str, args: impl IntoIterator<Item = OsString>) -> i32 {
62    let schema = command_schema(program_name);
63    match parse_args(args) {
64        Ok(CliAction::Help) => {
65            print!("{}", render_help(&schema));
66            0
67        }
68        Ok(CliAction::UiSchema) => match schema.render_json_pretty() {
69            Ok(payload) => {
70                print!("{payload}");
71                0
72            }
73            Err(error) => {
74                eprintln!("Error: failed to serialize --ui-schema output: {error}");
75                1
76            }
77        },
78        Ok(CliAction::JsonSchema) => {
79            match serde_json::to_string_pretty(&ImportVlaTaskSchemaBundle::current()) {
80                Ok(payload) => {
81                    print!("{payload}");
82                    0
83                }
84                Err(error) => {
85                    eprintln!("Error: failed to serialize --json-schema output: {error}");
86                    1
87                }
88            }
89        }
90        Ok(CliAction::ProtocolInfo) => {
91            match serde_json::to_string_pretty(&ImportVlaProtocolInfo::current()) {
92                Ok(payload) => {
93                    print!("{payload}");
94                    0
95                }
96                Err(error) => {
97                    eprintln!("Error: failed to serialize --protocol-info output: {error}");
98                    1
99                }
100            }
101        }
102        Ok(CliAction::JsonRun(source)) => match run_json_request(&source) {
103            Ok(()) => 0,
104            Err(error) => {
105                eprintln!("Error: {error}");
106                1
107            }
108        },
109        Ok(CliAction::Run { options, json }) => match run(options, json) {
110            Ok(()) => 0,
111            Err(error) => {
112                eprintln!("Error: {error}");
113                1
114            }
115        },
116        Err(error) => {
117            eprintln!("Error: {error}\n");
118            eprintln!("{}", render_help(&schema));
119            1
120        }
121    }
122}

124/// Build the machine-readable command schema for `importvla`.
125pub fn command_schema(program_name: &str) -> UiCommandSchema {
126    let bundle = casa_provider_contracts::builtin_surface_bundle("importvla")
127        .expect("built-in importvla parameter surface must remain valid");
128    let mut schema: UiCommandSchema =
129        serde_json::from_value(casa_provider_contracts::project_ui_schema(&bundle))
130            .expect("canonical importvla UI projection must match UiCommandSchema");
131    schema.invocation_name = program_name.to_string();
132    schema.usage = format!("{program_name} [parameters]");
133    schema
134}
135fn parse_args(args: impl IntoIterator<Item = OsString>) -> Result<CliAction, VlaError> {
136    let args = args.into_iter().collect::<Vec<_>>();
137    let string_args = args
138        .iter()
139        .map(|value| {
140            value.to_str().ok_or_else(|| VlaError::InvalidArgument {
141                argument: "argv",
142                message: "non-utf8 argument".to_string(),
143            })
144        })
145        .collect::<Result<Vec<_>, _>>()?;
146
147    if string_args
148        .iter()
149        .any(|arg| *arg == "-h" || *arg == "--help")
150    {
151        return Ok(CliAction::Help);
152    }
153    if string_args.contains(&"--ui-schema") {
154        return Ok(CliAction::UiSchema);
155    }
156    if string_args.contains(&"--json-schema") {
157        return Ok(CliAction::JsonSchema);
158    }
159    if string_args.contains(&"--protocol-info") {
160        return Ok(CliAction::ProtocolInfo);
161    }
162    if let Some(index) = string_args.iter().position(|arg| *arg == "--json-run") {
163        let source = string_args
164            .get(index + 1)
165            .ok_or_else(|| VlaError::InvalidArgument {
166                argument: "json-run",
167                message: "missing value".to_string(),
168            })?;
169        return Ok(CliAction::JsonRun((*source).to_string()));
170    }
171
172    parse_run_args(args)
173}

175fn parse_run_args(args: Vec<OsString>) -> Result<CliAction, VlaError> {
176    let mut args = args.into_iter();
177    let mut options = ImportVlaOptions::default();
178    let mut json = false;
179
180    while let Some(arg) = args.next() {
181        let Some(arg) = arg.to_str() else {
182            return Err(VlaError::InvalidArgument {
183                argument: "argv",
184                message: "non-utf8 argument".to_string(),
185            });
186        };
187        match arg {
188            "--json" => json = true,
189            "--archivefile" => {
190                let value = next_value(&mut args, "archivefile")?;
191                options.archivefiles.push(PathBuf::from(value));
192            }
193            "--archivefiles" => {
194                let value = next_value(&mut args, "archivefiles")?;
195                append_archivefiles(&mut options, &value);
196            }
197            "--vis" => {
198                options.vis = Some(PathBuf::from(next_value(&mut args, "vis")?));
199            }
200            "--bandname" => {
201                options.bandname = Some(next_value(&mut args, "bandname")?.parse()?);
202            }
203            "--frequencytol" => {
204                let value = next_value(&mut args, "frequencytol")?;
205                options.frequencytol_hz = ImportVlaOptions::parse_frequencytol(&value)?;
206            }
207            "--project" => {
208                options.project = Some(next_value(&mut args, "project")?.to_string());
209            }
210            "--starttime" => {
211                options.starttime = Some(next_value(&mut args, "starttime")?.to_string());
212            }
213            "--stoptime" => {
214                options.stoptime = Some(next_value(&mut args, "stoptime")?.to_string());
215            }
216            "--autocorr" => options.autocorr = true,
217            "--no-applytsys" => options.applytsys = false,
218            "--applytsys" => options.applytsys = true,
219            "--keepblanks" => options.keepblanks = true,
220            "--evlabands" => options.evlabands = true,
221            "--antnamescheme" => {
222                options.antnamescheme = next_value(&mut args, "antnamescheme")?.parse()?;
223            }
224            other => {
225                return Err(VlaError::InvalidArgument {
226                    argument: "argv",
227                    message: format!("unrecognized argument `{other}`"),
228                });
229            }
230        }
231    }
232
233    options.require_archivefiles()?;
234    Ok(CliAction::Run { options, json })
235}

237fn next_value(
238    args: &mut impl Iterator<Item = OsString>,
239    argument: &'static str,
240) -> Result<String, VlaError> {
241    let value = args.next().ok_or_else(|| VlaError::InvalidArgument {
242        argument,
243        message: "missing value".to_string(),
244    })?;
245    value.into_string().map_err(|_| VlaError::InvalidArgument {
246        argument,
247        message: "non-utf8 value".to_string(),
248    })
249}

251fn append_archivefiles(options: &mut ImportVlaOptions, value: &str) {
252    for item in value
253        .split(',')
254        .map(str::trim)
255        .filter(|item| !item.is_empty())
256    {
257        options.archivefiles.push(PathBuf::from(item));
258    }
259}

261fn run_json_request(source: &str) -> Result<(), String> {
262    let payload = read_json_source(source)?;
263    let request = serde_json::from_str::<ImportVlaTaskRequest>(&payload)
264        .map_err(|error| format!("parse importvla task request: {error}"))?;
265    let result = request.execute()?;
266    serde_json::to_string_pretty(&result)
267        .map(|json| {
268            println!("{json}");
269        })
270        .map_err(|error| format!("serialize importvla task result: {error}"))
271}
272
273fn read_json_source(source: &str) -> Result<String, String> {
274    if source == "-" {
275        let mut stdin = std::io::stdin();
276        let mut payload = String::new();
277        use std::io::Read as _;
278        stdin
279            .read_to_string(&mut payload)
280            .map_err(|error| format!("read importvla task request from stdin: {error}"))?;
281        Ok(payload)
282    } else {
283        fs::read_to_string(source)
284            .map_err(|error| format!("read importvla task request from {source}: {error}"))
285    }
286}

288fn run(options: ImportVlaOptions, json_output: bool) -> Result<(), VlaError> {
289    tracing::info!(
290        archive_count = options.archivefiles.len(),
291        vis = ?options.vis,
292        json_output,
293        "importvla run started"
294    );
295    let cleanup_vis = if options.vis.is_none() {
296        Some(options.effective_vis_for_import()?)
297    } else {
298        None
299    };
300    let report = match import_archive_files_to_measurement_set_from_options(&options) {
301        Ok(report) => report,
302        Err(error) => {
303            tracing::error!(
304                casa.priority = "SEVERE",
305                error = %error,
306                cleanup_vis = ?cleanup_vis,
307                "importvla import failed"
308            );
309            if let Some(path) = cleanup_vis.as_deref().filter(|path| path.exists()) {
310                cleanup_failed_import_output(path).map_err(|cleanup_error| {
311                    VlaError::import(format!(
312                        "{error}; cleanup failed for {}: {cleanup_error}",
313                        path.display()
314                    ))
315                })?;
316            }
317            return Err(error);
318        }
319    };
320    if json_output {
321        let payload = json!({
322            "mode": "disk-import",
323            "options": render_options_json(&options),
324            "report": report,
325        });
326        println!(
327            "{}",
328            serde_json::to_string_pretty(&payload).map_err(|error| VlaError::InvalidArgument {
329                argument: "json",
330                message: error.to_string(),
331            })?
332        );
333    } else {
334        render_import_text(&report);
335    }
336    tracing::info!(
337        archive_count = options.archivefiles.len(),
338        vis = ?report.vis,
339        rows = report.main_rows_written,
340        "importvla run completed"
341    );
342    Ok(())
343}
344
345fn cleanup_failed_import_output(path: &Path) -> Result<(), std::io::Error> {
346    if path.is_dir() {
347        fs::remove_dir_all(path)
348    } else {
349        fs::remove_file(path)
350    }
351}

373#[cfg_attr(not(test), allow(dead_code))]
374fn render_text(options: &ImportVlaOptions, summary: &crate::ArchiveSummary) {
375    println!("importvla disk scan");
376    if let Some(vis) = &options.vis {
377        println!("planned vis: {}", vis.display());
378    }
379    println!("archive files: {}", options.archivefiles.len());
380    println!("logical records: {}", summary.logical_records);
381    println!("logical bytes: {}", summary.logical_bytes);
382    println!();
383    for file in &summary.files {
384        println!("file: {}", file.path.display());
385        println!("  logical records: {}", file.logical_records);
386        println!("  logical bytes: {}", file.logical_bytes);
387        if let Some((min, max)) = file.revision_range {
388            println!("  revision range: {min}..{max}");
389        }
390        if let Some((min, max)) = file.obs_day_range {
391            println!("  obs day range: {min}..{max}");
392        }
393        println!("  max antennas: {}", file.max_antennas);
394        let cda_line = file
395            .used_cda_histogram
396            .iter()
397            .map(|(used, count)| format!("{used}=>{count}"))
398            .collect::<Vec<_>>()
399            .join(", ");
400        println!("  used CDA histogram: {cda_line}");
401        println!();
402    }
403    println!("note: this first wave scans disk archive files and reassembles logical records.");
404}

485    fn physical_block(
486        current: u16,
487        total: u16,
488        payload: &[u8],
489    ) -> [u8; crate::PHYSICAL_RECORD_SIZE] {
490        let mut block = [0_u8; crate::PHYSICAL_RECORD_SIZE];
491        block[0..2].copy_from_slice(&current.to_be_bytes());
492        block[2..4].copy_from_slice(&total.to_be_bytes());
493        block[4..4 + payload.len()].copy_from_slice(payload);
494        block
495    }

497    fn logical_record_bytes(length_bytes: usize, revision: u16, obs_day: u32) -> Vec<u8> {
498        let mut bytes = vec![0_u8; length_bytes];
499        bytes[0..4].copy_from_slice(&((length_bytes / 2) as i32).to_be_bytes());
500        bytes[2 * 3..2 * 3 + 2].copy_from_slice(&revision.to_be_bytes());
501        bytes[2 * 4..2 * 4 + 4].copy_from_slice(&obs_day.to_be_bytes());
502        bytes[2 * 17..2 * 17 + 2].copy_from_slice(&27_u16.to_be_bytes());
503        bytes
504    }

516    impl Drop for CurrentDirGuard {
517        fn drop(&mut self) {
518            let _ = std::env::set_current_dir(&self.0);

522    fn cwd_lock() -> &'static Mutex<()> {
523        static LOCK: OnceLock<Mutex<()>> = OnceLock::new();
524        LOCK.get_or_init(|| Mutex::new(()))
525    }

527    fn with_temp_cwd<T>(f: impl FnOnce(&Path) -> T) -> T {
528        let _lock = cwd_lock().lock().expect("cwd lock");
529        let restore = CurrentDirGuard(std::env::current_dir().expect("original cwd"));
530        let temp = tempdir().expect("temp cwd");
531        std::env::set_current_dir(temp.path()).expect("set temp cwd");
532        let result = f(temp.path());
533        drop(restore);
534        result
535    }
