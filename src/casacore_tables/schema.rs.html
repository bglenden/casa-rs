5/// Errors produced during schema construction or column option validation.
6///
7/// These are returned by [`TableSchema::new`] and [`ColumnSchema::with_options`]
8/// when the supplied configuration violates a schema invariant. They correspond
9/// to the exceptions that C++ `TableDesc::addColumn` and `ColumnDesc` constructors
10/// would throw at runtime.
11#[derive(Debug, Error, Clone, PartialEq, Eq)]
12pub enum SchemaError {
13    /// A column name appears more than once in the schema.
14    ///
15    /// Cf. C++ `TableDesc::addColumn`, which throws `TableError` on duplicates.
16    #[error("schema contains duplicate column name \"{0}\"")]
17    DuplicateColumn(String),
18
19    /// The `direct` option was set on a column that is not a [`ArrayShapeContract::Fixed`] array.
20    ///
21    /// Direct storage requires the cell size to be constant across all rows, which
22    /// is only possible when the array shape is fully specified at schema time.
23    /// Cf. C++ `ColumnDesc::Direct`, which is only valid alongside a fixed `IPosition`.
24    #[error("column \"{0}\" uses Direct but is not FixedShape array")]
25    DirectRequiresFixedShape(String),
26
27    /// The `undefined` option was set on a non-scalar column.
28    ///
29    /// Cf. C++ `ColumnDesc::Undefined`, which is only meaningful for scalar columns.
30    #[error("column \"{0}\" uses Undefined but is not scalar")]
31    UndefinedOnlyForScalar(String),
32
33    /// A column with the given name does not exist in the schema.
34    ///
35    /// Returned by [`TableSchema::remove_column`] and [`TableSchema::rename_column`]
36    /// when the specified column is not found.
37    #[error("column \"{0}\" not found in schema")]
38    ColumnNotFound(String),
39}
40
41/// Storage and behaviour flags for a single column.
42///
43/// These correspond to the bit flags carried by C++ `ColumnDesc` (specifically
44/// `ColumnDesc::Direct` and `ColumnDesc::Undefined`). Both default to `false`.
45#[derive(Debug, Clone, Copy, PartialEq, Eq, Default)]
46pub struct ColumnOptions {
47    /// Store cell data inline (directly) within the table file rather than
48    /// in a separate storage manager file.
49    ///
50    /// Only valid for [`ArrayShapeContract::Fixed`] array columns; setting this
51    /// on any other kind of column returns [`SchemaError::DirectRequiresFixedShape`].
52    /// Cf. C++ `ColumnDesc::Direct`.
53    pub direct: bool,
54
55    /// Allow cells in this column to be left in an undefined (empty) state.
56    ///
57    /// Only valid for [`ColumnType::Scalar`] columns; setting this on an array
58    /// or record column returns [`SchemaError::UndefinedOnlyForScalar`].
59    /// Cf. C++ `ColumnDesc::Undefined`.
60    pub undefined: bool,
61}

63/// Describes the shape contract of an array column.
64///
65/// This type is the payload of [`ColumnType::Array`] and distinguishes between
66/// columns where every row has the same shape and those where the shape may vary.
67#[derive(Debug, Clone, PartialEq, Eq)]
68pub enum ArrayShapeContract {
69    /// Every row stores an array with exactly this shape.
70    ///
71    /// Corresponds to `ArrayColumnDesc<T>` constructed with a fixed `IPosition`
72    /// and the `ColumnDesc::Direct` flag in C++. The shape is recorded in the
73    /// schema and enforced on every write.
74    Fixed { shape: Vec<usize> },
75
76    /// Each row may store an array with a different shape.
77    ///
78    /// Corresponds to C++ indirect array columns (`ArrayColumnDesc<T>` without a
79    /// fixed shape). `ndim`, when `Some`, constrains the number of dimensions but
80    /// not the extent along each axis; when `None` the dimensionality is also
81    /// unconstrained.
82    Variable { ndim: Option<usize> },
83}
84
85/// The kind of data a column holds.
86///
87/// Every [`ColumnSchema`] carries one of these three variants, mirroring the
88/// three column categories recognised by C++ casacore: scalar (`ScalarColumnDesc`),
89/// array (`ArrayColumnDesc`), and record (`RecordColumnDesc` / `TpRecord`).
90#[derive(Debug, Clone, PartialEq, Eq)]
91pub enum ColumnType {
92    /// A single primitive value per row.
93    Scalar,
94
95    /// A multi-dimensional array per row, with the given [`ArrayShapeContract`].
96    Array(ArrayShapeContract),
97
98    /// A nested keyword/value record per row.
99    ///
100    /// Record columns carry no numeric [`PrimitiveType`]; their data type is
101    /// implicitly `TpRecord` in C++ terms.
102    Record,
103}
104
105/// Describes a single column within a [`TableSchema`].
106///
107/// Corresponds to C++ `ColumnDesc`. A `ColumnSchema` captures the column name,
108/// its [`ColumnType`] (scalar, array, or record), the primitive [`PrimitiveType`]
109/// of the stored elements (absent for record columns), and any additional
110/// [`ColumnOptions`] flags.
111///
112/// Use the factory methods [`ColumnSchema::scalar`], [`ColumnSchema::record`],
113/// [`ColumnSchema::array_fixed`], and [`ColumnSchema::array_variable`] to
114/// construct instances, then optionally call [`ColumnSchema::with_options`] to
115/// attach non-default flags.
116#[derive(Debug, Clone, PartialEq, Eq)]
117pub struct ColumnSchema {
118    name: String,
119    column_type: ColumnType,
120    data_type: Option<PrimitiveType>,
121    options: ColumnOptions,
122}
123
124impl ColumnSchema {
125    /// Create a scalar column holding one primitive value per row.
126    ///
127    /// Corresponds to C++ `ScalarColumnDesc<T>`. The column has no shape
128    /// constraint and no special storage flags by default.
129    pub fn scalar(name: impl Into<String>, data_type: PrimitiveType) -> Self {
130        Self {
131            name: name.into(),
132            column_type: ColumnType::Scalar,
133            data_type: Some(data_type),
134            options: ColumnOptions::default(),
135        }
136    }
137
138    /// Create a record column holding a nested keyword/value record per row.
139    ///
140    /// Corresponds to a C++ column with data type `TpRecord`. Record columns
141    /// carry no numeric element type, so [`ColumnSchema::data_type`] returns
142    /// `None` for these columns.
143    pub fn record(name: impl Into<String>) -> Self {
144        Self {
145            name: name.into(),
146            column_type: ColumnType::Record,
147            data_type: None,
148            options: ColumnOptions::default(),
149        }
150    }
151
152    /// Create an array column with a fixed shape shared by every row.
153    ///
154    /// Corresponds to C++ `ArrayColumnDesc<T>` constructed with a fixed
155    /// `IPosition`. The supplied `shape` is embedded in the schema and every
156    /// cell must conform to it. This column kind is eligible for direct storage
157    /// via [`ColumnOptions::direct`].
158    pub fn array_fixed(
159        name: impl Into<String>,
160        data_type: PrimitiveType,
161        shape: Vec<usize>,
162    ) -> Self {
163        Self {
164            name: name.into(),
165            column_type: ColumnType::Array(ArrayShapeContract::Fixed { shape }),
166            data_type: Some(data_type),
167            options: ColumnOptions::default(),
168        }
169    }
170
171    /// Create an array column whose shape may differ between rows.
172    ///
173    /// Corresponds to C++ `ArrayColumnDesc<T>` without a fixed shape (indirect
174    /// array). `ndim` optionally constrains the number of dimensions; pass
175    /// `None` to leave both dimensionality and extents unconstrained.
176    pub fn array_variable(
177        name: impl Into<String>,
178        data_type: PrimitiveType,
179        ndim: Option<usize>,
180    ) -> Self {
181        Self {
182            name: name.into(),
183            column_type: ColumnType::Array(ArrayShapeContract::Variable { ndim }),
184            data_type: Some(data_type),
185            options: ColumnOptions::default(),
186        }
187    }
188
189    /// Attach additional storage/behaviour options to this column.
190    ///
191    /// Validates the combination of options against the column kind and returns
192    /// [`SchemaError::DirectRequiresFixedShape`] or
193    /// [`SchemaError::UndefinedOnlyForScalar`] if the options are incompatible.
194    /// On success returns `self` with the options applied, enabling builder-style
195    /// construction.
196    pub fn with_options(mut self, options: ColumnOptions) -> Result<Self, SchemaError> {
197        self.options = options;
198        self.validate_options()?;
199        Ok(self)
200    }
201
202    /// Return the column name.
203    pub fn name(&self) -> &str {
204        &self.name
205    }
206
207    /// Return the column kind (scalar, array, or record).
208    pub fn column_type(&self) -> &ColumnType {
209        &self.column_type
210    }
211
212    /// Return the primitive element type, or `None` for record columns.
213    pub fn data_type(&self) -> Option<PrimitiveType> {
214        self.data_type
215    }
216
217    /// Return the storage and behaviour flags for this column.
218    pub fn options(&self) -> ColumnOptions {
219        self.options
220    }
221
222    fn validate_options(&self) -> Result<(), SchemaError> {
223        if self.options.direct
224            && !matches!(
225                self.column_type,
226                ColumnType::Array(ArrayShapeContract::Fixed { .. })
227            )
228        {
229            return Err(SchemaError::DirectRequiresFixedShape(self.name.clone()));
230        }
231        if self.options.undefined && !matches!(self.column_type, ColumnType::Scalar) {
232            return Err(SchemaError::UndefinedOnlyForScalar(self.name.clone()));
233        }
234        Ok(())
235    }
236}

238/// Defines the structure of a casacore table: an ordered set of column descriptions.
239///
240/// Corresponds to C++ `TableDesc`. A `TableSchema` is constructed before the
241/// table itself and subsequently embedded in the on-disk table directory. It
242/// records all column names, types, shapes, and storage flags that govern how
243/// data is stored and retrieved.
244///
245/// From the C++ docs: "A table description consists of: name, version, comment,
246/// a set of column descriptions, and a keyword set." This Rust type covers the
247/// column-description portion; keywords are handled separately.
248///
249/// # Example
250///
251/// ```rust
252/// use casacore_tables::{ColumnSchema, TableSchema};
253/// use casacore_types::PrimitiveType;
254///
255/// let schema = TableSchema::new(vec![
256///     ColumnSchema::scalar("time", PrimitiveType::Float64),
257///     ColumnSchema::array_fixed("data", PrimitiveType::Float32, vec![4, 64]),
258/// ])
259/// .expect("no duplicate columns");
260/// ```
261#[derive(Debug, Clone, PartialEq, Eq, Default)]
262pub struct TableSchema {
263    columns: Vec<ColumnSchema>,
264}
265
266impl TableSchema {
267    /// Construct a `TableSchema` from an ordered list of column descriptions.
268    ///
269    /// Validates each [`ColumnSchema`]'s options (see [`ColumnSchema::with_options`])
270    /// and rejects the schema if any two columns share the same name, returning
271    /// [`SchemaError::DuplicateColumn`]. Cf. C++ `TableDesc::addColumn`, which
272    /// throws `TableError` on the first duplicate encountered.
273    pub fn new(columns: Vec<ColumnSchema>) -> Result<Self, SchemaError> {
274        for i in 0..columns.len() {
275            columns[i].validate_options()?;
276            if columns[(i + 1)..]
277                .iter()
278                .any(|other| other.name == columns[i].name)
279            {
280                return Err(SchemaError::DuplicateColumn(columns[i].name.clone()));
281            }
282        }
283        Ok(Self { columns })
284    }
285
286    /// Return the ordered slice of column descriptions.
287    ///
288    /// The order matches the order supplied to [`TableSchema::new`].
289    /// Cf. C++ `TableDesc::columnDescSet()`.
290    pub fn columns(&self) -> &[ColumnSchema] {
291        &self.columns
292    }
293
294    /// Look up a column description by name.
295    ///
296    /// Returns `None` if no column with the given name exists.
297    /// Cf. C++ `TableDesc::columnDesc(name)`, which throws on a missing column.
298    pub fn column(&self, name: &str) -> Option<&ColumnSchema> {
299        self.columns.iter().find(|column| column.name == name)
300    }
301
302    /// Return `true` if a column with the given name exists in this schema.
303    ///
304    /// Cf. C++ `TableDesc::isColumn(name)`.
305    pub fn contains_column(&self, name: &str) -> bool {
306        self.column(name).is_some()
307    }
308
309    /// Add a column to the schema.
310    ///
311    /// Returns [`SchemaError::DuplicateColumn`] if a column with the same
312    /// name already exists. Validates the column's options before inserting.
313    /// Cf. C++ `TableDesc::addColumn`.
314    pub fn add_column(&mut self, col: ColumnSchema) -> Result<(), SchemaError> {
315        if self.contains_column(col.name()) {
316            return Err(SchemaError::DuplicateColumn(col.name().to_string()));
317        }
318        col.validate_options()?;
319        self.columns.push(col);
320        Ok(())
321    }
322
323    /// Remove a column by name, returning the removed [`ColumnSchema`].
324    ///
325    /// Returns [`SchemaError::ColumnNotFound`] if no column with the given
326    /// name exists. Cf. C++ `TableDesc::removeColumn`.
327    pub fn remove_column(&mut self, name: &str) -> Result<ColumnSchema, SchemaError> {
328        let pos = self
329            .columns
330            .iter()
331            .position(|c| c.name == name)
332            .ok_or_else(|| SchemaError::ColumnNotFound(name.to_string()))?;
333        Ok(self.columns.remove(pos))
334    }
335
336    /// Rename a column.
337    ///
338    /// Returns [`SchemaError::ColumnNotFound`] if `old` does not exist, or
339    /// [`SchemaError::DuplicateColumn`] if `new` already exists.
340    /// Cf. C++ `TableDesc::renameColumn`.
341    pub fn rename_column(&mut self, old: &str, new: &str) -> Result<(), SchemaError> {
342        if self.contains_column(new) {
343            return Err(SchemaError::DuplicateColumn(new.to_string()));
344        }
345        let col = self
346            .columns
347            .iter_mut()
348            .find(|c| c.name == old)
349            .ok_or_else(|| SchemaError::ColumnNotFound(old.to_string()))?;
350        col.name = new.to_string();
351        Ok(())
352    }
353}

361    #[test]
362    fn schema_rejects_duplicate_columns() {
363        let schema = TableSchema::new(vec![
364            ColumnSchema::scalar("id", PrimitiveType::Int32),
365            ColumnSchema::scalar("id", PrimitiveType::Int32),
366        ]);
367        assert_eq!(schema, Err(SchemaError::DuplicateColumn("id".to_string())));
368    }
369
370    #[test]
371    fn direct_requires_fixed_array_shape() {
372        let column = ColumnSchema::array_variable("data", PrimitiveType::Float32, Some(2))
373            .with_options(ColumnOptions {
374                direct: true,
375                undefined: false,
376            });
377        assert_eq!(
378            column,
379            Err(SchemaError::DirectRequiresFixedShape("data".to_string()))
380        );
381    }
382
383    #[test]
384    fn undefined_only_applies_to_scalar_columns() {
385        let column = ColumnSchema::record("meta").with_options(ColumnOptions {
386            direct: false,
387            undefined: true,
388        });
389        assert_eq!(
390            column,
391            Err(SchemaError::UndefinedOnlyForScalar("meta".to_string()))
392        );
393    }

395    #[test]
396    fn add_column_to_existing_schema() {
397        let mut schema = TableSchema::new(vec![ColumnSchema::scalar("a", PrimitiveType::Int32)])
398            .expect("valid schema");
399        schema
400            .add_column(ColumnSchema::scalar("b", PrimitiveType::Float64))
401            .expect("add succeeds");

411        assert_eq!(result, Err(SchemaError::DuplicateColumn("a".to_string())));
412    }

416        let mut schema = TableSchema::new(vec![
