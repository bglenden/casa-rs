34pub(crate) fn parse_startup_args(
35    args: impl IntoIterator<Item = OsString>,
36) -> Result<StartupSelection, String> {
37    let mut args = args.into_iter();
38    let Some(first) = args.next() else {
39        return Ok(StartupSelection::Launcher);
40    };
41    let first = decode_arg(first, "casars argument")?;
42    if matches!(first.as_str(), "-h" | "--help") {
43        return Ok(StartupSelection::PrintText(render_casars_help()));
44    }
45
46    let (app_id, app_args) = if first == "--app" {
47        let app_id = args
48            .next()
49            .ok_or_else(|| "missing app id after --app".to_string())
50            .and_then(|value| decode_arg(value, "app id"))?;
51        (app_id, args.collect::<Vec<_>>())
52    } else if first.starts_with('-') {
53        return Err(format!(
54            "unknown casars option {first:?}; use --help to see supported startup syntax"
55        ));
56    } else {
57        (first, args.collect::<Vec<_>>())
58    };
59
60    let app = resolve_app(Some(&app_id))?;
61    if app_args.is_empty() {
62        return Ok(StartupSelection::App(StartupLaunch {
63            app,
64            prefill: Vec::new(),
65            auto_run: false,
66        }));
67    }
68
69    let schema = app.load_schema()?;
70    match parse_schema_prefill_args(&schema, app_args)? {
71        SchemaPrefillParse::PrintText(text) => Ok(StartupSelection::PrintText(text)),
72        SchemaPrefillParse::Prefill { values, auto_run } => {
73            Ok(StartupSelection::App(StartupLaunch {
74                app,
75                prefill: values,
76                auto_run,
77            }))
78        }
79    }
80}

83enum SchemaPrefillParse {
84    PrintText(String),
85    Prefill {
86        values: Vec<StartupPrefill>,
87        auto_run: bool,
88    },

91fn parse_schema_prefill_args(
92    schema: &UiCommandSchema,
93    args: Vec<OsString>,
94) -> Result<SchemaPrefillParse, String> {
95    let mut values = Vec::<StartupPrefill>::new();
96    let positional_ids = schema
97        .arguments
98        .iter()
99        .filter_map(|argument| match argument.parser {
100            UiArgumentParser::Positional { .. } if !argument.hidden_in_tui => {
101                Some(argument.id.as_str())
102            }
103            _ => None,
104        })
105        .collect::<Vec<_>>();
106    let mut positional_index = 0usize;
107    let mut raw_args = args
108        .into_iter()
109        .map(|value| decode_arg(value, &schema.command_id))
110        .collect::<Result<Vec<_>, _>>()?
111        .into_iter()
112        .peekable();
113    let mut end_of_options = false;
114    let mut saw_prefill = false;
115
116    while let Some(raw) = raw_args.next() {
117        if !end_of_options && raw == "--" {
118            end_of_options = true;
119            continue;
120        }
121
122        if !end_of_options && raw.starts_with('-') {
123            let Some(argument) = find_flag_argument(schema, &raw) else {
124                return Err(format!("unknown {} argument {raw:?}", schema.command_id));
125            };
126
127            match &argument.parser {
128                UiArgumentParser::Action { action, .. } => {
129                    if saw_prefill || raw_args.peek().is_some() {
130                        return Err(format!(
131                            "{raw} cannot be combined with other {} startup arguments",
132                            schema.command_id
133                        ));
134                    }
135                    let text = match action {
136                        UiActionKind::Help => schema.render_help(),
137                        UiActionKind::UiSchema => schema.render_json_pretty().map_err(|error| {
138                            format!("serialize {} ui schema: {error}", schema.command_id)
139                        })?,
140                    };
141                    return Ok(SchemaPrefillParse::PrintText(text));
142                }
143                _ if argument.hidden_in_tui => {
144                    return Err(format!(
145                        "{} is managed internally by casars and cannot be set on startup",
146                        argument.label
147                    ));
148                }
149                UiArgumentParser::Toggle { true_flags, .. } => {
150                    let enabled = true_flags.iter().any(|flag| flag == &raw);
151                    upsert_value(
152                        &mut values,
153                        argument.id.clone(),
154                        StartupValue::Toggle(enabled),
155                    );
156                    saw_prefill = true;
157                }
158                UiArgumentParser::Option { choices, .. } => {
159                    let value = raw_args.next().ok_or_else(|| {
160                        format!(
161                            "missing value for {raw} in {} startup arguments",
162                            schema.command_id
163                        )
164                    })?;
165                    if !choices.is_empty() && !choices.iter().any(|choice| choice == &value) {
166                        return Err(format!(
167                            "invalid value {value:?} for {}; expected one of: {}",
168                            argument.label,
169                            choices.join(", ")
170                        ));
171                    }
172                    upsert_value(&mut values, argument.id.clone(), StartupValue::Text(value));
173                    saw_prefill = true;
174                }
175                UiArgumentParser::Positional { .. } => {
176                    unreachable!("flags never map to positional arguments")
177                }
178            }
179            continue;
180        }
181
182        let positional_id = positional_ids.get(positional_index).ok_or_else(|| {
183            format!(
184                "unexpected extra positional argument {raw:?} for {}",
185                schema.command_id
186            )
187        })?;
188        upsert_value(
189            &mut values,
190            (*positional_id).to_string(),
191            StartupValue::Text(raw),
192        );
193        positional_index += 1;
194        saw_prefill = true;
195    }
196
197    Ok(SchemaPrefillParse::Prefill {
198        values,
199        auto_run: saw_prefill,
200    })
201}

203fn find_flag_argument<'a>(
204    schema: &'a UiCommandSchema,
205    raw: &str,
206) -> Option<&'a casacore_ms::listobs::cli::UiArgumentSchema> {
207    schema
208        .arguments
209        .iter()
210        .find(|argument| match &argument.parser {
211            UiArgumentParser::Option { flags, .. } | UiArgumentParser::Action { flags, .. } => {
212                flags.iter().any(|flag| flag == raw)
213            }
214            UiArgumentParser::Toggle {
215                true_flags,
216                false_flags,
217            } => true_flags
218                .iter()
219                .chain(false_flags.iter())
220                .any(|flag| flag == raw),
221            UiArgumentParser::Positional { .. } => false,
222        })
223}

225fn upsert_value(values: &mut Vec<StartupPrefill>, id: String, value: StartupValue) {
226    if let Some(existing) = values.iter_mut().find(|entry| entry.id == id) {
227        existing.value = value;
228    } else {
229        values.push(StartupPrefill { id, value });
230    }
231}
232
233fn decode_arg(value: OsString, context: &str) -> Result<String, String> {
234    value.into_string().map_err(|raw| {
235        format!(
236            "{context} must be valid UTF-8; received {:?}",
237            raw.to_string_lossy()
238        )
239    })
240}

242fn render_casars_help() -> String {
243    let mut out = String::from(
244        "casars - interactive launcher for CASA-oriented terminal apps\n\nUsage:\n  casars\n  casars <app-id> [app-args...]\n  casars --app <app-id> [app-args...]\n  casars --help\n\nApps:\n",
245    );
246    for app in registered_apps() {
247        out.push_str(&format!(
248            "  {:<12} {} / {}\n",
249            app.id, app.category, app.display_name
250        ));
251    }
252    out.push_str(
253        "\nExamples:\n  casars\n  casars tablebrowser /path/to/table\n  casars --app listobs /path/to.ms --field 3C286\n",
254    );
255    out
256}
