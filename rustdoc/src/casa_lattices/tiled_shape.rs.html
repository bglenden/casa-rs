2//! Lattice shape with a checked byte-aware physical tile layout.
3
4use casa_tables::TileLayoutPlanner;
5
6use crate::LatticeError;
7
8/// A lattice shape paired with its persisted physical tile shape.
9///
10/// Automatic layouts use the shared byte-aware storage planner. Axis zero is
11/// storage-major, matching casacore. For ordinary spectral cubes this fills
12/// the spatial axes before spectral axes, producing plane-oriented tiles whose
13/// non-plane extents are one. Existing files retain their recorded tile shape.
14#[derive(Debug, Clone)]
15pub struct TiledShape {
16    shape: Vec<usize>,
17    tile_shape: Vec<usize>,
18}
19
20impl TiledShape {
21    fn validate_lattice_shape(shape: &[usize]) -> Result<(), LatticeError> {
22        if let Some(axis) = shape.iter().position(|&extent| extent == 0) {
23            return Err(LatticeError::TileLayout(format!(
24                "lattice extent on axis {axis} must be positive"
25            )));
26        }
27        Ok(())
28    }
29
30    /// Plans a default physical tile shape for the actual element byte size.
31    pub fn new(shape: Vec<usize>, element_bytes: usize) -> Result<Self, LatticeError> {
32        Self::validate_lattice_shape(&shape)?;
33        let plan = TileLayoutPlanner::repository_default()
34            .plan_array(&shape, element_bytes)
35            .map_err(|error| LatticeError::TileLayout(error.to_string()))?;
36        Ok(Self {
37            shape,
38            tile_shape: plan.tile_shape().to_vec(),
39        })
40    }
41
42    /// Creates a shape with a strictly validated explicit physical tile.
43    pub fn with_tile_shape(
44        shape: Vec<usize>,
45        tile_shape: Vec<usize>,
46    ) -> Result<Self, LatticeError> {
47        Self::validate_lattice_shape(&shape)?;
48        let plan = TileLayoutPlanner::repository_default()
49            // The byte size does not affect an explicit plan; one avoids
50            // inventing a type-independent automatic layout.
51            .plan_explicit_array(&shape, 1, &tile_shape)
52            .map_err(|error| LatticeError::TileLayout(error.to_string()))?;
53        Ok(Self {
54            shape,
55            tile_shape: plan.tile_shape().to_vec(),
56        })
57    }
58
59    pub fn shape(&self) -> &[usize] {
60        &self.shape
61    }
62
63    pub fn tile_shape(&self) -> &[usize] {
64        &self.tile_shape
65    }
66}

68#[cfg(test)]
69mod tests {
70    use super::*;
71
72    #[test]
73    fn automatic_layout_uses_real_element_bytes() {
74        let f32_shape = TiledShape::new(vec![4096, 4096, 768], 4).unwrap();
75        let f64_shape = TiledShape::new(vec![4096, 4096, 768], 8).unwrap();
76        assert_eq!(f32_shape.tile_shape(), &[4096, 256, 1]);
77        assert_eq!(f64_shape.tile_shape(), &[4096, 128, 1]);
78    }
79
80    #[test]
81    fn small_shape_uses_full_cell() {
82        let tiled = TiledShape::new(vec![4, 4], 8).unwrap();
83        assert_eq!(tiled.tile_shape(), &[4, 4]);
84    }
85
86    #[test]
87    fn invalid_automatic_shapes_fail() {
88        assert!(TiledShape::new(vec![], 4).is_err());
89        assert!(TiledShape::new(vec![4, 0], 4).is_err());
90        assert!(TiledShape::new(vec![4], 0).is_err());
91    }
92
93    #[test]
94    fn explicit_shape_is_strict() {
95        let tiled = TiledShape::with_tile_shape(vec![100, 100], vec![32, 32]).unwrap();
96        assert_eq!(tiled.tile_shape(), &[32, 32]);
97        let padded = TiledShape::with_tile_shape(vec![10, 10], vec![32, 32]).unwrap();
98        assert_eq!(padded.tile_shape(), &[32, 32]);
99        assert!(TiledShape::with_tile_shape(vec![10, 10], vec![10, 0]).is_err());
100        assert!(TiledShape::with_tile_shape(vec![10, 10], vec![10]).is_err());
101    }
102}
