23/// A typed image/placement handle for a direct Kitty graphics layer.
24#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
25pub struct KittyLayerHandle {
26    image_id: NonZeroU32,
27    placement_id: NonZeroU32,
28}
29
30impl KittyLayerHandle {
31    /// Return the image id used for Kitty image uploads.
32    pub fn image_id(self) -> NonZeroU32 {
33        self.image_id
34    }
35
36    /// Return the placement id used for Kitty image display operations.
37    pub fn placement_id(self) -> NonZeroU32 {
38        self.placement_id
39    }
40}
41
42/// Placement configuration for a direct Kitty graphics layer.
43#[derive(Clone, Copy, Debug, PartialEq, Eq)]
44pub struct KittyPlacement {
45    /// The target terminal cell rectangle.
46    pub rect: Rect,
47    /// The Kitty z-index to use for this placement.
48    pub z_index: i32,
49    /// Whether the current cursor position should be restored after placement.
50    pub preserve_cursor: bool,
51}
52
53/// Errors returned when validating a Kitty placement.
54#[derive(Debug, Error, Clone, PartialEq, Eq)]
55pub enum KittyPlacementError {
56    /// Placements must target a non-empty cell rect.
57    #[error("kitty placement rect must be non-empty, got {width}x{height} cells")]
58    EmptyRect { width: u16, height: u16 },
59}
60
61/// Errors returned by direct Kitty layer operations.
62#[derive(Debug, Error)]
63pub enum KittyLayerError {
64    /// The manager ran out of image or placement ids.
65    #[error("kitty layer id space is exhausted")]
66    IdExhausted,
67    /// The placement was invalid.
68    #[error(transparent)]
69    InvalidPlacement(#[from] KittyPlacementError),
70    /// Writing Kitty escape sequences failed.
71    #[error("failed to write Kitty graphics sequence")]
72    Io(#[from] std::io::Error),
73}
74
75/// Allocates typed handles for direct Kitty graphics layers.
76#[derive(Debug, Default)]
77pub struct KittyLayerManager {
78    next_image_id: u32,
79    next_placement_id: u32,
80}
81
82impl KittyLayerManager {
83    /// Create a new layer manager with monotonic id allocation.
84    pub fn new() -> Self {
85        Self {
86            next_image_id: 1,
87            next_placement_id: 1,
88        }
89    }
90
91    /// Allocate a fresh image/placement handle pair.
92    pub fn allocate(&mut self) -> Result<KittyLayerHandle, KittyLayerError> {
93        let image_id = NonZeroU32::new(self.next_image_id).ok_or(KittyLayerError::IdExhausted)?;
94        let placement_id =
95            NonZeroU32::new(self.next_placement_id).ok_or(KittyLayerError::IdExhausted)?;
96        self.next_image_id = self
97            .next_image_id
98            .checked_add(1)
99            .ok_or(KittyLayerError::IdExhausted)?;
100        self.next_placement_id = self
101            .next_placement_id
102            .checked_add(1)
103            .ok_or(KittyLayerError::IdExhausted)?;
104
105        Ok(KittyLayerHandle {
106            image_id,
107            placement_id,
108        })
109    }
110
111    /// Upload RGBA image data to the terminal using the supplied handle.
112    pub fn upload_rgba<W: Write>(
113        &self,
114        out: &mut W,
115        handle: KittyLayerHandle,
116        image: &RgbaImage,
117    ) -> Result<(), KittyLayerError> {
118        let mut kitty_image: KittyImage<'static> = DynamicImage::ImageRgba8(image.clone()).into();
119        kitty_image.num_or_id = NumberOrId::Id(handle.image_id);
120        Action::Transmit(kitty_image).write_transmit_to(out, Verbosity::Silent)?;
121        Ok(())
122    }
123
124    /// Place a previously uploaded image on the terminal surface.
125    pub fn place<W: Write>(
126        &self,
127        out: &mut W,
128        handle: KittyLayerHandle,
129        placement: KittyPlacement,
130    ) -> Result<(), KittyLayerError> {
131        validate_placement(placement)?;
132
133        let config = DisplayConfig {
134            location: DisplayLocation {
135                columns: placement.rect.width,
136                rows: placement.rect.height,
137                z_index: placement.z_index,
138                ..DisplayLocation::default()
139            },
140            cursor_movement: if placement.preserve_cursor {
141                CursorMovementPolicy::DontMove
142            } else {
143                CursorMovementPolicy::MoveToAfterImage
144            },
145            ..DisplayConfig::default()
146        };
147
148        if placement.preserve_cursor {
149            execute!(
150                out,
151                SavePosition,
152                MoveTo(placement.rect.x, placement.rect.y)
153            )?;
154        } else {
155            execute!(out, MoveTo(placement.rect.x, placement.rect.y))?;
156        }
157
158        Action::Display {
159            image_id: handle.image_id,
160            placement_id: handle.placement_id,
161            config,
162        }
163        .write_transmit_to(&mut *out, Verbosity::Silent)?;
164
165        if placement.preserve_cursor {
166            execute!(out, RestorePosition)?;
167        }
168
169        Ok(())
170    }
171
172    /// Upload a new RGBA image and place it immediately.
173    pub fn upload_and_place_rgba<W: Write>(
174        &self,
175        out: &mut W,
176        handle: KittyLayerHandle,
177        image: &RgbaImage,
178        placement: KittyPlacement,
179    ) -> Result<(), KittyLayerError> {
180        self.upload_rgba(out, handle, image)?;
181        self.place(out, handle, placement)
182    }
183
184    /// Clear the current placement while keeping the uploaded image available in terminal memory.
185    pub fn clear_placement<W: Write>(
186        &self,
187        out: &mut W,
188        handle: KittyLayerHandle,
189    ) -> Result<(), KittyLayerError> {
190        let delete = Action::Delete(DeleteConfig {
191            effect: ClearOrDelete::Clear,
192            which: WhichToDelete::ImageId(handle.image_id, Some(handle.placement_id)),
193        });
194        delete.write_transmit_to(out, Verbosity::Silent)?;
195        Ok(())
196    }
197
198    /// Delete the uploaded image and all of its placements from terminal memory.
199    pub fn delete_image<W: Write>(
200        &self,
201        out: &mut W,
202        handle: KittyLayerHandle,
203    ) -> Result<(), KittyLayerError> {
204        let delete = Action::Delete(DeleteConfig {
205            effect: ClearOrDelete::Delete,
206            which: WhichToDelete::ImageId(handle.image_id, None),
207        });
208        delete.write_transmit_to(out, Verbosity::Silent)?;
209        Ok(())
210    }
211
212    /// Clear the placement and delete the uploaded image.
213    pub fn clear_and_delete<W: Write>(
214        &self,
215        out: &mut W,
216        handle: KittyLayerHandle,
217    ) -> Result<(), KittyLayerError> {
218        self.delete_image(out, handle)
219    }
220}
221
222fn validate_placement(placement: KittyPlacement) -> Result<(), KittyPlacementError> {
223    if placement.rect.is_empty() {
224        return Err(KittyPlacementError::EmptyRect {
225            width: placement.rect.width,
226            height: placement.rect.height,
227        });
228    }
229    Ok(())
230}

239    #[test]
240    fn allocate_returns_unique_handles() {
241        let mut manager = KittyLayerManager::new();
242        let first = manager.allocate().unwrap();
243        let second = manager.allocate().unwrap();
244
245        assert_ne!(first.image_id(), second.image_id());
246        assert_ne!(first.placement_id(), second.placement_id());
247    }

249    #[test]
250    fn reject_empty_rect_placements() {
251        let mut manager = KittyLayerManager::new();
252        let handle = manager.allocate().unwrap();
253        let mut out = Vec::new();
254        let err = manager
255            .place(
256                &mut out,
257                handle,
258                KittyPlacement {
259                    rect: Rect::new(0, 0, 0, 2),
260                    z_index: 1,
261                    preserve_cursor: true,
262                },
263            )
264            .unwrap_err();
265
266        assert!(matches!(
267            err,
268            super::KittyLayerError::InvalidPlacement(KittyPlacementError::EmptyRect {
269                width: 0,
270                height: 2
271            })
272        ));
273    }

275    #[test]
276    fn writes_upload_place_and_delete_sequences() {
277        let mut manager = KittyLayerManager::new();
278        let handle = manager.allocate().unwrap();
279        let image = RgbaImage::from_pixel(1, 1, Rgba([1, 2, 3, 255]));
280        let placement = KittyPlacement {
281            rect: Rect::new(4, 5, 1, 1),
282            z_index: 3,
283            preserve_cursor: true,
284        };
285
286        let mut upload = Vec::new();
287        manager.upload_rgba(&mut upload, handle, &image).unwrap();
288        let upload = String::from_utf8(upload).unwrap();
289        assert!(upload.contains("\u{1b}_Ga=t"));
290
291        let mut place = Vec::new();
292        manager.place(&mut place, handle, placement).unwrap();
293        let place = String::from_utf8(place).unwrap();
294        assert!(place.contains("\u{1b}_Ga=p"));
295        assert!(place.contains(",c=1"));
296        assert!(place.contains(",r=1"));
297        assert!(place.contains(",z=3"));
298
299        let mut clear = Vec::new();
300        manager.clear_placement(&mut clear, handle).unwrap();
301        let clear = String::from_utf8(clear).unwrap();
302        assert!(clear.contains("\u{1b}_Ga=d"));
303        assert!(clear.contains(",d=i"));
304
305        let mut delete = Vec::new();
306        manager.delete_image(&mut delete, handle).unwrap();
307        let delete = String::from_utf8(delete).unwrap();
308        assert!(delete.contains("\u{1b}_Ga=d"));
309        assert!(delete.contains(",d=I"));
310    }

312    #[test]

314        let mut manager = KittyLayerManager::new();

317        let placement = KittyPlacement {

323        let mut out = Vec::new();

327        let rendered = String::from_utf8(out).unwrap();
328        assert!(rendered.contains("\u{1b}_Ga=t"));
329        assert!(rendered.contains("\u{1b}_Ga=p"));

332        let mut delete = Vec::new();

334        let delete = String::from_utf8(delete).unwrap();
335        assert!(delete.contains("\u{1b}_Ga=d"));
336        assert!(delete.contains(",d=I"));
337    }
