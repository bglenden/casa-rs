140pub fn render_image_plane_image(
141    width: u32,
142    height: u32,
143    input: &ImagePlaneRenderInput,
144) -> Result<DynamicImage, String> {
145    ensure_plotters_font()?;
146    let width = width.max(1);
147    let height = height.max(1);
148    let raster_width = input.raster.width.max(1);
149    let raster_height = input.raster.height.max(1);
150    let mut image = RgbImage::new(width, height);
151    let layout = image_plane_layout(
152        width,
153        height,
154        input.display_aspect_ratio,
155        input.display_axes.len() >= 2,
156    );
157    let geometry = layout.image;
158    let highlight = image_browser_highlight(input.theme_mode);
159    let marker_halo = match input.theme_mode {
160        BrowserRenderTheme::DenseAnsi => [0, 0, 0],
161        BrowserRenderTheme::RichPanel => [15, 23, 42],
162    };
163    let marker_center = match input.theme_mode {
164        BrowserRenderTheme::DenseAnsi => [0, 0, 0],
165        BrowserRenderTheme::RichPanel => [248, 250, 252],
166    };
167
168    for out_y in geometry.y..geometry.y + geometry.height {
169        let relative_y = out_y - geometry.y;
170        let raster_y = ((relative_y as usize) * raster_height / geometry.height as usize)
171            .min(raster_height.saturating_sub(1));
172        for out_x in geometry.x..geometry.x + geometry.width {
173            let relative_x = out_x - geometry.x;
174            let raster_x = ((relative_x as usize) * raster_width / geometry.width as usize)
175                .min(raster_width.saturating_sub(1));
176            let value = input
177                .raster
178                .pixels_u8
179                .get(raster_y * raster_width + raster_x)
180                .copied()
181                .unwrap_or_default();
182            image.put_pixel(
183                out_x,
184                out_y,
185                Rgb(image_plane_pixel_color(value, input.colormap, input.invert)),
186            );
187        }
188    }
189
190    if let Some((sampled_width, sampled_height)) = input.sampled_shape {
191        let region_color = Rgb(image_region_overlay_color(input.theme_mode));
192        let region_halo = Rgb(marker_halo);
193        for shape in &input.region_overlay_shapes {
194            if shape.vertices.len() < 2 {
195                continue;
196            }
197            let screen_vertices = shape
198                .vertices
199                .iter()
200                .map(|vertex| {
201                    (
202                        sampled_coordinate_to_screen_x(
203                            geometry,
204                            sampled_width as u32,
205                            vertex.sampled_x,
206                        ),
207                        sampled_coordinate_to_screen_y(
208                            geometry,
209                            sampled_height as u32,
210                            vertex.sampled_y,
211                        ),
212                    )
213                })
214                .collect::<Vec<_>>();
215            for index in 0..screen_vertices.len().saturating_sub(1) {
216                let left = screen_vertices[index];
217                let right = screen_vertices[index + 1];
218                draw_region_segment(&mut image, left, right, region_halo, region_color);
219            }
220            if shape.closed
221                && let (Some(first), Some(last)) = (
222                    screen_vertices.first().copied(),
223                    screen_vertices.last().copied(),
224                )
225            {
226                draw_region_segment(&mut image, last, first, region_halo, region_color);
227            }
228            for &(x, y) in &screen_vertices {
229                draw_probe_marker(&mut image, x, y, 3, region_halo, region_color);
230            }
231        }
232    }
233
234    if let Some((cursor_x, cursor_y)) = input.cursor_sample {
235        let (sampled_width, sampled_height) =
236            input.sampled_shape.unwrap_or((raster_width, raster_height));
237        let marker = plane_cursor_marker_geometry(
238            geometry,
239            sampled_width as u32,
240            sampled_height as u32,
241            cursor_x as u32,
242            cursor_y as u32,
243        );
244        let center_x = marker.x + marker.width.saturating_sub(1) / 2;
245        let center_y = marker.y + marker.height.saturating_sub(1) / 2;
246        let guide_radius = (geometry.width.min(geometry.height) / 18).clamp(4, 12);
247        draw_cursor_guides(
248            &mut image,
249            geometry,
250            center_x,
251            center_y,
252            guide_radius,
253            Rgb(marker_halo),
254            Rgb(highlight),
255        );
256        draw_rect_outline(
257            &mut image,
258            marker.x,
259            marker.y,
260            marker.width,
261            marker.height,
262            Rgb(highlight),
263        );
264        let radius = if marker.width.max(marker.height) >= 8 {
265            2
266        } else {
267            1
268        };
269        let reticle = guide_radius.max(radius + 2);
270        let image_width = image.width();
271        let image_height = image.height();
272        let outer_x = center_x.saturating_sub(reticle + 1);
273        let outer_y = center_y.saturating_sub(reticle + 1);
274        let outer_width = (reticle * 2 + 3).min(image_width.saturating_sub(outer_x));
275        let outer_height = (reticle * 2 + 3).min(image_height.saturating_sub(outer_y));
276        draw_rect_outline(
277            &mut image,
278            outer_x,
279            outer_y,
280            outer_width,
281            outer_height,
282            Rgb(marker_halo),
283        );
284        let inner_x = center_x.saturating_sub(reticle);
285        let inner_y = center_y.saturating_sub(reticle);
286        let inner_width = (reticle * 2 + 1).min(image_width.saturating_sub(inner_x));
287        let inner_height = (reticle * 2 + 1).min(image_height.saturating_sub(inner_y));
288        draw_rect_outline(
289            &mut image,
290            inner_x,
291            inner_y,
292            inner_width,
293            inner_height,
294            Rgb(highlight),
295        );
296        draw_cross(&mut image, center_x, center_y, radius, Rgb(highlight));
297        image.put_pixel(center_x, center_y, Rgb(marker_center));
298    }
299
300    if let Some((sampled_width, sampled_height)) = input.sampled_shape {
301        let probe_radius = probe_marker_radius(geometry);
302        for overlay in &input.overlay_markers {
303            let marker = plane_cursor_marker_geometry(
304                geometry,
305                sampled_width as u32,
306                sampled_height as u32,
307                overlay.sample.0 as u32,
308                overlay.sample.1 as u32,
309            );
310            let color = Rgb(image_probe_overlay_color(
311                input.theme_mode,
312                overlay.color_index,
313            ));
314            let center_x = marker.x + marker.width.saturating_sub(1) / 2;
315            let center_y = marker.y + marker.height.saturating_sub(1) / 2;
316            draw_probe_marker(
317                &mut image,
318                center_x,
319                center_y,
320                probe_radius,
321                Rgb(marker_halo),
322                color,
323            );
324        }
325    }
326
327    draw_plane_frame_label(&mut image, geometry, input)?;
328
329    if input.display_axes.len() >= 2 {
330        draw_plane_axis_annotations(&mut image, layout, input)?;
331    }
332    if layout.right_gutter >= 48 && !input.raster.no_finite_values {
333        draw_plane_scale_sidebar(&mut image, layout, input)?;
334    }
335
336    Ok(DynamicImage::ImageRgb8(image))
337}

339pub fn render_image_spectrum_image(
340    width: u32,
341    height: u32,
342    input: &ImageSpectrumRenderInput,
343) -> Result<DynamicImage, String> {
344    ensure_plotters_font()?;
345    let width = width.max(1);
346    let height = height.max(1);
347    let mut image = RgbImage::new(width, height);
348    let highlight = image_browser_highlight(input.theme_mode);
349    let (background, border, axis_text, series, marker, guide_line, zero_line) =
350        match input.theme_mode {
351            BrowserRenderTheme::DenseAnsi => (
352                [0, 0, 0],
353                [96, 96, 96],
354                [224, 224, 224],
355                [96, 208, 255],
356                highlight,
357                highlight,
358                [48, 48, 48],
359            ),
360            BrowserRenderTheme::RichPanel => (
361                [15, 23, 42],
362                [94, 234, 212],
363                [226, 232, 240],
364                [125, 211, 252],
365                highlight,
366                highlight,
367                [55, 65, 81],
368            ),
369        };
370    fill_rect(&mut image, 0, 0, width, height, Rgb(background));
371
372    if width < 4 || height < 4 {
373        return Ok(DynamicImage::ImageRgb8(image));
374    }
375
376    let layout = image_spectrum_layout(width, height);
377    let plot_left = layout.plot.x;
378    let plot_top = layout.plot.y;
379    let plot_width = layout.plot.width;
380    let plot_height = layout.plot.height;
381    draw_rect_outline(
382        &mut image,
383        plot_left,
384        plot_top,
385        plot_width,
386        plot_height,
387        Rgb(border),
388    );
389
390    let x_domain = spectrum_x_domain(input);
391    let has_spectrum_values = spectrum_value_range(input).is_some();
392    let (mut min_value, mut max_value) = spectrum_value_range(input).unwrap_or((-1.0, 1.0));
393    if (max_value - min_value).abs() < f64::EPSILON {
394        min_value -= 1.0;
395        max_value += 1.0;
396    }
397
398    if min_value <= 0.0 && max_value >= 0.0 {
399        let zero_y = plot_value_y(0.0, min_value, max_value, plot_top, plot_height);
400        draw_horizontal_line(
401            &mut image,
402            plot_left,
403            plot_left + plot_width.saturating_sub(1),
404            zero_y,
405            Rgb(zero_line),
406        );
407    }
408
409    for overlay in &input.overlay_profiles {
410        draw_profile_series(
411            &mut image,
412            &overlay.profile,
413            plot_left,
414            plot_top,
415            plot_width,
416            plot_height,
417            x_domain,
418            min_value,
419            max_value,
420            Rgb(image_probe_overlay_color(
421                input.theme_mode,
422                overlay.color_index,
423            )),
424        );
425    }
426    draw_profile_series(
427        &mut image,
428        &input.profile,
429        plot_left,
430        plot_top,
431        plot_width,
432        plot_height,
433        x_domain,
434        min_value,
435        max_value,
436        Rgb(series),
437    );
438
439    draw_spectrum_axes(
440        &mut image, layout, input, x_domain, min_value, max_value, border, axis_text,
441    )?;
442    draw_selected_spectrum_marker(
443        &mut image,
444        &input.profile,
445        plot_left,
446        plot_top,
447        plot_width,
448        plot_height,
449        x_domain,
450        min_value,
451        max_value,
452        has_spectrum_values,
453        Rgb(guide_line),
454        Rgb(marker),
455    );
456
457    Ok(DynamicImage::ImageRgb8(image))
458}

460#[allow(clippy::too_many_arguments)]
461fn draw_spectrum_axes(
462    image: &mut RgbImage,
463    layout: ImageSpectrumLayout,
464    input: &ImageSpectrumRenderInput,
465    x_domain: (f64, f64),
466    min_value: f64,
467    max_value: f64,
468    border: [u8; 3],
469    axis_text: [u8; 3],
470) -> Result<(), String> {
471    let x_scale = spectrum_x_scale(input, x_domain);
472    let y_step = nice_numeric_step((max_value - min_value).abs() / 4.0);
473    let y_ticks = numeric_ticks(min_value, max_value, y_step, layout.plot.height)
474        .into_iter()
475        .map(|tick| SpectrumAxisTick {
476            value: tick.value,
477            label: format_spectrum_axis_numeric(tick.value, &input.profile.value_unit, y_step),
478            position_px: plot_value_y(tick.value, min_value, max_value, 0, layout.plot.height)
479                .min(layout.plot.height.saturating_sub(1)),
480        })
481        .collect::<Vec<_>>();
482    let x_ticks = spectrum_x_ticks(&x_scale, x_domain, layout.plot.width);
483
484    let border = RGBColor(border[0], border[1], border[2]);
485    let axis_text = RGBColor(axis_text[0], axis_text[1], axis_text[2]);
486    let image_width = image.width();
487    let image_height = image.height();
488    let area =
489        BitMapBackend::with_buffer(image.as_mut(), (image_width, image_height)).into_drawing_area();
490    let left_x = layout.plot.x as i32;
491    let bottom_y = (layout.plot.y + layout.plot.height.saturating_sub(1)) as i32;
492    let tick_font_size = spectrum_tick_font_size(layout.plot);
493    let axis_font_size = spectrum_axis_font_size(layout.plot);
494    let tick_font = ("sans-serif", tick_font_size).into_font().color(&axis_text);
495    let axis_font = ("sans-serif", axis_font_size).into_font().color(&axis_text);
496    let tick_length = spectrum_tick_length(layout.plot) as i32;
497
498    for tick in &x_ticks {
499        let x = (layout.plot.x + tick.position_px) as i32;
500        area.draw(&PathElement::new(
501            vec![(x, bottom_y), (x, bottom_y + tick_length)],
502            border,
503        ))
504        .map_err(|error| error.to_string())?;
505        area.draw(&Text::new(
506            tick.label.clone(),
507            (x, bottom_y + tick_length + 4),
508            tick_font.clone().pos(Pos::new(HPos::Center, VPos::Top)),
509        ))
510        .map_err(|error| error.to_string())?;
511    }
512    for tick in &y_ticks {
513        let y = (layout.plot.y + tick.position_px) as i32;
514        area.draw(&PathElement::new(
515            vec![(left_x - tick_length, y), (left_x, y)],
516            border,
517        ))
518        .map_err(|error| error.to_string())?;
519        area.draw(&Text::new(
520            tick.label.clone(),
521            (left_x - tick_length - 4, y),
522            tick_font.clone().pos(Pos::new(HPos::Right, VPos::Center)),
523        ))
524        .map_err(|error| error.to_string())?;
525    }
526
527    area.draw(&Text::new(
528        spectrum_x_axis_title(&x_scale, input),
529        (
530            (layout.plot.x + layout.plot.width / 2) as i32,
531            image_height.saturating_sub(4) as i32,
532        ),
533        axis_font.clone().pos(Pos::new(HPos::Center, VPos::Bottom)),
534    ))
535    .map_err(|error| error.to_string())?;
536    let y_title_x = spectrum_y_title_x(layout, tick_length, tick_font_size, axis_font_size);
537    area.draw(&Text::new(
538        spectrum_y_axis_title(input),
539        (y_title_x, (layout.plot.y + layout.plot.height / 2) as i32),
540        ("sans-serif", axis_font_size)
541            .into_font()
542            .transform(FontTransform::Rotate270)
543            .color(&axis_text)
544            .pos(Pos::new(HPos::Center, VPos::Center)),
545    ))
546    .map_err(|error| error.to_string())?;
547    area.present().map_err(|error| error.to_string())
548}

550pub fn image_plane_draw_geometry(
551    width: u32,
552    height: u32,
553    display_aspect_ratio: Option<f64>,
554) -> ImagePlaneDrawGeometry {
555    let width = width.max(1);
556    let height = height.max(1);
557    let Some(desired_aspect) =
558        display_aspect_ratio.filter(|ratio| ratio.is_finite() && *ratio > 0.0)
559    else {
560        return ImagePlaneDrawGeometry {
561            x: 0,
562            y: 0,
563            width,
564            height,
565        };
566    };
567    let canvas_aspect = width as f64 / height as f64;
568    if (canvas_aspect - desired_aspect).abs() < f64::EPSILON {
569        return ImagePlaneDrawGeometry {
570            x: 0,
571            y: 0,
572            width,
573            height,
574        };
575    }
576    if canvas_aspect > desired_aspect {
577        let draw_width = ((height as f64 * desired_aspect).round() as u32).clamp(1, width);
578        ImagePlaneDrawGeometry {
579            x: (width - draw_width) / 2,
580            y: 0,
581            width: draw_width,
582            height,
583        }
584    } else {
585        let draw_height = ((width as f64 / desired_aspect).round() as u32).clamp(1, height);
586        ImagePlaneDrawGeometry {
587            x: 0,
588            y: (height - draw_height) / 2,
589            width,
590            height: draw_height,
591        }
592    }
593}

595pub fn image_spectrum_layout(width: u32, height: u32) -> ImageSpectrumLayout {
596    let width = width.max(1);
597    let height = height.max(1);
598    let left_gutter = if width >= 180 {
599        72
600    } else if width >= 120 {
601        56
602    } else {
603        (width / 4).clamp(16, 32)
604    }
605    .min(width.saturating_sub(2));
606    let right_gutter = 8.min(width.saturating_sub(left_gutter + 1));
607    let top_gutter = 8.min(height.saturating_sub(2));
608    let bottom_gutter = if height >= 120 {
609        36
610    } else if height >= 72 {
611        28
612    } else {
613        (height / 4).clamp(12, 18)
614    }
615    .min(height.saturating_sub(top_gutter + 1));
616    let plot_width = width.saturating_sub(left_gutter + right_gutter).max(1);
617    let plot_height = height.saturating_sub(top_gutter + bottom_gutter).max(1);
618    ImageSpectrumLayout {
619        plot: ImagePlaneDrawGeometry {
620            x: left_gutter,
621            y: top_gutter,
622            width: plot_width,
623            height: plot_height,
624        },
625        left_gutter,
626        top_gutter,
627        right_gutter,
628        bottom_gutter,
629    }
630}

632pub fn image_plane_layout(
633    width: u32,
634    height: u32,
635    display_aspect_ratio: Option<f64>,
636    axis_annotations: bool,
637) -> ImagePlaneLayout {
638    let width = width.max(1);
639    let height = height.max(1);
640    let show_axes = axis_annotations && width >= 160 && height >= 120;
641    let (left_gutter, top_gutter, right_gutter, bottom_gutter) = if show_axes {
642        (
643            (width / 5).clamp(88, 144),
644            12,
645            (width / 6).clamp(88, 132),
646            (height / 4).clamp(52, 84),
647        )
648    } else {
649        (0, 0, 0, 0)
650    };
651    let inner_width = width.saturating_sub(left_gutter + right_gutter).max(1);
652    let inner_height = height.saturating_sub(top_gutter + bottom_gutter).max(1);
653    let base = image_plane_draw_geometry(inner_width, inner_height, display_aspect_ratio);

1126fn plane_axis_ticks(
1127    axis: &ImageDisplayAxisState,
1128    probe: Option<&ImageBrowserProbe>,
1129    geometry_len: u32,
1130    reverse: bool,
1131) -> Vec<PlaneAxisTick> {
1132    if axis.sampled_len == 0 || geometry_len == 0 {
1133        return Vec::new();
1134    }
1135    if let Some(ticks) = plane_axis_nice_ticks(axis, probe, geometry_len, reverse) {
1136        if !ticks.is_empty() {
1137            return ticks;
1138        }
1139    }
1140    let tick_count: usize = if geometry_len >= 520 {
1141        5
1142    } else if geometry_len >= 300 {
1143        4
1144    } else {
1145        3
1146    };
1147    let max_index = axis.sampled_len.saturating_sub(1);
1148    let mut indices = (0..tick_count)
1149        .map(|step| {
1150            if tick_count == 1 {
1151                0
1152            } else {
1153                ((step as f64 * max_index as f64) / (tick_count.saturating_sub(1) as f64)).round()
1154                    as usize
1155            }
1156        })
1157        .collect::<Vec<_>>();
1158    indices.dedup();
1159    indices
1160        .into_iter()
1161        .map(|sample_index| {
1162            let pixel = axis
1163                .blc
1164                .saturating_add(sample_index.saturating_mul(axis.inc.max(1)));
1165            let position_px = if axis.sampled_len <= 1 {
1166                geometry_len / 2
1167            } else {
1168                ((sample_index as f64) * geometry_len.saturating_sub(1) as f64 / max_index as f64)
1169                    .round() as u32
1170            };
1171            let position_px = if reverse {
1172                geometry_len.saturating_sub(1).saturating_sub(position_px)
1173            } else {
1174                position_px
1175            };
1176            PlaneAxisTick {
1177                pixel,
1178                label: plane_axis_tick_label(axis, probe, pixel),
1179                position_px,
1180            }
1181        })
1182        .collect()
1183}

1185fn plane_axis_nice_ticks(
1186    axis: &ImageDisplayAxisState,
1187    probe: Option<&ImageBrowserProbe>,
1188    geometry_len: u32,
1189    reverse: bool,
1190) -> Option<Vec<PlaneAxisTick>> {
1191    let probe = probe?;
1192    let probe_world = probe.world_axes.get(axis.axis)?.value;
1193    let probe_pixel = *probe.pixel_indices.get(axis.axis)? as f64;
1194    let increment = axis.world_increment?;
1195    if !probe_world.is_finite() || !increment.is_finite() || increment == 0.0 {
1196        return None;
1197    }
1198    let scale = plane_axis_scale(axis, probe_world, increment)?;
1199    let start_pixel = axis.blc as f64;
1200    let end_pixel = axis.trc as f64;
1201    let start_display = scale.display_at_pixel(probe_pixel, start_pixel);
1202    let end_display = scale.display_at_pixel(probe_pixel, end_pixel);
1203    let min_display = start_display.min(end_display);
1204    let max_display = start_display.max(end_display);
1205    let step = scale.nice_step(max_display - min_display)?;
1206    if !step.is_finite() || step <= 0.0 {
1207        return None;
1208    }
1209    let epsilon = step * 1e-6;
1210    let first = (min_display / step).ceil() * step;
1211    let mut value = first;
1212    let mut ticks = Vec::new();
1213    while value <= max_display + epsilon {
1214        let pixel = scale.pixel_at_display(probe_pixel, value);
1215        let sample_index = (pixel - axis.blc as f64) / axis.inc.max(1) as f64;
1216        if sample_index >= -epsilon
1217            && sample_index <= axis.sampled_len.saturating_sub(1) as f64 + epsilon
1218        {
1219            let position_px = if axis.sampled_len <= 1 {
1220                geometry_len / 2
1221            } else {
1222                ((sample_index.clamp(0.0, axis.sampled_len.saturating_sub(1) as f64)
1223                    / axis.sampled_len.saturating_sub(1) as f64)
1224                    * geometry_len.saturating_sub(1) as f64)
1225                    .round() as u32
1226            };
1227            let position_px = if reverse {
1228                geometry_len.saturating_sub(1).saturating_sub(position_px)
1229            } else {
1230                position_px
1231            };
1232            ticks.push(PlaneAxisTick {
1233                pixel: pixel.round().max(0.0) as usize,
1234                label: scale.label(value, step),
1235                position_px,
1236            });
1237        }
1238        value += step;
1239    }
1240    if ticks.len() >= 2 { Some(ticks) } else { None }
1241}

1253impl AxisDisplayScale {
1254    fn display_at_pixel(&self, probe_pixel: f64, pixel: f64) -> f64 {
1255        self.probe_display + (pixel - probe_pixel) * self.increment_display
1256    }
1257
1258    fn pixel_at_display(&self, probe_pixel: f64, display_value: f64) -> f64 {
1259        probe_pixel + (display_value - self.probe_display) / self.increment_display
1260    }
1261
1262    fn nice_step(&self, span: f64) -> Option<f64> {
1263        let span = span.abs();
1264        if !span.is_finite() || span == 0.0 {
1265            return None;
1266        }

1687    clip_max: f64,

1730fn nice_ticks_with_step(clip_min: f64, clip_max: f64, step: f64) -> Vec<PlaneSidebarTick> {
1731    if !step.is_finite() || step <= 0.0 {
1732        return Vec::new();
1733    }
1734    let epsilon = step * 1.0e-6;
1735    let start = (clip_min / step).ceil() * step;
1736    let end = (clip_max / step).floor() * step;
1737    if start > end + epsilon {
1738        return Vec::new();
1739    }
1740    let mut ticks = Vec::new();
1741    let mut value = start;
1742    while value <= end + epsilon {
1743        let clamped = value.clamp(clip_min, clip_max);
1744        let offset = ((clip_max - clamped) / (clip_max - clip_min)).clamp(0.0, 1.0);
1745        ticks.push(PlaneSidebarTick {
1746            offset,
1747            value: normalize_nice_tick_value(clamped, step),
1748        });
1749        value += step;
1750    }
1751    ticks.sort_by(|left, right| left.offset.total_cmp(&right.offset));
1752    ticks
1753}

1859fn interpolate_color_stops(value: u8, stops: &[[u8; 3]]) -> [u8; 3] {
1860    if stops.is_empty() {
1861        return [value, value, value];
1862    }
1863    if stops.len() == 1 {
1864        return stops[0];
1865    }
1866    let segment_count = stops.len() - 1;
1867    let scaled = usize::from(value) * segment_count * 256 / 255;
1868    let segment = (scaled / 256).min(segment_count - 1);
1869    let fraction = (scaled % 256) as u16;

2325fn draw_line(image: &mut RgbImage, x0: u32, y0: u32, x1: u32, y1: u32, color: Rgb<u8>) {

2327    let mut y0 = y0 as i32;
2328    let x1 = x1 as i32;
2329    let y1 = y1 as i32;
2330    let dx = (x1 - x0).abs();
2331    let sx = if x0 < x1 { 1 } else { -1 };
2332    let dy = -(y1 - y0).abs();
2333    let sy = if y0 < y1 { 1 } else { -1 };
2334    let mut error = dx + dy;
2335
2336    loop {
2337        if x0 >= 0 && y0 >= 0 && (x0 as u32) < image.width() && (y0 as u32) < image.height() {
2338            image.put_pixel(x0 as u32, y0 as u32, color);
2339        }
2340        if x0 == x1 && y0 == y1 {
2341            break;
2342        }
2343        let twice_error = 2 * error;
2344        if twice_error >= dy {
2345            error += dy;
2346            x0 += sx;
2347        }
2348        if twice_error <= dx {
2349            error += dx;
2350            y0 += sy;
2351        }
2352    }
