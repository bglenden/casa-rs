2//! CASA-table-backed measures runtime data for astronomical conversions.
3//!
4//! The runtime model follows CASA's layout and defaults:
5//!
6//! - default measures path: `~/.casa/data`
7//! - preferred source: an existing CASA/casaconfig-populated tree
8//! - fallback for the default path only: a packaged CASA-table snapshot shipped
9//!   with `casa-rs` and unpacked into `~/.casa/data` on first use
10//! - explicit override: `CASA_RS_MEASURESPATH` (with deprecated compatibility
11//!   alias `CASA_RS_DATA`)
12

67    }

72/// Errors raised while discovering or reading the standard measures runtime.
73#[derive(Debug, Clone)]
74pub enum MeasuresDataError {
75    /// The runtime measures path is unavailable or incomplete.
76    RuntimePath(String),
77    /// A packaged snapshot bootstrap step failed.
78    Bootstrap(String),
79    /// Reading a CASA table failed.
80    TableRead(String),
81    /// The packaged provenance metadata is invalid.
82    Provenance(String),
83    /// The requested data is unavailable from the runtime tree.
84    MissingData(String),
85}
86
87impl fmt::Display for MeasuresDataError {
88    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
89        match self {
90            Self::RuntimePath(msg) => write!(f, "measures runtime path error: {msg}"),
91            Self::Bootstrap(msg) => write!(f, "measures bootstrap error: {msg}"),
92            Self::TableRead(msg) => write!(f, "measures table read error: {msg}"),
93            Self::Provenance(msg) => write!(f, "measures provenance error: {msg}"),
94            Self::MissingData(msg) => write!(f, "measures data missing: {msg}"),
95        }
96    }
97}

192/// In-memory EOP table with daily entries and interpolation.
193#[derive(Debug, Clone)]
194pub struct EopTable {
195    entries: Vec<EopEntry>,
196    mjd_start: f64,
197    mjd_end: f64,
198}
199
200impl EopTable {
201    /// Parse an EOP table from raw `finals2000A.data` content.
202    pub fn from_finals2000a(content: &str) -> Result<Self, EopError> {
203        let entries = parser::parse_finals2000a(content)?;
204        let mjd_start = entries[0].mjd;
205        let mjd_end = entries[entries.len() - 1].mjd;
206        Ok(Self {
207            entries,
208            mjd_start,
209            mjd_end,
210        })
211    }
212
213    /// Load a raw `finals2000A.data` file.
214    pub fn from_file(path: impl AsRef<Path>) -> Result<Self, EopError> {
215        let content = std::fs::read_to_string(path.as_ref())
216            .map_err(|error| EopError::IoError(format!("{}: {error}", path.as_ref().display())))?;
217        Self::from_finals2000a(&content)
218    }
219
220    /// Return the standard runtime EOP table.
221    pub fn bundled() -> &'static Self {
222        load_eop().0
223    }
224
225    fn from_entries(entries: Vec<EopEntry>) -> Result<Self, MeasuresDataError> {
226        if entries.is_empty() {
227            return Err(MeasuresDataError::MissingData(
228                "EOP runtime tables produced no rows".to_string(),
229            ));
230        }
231        let mjd_start = entries[0].mjd;
232        let mjd_end = entries[entries.len() - 1].mjd;
233        Ok(Self {
234            entries,
235            mjd_start,
236            mjd_end,
237        })
238    }
239
240    /// Interpolate EOP values for the given MJD.
241    pub fn interpolate(&self, mjd: f64) -> Option<EopValues> {
242        interp::interpolate(self.entries(), mjd)
243    }
244
245    /// Returns the covered MJD range.
246    pub fn mjd_range(&self) -> (f64, f64) {
247        (self.mjd_start, self.mjd_end)
248    }
249
250    /// Returns a summary of the table contents.
251    pub fn summary(&self) -> EopSummary {
252        let num_entries = self.entries.len();
253        let num_measured = self
254            .entries
255            .iter()
256            .filter(|entry| !entry.is_predicted)
257            .count();
258        let num_predicted = num_entries - num_measured;
259        let last_measured_mjd = self
260            .entries
261            .iter()
262            .rev()
263            .find(|entry| !entry.is_predicted)
264            .map(|entry| entry.mjd)
265            .unwrap_or(self.mjd_start);
266        EopSummary {
267            num_entries,
268            mjd_start: self.mjd_start,
269            mjd_end: self.mjd_end,
270            last_measured_mjd,
271            num_measured,
272            num_predicted,
273        }
274    }
275
276    /// Returns all entries in the table.
277    pub fn entries(&self) -> &[EopEntry] {
278        &self.entries
279    }
280
281    /// Returns the number of days since the last measured entry.
282    pub fn days_since_last_measured(&self, current_mjd: f64) -> f64 {
283        let last = self
284            .entries
285            .iter()
286            .rev()
287            .find(|entry| !entry.is_predicted)
288            .map(|entry| entry.mjd)
289            .unwrap_or(self.mjd_start);
290        current_mjd - last
291    }
292}

294#[derive(Debug, Clone)]
295struct TaiUtcEntry {
296    mjd: f64,
297    d_utc: f64,
298    offset: f64,
299    multiplier: f64,
300}
301
302#[derive(Debug, Clone)]
303struct TaiUtcTable {
304    entries: Vec<TaiUtcEntry>,
305}
306
307impl TaiUtcTable {
308    fn tai_minus_utc_seconds(&self, utc_mjd: f64) -> Result<f64, MeasuresDataError> {
309        let entry = self
310            .entries
311            .iter()
312            .rev()
313            .find(|entry| utc_mjd >= entry.mjd)
314            .or_else(|| self.entries.first())
315            .ok_or_else(|| {
316                MeasuresDataError::MissingData("TAI_UTC table contains no rows".to_string())
317            })?;
318        Ok(entry.d_utc + (utc_mjd - entry.offset) * entry.multiplier)
319    }
320
321    fn utc_mjd_from_tai_mjd(&self, tai_mjd: f64) -> Result<f64, MeasuresDataError> {
322        let mut utc_mjd = tai_mjd;
323        for _ in 0..8 {
324            let offset_days = self.tai_minus_utc_seconds(utc_mjd)? / 86_400.0;
325            let next = tai_mjd - offset_days;
326            if (next - utc_mjd).abs() < 1e-14 {
327                return Ok(next);
328            }
329            utc_mjd = next;
330        }
331        Ok(utc_mjd)
332    }
333}

335#[derive(Debug, Clone)]
336struct IgrfTable {
337    years: Vec<f64>,
338    coeffs_by_year: Vec<Vec<f64>>,
339    secular_variation: Vec<f64>,
340    nmax: usize,
341}
342
343impl IgrfTable {
344    fn coefficients_for_decimal_year(
345        &self,
346        decimal_year: f64,
347    ) -> Result<(Vec<f64>, usize), MeasuresDataError> {
348        let min_year = *self.years.first().ok_or_else(|| {
349            MeasuresDataError::MissingData("IGRF table has no year rows".to_string())
350        })?;
351        let max_year = self.years.last().copied().unwrap_or(min_year) + 5.0;
352        if decimal_year < min_year || decimal_year > max_year {
353            return Err(MeasuresDataError::MissingData(format!(
354                "IGRF date must be between {min_year:.0}-01-01 and {max_year:.0}-12-31"
355            )));
356        }
357
358        let coeffs = if decimal_year >= *self.years.last().expect("years non-empty") {
359            extrapolate_coefficients(
360                decimal_year,
361                *self.years.last().expect("years non-empty"),
362                self.coeffs_by_year.last().expect("coefficients non-empty"),
363                &self.secular_variation,
364            )
365        } else {
366            let upper = self
367                .years
368                .iter()
369                .position(|year| *year > decimal_year)
370                .expect("decimal year below final row has upper interval");
371            interpolate_coefficients(
372                decimal_year,
373                self.years[upper - 1],
374                &self.coeffs_by_year[upper - 1],
375                self.years[upper],
376                &self.coeffs_by_year[upper],
377            )
378        };
379
380        Ok((coeffs, self.nmax))
381    }
382}

392/// Maximum age (in days) of the packaged snapshot's measures date before it is
393/// considered stale.
394pub const PACKAGED_STALENESS_THRESHOLD_DAYS: f64 = 180.0;
395
396/// Return the packaged snapshot provenance.
397pub fn packaged_snapshot_provenance() -> Result<&'static SnapshotProvenance, MeasuresDataError> {
398    match PACKAGED_PROVENANCE.get_or_init(|| {
399        serde_json::from_str(PACKAGED_SNAPSHOT_PROVENANCE_JSON).map_err(|error| error.to_string())
400    }) {
401        Ok(provenance) => Ok(provenance),
402        Err(error) => Err(MeasuresDataError::Provenance(error.clone())),
403    }
404}

406/// Check whether the packaged snapshot is stale relative to the given MJD.
407pub fn check_packaged_snapshot_freshness(current_mjd: f64) -> Result<f64, f64> {
408    let version = packaged_snapshot_provenance()
409        .ok()
410        .map(|provenance| provenance.measures_version.clone())
411        .unwrap_or_default();
412    let days_old = parse_measures_version_mjd_age(&version, current_mjd).unwrap_or(f64::INFINITY);
413    if days_old > PACKAGED_STALENESS_THRESHOLD_DAYS {
414        Err(days_old)
415    } else {
416        Ok(days_old)
417    }
418}

495fn resolve_standard_measures_root() -> Result<(PathBuf, &'static str), MeasuresDataError> {
496    let default_path = default_measures_path()?;
497    resolve_standard_measures_root_inner(
498        std::env::var(DEFAULT_MEASURES_ENV).ok().map(PathBuf::from),
499        std::env::var(LEGACY_MEASURES_ENV).ok().map(PathBuf::from),
500        default_path,
501    )
502}
503
504fn resolve_standard_measures_root_inner(
505    explicit_override: Option<PathBuf>,
506    legacy_override: Option<PathBuf>,
507    default_path: PathBuf,
508) -> Result<(PathBuf, &'static str), MeasuresDataError> {
509    if let Some(path) = explicit_override {
510        if measures_tree_complete(&path) {
511            return Ok((path, "$CASA_RS_MEASURESPATH"));
512        }
513        return Err(MeasuresDataError::RuntimePath(format!(
514            "{DEFAULT_MEASURES_ENV} points to an incomplete measures tree"
515        )));
516    }
517
518    if let Some(path) = legacy_override {
519        if measures_tree_complete(&path) {
520            return Ok((path, "$CASA_RS_DATA"));
521        }
522        return Err(MeasuresDataError::RuntimePath(format!(
523            "{LEGACY_MEASURES_ENV} points to an incomplete measures tree"
524        )));
525    }
526
527    let source = if measures_tree_complete(&default_path) {
528        "~/.casa/data"
529    } else {
530        bootstrap_packaged_snapshot(&default_path)?;
531        "~/.casa/data (bootstrapped)"
532    };
533    Ok((default_path, source))
534}
535
536fn default_measures_path() -> Result<PathBuf, MeasuresDataError> {
537    let home = std::env::var("HOME").map_err(|error| {
538        MeasuresDataError::RuntimePath(format!(
539            "HOME is not set for default measures path: {error}"
540        ))
541    })?;
542    Ok(PathBuf::from(home).join(DEFAULT_MEASURES_RELATIVE))
543}
544
545fn measures_tree_complete(root: &Path) -> bool {
546    REQUIRED_RELATIVE_PATHS
547        .iter()
548        .all(|relative| root.join(relative).is_file())
549        && REQUIRED_NONEMPTY_DIRS.iter().all(|relative| {
550            root.join(relative).is_dir()
551                && root
552                    .join(relative)
553                    .read_dir()
554                    .map(|mut entries| entries.next().is_some())
555                    .unwrap_or(false)
556        })
557}
558
559fn bootstrap_packaged_snapshot(root: &Path) -> Result<(), MeasuresDataError> {
560    std::fs::create_dir_all(root).map_err(|error| {
561        MeasuresDataError::Bootstrap(format!("creating {}: {error}", root.display()))
562    })?;
563
564    let decoder = GzDecoder::new(PACKAGED_SNAPSHOT_ARCHIVE);
565    let mut archive = Archive::new(decoder);
566    archive.unpack(root).map_err(|error| {
567        MeasuresDataError::Bootstrap(format!(
568            "extracting packaged snapshot into {}: {error}",
569            root.display()
570        ))
571    })?;
572
573    if !measures_tree_complete(root) {
574        return Err(MeasuresDataError::Bootstrap(format!(
575            "packaged snapshot extracted into {} but required tables are still missing",
576            root.display()
577        )));
578    }
579    Ok(())
580}

582fn load_standard_eop() -> Result<(EopTable, &'static str), MeasuresDataError> {
583    let (root, source) = standard_measures_path()?;
584    let measured = PlainTable::open(&root.join("geodetic/IERSeop2000"))?;
585    let predicted = PlainTable::open(&root.join("geodetic/IERSpredict2000"))?;
586
587    let measured_rows = eop_entries_from_table(&measured, false)?;
588    let mut combined = measured_rows;
589    let last_measured = combined
590        .last()
591        .map(|entry| entry.mjd)
592        .unwrap_or(f64::NEG_INFINITY);
593    for entry in eop_entries_from_table(&predicted, true)? {
594        if entry.mjd > last_measured {
595            combined.push(entry);
596        }
597    }
598
599    Ok((EopTable::from_entries(combined)?, source))
600}

602fn eop_entries_from_table(
603    table: &PlainTable,
604    is_predicted: bool,
605) -> Result<Vec<EopEntry>, MeasuresDataError> {
606    let mjd = table.scalar_f64("MJD")?;
607    let x = table.scalar_f64("x")?;
608    let y = table.scalar_f64("y")?;
609    let dut1 = table.scalar_f64("dUT1")?;
610    let lod = table.scalar_f64("LOD")?;
611    let dx = table.scalar_f64("dX")?;
612    let dy = table.scalar_f64("dY")?;
613    let mut entries = Vec::with_capacity(table.row_count());
614    for row in 0..table.row_count() {

685fn load_standard_igrf() -> Result<IgrfTable, MeasuresDataError> {
686    let (root, _source) = standard_measures_path()?;
687    let table = PlainTable::open(&root.join("geodetic/IGRF"))?;
688    let mjd = table.scalar_f64("MJD")?;
689    let mut years = Vec::with_capacity(table.row_count());
690    let mut coeffs_by_year = Vec::with_capacity(table.row_count());
691    let mut secular_variation = Vec::new();
692    let mut nmax = 0usize;
693
694    for (row, mjd_value) in mjd.iter().copied().enumerate().take(table.row_count()) {
695        years.push(decimal_year_from_mjd(mjd_value));
696        let coeffs = table.array_f64_cell("COEF", row)?.to_vec();
697        nmax = solve_igrf_nmax(coeffs.len())?;
698        coeffs_by_year.push(coeffs);
699        secular_variation = table.array_f64_cell("dCOEF", row)?.to_vec();
700    }
701
702    Ok(IgrfTable {
703        years,
704        coeffs_by_year,
705        secular_variation,
706        nmax,
707    })
708}

710fn decimal_year_from_mjd(mjd: f64) -> f64 {
711    let jd = mjd + 2_400_000.5;
712    let z = (jd + 0.5).floor();
713    let f = (jd + 0.5) - z;
714    let mut a = z;
715    if z >= 2_299_161.0 {
716        let alpha = ((z - 1_867_216.25) / 36_524.25).floor();
717        a = z + 1.0 + alpha - (alpha / 4.0).floor();
718    }
719    let b = a + 1524.0;
720    let c = ((b - 122.1) / 365.25).floor();
721    let d = (365.25 * c).floor();
722    let e = ((b - d) / 30.6001).floor();
723    let day = b - d - (30.6001 * e).floor() + f;
724    let month = if e < 14.0 { e - 1.0 } else { e - 13.0 };
725    let year = if month > 2.0 { c - 4716.0 } else { c - 4715.0 };
726
727    let year_i = year as i32;
728    let month_i = month as i32;
729    let day_i = day.floor() as i32;
730    let leap = if (year_i % 4 == 0) && ((year_i % 100 != 0) || (year_i % 400 == 0)) {
731        1
732    } else {
733        0
734    };
735    const DAYS: [i32; 12] = [0, 31, 59, 90, 120, 151, 181, 212, 243, 273, 304, 334];
736    let mut day_in_year = DAYS[(month_i - 1) as usize] + day_i;
737    if month_i > 2 {
738        day_in_year += leap;
739    }
740    f64::from(year_i) + (f64::from(day_in_year) / (365.0 + f64::from(leap)))
741}
742
743fn solve_igrf_nmax(coeff_len: usize) -> Result<usize, MeasuresDataError> {
744    for nmax in 1..32 {
745        if nmax * (nmax + 2) == coeff_len {
746            return Ok(nmax);
747        }
748    }
749    Err(MeasuresDataError::MissingData(format!(
750        "unable to derive IGRF nmax from coefficient length {coeff_len}"
751    )))
752}

754fn interpolate_coefficients(
755    date: f64,
756    year1: f64,
757    coeffs1: &[f64],
758    year2: f64,
759    coeffs2: &[f64],
760) -> Vec<f64> {
761    let factor = (date - year1) / (year2 - year1);
762    coeffs1
763        .iter()
764        .zip(coeffs2.iter())
765        .map(|(lhs, rhs)| lhs + factor * (rhs - lhs))
766        .collect()
767}
768
769fn extrapolate_coefficients(date: f64, base_year: f64, main: &[f64], sv: &[f64]) -> Vec<f64> {
770    let factor = date - base_year;
771    main.iter()
772        .zip(sv.iter())
773        .map(|(main_coeff, sv_coeff)| main_coeff + factor * sv_coeff)
774        .collect()
775}

777fn parse_measures_version_mjd_age(version: &str, current_mjd: f64) -> Option<f64> {
778    let marker = "_Measures_";
779    let idx = version.find(marker)?;
780    let date = version.get(idx + marker.len()..idx + marker.len() + 8)?;
781    let year = date.get(0..4)?.parse::<i32>().ok()?;
782    let month = date.get(4..6)?.parse::<u32>().ok()?;
783    let day = date.get(6..8)?.parse::<u32>().ok()?;
784    let mjd = mjd_from_calendar(year, month, day)?;
785    Some(current_mjd - mjd)
786}
787
788fn mjd_from_calendar(year: i32, month: u32, day: u32) -> Option<f64> {
789    if !(1..=12).contains(&month) || !(1..=31).contains(&day) {
790        return None;
791    }
792    let a = (14 - month as i32) / 12;
793    let y = year + 4800 - a;
794    let m = month as i32 + 12 * a - 3;
795    let jdn = day as i32 + ((153 * m + 2) / 5) + 365 * y + y / 4 - y / 100 + y / 400 - 32045;
796    Some(f64::from(jdn) - 2_400_001.0)
797}
