21/// Column name used by `PagedArray` — matches C++ casacore convention.
22const COLUMN_NAME: &str = "PagedArray";
23
24/// A disk-backed N-dimensional lattice using tiled table storage.
25///
26/// Corresponds to the C++ `PagedArray<T>` class. Data is stored in a
27/// one-row casacore table with a single array column named `"PagedArray"`,
28/// using `TiledCellStMan` for tiled I/O.
29///
30/// # On-disk format
31///
32/// The table has:
33/// - One column named `"PagedArray"` with fixed array shape
34/// - One row containing the entire lattice as a single array cell
35/// - `TiledCellStMan` storage manager with the specified tile shape
36///
37/// This format is fully compatible with C++ casacore's `PagedArray`.
38///
39/// # Examples
40///
41/// ```rust,no_run
42/// use casa_lattices::{PagedArray, TiledShape, Lattice, LatticeMut};
43///
44/// // Create a new 64x64 PagedArray on disk:
45/// let ts = TiledShape::new(vec![64, 64]);
46/// let mut pa = PagedArray::<f64>::create(ts, "/tmp/test_paged.table").unwrap();
47/// pa.set(1.0).unwrap();
48///
49/// // Reopen:
50/// let pa2 = PagedArray::<f64>::open("/tmp/test_paged.table").unwrap();
51/// assert_eq!(pa2.shape(), &[64, 64]);
52/// ```
53pub struct PagedArray<T: LatticeElement> {
54    /// Interior mutability for the table handle, enabling auto-reopen
55    /// from `&self` read methods after `temp_close()`.
56    table: RefCell<Option<Table>>,
57    /// Direct tile-level access to the underlying `TiledCellStMan` payload.
58    ///
59    /// This mirrors C++ `PagedArray`'s direct slice access through the tiled
60    /// array accessor instead of materializing the entire lattice for each
61    /// unit-stride slice operation.
62    tiled_io: RefCell<Option<TiledFileIO>>,
63    shape: Vec<usize>,
64    tile_shape: Vec<usize>,
65    max_cache_bytes: Cell<usize>,
66    path: Option<PathBuf>,
67    _phantom: std::marker::PhantomData<T>,
68}
69
70struct PagedArrayTraversalCacheScope<'a, T: LatticeElement> {
71    array: &'a PagedArray<T>,
72    previous_cache_bytes: usize,
73}
74
75impl<T: LatticeElement> TraversalCacheScope for PagedArrayTraversalCacheScope<'_, T> {}

77impl<T: LatticeElement> Drop for PagedArrayTraversalCacheScope<'_, T> {
78    fn drop(&mut self) {
79        let _ = self.array.set_cache_bytes_shared(self.previous_cache_bytes);
80    }
81}

92impl<T: LatticeElement> PagedArray<T> {
93    fn supports_tiled_io() -> bool {
94        matches!(
95            T::PRIMITIVE_TYPE,
96            PrimitiveType::Bool
97                | PrimitiveType::Float32
98                | PrimitiveType::Float64
99                | PrimitiveType::Complex32
100                | PrimitiveType::Complex64
101        )
102    }
103
104    fn open_tiled_io(path: &Path, max_cache_bytes: usize) -> Result<TiledFileIO, LatticeError> {
105        if max_cache_bytes == 0 {
106            TiledFileIO::open(path, 0).map_err(tiled_io_err)
107        } else {
108            TiledFileIO::open_with_cache_limit(path, 0, max_cache_bytes).map_err(tiled_io_err)
109        }
110    }
111
112    fn element_size_bytes() -> usize {
113        T::PRIMITIVE_TYPE.fixed_width_bytes().unwrap_or(0)
114    }
115
116    fn tile_pixels(&self) -> usize {
117        self.tile_shape.iter().product()
118    }
119
120    fn refresh_tiled_io(&mut self) -> Result<(), LatticeError> {
121        if self.path.is_none() || !Self::supports_tiled_io() {
122            return Ok(());
123        }
124
125        if let Some(ref mut tio) = *self.tiled_io.get_mut() {
126            tio.flush().map_err(tiled_io_err)?;
127        } else if self.table.get_mut().is_some() {
128            self.flush()?;
129        }
130        let path = self.path.as_ref().expect("persistent path");
131        let tiled_io = Self::open_tiled_io(path, self.max_cache_bytes.get())?;
132        *self.tiled_io.get_mut() = Some(tiled_io);
133        Ok(())
134    }
135
136    fn refresh_tiled_io_shared(&self) -> Result<(), LatticeError> {
137        if self.path.is_none() || !Self::supports_tiled_io() {
138            return Ok(());
139        }
140
141        if let Some(ref mut tio) = *self.tiled_io.borrow_mut() {
142            tio.flush().map_err(tiled_io_err)?;
143        } else if self.table.borrow().is_some() {
144            self.flush()?;
145        }
146        let path = self.path.as_ref().expect("persistent path");
147        let tiled_io = Self::open_tiled_io(path, self.max_cache_bytes.get())?;
148        *self.tiled_io.borrow_mut() = Some(tiled_io);
149        Ok(())
150    }
151
152    fn set_cache_bytes_shared(&self, max_cache_bytes: usize) -> Result<(), LatticeError> {
153        self.max_cache_bytes.set(max_cache_bytes);
154        self.refresh_tiled_io_shared()
155    }

157    fn cast_tiled_array<U: 'static>(data: ArrayD<U>) -> Result<ArrayD<T>, LatticeError> {
158        let boxed: Box<dyn Any> = Box::new(data);
159        boxed
160            .downcast::<ArrayD<T>>()
161            .map(|boxed| *boxed)
162            .map_err(|_| LatticeError::Table("internal PagedArray tiled type mismatch".to_string()))
163    }
164
165    fn tiled_get_slice(
166        tio: &mut TiledFileIO,
167        start: &[usize],
168        shape: &[usize],
169    ) -> Result<ArrayD<T>, LatticeError> {
170        match T::PRIMITIVE_TYPE {
171            PrimitiveType::Bool => {
172                Self::cast_tiled_array(tio.get_slice::<bool>(start, shape).map_err(tiled_io_err)?)
173            }
174            PrimitiveType::Float32 => {
175                Self::cast_tiled_array(tio.get_slice::<f32>(start, shape).map_err(tiled_io_err)?)
176            }
177            PrimitiveType::Float64 => {
178                Self::cast_tiled_array(tio.get_slice::<f64>(start, shape).map_err(tiled_io_err)?)
179            }
180            PrimitiveType::Complex32 => Self::cast_tiled_array(
181                tio.get_slice::<Complex32>(start, shape)
182                    .map_err(tiled_io_err)?,
183            ),
184            PrimitiveType::Complex64 => Self::cast_tiled_array(
185                tio.get_slice::<Complex64>(start, shape)
186                    .map_err(tiled_io_err)?,
187            ),
188            _ => Err(LatticeError::Table(
189                "tiled I/O is not supported for this PagedArray element type".to_string(),
190            )),
191        }
192    }
193
194    fn tiled_get_all(tio: &mut TiledFileIO) -> Result<ArrayD<T>, LatticeError> {
195        match T::PRIMITIVE_TYPE {
196            PrimitiveType::Bool => {
197                Self::cast_tiled_array(tio.get_all::<bool>().map_err(tiled_io_err)?)
198            }
199            PrimitiveType::Float32 => {
200                Self::cast_tiled_array(tio.get_all::<f32>().map_err(tiled_io_err)?)
201            }
202            PrimitiveType::Float64 => {
203                Self::cast_tiled_array(tio.get_all::<f64>().map_err(tiled_io_err)?)
204            }
205            PrimitiveType::Complex32 => {
206                Self::cast_tiled_array(tio.get_all::<Complex32>().map_err(tiled_io_err)?)
207            }
208            PrimitiveType::Complex64 => {
209                Self::cast_tiled_array(tio.get_all::<Complex64>().map_err(tiled_io_err)?)
210            }
211            _ => Err(LatticeError::Table(
212                "tiled I/O is not supported for this PagedArray element type".to_string(),
213            )),
214        }
215    }

217    fn tiled_put_slice(
218        tio: &mut TiledFileIO,
219        data: &ArrayD<T>,
220        start: &[usize],
221    ) -> Result<(), LatticeError> {
222        match T::PRIMITIVE_TYPE {
223            PrimitiveType::Bool => Self::tiled_put_slice_typed::<bool>(tio, data, start),
224            PrimitiveType::Float32 => Self::tiled_put_slice_typed::<f32>(tio, data, start),
225            PrimitiveType::Float64 => Self::tiled_put_slice_typed::<f64>(tio, data, start),
226            PrimitiveType::Complex32 => Self::tiled_put_slice_typed::<Complex32>(tio, data, start),
227            PrimitiveType::Complex64 => Self::tiled_put_slice_typed::<Complex64>(tio, data, start),
228            _ => Err(LatticeError::Table(
229                "tiled I/O is not supported for this PagedArray element type".to_string(),
230            )),
231        }
232    }
233
234    fn tiled_put_slice_typed<U: 'static + Clone + TilePixel>(
235        tio: &mut TiledFileIO,
236        data: &ArrayD<T>,
237        start: &[usize],
238    ) -> Result<(), LatticeError> {
239        let typed = (data as &dyn Any)
240            .downcast_ref::<ArrayD<U>>()
241            .ok_or_else(|| {
242                LatticeError::Table("internal PagedArray tiled type mismatch".to_string())
243            })?;
244        let fortran_view = typed.t();
245        if let Some(slice) = fortran_view.as_slice() {
246            tio.put_slice_fortran::<U>(slice, start, typed.shape())
247                .map_err(tiled_io_err)?;
248            return Ok(());
249        }
250        let contiguous = typed.as_standard_layout();
251        let slice = contiguous.as_slice().expect("contiguous C-order data");
252        tio.put_slice_c_order::<U>(slice, start, typed.shape())
253            .map_err(tiled_io_err)
254    }

256    /// Creates a new `PagedArray` at the given path with the specified shape.
257    ///
258    /// The table is created immediately on disk. The array is initialized
259    /// with zero/default values.
260    pub fn create(tiled_shape: TiledShape, path: impl AsRef<Path>) -> Result<Self, LatticeError> {
261        let path = path.as_ref();
262        let shape = tiled_shape.shape().to_vec();
263        let tile_shape = tiled_shape.tile_shape();
264        let ndim = shape.len();
265
266        let schema = TableSchema::new(vec![ColumnSchema::array_fixed(
267            COLUMN_NAME,
268            T::PRIMITIVE_TYPE,
269            shape.clone(),
270        )])
271        .map_err(|e| LatticeError::Table(e.to_string()))?;
272
273        let mut table = Table::with_schema(schema);
274
275        // Add one row with a default-filled array.
276        let data = ArrayD::from_elem(IxDyn(&shape), T::default_value());
277        let array_value = value_bridge::to_array_value(&data);
278        let row = RecordValue::new(vec![RecordField::new(
279            COLUMN_NAME,
280            Value::Array(array_value),
281        )]);
282        table.add_row(row).map_err(table_err)?;
283
284        table
285            .save(
286                TableOptions::new(path)
287                    .with_data_manager(DataManagerKind::TiledCellStMan)
288                    .with_tile_shape(if ndim > 0 {
289                        tile_shape.clone()
290                    } else {
291                        Vec::new()
292                    }),
293            )
294            .map_err(table_err)?;
295
296        let tiled_io = if Self::supports_tiled_io() {
297            Some(Self::open_tiled_io(path, 0)?)
298        } else {
299            None
300        };
301
302        Ok(Self {
303            table: RefCell::new(Some(table)),
304            tiled_io: RefCell::new(tiled_io),
305            shape,
306            tile_shape,
307            max_cache_bytes: Cell::new(0),
308            path: Some(path.to_path_buf()),
309            _phantom: std::marker::PhantomData,
310        })
311    }

313    /// Creates a scratch (temporary) `PagedArray` that is not persisted.
314    ///
315    /// The data lives in a memory-backed table. Equivalent to C++
316    /// `PagedArray(shape)` with no path argument.

318        let shape = tiled_shape.shape().to_vec();
319        let tile_shape = tiled_shape.tile_shape();

321        let schema = TableSchema::new(vec![ColumnSchema::array_fixed(
322            COLUMN_NAME,

324            shape.clone(),
325        )])
326        .map_err(|e| LatticeError::Table(e.to_string()))?;
327
328        let mut table = Table::with_schema_memory(schema);
329
330        let data = ArrayD::from_elem(IxDyn(&shape), T::default_value());
331        let array_value = value_bridge::to_array_value(&data);
332        let row = RecordValue::new(vec![RecordField::new(
333            COLUMN_NAME,
334            Value::Array(array_value),
335        )]);
336        table.add_row(row).map_err(table_err)?;
337
338        Ok(Self {
339            table: RefCell::new(Some(table)),
340            tiled_io: RefCell::new(None),
341            shape,
342            tile_shape,
343            max_cache_bytes: Cell::new(0),
344            path: None,
345            _phantom: std::marker::PhantomData,
346        })
347    }

349    /// Opens an existing `PagedArray` from disk.
350    ///
351    /// The table must have been created by a `PagedArray` (Rust or C++),
352    /// with a single column named `"PagedArray"`.
353    pub fn open(path: impl AsRef<Path>) -> Result<Self, LatticeError> {
354        Self::open_with_cache(path, 0)
355    }
356
357    /// Opens an existing `PagedArray` from disk with an explicit cache size.
358    ///
359    /// A `max_cache_bytes` value of `0` keeps the existing "no explicit limit"
360    /// behavior, while non-zero values allow slice-heavy callers to force the
361    /// bounded LRU path instead of fully warming the backing tiled payload.
362    pub fn open_with_cache(
363        path: impl AsRef<Path>,
364        max_cache_bytes: usize,
365    ) -> Result<Self, LatticeError> {
366        let path = path.as_ref();
367        let table = Table::open(TableOptions::new(path)).map_err(table_err)?;
368
369        // Read the shape from the cell.
370        let cell = table
371            .cell_accessor(0, COLUMN_NAME)
372            .map_err(table_err)?
373            .value()
374            .map_err(table_err)?
375            .ok_or_else(|| {
376                LatticeError::Table("PagedArray column not found or no rows".to_string())
377            })?;
378
379        let shape = match cell {
380            Value::Array(av) => av.shape().to_vec(),
381            _ => {
382                return Err(LatticeError::Table(
383                    "PagedArray column is not an array".to_string(),
384                ));
385            }
386        };
387
388        let tiled_io = if Self::supports_tiled_io() {
389            Some(Self::open_tiled_io(path, max_cache_bytes)?)
390        } else {
391            None
392        };
393        let (shape, tile_shape) = if let Some(ref tio) = tiled_io {
394            (tio.cube_shape().to_vec(), tio.tile_shape().to_vec())
395        } else {
396            (shape.clone(), TiledShape::default_tile_shape(&shape))
397        };
398
399        Ok(Self {
400            table: RefCell::new(Some(table)),
401            tiled_io: RefCell::new(tiled_io),
402            shape,
403            tile_shape,
404            max_cache_bytes: Cell::new(max_cache_bytes),
405            path: Some(path.to_path_buf()),
406            _phantom: std::marker::PhantomData,
407        })
408    }

420    /// Returns the configured maximum tile-cache size in pixels.
421    ///
422    /// A value of `0` means "no explicit maximum", matching C++ casacore's
423    /// `PagedArray::maximumCacheSize()` convention.
424    pub fn maximum_cache_size_pixels(&self) -> usize {
425        let elem_size = Self::element_size_bytes();
426        let max_cache_bytes = self.max_cache_bytes.get();
427        if elem_size == 0 || max_cache_bytes == 0 {
428            0
429        } else {
430            max_cache_bytes / elem_size
431        }
432    }
433
434    /// Sets the maximum tile-cache size in pixels.
435    ///
436    /// A value of `0` removes the explicit limit. Persistent arrays reopen
437    /// their tiled I/O handle so subsequent reads use the new cache policy.
438    ///
439    /// Mirrors C++ `PagedArray::setMaximumCacheSize`.
440    pub fn set_maximum_cache_size_pixels(
441        &mut self,
442        how_many_pixels: usize,
443    ) -> Result<(), LatticeError> {
444        let elem_size = Self::element_size_bytes();
445        let max_cache_bytes = if elem_size == 0 {
446            0
447        } else {
448            how_many_pixels.saturating_mul(elem_size)
449        };
450        self.max_cache_bytes.set(max_cache_bytes);
451        self.refresh_tiled_io()
452    }
453
454    /// Sets the cache size to hold approximately `how_many_tiles` tiles.
455    ///
456    /// A value of `0` removes the explicit maximum. Persistent arrays reopen
457    /// their tiled I/O handle so subsequent reads use the new cache policy.
458    ///
459    /// Mirrors C++ `PagedArray::setCacheSizeInTiles`.
460    pub fn set_cache_size_in_tiles(&mut self, how_many_tiles: usize) -> Result<(), LatticeError> {
461        if how_many_tiles == 0 {
462            return self.set_maximum_cache_size_pixels(0);
463        }
464        let pixels = self.tile_pixels().saturating_mul(how_many_tiles);
465        self.set_maximum_cache_size_pixels(pixels)
466    }

468    /// Returns `true` if the array has been temp-closed.
469    pub fn is_temp_closed(&self) -> bool {
470        self.table.borrow().is_none()
471    }
472
473    /// Releases the in-memory table, flushing to disk first if persistent.
474    ///
475    /// For scratch (no-path) arrays this is a no-op. After `temp_close()`,
476    /// subsequent reads/writes will auto-reopen from disk.
477    pub fn temp_close(&mut self) -> Result<(), LatticeError> {
478        if self.path.is_none() {
479            return Ok(());
480        }
481        self.flush()?;
482        *self.table.get_mut() = None;
483        *self.tiled_io.get_mut() = None;
484        Ok(())
485    }
486
487    /// Explicitly reopens a temp-closed array from disk.
488    ///
489    /// No-op if the table is already open or if this is a scratch array.
490    pub fn reopen(&mut self) -> Result<(), LatticeError> {
491        if (self.table.get_mut().is_some() && self.tiled_io.get_mut().is_some())
492            || self.path.is_none()
493        {
494            return Ok(());
495        }
496        self.auto_reopen()
497    }
498
499    /// Ensures the table is open, reopening from disk if needed.
500    ///
501    /// Uses interior mutability so that `&self` read methods can
502    /// transparently reopen after `temp_close()`.
503    fn auto_reopen(&self) -> Result<(), LatticeError> {
504        let needs_tiled_io = self.path.is_some() && Self::supports_tiled_io();
505        if self.table.borrow().is_none() || (needs_tiled_io && self.tiled_io.borrow().is_none()) {
506            let Some(path) = &self.path else {
507                return Err(LatticeError::Table(
508                    "cannot reopen scratch PagedArray after temp_close".to_string(),
509                ));
510            };
511            if self.table.borrow().is_none() {
512                let table = Table::open(TableOptions::new(path)).map_err(table_err)?;
513                *self.table.borrow_mut() = Some(table);
514            }
515            if needs_tiled_io && self.tiled_io.borrow().is_none() {
516                let tiled_io = Self::open_tiled_io(path, self.max_cache_bytes.get())?;
517                *self.tiled_io.borrow_mut() = Some(tiled_io);
518            }
519        }
520        Ok(())
521    }
522
523    fn auto_reopen_tiled_io(&self) -> Result<(), LatticeError> {
524        if self.path.is_none() || !Self::supports_tiled_io() {
525            return Ok(());
526        }
527        if self.tiled_io.borrow().is_none() {
528            let path = self.path.as_ref().expect("persistent path");
529            let tiled_io = Self::open_tiled_io(path, self.max_cache_bytes.get())?;
530            *self.tiled_io.borrow_mut() = Some(tiled_io);
531        }
532        Ok(())
533    }
534
535    fn save_options(&self, path: &Path) -> TableOptions {
536        TableOptions::new(path)
537            .with_data_manager(DataManagerKind::TiledCellStMan)
538            .with_tile_shape(self.tile_shape.clone())
539    }
540
541    /// Flushes changes to disk (for persistent arrays).
542    pub fn flush(&self) -> Result<(), LatticeError> {
543        if let Some(path) = &self.path {
544            self.auto_reopen()?;
545            if let Some(ref mut tio) = *self.tiled_io.borrow_mut() {
546                tio.flush().map_err(tiled_io_err)?;
547            } else if self.table.borrow().is_some() {
548                self.table
549                    .borrow()
550                    .as_ref()
551                    .unwrap()
552                    .save(self.save_options(path))
553                    .map_err(table_err)?;
554            }
555        }
556        Ok(())
557    }
558
559    /// Reads the full array, auto-reopening if temp-closed.
560    fn read_full_array(&self) -> Result<ArrayD<T>, LatticeError> {
561        self.auto_reopen()?;
562        let table_ref = self.table.borrow();
563        let table = table_ref.as_ref().unwrap();
564        let cell = table
565            .cell_accessor(0, COLUMN_NAME)
566            .map_err(table_err)?
567            .value()
568            .map_err(table_err)?
569            .ok_or_else(|| LatticeError::Table("PagedArray cell not found".to_string()))?;
570        match cell {
571            Value::Array(av) => value_bridge::from_array_value(av.clone()),
572            _ => Err(LatticeError::Table("unexpected scalar cell".to_string())),
573        }
574    }
575
576    /// Writes the full array to the table cell.
577    fn write_full_array(&mut self, data: &ArrayD<T>) -> Result<(), LatticeError> {
578        self.auto_reopen()?;
579        let array_value = value_bridge::to_array_value(data);
580        self.table
581            .get_mut()
582            .as_mut()
583            .unwrap()
584            .cell_accessor_mut(0, COLUMN_NAME)
585            .map_err(table_err)?
586            .set(Value::Array(array_value))
587            .map_err(table_err)
588    }
589}

598    }

600    fn is_paged(&self) -> bool {
601        true
602    }
603
604    fn get_at(&self, position: &[usize]) -> Result<T, LatticeError> {
605        if position.len() != self.shape.len() {
606            return Err(LatticeError::NdimMismatch {
607                expected: self.shape.len(),
608                got: position.len(),
609            });
610        }
611        for (&p, &s) in position.iter().zip(self.shape.iter()) {
612            if p >= s {
613                return Err(LatticeError::IndexOutOfBounds {
614                    index: position.to_vec(),
615                    shape: self.shape.clone(),
616                });
617            }
618        }
619
620        if self.path.is_some() {
621            self.auto_reopen_tiled_io()?;
622            if let Some(ref mut tio) = *self.tiled_io.borrow_mut() {
623                let ones = vec![1; self.shape.len()];
624                let arr = Self::tiled_get_slice(tio, position, &ones)?;
625                return arr
626                    .into_iter()
627                    .next()
628                    .ok_or_else(|| LatticeError::Table("empty tiled slice".to_string()));
629            }
630        }
631
632        let arr = self.read_full_array()?;
633        Ok(arr[IxDyn(position)].clone())
634    }
635
636    fn get_slice(
637        &self,
638        start: &[usize],
639        shape: &[usize],
640        stride: &[usize],
641    ) -> Result<ArrayD<T>, LatticeError> {
642        let ndim = self.shape.len();
643        if start.len() != ndim || shape.len() != ndim || stride.len() != ndim {
644            return Err(LatticeError::NdimMismatch {
645                expected: ndim,
646                got: start.len(),
647            });
648        }
649
650        let is_unit_stride = stride.iter().all(|&s| s == 1);
651        if self.path.is_some() && is_unit_stride {
652            self.auto_reopen_tiled_io()?;
653            if let Some(ref mut tio) = *self.tiled_io.borrow_mut() {
654                return Self::tiled_get_slice(tio, start, shape);
655            }
656        }
657
658        let arr = self.read_full_array()?;
659
660        let slice_info: Vec<ndarray::SliceInfoElem> = start
661            .iter()
662            .zip(shape.iter())
663            .zip(stride.iter())
664            .map(|((&s, &n), &st)| {
665                let end = if n == 0 { s } else { s + n * st };
666                ndarray::SliceInfoElem::Slice {
667                    start: s as isize,
668                    end: Some(end as isize),
669                    step: st as isize,
670                }
671            })
672            .collect();
673
674        let view = arr.slice(slice_info.as_slice());
675        Ok(view.to_owned())
676    }
677
678    fn get(&self) -> Result<ArrayD<T>, LatticeError> {
679        if self.path.is_some() {
680            self.auto_reopen_tiled_io()?;
681            if let Some(ref mut tio) = *self.tiled_io.borrow_mut() {
682                return Self::tiled_get_all(tio);
683            }
684        }
685        self.read_full_array()
686    }
687
688    fn advised_max_pixels(&self) -> usize {
689        self.tile_shape.iter().product()
690    }
691
692    fn nice_cursor_shape(&self) -> Vec<usize> {
693        self.tile_shape.clone()
694    }
695
696    fn enter_traversal_cache_scope<'a>(
697        &'a self,
698        hint: &TraversalCacheHint,
699    ) -> Result<Option<Box<dyn TraversalCacheScope + 'a>>, LatticeError> {
700        if self.path.is_none() || !Self::supports_tiled_io() {
701            return Ok(None);
702        }
703        let previous_cache_bytes = self.max_cache_bytes.get();
704        let elem_size = Self::element_size_bytes().max(1);
705        let recommended_tiles =
706            recommended_tile_cache_size(&self.shape, &self.tile_shape, hint, None).max(1);
707        let recommended_bytes = recommended_tiles
708            .saturating_mul(self.tile_pixels())
709            .saturating_mul(elem_size);
710        if previous_cache_bytes == recommended_bytes {
711            return Ok(None);
712        }
713        self.set_cache_bytes_shared(recommended_bytes)?;
714        Ok(Some(Box::new(PagedArrayTraversalCacheScope {
715            array: self,
716            previous_cache_bytes,
717        })))
718    }
719}

721impl<T: LatticeElement> LatticeMut<T> for PagedArray<T> {
722    fn with_traversal_cache_hint_mut<R>(
723        &mut self,
724        hint: &TraversalCacheHint,
725        f: impl FnOnce(&mut Self) -> Result<R, LatticeError>,
726    ) -> Result<R, LatticeError> {
727        if self.path.is_none() || !Self::supports_tiled_io() {
728            return f(self);
729        }
730        let previous_cache_bytes = self.max_cache_bytes.get();
731        let elem_size = Self::element_size_bytes().max(1);
732        let recommended_tiles =
733            recommended_tile_cache_size(&self.shape, &self.tile_shape, hint, None).max(1);
734        let recommended_bytes = recommended_tiles
735            .saturating_mul(self.tile_pixels())
736            .saturating_mul(elem_size);
737        if previous_cache_bytes == recommended_bytes {
738            return f(self);
739        }
740
741        self.set_cache_bytes_shared(recommended_bytes)?;
742        let result = f(self);
743        let restore = self.set_cache_bytes_shared(previous_cache_bytes);
744        match (result, restore) {
745            (Err(err), _) => Err(err),
746            (Ok(_), Err(err)) => Err(err),
747            (Ok(value), Ok(())) => Ok(value),
748        }
749    }
750
751    fn put_at(&mut self, value: T, position: &[usize]) -> Result<(), LatticeError> {
752        let data = ArrayD::from_elem(IxDyn(&vec![1; position.len()]), value);
753        self.put_slice(&data, position)
754    }
755
756    fn put_slice(&mut self, data: &ArrayD<T>, start: &[usize]) -> Result<(), LatticeError> {
757        let ndim = self.shape.len();
758        if start.len() != ndim {
759            return Err(LatticeError::NdimMismatch {
760                expected: ndim,
761                got: start.len(),
762            });
763        }
764        let end: Vec<usize> = start
765            .iter()
766            .zip(data.shape().iter())
767            .map(|(&s, &n)| s + n)
768            .collect();
769        for (&limit, &dim) in end.iter().zip(self.shape.iter()) {
770            if limit > dim {
771                return Err(LatticeError::ShapeMismatch {
772                    expected: self.shape.clone(),
773                    got: end,
774                });
775            }
776        }
777
778        if self.path.is_some() {
779            self.auto_reopen()?;
780            if let Some(ref mut tio) = *self.tiled_io.borrow_mut() {
781                Self::tiled_put_slice(tio, data, start)?;
782                return Ok(());
783            }
784        }
785
786        let mut current = self.read_full_array()?;
787
788        let slice_info: Vec<ndarray::SliceInfoElem> = start
789            .iter()
790            .zip(data.shape().iter())
791            .map(|(&s, &n)| ndarray::SliceInfoElem::Slice {
792                start: s as isize,
793                end: Some((s + n) as isize),
794                step: 1,
795            })
796            .collect();
797
798        let mut view = current.slice_mut(slice_info.as_slice());
799        view.assign(data);
800
801        self.write_full_array(&current)
802    }
803
804    fn set(&mut self, value: T) -> Result<(), LatticeError> {
805        let data = ArrayD::from_elem(IxDyn(&self.shape), value);
806        let start = vec![0; self.shape.len()];
807        self.put_slice(&data, &start)
808    }
809}
