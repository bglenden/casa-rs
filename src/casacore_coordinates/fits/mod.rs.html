2//! FITS/WCS coordinate interoperability.
3//!
4//! This module provides conversion between FITS World Coordinate System (WCS)
5//! headers and the coordinate types defined in this crate. It can parse
6//! standard FITS WCS keywords (CTYPEn, CRVALn, CRPIXn, CDELTn, PCi_j, etc.)
7//! into a [`CoordinateSystem`](crate::CoordinateSystem), and emit those
8//! keywords from an existing coordinate system.
9//!
10//! The implementation follows:
11//! - Greisen & Calabretta (2002), "Representations of world coordinates in FITS"
12//! - Calabretta & Greisen (2002), "Representations of celestial coordinates in FITS"
13//! - Greisen et al. (2006), "Representations of spectral coordinates in FITS"
14//!
15//! # Examples
16//!
17//! ```rust
18//! use casacore_coordinates::fits::{FitsHeader, to_fits_header, from_fits_header};
19//! use casacore_coordinates::{
20//!     CoordinateSystem, DirectionCoordinate, SpectralCoordinate,
21//!     StokesCoordinate, StokesType, Projection, ProjectionType,
22//! };
23//! use casacore_types::measures::direction::DirectionRef;
24//! use casacore_types::measures::frequency::FrequencyRef;
25//!
26//! // Build a coordinate system
27//! let mut cs = CoordinateSystem::new();
28//! let proj = Projection::new(ProjectionType::SIN);
29//! let dir = DirectionCoordinate::new(
30//!     DirectionRef::J2000, proj,
31//!     [0.0, std::f64::consts::FRAC_PI_4],
32//!     [-1e-4, 1e-4], [512.0, 512.0],
33//! );
34//! cs.add_coordinate(Box::new(dir));
35//! let spec = SpectralCoordinate::new(FrequencyRef::LSRK, 1.42e9, 1e6, 0.0, 1.42040575e9);
36//! cs.add_coordinate(Box::new(spec));
37//! let stokes = StokesCoordinate::new(vec![StokesType::I, StokesType::V]);
38//! cs.add_coordinate(Box::new(stokes));
39//!
40//! // Convert to FITS header
41//! let shape = [1024, 1024, 256, 2];
42//! let header = to_fits_header(&cs, &shape);
43//!
44//! // Round-trip back
45//! let cs2 = from_fits_header(&header, &shape).unwrap();
46//! assert_eq!(cs2.n_pixel_axes(), cs.n_pixel_axes());
47//! ```
48
49pub mod coordinate_util;
50pub mod header;

53pub use header::{FitsHeader, FitsKeyword, FitsValue};
