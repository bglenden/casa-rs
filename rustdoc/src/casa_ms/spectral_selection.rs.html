47/// Typed cube-axis value corresponding to CASA `start` / `width`.
48#[derive(Debug, Clone, PartialEq)]
49pub enum CubeAxisValue {
50    /// Channel-number selection in the source SPW.
51    Channel(i32),
52    /// Frequency-like quantity in Hz.
53    FrequencyHz {
54        /// Frequency value in Hz.
55        hz: f64,
56        /// Explicit frame carried by a CASA measure.
57        frame: Option<FrequencyRef>,
58    },
59    /// Velocity-like quantity in m/s.
60    VelocityMs {
61        /// Velocity value in m/s.
62        ///
63        /// CASA measure-valued `start` / `width` inputs such as
64        /// `me.radialvelocity(...)` and many `me.todoppler(...)` results surface
65        /// here, because the casatools layer hands them around as velocity-like
66        /// quantities in `m/s` rather than as dimensionless algebraic doppler
67        /// parameters.
68        ms: f64,
69        /// Explicit frame carried by a CASA measure.
70        frame: Option<FrequencyRef>,
71    },
72    /// Doppler value in the given convention.
73    Doppler {
74        /// Dimensionless algebraic doppler value in the given convention.
75        ///
76        /// This variant is for callers that already have the convention's
77        /// normalized scalar. It is not the usual representation of CASA
78        /// `me.todoppler(...)` records in cube-axis task inputs; those are more
79        /// commonly handled as [`VelocityMs`](Self::VelocityMs).
80        value: f64,
81        /// Doppler convention.
82        convention: DopplerRef,
83    },
84}
85
86impl CubeAxisValue {
87    /// Parse a simple CASA-style cube-axis value.
88    ///
89    /// Supported forms currently include plain integers (channel numbers),
90    /// frequency quantities such as `1.1GHz`, and velocity quantities such as
91    /// `11991.7km/s`.
92    pub fn parse(text: &str, veltype: DopplerRef) -> MsResult<Self> {
93        let text = text.trim();
94        if text.is_empty() {
95            return Err(MsError::VersionError(
96                "cube axis value was empty".to_string(),
97            ));
98        }
99        if let Ok(channel) = text.parse::<i32>() {
100            return Ok(Self::Channel(channel));
101        }
102
103        let quantity = Quantity::from_str(text).map_err(|error| {
104            MsError::VersionError(format!("parse cube axis value {text:?}: {error}"))
105        })?;
106        let hz = Unit::new("Hz").expect("built-in Hz unit must parse");
107        let velocity = Unit::new("m/s").expect("built-in velocity unit must parse");
108        let dimensionless = Unit::dimensionless();
109        if quantity.unit().conformant(&hz) {
110            return Ok(Self::FrequencyHz {
111                hz: quantity.get_value_in(&hz).map_err(|error| {
112                    MsError::VersionError(format!(
113                        "convert cube axis frequency {text:?} to Hz: {error}"
114                    ))
115                })?,
116                frame: None,
117            });
118        }
119        if quantity.unit().conformant(&velocity) {
120            return Ok(Self::VelocityMs {
121                ms: quantity.get_value_in(&velocity).map_err(|error| {
122                    MsError::VersionError(format!(
123                        "convert cube axis velocity {text:?} to m/s: {error}"
124                    ))
125                })?,
126                frame: None,
127            });
128        }
129        if quantity.unit().conformant(&dimensionless) {
130            return Ok(Self::Doppler {
131                value: quantity.value(),
132                convention: veltype,
133            });
134        }
135
136        Err(MsError::VersionError(format!(
137            "cube axis value {text:?} is neither a channel id, frequency, velocity, nor dimensionless doppler value"
138        )))
139    }
140}

142/// Parse a CASA-style rest frequency quantity into Hz.
143///
144/// This accepts explicit frequency units such as `1.25GHz`. Bare numeric values
145/// follow CASA task semantics and are interpreted as MHz.
146pub fn parse_rest_frequency_hz(value: &str) -> MsResult<f64> {
147    let quantity = value.trim().parse::<Quantity>().map_err(|error| {
148        MsError::VersionError(format!("invalid rest frequency {value:?}: {error}"))
149    })?;
150    let hz = if quantity.unit().name().is_empty() {
151        quantity.value() * 1.0e6
152    } else {
153        quantity
154            .get_value_in(&Unit::new("Hz").expect("built-in Hz unit must parse"))
155            .map_err(|error| {
156                MsError::VersionError(format!("convert rest frequency {value:?} to Hz: {error}"))
157            })?
158    };
159    if hz <= 0.0 || !hz.is_finite() {
160        return Err(MsError::VersionError(format!(
161            "rest frequency must resolve to a positive finite Hz value, got {hz}"
162        )));
163    }
164    Ok(hz)
165}

292fn channel_edges_from_centers(all_source_frequencies_hz: &[f64]) -> MsResult<Vec<f64>> {
293    if all_source_frequencies_hz.is_empty() {
294        return Err(MsError::VersionError(
295            "cannot build cube channel axis from an empty SPW".to_string(),
296        ));
297    }
298    if all_source_frequencies_hz.len() == 1 {
299        let center = all_source_frequencies_hz[0];
300        return Ok(vec![center - 0.5, center + 0.5]);
301    }
302
303    let mut edges = Vec::with_capacity(all_source_frequencies_hz.len() + 1);
304    let first_step = all_source_frequencies_hz[1] - all_source_frequencies_hz[0];
305    edges.push(all_source_frequencies_hz[0] - first_step / 2.0);
306    for pair in all_source_frequencies_hz.windows(2) {
307        edges.push(0.5 * (pair[0] + pair[1]));
308    }
309    let last = all_source_frequencies_hz.len() - 1;
310    let last_step = all_source_frequencies_hz[last] - all_source_frequencies_hz[last - 1];
311    edges.push(all_source_frequencies_hz[last] + last_step / 2.0);
312    Ok(edges)
313}

315fn channel_widths_from_centers(channel_frequencies_hz: &[f64]) -> MsResult<Vec<f64>> {
316    let edges = channel_edges_from_centers(channel_frequencies_hz)?;
317    Ok(edges
318        .windows(2)
319        .map(|pair| (pair[1] - pair[0]).abs())
320        .collect())
321}

323fn channel_mode_single_output_width_hz(
324    source_frequencies_hz: &[f64],
325    source_channel_widths_hz: &[f64],
326    start_channel: i32,
327    width_channels: i32,
328) -> MsResult<f64> {
329    if width_channels == 0 {
330        return Err(MsError::VersionError(
331            "cube channel width must not be zero".to_string(),
332        ));
333    }
334    let nchannels = i32::try_from(source_frequencies_hz.len()).map_err(|_| {
335        MsError::VersionError("spectral window channel count exceeds i32".to_string())
336    })?;
337    if source_channel_widths_hz.len() != source_frequencies_hz.len() {
338        return Err(MsError::VersionError(format!(
339            "channel-mode cube axis requires matching frequency/width arrays, got {} and {}",
340            source_frequencies_hz.len(),
341            source_channel_widths_hz.len()
342        )));
343    }
344    let (low, high) = if width_channels > 0 {
345        let low = start_channel;
346        let high = low + width_channels;
347        (low, high)
348    } else {
349        let high = start_channel + 1;
350        let low = high + width_channels;
351        (low, high)
352    };
353    if low < 0 || high < 0 || low >= nchannels || high > nchannels || low >= high {
354        return Err(MsError::VersionError(format!(
355            "cube channel axis bin [{low}, {high}) is outside the SPW with {nchannels} channels"
356        )));
357    }
358    let selected = usize::try_from(low).expect("validated low bound")
359        ..usize::try_from(high).expect("validated high bound");
360    let low_edge_hz = selected
361        .clone()
362        .map(|index| source_frequencies_hz[index] - source_channel_widths_hz[index].abs() / 2.0)
363        .reduce(f64::min)
364        .expect("validated non-empty channel range");
365    let high_edge_hz = selected
366        .map(|index| source_frequencies_hz[index] + source_channel_widths_hz[index].abs() / 2.0)
367        .reduce(f64::max)
368        .expect("validated non-empty channel range");
369    Ok((high_edge_hz - low_edge_hz).abs())
370}

372fn channel_mode_output_centers(
373    all_source_frequencies_hz: &[f64],
374    all_source_channel_widths_hz: &[f64],
375    start_channel: i32,
376    width_channels: i32,
377    nchan: usize,
378) -> MsResult<Vec<f64>> {
379    if width_channels == 0 {
380        return Err(MsError::VersionError(
381            "cube channel width must not be zero".to_string(),
382        ));
383    }
384    let nchannels = i32::try_from(all_source_frequencies_hz.len()).map_err(|_| {
385        MsError::VersionError("spectral window channel count exceeds i32".to_string())
386    })?;
387    if all_source_channel_widths_hz.len() != all_source_frequencies_hz.len() {
388        return Err(MsError::VersionError(format!(
389            "channel-mode cube axis requires matching frequency/width arrays, got {} and {}",
390            all_source_frequencies_hz.len(),
391            all_source_channel_widths_hz.len()
392        )));
393    }
394    if all_source_frequencies_hz.first() <= all_source_frequencies_hz.last() {
395        return channel_mode_output_centers_simple(
396            all_source_frequencies_hz,
397            all_source_channel_widths_hz,
398            start_channel,
399            width_channels,
400            nchan,
401        );
402    }
403
404    let mut trans_freq_hz = all_source_frequencies_hz.to_vec();
405    let mut trans_width_hz = all_source_channel_widths_hz
406        .iter()
407        .map(|width_hz| width_hz.abs())
408        .collect::<Vec<_>>();
409    let mut start = start_channel;
410    let mut start_is_end = width_channels < 0;
411
412    if trans_freq_hz.first() > trans_freq_hz.last() {
413        trans_freq_hz.reverse();
414        trans_width_hz.reverse();
415        start = nchannels - 1 - start;
416        start_is_end = !start_is_end;
417    }
418
419    if start < 0 || start >= nchannels {
420        return Err(MsError::VersionError(format!(
421            "cube channel start {start_channel} is outside the SPW with {nchannels} channels"
422        )));
423    }
424
425    let requested_nchan = i32::try_from(nchan)
426        .map_err(|_| MsError::VersionError("cube channel count exceeds i32".to_string()))?;
427    let mut bandwidth_channels = requested_nchan;
428    let mut center_channel = start;
429    if start_is_end {
430        center_channel -= bandwidth_channels / 2;
431    } else {
432        center_channel += bandwidth_channels / 2;
433    }
434
435    if center_channel - bandwidth_channels / 2 < 0 {
436        bandwidth_channels = 2 * center_channel + 1;
437    }
438    if nchannels < center_channel + bandwidth_channels / 2 {
439        bandwidth_channels = 2 * (nchannels - center_channel);
440    }
441
442    let requested_width_channels = width_channels.abs();
443    let channel_width = if requested_width_channels < 1 {
444        1
445    } else if requested_width_channels > bandwidth_channels {
446        bandwidth_channels
447    } else {
448        requested_width_channels
449    };
450    bandwidth_channels = requested_nchan * channel_width;
451
452    let bandwidth_lower_end_channel = center_channel - bandwidth_channels / 2;
453    let bandwidth_upper_end_channel = bandwidth_lower_end_channel + bandwidth_channels - 1;
454
455    if channel_width == bandwidth_channels {
456        let low = usize::try_from(bandwidth_lower_end_channel)
457            .expect("validated lower single-channel bound");
458        let high = usize::try_from(bandwidth_upper_end_channel)
459            .expect("validated upper single-channel bound");
460        return Ok(vec![
461            0.5 * ((trans_freq_hz[low] - trans_width_hz[low] / 2.0)
462                + (trans_freq_hz[high] + trans_width_hz[high] / 2.0)),
463        ]);
464    }
465
466    let mut lower_bounds_hz = Vec::new();
467    let mut upper_bounds_hz = Vec::new();
468    let channel_ratio = bandwidth_channels / channel_width;
469    let start_channel_index = if channel_ratio % 2 != 0 {
470        center_channel - channel_width / 2
471    } else {
472        center_channel
473    };
474
475    let mut lower_indices_up = Vec::new();
476    let mut upper_indices_up = Vec::new();
477    let mut index = start_channel_index;
478    while index <= bandwidth_upper_end_channel {
479        lower_indices_up.push(index);
480        upper_indices_up.push((index + channel_width - 1).min(bandwidth_upper_end_channel));
481        index += channel_width;
482    }
483
484    let mut lower_indices_down = Vec::new();
485    let mut upper_indices_down = Vec::new();
486    let mut index = start_channel_index - 1;
487    while index >= bandwidth_lower_end_channel {
488        upper_indices_down.push(index);
489        lower_indices_down.push((index - channel_width + 1).max(bandwidth_lower_end_channel));
490        index -= channel_width;
491    }
492
493    for reverse_index in (0..lower_indices_down.len()).rev() {
494        let low = usize::try_from(lower_indices_down[reverse_index])
495            .expect("validated lower descending bound");
496        let high = usize::try_from(upper_indices_down[reverse_index])
497            .expect("validated upper descending bound");
498        lower_bounds_hz.push(trans_freq_hz[low] - trans_width_hz[low] / 2.0);
499        upper_bounds_hz.push(trans_freq_hz[high] + trans_width_hz[high] / 2.0);
500    }
501    for index in 0..lower_indices_up.len() {
502        let low =
503            usize::try_from(lower_indices_up[index]).expect("validated lower ascending bound");
504        let high =
505            usize::try_from(upper_indices_up[index]).expect("validated upper ascending bound");
506        lower_bounds_hz.push(trans_freq_hz[low] - trans_width_hz[low] / 2.0);
507        upper_bounds_hz.push(trans_freq_hz[high] + trans_width_hz[high] / 2.0);
508    }
509
510    Ok(lower_bounds_hz
511        .into_iter()
512        .zip(upper_bounds_hz)
513        .map(|(low_hz, high_hz)| 0.5 * (low_hz + high_hz))
514        .collect())
515}

539fn casa_transformed_channel_mode_output_centers(
540    transformed_source_frequencies_hz: &[f64],
541    transformed_source_channel_widths_hz: &[f64],
542    start_channel: i32,
543    width_channels: i32,
544    nchan: usize,
545) -> MsResult<Vec<f64>> {
546    if width_channels == 0 {
547        return Err(MsError::VersionError(
548            "cube channel width must not be zero".to_string(),
549        ));
550    }
551    if transformed_source_frequencies_hz.len() != transformed_source_channel_widths_hz.len() {
552        return Err(MsError::VersionError(format!(
553            "CASA transformed cube axis requires matching frequency/width arrays, got {} and {}",
554            transformed_source_frequencies_hz.len(),
555            transformed_source_channel_widths_hz.len()
556        )));
557    }
558    if transformed_source_frequencies_hz.is_empty() {
559        return Err(MsError::VersionError(
560            "cannot build CASA transformed cube axis from an empty SPW".to_string(),
561        ));
562    }
563
564    let nchannels = i32::try_from(transformed_source_frequencies_hz.len()).map_err(|_| {
565        MsError::VersionError("spectral window channel count exceeds i32".to_string())
566    })?;
567    let mut transformed_frequencies_hz = transformed_source_frequencies_hz.to_vec();
568    let mut transformed_widths_hz = transformed_source_channel_widths_hz
569        .iter()
570        .map(|width_hz| width_hz.abs())
571        .collect::<Vec<_>>();
572    let mut first_channel = start_channel;
573    let mut start_is_end = width_channels < 0;
574    let descending = transformed_frequencies_hz.first() > transformed_frequencies_hz.last();
575
576    if descending {
577        transformed_frequencies_hz.reverse();
578        transformed_widths_hz.reverse();
579        first_channel = nchannels - 1 - first_channel;
580        start_is_end = !start_is_end;
581    }
582
583    if first_channel < 0 || first_channel >= nchannels {
584        return Err(MsError::VersionError(format!(
585            "cube channel start {start_channel} is outside the SPW with {nchannels} channels"
586        )));
587    }
588
589    let output_channel_width_hz =
590        transformed_widths_hz[0] * f64::from(width_channels.unsigned_abs());
591    let output_bandwidth_hz = output_channel_width_hz * nchan as f64;
592    let first_channel = usize::try_from(first_channel).expect("validated channel index");
593    let start_edge_hz = if start_is_end {
594        transformed_frequencies_hz[first_channel] + transformed_widths_hz[first_channel] / 2.0
595            - output_bandwidth_hz
596    } else {
597        transformed_frequencies_hz[first_channel] - transformed_widths_hz[first_channel] / 2.0
598    };
599    let mut centers_hz = (0..nchan)
600        .map(|index| start_edge_hz + (index as f64 + 0.5) * output_channel_width_hz)
601        .collect::<Vec<_>>();
602    if descending ^ (width_channels < 0) {
603        centers_hz.reverse();
604    }
605    Ok(centers_hz)
606}

608fn channel_mode_output_centers_simple(
609    all_source_frequencies_hz: &[f64],
610    all_source_channel_widths_hz: &[f64],
611    start_channel: i32,
612    width_channels: i32,
613    nchan: usize,
614) -> MsResult<Vec<f64>> {
615    let nchannels = i32::try_from(all_source_frequencies_hz.len()).map_err(|_| {
616        MsError::VersionError("spectral window channel count exceeds i32".to_string())
617    })?;
618    let mut centers = Vec::with_capacity(nchan);
619    for offset in 0..nchan {
620        let offset = i32::try_from(offset)
621            .map_err(|_| MsError::VersionError("cube channel count exceeds i32".to_string()))?;
622        let (low, high) = if width_channels > 0 {
623            let low = start_channel + offset * width_channels;
624            let high = low + width_channels;
625            (low, high)
626        } else {
627            let high = start_channel + 1 + offset * width_channels;
628            let low = high + width_channels;
629            (low, high)
630        };
631        if low < 0 || high < 0 || low >= nchannels || high > nchannels || low >= high {
632            return Err(MsError::VersionError(format!(
633                "cube channel axis bin [{low}, {high}) is outside the SPW with {nchannels} channels"
634            )));
635        }
636        let selected = usize::try_from(low).expect("validated low bound")
637            ..usize::try_from(high).expect("validated high bound");
638        let low_edge_hz = selected
639            .clone()
640            .map(|index| {
641                all_source_frequencies_hz[index] - all_source_channel_widths_hz[index].abs() / 2.0
642            })
643            .reduce(f64::min)
644            .expect("validated non-empty channel range");
645        let high_edge_hz = selected
646            .map(|index| {
647                all_source_frequencies_hz[index] + all_source_channel_widths_hz[index].abs() / 2.0
648            })
649            .reduce(f64::max)
650            .expect("validated non-empty channel range");
651        centers.push(0.5 * (low_edge_hz + high_edge_hz));
652    }
653    Ok(centers)
654}

656/// Resolve a simple contiguous source-channel selection.
657pub fn resolve_contiguous_channel_selection(
658    all_frequencies_hz: &[f64],
659    start: Option<usize>,
660    count: Option<usize>,
661) -> MsResult<ResolvedChannelSelection> {
662    let start = start.unwrap_or(0);
663    if start >= all_frequencies_hz.len() {
664        return Err(MsError::VersionError(format!(
665            "channel start {start} is outside spectral window with {} channels",
666            all_frequencies_hz.len()
667        )));
668    }
669    let count = count.unwrap_or(all_frequencies_hz.len() - start);
670    if count == 0 || start + count > all_frequencies_hz.len() {
671        return Err(MsError::VersionError(format!(
672            "channel selection [{start}, {}) is outside spectral window with {} channels",
673            start + count,
674            all_frequencies_hz.len()
675        )));
676    }
677    Ok(ResolvedChannelSelection {
678        indices: (start..start + count).collect(),
679        frequencies_hz: all_frequencies_hz[start..start + count].to_vec(),
680    })
681}

698            .collect(),

865            source_freq_ref,
866            output_freq_ref,

868            interpolation_uses_native_source_frequencies: uses_native_source_interpolation(

874            output_channel_frequencies_hz,
875        };
876        let support = cube_source_channel_support(
877            all_source_frequencies_hz,
878            all_source_channel_widths_hz,
879            &setup,
880            time_bounds_mjd_sec,
881            field_id,
882            derived_engine,
883        )?;
884        Ok((setup, support))

965                )
966                .map_err(|error| {
967                    MsError::VersionError(format!(

990                        output_freq_ref,

1005                        &source_channel_widths_in_output_frame,
1006                        start_channel,
1007                        width_channels,

1009                } else {
1010                    channel_widths_from_centers(&output_channel_frequencies_hz)?
1011                };

1013            };
1014        let setup = Self {
1015            source_freq_ref,
1016            output_freq_ref,
1017            interpolation: axis_config.interpolation,

1019                axis_config,
1020            ),
1021            output_frame_reference_time_mjd_sec: reference_row_time_mjd_sec,
1022            output_frame_field_id: field_id,
1023            output_channel_widths_hz,
1024            output_channel_frequencies_hz,

1027            all_source_frequencies_hz,

1180            &source_channel_widths_for_interpolation,
1181            &padded_output_frequencies_for_interpolation,
1182            &[],
1183            self.interpolation,
1184        );
1185        let source_channel_model_contributions = source_frequencies_for_interpolation
1186            .into_iter()
1187            .map(|source_frequency_hz| match self.interpolation {
1188                CubeInterpolation::Nearest => nearest_channel_index(
1189                    &output_frequencies_for_interpolation,
1190                    source_frequency_hz,
1191                )
1192                .map(|index| {
1193                    vec![CubeChannelContribution {
1194                        source_channel: index,
1195                        source_frequency_hz: self.output_channel_frequencies_hz[index],
1196                        factor: 1.0,
1197                    }]
1198                })
1199                .unwrap_or_default(),
1200                CubeInterpolation::Linear | CubeInterpolation::Cubic => {

1202                        &self.output_channel_frequencies_hz,

1204                        &output_channel_widths_for_interpolation,
1205                        source_frequency_hz,
1206                    )
1207                }
1208            })
1209            .collect();
1210        Ok(CubeRowSpectralContributions {
1211            output_channel_contributions,
1212            source_channel_output_map,
1213            padded_source_channel_output_map,
1214            padded_grid_channel_contributions,

1361fn effective_output_frequency_ref(
1362    source_freq_ref: FrequencyRef,
1363    axis_config: &CubeAxisConfig,
1364) -> MsResult<FrequencyRef> {
1365    if matches!(axis_config.specmode, CubeSpecMode::Cubedata) {
1366        return Ok(source_freq_ref);
1367    }
1368    let start_frame = cube_axis_explicit_frame(axis_config.start.as_ref());
1369    let width_frame = cube_axis_explicit_frame(axis_config.width.as_ref());
1370    match (start_frame, width_frame) {
1371        (Some(left), Some(right)) if left != right => Err(MsError::VersionError(format!(
1372            "cube axis start frame {left} and width frame {right} disagree"
1373        ))),
1374        (Some(frame), _) | (_, Some(frame)) => Ok(frame),
1375        (None, None) => Ok(axis_config.outframe),
1376    }
1377}

1399fn cube_axis_frequency_hz_from_value(
1400    axis_value: &CubeAxisValue,
1401    output_freq_ref: FrequencyRef,
1402    rest_frequency_hz: Option<f64>,
1403    veltype: DopplerRef,
1404    row_time_mjd_sec: f64,
1405    field_id: usize,
1406    derived_engine: &MsCalEngine,
1407) -> MsResult<f64> {
1408    match axis_value {
1409        CubeAxisValue::Channel(channel) => Err(MsError::VersionError(format!(
1410            "expected frequency-like cube axis value, found channel {channel}"
1411        ))),
1412        CubeAxisValue::FrequencyHz { hz, frame } => {
1413            let source_freq_ref = frame.unwrap_or(output_freq_ref);
1414            convert_frequency_to_frame(
1415                source_freq_ref,
1416                output_freq_ref,
1417                *hz,
1418                row_time_mjd_sec,
1419                field_id,
1420                derived_engine,
1421            )
1422        }
1423        CubeAxisValue::VelocityMs { ms, frame } => {
1424            let rest_frequency_hz = required_rest_frequency_hz(rest_frequency_hz, axis_value)?;
1425            let doppler = MDoppler::new(ms / SPEED_OF_LIGHT_M_S, veltype)
1426                .convert_to(DopplerRef::RATIO, &MeasFrame::new())
1427                .map_err(|error| {
1428                    MsError::VersionError(format!(
1429                        "convert cube-axis velocity {ms} m/s to frequency ratio: {error}"
1430                    ))
1431                })?;
1432            let frequency_hz = rest_frequency_hz * doppler.value();
1433            let source_freq_ref = frame.unwrap_or(output_freq_ref);
1434            convert_frequency_to_frame(
1435                source_freq_ref,
1436                output_freq_ref,
1437                frequency_hz,
1438                row_time_mjd_sec,
1439                field_id,
1440                derived_engine,
1441            )
1442        }
1443        CubeAxisValue::Doppler { value, convention } => {
1444            let rest_frequency_hz = required_rest_frequency_hz(rest_frequency_hz, axis_value)?;
1445            let doppler = MDoppler::new(*value, *convention)
1446                .convert_to(DopplerRef::RATIO, &MeasFrame::new())
1447                .map_err(|error| {
1448                    MsError::VersionError(format!(
1449                        "convert cube-axis doppler value {value} to frequency ratio: {error}"
1450                    ))
1451                })?;
1452            Ok(rest_frequency_hz * doppler.value())
1453        }
1454    }
1455}

1458    axis_value: &CubeAxisValue,
1459    rest_frequency_hz: Option<f64>,
1460    veltype: DopplerRef,
1461) -> MsResult<f64> {
1462    match axis_value {
1463        CubeAxisValue::Channel(channel_width) => Err(MsError::VersionError(format!(
1464            "expected frequency-like cube increment, found channel width {channel_width}"
1465        ))),
1466        CubeAxisValue::FrequencyHz { hz, .. } => Ok(*hz),
1467        CubeAxisValue::VelocityMs { ms, .. } => {
1468            let rest_frequency_hz = required_rest_frequency_hz(rest_frequency_hz, axis_value)?;
1469            let doppler = MDoppler::new(ms / SPEED_OF_LIGHT_M_S, veltype)
1470                .convert_to(DopplerRef::RATIO, &MeasFrame::new())

1473                        "convert cube-axis velocity width {ms} m/s to frequency ratio: {error}"
1474                    ))
1475                })?;
1476            Ok(rest_frequency_hz * (doppler.value() - 1.0))
1477        }
1478        CubeAxisValue::Doppler { value, convention } => {

1480            let doppler = MDoppler::new(*value, *convention)
1481                .convert_to(DopplerRef::RATIO, &MeasFrame::new())
1482                .map_err(|error| {
1483                    MsError::VersionError(format!(
1484                        "convert cube-axis doppler width {value} to frequency ratio: {error}"

1487            Ok(rest_frequency_hz * (doppler.value() - 1.0))
1488        }
1489    }

1492fn frequency_hz_from_vopt_ms(optical_velocity_ms: f64, rest_frequency_hz: f64) -> f64 {
1493    rest_frequency_hz / (1.0 + optical_velocity_ms / SPEED_OF_LIGHT_M_S)
1494}

1496fn optical_velocity_ms_from_frequency_hz(frequency_hz: f64, rest_frequency_hz: f64) -> f64 {
1497    SPEED_OF_LIGHT_M_S * (rest_frequency_hz / frequency_hz - 1.0)
1498}

1500fn optical_velocity_ms_from_axis_value(
1501    axis_value: &CubeAxisValue,
1502    rest_frequency_hz: f64,
1503) -> MsResult<f64> {
1504    match axis_value {
1505        CubeAxisValue::VelocityMs { ms, .. } => Ok(*ms),
1506        CubeAxisValue::Doppler { value, convention } => {
1507            let doppler = MDoppler::new(*value, *convention)
1508                .convert_to(DopplerRef::RATIO, &MeasFrame::new())
1509                .map_err(|error| {
1510                    MsError::VersionError(format!(
1511                        "convert cube-axis optical doppler {value} to ratio: {error}"
1512                    ))
1513                })?;
1514            Ok(optical_velocity_ms_from_frequency_hz(
1515                rest_frequency_hz * doppler.value(),
1516                rest_frequency_hz,
1517            ))
1518        }
1519        other => Err(MsError::VersionError(format!(
1520            "expected optical velocity-like cube axis value, found {other:?}"
1521        ))),
1522    }
1523}

1525fn optical_velocity_output_centers(
1526    source_frequencies_in_output_frame_hz: &[f64],
1527    source_channel_widths_hz: &[f64],
1528    nchan: usize,
1529    axis_config: &CubeAxisConfig,
1530) -> MsResult<Vec<f64>> {
1531    if nchan == 0 {
1532        return Err(MsError::VersionError(
1533            "cube output channel count must be positive".to_string(),
1534        ));
1535    }
1536    if source_frequencies_in_output_frame_hz.is_empty()
1537        || source_channel_widths_hz.len() != source_frequencies_in_output_frame_hz.len()
1538    {
1539        return Err(MsError::VersionError(
1540            "optical velocity axis requires non-empty source frequencies and matching widths"
1541                .to_string(),
1542        ));
1543    }
1544    let rest_frequency_hz =
1545        required_rest_frequency_hz(axis_config.rest_frequency_hz, &axis_config.start)?;
1546    let mut source_frequencies_hz = source_frequencies_in_output_frame_hz.to_vec();
1547    let mut source_widths_hz = source_channel_widths_hz.to_vec();
1548    let mut reverse_output = false;
1549    if source_frequencies_hz.len() > 1 && source_frequencies_hz[1] < source_frequencies_hz[0] {
1550        source_frequencies_hz.reverse();
1551        source_widths_hz.reverse();
1552        reverse_output = true;
1553    }
1554
1555    let width_ms = match axis_config.width.as_ref() {
1556        Some(value @ (CubeAxisValue::VelocityMs { .. } | CubeAxisValue::Doppler { .. })) => {
1557            optical_velocity_ms_from_axis_value(value, rest_frequency_hz)?.abs()
1558        }
1559        Some(other) => {
1560            return Err(MsError::VersionError(format!(
1561                "optical velocity cube width expects a velocity-like value, found {other:?}"
1562            )));
1563        }
1564        None => {
1565            let first_center_hz = source_frequencies_hz[0];
1566            let first_width_hz = source_widths_hz[0];
1567            let upper_edge = optical_velocity_ms_from_frequency_hz(
1568                first_center_hz - first_width_hz,
1569                rest_frequency_hz,
1570            );
1571            let lower_edge = optical_velocity_ms_from_frequency_hz(
1572                first_center_hz + first_width_hz,
1573                rest_frequency_hz,
1574            );
1575            (upper_edge - lower_edge).abs()
1576        }
1577    };
1578    if !(width_ms.is_finite() && width_ms > 0.0) {
1579        return Err(MsError::VersionError(
1580            "optical velocity cube width must be positive".to_string(),
1581        ));
1582    }
1583
1584    let start_is_end = match axis_config.width.as_ref() {
1585        Some(CubeAxisValue::VelocityMs { ms, .. }) => *ms >= 0.0,
1586        Some(CubeAxisValue::Doppler { value, .. }) => *value >= 0.0,
1587        None => true,
1588        Some(other) => {
1589            return Err(MsError::VersionError(format!(
1590                "optical velocity cube width expects a velocity-like value, found {other:?}"
1591            )));
1592        }
1593    };
1594
1595    let center_is_start = axis_config.start.is_some();
1596    let mut regrid_center_hz;
1597    let regrid_center_velocity_ms;
1598    if let Some(start_value) = axis_config.start.as_ref() {
1599        let mut start_velocity_ms =
1600            optical_velocity_ms_from_axis_value(start_value, rest_frequency_hz)?;
1601        if center_is_start {
1602            if start_is_end {
1603                start_velocity_ms -= width_ms / 2.0;
1604            } else {
1605                start_velocity_ms += width_ms / 2.0;
1606            }
1607        }
1608        regrid_center_hz = frequency_hz_from_vopt_ms(start_velocity_ms, rest_frequency_hz);
1609        regrid_center_velocity_ms = start_velocity_ms;
1610    } else {
1611        regrid_center_hz = (source_frequencies_hz[0] - source_widths_hz[0]
1612            + source_frequencies_hz[source_frequencies_hz.len() - 1]
1613            + source_widths_hz[source_widths_hz.len() - 1])
1614            / 2.0;
1615        regrid_center_velocity_ms =
1616            optical_velocity_ms_from_frequency_hz(regrid_center_hz, rest_frequency_hz);
1617    }
1618
1619    let regrid_bandwidth_hz = {
1620        let div_by_two = if center_is_start { 1.0 } else { 0.5 };
1621        let bw_edge_hz = if center_is_start && !start_is_end {
1622            frequency_hz_from_vopt_ms(
1623                regrid_center_velocity_ms - nchan as f64 * width_ms * div_by_two,
1624                rest_frequency_hz,
1625            )
1626        } else {
1627            frequency_hz_from_vopt_ms(
1628                regrid_center_velocity_ms + nchan as f64 * width_ms * div_by_two,
1629                rest_frequency_hz,
1630            )
1631        };
1632        (bw_edge_hz - regrid_center_hz).abs() / div_by_two
1633    };
1634    if center_is_start {
1635        let center_velocity_ms = if start_is_end {
1636            regrid_center_velocity_ms + nchan as f64 * width_ms / 2.0
1637        } else {
1638            regrid_center_velocity_ms - nchan as f64 * width_ms / 2.0
1639        };
1640        regrid_center_hz = frequency_hz_from_vopt_ms(center_velocity_ms, rest_frequency_hz);
1641    }
1642
1643    let channel_upper_edge_hz = frequency_hz_from_vopt_ms(
1644        regrid_center_velocity_ms - width_ms / 2.0,
1645        rest_frequency_hz,
1646    );
1647    let regrid_channel_width_hz = 2.0
1648        * (channel_upper_edge_hz
1649            - frequency_hz_from_vopt_ms(regrid_center_velocity_ms, rest_frequency_hz));
1650
1651    let lower_source_edge_hz = source_frequencies_hz[0] - source_widths_hz[0] / 2.0;
1652    let upper_source_edge_hz = source_frequencies_hz[source_frequencies_hz.len() - 1]
1653        + source_widths_hz[source_widths_hz.len() - 1] / 2.0;
1654
1655    let mut the_regrid_center_hz = regrid_center_hz;
1656    if (the_regrid_center_hz - upper_source_edge_hz) > 1.0 {
1657        the_regrid_center_hz = upper_source_edge_hz;
1658    } else if the_regrid_center_hz < lower_source_edge_hz {
1659        the_regrid_center_hz = lower_source_edge_hz;
1660    }
1661
1662    let mut the_regrid_bandwidth_hz = regrid_bandwidth_hz;
1663    let range_tolerance_hz = source_widths_hz[0];
1664    if (the_regrid_center_hz + the_regrid_bandwidth_hz / 2.0) - upper_source_edge_hz
1665        > range_tolerance_hz
1666    {
1667        the_regrid_bandwidth_hz = (upper_source_edge_hz - the_regrid_center_hz)
1668            .abs()
1669            .min((the_regrid_center_hz - lower_source_edge_hz).abs())
1670            * 2.0;
1671    }
1672    if (the_regrid_center_hz - the_regrid_bandwidth_hz / 2.0) - lower_source_edge_hz
1673        < -range_tolerance_hz
1674    {
1675        the_regrid_bandwidth_hz = (upper_source_edge_hz - the_regrid_center_hz)
1676            .abs()
1677            .min((the_regrid_center_hz - lower_source_edge_hz).abs())
1678            * 2.0;
1679    }
1680
1681    let central_channel_width_hz = regrid_channel_width_hz;
1682
1683    let mut lo_fb_up = Vec::new();
1684    let mut hi_fb_up = Vec::new();
1685    let mut lo_fb_down = Vec::new();
1686    let mut hi_fb_down = Vec::new();
1687    let edge_tolerance_hz = central_channel_width_hz.abs() * 0.01;
1688    let upper_end_hz = the_regrid_center_hz + the_regrid_bandwidth_hz / 2.0;
1689    let lower_end_hz = the_regrid_center_hz - the_regrid_bandwidth_hz / 2.0;
1690    let upper_end_velocity_ms =
1691        optical_velocity_ms_from_frequency_hz(upper_end_hz, rest_frequency_hz);
1692    let lower_end_velocity_ms =
1693        optical_velocity_ms_from_frequency_hz(lower_end_hz, rest_frequency_hz);
1694    let tnum_chan =
1695        ((the_regrid_bandwidth_hz + edge_tolerance_hz) / central_channel_width_hz).floor();
1696    if (tnum_chan as i64) % 2 != 0 {
1697        lo_fb_up.push(the_regrid_center_hz - central_channel_width_hz / 2.0);
1698        hi_fb_up.push(the_regrid_center_hz + central_channel_width_hz / 2.0);
1699        lo_fb_down.push(the_regrid_center_hz - central_channel_width_hz / 2.0);
1700        hi_fb_down.push(the_regrid_center_hz + central_channel_width_hz / 2.0);
1701    } else {
1702        lo_fb_up.push(the_regrid_center_hz);
1703        hi_fb_up.push(the_regrid_center_hz + central_channel_width_hz);
1704        lo_fb_down.push(the_regrid_center_hz);
1705        hi_fb_down.push(the_regrid_center_hz + central_channel_width_hz);
1706    }
1707
1708    let velocity_channel_width_ms =
1709        optical_velocity_ms_from_frequency_hz(lo_fb_up[0], rest_frequency_hz)
1710            - optical_velocity_ms_from_frequency_hz(hi_fb_up[0], rest_frequency_hz);
1711    let mut lower_velocity_ms =
1712        optical_velocity_ms_from_frequency_hz(*hi_fb_up.last().unwrap(), rest_frequency_hz);
1713    let mut upper_velocity_ms = lower_velocity_ms - velocity_channel_width_ms;
1714    while upper_end_velocity_ms - upper_velocity_ms < velocity_channel_width_ms / 10.0 {
1715        let upper_frequency_hz = frequency_hz_from_vopt_ms(upper_velocity_ms, rest_frequency_hz);
1716        if upper_frequency_hz <= upper_end_hz + edge_tolerance_hz {
1717            lo_fb_up.push(*hi_fb_up.last().unwrap());
1718            hi_fb_up.push(upper_frequency_hz);
1719        } else if upper_frequency_hz < upper_end_hz + edge_tolerance_hz {
1720            lo_fb_up.push(*hi_fb_up.last().unwrap());
1721            hi_fb_up.push(upper_end_hz);
1722            break;
1723        } else {
1724            break;
1725        }
1726        lower_velocity_ms =
1727            optical_velocity_ms_from_frequency_hz(*hi_fb_up.last().unwrap(), rest_frequency_hz);
1728        upper_velocity_ms = lower_velocity_ms - velocity_channel_width_ms;
1729    }
1730
1731    let mut upper_velocity_ms =
1732        optical_velocity_ms_from_frequency_hz(*lo_fb_down.last().unwrap(), rest_frequency_hz);
1733    let mut lower_velocity_ms = upper_velocity_ms + velocity_channel_width_ms;
1734    while lower_velocity_ms - lower_end_velocity_ms < velocity_channel_width_ms / 10.0 {
1735        let lower_frequency_hz = frequency_hz_from_vopt_ms(lower_velocity_ms, rest_frequency_hz);
1736        if lower_frequency_hz >= lower_end_hz - edge_tolerance_hz {
1737            hi_fb_down.push(*lo_fb_down.last().unwrap());
1738            lo_fb_down.push(lower_frequency_hz);
1739        } else if lower_frequency_hz > lower_end_hz - edge_tolerance_hz {
1740            hi_fb_down.push(*lo_fb_down.last().unwrap());
1741            lo_fb_down.push(lower_end_hz);
1742            break;
1743        } else {
1744            break;
1745        }
1746        upper_velocity_ms =
1747            optical_velocity_ms_from_frequency_hz(*lo_fb_down.last().unwrap(), rest_frequency_hz);
1748        lower_velocity_ms = upper_velocity_ms + velocity_channel_width_ms;
1749    }
1750
1751    let mut centers = Vec::with_capacity(lo_fb_down.len() + lo_fb_up.len());
1752    for index in (0..lo_fb_down.len()).rev() {
1753        centers.push((lo_fb_down[index] + hi_fb_down[index]) / 2.0);
1754    }
1755    centers.extend(
1756        lo_fb_up
1757            .iter()
1758            .zip(hi_fb_up.iter())
1759            .enumerate()
1760            .filter(|(index, _)| *index > 0)
1761            .map(|(_, (low_hz, high_hz))| (low_hz + high_hz) / 2.0),
1762    );
1763    if reverse_output {
1764        centers.reverse();
1765    }
1766    Ok(centers)
1767}

1868fn convert_frequency_to_frame_with_frames(
1869    source_freq_ref: FrequencyRef,
1870    target_freq_ref: FrequencyRef,
1871    frequency_hz: f64,
1872    source_frame: Option<&MeasFrame>,
1873    target_frame: Option<&MeasFrame>,
1874) -> MsResult<f64> {
1875    if source_freq_ref == target_freq_ref {
1876        return Ok(frequency_hz);
1877    }
1878    let source_frame = source_frame.ok_or_else(|| {
1879        MsError::VersionError(format!(
1880            "source frame required for cross-frame frequency conversion from {source_freq_ref} to {target_freq_ref}"
1881        ))
1882    })?;
1883    let target_frame = target_frame.ok_or_else(|| {
1884        MsError::VersionError(format!(
1885            "target frame required for cross-frame frequency conversion from {source_freq_ref} to {target_freq_ref}"
1886        ))
1887    })?;
1888    let mut current_frequency_hz = frequency_hz;
1889    let mut current_ref = source_freq_ref;
1890    for next_ref in direct_frequency_hop_path(source_freq_ref, target_freq_ref)? {
1891        let hop_frame = if direct_frequency_hop_uses_target_frame(current_ref, next_ref) {
1892            target_frame
1893        } else {
1894            source_frame
1895        };
1896        current_frequency_hz = MFrequency::new(current_frequency_hz, current_ref)
1897            .convert_to(next_ref, hop_frame)
1898            .map(|frequency| frequency.hz())
1899            .map_err(|error| {
1900                MsError::VersionError(format!(
1901                    "convert frequency {current_frequency_hz} Hz from {current_ref} to {next_ref}: {error}"
1902                ))
1903            })?;
1904        current_ref = next_ref;
1905    }
1906    Ok(current_frequency_hz)
1907}

1944    source_freq_ref: FrequencyRef,
1945    target_freq_ref: FrequencyRef,
1946    source_channel_frequencies_hz: &[f64],
1947    source_channel_widths_hz: &[f64],

1949    target_frame: &MeasFrame,

1979                target_freq_ref,
1980                frequency_hz - half_width_hz,
1981                Some(source_frame),

1989const DIRECT_FREQUENCY_HOP_EDGES: &[(FrequencyRef, FrequencyRef)] = &[
1990    (FrequencyRef::REST, FrequencyRef::LSRK),
1991    (FrequencyRef::LSRK, FrequencyRef::BARY),
1992    (FrequencyRef::LSRD, FrequencyRef::BARY),
1993    (FrequencyRef::LSRD, FrequencyRef::GALACTO),
1994    (FrequencyRef::BARY, FrequencyRef::LGROUP),
1995    (FrequencyRef::BARY, FrequencyRef::CMB),
1996    (FrequencyRef::BARY, FrequencyRef::GEO),
1997    (FrequencyRef::GEO, FrequencyRef::TOPO),
1998];
1999
2000fn direct_frequency_hop_path(
2001    source_freq_ref: FrequencyRef,
2002    target_freq_ref: FrequencyRef,
2003) -> MsResult<Vec<FrequencyRef>> {
2004    use std::collections::{HashMap, HashSet, VecDeque};
2005
2006    if source_freq_ref == target_freq_ref {
2007        return Ok(Vec::new());
2008    }
2009
2010    let mut queue = VecDeque::from([source_freq_ref]);
2011    let mut visited = HashSet::from([source_freq_ref]);
2012    let mut parent = HashMap::<FrequencyRef, FrequencyRef>::new();
2013
2014    while let Some(current) = queue.pop_front() {
2015        for &(left, right) in DIRECT_FREQUENCY_HOP_EDGES {
2016            let next = if left == current {
2017                right
2018            } else if right == current {
2019                left
2020            } else {
2021                continue;
2022            };
2023            if !visited.insert(next) {
2024                continue;
2025            }
2026            parent.insert(next, current);
2027            if next == target_freq_ref {
2028                let mut path = Vec::new();
2029                let mut cursor = target_freq_ref;
2030                while cursor != source_freq_ref {
2031                    path.push(cursor);
2032                    cursor = *parent
2033                        .get(&cursor)
2034                        .expect("BFS parent must exist for visited node");
2035                }
2036                path.reverse();
2037                return Ok(path);
2038            }
2039            queue.push_back(next);
2040        }
2041    }
2042
2043    Err(MsError::VersionError(format!(
2044        "no direct-hop route found from {source_freq_ref} to {target_freq_ref}"
2045    )))
2046}

2062fn nearest_channel_index(channel_frequencies_hz: &[f64], frequency_hz: f64) -> Option<usize> {
2063    let (best_index, best_diff) = channel_frequencies_hz
2064        .iter()
2065        .enumerate()
2066        .map(|(index, channel_frequency_hz)| (index, (channel_frequency_hz - frequency_hz).abs()))
2067        .min_by(|left, right| {
2068            left.1
2069                .partial_cmp(&right.1)
2070                .unwrap_or(std::cmp::Ordering::Equal)
2071        })?;
2072    let max_allowed_diff = if channel_frequencies_hz.len() <= 1 {
2073        f64::INFINITY
2074    } else {
2075        let lower_spacing = best_index
2076            .checked_sub(1)
2077            .map(|index| (channel_frequencies_hz[best_index] - channel_frequencies_hz[index]).abs())
2078            .unwrap_or(f64::INFINITY);
2079        let upper_spacing = channel_frequencies_hz
2080            .get(best_index + 1)
2081            .map(|frequency| (frequency - channel_frequencies_hz[best_index]).abs())
2082            .unwrap_or(f64::INFINITY);
2083        0.5 * lower_spacing.min(upper_spacing)
2084    };
2085    if best_diff <= max_allowed_diff {
2086        Some(best_index)
2087    } else {
2088        None
2089    }
2090}

2273fn casa_grid_frequency_channel_map(
2274    source_channel_frequencies_hz: &[f64],
2275    output_channel_frequencies_hz: &[f64],
2276    output_channel_widths_hz: &[f64],
2277    interpolation: CubeInterpolation,
2278) -> (Vec<f64>, Vec<Option<usize>>) {
2279    if output_channel_frequencies_hz.is_empty() {
2280        return (Vec::new(), Vec::new());
2281    }
2282    if output_channel_frequencies_hz.len() == 1
2283        || interpolation == CubeInterpolation::Nearest
2284        || source_channel_frequencies_hz.len() == 1
2285    {
2286        let channel_map = build_source_channel_output_map(
2287            source_channel_frequencies_hz,
2288            output_channel_frequencies_hz,
2289            output_channel_widths_hz,
2290        );
2291        return (source_channel_frequencies_hz.to_vec(), channel_map);
2292    }
2293
2294    let image_increment_hz = output_channel_frequencies_hz[1] - output_channel_frequencies_hz[0];
2295    let source_increment_hz = source_channel_frequencies_hz[1] - source_channel_frequencies_hz[0];
2296    if !(image_increment_hz.is_finite()
2297        && source_increment_hz.is_finite()
2298        && image_increment_hz != 0.0
2299        && source_increment_hz != 0.0)
2300    {
2301        return (
2302            output_channel_frequencies_hz.to_vec(),
2303            (0..output_channel_frequencies_hz.len()).map(Some).collect(),
2304        );
2305    }
2306
2307    let width = image_increment_hz.abs() / source_increment_hz.abs();
2308    let direct_width_limit = match interpolation {
2309        CubeInterpolation::Linear => 2.0,
2310        CubeInterpolation::Cubic => 4.0,
2311        CubeInterpolation::Nearest => 1.0,
2312    };
2313    if width <= direct_width_limit {
2314        return (
2315            output_channel_frequencies_hz.to_vec(),
2316            (0..output_channel_frequencies_hz.len()).map(Some).collect(),
2317        );
2318    }
2319
2320    let min_vis_hz = source_channel_frequencies_hz
2321        .iter()
2322        .copied()
2323        .fold(f64::INFINITY, f64::min);
2324    let max_vis_hz = source_channel_frequencies_hz
2325        .iter()
2326        .copied()
2327        .fold(f64::NEG_INFINITY, f64::max);
2328    let mut min_image_hz = output_channel_frequencies_hz
2329        .iter()
2330        .copied()
2331        .fold(f64::INFINITY, f64::min);
2332    let mut max_image_hz = output_channel_frequencies_hz
2333        .iter()
2334        .copied()
2335        .fold(f64::NEG_INFINITY, f64::max);
2336    let image_width_hz = image_increment_hz.abs();
2337    if !((min_image_hz - image_width_hz / 2.0) <= max_vis_hz
2338        && (max_image_hz + image_width_hz / 2.0) >= min_vis_hz)
2339    {
2340        return (
2341            output_channel_frequencies_hz.to_vec(),
2342            output_channel_frequencies_hz.iter().map(|_| None).collect(),
2343        );
2344    }
2345
2346    if min_image_hz < min_vis_hz {
2347        if let Some(value) =
2348            nearest_bracketed_output_frequency(output_channel_frequencies_hz, min_vis_hz)
2349        {
2350            min_image_hz = value;
2351        }
2352    }
2353    if max_image_hz > max_vis_hz {
2354        if let Some(value) =
2355            nearest_bracketed_output_frequency(output_channel_frequencies_hz, max_vis_hz)
2356        {
2357            max_image_hz = value;
2358        }
2359    }
2360
2361    let interp_width_hz = (image_width_hz / width.floor()).copysign(source_increment_hz);
2362    let ninterp = (((max_image_hz - min_image_hz).abs() + image_width_hz) / interp_width_hz.abs())
2363        .ceil() as usize
2364        + 2;
2365    let mut first = if interp_width_hz > 0.0 {
2366        min_image_hz
2367    } else {
2368        max_image_hz
2369    };
2370    if interpolation == CubeInterpolation::Linear {
2371        first -= interp_width_hz;
2372    } else if interpolation == CubeInterpolation::Cubic {
2373        first -= 2.0 * interp_width_hz;
2374    }
2375    let start_edge_hz = image_width_hz / 2.0 - interp_width_hz.abs() / 2.0;
2376    if interp_width_hz > 0.0 {
2377        first -= start_edge_hz;
2378    } else {
2379        first += start_edge_hz;
2380    }
2381    let grid_frequencies_hz = (0..ninterp)
2382        .map(|index| first + index as f64 * interp_width_hz)
2383        .collect::<Vec<_>>();
2384    let half_image_width_hz = image_width_hz / 2.0;
2385    let channel_map = grid_frequencies_hz
2386        .iter()
2387        .copied()
2388        .map(|frequency_hz| {
2389            output_channel_frequencies_hz.iter().position(|output_hz| {
2390                frequency_hz >= output_hz - half_image_width_hz
2391                    && frequency_hz < output_hz + half_image_width_hz
2392            })
2393        })
2394        .collect();
2395    (grid_frequencies_hz, channel_map)
2396}

2443fn linear_channel_contributions_impl(
2444    native_source_channel_frequencies_hz: &[f64],
2445    source_channel_frequencies_hz: &[f64],
2446    source_channel_widths_hz: &[f64],
2447    frequency_hz: f64,
2448    clip_to_edge_channel: bool,
2449) -> Vec<CubeChannelContribution> {
2450    let nchan = source_channel_frequencies_hz.len();
2451    if nchan == 0 {
2452        return Vec::new();
2453    }
2454    if native_source_channel_frequencies_hz.len() != nchan {
2455        return Vec::new();
2456    }
2457    if source_channel_widths_hz.len() != nchan {
2458        return Vec::new();
2459    }
2460    if nchan == 1 {
2461        let low_edge_hz =
2462            source_channel_frequencies_hz[0] - source_channel_widths_hz[0].abs() / 2.0;
2463        let high_edge_hz =
2464            source_channel_frequencies_hz[0] + source_channel_widths_hz[0].abs() / 2.0;
2465        if frequency_hz < low_edge_hz || frequency_hz > high_edge_hz {
2466            return Vec::new();
2467        }
2468        return Some(0)
2469            .map(|index| {
2470                vec![CubeChannelContribution {
2471                    source_channel: index,
2472                    source_frequency_hz: native_source_channel_frequencies_hz[index],
2473                    factor: 1.0,
2474                }]
2475            })
2476            .unwrap_or_default();
2477    }
2478
2479    let min_spacing_hz = source_channel_frequencies_hz
2480        .windows(2)
2481        .map(|pair| (pair[1] - pair[0]).abs())
2482        .filter(|spacing| spacing.is_finite() && *spacing > 0.0)
2483        .fold(f64::INFINITY, f64::min);
2484    let center_tolerance_hz = if min_spacing_hz.is_finite() {
2485        min_spacing_hz * 1.0e-12
2486    } else {
2487        f64::EPSILON
2488    };
2489
2490    if let Some(index) = source_channel_frequencies_hz
2491        .iter()
2492        .position(|center| (*center - frequency_hz).abs() <= center_tolerance_hz)
2493    {
2494        return vec![CubeChannelContribution {
2495            source_channel: index,
2496            source_frequency_hz: native_source_channel_frequencies_hz[index],
2497            factor: 1.0,
2498        }];
2499    }
2500
2501    let ascending = source_channel_frequencies_hz[nchan - 1] >= source_channel_frequencies_hz[0];
2502    let first_center_hz = source_channel_frequencies_hz[0];
2503    let last_index = nchan - 1;
2504    let last_center_hz = source_channel_frequencies_hz[last_index];
2505    if clip_to_edge_channel {
2506        if ascending {
2507            let first_outer_edge_hz = first_center_hz - source_channel_widths_hz[0].abs() / 2.0;
2508            let last_outer_edge_hz =
2509                last_center_hz + source_channel_widths_hz[last_index].abs() / 2.0;
2510            if first_outer_edge_hz <= frequency_hz && frequency_hz < first_center_hz {
2511                return vec![CubeChannelContribution {
2512                    source_channel: 0,
2513                    source_frequency_hz: native_source_channel_frequencies_hz[0],
2514                    factor: 1.0,
2515                }];
2516            }
2517            if last_center_hz < frequency_hz && frequency_hz <= last_outer_edge_hz {
2518                return vec![CubeChannelContribution {
2519                    source_channel: last_index,
2520                    source_frequency_hz: native_source_channel_frequencies_hz[last_index],
2521                    factor: 1.0,
2522                }];
2523            }
2524        } else {
2525            let first_outer_edge_hz = first_center_hz + source_channel_widths_hz[0].abs() / 2.0;
2526            let last_outer_edge_hz =
2527                last_center_hz - source_channel_widths_hz[last_index].abs() / 2.0;
2528            if first_center_hz < frequency_hz && frequency_hz <= first_outer_edge_hz {
2529                return vec![CubeChannelContribution {
2530                    source_channel: 0,
2531                    source_frequency_hz: native_source_channel_frequencies_hz[0],
2532                    factor: 1.0,
2533                }];
2534            }
2535            if last_outer_edge_hz <= frequency_hz && frequency_hz < last_center_hz {
2536                return vec![CubeChannelContribution {
2537                    source_channel: last_index,
2538                    source_frequency_hz: native_source_channel_frequencies_hz[last_index],
2539                    factor: 1.0,
2540                }];
2541            }
2542        }
2543    }
2544
2545    let interval = if ascending {
2546        source_channel_frequencies_hz
2547            .windows(2)
2548            .position(|pair| pair[0] <= frequency_hz && frequency_hz <= pair[1])
2549    } else {
2550        source_channel_frequencies_hz
2551            .windows(2)
2552            .position(|pair| pair[1] <= frequency_hz && frequency_hz <= pair[0])
2553    };
2554    let Some(lower_index) = interval else {
2555        return Vec::new();
2556    };
2557
2558    let upper_index = lower_index + 1;
2559    let lower_freq = source_channel_frequencies_hz[lower_index];
2560    let upper_freq = source_channel_frequencies_hz[upper_index];
2561    let span = upper_freq - lower_freq;
2562    if !span.is_finite() || span == 0.0 {
2563        return vec![CubeChannelContribution {
2564            source_channel: lower_index,
2565            source_frequency_hz: native_source_channel_frequencies_hz[lower_index],
2566            factor: 1.0,
2567        }];
2568    }
2569
2570    let upper_factor = ((frequency_hz - lower_freq) / span) as f32;
2571    let lower_factor = 1.0 - upper_factor;
2572    let mut contributions = Vec::with_capacity(2);
2573    if lower_factor > 0.0 {
2574        contributions.push(CubeChannelContribution {
2575            source_channel: lower_index,
2576            source_frequency_hz: native_source_channel_frequencies_hz[lower_index],
2577            factor: lower_factor,
2578        });
2579    }
2580    if upper_factor > 0.0 {
2581        contributions.push(CubeChannelContribution {
2582            source_channel: upper_index,
2583            source_frequency_hz: native_source_channel_frequencies_hz[upper_index],
2584            factor: upper_factor,
2585        });
2586    }
2587    contributions
2588}

2591    all_source_frequencies_hz: &[f64],
2592    all_source_channel_widths_hz: &[f64],
2593    cube_setup: &CubeSpectralSetup,
2594    time_bounds_mjd_sec: [f64; 2],
2595    field_id: usize,
2596    derived_engine: &MsCalEngine,
2597) -> MsResult<ResolvedChannelSelection> {
2598    let mut first_index = None::<usize>;
2599    let mut last_index = None::<usize>;
2600    for row_time_mjd_sec in [time_bounds_mjd_sec[0], time_bounds_mjd_sec[1]] {
2601        let contributions = cube_setup.row_spectral_contributions(
2602            all_source_frequencies_hz,
2603            all_source_channel_widths_hz,
2604            row_time_mjd_sec,
2605            field_id,
2606            derived_engine,
2607        )?;
2608        for source_index in contributions
2609            .output_channel_contributions
2610            .into_iter()
2611            .flatten()
2612            .chain(
2613                contributions
2614                    .grid_channel_contributions
2615                    .into_iter()
2616                    .flat_map(|grid| grid.contributions),
2617            )
2618            .chain(
2619                contributions
2620                    .padded_grid_channel_contributions
2621                    .into_iter()
2622                    .flat_map(|grid| grid.contributions),
2623            )
2624            .map(|contribution| contribution.source_channel)
2625        {
2626            first_index =
2627                Some(first_index.map_or(source_index, |current| current.min(source_index)));
2628            last_index = Some(last_index.map_or(source_index, |current| current.max(source_index)));
2629        }
2630    }
2631    let mut first_index = first_index.ok_or_else(|| {
2632        MsError::VersionError(
2633            "cube channel selection produced no supporting source channels".to_string(),
2634        )
2635    })?;
2636    let mut last_index = last_index.expect("first_index implies last_index");
2637    first_index = first_index.saturating_sub(1);
2638    last_index = (last_index + 2).min(all_source_frequencies_hz.len().saturating_sub(1));
2639    Ok(ResolvedChannelSelection {
2640        indices: (first_index..=last_index).collect(),
2641        frequencies_hz: all_source_frequencies_hz[first_index..=last_index].to_vec(),
2642    })
