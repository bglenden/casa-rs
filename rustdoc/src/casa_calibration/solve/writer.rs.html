27pub(crate) fn write_gain_caltable(
28    ms: &MeasurementSet,
29    request: &GainSolveRequest,
30    refant_id: i32,
31    rows: &[SolutionRow],
32) -> Result<GainSolveReport, GainSolveError> {
33    prepare_output_root(&request.output_table)?;
34
35    let schema = TableSchema::new(vec![
36        ColumnSchema::scalar(COL_TIME, casa_types::PrimitiveType::Float64),
37        ColumnSchema::scalar(COL_FIELD_ID, casa_types::PrimitiveType::Int32),
38        ColumnSchema::scalar(COL_SPECTRAL_WINDOW_ID, casa_types::PrimitiveType::Int32),
39        ColumnSchema::scalar(COL_ANTENNA1, casa_types::PrimitiveType::Int32),
40        ColumnSchema::scalar(COL_ANTENNA2, casa_types::PrimitiveType::Int32),
41        ColumnSchema::scalar(COL_INTERVAL, casa_types::PrimitiveType::Float64),
42        ColumnSchema::scalar(COL_SCAN_NUMBER, casa_types::PrimitiveType::Int32),
43        ColumnSchema::scalar(COL_OBSERVATION_ID, casa_types::PrimitiveType::Int32),
44        ColumnSchema::scalar(COL_ARRAY_ID, casa_types::PrimitiveType::Int32),
45        ColumnSchema::scalar(COL_TIME_EXTRA_PREC, casa_types::PrimitiveType::Float64),
46        ColumnSchema::array_variable(COL_CPARAM, casa_types::PrimitiveType::Complex32, Some(2)),
47        ColumnSchema::array_variable(COL_PARAMERR, casa_types::PrimitiveType::Float32, Some(2)),
48        ColumnSchema::array_variable(COL_FLAG, casa_types::PrimitiveType::Bool, Some(2)),
49        ColumnSchema::array_variable(COL_SNR, casa_types::PrimitiveType::Float32, Some(2)),
50        ColumnSchema::array_variable(COL_WEIGHT, casa_types::PrimitiveType::Float32, Some(2)),
51    ])
52    .expect("valid gain caltable schema");
53    let mut table = Table::with_schema(schema);
54    table.set_info(TableInfo {
55        table_type: TABLE_INFO_TYPE.to_string(),
56        sub_type: request.gain_type.vis_cal().to_string(),
57        readme: Vec::new(),
58    });
59    table.keywords_mut().upsert(
60        KEY_PAR_TYPE,
61        Value::Scalar(ScalarValue::String("Complex".to_string())),
62    );
63    table.keywords_mut().upsert(
64        KEY_VIS_CAL,
65        Value::Scalar(ScalarValue::String(request.gain_type.vis_cal().to_string())),
66    );
67    table.keywords_mut().upsert(
68        KEY_MS_NAME,
69        Value::Scalar(ScalarValue::String(
70            ms.path()
71                .map(|path| path.display().to_string())
72                .unwrap_or_else(|| "<in-memory>".to_string()),
73        )),
74    );
75    table.keywords_mut().upsert(
76        KEY_POL_BASIS,
77        Value::Scalar(ScalarValue::String("unknown".to_string())),
78    );
79    table.keywords_mut().upsert(
80        KEY_CASA_VERSION,
81        Value::Scalar(ScalarValue::String("casa-rs".to_string())),
82    );
83    set_fixed_unit_keyword(&mut table, COL_TIME, &["s"]);
84    set_measinfo_keyword(&mut table, COL_TIME, "epoch", Some("UTC"));
85    set_fixed_unit_keyword(&mut table, COL_INTERVAL, &["s"]);
86    set_fixed_unit_keyword(&mut table, COL_TIME_EXTRA_PREC, &["s"]);
87    for name in STANDARD_SUBTABLE_KEYWORDS {
88        table.keywords_mut().upsert(
89            *name,
90            Value::table_ref(subtable_keyword_value(
91                &request.output_table,
92                &request.output_table.join(name),
93            )),
94        );
95    }
96
97    for row in rows {
98        table
99            .add_row(RecordValue::new(vec![
100                RecordField::new(
101                    COL_TIME,
102                    Value::Scalar(ScalarValue::Float64(row.time_seconds)),
103                ),
104                RecordField::new(
105                    COL_FIELD_ID,
106                    Value::Scalar(ScalarValue::Int32(row.field_id)),
107                ),
108                RecordField::new(
109                    COL_SPECTRAL_WINDOW_ID,
110                    Value::Scalar(ScalarValue::Int32(row.spw_id)),
111                ),
112                RecordField::new(
113                    COL_ANTENNA1,
114                    Value::Scalar(ScalarValue::Int32(row.antenna_id)),
115                ),
116                RecordField::new(
117                    COL_ANTENNA2,
118                    Value::Scalar(ScalarValue::Int32(row.refant_id)),
119                ),
120                RecordField::new(
121                    COL_INTERVAL,
122                    Value::Scalar(ScalarValue::Float64(row.interval_seconds)),
123                ),
124                RecordField::new(
125                    COL_SCAN_NUMBER,
126                    Value::Scalar(ScalarValue::Int32(row.scan_number)),
127                ),
128                RecordField::new(
129                    COL_OBSERVATION_ID,
130                    Value::Scalar(ScalarValue::Int32(row.observation_id)),
131                ),
132                RecordField::new(COL_ARRAY_ID, Value::Scalar(ScalarValue::Int32(0))),
133                RecordField::new(
134                    COL_TIME_EXTRA_PREC,
135                    Value::Scalar(ScalarValue::Float64(0.0)),
136                ),
137                RecordField::new(
138                    COL_CPARAM,
139                    Value::Array(ArrayValue::Complex32(
140                        ArrayD::from_shape_vec(IxDyn(&[row.gains.len(), 1]).f(), row.gains.clone())
141                            .expect("gain vector should reshape to receptor x channel"),
142                    )),
143                ),
144                RecordField::new(
145                    COL_PARAMERR,
146                    Value::Array(ArrayValue::Float32(
147                        ArrayD::from_shape_vec(
148                            IxDyn(&[row.gains.len(), 1]).f(),
149                            row.param_errors.clone(),
150                        )
151                        .expect("paramerr vector should reshape to receptor x channel"),
152                    )),
153                ),
154                RecordField::new(
155                    COL_FLAG,
156                    Value::Array(ArrayValue::Bool(
157                        ArrayD::from_shape_vec(IxDyn(&[row.flags.len(), 1]).f(), row.flags.clone())
158                            .expect("flag vector should reshape to receptor x channel"),
159                    )),
160                ),
161                RecordField::new(
162                    COL_SNR,
163                    Value::Array(ArrayValue::Float32(
164                        ArrayD::from_shape_vec(IxDyn(&[row.gains.len(), 1]).f(), row.snrs.clone())
165                            .expect("snr vector should reshape to receptor x channel"),
166                    )),
167                ),
168                RecordField::new(
169                    COL_WEIGHT,
170                    Value::Array(ArrayValue::Float32(
171                        ArrayD::from_shape_vec(
172                            IxDyn(&[row.gains.len(), 1]).f(),
173                            row.weights.clone(),
174                        )
175                        .expect("weight vector should reshape to receptor x channel"),
176                    )),
177                ),
178            ]))
179            .expect("insert gain solution row");
180    }
181
182    table
183        .save(
184            TableOptions::new(&request.output_table)
185                .with_data_manager(DataManagerKind::StandardStMan),
186        )
187        .map_err(|source| GainSolveError::SaveCalibrationTable {
188            path: request.output_table.display().to_string(),
189            source: Box::new(source),
190        })?;
191
192    for (id, name) in [
193        (SubtableId::Observation, "OBSERVATION"),
194        (SubtableId::Antenna, "ANTENNA"),
195        (SubtableId::Field, "FIELD"),
196        (SubtableId::History, "HISTORY"),
197    ] {
198        ms.subtable(id)
199            .expect("required subtable available")
200            .save(TableOptions::new(request.output_table.join(name)))
201            .map_err(|source| GainSolveError::CopySubtable {
202                subtable: name.to_string(),
203                path: request.output_table.display().to_string(),
204                source: Box::new(source),
205            })?;
206    }
207    write_gain_spectral_window_subtable(ms, &request.output_table, rows)?;
208
209    let field_ids = rows.iter().map(|row| row.field_id).collect::<BTreeSet<_>>();
210    let spw_ids = rows.iter().map(|row| row.spw_id).collect::<BTreeSet<_>>();
211    Ok(GainSolveReport {
212        output_table: request.output_table.clone(),
213        gain_type: request.gain_type,
214        refant_antenna_id: refant_id,
215        field_ids: field_ids.iter().copied().collect(),
216        spectral_window_ids: spw_ids.iter().copied().collect(),
217        solution_row_count: rows.len(),
218    })
219}

221pub(crate) fn prepare_output_root(path: &Path) -> Result<(), GainSolveError> {
222    if path.exists() {
223        fs::remove_dir_all(path)
224            .or_else(|_| fs::remove_file(path))
225            .map_err(|error| GainSolveError::PrepareOutput {
226                path: path.display().to_string(),
227                reason: error.to_string(),
228            })?;
229    }
230    if let Some(parent) = path.parent() {
231        fs::create_dir_all(parent).map_err(|error| GainSolveError::PrepareOutput {
232            path: path.display().to_string(),
233            reason: error.to_string(),
234        })?;
235    }
236    Ok(())
237}

239pub(crate) fn set_fixed_unit_keyword(table: &mut Table, column: &str, units: &[&str]) {
240    let mut keywords = table.column_keywords(column).cloned().unwrap_or_default();
241    keywords.upsert(
242        "QuantumUnits",
243        Value::Array(ArrayValue::from_string_vec(
244            units.iter().map(|unit| (*unit).to_string()).collect(),
245        )),
246    );
247    table.set_column_keywords(column, keywords);
248}

250pub(crate) fn set_measinfo_keyword(
251    table: &mut Table,
252    column: &str,
253    measure_type: &str,
254    measure_ref: Option<&str>,
255) {
256    let mut keywords = table.column_keywords(column).cloned().unwrap_or_default();
257    let mut fields = vec![RecordField::new(
258        "type",
259        Value::Scalar(ScalarValue::String(measure_type.to_string())),
260    )];
261    if let Some(measure_ref) = measure_ref {
262        fields.push(RecordField::new(
263            "Ref",
264            Value::Scalar(ScalarValue::String(measure_ref.to_string())),
265        ));
266    }
267    keywords.upsert("MEASINFO", Value::Record(RecordValue::new(fields)));
268    table.set_column_keywords(column, keywords);
269}

271fn write_gain_spectral_window_subtable(
272    ms: &MeasurementSet,
273    output_root: &Path,
274    rows: &[SolutionRow],
275) -> Result<(), GainSolveError> {
276    let destination = output_root.join("SPECTRAL_WINDOW");
277    ms.subtable(SubtableId::SpectralWindow)
278        .expect("required subtable available")
279        .save(TableOptions::new(&destination))
280        .map_err(|source| GainSolveError::CopySubtable {
281            subtable: "SPECTRAL_WINDOW".to_string(),
282            path: output_root.display().to_string(),
283            source: Box::new(source),
284        })?;
285
286    let solved_spw_ids = rows.iter().map(|row| row.spw_id).collect::<BTreeSet<_>>();
287    let mut table = Table::open(TableOptions::new(&destination)).map_err(|source| {
288        GainSolveError::CopySubtable {
289            subtable: "SPECTRAL_WINDOW".to_string(),
290            path: output_root.display().to_string(),
291            source: Box::new(source),
292        }
293    })?;
294
295    for spw_id in solved_spw_ids {
296        let row =
297            usize::try_from(spw_id).map_err(|_| GainSolveError::UnsupportedParameterShape {
298                path: "SPECTRAL_WINDOW row id".to_string(),
299                shape: vec![spw_id as usize],
300            })?;
301        let chan_freq = get_f64_array(&table, row, "CHAN_FREQ")?;
302        let chan_width = get_f64_array(&table, row, "CHAN_WIDTH")?;
303        let effective_bw = get_f64_array(&table, row, "EFFECTIVE_BW")?;
304        let resolution = get_f64_array(&table, row, "RESOLUTION")?;
305
306        if chan_freq.len() <= 1 {
307            continue;
308        }
309
310        let center_index = chan_freq.len() / 2;
311        table
312            .cell_accessor_mut(row, "NUM_CHAN")
313            .and_then(|mut cell| cell.set(Value::Scalar(ScalarValue::Int32(1))))
314            .map_err(|source| GainSolveError::CopySubtable {
315                subtable: "SPECTRAL_WINDOW".to_string(),
316                path: output_root.display().to_string(),
317                source: Box::new(source),
318            })?;
319        table
320            .cell_accessor_mut(row, "CHAN_FREQ")
321            .and_then(|mut cell| cell.set(Value::Array(f64_array(&[chan_freq[center_index]]))))
322            .map_err(|source| GainSolveError::CopySubtable {
323                subtable: "SPECTRAL_WINDOW".to_string(),
324                path: output_root.display().to_string(),
325                source: Box::new(source),
326            })?;
327        table
328            .cell_accessor_mut(row, "CHAN_WIDTH")
329            .and_then(|mut cell| {
330                cell.set(Value::Array(f64_array(&[chan_width.iter().copied().sum()])))
331            })
332            .map_err(|source| GainSolveError::CopySubtable {
333                subtable: "SPECTRAL_WINDOW".to_string(),
334                path: output_root.display().to_string(),
335                source: Box::new(source),
336            })?;
337        table
338            .cell_accessor_mut(row, "EFFECTIVE_BW")
339            .and_then(|mut cell| {
340                cell.set(Value::Array(f64_array(&[effective_bw
341                    .iter()
342                    .copied()
343                    .sum()])))
344            })
345            .map_err(|source| GainSolveError::CopySubtable {
346                subtable: "SPECTRAL_WINDOW".to_string(),
347                path: output_root.display().to_string(),
348                source: Box::new(source),
349            })?;
350        table
351            .cell_accessor_mut(row, "RESOLUTION")
352            .and_then(|mut cell| {
353                cell.set(Value::Array(f64_array(&[resolution.iter().copied().sum()])))
354            })
355            .map_err(|source| GainSolveError::CopySubtable {
356                subtable: "SPECTRAL_WINDOW".to_string(),
357                path: output_root.display().to_string(),
358                source: Box::new(source),
359            })?;
360    }
361
362    table
363        .save(TableOptions::new(&destination))
364        .map_err(|source| GainSolveError::CopySubtable {
365            subtable: "SPECTRAL_WINDOW".to_string(),
366            path: output_root.display().to_string(),
367            source: Box::new(source),
368        })?;
369
370    Ok(())
371}

373fn get_f64_array(table: &Table, row: usize, column: &str) -> Result<Vec<f64>, GainSolveError> {
374    match table
375        .cell_accessor(row, column)
376        .and_then(|cell| cell.array())
377        .map_err(MsError::from)
378        .map_err(|source| GainSolveError::OpenMeasurementSet {
379            path: column.to_string(),
380            source,
381        })? {
382        ArrayValue::Float64(values) => Ok(values.iter().copied().collect()),
383        other => Err(GainSolveError::UnsupportedParameterShape {
384            path: format!("{column} array"),
385            shape: other.shape().to_vec(),
386        }),
387    }
388}
389
390fn f64_array(values: &[f64]) -> ArrayValue {
391    ArrayValue::Float64(
392        ArrayD::from_shape_vec(IxDyn(&[values.len()]).f(), values.to_vec())
393            .expect("f64 vector should reshape to 1-D array"),
394    )
395}

397pub(crate) fn subtable_keyword_value(base_path: &Path, subtable_path: &Path) -> String {
398    if let Ok(relative) = subtable_path.strip_prefix(base_path) {
399        let rel = relative.to_string_lossy();
400        return format!("././{rel}");
401    }
402    if let Some(parent) = base_path.parent()
403        && let Ok(relative) = subtable_path.strip_prefix(parent)
404    {
405        let rel = relative.to_string_lossy();
406        return format!("./{rel}");
407    }
408    if subtable_path.is_relative() {
409        let rel = subtable_path.to_string_lossy();
410        return format!("././{}", rel.trim_start_matches("./"));
411    }
412    subtable_path.to_string_lossy().to_string()
413}
