2//! Recursive descent parser for unit expressions.
3//!
4//! Parses strings like `"km/s"`, `"kg.m.s-2"`, `"Jy/beam"` into a
5//! [`UnitVal`].  The grammar matches the C++ `UnitVal::create()` parser.
6//!
7//! # Grammar
8//!
9//! ```text
10//! unit_expr = term { ('.' | '/') term }*
11//! term      = '(' unit_expr ')' [exponent]
12//!           | field [exponent]
13//! field     = known_unit                   # try full name
14//!           | prefix(1) known_unit         # 1-char prefix + rest
15//!           | prefix(2) known_unit         # 2-char prefix + rest
16//! exponent  = ['-'] digits
17//! ```
18//!
19//! The separator between terms is `.` (multiply) or `/` (divide).
20//! Terms without a separator are multiplied.
21
22use crate::quanta::error::UnitError;
23use crate::quanta::registry::global_registry;
24use crate::quanta::unit_val::UnitVal;
25
26/// Parses a unit string into a [`UnitVal`].
27///
28/// On success the result is cached in the global registry for future lookups.
29/// Returns `Err(UnitError::InvalidUnit)` if the string cannot be parsed.
30pub fn parse_unit(input: &str) -> Result<UnitVal, UnitError> {
31    let input = input.trim();
32    if input.is_empty() {
33        return Ok(UnitVal::NODIM);
34    }
35
36    // Check the registry cache first.
37    let reg = global_registry();
38    if let Some(val) = reg.lookup_unit(input) {
39        return Ok(val);
40    }
41
42    let bytes = input.as_bytes();
43    let mut pos = 0;
44    let result = parse_expr(bytes, &mut pos)?;
45
46    if pos != bytes.len() {
47        return Err(UnitError::InvalidUnit {
48            input: input.to_owned(),
49        });
50    }
51
52    // Cache the result.
53    reg.cache_put(input, result);
54    Ok(result)
55}
56
57/// Parse a full expression: sequence of terms separated by `.` or `/`.
58/// A leading `/` is treated as `1/…` (e.g. `/mol` → mol⁻¹).
59fn parse_expr(bytes: &[u8], pos: &mut usize) -> Result<UnitVal, UnitError> {
60    let mut result = if *pos < bytes.len() && bytes[*pos] == b'/' {
61        UnitVal::NODIM
62    } else {
63        parse_term(bytes, pos)?
64    };
65
66    while *pos < bytes.len() {
67        match bytes[*pos] {
68            b'.' => {
69                *pos += 1;
70                let term = parse_term(bytes, pos)?;
71                result = result * term;
72            }
73            b'/' => {
74                *pos += 1;
75                let term = parse_term(bytes, pos)?;
76                result = result / term;
77            }
78            b')' => break,
79            _ => break,
80        }
81    }
82    Ok(result)
83}

85/// Parse a single term: parenthesised expression or field, optionally with exponent.
86fn parse_term(bytes: &[u8], pos: &mut usize) -> Result<UnitVal, UnitError> {
87    if *pos >= bytes.len() {
88        return Ok(UnitVal::NODIM);
89    }
90
91    let val = if bytes[*pos] == b'(' {
92        *pos += 1;
93        let inner = parse_expr(bytes, pos)?;
94        if *pos >= bytes.len() || bytes[*pos] != b')' {
95            return Err(UnitError::InvalidUnit {
96                input: String::from_utf8_lossy(bytes).into_owned(),
97            });
98        }
99        *pos += 1;
100        inner
101    } else {
102        parse_field(bytes, pos)?
103    };
104
105    // Optional exponent.
106    let exp = parse_exponent(bytes, pos);
107    if exp != 1 { Ok(val.pow(exp)) } else { Ok(val) }
108}

110/// Parse a field: a known unit name, optionally preceded by a prefix.
111fn parse_field(bytes: &[u8], pos: &mut usize) -> Result<UnitVal, UnitError> {
112    let start = *pos;
113    let reg = global_registry();
114
115    // Collect the field token: alphanumeric + special chars.
116    let end = scan_field(bytes, *pos);
117    if end == start {
118        return Err(UnitError::InvalidUnit {
119            input: String::from_utf8_lossy(bytes).into_owned(),
120        });
121    }
122
123    let field = &bytes[start..end];
124    let field_str = std::str::from_utf8(field).unwrap_or("");
125
126    // 1) Try the full field as a unit name.
127    if let Some(val) = reg.lookup_unit(field_str) {
128        *pos = end;
129        return Ok(val);
130    }
131
132    // 2) Try 2-char prefix + rest (try longer prefix first to match "da" before "d").
133    if field.len() > 2 {
134        let prefix2 = std::str::from_utf8(&field[..2]).unwrap_or("");
135        let rest2 = std::str::from_utf8(&field[2..]).unwrap_or("");
136        if let Some(pfactor) = reg.lookup_prefix(prefix2) {
137            if let Some(uval) = reg.lookup_unit(rest2) {
138                *pos = end;
139                return Ok(UnitVal::new(pfactor * uval.factor, uval.dim));
140            }
141        }
142    }
143
144    // 3) Try 1-char prefix + rest.
145    if field.len() > 1 {
146        let prefix1 = std::str::from_utf8(&field[..1]).unwrap_or("");
147        let rest1 = std::str::from_utf8(&field[1..]).unwrap_or("");
148        if let Some(pfactor) = reg.lookup_prefix(prefix1) {
149            if let Some(uval) = reg.lookup_unit(rest1) {
150                *pos = end;
151                return Ok(UnitVal::new(pfactor * uval.factor, uval.dim));
152            }
153        }
154    }
155
156    // Not found.
157    Err(UnitError::UnknownUnit {
158        name: field_str.to_owned(),
159    })
160}

162/// Scan forward to find the end of a field token.
163///
164/// A field consists of alphabetic characters plus the special characters
165/// `'`, `"`, `_`, `$`, `%`.  Digits following alphabetic characters are
166/// included only when the resulting token (alpha+digit) is a registered unit
167/// name (e.g. `S0`, `M0`, `R0`).  Otherwise digits are parsed separately as
168/// exponents, matching the C++ `UnitVal::create()` behaviour where `"m2"` is
169/// field `"m"` with exponent `2`.
170fn scan_field(bytes: &[u8], start: usize) -> usize {
171    let mut i = start;
172
173    // Special: handle `:` sequences (`:`, `::`, `:::`)
174    if i < bytes.len() && bytes[i] == b':' {
175        while i < bytes.len() && bytes[i] == b':' {
176            i += 1;
177        }
178        return i;
179    }
180
181    while i < bytes.len() {
182        let b = bytes[i];
183        if b.is_ascii_alphabetic() || b == b'_' || b == b'\'' || b == b'"' || b == b'$' || b == b'%'
184        {
185            i += 1;
186        } else {
187            break;
188        }
189    }
190
191    // Try extending with trailing digits if the result is a known unit.
192    // This handles units like S0, M0, R0 whose names end in digits.
193    if i > start && i < bytes.len() && bytes[i].is_ascii_digit() {
194        let mut j = i;
195        while j < bytes.len() && bytes[j].is_ascii_digit() {
196            j += 1;
197        }
198        if let Ok(candidate) = std::str::from_utf8(&bytes[start..j]) {
199            let reg = crate::quanta::registry::global_registry();
200            if reg.lookup_unit(candidate).is_some() {
201                return j;
202            }
203        }
204    }
205
206    i
207}

209/// Parse an optional integer exponent (possibly negative).
210fn parse_exponent(bytes: &[u8], pos: &mut usize) -> i32 {
211    if *pos >= bytes.len() {
212        return 1;
213    }
214
215    let mut sign = 1i32;
216    let mut has_digits = false;
217    let mut value = 0i32;
218
219    // Check for negative sign.
220    if bytes[*pos] == b'-' {
221        sign = -1;
222        *pos += 1;
223        // A bare `-` without digits is not an exponent — undo.
224        if *pos >= bytes.len() || !bytes[*pos].is_ascii_digit() {
225            *pos -= 1;
226            return 1;
227        }
228    } else if bytes[*pos] == b'+' {
229        *pos += 1;
230        if *pos >= bytes.len() || !bytes[*pos].is_ascii_digit() {
231            *pos -= 1;
232            return 1;
233        }
234    }
235
236    while *pos < bytes.len() && bytes[*pos].is_ascii_digit() {
237        value = value * 10 + (bytes[*pos] - b'0') as i32;
238        has_digits = true;
239        *pos += 1;
240    }
241
242    if has_digits { sign * value } else { 1 }
243}

250    fn assert_close(a: f64, b: f64, tol: f64) {
251        let diff = (a - b).abs();
252        let scale = a.abs().max(b.abs()).max(1e-30);
253        assert!(
254            diff / scale < tol,
255            "expected {a} ≈ {b} (rel diff {:.2e})",
256            diff / scale,
257        );
258    }
