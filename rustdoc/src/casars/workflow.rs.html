92pub(crate) fn render_workflow_artifact_groups(
93    groups: &[WorkflowArtifactGroupDisplay],
94    empty_message: &str,
95) -> Vec<String> {
96    let mut lines = vec!["Products".to_string()];
97    let has_items = groups.iter().any(|group| !group.items.is_empty());
98    if !has_items {
99        lines.push(empty_message.to_string());
100        return lines;
101    }
102    for group in groups.iter().filter(|group| !group.items.is_empty()) {
103        lines.push(group.title.clone());
104        for item in &group.items {
105            lines.push(format!("  {}", item.heading));
106            lines.extend(item.detail_lines.iter().map(|line| format!("     {line}")));
107        }
108    }
109    lines
110}

130pub(crate) fn render_workflow_stage_display(entry: &WorkflowStageDisplay) -> String {
131    let mut parts = vec![entry.status.label().to_string()];
132    if let Some(revision) = entry.latest_revision {
133        parts.push(format!("r{revision}"));
134    }
135    if entry.stale_revisions > 0 {
136        parts.push(format!("stale:{}", entry.stale_revisions));
137    }
138    if entry.run_count > 0 {
139        parts.push(format!("runs:{}", entry.run_count));
140    }
141    if entry.recommended {
142        parts.push("next".to_string());
143    }
144    if entry.selected {
145        parts.push("selected".to_string());
146    }
147    let suffix = if parts.is_empty() {
148        String::new()
149    } else {
150        format!("  [{}]", parts.join(" | "))
151    };
152    format!(
153        "{:<width$} {}{}",
154        "Stage",
155        entry.label,
156        suffix,
157        width = WORKFLOW_VALUE_LABEL_WIDTH
158    )
159}

161pub(crate) fn render_workflow_overview_lines(entry: &WorkflowOverviewDisplay) -> Vec<String> {
162    let mut lines = vec!["Workflow Session".to_string()];
163    if let Some(path) = &entry.dataset_path {
164        lines.push(format!("Dataset: {path}"));
165    }
166    if let Some(stage) = &entry.recommended_stage {
167        lines.push(format!("Recommended next stage: {stage}"));
168    }
169    lines.push(format!("Selected stage: {}", entry.selected_stage));
170    lines.push(format!(
171        "Products: active={} stale={} total={}",
172        entry.active_products, entry.stale_products, entry.total_products
173    ));
174    lines.push(String::new());
175    lines.push("Workflow shell".to_string());
176    lines.push(entry.guidance.clone());
177    if !entry.latest_run_lines.is_empty() {
178        lines.push(String::new());
179        lines.push("Latest run".to_string());
180        lines.extend(
181            entry
182                .latest_run_lines
183                .iter()
184                .map(|line| format!("  {line}")),
185        );
186    }
187    lines
188}

269pub(crate) fn derive_stage_states(
270    specs: &[WorkflowStageSpec],
271    runs: &[WorkflowRunSnapshot],
272    products: &[WorkflowProductSnapshot],
273) -> Vec<WorkflowStageState> {
274    let latest_active_revision = latest_active_revisions(products);
275    let latest_run = latest_run_sequence(runs);
276    let mut states = specs
277        .iter()
278        .map(|spec| {
279            let run_count = runs.iter().filter(|run| run.stage_id == spec.id).count();
280            let latest_product = latest_product_for_stage(products, spec.id);
281            let latest_revision = latest_product.map(|product| product.revision);
282            let stale_revisions = products
283                .iter()
284                .filter(|product| {
285                    product.stage_id == spec.id && product.status == WorkflowProductStatus::Stale
286                })
287                .count();
288            let status = if latest_product
289                .is_some_and(|product| product.status == WorkflowProductStatus::Stale)
290            {
291                WorkflowStageStatus::Stale
292            } else if spec.produces_product {
293                if latest_product
294                    .is_some_and(|product| product.status == WorkflowProductStatus::Active)
295                {
296                    WorkflowStageStatus::Completed
297                } else if dependencies_satisfied(
298                    spec.depends_on,
299                    &latest_active_revision,
300                    &latest_run,
301                ) {
302                    WorkflowStageStatus::Ready
303                } else {
304                    WorkflowStageStatus::Blocked
305                }
306            } else if run_count > 0
307                && dependencies_satisfied(spec.depends_on, &latest_active_revision, &latest_run)
308            {
309                WorkflowStageStatus::Completed
310            } else if dependencies_satisfied(spec.depends_on, &latest_active_revision, &latest_run)
311            {
312                WorkflowStageStatus::Ready
313            } else {
314                WorkflowStageStatus::Blocked
315            };
316
317            WorkflowStageState {
318                id: spec.id,
319                label: spec.label,
320                status,
321                recommended: false,
322                run_count,
323                latest_revision,
324                stale_revisions,
325            }
326        })
327        .collect::<Vec<_>>();
328
329    if let Some(index) = states
330        .iter()
331        .position(|state| state.status == WorkflowStageStatus::Stale)
332        .or_else(|| {
333            states
334                .iter()
335                .position(|state| state.status == WorkflowStageStatus::Ready)
336        })
337        && let Some(state) = states.get_mut(index)
338    {
339        state.recommended = true;
340    }
341
342    states
343}

345pub(crate) fn stale_descendant_product_indices(
346    specs: &[WorkflowStageSpec],
347    products: &[WorkflowProductSnapshot],
348    changed_stage_id: &'static str,
349    changed_revision: usize,
350) -> Vec<usize> {
351    let mut descendants = BTreeSet::new();
352    let mut frontier = vec![changed_stage_id];
353    while let Some(stage_id) = frontier.pop() {
354        for spec in specs {
355            if spec.depends_on.contains(&stage_id) && descendants.insert(spec.id) {
356                frontier.push(spec.id);
357            }
358        }
359    }
360
361    products
362        .iter()
363        .enumerate()
364        .filter_map(|(index, product)| {
365            if product.status != WorkflowProductStatus::Active {
366                return None;
367            }
368            if !descendants.contains(product.stage_id) {
369                return None;
370            }
371            let dependency_revision = product
372                .dependency_revisions
373                .get(changed_stage_id)
374                .copied()
375                .unwrap_or(0);
376            (dependency_revision < changed_revision).then_some(index)
377        })
378        .collect()
379}
380
381fn latest_active_revisions(products: &[WorkflowProductSnapshot]) -> BTreeMap<&'static str, usize> {
382    let mut revisions = BTreeMap::new();
383    for product in products
384        .iter()
385        .filter(|product| product.status == WorkflowProductStatus::Active)
386    {
387        revisions
388            .entry(product.stage_id)
389            .and_modify(|current: &mut usize| *current = (*current).max(product.revision))
390            .or_insert(product.revision);
391    }
392    revisions
393}
394
395fn latest_run_sequence(runs: &[WorkflowRunSnapshot]) -> BTreeMap<&'static str, usize> {
396    let mut sequences = BTreeMap::new();
397    for run in runs {
398        sequences
399            .entry(run.stage_id)
400            .and_modify(|current: &mut usize| *current = (*current).max(run.sequence))
401            .or_insert(run.sequence);
402    }
403    sequences
404}
405
406fn latest_product_for_stage<'a>(
407    products: &'a [WorkflowProductSnapshot],
408    stage_id: &'static str,
409) -> Option<&'a WorkflowProductSnapshot> {
410    products
411        .iter()
412        .filter(|product| product.stage_id == stage_id)
413        .max_by_key(|product| product.revision)
414}
415
416fn dependencies_satisfied(
417    dependencies: &[&'static str],
418    latest_active_revision: &BTreeMap<&'static str, usize>,
419    latest_run: &BTreeMap<&'static str, usize>,
420) -> bool {
421    dependencies.iter().all(|dependency| {
422        latest_active_revision.contains_key(dependency) || latest_run.contains_key(dependency)
423    })
424}
