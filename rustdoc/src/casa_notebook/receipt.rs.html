23#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
24pub struct PythonEnvironmentIdentity {
25    pub environment_id: String,
26    pub interpreter: PathBuf,
27    pub implementation: String,
28    pub version: String,
29    #[serde(default, skip_serializing_if = "Option::is_none")]
30    pub casa_rs_version: Option<String>,
31    #[serde(default)]
32    pub packages: BTreeMap<String, String>,
33    pub fingerprint_sha256: String,
34}
35
36impl PythonEnvironmentIdentity {
37    #[must_use]
38    pub fn new(
39        environment_id: impl Into<String>,
40        interpreter: PathBuf,
41        implementation: impl Into<String>,
42        version: impl Into<String>,
43        casa_rs_version: Option<String>,
44        packages: BTreeMap<String, String>,
45    ) -> Self {
46        let mut identity = Self {
47            environment_id: environment_id.into(),
48            interpreter,
49            implementation: implementation.into(),
50            version: version.into(),
51            casa_rs_version,
52            packages,
53            fingerprint_sha256: String::new(),
54        };
55        identity.fingerprint_sha256 = identity.computed_fingerprint();
56        identity
57    }
58
59    #[must_use]
60    pub fn has_valid_fingerprint(&self) -> bool {
61        self.fingerprint_sha256 == self.computed_fingerprint()
62    }
63
64    fn computed_fingerprint(&self) -> String {
65        let mut digest = Sha256::new();
66        hash_field(&mut digest, self.environment_id.as_bytes());
67        hash_field(&mut digest, self.interpreter.as_os_str().as_encoded_bytes());
68        hash_field(&mut digest, self.implementation.as_bytes());
69        hash_field(&mut digest, self.version.as_bytes());
70        match &self.casa_rs_version {
71            Some(version) => {
72                digest.update([1]);
73                hash_field(&mut digest, version.as_bytes());
74            }
75            None => digest.update([0]),
76        }
77        for (name, version) in &self.packages {
78            hash_field(&mut digest, name.as_bytes());
79            hash_field(&mut digest, version.as_bytes());
80        }
81        format!("{:x}", digest.finalize())
82    }
83}

86pub struct PythonExecutionInput {
87    pub source: String,
88    pub source_sha256: String,
89    pub authority: PythonExecutionAuthority,
90    #[serde(default)]
91    pub input_references: Vec<PathBuf>,
92    pub environment: PythonEnvironmentIdentity,
93}
94
95impl PythonExecutionInput {
96    #[must_use]
97    pub fn new(
98        source: impl Into<String>,
99        authority: PythonExecutionAuthority,
100        input_references: Vec<PathBuf>,
101        environment: PythonEnvironmentIdentity,
102    ) -> Self {
103        let source = source.into();
104        let source_sha256 = sha256(source.as_bytes());
105        Self {
106            source,
107            source_sha256,
108            authority,
109            input_references,
110            environment,
111        }
112    }
113
114    #[must_use]
115    pub fn has_valid_source_hash(&self) -> bool {
116        self.source_sha256 == sha256(self.source.as_bytes())
117    }
118}

120#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
121#[serde(tag = "kind", content = "details", rename_all = "snake_case")]
122pub enum ExecutionInput {
123    Python(PythonExecutionInput),
124}
125
126impl ExecutionInput {
127    #[must_use]
128    pub fn validation_error(&self) -> Option<&'static str> {
129        match self {
130            Self::Python(input) if !input.has_valid_source_hash() => {
131                Some("Python source SHA-256 does not match the exact source")
132            }
133            Self::Python(input) if !input.environment.has_valid_fingerprint() => {
134                Some("Python environment fingerprint does not match its identity fields")
135            }
136            Self::Python(_) => None,
137        }
138    }
139}

141fn hash_field(digest: &mut Sha256, value: &[u8]) {
142    digest.update(u64::try_from(value.len()).unwrap_or(u64::MAX).to_le_bytes());
143    digest.update(value);
144}
145
146fn sha256(value: &[u8]) -> String {
147    format!("{:x}", Sha256::digest(value))
148}

150#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Serialize, Deserialize)]

152pub struct Timestamp(pub u64);
153
154impl Timestamp {
155    #[must_use]
156    pub fn now() -> Self {
157        let millis = SystemTime::now()
158            .duration_since(UNIX_EPOCH)
159            .unwrap_or_default()
160            .as_millis();
161        Self(u64::try_from(millis).unwrap_or(u64::MAX))
162    }
163}

277#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
278pub struct ReplayAssessment {
279    pub parameters: BTreeMap<String, toml::Value>,
280    pub warnings: Vec<String>,
281}
282
283impl ExecutionReceipt {
284    #[must_use]
285    pub fn assess_replay(
286        &self,
287        current_contract_version: u32,
288        current_resolved_parameters: &BTreeMap<String, serde_json::Value>,
289    ) -> ReplayAssessment {
290        let mut warnings = Vec::new();
291        if current_contract_version != self.provider_contract_version {
292            warnings.push(format!(
293                "provider contract changed from {} to {}; validate the typed parameter diff",
294                self.provider_contract_version, current_contract_version
295            ));
296        }
297        if current_resolved_parameters != &self.resolved_parameters {
298            warnings.push(
299                "current defaults or context resolve differently from this historical run"
300                    .to_owned(),
301            );
302        }
303        ReplayAssessment {
304            parameters: self
305                .sparse_intent
306                .as_ref()
307                .map(|intent| intent.parameters.clone())
308                .unwrap_or_default(),
309            warnings,
310        }
311    }
312}
