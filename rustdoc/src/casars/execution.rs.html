13#[derive(Debug, Clone)]
14pub(crate) struct ExecutionPlan {
15    pub command: ResolvedCommand,
16    pub arguments: Vec<OsString>,
17    pub stdin: Option<String>,
18    pub working_directory: PathBuf,
19    pub renderer: Option<String>,
20    pub file_output_path: Option<String>,
21}

23#[derive(Debug)]
24pub(crate) enum ExecutionEvent {
25    Stdout(String),
26    Stderr(String),
27    Exited(ExecutionExit),
28}
29
30#[derive(Debug, Clone, Copy)]
31pub(crate) struct ExecutionExit {
32    pub code: Option<i32>,
33    pub success: bool,
34}
35
36#[derive(Debug)]
37pub(crate) struct RunningProcess {
38    receiver: Receiver<ExecutionEvent>,
39    child: Arc<Mutex<Child>>,
40}
41
42impl RunningProcess {
43    pub(crate) fn try_recv(&self) -> Result<ExecutionEvent, mpsc::TryRecvError> {
44        self.receiver.try_recv()
45    }
46
47    pub(crate) fn cancel(&self) -> Result<(), String> {
48        self.child
49            .lock()
50            .map_err(|_| "failed to acquire child-process lock".to_string())?
51            .kill()
52            .map_err(|error| format!("terminate child process: {error}"))
53    }
54}

56pub(crate) fn spawn_process(plan: &ExecutionPlan) -> Result<RunningProcess, String> {
57    let mut command = plan.command.command();
58    command
59        .args(&plan.arguments)
60        .current_dir(&plan.working_directory)
61        .stdin(if plan.stdin.is_some() {
62            Stdio::piped()
63        } else {
64            Stdio::null()
65        })
66        .stdout(Stdio::piped())
67        .stderr(Stdio::piped());
68
69    let mut child = command
70        .spawn()
71        .map_err(|error| format!("spawn subprocess: {error}"))?;
72    if let Some(payload) = plan.stdin.as_deref() {
73        child
74            .stdin
75            .take()
76            .ok_or_else(|| "subprocess stdin was not captured".to_string())?
77            .write_all(payload.as_bytes())
78            .map_err(|error| format!("write subprocess stdin: {error}"))?;
79    }
80    let stdout = child
81        .stdout
82        .take()
83        .ok_or_else(|| "subprocess stdout was not captured".to_string())?;
84    let stderr = child
85        .stderr
86        .take()
87        .ok_or_else(|| "subprocess stderr was not captured".to_string())?;
88
89    let child = Arc::new(Mutex::new(child));
90    let (tx, rx) = mpsc::channel();
91
92    let stdout_tx = tx.clone();
93    let stdout_handle = thread::spawn(move || {
94        read_stream(stdout, move |chunk| {
95            let _ = stdout_tx.send(ExecutionEvent::Stdout(chunk));
96        })
97    });
98
99    let stderr_tx = tx.clone();
100    let stderr_handle = thread::spawn(move || {
101        read_stream(stderr, move |chunk| {
102            let _ = stderr_tx.send(ExecutionEvent::Stderr(chunk));
103        })
104    });
105
106    let child_for_wait = Arc::clone(&child);
107    thread::spawn(move || {
108        let exit_status = wait_for_child_exit(&child_for_wait);
109        let _ = stdout_handle.join();
110        let _ = stderr_handle.join();
111        let exit = match exit_status {
112            Ok(status) => exit_from_status(status),
113            Err(_error) => ExecutionExit {
114                code: None,
115                success: false,
116            },
117        };
118        let _ = tx.send(ExecutionEvent::Exited(exit));
119    });
120
121    Ok(RunningProcess {
122        receiver: rx,
123        child,
124    })
125}
126
127fn wait_for_child_exit(child: &Arc<Mutex<Child>>) -> Result<ExitStatus, String> {
128    loop {
129        let status = child
130            .lock()
131            .map_err(|_| "failed to acquire child-process lock".to_string())?
132            .try_wait()
133            .map_err(|error| format!("wait for child: {error}"))?;
134        if let Some(status) = status {
135            return Ok(status);
136        }
137        thread::sleep(Duration::from_millis(25));
138    }
139}
140
141fn read_stream<R>(stream: R, mut send: impl FnMut(String))
142where
143    R: Read,
144{
145    let mut reader = BufReader::new(stream);
146    let mut line = String::new();
147    loop {
148        line.clear();
149        match reader.read_line(&mut line) {
150            Ok(0) => break,
151            Ok(_) => send(line.clone()),
152            Err(_) => break,
153        }
154    }
155}
156
157fn exit_from_status(status: ExitStatus) -> ExecutionExit {
158    ExecutionExit {
159        code: status.code(),
160        success: status.success(),
161    }
162}

170    #[test]
171    fn spawn_process_reports_stdout_stderr_and_exit() {
172        let plan = ExecutionPlan {
173            command: ResolvedCommand::direct("sh"),
174            arguments: vec![
175                "-c".into(),
176                "printf 'hello\\n'; printf 'oops\\n' >&2; exit 3".into(),
177            ],
178            stdin: None,
179            working_directory: PathBuf::from("."),
180            renderer: None,
181            file_output_path: None,
182        };
183
184        let process = spawn_process(&plan).expect("spawn process");
185        let mut saw_stdout = false;
186        let mut saw_stderr = false;
187        let mut exit = None;
188        for _ in 0..80 {
189            match process.try_recv() {
190                Ok(ExecutionEvent::Stdout(chunk)) => saw_stdout |= chunk.contains("hello"),
191                Ok(ExecutionEvent::Stderr(chunk)) => saw_stderr |= chunk.contains("oops"),
192                Ok(ExecutionEvent::Exited(status)) => {
193                    exit = Some(status);
194                    break;
195                }
196                Err(mpsc::TryRecvError::Empty) => thread::sleep(Duration::from_millis(25)),
197                Err(error) => panic!("unexpected channel error: {error}"),
198            }
199        }
200
201        let exit = exit.expect("process exit event");
202        assert!(saw_stdout);
203        assert!(saw_stderr);
204        assert_eq!(exit.code, Some(3));
205        assert!(!exit.success);
206    }

208    #[test]
209    fn cancel_stops_running_child() {
210        let plan = ExecutionPlan {
211            command: ResolvedCommand::direct("sh"),
212            arguments: vec!["-c".into(), "sleep 5".into()],
213            stdin: None,
214            working_directory: PathBuf::from("."),
215            renderer: None,
216            file_output_path: None,
217        };
218
219        let process = spawn_process(&plan).expect("spawn process");
220        process.cancel().expect("cancel process");
221
222        let mut exit = None;
223        let deadline = Instant::now() + Duration::from_secs(10);
224        while Instant::now() < deadline {
225            match process.try_recv() {
226                Ok(ExecutionEvent::Exited(status)) => {
227                    exit = Some(status);
228                    break;
229                }
230                Ok(_) | Err(mpsc::TryRecvError::Empty) => {
231                    thread::sleep(Duration::from_millis(25));
232                }
233                Err(error) => panic!("unexpected channel error: {error}"),
234            }
235        }
236
237        assert!(!exit.expect("exit after cancel").success);
238    }

240    #[test]
241    fn spawn_process_uses_workspace_and_projected_stdin() {
242        let workspace = tempfile::tempdir().unwrap();
243        let plan = ExecutionPlan {
244            command: ResolvedCommand::direct("sh"),
245            arguments: vec![
246                "-c".into(),
247                "pwd; IFS= read -r line; printf 'stdin=%s\\n' \"$line\"".into(),
248            ],
249            stdin: Some("family-request\n".to_string()),
250            working_directory: workspace.path().to_path_buf(),
251            renderer: None,
252            file_output_path: None,
253        };
254
255        let process = spawn_process(&plan).expect("spawn process");
256        let mut stdout = String::new();
257        let mut exit = None;
258        for _ in 0..80 {
259            match process.try_recv() {
260                Ok(ExecutionEvent::Stdout(chunk)) => stdout.push_str(&chunk),
261                Ok(ExecutionEvent::Exited(status)) => {
262                    exit = Some(status);
263                    break;
264                }
265                Ok(ExecutionEvent::Stderr(_)) | Err(mpsc::TryRecvError::Empty) => {
266                    thread::sleep(Duration::from_millis(25));
267                }
268                Err(error) => panic!("unexpected channel error: {error}"),
269            }
270        }
271
272        assert!(exit.expect("process exit").success);
273        assert!(stdout.contains(&workspace.path().to_string_lossy().into_owned()));
274        assert!(stdout.contains("stdin=family-request"));
275    }
