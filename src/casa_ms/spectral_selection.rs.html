46/// Typed cube-axis value corresponding to CASA `start` / `width`.
47#[derive(Debug, Clone, PartialEq)]
48pub enum CubeAxisValue {
49    /// Channel-number selection in the source SPW.
50    Channel(i32),
51    /// Frequency-like quantity in Hz.
52    FrequencyHz {
53        /// Frequency value in Hz.
54        hz: f64,
55        /// Explicit frame carried by a CASA measure.
56        frame: Option<FrequencyRef>,
57    },
58    /// Velocity-like quantity in m/s.
59    VelocityMs {
60        /// Velocity value in m/s.
61        ///
62        /// CASA measure-valued `start` / `width` inputs such as
63        /// `me.radialvelocity(...)` and many `me.todoppler(...)` results surface
64        /// here, because the casatools layer hands them around as velocity-like
65        /// quantities in `m/s` rather than as dimensionless algebraic doppler
66        /// parameters.
67        ms: f64,
68        /// Explicit frame carried by a CASA measure.
69        frame: Option<FrequencyRef>,
70    },
71    /// Doppler value in the given convention.
72    Doppler {
73        /// Dimensionless algebraic doppler value in the given convention.
74        ///
75        /// This variant is for callers that already have the convention's
76        /// normalized scalar. It is not the usual representation of CASA
77        /// `me.todoppler(...)` records in cube-axis task inputs; those are more
78        /// commonly handled as [`VelocityMs`](Self::VelocityMs).
79        value: f64,
80        /// Doppler convention.
81        convention: DopplerRef,
82    },
83}
84
85impl CubeAxisValue {
86    /// Parse a simple CASA-style cube-axis value.
87    ///
88    /// Supported forms currently include plain integers (channel numbers),
89    /// frequency quantities such as `1.1GHz`, and velocity quantities such as
90    /// `11991.7km/s`.
91    pub fn parse(text: &str, veltype: DopplerRef) -> MsResult<Self> {
92        let text = text.trim();
93        if text.is_empty() {
94            return Err(MsError::VersionError(
95                "cube axis value was empty".to_string(),
96            ));
97        }
98        if let Ok(channel) = text.parse::<i32>() {
99            return Ok(Self::Channel(channel));
100        }
101
102        let quantity = Quantity::from_str(text).map_err(|error| {
103            MsError::VersionError(format!("parse cube axis value {text:?}: {error}"))
104        })?;
105        let hz = Unit::new("Hz").expect("built-in Hz unit must parse");
106        let velocity = Unit::new("m/s").expect("built-in velocity unit must parse");
107        let dimensionless = Unit::dimensionless();
108        if quantity.unit().conformant(&hz) {
109            return Ok(Self::FrequencyHz {
110                hz: quantity.get_value_in(&hz).map_err(|error| {
111                    MsError::VersionError(format!(
112                        "convert cube axis frequency {text:?} to Hz: {error}"
113                    ))
114                })?,
115                frame: None,
116            });
117        }
118        if quantity.unit().conformant(&velocity) {
119            return Ok(Self::VelocityMs {
120                ms: quantity.get_value_in(&velocity).map_err(|error| {
121                    MsError::VersionError(format!(
122                        "convert cube axis velocity {text:?} to m/s: {error}"
123                    ))
124                })?,
125                frame: None,
126            });
127        }
128        if quantity.unit().conformant(&dimensionless) {
129            return Ok(Self::Doppler {
130                value: quantity.value(),
131                convention: veltype,
132            });
133        }
134
135        Err(MsError::VersionError(format!(
136            "cube axis value {text:?} is neither a channel id, frequency, velocity, nor dimensionless doppler value"
137        )))
138    }
139}

213fn channel_edges_from_centers(all_source_frequencies_hz: &[f64]) -> MsResult<Vec<f64>> {
214    if all_source_frequencies_hz.is_empty() {
215        return Err(MsError::VersionError(
216            "cannot build cube channel axis from an empty SPW".to_string(),
217        ));
218    }
219    if all_source_frequencies_hz.len() == 1 {
220        let center = all_source_frequencies_hz[0];
221        return Ok(vec![center - 0.5, center + 0.5]);
222    }
223
224    let mut edges = Vec::with_capacity(all_source_frequencies_hz.len() + 1);
225    let first_step = all_source_frequencies_hz[1] - all_source_frequencies_hz[0];
226    edges.push(all_source_frequencies_hz[0] - first_step / 2.0);
227    for pair in all_source_frequencies_hz.windows(2) {
228        edges.push(0.5 * (pair[0] + pair[1]));
229    }
230    let last = all_source_frequencies_hz.len() - 1;
231    let last_step = all_source_frequencies_hz[last] - all_source_frequencies_hz[last - 1];
232    edges.push(all_source_frequencies_hz[last] + last_step / 2.0);
233    Ok(edges)
234}

236fn channel_mode_output_centers(
237    all_source_frequencies_hz: &[f64],
238    all_source_channel_widths_hz: &[f64],
239    start_channel: i32,
240    width_channels: i32,
241    nchan: usize,
242) -> MsResult<Vec<f64>> {
243    if width_channels == 0 {
244        return Err(MsError::VersionError(
245            "cube channel width must not be zero".to_string(),
246        ));
247    }
248    let nchannels = i32::try_from(all_source_frequencies_hz.len()).map_err(|_| {
249        MsError::VersionError("spectral window channel count exceeds i32".to_string())
250    })?;
251    if all_source_channel_widths_hz.len() != all_source_frequencies_hz.len() {
252        return Err(MsError::VersionError(format!(
253            "channel-mode cube axis requires matching frequency/width arrays, got {} and {}",
254            all_source_frequencies_hz.len(),
255            all_source_channel_widths_hz.len()
256        )));
257    }
258    let mut centers = Vec::with_capacity(nchan);
259    for offset in 0..nchan {
260        let offset = i32::try_from(offset)
261            .map_err(|_| MsError::VersionError("cube channel count exceeds i32".to_string()))?;
262        let (low, high) = if width_channels > 0 {
263            let low = start_channel + offset * width_channels;
264            let high = low + width_channels;
265            (low, high)
266        } else {
267            let high = start_channel + 1 + offset * width_channels;
268            let low = high + width_channels;
269            (low, high)
270        };
271        if low < 0 || high < 0 || low >= nchannels || high > nchannels || low >= high {
272            return Err(MsError::VersionError(format!(
273                "cube channel axis bin [{low}, {high}) is outside the SPW with {nchannels} channels"
274            )));
275        }
276        let selected = usize::try_from(low).expect("validated low bound")
277            ..usize::try_from(high).expect("validated high bound");
278        let low_edge_hz = selected
279            .clone()
280            .map(|index| {
281                all_source_frequencies_hz[index] - all_source_channel_widths_hz[index].abs() / 2.0
282            })
283            .reduce(f64::min)
284            .expect("validated non-empty channel range");
285        let high_edge_hz = selected
286            .map(|index| {
287                all_source_frequencies_hz[index] + all_source_channel_widths_hz[index].abs() / 2.0
288            })
289            .reduce(f64::max)
290            .expect("validated non-empty channel range");
291        centers.push(0.5 * (low_edge_hz + high_edge_hz));
292    }
293    Ok(centers)
294}

296/// Resolve a simple contiguous source-channel selection.
297pub fn resolve_contiguous_channel_selection(
298    all_frequencies_hz: &[f64],
299    start: Option<usize>,
300    count: Option<usize>,
301) -> MsResult<ResolvedChannelSelection> {
302    let start = start.unwrap_or(0);
303    if start >= all_frequencies_hz.len() {
304        return Err(MsError::VersionError(format!(
305            "channel start {start} is outside spectral window with {} channels",
306            all_frequencies_hz.len()
307        )));
308    }
309    let count = count.unwrap_or(all_frequencies_hz.len() - start);
310    if count == 0 || start + count > all_frequencies_hz.len() {
311        return Err(MsError::VersionError(format!(
312            "channel selection [{start}, {}) is outside spectral window with {} channels",
313            start + count,
314            all_frequencies_hz.len()
315        )));
316    }
317    Ok(ResolvedChannelSelection {
318        indices: (start..start + count).collect(),
319        frequencies_hz: all_frequencies_hz[start..start + count].to_vec(),
320    })
321}

323/// Resolve an explicit CASA channel selector such as `4~9;12~14` or `0~10^2`.
324pub fn resolve_channel_selector_selection(
325    all_frequencies_hz: &[f64],
326    selector: &ChannelSelection,
327) -> MsResult<ResolvedChannelSelection> {
328    let indices = selector.indices(all_frequencies_hz.len())?;
329    if indices.is_empty() {
330        return Err(MsError::VersionError(
331            "channel selector resolved to zero channels".to_string(),
332        ));
333    }
334    Ok(ResolvedChannelSelection {
335        frequencies_hz: indices
336            .iter()
337            .map(|&index| all_frequencies_hz[index])
338            .collect(),
339        indices,
340    })
341}

343impl CubeSpectralSetup {
344    /// Build a CASA-style output cube axis.
345    #[allow(clippy::too_many_arguments)]
346    pub fn for_casa_cube_axis(
347        source_freq_ref: FrequencyRef,
348        all_source_frequencies_hz: &[f64],
349        all_source_channel_widths_hz: &[f64],
350        nchan: usize,
351        axis_config: &CubeAxisConfig,
352        reference_row_time_mjd_sec: f64,
353        field_id: usize,
354        time_bounds_mjd_sec: [f64; 2],
355        derived_engine: &MsCalEngine,
356    ) -> MsResult<(Self, ResolvedChannelSelection)> {
357        let channel_mode = axis_config.start.is_none() && axis_config.width.is_none()
358            || matches!(axis_config.start, Some(CubeAxisValue::Channel(_)))
359                && axis_config.width.is_none()
360            || matches!(
361                (&axis_config.start, &axis_config.width),
362                (
363                    Some(CubeAxisValue::Channel(_)) | None,
364                    Some(CubeAxisValue::Channel(_)) | None
365                )
366            );
367        if channel_mode {
368            return Self::for_casa_cube_channel_axis(
369                source_freq_ref,
370                all_source_frequencies_hz,
371                all_source_channel_widths_hz,
372                nchan,
373                axis_config,
374                reference_row_time_mjd_sec,
375                field_id,
376                time_bounds_mjd_sec,
377                derived_engine,
378            );
379        }
380
381        if nchan == 0 {
382            return Err(MsError::VersionError(
383                "cube output channel count must be positive".to_string(),
384            ));
385        }
386
387        let output_freq_ref = effective_output_frequency_ref(source_freq_ref, axis_config)?;
388        let frame = if matches!(axis_config.specmode, CubeSpecMode::Cubedata)
389            || source_freq_ref == output_freq_ref
390        {
391            None
392        } else {
393            Some(
394                derived_engine
395                    .spectral_frame_observatory(reference_row_time_mjd_sec, field_id)
396                    .map_err(|error| {
397                        MsError::VersionError(format!(
398                            "build spectral frame for field {field_id}: {error}"
399                        ))
400                    })?,
401            )
402        };
403        let source_frequencies_in_output_frame =
404            if matches!(axis_config.specmode, CubeSpecMode::Cubedata) {
405                all_source_frequencies_hz.to_vec()
406            } else {
407                all_source_frequencies_hz
408                    .iter()
409                    .copied()
410                    .map(|frequency_hz| {
411                        convert_frequency_to_frame_with_frame(
412                            source_freq_ref,
413                            output_freq_ref,
414                            frequency_hz,
415                            frame.as_ref(),
416                        )
417                    })
418                    .collect::<MsResult<Vec<_>>>()?
419            };
420        if uses_optical_velocity_axis(axis_config) {
421            let output_channel_frequencies_hz = optical_velocity_output_centers(
422                &source_frequencies_in_output_frame,
423                all_source_channel_widths_hz,
424                nchan,
425                axis_config,
426            )?;
427            let setup = Self {
428                source_freq_ref,
429                output_freq_ref,
430                interpolation: axis_config.interpolation,
431                interpolation_uses_native_source_frequencies: uses_native_source_interpolation(
432                    axis_config,
433                ),
434                output_channel_frequencies_hz,
435            };
436            let support = cube_source_channel_support(
437                all_source_frequencies_hz,
438                all_source_channel_widths_hz,
439                &setup,
440                time_bounds_mjd_sec,
441                field_id,
442                derived_engine,
443            )?;
444            return Ok((setup, support));
445        }
446        let native_spacing_hz = infer_native_spacing_hz(&source_frequencies_in_output_frame)?;
447
448        let delta_hz = match axis_config.width.as_ref() {
449            Some(value) => cube_axis_delta_hz_from_value(
450                value,
451                axis_config.rest_frequency_hz,
452                axis_config.veltype,
453            )?,
454            None => match axis_config.start.as_ref() {
455                Some(CubeAxisValue::FrequencyHz { .. }) => native_spacing_hz.abs(),
456                Some(CubeAxisValue::VelocityMs { .. }) | Some(CubeAxisValue::Doppler { .. }) => {
457                    -native_spacing_hz.abs()
458                }
459                _ => native_spacing_hz,
460            },
461        };
462        if delta_hz == 0.0 {
463            return Err(MsError::VersionError(
464                "cube output frequency increment must not be zero".to_string(),
465            ));
466        }
467
468        let start_frequency_hz = match axis_config.start.as_ref() {
469            Some(value) => cube_axis_frequency_hz_from_value(
470                value,
471                output_freq_ref,
472                axis_config.rest_frequency_hz,
473                axis_config.veltype,
474                reference_row_time_mjd_sec,
475                field_id,
476                derived_engine,
477            )?,
478            None => {
479                default_generic_axis_start_hz(&source_frequencies_in_output_frame, delta_hz, nchan)?
480            }
481        };
482
483        let start_frequency_hz = snap_default_frequency_like_start_to_source_grid(
484            axis_config,
485            &source_frequencies_in_output_frame,
486            start_frequency_hz,
487        );
488
489        let output_channel_frequencies_hz = (0..nchan)
490            .map(|index| start_frequency_hz + index as f64 * delta_hz)
491            .collect::<Vec<_>>();
492        let setup = Self {
493            source_freq_ref,
494            output_freq_ref,
495            interpolation: axis_config.interpolation,
496            interpolation_uses_native_source_frequencies: uses_native_source_interpolation(
497                axis_config,
498            ),
499            output_channel_frequencies_hz,
500        };
501        let support = cube_source_channel_support(
502            all_source_frequencies_hz,
503            all_source_channel_widths_hz,
504            &setup,
505            time_bounds_mjd_sec,
506            field_id,
507            derived_engine,
508        )?;
509        Ok((setup, support))
510    }

512    /// Build a CASA-style output cube axis in channel mode.
513    ///
514    /// This models the source-backed `tclean` cube tests that specify
515    /// `start`/`width` as channel numbers. Output channels are defined on the
516    /// full SPW channel axis; any SPW channel selector affects which input data
517    /// are available, not the output channel locations themselves.
518    #[allow(clippy::too_many_arguments)]
519    pub fn for_casa_cube_channel_axis(
520        source_freq_ref: FrequencyRef,
521        all_source_frequencies_hz: &[f64],
522        all_source_channel_widths_hz: &[f64],
523        nchan: usize,
524        axis_config: &CubeAxisConfig,
525        reference_row_time_mjd_sec: f64,
526        field_id: usize,
527        time_bounds_mjd_sec: [f64; 2],
528        derived_engine: &MsCalEngine,
529    ) -> MsResult<(Self, ResolvedChannelSelection)> {
530        if nchan == 0 {
531            return Err(MsError::VersionError(
532                "cube output channel count must be positive".to_string(),
533            ));
534        }
535        let start_channel = match axis_config.start.as_ref() {
536            None => 0,
537            Some(CubeAxisValue::Channel(channel)) => *channel,
538            Some(other) => {
539                return Err(MsError::VersionError(format!(
540                    "channel-mode cube axis currently expects a channel-valued start, found {other:?}"
541                )));
542            }
543        };
544        let width_channels = match axis_config.width.as_ref() {
545            None => 1,
546            Some(CubeAxisValue::Channel(channel_width)) => *channel_width,
547            Some(other) => {
548                return Err(MsError::VersionError(format!(
549                    "channel-mode cube axis currently expects a channel-valued width, found {other:?}"
550                )));
551            }
552        };
553        if width_channels == 0 {
554            return Err(MsError::VersionError(
555                "cube channel width must not be zero".to_string(),
556            ));
557        }
558        let output_freq_ref = match axis_config.specmode {
559            CubeSpecMode::Cube => axis_config.outframe,
560            CubeSpecMode::Cubedata => source_freq_ref,
561        };
562        let output_channel_frequencies_hz =
563            if matches!(axis_config.specmode, CubeSpecMode::Cubedata) {
564                channel_mode_output_centers(
565                    all_source_frequencies_hz,
566                    all_source_channel_widths_hz,
567                    start_channel,
568                    width_channels,
569                    nchan,
570                )?
571            } else {
572                let frame = derived_engine
573                    .spectral_frame_observatory(reference_row_time_mjd_sec, field_id)
574                    .map_err(|error| {
575                        MsError::VersionError(format!(
576                            "build spectral frame for field {field_id}: {error}"
577                        ))
578                    })?;
579                let source_frequencies_in_output_frame = all_source_frequencies_hz
580                    .iter()
581                    .copied()
582                    .map(|frequency_hz| {
583                        convert_frequency_to_frame_with_frame(
584                            source_freq_ref,
585                            output_freq_ref,
586                            frequency_hz,
587                            Some(&frame),
588                        )
589                    })
590                    .collect::<MsResult<Vec<_>>>()?;
591                let source_channel_widths_in_output_frame =
592                    convert_channel_widths_to_frame_with_frame(
593                        source_freq_ref,
594                        output_freq_ref,
595                        all_source_frequencies_hz,
596                        all_source_channel_widths_hz,
597                        &frame,
598                    )?;
599                channel_mode_output_centers(
600                    &source_frequencies_in_output_frame,
601                    &source_channel_widths_in_output_frame,
602                    start_channel,
603                    width_channels,
604                    nchan,
605                )?
606            };
607        let setup = Self {
608            source_freq_ref,
609            output_freq_ref,
610            interpolation: axis_config.interpolation,
611            interpolation_uses_native_source_frequencies: uses_native_source_interpolation(
612                axis_config,
613            ),
614            output_channel_frequencies_hz,
615        };
616        let support = cube_source_channel_support(
617            all_source_frequencies_hz,
618            all_source_channel_widths_hz,
619            &setup,
620            time_bounds_mjd_sec,
621            field_id,
622            derived_engine,
623        )?;
624        Ok((setup, support))
625    }

627    /// Build output-channel interpolation contributions for one row, reusing
628    /// the per-row spectral frame conversion state.
629    pub fn row_output_channel_contributions_batch(
630        &self,
631        source_frequencies_hz: &[f64],
632        source_channel_widths_hz: &[f64],
633        row_time_mjd_sec: f64,
634        field_id: usize,
635        derived_engine: &MsCalEngine,
636    ) -> MsResult<Vec<Vec<CubeChannelContribution>>> {
637        if source_channel_widths_hz.len() != source_frequencies_hz.len() {
638            return Err(MsError::VersionError(format!(
639                "row cube interpolation requires matching frequency/width arrays, got {} and {}",
640                source_frequencies_hz.len(),
641                source_channel_widths_hz.len()
642            )));
643        }
644        let source_frequencies_for_interpolation =
645            if self.interpolation_uses_native_source_frequencies {
646                source_frequencies_hz.to_vec()
647            } else {
648                let spectral_frame = if self.source_freq_ref == self.output_freq_ref {
649                    None
650                } else {
651                    Some(
652                        derived_engine
653                            .spectral_frame_observatory(row_time_mjd_sec, field_id)
654                            .map_err(|error| {
655                                MsError::VersionError(format!(
656                                    "build spectral frame for field {field_id}: {error}"
657                                ))
658                            })?,
659                    )
660                };
661                source_frequencies_hz
662                    .iter()
663                    .copied()
664                    .map(|source_frequency_hz| {
665                        convert_frequency_to_frame_with_frame(
666                            self.source_freq_ref,
667                            self.output_freq_ref,
668                            source_frequency_hz,
669                            spectral_frame.as_ref(),
670                        )
671                    })
672                    .collect::<MsResult<Vec<_>>>()?
673            };
674        let source_channel_widths_for_interpolation =
675            if self.interpolation_uses_native_source_frequencies {
676                source_channel_widths_hz.to_vec()
677            } else {
678                let spectral_frame = if self.source_freq_ref == self.output_freq_ref {
679                    None
680                } else {
681                    Some(
682                        derived_engine
683                            .spectral_frame_observatory(row_time_mjd_sec, field_id)
684                            .map_err(|error| {
685                                MsError::VersionError(format!(
686                                    "build spectral frame for field {field_id}: {error}"
687                                ))
688                            })?,
689                    )
690                };
691                match spectral_frame.as_ref() {
692                    Some(frame) => convert_channel_widths_to_frame_with_frame(
693                        self.source_freq_ref,
694                        self.output_freq_ref,
695                        source_frequencies_hz,
696                        source_channel_widths_hz,
697                        frame,
698                    )?,
699                    None => source_channel_widths_hz.to_vec(),
700                }
701            };
702        Ok(self
703            .output_channel_frequencies_hz
704            .iter()
705            .copied()
706            .map(|output_frequency_hz| {
707                spectral_interpolation_contributions(
708                    source_frequencies_hz,
709                    &source_frequencies_for_interpolation,
710                    &source_channel_widths_for_interpolation,
711                    self.interpolation,
712                    output_frequency_hz,
713                )
714            })
715            .collect())
716    }
717}

744fn effective_output_frequency_ref(
745    source_freq_ref: FrequencyRef,
746    axis_config: &CubeAxisConfig,
747) -> MsResult<FrequencyRef> {
748    if matches!(axis_config.specmode, CubeSpecMode::Cubedata) {
749        return Ok(source_freq_ref);
750    }
751    let start_frame = cube_axis_explicit_frame(axis_config.start.as_ref());
752    let width_frame = cube_axis_explicit_frame(axis_config.width.as_ref());
753    match (start_frame, width_frame) {
754        (Some(left), Some(right)) if left != right => Err(MsError::VersionError(format!(
755            "cube axis start frame {left} and width frame {right} disagree"
756        ))),
757        (Some(frame), _) | (_, Some(frame)) => Ok(frame),
758        (None, None) => Ok(axis_config.outframe),
759    }
760}

762fn cube_axis_explicit_frame(axis_value: Option<&CubeAxisValue>) -> Option<FrequencyRef> {
763    match axis_value {
764        Some(CubeAxisValue::FrequencyHz { frame, .. })
765        | Some(CubeAxisValue::VelocityMs { frame, .. }) => *frame,
766        Some(CubeAxisValue::Channel(_)) | Some(CubeAxisValue::Doppler { .. }) | None => None,
767    }

782fn cube_axis_frequency_hz_from_value(
783    axis_value: &CubeAxisValue,
784    output_freq_ref: FrequencyRef,
785    rest_frequency_hz: Option<f64>,
786    veltype: DopplerRef,
787    row_time_mjd_sec: f64,
788    field_id: usize,
789    derived_engine: &MsCalEngine,
790) -> MsResult<f64> {
791    match axis_value {
792        CubeAxisValue::Channel(channel) => Err(MsError::VersionError(format!(
793            "expected frequency-like cube axis value, found channel {channel}"
794        ))),
795        CubeAxisValue::FrequencyHz { hz, frame } => {
796            let source_freq_ref = frame.unwrap_or(output_freq_ref);
797            convert_frequency_to_frame(
798                source_freq_ref,
799                output_freq_ref,
800                *hz,
801                row_time_mjd_sec,
802                field_id,
803                derived_engine,
804            )
805        }
806        CubeAxisValue::VelocityMs { ms, frame } => {
807            let rest_frequency_hz = required_rest_frequency_hz(rest_frequency_hz, axis_value)?;
808            let doppler = MDoppler::new(ms / SPEED_OF_LIGHT_M_S, veltype)
809                .convert_to(DopplerRef::RATIO, &MeasFrame::new())
810                .map_err(|error| {
811                    MsError::VersionError(format!(
812                        "convert cube-axis velocity {ms} m/s to frequency ratio: {error}"
813                    ))
814                })?;
815            let frequency_hz = rest_frequency_hz * doppler.value();
816            let source_freq_ref = frame.unwrap_or(output_freq_ref);
817            convert_frequency_to_frame(
818                source_freq_ref,
819                output_freq_ref,
820                frequency_hz,
821                row_time_mjd_sec,
822                field_id,
823                derived_engine,
824            )
825        }
826        CubeAxisValue::Doppler { value, convention } => {
827            let rest_frequency_hz = required_rest_frequency_hz(rest_frequency_hz, axis_value)?;
828            let doppler = MDoppler::new(*value, *convention)
829                .convert_to(DopplerRef::RATIO, &MeasFrame::new())
830                .map_err(|error| {
831                    MsError::VersionError(format!(
832                        "convert cube-axis doppler value {value} to frequency ratio: {error}"
833                    ))
834                })?;
835            Ok(rest_frequency_hz * doppler.value())
836        }
837    }
838}

840fn cube_axis_delta_hz_from_value(
841    axis_value: &CubeAxisValue,
842    rest_frequency_hz: Option<f64>,
843    veltype: DopplerRef,
844) -> MsResult<f64> {
845    match axis_value {
846        CubeAxisValue::Channel(channel_width) => Err(MsError::VersionError(format!(
847            "expected frequency-like cube increment, found channel width {channel_width}"
848        ))),
849        CubeAxisValue::FrequencyHz { hz, .. } => Ok(*hz),
850        CubeAxisValue::VelocityMs { ms, .. } => {
851            let rest_frequency_hz = required_rest_frequency_hz(rest_frequency_hz, axis_value)?;
852            let doppler = MDoppler::new(ms / SPEED_OF_LIGHT_M_S, veltype)
853                .convert_to(DopplerRef::RATIO, &MeasFrame::new())
854                .map_err(|error| {
855                    MsError::VersionError(format!(
856                        "convert cube-axis velocity width {ms} m/s to frequency ratio: {error}"
857                    ))
858                })?;
859            Ok(rest_frequency_hz * (doppler.value() - 1.0))
860        }
861        CubeAxisValue::Doppler { value, convention } => {
862            let rest_frequency_hz = required_rest_frequency_hz(rest_frequency_hz, axis_value)?;
863            let doppler = MDoppler::new(*value, *convention)
864                .convert_to(DopplerRef::RATIO, &MeasFrame::new())
865                .map_err(|error| {
866                    MsError::VersionError(format!(
867                        "convert cube-axis doppler width {value} to frequency ratio: {error}"
868                    ))
869                })?;
870            Ok(rest_frequency_hz * (doppler.value() - 1.0))
871        }
872    }
873}

875fn frequency_hz_from_vopt_ms(optical_velocity_ms: f64, rest_frequency_hz: f64) -> f64 {
876    rest_frequency_hz / (1.0 + optical_velocity_ms / SPEED_OF_LIGHT_M_S)
877}
878
879fn optical_velocity_ms_from_frequency_hz(frequency_hz: f64, rest_frequency_hz: f64) -> f64 {
880    SPEED_OF_LIGHT_M_S * (rest_frequency_hz / frequency_hz - 1.0)
881}
882
883fn optical_velocity_ms_from_axis_value(
884    axis_value: &CubeAxisValue,
885    rest_frequency_hz: f64,
886) -> MsResult<f64> {
887    match axis_value {
888        CubeAxisValue::VelocityMs { ms, .. } => Ok(*ms),
889        CubeAxisValue::Doppler { value, convention } => {
890            let doppler = MDoppler::new(*value, *convention)
891                .convert_to(DopplerRef::RATIO, &MeasFrame::new())
892                .map_err(|error| {
893                    MsError::VersionError(format!(
894                        "convert cube-axis optical doppler {value} to ratio: {error}"
895                    ))
896                })?;
897            Ok(optical_velocity_ms_from_frequency_hz(
898                rest_frequency_hz * doppler.value(),
899                rest_frequency_hz,
900            ))
901        }
902        other => Err(MsError::VersionError(format!(
903            "expected optical velocity-like cube axis value, found {other:?}"
904        ))),
905    }

908fn optical_velocity_output_centers(
909    source_frequencies_in_output_frame_hz: &[f64],
910    source_channel_widths_hz: &[f64],
911    nchan: usize,
912    axis_config: &CubeAxisConfig,
913) -> MsResult<Vec<f64>> {
914    if nchan == 0 {
915        return Err(MsError::VersionError(
916            "cube output channel count must be positive".to_string(),
917        ));
918    }
919    if source_frequencies_in_output_frame_hz.is_empty()
920        || source_channel_widths_hz.len() != source_frequencies_in_output_frame_hz.len()
921    {
922        return Err(MsError::VersionError(
923            "optical velocity axis requires non-empty source frequencies and matching widths"
924                .to_string(),
925        ));
926    }
927    let rest_frequency_hz =
928        required_rest_frequency_hz(axis_config.rest_frequency_hz, &axis_config.start)?;
929    let mut source_frequencies_hz = source_frequencies_in_output_frame_hz.to_vec();
930    let mut source_widths_hz = source_channel_widths_hz.to_vec();
931    let mut reverse_output = false;
932    if source_frequencies_hz.len() > 1 && source_frequencies_hz[1] < source_frequencies_hz[0] {
933        source_frequencies_hz.reverse();
934        source_widths_hz.reverse();
935        reverse_output = true;
936    }
937
938    let width_ms = match axis_config.width.as_ref() {
939        Some(value @ (CubeAxisValue::VelocityMs { .. } | CubeAxisValue::Doppler { .. })) => {
940            optical_velocity_ms_from_axis_value(value, rest_frequency_hz)?.abs()
941        }
942        Some(other) => {
943            return Err(MsError::VersionError(format!(
944                "optical velocity cube width expects a velocity-like value, found {other:?}"
945            )));
946        }
947        None => {
948            let first_center_hz = source_frequencies_hz[0];
949            let first_width_hz = source_widths_hz[0];
950            let upper_edge = optical_velocity_ms_from_frequency_hz(
951                first_center_hz - first_width_hz,
952                rest_frequency_hz,
953            );
954            let lower_edge = optical_velocity_ms_from_frequency_hz(
955                first_center_hz + first_width_hz,
956                rest_frequency_hz,
957            );
958            (upper_edge - lower_edge).abs()
959        }
960    };
961    if !(width_ms.is_finite() && width_ms > 0.0) {
962        return Err(MsError::VersionError(
963            "optical velocity cube width must be positive".to_string(),
964        ));
965    }
966
967    let start_is_end = match axis_config.width.as_ref() {
968        Some(CubeAxisValue::VelocityMs { ms, .. }) => *ms >= 0.0,
969        Some(CubeAxisValue::Doppler { value, .. }) => *value >= 0.0,
970        None => true,
971        Some(other) => {
972            return Err(MsError::VersionError(format!(
973                "optical velocity cube width expects a velocity-like value, found {other:?}"
974            )));
975        }
976    };
977
978    let center_is_start = axis_config.start.is_some();
979    let mut regrid_center_hz;
980    let regrid_center_velocity_ms;
981    if let Some(start_value) = axis_config.start.as_ref() {
982        let mut start_velocity_ms =
983            optical_velocity_ms_from_axis_value(start_value, rest_frequency_hz)?;
984        if center_is_start {
985            if start_is_end {
986                start_velocity_ms -= width_ms / 2.0;
987            } else {
988                start_velocity_ms += width_ms / 2.0;
989            }
990        }
991        regrid_center_hz = frequency_hz_from_vopt_ms(start_velocity_ms, rest_frequency_hz);
992        regrid_center_velocity_ms = start_velocity_ms;
993    } else {
994        regrid_center_hz = (source_frequencies_hz[0] - source_widths_hz[0]
995            + source_frequencies_hz[source_frequencies_hz.len() - 1]
996            + source_widths_hz[source_widths_hz.len() - 1])
997            / 2.0;
998        regrid_center_velocity_ms =
999            optical_velocity_ms_from_frequency_hz(regrid_center_hz, rest_frequency_hz);
1000    }
1001
1002    let regrid_bandwidth_hz = {
1003        let div_by_two = if center_is_start { 1.0 } else { 0.5 };
1004        let bw_edge_hz = if center_is_start && !start_is_end {
1005            frequency_hz_from_vopt_ms(
1006                regrid_center_velocity_ms - nchan as f64 * width_ms * div_by_two,
1007                rest_frequency_hz,
1008            )
1009        } else {
1010            frequency_hz_from_vopt_ms(
1011                regrid_center_velocity_ms + nchan as f64 * width_ms * div_by_two,
1012                rest_frequency_hz,
1013            )
1014        };
1015        (bw_edge_hz - regrid_center_hz).abs() / div_by_two
1016    };
1017    if center_is_start {
1018        let center_velocity_ms = if start_is_end {
1019            regrid_center_velocity_ms + nchan as f64 * width_ms / 2.0
1020        } else {
1021            regrid_center_velocity_ms - nchan as f64 * width_ms / 2.0
1022        };
1023        regrid_center_hz = frequency_hz_from_vopt_ms(center_velocity_ms, rest_frequency_hz);
1024    }
1025
1026    let channel_upper_edge_hz = frequency_hz_from_vopt_ms(
1027        regrid_center_velocity_ms - width_ms / 2.0,
1028        rest_frequency_hz,
1029    );
1030    let regrid_channel_width_hz = 2.0
1031        * (channel_upper_edge_hz
1032            - frequency_hz_from_vopt_ms(regrid_center_velocity_ms, rest_frequency_hz));
1033
1034    let lower_source_edge_hz = source_frequencies_hz[0] - source_widths_hz[0] / 2.0;
1035    let upper_source_edge_hz = source_frequencies_hz[source_frequencies_hz.len() - 1]
1036        + source_widths_hz[source_widths_hz.len() - 1] / 2.0;
1037
1038    let mut the_regrid_center_hz = regrid_center_hz;
1039    if (the_regrid_center_hz - upper_source_edge_hz) > 1.0 {
1040        the_regrid_center_hz = upper_source_edge_hz;
1041    } else if the_regrid_center_hz < lower_source_edge_hz {
1042        the_regrid_center_hz = lower_source_edge_hz;
1043    }
1044
1045    let mut the_regrid_bandwidth_hz = regrid_bandwidth_hz;
1046    let range_tolerance_hz = source_widths_hz[0];
1047    if (the_regrid_center_hz + the_regrid_bandwidth_hz / 2.0) - upper_source_edge_hz
1048        > range_tolerance_hz
1049    {
1050        the_regrid_bandwidth_hz = (upper_source_edge_hz - the_regrid_center_hz)
1051            .abs()
1052            .min((the_regrid_center_hz - lower_source_edge_hz).abs())
1053            * 2.0;
1054    }
1055    if (the_regrid_center_hz - the_regrid_bandwidth_hz / 2.0) - lower_source_edge_hz
1056        < -range_tolerance_hz
1057    {
1058        the_regrid_bandwidth_hz = (upper_source_edge_hz - the_regrid_center_hz)
1059            .abs()
1060            .min((the_regrid_center_hz - lower_source_edge_hz).abs())
1061            * 2.0;
1062    }
1063
1064    let central_channel_width_hz = regrid_channel_width_hz;
1065
1066    let mut lo_fb_up = Vec::new();
1067    let mut hi_fb_up = Vec::new();
1068    let mut lo_fb_down = Vec::new();
1069    let mut hi_fb_down = Vec::new();
1070    let edge_tolerance_hz = central_channel_width_hz.abs() * 0.01;
1071    let upper_end_hz = the_regrid_center_hz + the_regrid_bandwidth_hz / 2.0;
1072    let lower_end_hz = the_regrid_center_hz - the_regrid_bandwidth_hz / 2.0;
1073    let upper_end_velocity_ms =
1074        optical_velocity_ms_from_frequency_hz(upper_end_hz, rest_frequency_hz);
1075    let lower_end_velocity_ms =
1076        optical_velocity_ms_from_frequency_hz(lower_end_hz, rest_frequency_hz);
1077    let tnum_chan =
1078        ((the_regrid_bandwidth_hz + edge_tolerance_hz) / central_channel_width_hz).floor();
1079    if (tnum_chan as i64) % 2 != 0 {
1080        lo_fb_up.push(the_regrid_center_hz - central_channel_width_hz / 2.0);
1081        hi_fb_up.push(the_regrid_center_hz + central_channel_width_hz / 2.0);
1082        lo_fb_down.push(the_regrid_center_hz - central_channel_width_hz / 2.0);
1083        hi_fb_down.push(the_regrid_center_hz + central_channel_width_hz / 2.0);
1084    } else {
1085        lo_fb_up.push(the_regrid_center_hz);
1086        hi_fb_up.push(the_regrid_center_hz + central_channel_width_hz);
1087        lo_fb_down.push(the_regrid_center_hz);
1088        hi_fb_down.push(the_regrid_center_hz + central_channel_width_hz);
1089    }
1090
1091    let velocity_channel_width_ms =
1092        optical_velocity_ms_from_frequency_hz(lo_fb_up[0], rest_frequency_hz)
1093            - optical_velocity_ms_from_frequency_hz(hi_fb_up[0], rest_frequency_hz);
1094    let mut lower_velocity_ms =
1095        optical_velocity_ms_from_frequency_hz(*hi_fb_up.last().unwrap(), rest_frequency_hz);
1096    let mut upper_velocity_ms = lower_velocity_ms - velocity_channel_width_ms;
1097    while upper_end_velocity_ms - upper_velocity_ms < velocity_channel_width_ms / 10.0 {
1098        let upper_frequency_hz = frequency_hz_from_vopt_ms(upper_velocity_ms, rest_frequency_hz);
1099        if upper_frequency_hz <= upper_end_hz + edge_tolerance_hz {
1100            lo_fb_up.push(*hi_fb_up.last().unwrap());
1101            hi_fb_up.push(upper_frequency_hz);
1102        } else if upper_frequency_hz < upper_end_hz + edge_tolerance_hz {
1103            lo_fb_up.push(*hi_fb_up.last().unwrap());
1104            hi_fb_up.push(upper_end_hz);
1105            break;
1106        } else {
1107            break;
1108        }
1109        lower_velocity_ms =
1110            optical_velocity_ms_from_frequency_hz(*hi_fb_up.last().unwrap(), rest_frequency_hz);
1111        upper_velocity_ms = lower_velocity_ms - velocity_channel_width_ms;
1112    }
1113
1114    let mut upper_velocity_ms =
1115        optical_velocity_ms_from_frequency_hz(*lo_fb_down.last().unwrap(), rest_frequency_hz);
1116    let mut lower_velocity_ms = upper_velocity_ms + velocity_channel_width_ms;
1117    while lower_velocity_ms - lower_end_velocity_ms < velocity_channel_width_ms / 10.0 {
1118        let lower_frequency_hz = frequency_hz_from_vopt_ms(lower_velocity_ms, rest_frequency_hz);
1119        if lower_frequency_hz >= lower_end_hz - edge_tolerance_hz {
1120            hi_fb_down.push(*lo_fb_down.last().unwrap());
1121            lo_fb_down.push(lower_frequency_hz);
1122        } else if lower_frequency_hz > lower_end_hz - edge_tolerance_hz {
1123            hi_fb_down.push(*lo_fb_down.last().unwrap());
1124            lo_fb_down.push(lower_end_hz);
1125            break;
1126        } else {
1127            break;
1128        }
1129        upper_velocity_ms =
1130            optical_velocity_ms_from_frequency_hz(*lo_fb_down.last().unwrap(), rest_frequency_hz);
1131        lower_velocity_ms = upper_velocity_ms + velocity_channel_width_ms;
1132    }
1133
1134    let mut centers = Vec::with_capacity(lo_fb_down.len() + lo_fb_up.len());
1135    for index in (0..lo_fb_down.len()).rev() {
1136        centers.push((lo_fb_down[index] + hi_fb_down[index]) / 2.0);
1137    }
1138    centers.extend(
1139        lo_fb_up
1140            .iter()
1141            .zip(hi_fb_up.iter())
1142            .enumerate()
1143            .filter(|(index, _)| *index > 0)
1144            .map(|(_, (low_hz, high_hz))| (low_hz + high_hz) / 2.0),
1145    );
1146    if reverse_output {
1147        centers.reverse();
1148    }
1149    Ok(centers)
1150}

1152fn required_rest_frequency_hz(
1153    rest_frequency_hz: Option<f64>,
1154    value: &impl std::fmt::Debug,
1155) -> MsResult<f64> {
1156    rest_frequency_hz.ok_or_else(|| {

1166    nchan: usize,
1167) -> MsResult<f64> {

1180        low_edge_hz + 0.5 * delta_hz

1229fn convert_frequency_to_frame_with_frame(
1230    source_freq_ref: FrequencyRef,
1231    target_freq_ref: FrequencyRef,
1232    frequency_hz: f64,
1233    frame: Option<&MeasFrame>,
1234) -> MsResult<f64> {
1235    if source_freq_ref == target_freq_ref {
1236        return Ok(frequency_hz);

1239    MFrequency::new(frequency_hz, source_freq_ref)
1240        .convert_to(target_freq_ref, frame)
1241        .map(|frequency| frequency.hz())
1242        .map_err(|error| {
1243            MsError::VersionError(format!(
1244                "convert frequency {frequency_hz} Hz from {source_freq_ref} to {target_freq_ref}: {error}"
1245            ))
1246        })

1249fn convert_channel_widths_to_frame_with_frame(
1250    source_freq_ref: FrequencyRef,
1251    target_freq_ref: FrequencyRef,
1252    source_channel_frequencies_hz: &[f64],
1253    source_channel_widths_hz: &[f64],
1254    frame: &MeasFrame,
1255) -> MsResult<Vec<f64>> {
1256    if source_channel_frequencies_hz.len() != source_channel_widths_hz.len() {
1257        return Err(MsError::VersionError(format!(
1258            "frequency/width arrays must match for frame conversion, got {} and {}",
1259            source_channel_frequencies_hz.len(),
1260            source_channel_widths_hz.len()
1261        )));
1262    }
1263    if source_freq_ref == target_freq_ref {
1264        return Ok(source_channel_widths_hz
1265            .iter()
1266            .map(|width_hz| width_hz.abs())
1267            .collect());
1268    }
1269    source_channel_frequencies_hz
1270        .iter()
1271        .copied()
1272        .zip(source_channel_widths_hz.iter().copied())
1273        .map(|(frequency_hz, width_hz)| {
1274            let half_width_hz = width_hz.abs() / 2.0;
1275            let upper_hz = convert_frequency_to_frame_with_frame(
1276                source_freq_ref,
1277                target_freq_ref,
1278                frequency_hz + half_width_hz,
1279                Some(frame),
1280            )?;
1281            let lower_hz = convert_frequency_to_frame_with_frame(
1282                source_freq_ref,
1283                target_freq_ref,
1284                frequency_hz - half_width_hz,
1285                Some(frame),
1286            )?;
1287            Ok((upper_hz - lower_hz).abs())
1288        })
1289        .collect()
1290}

1292fn nearest_channel_index(channel_frequencies_hz: &[f64], frequency_hz: f64) -> Option<usize> {
1293    let (best_index, best_diff) = channel_frequencies_hz
1294        .iter()
1295        .enumerate()
1296        .map(|(index, channel_frequency_hz)| (index, (channel_frequency_hz - frequency_hz).abs()))
1297        .min_by(|left, right| {
1298            left.1
1299                .partial_cmp(&right.1)
1300                .unwrap_or(std::cmp::Ordering::Equal)
1301        })?;
1302    let max_allowed_diff = if channel_frequencies_hz.len() <= 1 {
1303        f64::INFINITY
1304    } else {
1305        let lower_spacing = best_index
1306            .checked_sub(1)
1307            .map(|index| (channel_frequencies_hz[best_index] - channel_frequencies_hz[index]).abs())
1308            .unwrap_or(f64::INFINITY);
1309        let upper_spacing = channel_frequencies_hz
1310            .get(best_index + 1)
1311            .map(|frequency| (frequency - channel_frequencies_hz[best_index]).abs())
1312            .unwrap_or(f64::INFINITY);
1313        0.5 * lower_spacing.min(upper_spacing)
1314    };
1315    if best_diff <= max_allowed_diff {
1316        Some(best_index)
1317    } else {
1318        None
1319    }
1320}

1350fn linear_channel_contributions(
1351    native_source_channel_frequencies_hz: &[f64],
1352    source_channel_frequencies_hz: &[f64],
1353    source_channel_widths_hz: &[f64],
1354    frequency_hz: f64,
1355) -> Vec<CubeChannelContribution> {
1356    let nchan = source_channel_frequencies_hz.len();
1357    if nchan == 0 {
1358        return Vec::new();
1359    }
1360    if native_source_channel_frequencies_hz.len() != nchan {
1361        return Vec::new();
1362    }
1363    if source_channel_widths_hz.len() != nchan {
1364        return Vec::new();
1365    }
1366    if nchan == 1 {
1367        let low_edge_hz =
1368            source_channel_frequencies_hz[0] - source_channel_widths_hz[0].abs() / 2.0;
1369        let high_edge_hz =
1370            source_channel_frequencies_hz[0] + source_channel_widths_hz[0].abs() / 2.0;
1371        if frequency_hz < low_edge_hz || frequency_hz > high_edge_hz {
1372            return Vec::new();
1373        }
1374        return Some(0)
1375            .map(|index| {
1376                vec![CubeChannelContribution {
1377                    source_channel: index,
1378                    source_frequency_hz: native_source_channel_frequencies_hz[index],
1379                    factor: 1.0,
1380                }]
1381            })
1382            .unwrap_or_default();
1383    }
1384
1385    let min_spacing_hz = source_channel_frequencies_hz
1386        .windows(2)
1387        .map(|pair| (pair[1] - pair[0]).abs())
1388        .filter(|spacing| spacing.is_finite() && *spacing > 0.0)
1389        .fold(f64::INFINITY, f64::min);
1390    let center_tolerance_hz = if min_spacing_hz.is_finite() {
1391        min_spacing_hz * 1.0e-12
1392    } else {
1393        f64::EPSILON
1394    };
1395
1396    if let Some(index) = source_channel_frequencies_hz
1397        .iter()
1398        .position(|center| (*center - frequency_hz).abs() <= center_tolerance_hz)
1399    {
1400        return vec![CubeChannelContribution {
1401            source_channel: index,
1402            source_frequency_hz: native_source_channel_frequencies_hz[index],
1403            factor: 1.0,
1404        }];
1405    }
1406
1407    let ascending = source_channel_frequencies_hz[nchan - 1] >= source_channel_frequencies_hz[0];
1408    let interval = if ascending {
1409        source_channel_frequencies_hz
1410            .windows(2)
1411            .position(|pair| pair[0] <= frequency_hz && frequency_hz <= pair[1])
1412    } else {
1413        source_channel_frequencies_hz
1414            .windows(2)
1415            .position(|pair| pair[1] <= frequency_hz && frequency_hz <= pair[0])
1416    };
1417    let Some(lower_index) = interval else {
1418        return Vec::new();
1419    };
1420
1421    let upper_index = lower_index + 1;
1422    let lower_freq = source_channel_frequencies_hz[lower_index];
1423    let upper_freq = source_channel_frequencies_hz[upper_index];
1424    let span = upper_freq - lower_freq;
1425    if !span.is_finite() || span == 0.0 {
1426        return vec![CubeChannelContribution {
1427            source_channel: lower_index,
1428            source_frequency_hz: native_source_channel_frequencies_hz[lower_index],
1429            factor: 1.0,
1430        }];
1431    }
1432
1433    let upper_factor = ((frequency_hz - lower_freq) / span) as f32;
1434    let lower_factor = 1.0 - upper_factor;
1435    let mut contributions = Vec::with_capacity(2);
1436    if lower_factor > 0.0 {
1437        contributions.push(CubeChannelContribution {
1438            source_channel: lower_index,
1439            source_frequency_hz: native_source_channel_frequencies_hz[lower_index],
1440            factor: lower_factor,
1441        });
1442    }
1443    if upper_factor > 0.0 {
1444        contributions.push(CubeChannelContribution {
1445            source_channel: upper_index,
1446            source_frequency_hz: native_source_channel_frequencies_hz[upper_index],
1447            factor: upper_factor,
1448        });
1449    }
1450    contributions
1451}

1453fn cube_source_channel_support(
1454    all_source_frequencies_hz: &[f64],
1455    all_source_channel_widths_hz: &[f64],
1456    cube_setup: &CubeSpectralSetup,
1457    time_bounds_mjd_sec: [f64; 2],
1458    field_id: usize,
1459    derived_engine: &MsCalEngine,
1460) -> MsResult<ResolvedChannelSelection> {
1461    let mut first_index = None::<usize>;
1462    let mut last_index = None::<usize>;
1463    for row_time_mjd_sec in [time_bounds_mjd_sec[0], time_bounds_mjd_sec[1]] {
1464        let contributions = cube_setup.row_output_channel_contributions_batch(
1465            all_source_frequencies_hz,
1466            all_source_channel_widths_hz,
1467            row_time_mjd_sec,
1468            field_id,
1469            derived_engine,
1470        )?;
1471        for source_index in contributions
1472            .into_iter()
1473            .flatten()
1474            .map(|contribution| contribution.source_channel)
1475        {
1476            first_index =
1477                Some(first_index.map_or(source_index, |current| current.min(source_index)));
1478            last_index = Some(last_index.map_or(source_index, |current| current.max(source_index)));
1479        }
1480    }
1481    let first_index = first_index.ok_or_else(|| {
1482        MsError::VersionError(
1483            "cube channel selection produced no supporting source channels".to_string(),
1484        )
1485    })?;
1486    let last_index = last_index.expect("first_index implies last_index");
1487    Ok(ResolvedChannelSelection {
1488        indices: (first_index..=last_index).collect(),
1489        frequencies_hz: all_source_frequencies_hz[first_index..=last_index].to_vec(),
1490    })
1491}
