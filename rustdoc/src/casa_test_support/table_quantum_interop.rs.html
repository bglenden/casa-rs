2//! Typed oracle facade for casacore table quantum metadata.
3
4#[cfg(has_casacore_cpp)]
5use crate::oracle_runtime::CasacoreOracleRuntime;
6use crate::oracle_runtime::{OracleError, oracle_operation};
7
8#[cfg(has_casacore_cpp)]
9unsafe extern "C" {
10    fn table_quantum_create_cpp(path: *const std::ffi::c_char) -> i32;
11    fn table_quantum_read_cpp(
12        path: *const std::ffi::c_char,
13        sca_fixed_out: *mut f64,
14        sca_var_out: *mut f64,
15        sca_var_units_out: *mut std::ffi::c_char,
16        unit_buf_len: i32,
17    ) -> i32;
18    fn table_quantum_verify_cpp(path: *const std::ffi::c_char, ok_out: *mut i32) -> i32;
19    fn table_quantum_bench_scalar_read_cpp(
20        path: *const std::ffi::c_char,
21        column: *const std::ffi::c_char,
22        iterations: i32,
23        elapsed_ns_out: *mut u64,
24    ) -> i32;
25    fn table_quantum_bench_array_read_cpp(
26        path: *const std::ffi::c_char,
27        column: *const std::ffi::c_char,
28        iterations: i32,
29        elapsed_ns_out: *mut u64,
30    ) -> i32;
31}
32
33macro_rules! table_quantum_operation {
34    ($capability:expr, $body:block) => {{ oracle_operation!($capability, $body) }};
35}
36
37/// Rust-facing access to casacore table quantum operations.
38pub struct TableQuantumOracle;
39
40#[cfg_attr(not(has_casacore_cpp), allow(unused_variables))]
41impl TableQuantumOracle {
42    pub fn create(path: &str) -> Result<(), OracleError> {
43        table_quantum_operation!("table_quantum.create", {
44            let path = CasacoreOracleRuntime::c_string("table path", path)?;
45            CasacoreOracleRuntime::status("table_quantum.create", unsafe {
46                table_quantum_create_cpp(path.as_ptr())
47            })
48        })
49    }
50
51    #[allow(clippy::type_complexity)]
52    pub fn read(path: &str) -> Result<(Vec<f64>, Vec<f64>, Vec<String>), OracleError> {
53        table_quantum_operation!("table_quantum.read", {
54            let path = CasacoreOracleRuntime::c_string("table path", path)?;
55            let mut fixed = [0.0; 3];
56            let mut variable = [0.0; 3];
57            const UNIT_BUFFER_LENGTH: i32 = 64;
58            let mut unit_buffer = vec![0; 3 * UNIT_BUFFER_LENGTH as usize];
59            let status = unsafe {
60                table_quantum_read_cpp(
61                    path.as_ptr(),
62                    fixed.as_mut_ptr(),
63                    variable.as_mut_ptr(),
64                    unit_buffer.as_mut_ptr().cast(),
65                    UNIT_BUFFER_LENGTH,
66                )
67            };
68            CasacoreOracleRuntime::status("table_quantum.read", status)?;
69            let units = unit_buffer
70                .chunks(UNIT_BUFFER_LENGTH as usize)
71                .map(|bytes| CasacoreOracleRuntime::output_string("table_quantum.read", bytes))
72                .collect::<Result<Vec<_>, _>>()?;
73            Ok((fixed.to_vec(), variable.to_vec(), units))
74        })
75    }
76
77    pub fn verify(path: &str) -> Result<bool, OracleError> {
78        table_quantum_operation!("table_quantum.verify", {
79            let path = CasacoreOracleRuntime::c_string("table path", path)?;
80            let mut valid = 0;
81            let status = unsafe { table_quantum_verify_cpp(path.as_ptr(), &mut valid) };
82            CasacoreOracleRuntime::status("table_quantum.verify", status)?;
83            Ok(valid == 1)
84        })
85    }
86
87    pub fn bench_scalar_read(
88        path: &str,
89        column: &str,
90        iterations: i32,
91    ) -> Result<u64, OracleError> {
92        table_quantum_operation!("table_quantum.bench_scalar_read", {
93            let path = CasacoreOracleRuntime::c_string("table path", path)?;
94            let column = CasacoreOracleRuntime::c_string("column", column)?;
95            let mut elapsed_ns = 0;
96            let status = unsafe {
97                table_quantum_bench_scalar_read_cpp(
98                    path.as_ptr(),
99                    column.as_ptr(),
100                    iterations,
101                    &mut elapsed_ns,
102                )
103            };
104            CasacoreOracleRuntime::status("table_quantum.bench_scalar_read", status)?;
105            Ok(elapsed_ns)
106        })
107    }
108
109    pub fn bench_array_read(path: &str, column: &str, iterations: i32) -> Result<u64, OracleError> {
110        table_quantum_operation!("table_quantum.bench_array_read", {
111            let path = CasacoreOracleRuntime::c_string("table path", path)?;
112            let column = CasacoreOracleRuntime::c_string("column", column)?;
113            let mut elapsed_ns = 0;
114            let status = unsafe {
115                table_quantum_bench_array_read_cpp(
116                    path.as_ptr(),
117                    column.as_ptr(),
118                    iterations,
119                    &mut elapsed_ns,
120                )
121            };
122            CasacoreOracleRuntime::status("table_quantum.bench_array_read", status)?;
123            Ok(elapsed_ns)
124        })
125    }
126}
