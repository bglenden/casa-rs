26fn run() -> Result<(), String> {
27    let args = env::args().skip(1).collect::<Vec<_>>();
28    if args.iter().any(|arg| arg == "--help" || arg == "-h") {
29        println!("{}", command_schema("flagdata").render_help());
30        return Ok(());
31    }
32    if args.iter().any(|arg| arg == "--ui-schema") {
33        println!(
34            "{}",
35            command_schema("flagdata")
36                .render_json_pretty()
37                .map_err(|error| error.to_string())?
38        );
39        return Ok(());
40    }
41    if args.iter().any(|arg| arg == "--json-schema") {
42        println!(
43            "{}",
44            serde_json::to_string_pretty(&schema_bundle("flagdata"))
45                .map_err(|error| error.to_string())?
46        );
47        return Ok(());
48    }
49    let (vis, request) = parse_args(&args)?;
50    let report = flagdata_path(vis, &request).map_err(|error| error.to_string())?;
51    println!(
52        "{}",
53        serde_json::to_string_pretty(&report).map_err(|error| error.to_string())?
54    );
55    Ok(())
56}

58fn parse_args(args: &[String]) -> Result<(PathBuf, FlagDataRequest), String> {
59    let mut vis = None;
60    let mut request = FlagDataRequest::default();
61    let mut selection = MsSelection::new();
62    let mut index = 0usize;
63    while index < args.len() {
64        match args[index].as_str() {
65            "--vis" | "--ms" => {
66                index += 1;
67                vis = Some(PathBuf::from(args.get(index).ok_or_else(usage)?));
68            }
69            "--mode" => {
70                index += 1;
71                request.mode = parse_mode(args.get(index).ok_or_else(usage)?)?;
72            }
73            "--spw" => {
74                index += 1;
75                let spw = args.get(index).ok_or_else(usage)?.clone();
76                selection = selection
77                    .spw_selector(&spw)
78                    .map_err(|error| error.to_string())?;
79                request.spw = Some(spw);
80            }
81            "--field" => {
82                index += 1;
83                selection = selection.field(
84                    &parse_numeric_id_selector(args.get(index).ok_or_else(usage)?, "field")
85                        .map_err(|error| error.to_string())?,
86                );
87            }
88            "--scan" => {
89                index += 1;
90                selection = selection.scan(
91                    &parse_numeric_id_selector(args.get(index).ok_or_else(usage)?, "scan")
92                        .map_err(|error| error.to_string())?,
93                );
94            }
95            "--antenna" => {
96                index += 1;
97                let value = args.get(index).ok_or_else(usage)?;
98                selection = apply_antenna_selection(selection, value)?;
99            }
100            "--datacolumn" => {
101                index += 1;
102                request.data_column = parse_data_column(args.get(index).ok_or_else(usage)?)?;
103            }
104            "--action" => {
105                index += 1;
106                request.action = parse_action(args.get(index).ok_or_else(usage)?)?;
107            }
108            "--clipzeros" => request.clipzeros = true,
109            "--quackinterval" => {
110                index += 1;
111                request.quackinterval =
112                    parse_f64(args.get(index).ok_or_else(usage)?, "quackinterval")?;
113            }
114            "--quackmode" => {
115                index += 1;
116                request.quackmode = parse_quackmode(args.get(index).ok_or_else(usage)?)?;
117            }
118            "--timecutoff" => {
119                index += 1;
120                request.timecutoff = parse_f64(args.get(index).ok_or_else(usage)?, "timecutoff")?;
121            }
122            "--freqcutoff" => {
123                index += 1;
124                request.freqcutoff = parse_f64(args.get(index).ok_or_else(usage)?, "freqcutoff")?;
125            }
126            "--timedev" => {
127                index += 1;
128                request.timedev = Some(parse_f64(args.get(index).ok_or_else(usage)?, "timedev")?);
129            }
130            "--freqdev" => {
131                index += 1;
132                request.freqdev = Some(parse_f64(args.get(index).ok_or_else(usage)?, "freqdev")?);
133            }
134            "--timedevscale" => {
135                index += 1;
136                request.timedevscale =
137                    parse_f64(args.get(index).ok_or_else(usage)?, "timedevscale")?;
138            }
139            "--freqdevscale" => {
140                index += 1;
141                request.freqdevscale =
142                    parse_f64(args.get(index).ok_or_else(usage)?, "freqdevscale")?;
143            }
144            "--spectralmax" => {
145                index += 1;
146                request.spectralmax = parse_f64(args.get(index).ok_or_else(usage)?, "spectralmax")?;
147            }
148            "--spectralmin" => {
149                index += 1;
150                request.spectralmin = parse_f64(args.get(index).ok_or_else(usage)?, "spectralmin")?;
151            }
152            "--extendflags" => request.extendflags = true,
153            "--no-extendflags" | "--extendflags=false" => request.extendflags = false,
154            "--extendpols" => request.extendpols = true,
155            "--growtime" => {
156                index += 1;
157                request.growtime = parse_f64(args.get(index).ok_or_else(usage)?, "growtime")?;
158            }
159            "--growfreq" => {
160                index += 1;
161                request.growfreq = parse_f64(args.get(index).ok_or_else(usage)?, "growfreq")?;
162            }
163            "--flagbackup" => request.flagbackup = true,
164            "--no-flagbackup" | "--flagbackup=false" => request.flagbackup = false,
165            other => return Err(format!("unknown argument {other:?}\n{}", usage())),
166        }
167        index += 1;
168    }
169    request.selection = selection;
170    Ok((vis.ok_or_else(usage)?, request))
171}

173fn apply_antenna_selection(selection: MsSelection, value: &str) -> Result<MsSelection, String> {
174    let parts = value
175        .split(',')
176        .map(str::trim)
177        .filter(|part| !part.is_empty())
178        .collect::<Vec<_>>();
179    if parts.iter().all(|part| part.parse::<i32>().is_ok()) {
180        Ok(selection.antenna(
181            &parts
182                .iter()
183                .map(|part| part.parse::<i32>().expect("checked int"))
184                .collect::<Vec<_>>(),
185        ))
186    } else {
187        Ok(selection.antenna_name(&parts))
188    }
189}

191fn parse_mode(value: &str) -> Result<FlagDataMode, String> {
192    match value.to_ascii_lowercase().as_str() {
193        "manual" => Ok(FlagDataMode::Manual),
194        "clip" => Ok(FlagDataMode::Clip),
195        "quack" => Ok(FlagDataMode::Quack),
196        "tfcrop" => Ok(FlagDataMode::Tfcrop),
197        "rflag" => Ok(FlagDataMode::Rflag),
198        "extend" => Ok(FlagDataMode::Extend),
199        "summary" => Ok(FlagDataMode::Summary),
200        other => Err(format!("unsupported mode {other:?}")),
201    }
202}
203
204fn parse_action(value: &str) -> Result<FlagDataAction, String> {
205    match value.to_ascii_lowercase().as_str() {
206        "flag" | "apply" => Ok(FlagDataAction::Flag),
207        "unflag" => Ok(FlagDataAction::Unflag),
208        other => Err(format!("unsupported action {other:?}")),
209    }
210}
211
212fn parse_data_column(value: &str) -> Result<FlagDataColumn, String> {
213    match value.to_ascii_lowercase().as_str() {
214        "data" => Ok(FlagDataColumn::Data),
215        "corrected" | "corrected_data" => Ok(FlagDataColumn::CorrectedData),
216        other => Err(format!("unsupported datacolumn {other:?}")),
217    }
218}
219
220fn parse_quackmode(value: &str) -> Result<QuackMode, String> {
221    match value.to_ascii_lowercase().as_str() {
222        "beg" => Ok(QuackMode::Beg),
223        "end" => Ok(QuackMode::End),
224        other => Err(format!("unsupported quackmode {other:?}")),
225    }
226}

228fn parse_f64(value: &str, label: &str) -> Result<f64, String> {
229    value
230        .parse::<f64>()
231        .map_err(|error| format!("invalid {label} {value:?}: {error}"))
232}

485fn schema_bundle(program_name: &str) -> serde_json::Value {
486    json!({
487        "protocol": {
488            "protocol_name": "casa_ms_flagdata_task",
489            "protocol_version": 1,
490            "surface_kind": "task"
491        },
492        "projections": {
493            "ui_schema": command_schema(program_name)
494        },
495        "request_schema": {
496            "type": "object",
497            "required": ["vis", "mode"],
498            "properties": {
499                "vis": {"type": "string"},
500                "mode": {"type": "string", "enum": ["manual", "clip", "quack", "tfcrop", "rflag", "extend", "summary"]},
501                "spw": {"type": "string"},
502                "field": {"type": "string"},
503                "scan": {"type": "string"},
504                "antenna": {"type": "string"},
505                "datacolumn": {"type": "string", "enum": ["data", "corrected"]},
506                "flagbackup": {"type": "boolean"}
507            }
508        },
509        "result_schema": schema_for!(casa_ms::FlagDataReport)
510    })
511}

513struct OptionConfig<'a> {

516    order: usize,
517    flags: &'a [&'a str],
518    metavar: &'a str,
519    value_kind: UiValueKind,

528    UiArgumentSchema {
529        id: config.id.to_string(),
530        label: config.label.to_string(),
531        order: config.order,
532        parser: UiArgumentParser::Option {
533            flags: config
534                .flags
535                .iter()
536                .map(|flag| (*flag).to_string())
537                .collect(),
538            metavar: config.metavar.to_string(),
539            choices: config
540                .choices
541                .iter()
542                .map(|choice| (*choice).to_string())
543                .collect(),
544        },
545        value_kind: config.value_kind,
546        required: config.required,
547        default: config.default.map(str::to_string),
548        help: config.help.to_string(),
549        group: config.group.to_string(),
550        advanced: false,
551        hidden_in_tui: false,
552    }
553}
