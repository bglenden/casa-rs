2//! Parser for IERS finals2000A.data fixed-column ASCII format.
3//!
4//! The finals2000A.data file contains one row per MJD day with Earth
5//! Orientation Parameters (EOP) in fixed-column format. This parser
6//! extracts the key parameters needed for astronomical coordinate
7//! conversions.
8//!
9//! # Format reference
10//!
11//! Columns are 1-indexed per the IERS specification:
12//!
13//! | Parameter | Columns | Unit | Notes |
14//! |-----------|---------|------|-------|
15//! | MJD       | 8–15   | days | UTC epoch |
16//! | x (PM)    | 19–27  | arcsec | Polar motion X |
17//! | y (PM)    | 38–46  | arcsec | Polar motion Y |
18//! | dUT1      | 59–68  | seconds | UT1−UTC |
19//! | LOD       | 80–86  | ms | Length of day excess |
20//! | Flag      | 17, 58 | char | I=measured, P=predicted |
21//! | dX (nut)  | 98–106 | mas | Nutation X (IAU2000A) |
22//! | dY (nut)  | 117–125| mas | Nutation Y (IAU2000A) |
23
24use super::{EopEntry, EopError};

26/// Parse a single line from finals2000A.data into an [`EopEntry`].
27///
28/// Returns `None` for blank/short lines or lines where the MJD field is empty.
29/// Returns `Err` for lines that have a valid MJD but malformed data columns.
30pub(crate) fn parse_line(line: &str) -> Result<Option<EopEntry>, EopError> {
31    // Lines shorter than 68 chars can't have dUT1 — skip them
32    if line.len() < 68 {
33        return Ok(None);
34    }
35
36    // MJD: columns 8–15 (0-indexed: 7..15)
37    let mjd_str = line.get(7..15).unwrap_or("").trim();
38    if mjd_str.is_empty() {
39        return Ok(None);
40    }
41    let mjd: f64 = mjd_str
42        .parse()
43        .map_err(|_| EopError::ParseError(format!("bad MJD: {mjd_str:?}")))?;
44
45    // PM flag: column 17 (0-indexed: 16)
46    let pm_flag = line.as_bytes().get(16).copied().unwrap_or(b' ');
47
48    // Polar motion X: columns 19–27 (0-indexed: 18..27)
49    let x_arcsec = parse_field(line, 18, 27).unwrap_or(0.0);
50
51    // Polar motion Y: columns 38–46 (0-indexed: 37..46)
52    let y_arcsec = parse_field(line, 37, 46).unwrap_or(0.0);
53
54    // dUT1 flag: column 58 (0-indexed: 57)
55    let dut1_flag = line.as_bytes().get(57).copied().unwrap_or(b' ');
56
57    // dUT1: columns 59–68 (0-indexed: 58..68)
58    let dut1_seconds = match parse_field(line, 58, 68) {
59        Some(v) => v,
60        None => return Ok(None), // No dUT1 means this row isn't useful
61    };
62
63    // LOD: columns 80–86 (0-indexed: 79..86) — in milliseconds
64    let lod_ms = parse_field(line, 79, 86).unwrap_or(0.0);
65    let lod_seconds = lod_ms / 1000.0;
66
67    // dX: columns 98–106 (0-indexed: 97..106) — in mas
68    let dx_mas = if line.len() >= 106 {
69        parse_field(line, 97, 106).unwrap_or(0.0)
70    } else {
71        0.0
72    };
73
74    // dY: columns 117–125 (0-indexed: 116..125) — in mas
75    let dy_mas = if line.len() >= 125 {
76        parse_field(line, 116, 125).unwrap_or(0.0)
77    } else {
78        0.0
79    };
80
81    // Use dUT1 flag primarily; fall back to PM flag
82    let is_predicted = dut1_flag == b'P' || (dut1_flag == b' ' && pm_flag == b'P');
83
84    Ok(Some(EopEntry {
85        mjd,
86        x_arcsec,
87        y_arcsec,
88        dut1_seconds,
89        lod_seconds,
90        dx_mas,
91        dy_mas,
92        is_predicted,
93    }))
94}

96/// Parse a floating-point field from a fixed-column substring.
97fn parse_field(line: &str, start: usize, end: usize) -> Option<f64> {
98    let s = line.get(start..end)?.trim();
99    if s.is_empty() {
100        return None;
101    }
102    s.parse().ok()
103}

105/// Parse all lines from finals2000A.data content into a vector of entries.
106///
107/// Skips blank/short lines and lines without valid data.
108pub(crate) fn parse_finals2000a(content: &str) -> Result<Vec<EopEntry>, EopError> {
109    let mut entries = Vec::with_capacity(15000);
110
111    for (line_num, line) in content.lines().enumerate() {
112        match parse_line(line) {
113            Ok(Some(entry)) => entries.push(entry),
114            Ok(None) => {} // skip blank/incomplete lines
115            Err(e) => {
116                return Err(EopError::ParseError(format!("line {}: {e}", line_num + 1)));
117            }
118        }
119    }
120
121    if entries.is_empty() {
122        return Err(EopError::ParseError(
123            "no valid EOP entries found".to_string(),
124        ));
125    }
126
127    // Sort by MJD (should already be sorted, but be safe)
128    entries.sort_by(|a, b| a.mjd.partial_cmp(&b.mjd).unwrap());
129
130    Ok(entries)
131}

133#[cfg(test)]
134mod tests {
135    use super::*;
136
137    // Real line from finals2000A.data
138    const SAMPLE_LINE: &str = "92 1 1 48622.00 I  0.182987 0.000672  0.168775 0.000345  I-0.1251659 0.0000207  1.8335 0.0201  I    -0.086    0.202     0.130    0.165   .182400   .167900  -.1253000     0.129    -0.653  ";
139
140    #[test]
141    fn parse_sample_line() {
142        let entry = parse_line(SAMPLE_LINE).unwrap().unwrap();
143        assert!((entry.mjd - 48622.0).abs() < 1e-10);
144        assert!((entry.x_arcsec - 0.182987).abs() < 1e-7);
145        assert!((entry.y_arcsec - 0.168775).abs() < 1e-7);
146        assert!((entry.dut1_seconds - (-0.1251659)).abs() < 1e-8);
147        assert!((entry.lod_seconds - 0.0018335).abs() < 1e-7);
148        assert!((entry.dx_mas - (-0.086)).abs() < 1e-4);
149        assert!((entry.dy_mas - 0.130).abs() < 1e-4);
150        assert!(!entry.is_predicted);
151    }
152
153    #[test]
154    fn parse_short_line_returns_none() {
155        assert!(parse_line("short").unwrap().is_none());
156        assert!(parse_line("").unwrap().is_none());
157    }
158
159    #[test]
160    fn parse_predicted_line() {
161        // Create a line with P flag at column 58 (0-indexed 57)
162        let mut line = SAMPLE_LINE.to_string();
163        // Replace char at position 57 (dUT1 flag)
164        let bytes = unsafe { line.as_bytes_mut() };
165        bytes[57] = b'P';
166        let entry = parse_line(&line).unwrap().unwrap();
167        assert!(entry.is_predicted);
168    }
