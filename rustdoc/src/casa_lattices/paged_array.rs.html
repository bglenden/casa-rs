23const COLUMN_NAME: &str = "PagedArray";

25/// A disk-backed N-dimensional lattice using tiled table storage.
26///
27/// Corresponds to the C++ `PagedArray<T>` class. Data is stored in a
28/// one-row casacore table with a single array column named `"PagedArray"`,
29/// using `TiledCellStMan` for tiled I/O.
30///
31/// # On-disk format
32///
33/// The table has:
34/// - One column named `"PagedArray"` with fixed array shape
35/// - One row containing the entire lattice as a single array cell
36/// - `TiledCellStMan` storage manager with the specified tile shape
37///
38/// This format is fully compatible with C++ casacore's `PagedArray`.
39///
40/// # Examples
41///
42/// ```rust,no_run
43/// use casa_lattices::{PagedArray, TiledShape, Lattice, LatticeMut};
44///
45/// // Create a new 64x64 PagedArray on disk:
46/// let ts = TiledShape::new(vec![64, 64], std::mem::size_of::<f64>()).unwrap();
47/// let mut pa = PagedArray::<f64>::create(ts, "/tmp/test_paged.table").unwrap();
48/// pa.set(1.0).unwrap();
49///
50/// // Reopen:
51/// let pa2 = PagedArray::<f64>::open("/tmp/test_paged.table").unwrap();
52/// assert_eq!(pa2.shape(), &[64, 64]);
53/// ```
54pub struct PagedArray<T: LatticeElement> {
55    /// Interior mutability for the table handle, enabling auto-reopen
56    /// from `&self` read methods after `temp_close()`.
57    table: RefCell<Option<Table>>,
58    /// Direct tile-level access to the underlying `TiledCellStMan` payload.
59    ///
60    /// This mirrors C++ `PagedArray`'s direct slice access through the tiled
61    /// array accessor instead of materializing the entire lattice for each
62    /// unit-stride slice operation.
63    tiled_io: RefCell<Option<TiledArrayStorage>>,
64    shape: Vec<usize>,
65    tile_shape: Vec<usize>,
66    max_cache_bytes: Cell<usize>,
67    path: Option<PathBuf>,
68    _phantom: std::marker::PhantomData<T>,
69}
70
71struct PagedArrayTraversalCacheScope<'a, T: LatticeElement> {
72    array: &'a PagedArray<T>,
73    previous_cache_bytes: usize,
74}

127    fn refresh_tiled_io(&mut self) -> Result<(), LatticeError> {
128        if self.path.is_none() || !Self::supports_tiled_io() {
129            return Ok(());
130        }
131
132        if let Some(ref mut tio) = *self.tiled_io.get_mut() {
133            tio.flush().map_err(tiled_io_err)?;
134        } else if self.table.get_mut().is_some() {
135            self.flush()?;
136        }
137        let path = self.path.as_ref().expect("persistent path");
138        let tiled_io = Self::open_tiled_io(path, self.max_cache_bytes.get())?;
139        *self.tiled_io.get_mut() = Some(tiled_io);
140        Ok(())
141    }

143    fn refresh_tiled_io_shared(&self) -> Result<(), LatticeError> {
144        if self.path.is_none() || !Self::supports_tiled_io() {
145            return Ok(());
146        }
147
148        if let Some(ref mut tio) = *self.tiled_io.borrow_mut() {
149            tio.flush().map_err(tiled_io_err)?;
150        } else if self.table.borrow().is_some() {
151            self.flush()?;
152        }
153        let path = self.path.as_ref().expect("persistent path");
154        let tiled_io = Self::open_tiled_io(path, self.max_cache_bytes.get())?;
155        *self.tiled_io.borrow_mut() = Some(tiled_io);
156        Ok(())
157    }
158
159    fn set_cache_bytes_shared(&self, max_cache_bytes: usize) -> Result<(), LatticeError> {
160        self.max_cache_bytes.set(max_cache_bytes);
161        self.refresh_tiled_io_shared()
162    }

283    fn tiled_put_slice_typed<U: 'static + Clone + TilePixel>(
284        tio: &mut TiledArrayStorage,
285        data: &ArrayD<T>,
286        start: &[usize],
287    ) -> Result<(), LatticeError> {
288        let typed = (data as &dyn Any)
289            .downcast_ref::<ArrayD<U>>()
290            .ok_or_else(|| {
291                LatticeError::Table("internal PagedArray tiled type mismatch".to_string())
292            })?;
293        if let Some(slice) = typed.as_slice() {
294            tio.put_slice_c_order::<U>(slice, start, typed.shape())
295                .map_err(tiled_io_err)?;
296            return Ok(());
297        }
298        if let Some(slice) = typed.as_slice_memory_order() {
299            tio.put_slice_fortran::<U>(slice, start, typed.shape())
300                .map_err(tiled_io_err)?;
301            return Ok(());
302        }
303        let contiguous = typed.as_standard_layout();
304        tio.put_slice_c_order::<U>(
305            contiguous.as_slice().expect("contiguous C-order data"),
306            start,
307            typed.shape(),
308        )
309        .map_err(tiled_io_err)
310    }

312    /// Creates a new `PagedArray` at the given path with the specified shape.
313    ///
314    /// The table is created immediately on disk. The array is initialized
315    /// with zero/default values.
316    pub fn create(tiled_shape: TiledShape, path: impl AsRef<Path>) -> Result<Self, LatticeError> {
317        let path = path.as_ref();
318        let shape = tiled_shape.shape().to_vec();
319        let tile_shape = tiled_shape.tile_shape().to_vec();
320        let ndim = shape.len();
321
322        let schema = TableSchema::new(vec![ColumnSchema::array_variable(
323            COLUMN_NAME,
324            T::PRIMITIVE_TYPE,
325            Some(ndim),
326        )])
327        .map_err(|e| LatticeError::Table(e.to_string()))?;
328
329        let mut table = Table::with_schema(schema);
330
331        // The row is only a schema placeholder. The typed tiled seam below
332        // installs the real cube shape without allocating the full lattice.
333        let data = ArrayD::from_elem(IxDyn(&tile_shape), T::default_value());
334        let array_value = value_bridge::to_array_value(&data);
335        let row = RecordValue::new(vec![RecordField::new(
336            COLUMN_NAME,
337            Value::Array(array_value),
338        )]);
339        table.add_row(row).map_err(table_err)?;
340
341        table
342            .save(
343                TableOptions::new(path)
344                    .with_data_manager(DataManagerKind::TiledCellStMan)
345                    .with_tile_shape(if ndim > 0 {
346                        tile_shape.clone()
347                    } else {
348                        Vec::new()
349                    }),
350            )
351            .map_err(table_err)?;
352
353        let tiled_io = if Self::supports_tiled_io() {
354            Some(
355                TiledArrayStorage::create_for_pixel_type(
356                    path,
357                    &shape,
358                    &tile_shape,
359                    T::PRIMITIVE_TYPE,
360                    cfg!(target_endian = "big"),
361                    0,
362                    COLUMN_NAME,
363                    DEFAULT_TILED_ARRAY_CACHE_BYTES,
364                )
365                .map_err(tiled_io_err)?,
366            )
367        } else {
368            return Err(LatticeError::Table(format!(
369                "PagedArray does not support tiled storage for {:?}",
370                T::PRIMITIVE_TYPE
371            )));
372        };
373
374        Ok(Self {
375            table: RefCell::new(Some(table)),
376            tiled_io: RefCell::new(tiled_io),
377            shape,
378            tile_shape,
379            max_cache_bytes: Cell::new(DEFAULT_TILED_ARRAY_CACHE_BYTES),
380            path: Some(path.to_path_buf()),
381            _phantom: std::marker::PhantomData,
382        })
383    }

385    /// Creates a scratch (temporary) `PagedArray` that is not persisted.
386    ///
387    /// The data lives in a memory-backed table. Equivalent to C++
388    /// `PagedArray(shape)` with no path argument.
389    pub fn new_scratch(tiled_shape: TiledShape) -> Result<Self, LatticeError> {
390        let shape = tiled_shape.shape().to_vec();
391        let tile_shape = tiled_shape.tile_shape().to_vec();
392
393        let schema = TableSchema::new(vec![ColumnSchema::array_fixed(
394            COLUMN_NAME,
395            T::PRIMITIVE_TYPE,
396            shape.clone(),
397        )])
398        .map_err(|e| LatticeError::Table(e.to_string()))?;
399
400        let mut table = Table::with_schema_memory(schema);
401
402        let data = ArrayD::from_elem(IxDyn(&shape), T::default_value());
403        let array_value = value_bridge::to_array_value(&data);
404        let row = RecordValue::new(vec![RecordField::new(
405            COLUMN_NAME,
406            Value::Array(array_value),
407        )]);
408        table.add_row(row).map_err(table_err)?;
409
410        Ok(Self {
411            table: RefCell::new(Some(table)),
412            tiled_io: RefCell::new(None),
413            shape,
414            tile_shape,
415            max_cache_bytes: Cell::new(0),
416            path: None,
417            _phantom: std::marker::PhantomData,
418        })
419    }

429    /// Opens an existing `PagedArray` from disk with an explicit cache size.
430    ///
431    /// A `max_cache_bytes` value of `0` disables tile retention. Use [`Self::open`]
432    /// for the repository's fixed 64 MiB default.
433    pub fn open_with_cache(
434        path: impl AsRef<Path>,
435        max_cache_bytes: usize,
436    ) -> Result<Self, LatticeError> {
437        let path = path.as_ref();
438        let tiled_io = if Self::supports_tiled_io() {
439            Some(Self::open_tiled_io(path, max_cache_bytes)?)
440        } else {
441            return Err(LatticeError::Table(format!(
442                "PagedArray does not support tiled storage for {:?}",
443                T::PRIMITIVE_TYPE
444            )));
445        };
446        let tio = tiled_io.as_ref().expect("supported tiled PagedArray");
447        let shape = tio.cube_shape().to_vec();
448        let tile_shape = tio.tile_shape().to_vec();
449        let table = Table::open_metadata_only(TableOptions::new(path)).map_err(table_err)?;
450
451        Ok(Self {
452            table: RefCell::new(Some(table)),
453            tiled_io: RefCell::new(tiled_io),
454            shape,
455            tile_shape,
456            max_cache_bytes: Cell::new(max_cache_bytes),
457            path: Some(path.to_path_buf()),
458            _phantom: std::marker::PhantomData,
459        })
460    }

472    /// Returns the configured maximum tile-cache size in pixels.
473    ///
474    /// Persistent arrays always report their effective bounded cache.
475    pub fn maximum_cache_size_pixels(&self) -> usize {
476        let elem_size = Self::element_size_bytes();
477        let max_cache_bytes = self.max_cache_bytes.get();
478        if elem_size == 0 || max_cache_bytes == 0 {
479            0
480        } else {
481            max_cache_bytes / elem_size
482        }
483    }
484
485    /// Sets the maximum tile-cache size in pixels.
486    ///
487    /// A value of `0` disables tile retention.
488    ///
489    /// Mirrors C++ `PagedArray::setMaximumCacheSize`.
490    pub fn set_maximum_cache_size_pixels(
491        &mut self,
492        how_many_pixels: usize,
493    ) -> Result<(), LatticeError> {
494        let elem_size = Self::element_size_bytes();
495        let max_cache_bytes = if elem_size == 0 {
496            0
497        } else {
498            how_many_pixels.saturating_mul(elem_size)
499        };
500        self.max_cache_bytes.set(max_cache_bytes);
501        self.refresh_tiled_io()
502    }
503
504    /// Sets the cache size to hold approximately `how_many_tiles` tiles.
505    ///
506    /// A value of `0` disables tile retention.
507    ///
508    /// Mirrors C++ `PagedArray::setCacheSizeInTiles`.
509    pub fn set_cache_size_in_tiles(&mut self, how_many_tiles: usize) -> Result<(), LatticeError> {
510        if how_many_tiles == 0 {
511            return self.set_maximum_cache_size_pixels(0);
512        }
513        let pixels = self.tile_pixels().saturating_mul(how_many_tiles);
514        self.set_maximum_cache_size_pixels(pixels)
515    }

517    /// Returns `true` if the array has been temp-closed.
518    pub fn is_temp_closed(&self) -> bool {
519        self.table.borrow().is_none()
520    }

522    /// Releases the in-memory table, flushing to disk first if persistent.
523    ///
524    /// For scratch (no-path) arrays this is a no-op. After `temp_close()`,
525    /// subsequent reads/writes will auto-reopen from disk.
526    pub fn temp_close(&mut self) -> Result<(), LatticeError> {
527        if self.path.is_none() {
528            return Ok(());
529        }
530        self.flush()?;
531        *self.table.get_mut() = None;
532        *self.tiled_io.get_mut() = None;
533        Ok(())
534    }
535
536    /// Explicitly reopens a temp-closed array from disk.
537    ///
538    /// No-op if the table is already open or if this is a scratch array.
539    pub fn reopen(&mut self) -> Result<(), LatticeError> {
540        if (self.table.get_mut().is_some() && self.tiled_io.get_mut().is_some())
541            || self.path.is_none()
542        {
543            return Ok(());
544        }
545        self.auto_reopen()
546    }
547
548    /// Ensures the table is open, reopening from disk if needed.
549    ///
550    /// Uses interior mutability so that `&self` read methods can
551    /// transparently reopen after `temp_close()`.
552    fn auto_reopen(&self) -> Result<(), LatticeError> {
553        let needs_tiled_io = self.path.is_some() && Self::supports_tiled_io();
554        if self.table.borrow().is_none() || (needs_tiled_io && self.tiled_io.borrow().is_none()) {
555            let Some(path) = &self.path else {
556                return Err(LatticeError::Table(
557                    "cannot reopen scratch PagedArray after temp_close".to_string(),
558                ));
559            };
560            if self.table.borrow().is_none() {
561                let table = Table::open(TableOptions::new(path)).map_err(table_err)?;
562                *self.table.borrow_mut() = Some(table);
563            }
564            if needs_tiled_io && self.tiled_io.borrow().is_none() {
565                let tiled_io = Self::open_tiled_io(path, self.max_cache_bytes.get())?;
566                *self.tiled_io.borrow_mut() = Some(tiled_io);
567            }
568        }
569        Ok(())
570    }

572    fn auto_reopen_tiled_io(&self) -> Result<(), LatticeError> {
573        if self.path.is_none() || !Self::supports_tiled_io() {
574            return Ok(());
575        }
576        if self.tiled_io.borrow().is_none() {
577            let path = self.path.as_ref().expect("persistent path");
578            let tiled_io = Self::open_tiled_io(path, self.max_cache_bytes.get())?;
579            *self.tiled_io.borrow_mut() = Some(tiled_io);
580        }
581        Ok(())
582    }
583
584    fn save_options(&self, path: &Path) -> TableOptions {
585        TableOptions::new(path)
586            .with_data_manager(DataManagerKind::TiledCellStMan)
587            .with_tile_shape(self.tile_shape.clone())
588    }
589
590    /// Flushes changes to disk (for persistent arrays).
591    pub fn flush(&self) -> Result<(), LatticeError> {
592        if let Some(path) = &self.path {
593            self.auto_reopen()?;
594            if let Some(ref mut tio) = *self.tiled_io.borrow_mut() {
595                tio.flush().map_err(tiled_io_err)?;
596            } else if self.table.borrow().is_some() {
597                self.table
598                    .borrow()
599                    .as_ref()
600                    .unwrap()
601                    .save(self.save_options(path))
602                    .map_err(table_err)?;
603            }
604        }
605        Ok(())
606    }
607
608    /// Reads the full array, auto-reopening if temp-closed.
609    fn read_full_array(&self) -> Result<ArrayD<T>, LatticeError> {
610        self.auto_reopen()?;
611        let table_ref = self.table.borrow();
612        let table = table_ref.as_ref().unwrap();
613        let cell = table
614            .cell_accessor(0, COLUMN_NAME)
615            .map_err(table_err)?
616            .value()
617            .map_err(table_err)?
618            .ok_or_else(|| LatticeError::Table("PagedArray cell not found".to_string()))?;
619        match cell {
620            Value::Array(av) => value_bridge::from_array_value(av.clone()),
621            _ => Err(LatticeError::Table("unexpected scalar cell".to_string())),
622        }
623    }
624
625    /// Writes the full array to the table cell.
626    fn write_full_array(&mut self, data: &ArrayD<T>) -> Result<(), LatticeError> {
627        self.auto_reopen()?;
628        let array_value = value_bridge::to_array_value(data);
629        self.table
630            .get_mut()
631            .as_mut()
632            .unwrap()
633            .cell_accessor_mut(0, COLUMN_NAME)
634            .map_err(table_err)?
635            .set(Value::Array(array_value))
636            .map_err(table_err)
637    }
638}

640impl<T: LatticeElement> Lattice<T> for PagedArray<T> {
641    fn shape(&self) -> &[usize] {
642        &self.shape
643    }
644
645    fn is_persistent(&self) -> bool {
646        self.path.is_some()
647    }
648
649    fn is_paged(&self) -> bool {
650        true
651    }
652
653    fn get_at(&self, position: &[usize]) -> Result<T, LatticeError> {
654        if position.len() != self.shape.len() {
655            return Err(LatticeError::NdimMismatch {
656                expected: self.shape.len(),
657                got: position.len(),
658            });
659        }
660        for (&p, &s) in position.iter().zip(self.shape.iter()) {
661            if p >= s {
662                return Err(LatticeError::IndexOutOfBounds {
663                    index: position.to_vec(),
664                    shape: self.shape.clone(),
665                });
666            }
667        }
668
669        if self.path.is_some() {
670            self.auto_reopen_tiled_io()?;
671            if let Some(ref mut tio) = *self.tiled_io.borrow_mut() {
672                let ones = vec![1; self.shape.len()];
673                let arr = Self::tiled_get_slice(tio, position, &ones)?;
674                return arr
675                    .into_iter()
676                    .next()
677                    .ok_or_else(|| LatticeError::Table("empty tiled slice".to_string()));
678            }
679        }
680
681        let arr = self.read_full_array()?;
682        Ok(arr[IxDyn(position)].clone())
683    }
684
685    fn get_slice(
686        &self,
687        start: &[usize],
688        shape: &[usize],
689        stride: &[usize],
690    ) -> Result<ArrayD<T>, LatticeError> {
691        let ndim = self.shape.len();
692        if start.len() != ndim || shape.len() != ndim || stride.len() != ndim {
693            return Err(LatticeError::NdimMismatch {
694                expected: ndim,
695                got: start.len(),
696            });
697        }
698
699        let is_unit_stride = stride.iter().all(|&s| s == 1);
700        if self.path.is_some() {
701            self.auto_reopen_tiled_io()?;
702            if let Some(ref mut tio) = *self.tiled_io.borrow_mut() {
703                if is_unit_stride {
704                    return Self::tiled_get_slice(tio, start, shape);
705                }
706                let bounding_shape = shape
707                    .iter()
708                    .zip(stride.iter())
709                    .map(|(&count, &step)| {
710                        if step == 0 {
711                            return Err(LatticeError::InvalidTraversal(
712                                "slice stride must be positive".into(),
713                            ));
714                        }
715                        count
716                            .saturating_sub(1)
717                            .checked_mul(step)
718                            .and_then(|extent| extent.checked_add(usize::from(count > 0)))
719                            .ok_or_else(|| {
720                                LatticeError::InvalidTraversal(
721                                    "strided slice extent overflow".into(),
722                                )
723                            })
724                    })
725                    .collect::<Result<Vec<_>, _>>()?;
726                let bounding = Self::tiled_get_slice(tio, start, &bounding_shape)?;
727                let slice_info: Vec<ndarray::SliceInfoElem> = bounding_shape
728                    .iter()
729                    .zip(stride.iter())
730                    .map(|(&extent, &step)| ndarray::SliceInfoElem::Slice {
731                        start: 0,
732                        end: Some(extent as isize),
733                        step: step as isize,
734                    })
735                    .collect();
736                return Ok(bounding.slice(slice_info.as_slice()).to_owned());
737            }
738        }
739
740        let arr = self.read_full_array()?;
741
742        let slice_info: Vec<ndarray::SliceInfoElem> = start
743            .iter()
744            .zip(shape.iter())
745            .zip(stride.iter())
746            .map(|((&s, &n), &st)| {
747                let end = if n == 0 { s } else { s + n * st };
748                ndarray::SliceInfoElem::Slice {
749                    start: s as isize,
750                    end: Some(end as isize),
751                    step: st as isize,
752                }
753            })
754            .collect();
755
756        let view = arr.slice(slice_info.as_slice());
757        Ok(view.to_owned())
758    }
759
760    fn get(&self) -> Result<ArrayD<T>, LatticeError> {
761        if self.path.is_some() {
762            self.auto_reopen_tiled_io()?;
763            if let Some(ref mut tio) = *self.tiled_io.borrow_mut() {
764                return Self::tiled_get_all(tio);
765            }
766        }
767        self.read_full_array()
768    }
769
770    fn advised_max_pixels(&self) -> usize {
771        self.tile_shape.iter().product()
772    }
773
774    fn nice_cursor_shape(&self) -> Vec<usize> {
775        self.tile_shape.clone()
776    }
777
778    fn enter_traversal_cache_scope<'a>(
779        &'a self,
780        hint: &TraversalCacheHint,
781    ) -> Result<Option<Box<dyn TraversalCacheScope + 'a>>, LatticeError> {
782        if self.path.is_none() || !Self::supports_tiled_io() {
783            return Ok(None);
784        }
785        let previous_cache_bytes = self.max_cache_bytes.get();
786        let elem_size = Self::element_size_bytes().max(1);
787        let recommended_tiles =
788            recommended_tile_cache_size(&self.shape, &self.tile_shape, hint, None).max(1);
789        let recommended_bytes = recommended_tiles
790            .saturating_mul(self.tile_pixels())
791            .saturating_mul(elem_size);
792        if previous_cache_bytes == recommended_bytes {
793            return Ok(None);
794        }
795        self.set_cache_bytes_shared(recommended_bytes)?;
796        Ok(Some(Box::new(PagedArrayTraversalCacheScope {
797            array: self,
798            previous_cache_bytes,
799        })))
800    }
801}

803impl<T: LatticeElement> LatticeMut<T> for PagedArray<T> {
804    fn with_traversal_cache_hint_mut<R>(
805        &mut self,
806        hint: &TraversalCacheHint,
807        f: impl FnOnce(&mut Self) -> Result<R, LatticeError>,
808    ) -> Result<R, LatticeError> {
809        if self.path.is_none() || !Self::supports_tiled_io() {
810            return f(self);
811        }
812        let previous_cache_bytes = self.max_cache_bytes.get();
813        let elem_size = Self::element_size_bytes().max(1);
814        let recommended_tiles =
815            recommended_tile_cache_size(&self.shape, &self.tile_shape, hint, None).max(1);
816        let recommended_bytes = recommended_tiles
817            .saturating_mul(self.tile_pixels())
818            .saturating_mul(elem_size);
819        if previous_cache_bytes == recommended_bytes {
820            return f(self);
821        }
822
823        self.set_cache_bytes_shared(recommended_bytes)?;
824        let result = f(self);
825        let restore = self.set_cache_bytes_shared(previous_cache_bytes);
826        match (result, restore) {
827            (Err(err), _) => Err(err),
828            (Ok(_), Err(err)) => Err(err),
829            (Ok(value), Ok(())) => Ok(value),
830        }
831    }
832
833    fn put_at(&mut self, value: T, position: &[usize]) -> Result<(), LatticeError> {
834        let data = ArrayD::from_elem(IxDyn(&vec![1; position.len()]), value);
835        self.put_slice(&data, position)
836    }
837
838    fn put_slice(&mut self, data: &ArrayD<T>, start: &[usize]) -> Result<(), LatticeError> {
839        let ndim = self.shape.len();
840        if start.len() != ndim {
841            return Err(LatticeError::NdimMismatch {
842                expected: ndim,
843                got: start.len(),
844            });
845        }
846        let end: Vec<usize> = start
847            .iter()
848            .zip(data.shape().iter())
849            .map(|(&s, &n)| s + n)
850            .collect();
851        for (&limit, &dim) in end.iter().zip(self.shape.iter()) {
852            if limit > dim {
853                return Err(LatticeError::ShapeMismatch {
854                    expected: self.shape.clone(),
855                    got: end,
856                });
857            }
858        }
859
860        if self.path.is_some() {
861            self.auto_reopen()?;
862            if let Some(ref mut tio) = *self.tiled_io.borrow_mut() {
863                Self::tiled_put_slice(tio, data, start)?;
864                return Ok(());
865            }
866        }
867
868        let mut current = self.read_full_array()?;
869
870        let slice_info: Vec<ndarray::SliceInfoElem> = start
871            .iter()
872            .zip(data.shape().iter())
873            .map(|(&s, &n)| ndarray::SliceInfoElem::Slice {
874                start: s as isize,
875                end: Some((s + n) as isize),
876                step: 1,
877            })
878            .collect();
879
880        let mut view = current.slice_mut(slice_info.as_slice());
881        view.assign(data);
882
883        self.write_full_array(&current)
884    }
885
886    fn set(&mut self, value: T) -> Result<(), LatticeError> {
887        if self.tiled_io.borrow().is_some() {
888            let cursors = crate::TraversalCursorIter::new(
889                self.shape.clone(),
890                self.tile_shape.clone(),
891                crate::TraversalSpec::tiles(),
892            );
893            for cursor in cursors {
894                let cursor = cursor?;
895                let data = ArrayD::from_elem(IxDyn(&cursor.shape), value.clone());
896                self.put_slice(&data, &cursor.position)?;
897            }
898            Ok(())
899        } else {
900            let data = ArrayD::from_elem(IxDyn(&self.shape), value);
901            let start = vec![0; self.shape.len()];
902            self.put_slice(&data, &start)
903        }
904    }
905}
