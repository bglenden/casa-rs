118pub(crate) fn read_stman_file(
119    path: &Path,
120    col_info: &[StManColumnInfo],
121    byte_order: ByteOrder,
122) -> Result<StManAipsIOFile, StorageError> {
123    let mut io = AipsIo::open_with_order(path, AipsOpenOption::Old, byte_order)?;
124
125    let version = io.getstart("StManAipsIO")?;
126
127    let name = if version >= 2 {
128        io.get_string()?
129    } else {
130        String::new()
131    };
132
133    let seq_nr = io.get_u32()?;
134    let uniq_nr = io.get_u32()?;
135    let nrrow = io.get_u32()?;
136    let ncol = io.get_u32()?;
137
138    // Read per-column data types
139    let mut data_types = Vec::with_capacity(ncol as usize);
140    for _ in 0..ncol {
141        let dt_i32 = io.get_i32()?;
142        let dt = CasacoreDataType::from_i32(dt_i32).ok_or_else(|| {
143            StorageError::FormatMismatch(format!("unknown StManAipsIO column data type: {dt_i32}"))
144        })?;
145        data_types.push(dt);
146    }
147
148    // Read per-column data blocks, using get_next_type() to dispatch.
149    let mut columns = Vec::with_capacity(ncol as usize);
150    let big_endian = byte_order == ByteOrder::BigEndian;
151    for (i, dt) in data_types.iter().enumerate() {
152        let info = col_info.get(i).ok_or_else(|| {
153            StorageError::FormatMismatch(format!("missing column info for StManAipsIO col {i}"))
154        })?;
155
156        // Peek at the next AipsIO object type to determine column kind.
157        let next_type = io.get_next_type()?;
158        let col_data = match next_type.as_str() {
159            "StManColumnIndArrayAipsIO" => {
160                read_stman_indirect_column(&mut io, *dt, path, big_endian)?
161            }
162            "StManColumnArrayAipsIO" => {
163                StManColumnData::Flat(read_stman_array_column(&mut io, *dt, info.nrelem)?)
164            }
165            "StManColumnAipsIO" => {
166                if *dt == CasacoreDataType::TpRecord {
167                    read_stman_record_column(&mut io)?
168                } else {
169                    StManColumnData::Flat(read_stman_column(&mut io, *dt)?)
170                }
171            }
172            other => {
173                return Err(StorageError::FormatMismatch(format!(
174                    "unexpected StManAipsIO column type: {other}"
175                )));
176            }
177        };
178        columns.push(col_data);
179    }
180
181    io.getend()?;
182    io.close()?;
183
184    Ok(StManAipsIOFile {
185        name,
186        seq_nr,
187        uniq_nr,
188        nrrow,
189        ncol,
190        data_types,
191        columns,
192    })
193}

195fn read_stman_column(io: &mut AipsIo, dt: CasacoreDataType) -> Result<ColumnRawData, StorageError> {
196    let _version = io.getstart("StManColumnAipsIO")?;
197    let nrval = io.get_u32()?;
198
199    let data = if nrval > 0 {
200        // C++ StManColumnAipsIO organises data into extent blocks.
201        // Each extent has: u32(extent_count) + ios.put(extent_count, data).
202        // ios.put writes a count prefix + typed data, which getnew reads.
203        read_extent_data(io, dt, nrval)?
204    } else {
205        empty_column_data(dt)
206    };
207
208    io.getend()?;
209    Ok(data)
210}

212/// Read a record column from StManColumnAipsIO.
213///
214/// C++ `StManColumnAipsIO::getData` for `TpRecord` reads individual
215/// `TableRecord` objects (no count prefix, unlike typed columns). Each
216/// extent contains `extent_count` serialized `Record` objects.
217///
218/// Corresponds to C++ `ScalarRecordColumnData` with `StManColumnAipsIO`.
219fn read_stman_record_column(io: &mut AipsIo) -> Result<StManColumnData, StorageError> {
220    let _version = io.getstart("StManColumnAipsIO")?;
221    let nrval = io.get_u32()?;
222
223    let mut records = Vec::with_capacity(nrval as usize);
224    if nrval > 0 {
225        let mut nrd = 0u32;
226        while nrd < nrval {
227            let mut extent_count = io.get_u32()?;
228            if extent_count == 0 {
229                extent_count = nrval - nrd;
230            }
231            for _ in 0..extent_count {
232                let record = table_control::read_record(io)?;
233                records.push(Some(Value::Record(record)));
234            }
235            nrd += extent_count;
236        }
237    }
238
239    io.getend()?;
240    Ok(StManColumnData::Indirect(records))
241}

243/// Read an array column wrapped in StManColumnArrayAipsIO > StManColumnAipsIO.
244fn read_stman_array_column(
245    io: &mut AipsIo,
246    dt: CasacoreDataType,
247    nrelem: usize,
248) -> Result<ColumnRawData, StorageError> {
249    let _arr_version = io.getstart("StManColumnArrayAipsIO")?;
250
251    let _inner_version = io.getstart("StManColumnAipsIO")?;
252    let nrval = io.get_u32()?;
253
254    let data = if nrval > 0 {
255        read_array_extent_data(io, dt, nrval, nrelem)?
256    } else {
257        empty_column_data(dt)
258    };
259
260    io.getend()?; // StManColumnAipsIO
261    io.getend()?; // StManColumnArrayAipsIO
262    Ok(data)
263}

265/// Marker value for file offsets > 2GB in indirect AipsIO columns.
266const LARGE_OFFSET_MARKER: u32 = 2u32 * 1024 * 1024 * 1024 + 1;
267const AIPSIO_TOP_LEVEL_MAGIC: u32 = 0xbebebebe;

269/// Read a variable-shape (indirect) array column.
270///
271/// Corresponds to C++ `StManColumnIndArrayAipsIO::getFile`.
272/// The AipsIO stream contains:
273/// ```text
274/// StManColumnIndArrayAipsIO v2 {
275///     dtype: i32,
276///     seqnr: i32,
277///     StManColumnAipsIO {
278///         nrval,
279///         extents containing file offsets (not data)
280///     }
281/// }
282/// ```
283fn read_stman_indirect_column(
284    io: &mut AipsIo,
285    dt: CasacoreDataType,
286    data_path: &Path,
287    big_endian: bool,
288) -> Result<StManColumnData, StorageError> {
289    let ind_version = io.getstart("StManColumnIndArrayAipsIO")?;
290    let _dtype_compat = io.get_i32()?; // backward-compat dtype
291    let _seqnr = io.get_i32()?;
292
293    // Read inner StManColumnAipsIO containing file offsets.
294    let _inner_version = io.getstart("StManColumnAipsIO")?;
295    let nrval = io.get_u32()?;
296
297    let mut offsets: Vec<i64> = Vec::with_capacity(nrval as usize);
298    if nrval > 0 {
299        // Read extent blocks. Each extent: u32(extent_count) + getnew(count_prefix + u32 offsets).
300        // But the indirect column stores offsets via putData, which writes directly without
301        // count prefix — it uses the same extent structure as scalar columns.
302        let mut nrd: u32 = 0;
303        while nrd < nrval {
304            let mut extent_count = io.get_u32()?;
305            if extent_count == 0 {
306                extent_count = nrval - nrd;
307            }
308
309            // Read file offsets for this extent.
310            // The C++ putData writes u32 offsets (or marker+i64 for >2GB).
311            // The C++ getData reads them via ios >> off pattern.
312            // Since this is inside a scalar-column extent, it uses getnew (count-prefixed).
313            // Actually, looking at the C++ code more carefully:
314            // StManColumnAipsIO::putFile calls putData per extent, and putData/getData
315            // are virtual — for indirect columns they write/read file offsets.
316            // The getData reads raw u32 values from the AipsIO stream.
317            let extent_offsets = read_indirect_extent_offsets(io, extent_count as usize)?;
318            offsets.extend(extent_offsets);
319
320            nrd += extent_count;
321        }
322    }
323
324    io.getend()?; // StManColumnAipsIO
325    io.getend()?; // StManColumnIndArrayAipsIO
326
327    // Determine the array file path.
328    // Version >= 2: shared file = data_path + "i"
329    // Version <= 1: per-column file = data_path + "i" + seqnr (not commonly seen)
330    let array_file_path = if ind_version >= 2 {
331        let mut p = data_path.as_os_str().to_os_string();
332        p.push("i");
333        std::path::PathBuf::from(p)
334    } else {
335        let mut p = data_path.as_os_str().to_os_string();
336        p.push(format!("i{}", _seqnr));
337        std::path::PathBuf::from(p)
338    };
339
340    // Open the array file and read each row's data.
341    let mut reader = StManArrayFileReader::open(&array_file_path, big_endian)?;
342    let mut values = Vec::with_capacity(offsets.len());
343    for &offset in &offsets {
344        values.push(reader.read_array_at(offset, dt)?);
345    }
346
347    Ok(StManColumnData::Indirect(values))
348}

350/// Read file offsets from an indirect column extent.
351///
352/// C++ `StManColumnIndArrayAipsIO::getData` reads individual `ios >> uInt` per
353/// row (no count prefix). For offsets > 2GB a marker u32 is followed by an i64.
354fn read_indirect_extent_offsets(io: &mut AipsIo, count: usize) -> Result<Vec<i64>, StorageError> {
355    let mut offsets = Vec::with_capacity(count);
356    for _ in 0..count {
357        let off = io.get_u32()?;
358        if off == LARGE_OFFSET_MARKER {
359            let big = io.get_i64()?;
360            offsets.push(big);
361        } else {
362            offsets.push(off as i64);
363        }
364    }
365    Ok(offsets)
366}

368/// Read array column data organized in extent blocks.
369/// Each extent has: u32(extent_count) + u32(total_elements) + per-row raw data.
370fn read_array_extent_data(
371    io: &mut AipsIo,
372    dt: CasacoreDataType,
373    nrval: u32,
374    nrelem: usize,
375) -> Result<ColumnRawData, StorageError> {
376    let mut nrd: u32 = 0;
377    let mut result = empty_column_data(dt);
378
379    while nrd < nrval {
380        let mut extent_count = io.get_u32()?;
381        if extent_count == 0 {
382            extent_count = nrval - nrd;
383        }
384        if extent_count + nrd > nrval {
385            return Err(StorageError::FormatMismatch(format!(
386                "StManColumnArrayAipsIO: extent overrun ({extent_count}+{nrd} > {nrval})"
387            )));
388        }
389
390        // Read u32(total_elements) — sanity check, then per-row data
391        let _total_elements = io.get_u32()?;
392        let total_values = extent_count as usize * nrelem;
393
394        let extent = read_raw_typed_values(io, dt, total_values)?;
395        append_column_data(&mut result, extent);
396        nrd += extent_count;
397    }
398
399    Ok(result)
400}

402/// Read typed values without a count prefix (used for array column data).
403fn read_raw_typed_values(
404    io: &mut AipsIo,
405    dt: CasacoreDataType,
406    count: usize,
407) -> Result<ColumnRawData, StorageError> {
408    match dt {
409        CasacoreDataType::TpBool => {
410            let mut v = vec![false; count];
411            io.get_bool_into(&mut v)?;
412            Ok(ColumnRawData::Bool(v))
413        }
414        CasacoreDataType::TpUChar => {
415            let mut v = vec![0u8; count];
416            io.get_u8_into(&mut v)?;
417            Ok(ColumnRawData::UInt8(v))
418        }
419        CasacoreDataType::TpShort => {
420            let mut v = vec![0i16; count];
421            io.get_i16_into(&mut v)?;
422            Ok(ColumnRawData::Int16(v))
423        }
424        CasacoreDataType::TpUShort => {
425            let mut v = vec![0u16; count];
426            io.get_u16_into(&mut v)?;
427            Ok(ColumnRawData::UInt16(v))
428        }
429        CasacoreDataType::TpInt => {
430            let mut v = vec![0i32; count];
431            io.get_i32_into(&mut v)?;
432            Ok(ColumnRawData::Int32(v))
433        }
434        CasacoreDataType::TpUInt => {
435            let mut v = vec![0u32; count];
436            io.get_u32_into(&mut v)?;
437            Ok(ColumnRawData::UInt32(v))
438        }
439        CasacoreDataType::TpFloat => {
440            let mut v = vec![0f32; count];
441            io.get_f32_into(&mut v)?;
442            Ok(ColumnRawData::Float32(v))
443        }
444        CasacoreDataType::TpDouble => {
445            let mut v = vec![0f64; count];
446            io.get_f64_into(&mut v)?;
447            Ok(ColumnRawData::Float64(v))
448        }
449        CasacoreDataType::TpComplex => {
450            let mut v = vec![casa_types::Complex32::new(0.0, 0.0); count];
451            io.get_complex32_into(&mut v)?;
452            Ok(ColumnRawData::Complex32(v))
453        }
454        CasacoreDataType::TpDComplex => {
455            let mut v = vec![casa_types::Complex64::new(0.0, 0.0); count];
456            io.get_complex64_into(&mut v)?;
457            Ok(ColumnRawData::Complex64(v))
458        }
459        CasacoreDataType::TpString => {
460            let mut v = Vec::with_capacity(count);
461            for _ in 0..count {
462                v.push(io.get_string()?);
463            }
464            Ok(ColumnRawData::String(v))
465        }
466        CasacoreDataType::TpInt64 => {
467            let mut v = vec![0i64; count];
468            io.get_i64_into(&mut v)?;
469            Ok(ColumnRawData::Int64(v))
470        }
471        _ => Err(StorageError::FormatMismatch(format!(
472            "unsupported StManAipsIO array column type: {dt:?}"
473        ))),
474    }
475}

477/// Read column data organized in extent blocks as written by C++ StManColumnAipsIO.
478/// Format per extent: u32(extent_count) + getnew(count_prefix + typed_data).
479fn read_extent_data(
480    io: &mut AipsIo,
481    dt: CasacoreDataType,
482    nrval: u32,
483) -> Result<ColumnRawData, StorageError> {
484    let mut nrd: u32 = 0;
485    let mut result = empty_column_data(dt);
486
487    while nrd < nrval {
488        let mut extent_count = io.get_u32()?;
489        if extent_count == 0 {
490            extent_count = nrval - nrd;
491        }
492        if extent_count + nrd > nrval {
493            return Err(StorageError::FormatMismatch(format!(
494                "StManColumnAipsIO: extent overrun ({extent_count}+{nrd} > {nrval})"
495            )));
496        }
497
498        // getnew reads the count prefix (written by ios.put) + typed data
499        let extent = read_typed_extent(io, dt)?;
500        append_column_data(&mut result, extent);
501        nrd += extent_count;
502    }
503
504    Ok(result)
505}

507/// Read a single extent's typed data via getnew (reads count prefix + data).
508fn read_typed_extent(io: &mut AipsIo, dt: CasacoreDataType) -> Result<ColumnRawData, StorageError> {
509    match dt {
510        CasacoreDataType::TpBool => Ok(ColumnRawData::Bool(io.getnew_bool()?)),
511        CasacoreDataType::TpUChar => Ok(ColumnRawData::UInt8(io.getnew_u8()?)),
512        CasacoreDataType::TpShort => Ok(ColumnRawData::Int16(io.getnew_i16()?)),
513        CasacoreDataType::TpUShort => Ok(ColumnRawData::UInt16(io.getnew_u16()?)),
514        CasacoreDataType::TpInt => Ok(ColumnRawData::Int32(io.getnew_i32()?)),
515        CasacoreDataType::TpUInt => Ok(ColumnRawData::UInt32(io.getnew_u32()?)),
516        CasacoreDataType::TpFloat => Ok(ColumnRawData::Float32(io.getnew_f32()?)),
517        CasacoreDataType::TpDouble => Ok(ColumnRawData::Float64(io.getnew_f64()?)),
518        CasacoreDataType::TpComplex => Ok(ColumnRawData::Complex32(io.getnew_complex32()?)),
519        CasacoreDataType::TpDComplex => Ok(ColumnRawData::Complex64(io.getnew_complex64()?)),
520        CasacoreDataType::TpString => Ok(ColumnRawData::String(io.getnew_string()?)),
521        CasacoreDataType::TpInt64 => Ok(ColumnRawData::Int64(io.getnew_i64()?)),
522        _ => Err(StorageError::FormatMismatch(format!(
523            "unsupported StManAipsIO column type: {dt:?}"
524        ))),
525    }
526}

528/// Append extent data to the accumulated column data.
529fn append_column_data(target: &mut ColumnRawData, source: ColumnRawData) {
530    match (target, source) {
531        (ColumnRawData::Bool(t), ColumnRawData::Bool(s)) => t.extend(s),
532        (ColumnRawData::UInt8(t), ColumnRawData::UInt8(s)) => t.extend(s),
533        (ColumnRawData::Int16(t), ColumnRawData::Int16(s)) => t.extend(s),
534        (ColumnRawData::UInt16(t), ColumnRawData::UInt16(s)) => t.extend(s),
535        (ColumnRawData::Int32(t), ColumnRawData::Int32(s)) => t.extend(s),
536        (ColumnRawData::UInt32(t), ColumnRawData::UInt32(s)) => t.extend(s),
537        (ColumnRawData::Int64(t), ColumnRawData::Int64(s)) => t.extend(s),
538        (ColumnRawData::Float32(t), ColumnRawData::Float32(s)) => t.extend(s),
539        (ColumnRawData::Float64(t), ColumnRawData::Float64(s)) => t.extend(s),
540        (ColumnRawData::Complex32(t), ColumnRawData::Complex32(s)) => t.extend(s),
541        (ColumnRawData::Complex64(t), ColumnRawData::Complex64(s)) => t.extend(s),
542        (ColumnRawData::String(t), ColumnRawData::String(s)) => t.extend(s),
543        _ => {} // type mismatch — should not happen
544    }
545}

547fn empty_column_data(dt: CasacoreDataType) -> ColumnRawData {

549        CasacoreDataType::TpBool => ColumnRawData::Bool(vec![]),

551        CasacoreDataType::TpShort => ColumnRawData::Int16(vec![]),
552        CasacoreDataType::TpUShort => ColumnRawData::UInt16(vec![]),
553        CasacoreDataType::TpInt => ColumnRawData::Int32(vec![]),
554        CasacoreDataType::TpUInt => ColumnRawData::UInt32(vec![]),
555        CasacoreDataType::TpFloat => ColumnRawData::Float32(vec![]),
556        CasacoreDataType::TpDouble => ColumnRawData::Float64(vec![]),
557        CasacoreDataType::TpComplex => ColumnRawData::Complex32(vec![]),
558        CasacoreDataType::TpDComplex => ColumnRawData::Complex64(vec![]),

560        CasacoreDataType::TpInt64 => ColumnRawData::Int64(vec![]),
561        _ => ColumnRawData::Int32(vec![]),
562    }
563}

569pub(crate) fn save_stman_file_rows_in_place(
570    path: &Path,
571    columns: &[ColumnDescContents],
572    changed_values: &HashMap<&str, Vec<(usize, Option<Value>)>>,
573    byte_order: ByteOrder,
574) -> Result<bool, StorageError> {
575    if changed_values.is_empty() {
576        return Ok(true);
577    }
578
579    let layouts = parse_sparse_column_layouts(path, columns, byte_order)?;
580    if layouts.len() != columns.len() {
581        return Ok(false);
582    }
583
584    let mut main_file = OpenOptions::new().read(true).write(true).open(path)?;
585    let array_file_path = indirect_array_file_path(path);
586    let array_file_len = if array_file_path.exists() {
587        Some(std::fs::metadata(&array_file_path)?.len())
588    } else {
589        None
590    };
591    let mut array_writer: Option<StManArrayFileWriter> = None;
592
593    for (col_index, col_desc) in columns.iter().enumerate() {
594        let Some(row_values) = changed_values.get(col_desc.col_name.as_str()) else {
595            continue;
596        };
597        let Some(layout) = layouts[col_index].as_ref() else {
598            return Ok(false);
599        };
600        match layout {
601            SparseColumnLayout::ScalarBoolPacked { extents } => {
602                for &(row_index, ref value) in row_values {
603                    let Some(Value::Scalar(ScalarValue::Bool(flag))) = value else {
604                        return Ok(false);
605                    };
606                    let (byte_pos, bit_index) = locate_bit_extent_slot(extents, row_index)?;
607                    main_file.seek(SeekFrom::Start(byte_pos))?;
608                    let mut current = [0u8; 1];
609                    main_file.read_exact(&mut current)?;
610                    if *flag {
611                        current[0] |= 1 << bit_index;
612                    } else {
613                        current[0] &= !(1 << bit_index);
614                    }
615                    main_file.seek(SeekFrom::Start(byte_pos))?;
616                    main_file.write_all(&current)?;
617                }
618            }
619            SparseColumnLayout::IndirectArrayOffsetsU32 { data_type, extents } => {
620                let existing_len = array_file_len.unwrap_or(0);
621                if existing_len >= LARGE_OFFSET_MARKER as u64 {
622                    return Ok(false);
623                }
624                let writer = match array_writer.as_mut() {
625                    Some(writer) => writer,
626                    None => {
627                        array_writer = Some(StManArrayFileWriter::open_append(
628                            &array_file_path,
629                            byte_order == ByteOrder::BigEndian,
630                        )?);
631                        array_writer.as_mut().expect("writer just created")
632                    }
633                };
634                for &(row_index, ref value) in row_values {
635                    let offset = match value {
636                        Some(value @ Value::Array(_)) => writer.write_array(value, *data_type)?,
637                        None => 0,
638                        _ => return Ok(false),
639                    };
640                    let offset_u32 = u32::try_from(offset).map_err(|_| {
641                        StorageError::FormatMismatch(format!(
642                            "sparse StManAipsIO offset exceeds u32 for column {} row {}",
643                            col_desc.col_name, row_index
644                        ))
645                    })?;
646                    let slot = locate_extent_slot(extents, row_index)?;
647                    main_file.seek(SeekFrom::Start(slot))?;
648                    main_file.write_all(&offset_u32.to_be_bytes())?;
649                }
650            }
651            SparseColumnLayout::ScalarFixed { data_type, extents } => {
652                for &(row_index, ref value) in row_values {
653                    let Some(Value::Scalar(scalar)) = value else {
654                        return Ok(false);
655                    };
656                    let encoded = encode_fixed_scalar_value(*data_type, scalar)?;
657                    let slot = locate_extent_slot(extents, row_index)?;
658                    main_file.seek(SeekFrom::Start(slot))?;
659                    main_file.write_all(&encoded)?;
660                }
661            }
662            SparseColumnLayout::DirectArrayFixed { data_type, extents } => {
663                for &(row_index, ref value) in row_values {
664                    let Some(Value::Array(array)) = value else {
665                        return Ok(false);
666                    };
667                    let encoded = encode_fixed_array_row(*data_type, array)?;
668                    let slot = locate_extent_slot(extents, row_index)?;
669                    main_file.seek(SeekFrom::Start(slot))?;
670                    main_file.write_all(&encoded)?;
671                }
672            }
673        }
674    }
675
676    if let Some(writer) = array_writer.as_mut() {
677        writer.finish()?;
678    }
679    main_file.flush()?;
680    Ok(true)
681}

683pub(crate) fn read_stman_array_column_rows(
684    path: &Path,
685    columns: &[ColumnDescContents],
686    target_col_idx: usize,
687    selected_rows: &[usize],
688    byte_order: ByteOrder,
689) -> Result<Option<Vec<Option<ArrayValue>>>, StorageError> {
690    if selected_rows.is_empty() {
691        return Ok(Some(Vec::new()));
692    }
693    let layouts = parse_sparse_column_layouts(path, columns, byte_order)?;
694    let Some(SparseColumnLayout::IndirectArrayOffsetsU32 { data_type, extents }) = layouts
695        .get(target_col_idx)
696        .and_then(|layout| layout.as_ref())
697    else {
698        return Ok(None);
699    };
700
701    let array_file_path = indirect_array_file_path(path);
702    if std::fs::metadata(&array_file_path)
703        .map(|meta| meta.len())
704        .unwrap_or(0)
705        >= LARGE_OFFSET_MARKER as u64
706    {
707        return Ok(None);
708    }
709
710    let mut main_file = std::fs::File::open(path)?;
711    let mut requests: Vec<(usize, usize)> = selected_rows
712        .iter()
713        .copied()
714        .enumerate()
715        .map(|(out_idx, row_index)| (row_index, out_idx))
716        .collect();
717    requests.sort_unstable_by_key(|&(row_index, _)| row_index);
718
719    let mut offsets = vec![0u32; selected_rows.len()];
720    for (row_index, out_idx) in requests {
721        let slot = locate_extent_slot(extents, row_index)?;
722        main_file.seek(SeekFrom::Start(slot))?;
723        offsets[out_idx] = read_be_u32(&mut main_file)?;
724    }
725
726    let mut reader =
727        StManArrayFileReader::open(&array_file_path, byte_order == ByteOrder::BigEndian)?;
728    let mut values = Vec::with_capacity(offsets.len());
729    for offset in offsets {
730        let value = reader
731            .read_array_at(offset as i64, *data_type)?
732            .map(|value| match value {
733                Value::Array(array) => Ok(array),
734                other => Err(StorageError::FormatMismatch(format!(
735                    "expected array value in indirect StManAipsIO column, found {:?}",
736                    other.kind()
737                ))),
738            })
739            .transpose()?;
740        values.push(value);
741    }
742
743    Ok(Some(values))
744}

746pub(crate) fn read_stman_scalar_column(
747    path: &Path,
748    columns: &[ColumnDescContents],
749    target_col_idx: usize,
750    byte_order: ByteOrder,
751) -> Result<Option<Vec<Option<ScalarValue>>>, StorageError> {
752    let layouts = parse_sparse_column_layouts(path, columns, byte_order)?;
753    let Some(layout) = layouts
754        .get(target_col_idx)
755        .and_then(|layout| layout.as_ref())
756    else {
757        return Ok(None);
758    };
759    let column = columns.get(target_col_idx).ok_or_else(|| {
760        StorageError::FormatMismatch(format!(
761            "target StManAipsIO scalar column index {target_col_idx} is out of range"
762        ))
763    })?;
764
765    let mut file = std::fs::File::open(path)?;
766    let mut values = Vec::new();
767    match layout {
768        SparseColumnLayout::ScalarBoolPacked { extents } => {
769            for extent in extents {
770                file.seek(SeekFrom::Start(extent.values_start))?;
771                let mut packed = vec![0u8; extent.row_count.div_ceil(8)];
772                file.read_exact(&mut packed)?;
773                for row_offset in 0..extent.row_count {
774                    let byte = packed[row_offset / 8];
775                    let flag = ((byte >> (row_offset % 8)) & 1) != 0;
776                    let scalar = ScalarValue::Bool(flag);
777                    if (column.option & 2) != 0
778                        && scalar_value_is_default(
779                            &Value::Scalar(scalar.clone()),
780                            column.require_primitive_type()?,
781                        )
782                    {
783                        values.push(None);
784                    } else {
785                        values.push(Some(scalar));
786                    }
787                }
788            }
789        }
790        SparseColumnLayout::ScalarFixed { data_type, extents } => {
791            for extent in extents {
792                file.seek(SeekFrom::Start(extent.values_start))?;
793                let mut raw = vec![0u8; extent.row_count * extent.row_width_bytes];
794                file.read_exact(&mut raw)?;
795                for chunk in raw.chunks_exact(extent.row_width_bytes) {
796                    let scalar = decode_fixed_scalar_value(*data_type, chunk)?;
797                    if (column.option & 2) != 0
798                        && scalar_value_is_default(
799                            &Value::Scalar(scalar.clone()),
800                            column.require_primitive_type()?,
801                        )
802                    {
803                        values.push(None);
804                    } else {
805                        values.push(Some(scalar));
806                    }
807                }
808            }
809        }
810        _ => return Ok(None),
811    }
812    Ok(Some(values))
813}

815fn parse_sparse_column_layouts(
816    path: &Path,
817    columns: &[ColumnDescContents],
818    byte_order: ByteOrder,
819) -> Result<Vec<Option<SparseColumnLayout>>, StorageError> {
820    if byte_order != ByteOrder::BigEndian {
821        return Ok(vec![None; columns.len()]);
822    }
823
824    let mut file = std::fs::File::open(path)?;
825    let magic = read_be_u32(&mut file)?;
826    if magic != AIPSIO_TOP_LEVEL_MAGIC {
827        return Err(StorageError::FormatMismatch(format!(
828            "StManAipsIO file missing top-level magic: {magic:#x}"
829        )));
830    }
831    let top = read_object_header(&mut file)?;
832    if top.type_name != "StManAipsIO" {
833        return Err(StorageError::FormatMismatch(format!(
834            "expected StManAipsIO top-level object, found {}",
835            top.type_name
836        )));
837    }
838
839    let _name = read_be_string(&mut file)?;
840    let _seq_nr = read_be_u32(&mut file)?;
841    let _uniq_nr = read_be_u32(&mut file)?;
842    let nrrow = read_be_u32(&mut file)? as usize;
843    let ncol = read_be_u32(&mut file)? as usize;
844    if ncol != columns.len() {
845        return Err(StorageError::FormatMismatch(format!(
846            "StManAipsIO column count mismatch: file has {ncol}, expected {}",
847            columns.len()
848        )));
849    }
850
851    let mut data_types = Vec::with_capacity(ncol);
852    for _ in 0..ncol {
853        let dt_i32 = read_be_i32(&mut file)?;
854        let dt = CasacoreDataType::from_i32(dt_i32).ok_or_else(|| {
855            StorageError::FormatMismatch(format!("unknown StManAipsIO dtype tag: {dt_i32}"))
856        })?;
857        data_types.push(dt);
858    }
859
860    let mut layouts = Vec::with_capacity(ncol);
861    for (col_desc, data_type) in columns.iter().zip(data_types) {
862        let column_header = read_object_header(&mut file)?;
863        let layout = match column_header.type_name.as_str() {
864            "StManColumnIndArrayAipsIO" => {
865                let _compat_dtype = read_be_i32(&mut file)?;
866                let _seqnr = read_be_i32(&mut file)?;
867                let inner = read_object_header(&mut file)?;
868                if inner.type_name != "StManColumnAipsIO" {
869                    None
870                } else {
871                    let stored_nrrow = read_be_u32(&mut file)? as usize;
872                    if stored_nrrow != nrrow {
873                        None
874                    } else {
875                        parse_indirect_offsets_layout(&mut file, stored_nrrow, data_type)?
876                    }
877                }
878            }
879            "StManColumnAipsIO" => {
880                let stored_nrrow = read_be_u32(&mut file)? as usize;
881                if stored_nrrow != nrrow {
882                    None
883                } else {
884                    parse_scalar_layout(&mut file, stored_nrrow, data_type)?
885                }
886            }
887            "StManColumnArrayAipsIO" => {
888                let inner = read_object_header(&mut file)?;
889                if inner.type_name != "StManColumnAipsIO" {
890                    None
891                } else {
892                    let stored_nrrow = read_be_u32(&mut file)? as usize;
893                    if stored_nrrow != nrrow {
894                        None
895                    } else {
896                        parse_direct_array_layout(&mut file, stored_nrrow, col_desc, data_type)?
897                    }
898                }
899            }
900            _ => None,
901        };
902        file.seek(SeekFrom::Start(column_header.object_end))?;
903        layouts.push(layout);
904    }
905
906    file.seek(SeekFrom::Start(top.object_end))?;
907    Ok(layouts)
908}

910fn parse_indirect_offsets_layout(
911    file: &mut std::fs::File,
912    nrrow: usize,
913    data_type: CasacoreDataType,
914) -> Result<Option<SparseColumnLayout>, StorageError> {
915    let mut extents = Vec::new();
916    let mut row_start = 0usize;
917    while row_start < nrrow {
918        let raw_count = read_be_u32(file)? as usize;
919        let row_count = if raw_count == 0 {
920            nrrow - row_start
921        } else {
922            raw_count
923        };
924        let values_start = file.stream_position()?;
925        file.seek(SeekFrom::Current(
926            (row_count * std::mem::size_of::<u32>()) as i64,
927        ))?;
928        extents.push(ExtentLayout {
929            row_start,
930            row_count,
931            values_start,
932            row_width_bytes: std::mem::size_of::<u32>(),
933        });
934        row_start += row_count;
935    }
936    Ok(Some(SparseColumnLayout::IndirectArrayOffsetsU32 {
937        data_type,
938        extents,
939    }))
940}

942fn parse_scalar_layout(
943    file: &mut std::fs::File,
944    nrrow: usize,
945    data_type: CasacoreDataType,
946) -> Result<Option<SparseColumnLayout>, StorageError> {
947    if data_type == CasacoreDataType::TpBool {
948        let mut extents = Vec::new();
949        let mut row_start = 0usize;
950        while row_start < nrrow {
951            let raw_count = read_be_u32(file)? as usize;
952            let row_count = if raw_count == 0 {
953                nrrow - row_start
954            } else {
955                raw_count
956            };
957            let stored_count = read_be_u32(file)? as usize;
958            if stored_count != row_count {
959                return Ok(None);
960            }
961            let values_start = file.stream_position()?;
962            let packed_bytes = row_count.div_ceil(8);
963            file.seek(SeekFrom::Current(packed_bytes as i64))?;
964            extents.push(BitExtentLayout {
965                row_start,
966                row_count,
967                values_start,
968            });
969            row_start += row_count;
970        }
971        return Ok(Some(SparseColumnLayout::ScalarBoolPacked { extents }));
972    }
973
974    let Some(row_width_bytes) = scalar_fixed_width_bytes(data_type) else {
975        return Ok(None);
976    };
977    let mut extents = Vec::new();
978    let mut row_start = 0usize;
979    while row_start < nrrow {
980        let raw_count = read_be_u32(file)? as usize;
981        let row_count = if raw_count == 0 {
982            nrrow - row_start
983        } else {
984            raw_count
985        };
986        let stored_count = read_be_u32(file)? as usize;
987        if stored_count != row_count {
988            return Ok(None);
989        }
990        let values_start = file.stream_position()?;
991        file.seek(SeekFrom::Current((row_count * row_width_bytes) as i64))?;
992        extents.push(ExtentLayout {
993            row_start,
994            row_count,
995            values_start,
996            row_width_bytes,
997        });
998        row_start += row_count;
999    }
1000    Ok(Some(SparseColumnLayout::ScalarFixed { data_type, extents }))
1001}

1003fn parse_direct_array_layout(
1004    file: &mut std::fs::File,
1005    nrrow: usize,
1006    col_desc: &ColumnDescContents,
1007    data_type: CasacoreDataType,
1008) -> Result<Option<SparseColumnLayout>, StorageError> {
1009    let Some(elem_width) = scalar_fixed_width_bytes(data_type) else {
1010        return Ok(None);
1011    };
1012    let elements_per_row: usize = col_desc.shape.iter().map(|&dim| dim as usize).product();
1013    if elements_per_row == 0 {
1014        return Ok(None);
1015    }
1016    let row_width_bytes = elements_per_row.checked_mul(elem_width).ok_or_else(|| {
1017        StorageError::FormatMismatch("direct-array row width overflow".to_string())
1018    })?;
1019    let mut extents = Vec::new();
1020    let mut row_start = 0usize;
1021    while row_start < nrrow {
1022        let raw_count = read_be_u32(file)? as usize;
1023        let row_count = if raw_count == 0 {
1024            nrrow - row_start
1025        } else {
1026            raw_count
1027        };
1028        let total_elements = read_be_u32(file)? as usize;
1029        if total_elements != row_count * elements_per_row {
1030            return Ok(None);
1031        }
1032        let values_start = file.stream_position()?;
1033        file.seek(SeekFrom::Current((row_count * row_width_bytes) as i64))?;
1034        extents.push(ExtentLayout {
1035            row_start,
1036            row_count,
1037            values_start,
1038            row_width_bytes,
1039        });
1040        row_start += row_count;
1041    }
1042    Ok(Some(SparseColumnLayout::DirectArrayFixed {
1043        data_type,
1044        extents,
1045    }))
1046}

1064fn locate_extent_slot(extents: &[ExtentLayout], row_index: usize) -> Result<u64, StorageError> {
1065    for extent in extents {
1066        if row_index >= extent.row_start && row_index < extent.row_start + extent.row_count {
1067            return Ok(extent.values_start
1068                + ((row_index - extent.row_start) * extent.row_width_bytes) as u64);
1069        }
1070    }
1071    Err(StorageError::FormatMismatch(format!(
1072        "row {row_index} is outside sparse-patch extents"
1073    )))
1074}
1075
1076fn locate_bit_extent_slot(
1077    extents: &[BitExtentLayout],
1078    row_index: usize,
1079) -> Result<(u64, usize), StorageError> {
1080    for extent in extents {
1081        if row_index >= extent.row_start && row_index < extent.row_start + extent.row_count {
1082            let row_offset = row_index - extent.row_start;
1083            return Ok((
1084                extent.values_start + (row_offset / 8) as u64,
1085                row_offset % 8,
1086            ));
1087        }
1088    }
1089    Err(StorageError::FormatMismatch(format!(
1090        "row {row_index} is outside sparse-patch bit extents"
1091    )))
1092}

1094fn scalar_fixed_width_bytes(data_type: CasacoreDataType) -> Option<usize> {
1095    match data_type {
1096        CasacoreDataType::TpBool | CasacoreDataType::TpUChar | CasacoreDataType::TpChar => Some(1),
1097        CasacoreDataType::TpShort | CasacoreDataType::TpUShort => Some(2),
1098        CasacoreDataType::TpInt | CasacoreDataType::TpUInt | CasacoreDataType::TpFloat => Some(4),
1099        CasacoreDataType::TpDouble | CasacoreDataType::TpInt64 | CasacoreDataType::TpComplex => {
1100            Some(8)
1101        }
1102        CasacoreDataType::TpDComplex => Some(16),
1103        _ => None,
1104    }
1105}

1107fn decode_fixed_scalar_value(
1108    data_type: CasacoreDataType,
1109    bytes: &[u8],
1110) -> Result<ScalarValue, StorageError> {
1111    let scalar = match data_type {
1112        CasacoreDataType::TpUChar | CasacoreDataType::TpChar => {
1113            ScalarValue::UInt8(*bytes.first().ok_or_else(|| {
1114                StorageError::FormatMismatch("missing u8 scalar bytes".to_string())
1115            })?)
1116        }
1117        CasacoreDataType::TpShort => {
1118            ScalarValue::Int16(i16::from_be_bytes(bytes.try_into().map_err(|_| {
1119                StorageError::FormatMismatch("invalid i16 scalar width".to_string())
1120            })?))
1121        }
1122        CasacoreDataType::TpUShort => {
1123            ScalarValue::UInt16(u16::from_be_bytes(bytes.try_into().map_err(|_| {
1124                StorageError::FormatMismatch("invalid u16 scalar width".to_string())
1125            })?))
1126        }
1127        CasacoreDataType::TpInt => {
1128            ScalarValue::Int32(i32::from_be_bytes(bytes.try_into().map_err(|_| {
1129                StorageError::FormatMismatch("invalid i32 scalar width".to_string())
1130            })?))
1131        }
1132        CasacoreDataType::TpUInt => {
1133            ScalarValue::UInt32(u32::from_be_bytes(bytes.try_into().map_err(|_| {
1134                StorageError::FormatMismatch("invalid u32 scalar width".to_string())
1135            })?))
1136        }
1137        CasacoreDataType::TpInt64 => {
1138            ScalarValue::Int64(i64::from_be_bytes(bytes.try_into().map_err(|_| {
1139                StorageError::FormatMismatch("invalid i64 scalar width".to_string())
1140            })?))
1141        }
1142        CasacoreDataType::TpFloat => {
1143            ScalarValue::Float32(f32::from_be_bytes(bytes.try_into().map_err(|_| {
1144                StorageError::FormatMismatch("invalid f32 scalar width".to_string())
1145            })?))
1146        }
1147        CasacoreDataType::TpDouble => {
1148            ScalarValue::Float64(f64::from_be_bytes(bytes.try_into().map_err(|_| {
1149                StorageError::FormatMismatch("invalid f64 scalar width".to_string())
1150            })?))
1151        }
1152        CasacoreDataType::TpComplex => {
1153            if bytes.len() != 8 {
1154                return Err(StorageError::FormatMismatch(
1155                    "invalid complex32 scalar width".to_string(),
1156                ));
1157            }
1158            ScalarValue::Complex32(casa_types::Complex32 {
1159                re: f32::from_be_bytes(bytes[0..4].try_into().expect("slice width checked")),
1160                im: f32::from_be_bytes(bytes[4..8].try_into().expect("slice width checked")),
1161            })
1162        }
1163        CasacoreDataType::TpDComplex => {
1164            if bytes.len() != 16 {
1165                return Err(StorageError::FormatMismatch(
1166                    "invalid complex64 scalar width".to_string(),
1167                ));
1168            }
1169            ScalarValue::Complex64(casa_types::Complex64 {
1170                re: f64::from_be_bytes(bytes[0..8].try_into().expect("slice width checked")),
1171                im: f64::from_be_bytes(bytes[8..16].try_into().expect("slice width checked")),
1172            })
1173        }
1174        other => {
1175            return Err(StorageError::FormatMismatch(format!(
1176                "unsupported direct StManAipsIO scalar decode for {other:?}"
1177            )));
1178        }
1179    };
1180    Ok(scalar)
1181}

1183fn encode_fixed_scalar_value(
1184    data_type: CasacoreDataType,
1185    value: &ScalarValue,
1186) -> Result<Vec<u8>, StorageError> {
1187    let bytes = match (data_type, value) {
1188        (CasacoreDataType::TpBool, ScalarValue::Bool(v)) => vec![u8::from(*v)],
1189        (CasacoreDataType::TpUChar, ScalarValue::UInt8(v))
1190        | (CasacoreDataType::TpChar, ScalarValue::UInt8(v)) => vec![*v],
1191        (CasacoreDataType::TpShort, ScalarValue::Int16(v)) => v.to_be_bytes().to_vec(),
1192        (CasacoreDataType::TpUShort, ScalarValue::UInt16(v)) => v.to_be_bytes().to_vec(),
1193        (CasacoreDataType::TpInt, ScalarValue::Int32(v)) => v.to_be_bytes().to_vec(),
1194        (CasacoreDataType::TpUInt, ScalarValue::UInt32(v)) => v.to_be_bytes().to_vec(),

1447pub(crate) fn write_stman_file(
1448    path: &Path,
1449    columns: &[ColumnDescContents],
1450    rows: &[casa_types::RecordValue],
1451    byte_order: ByteOrder,
1452) -> Result<(), StorageError> {
1453    let nrrow = rows.len() as u32;
1454    let ncol = columns.len() as u32;
1455    let big_endian = byte_order == ByteOrder::BigEndian;
1456
1457    // Check if any columns are stored indirectly or are record columns.
1458    // Record columns use indirect Vector<uChar> storage (like C++ ScaRecordColData).
1459    let has_indirect = columns
1460        .iter()
1461        .any(|c| c.is_record() || (c.is_array && (c.option & 1) == 0));
1462
1463    // Create shared array file for indirect columns (version 2 format: path + "i").
1464    let mut array_writer = if has_indirect {
1465        let mut arr_path = path.as_os_str().to_os_string();
1466        arr_path.push("i");
1467        // StManAipsIO uses version 1 (with refcount) by default.
1468        Some(StManArrayFileWriter::create(
1469            &std::path::PathBuf::from(arr_path),
1470            big_endian,
1471            1,
1472        )?)
1473    } else {
1474        None
1475    };
1476
1477    let mut io = AipsIo::open_with_order(path, AipsOpenOption::New, byte_order)?;
1478    io.putstart("StManAipsIO", 2)?;
1479
1480    // C++ casacore writes an empty stmanName for StManAipsIO v2
1481    io.put_string("")?; // stmanName
1482    io.put_u32(0)?; // seqNr
1483    io.put_u32(0)?; // uniqNr
1484    io.put_u32(nrrow)?;
1485    io.put_u32(ncol)?;
1486
1487    // Per-column data types.
1488    // C++ stores the DM-level data type: TpUChar for record columns
1489    // (since ScaRecordColData uses createIndArrColumn(name, TpUChar, "")).
1490    for col in columns {
1491        let scalar_dt = if col.is_record() {
1492            CasacoreDataType::TpUChar
1493        } else {
1494            CasacoreDataType::from_primitive_type(col.require_primitive_type()?, false)
1495        };
1496        io.put_i32(scalar_dt as i32)?;
1497    }
1498
1499    // Per-column data blocks
1500    for (col_idx, col) in columns.iter().enumerate() {
1501        write_stman_column(&mut io, col, rows, col_idx, &mut array_writer)?;
1502    }
1503
1504    // Flush the array file header and buffer if we wrote indirect arrays.
1505    if let Some(ref mut w) = array_writer {
1506        w.finish()?;
1507    }
1508
1509    io.putend()?;
1510    io.close()?;
1511    Ok(())
1512}

1514fn write_stman_column(
1515    io: &mut AipsIo,
1516    col_desc: &ColumnDescContents,
1517    rows: &[casa_types::RecordValue],
1518    col_idx: usize,
1519    array_writer: &mut Option<StManArrayFileWriter>,
1520) -> Result<(), StorageError> {
1521    let col_name = &col_desc.col_name;
1522    let is_direct_array = col_desc.is_array && (col_desc.option & 1) != 0;
1523    let is_indirect_array = col_desc.is_array && !is_direct_array;
1524
1525    if col_desc.is_record() {
1526        // Record column: serialize as indirect Vector<uChar>, matching C++ ScaRecordColData.
1527        write_stman_record_as_indirect(io, rows, col_name, array_writer)?;
1528    } else if is_indirect_array {
1529        // Non-direct array column: StManColumnIndArrayAipsIO. This includes
1530        // both variable-shape arrays and fixed-shape arrays without the Direct
1531        // option bit.
1532        write_stman_indirect_column(io, col_desc, rows, col_name, array_writer)?;
1533    } else if is_direct_array {
1534        // Fixed-shape array: StManColumnArrayAipsIO wrapping StManColumnAipsIO
1535        let nrrow = rows.len() as u32;
1536        let elements_per_row: usize = col_desc.shape.iter().map(|&s| s as usize).product();
1537
1538        io.putstart("StManColumnArrayAipsIO", 2)?;
1539        io.putstart("StManColumnAipsIO", 2)?;
1540
1541        io.put_u32(nrrow)?; // nrval = number of rows
1542
1543        if nrrow > 0 {
1544            // Single extent: extent_count + total_elements + per-row data (no count prefix)
1545            io.put_u32(nrrow)?; // extent count
1546            io.put_u32(nrrow * elements_per_row as u32)?; // total elements
1547            write_flat_array_column_raw(io, col_desc, rows, col_name)?;
1548        }
1549
1550        io.putend()?; // StManColumnAipsIO
1551        io.putend()?; // StManColumnArrayAipsIO
1552    } else {
1553        // Scalar columns: StManColumnAipsIO only
1554        let nrval = rows.len() as u32;
1555
1556        io.putstart("StManColumnAipsIO", 2)?;
1557        io.put_u32(nrval)?;
1558
1559        if nrval > 0 {
1560            // Single extent: extent_count + data (with count prefix)
1561            io.put_u32(nrval)?;
1562            write_scalar_column(io, col_desc, rows, col_name, col_idx)?;
1563        }
1564
1565        io.putend()?;
1566    }
1567
1568    Ok(())
1569}

1571/// Write a variable-shape (indirect) column.
1572///
1573/// Corresponds to C++ `StManColumnIndArrayAipsIO::putFile`.
1574fn write_stman_indirect_column(
1575    io: &mut AipsIo,
1576    col_desc: &ColumnDescContents,
1577    rows: &[casa_types::RecordValue],
1578    col_name: &str,
1579    array_writer: &mut Option<StManArrayFileWriter>,
1580) -> Result<(), StorageError> {
1581    let nrrow = rows.len() as u32;
1582    let dt = CasacoreDataType::from_primitive_type(col_desc.require_primitive_type()?, false);
1583
1584    let writer = array_writer.as_mut().ok_or_else(|| {
1585        StorageError::FormatMismatch("no array file writer for indirect column".to_string())
1586    })?;
1587
1588    // Write each row's array to the array file, collecting offsets.
1589    let mut offsets = Vec::with_capacity(nrrow as usize);
1590    for row in rows {
1591        match row.get(col_name) {
1592            Some(val @ Value::Array(_)) => {
1593                let offset = writer.write_array(val, dt)?;
1594                offsets.push(offset);
1595            }
1596            _ => {
1597                offsets.push(0i64); // undefined cell
1598            }
1599        }
1600    }
1601
1602    // Write AipsIO envelope.
1603    io.putstart("StManColumnIndArrayAipsIO", 2)?;
1604    io.put_i32(dt as i32)?; // dtype for backward compatibility
1605    io.put_i32(0)?; // seqnr (not significant for v2)
1606
1607    // Inner StManColumnAipsIO with file offsets as scalar data.
1608    io.putstart("StManColumnAipsIO", 2)?;
1609    io.put_u32(nrrow)?; // nrval
1610
1611    if nrrow > 0 {
1612        // Single extent: extent_count + individual u32 offsets (no count prefix).
1613        // C++ StManColumnIndArrayAipsIO::putData writes `ios << uInt(off)` per row.
1614        io.put_u32(nrrow)?; // extent_count
1615
1616        for &off in &offsets {
1617            if off == 0 {
1618                io.put_u32(0)?;
1619            } else if off <= 2u64.pow(31) as i64 {
1620                io.put_u32(off as u32)?;
1621            } else {
1622                io.put_u32(LARGE_OFFSET_MARKER)?;
1623                io.put_i64(off)?;
1624            }
1625        }
1626    }
1627
1628    io.putend()?; // StManColumnAipsIO
1629    io.putend()?; // StManColumnIndArrayAipsIO
1630
1631    Ok(())
1632}

1634/// Write a record column as `StManColumnIndArrayAipsIO` with `TpUChar` data.
1635///
1636/// Matches C++ `ScalarRecordColumnData` which calls `createIndArrColumn(name, TpUChar, "")`,
1637/// storing each record as an AipsIO-serialized `Vector<uChar>` in the indirect array file.
1638fn write_stman_record_as_indirect(
1639    io: &mut AipsIo,
1640    rows: &[casa_types::RecordValue],
1641    col_name: &str,
1642    array_writer: &mut Option<StManArrayFileWriter>,
1643) -> Result<(), StorageError> {
1644    let nrrow = rows.len() as u32;
1645    let dt = CasacoreDataType::TpUChar;
1646
1647    // Serialize each record to uChar bytes and write to the array file.
1648    let mut offsets = Vec::with_capacity(nrrow as usize);
1649    for row in rows {
1650        let record = match row.get(col_name) {
1651            Some(Value::Record(r)) => r.clone(),
1652            _ => RecordValue::default(),
1653        };
1654        let bytes = table_control::serialize_record_to_uchar(&record)?;
1655        // Write as 1-D Array<uChar> to the array file.
1656        let arr = ndarray::ArrayD::from_shape_vec(ndarray::IxDyn(&[bytes.len()]), bytes)
1657            .map_err(|e| StorageError::FormatMismatch(format!("record serialize shape: {e}")))?;
1658        let value = Value::Array(casa_types::ArrayValue::UInt8(arr));
1659        let writer = array_writer.as_mut().ok_or_else(|| {
1660            StorageError::FormatMismatch(
1661                "record column requires indirect array writer but none was created".into(),
1662            )
1663        })?;
1664        let offset = writer.write_array(&value, dt)?;
1665        offsets.push(offset);
1666    }
1667
1668    // Write AipsIO envelope: StManColumnIndArrayAipsIO.
1669    io.putstart("StManColumnIndArrayAipsIO", 2)?;
1670    io.put_i32(dt as i32)?; // dtype for backward compatibility
1671    io.put_i32(0)?; // seqnr
1672
1673    // Inner StManColumnAipsIO with file offsets.
1674    io.putstart("StManColumnAipsIO", 2)?;
1675    io.put_u32(nrrow)?;
1676
1677    if nrrow > 0 {
1678        io.put_u32(nrrow)?; // extent_count
1679        for &off in &offsets {
1680            if off == 0 {
1681                io.put_u32(0)?;
1682            } else if off <= 2u64.pow(31) as i64 {
1683                io.put_u32(off as u32)?;
1684            } else {
1685                io.put_u32(LARGE_OFFSET_MARKER)?;
1686                io.put_i64(off)?;
1687            }
1688        }
1689    }
1690
1691    io.putend()?; // StManColumnAipsIO
1692    io.putend()?; // StManColumnIndArrayAipsIO
1693
1694    Ok(())
1695}

1697fn write_scalar_column(
1698    io: &mut AipsIo,
1699    col_desc: &ColumnDescContents,
1700    rows: &[casa_types::RecordValue],
1701    col_name: &str,
1702    _col_idx: usize,
1703) -> Result<(), StorageError> {
1704    // Collect all scalar values into a typed vec and write as array
1705    let pt = col_desc.require_primitive_type()?;
1706    let allow_undefined = (col_desc.option & 2) != 0;
1707    match pt {
1708        PrimitiveType::Bool => {

2099/// Iterate over an ndarray in Fortran (column-major) order.
2100fn fortran_flat_iter<T: Clone>(arr: &ArrayD<T>) -> Vec<T> {
2101    let shape = arr.shape();
2102    if shape.is_empty() {
2103        return vec![];
2104    }
2105    let total: usize = shape.iter().product();
2106    let ndim = shape.len();
2107    let mut result = Vec::with_capacity(total);
2108    let mut indices = vec![0usize; ndim];
2109
2110    for _ in 0..total {
2111        result.push(arr[IxDyn(&indices)].clone());
2112        // Increment indices in Fortran order (first dimension changes fastest)
2113        for d in 0..ndim {
2114            indices[d] += 1;
2115            if indices[d] < shape[d] {
2116                break;
2117            }
2118            indices[d] = 0;
2119        }
2120    }
2121    result
