2//! Typed wrapper for the FIELD subtable.
3//!
4//! Provides read access to field names, positions (phase/delay/reference
5//! direction polynomials), and source IDs.
6//!
7//! Cf. C++ `MSFieldColumns`.
8
9use casacore_tables::Table;
10use casacore_types::{ArrayValue, ScalarValue};
11
12use crate::column_def::ColumnDef;
13use crate::error::MsResult;
14use crate::schema::{self, SubtableId};
15use crate::subtables::{
16    SubTable, get_array, get_bool, get_f64, get_i32, get_string, set_array, set_scalar,
17};
18
19/// Read-only typed wrapper for the FIELD subtable.
20pub struct MsField<'a> {
21    table: &'a Table,
22}
23
24impl<'a> MsField<'a> {
25    /// Wrap an existing table as a FIELD subtable.
26    pub fn new(table: &'a Table) -> Self {
27        Self { table }
28    }
29
30    /// Number of field rows.
31    pub fn row_count(&self) -> usize {
32        self.table.row_count()
33    }
34
35    /// Field name.
36    pub fn name(&self, row: usize) -> MsResult<String> {
37        get_string(self.table, row, "NAME")
38    }
39
40    /// Special characteristics code (e.g. `"C"` for calibrator).
41    pub fn code(&self, row: usize) -> MsResult<String> {
42        get_string(self.table, row, "CODE")
43    }
44
45    /// Polynomial order of the direction columns.
46    pub fn num_poly(&self, row: usize) -> MsResult<i32> {
47        get_i32(self.table, row, "NUM_POLY")
48    }
49
50    /// Delay direction as polynomial coefficients, shape `[2, NUM_POLY+1]`.
51    pub fn delay_dir(&self, row: usize) -> MsResult<&ArrayValue> {
52        get_array(self.table, row, "DELAY_DIR")
53    }
54
55    /// Phase direction as polynomial coefficients, shape `[2, NUM_POLY+1]`.
56    pub fn phase_dir(&self, row: usize) -> MsResult<&ArrayValue> {
57        get_array(self.table, row, "PHASE_DIR")
58    }
59
60    /// Reference direction as polynomial coefficients, shape `[2, NUM_POLY+1]`.
61    pub fn reference_dir(&self, row: usize) -> MsResult<&ArrayValue> {
62        get_array(self.table, row, "REFERENCE_DIR")
63    }
64
65    /// Source ID (index into SOURCE subtable, or -1 if not specified).
66    pub fn source_id(&self, row: usize) -> MsResult<i32> {
67        get_i32(self.table, row, "SOURCE_ID")
68    }
69
70    /// Time origin for the directions and rates (MJD seconds, UTC).
71    pub fn time(&self, row: usize) -> MsResult<f64> {
72        get_f64(self.table, row, "TIME")
73    }
74
75    /// Row flag.
76    pub fn flag_row(&self, row: usize) -> MsResult<bool> {
77        get_bool(self.table, row, "FLAG_ROW")
78    }
79}
80
81impl SubTable for MsField<'_> {
82    fn id() -> SubtableId {
83        SubtableId::Field
84    }
85    fn required_columns() -> &'static [ColumnDef] {
86        schema::field::REQUIRED_COLUMNS
87    }
88    fn optional_columns() -> &'static [ColumnDef] {
89        schema::field::OPTIONAL_COLUMNS
90    }
91    fn table(&self) -> &Table {
92        self.table
93    }
94}

96/// Mutable typed wrapper for the FIELD subtable.
97pub struct MsFieldMut<'a> {
98    table: &'a mut Table,
99}
100
101impl<'a> MsFieldMut<'a> {
102    /// Wrap an existing mutable table as a FIELD subtable.
103    pub fn new(table: &'a mut Table) -> Self {
104        Self { table }
105    }
106
107    /// Number of field rows.
108    pub fn row_count(&self) -> usize {
109        self.table.row_count()
110    }
111
112    /// View the table through the read-only wrapper.
113    pub fn as_ref(&self) -> MsField<'_> {
114        MsField { table: self.table }
115    }
116
117    /// Access the underlying mutable table.
118    pub fn table_mut(&mut self) -> &mut Table {
119        self.table
120    }
121
122    /// Set a scalar `String` column by name.
123    pub fn set_string(&mut self, row: usize, col: &str, value: impl Into<String>) -> MsResult<()> {
124        set_scalar(self.table, row, col, ScalarValue::String(value.into()))
125    }
126
127    /// Set a scalar `i32` column by name.
128    pub fn set_i32(&mut self, row: usize, col: &str, value: i32) -> MsResult<()> {
129        set_scalar(self.table, row, col, ScalarValue::Int32(value))
130    }
131
132    /// Set a scalar `f64` column by name.
133    pub fn set_f64(&mut self, row: usize, col: &str, value: f64) -> MsResult<()> {
134        set_scalar(self.table, row, col, ScalarValue::Float64(value))
135    }
136
137    /// Set a scalar `bool` column by name.
138    pub fn set_bool(&mut self, row: usize, col: &str, value: bool) -> MsResult<()> {
139        set_scalar(self.table, row, col, ScalarValue::Bool(value))
140    }
141
142    /// Set an array column by name.
143    pub fn set_array(&mut self, row: usize, col: &str, value: ArrayValue) -> MsResult<()> {
144        set_array(self.table, row, col, value)
145    }
146}
147
148impl SubTable for MsFieldMut<'_> {
149    fn id() -> SubtableId {
150        SubtableId::Field
151    }
152    fn required_columns() -> &'static [ColumnDef] {
153        schema::field::REQUIRED_COLUMNS
154    }
155    fn optional_columns() -> &'static [ColumnDef] {
156        schema::field::OPTIONAL_COLUMNS
157    }
158    fn table(&self) -> &Table {
159        self.table
160    }
161}

163#[cfg(test)]
164mod tests {
165    use super::*;
166    use crate::column_def::build_table_schema;
167    use casacore_types::{RecordField, RecordValue, Value};
168    use ndarray::ArrayD;
169
170    fn make_field_table() -> Table {
171        let schema = build_table_schema(schema::field::REQUIRED_COLUMNS).expect("valid schema");
172        Table::with_schema(schema)
173    }
174
175    fn make_dir_array(ra: f64, dec: f64) -> ArrayValue {
176        ArrayValue::Float64(ArrayD::from_shape_vec(vec![2, 1], vec![ra, dec]).unwrap())
177    }
178
179    #[test]
180    fn read_field_row() {
181        let mut table = make_field_table();
182        let dir = make_dir_array(1.0, 0.5);
183
184        let row = RecordValue::new(vec![
185            RecordField::new(
186                "CODE",
187                Value::Scalar(casacore_types::ScalarValue::String("T".to_string())),
188            ),
189            RecordField::new("DELAY_DIR", Value::Array(dir.clone())),
190            RecordField::new(
191                "FLAG_ROW",
192                Value::Scalar(casacore_types::ScalarValue::Bool(false)),
193            ),
194            RecordField::new(
195                "NAME",
196                Value::Scalar(casacore_types::ScalarValue::String("3C286".to_string())),
197            ),
198            RecordField::new(
199                "NUM_POLY",
200                Value::Scalar(casacore_types::ScalarValue::Int32(0)),
201            ),
202            RecordField::new("PHASE_DIR", Value::Array(dir.clone())),
203            RecordField::new("REFERENCE_DIR", Value::Array(dir)),
204            RecordField::new(
205                "SOURCE_ID",
206                Value::Scalar(casacore_types::ScalarValue::Int32(0)),
207            ),
208            RecordField::new(
209                "TIME",
210                Value::Scalar(casacore_types::ScalarValue::Float64(4.8e9)),
211            ),
212        ]);
213        table.add_row(row).unwrap();
214
215        let field = MsField::new(&table);
216        assert_eq!(field.row_count(), 1);
217        assert_eq!(field.name(0).unwrap(), "3C286");
218        assert_eq!(field.code(0).unwrap(), "T");
219        assert_eq!(field.num_poly(0).unwrap(), 0);
220        assert_eq!(field.source_id(0).unwrap(), 0);
221        assert!(!field.flag_row(0).unwrap());
222    }
223}
