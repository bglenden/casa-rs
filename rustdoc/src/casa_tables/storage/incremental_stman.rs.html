2//! IncrementalStMan (ISM) delta-compression storage manager reader/writer.
3//!
4//! The ISM stores column values only when they change from the previous row,
5//! making it space-efficient for slowly-changing columns (e.g. `ANTENNA1`,
6//! `FEED_ID`, `SCAN_NUMBER` in a MeasurementSet).
7//!
8//! On-disk format: `table.fN` file with 512-byte AipsIO header at offset 0,
9//! followed by fixed-size buckets, followed by an ISMIndex AipsIO object.
10//!
11//! The ISM file header, bucket data, and index all use the table's endian
12//! setting (BE for v4 tables, LE for v5). The DM blob in `table.dat` always
13//! uses big-endian AipsIO (file-based).
14//!
15//! C++ equivalent: `casacore/tables/DataMan/ISMBase`, `ISMBucket`, `ISMIndex`,
16//! `ISMColumn`.
17

72/// Minimal reader for AipsIO-framed data in either byte order.
73struct AipsIoBuf<'a> {
74    data: &'a [u8],
75    pos: usize,
76    order: ByteOrder,
77    level: usize,
78}
79
80impl<'a> AipsIoBuf<'a> {
81    fn new(data: &'a [u8], order: ByteOrder) -> Self {
82        Self {
83            data,
84            pos: 0,
85            order,
86            level: 0,
87        }
88    }
89
90    fn read_bytes(&mut self, n: usize) -> Result<&'a [u8], StorageError> {
91        if self.pos + n > self.data.len() {
92            return Err(StorageError::FormatMismatch(
93                "ISM AipsIO buffer underrun".to_string(),
94            ));
95        }
96        let slice = &self.data[self.pos..self.pos + n];
97        self.pos += n;
98        Ok(slice)
99    }
100
101    fn read_u32(&mut self) -> Result<u32, StorageError> {
102        let b = self.read_bytes(4)?;
103        Ok(match self.order {
104            ByteOrder::BigEndian => u32::from_be_bytes([b[0], b[1], b[2], b[3]]),
105            ByteOrder::LittleEndian => u32::from_le_bytes([b[0], b[1], b[2], b[3]]),
106        })
107    }
108
109    fn read_i32(&mut self) -> Result<i32, StorageError> {
110        let b = self.read_bytes(4)?;
111        Ok(match self.order {
112            ByteOrder::BigEndian => i32::from_be_bytes([b[0], b[1], b[2], b[3]]),
113            ByteOrder::LittleEndian => i32::from_le_bytes([b[0], b[1], b[2], b[3]]),
114        })
115    }
116
117    fn read_u64(&mut self) -> Result<u64, StorageError> {
118        let b = self.read_bytes(8)?;
119        Ok(match self.order {
120            ByteOrder::BigEndian => {
121                u64::from_be_bytes([b[0], b[1], b[2], b[3], b[4], b[5], b[6], b[7]])
122            }
123            ByteOrder::LittleEndian => {
124                u64::from_le_bytes([b[0], b[1], b[2], b[3], b[4], b[5], b[6], b[7]])
125            }
126        })
127    }
128
129    fn read_bool(&mut self) -> Result<bool, StorageError> {
130        let b = self.read_bytes(1)?;
131        Ok(b[0] != 0)
132    }
133
134    fn read_string(&mut self) -> Result<String, StorageError> {
135        let len = self.read_u32()? as usize;
136        let b = self.read_bytes(len)?;
137        String::from_utf8(b.to_vec())
138            .map_err(|e| StorageError::FormatMismatch(format!("invalid UTF-8 in ISM: {e}")))
139    }
140
141    fn getstart(&mut self, expected_type: &str) -> Result<u32, StorageError> {
142        if self.level == 0 {
143            let magic = self.read_u32()?;
144            if magic != AIPSIO_MAGIC {
145                return Err(StorageError::FormatMismatch(format!(
146                    "ISM AipsIO magic mismatch: expected 0x{AIPSIO_MAGIC:08x}, got 0x{magic:08x}"
147                )));
148            }
149        }
150        self.level += 1;
151        let _obj_len = self.read_u32()?;
152        let type_name = self.read_string()?;
153        if type_name != expected_type {
154            return Err(StorageError::FormatMismatch(format!(
155                "ISM AipsIO type mismatch: expected '{expected_type}', got '{type_name}'"
156            )));
157        }
158        self.read_u32()
159    }
160
161    fn getend(&mut self) {
162        if self.level > 0 {
163            self.level -= 1;
164        }
165    }
166
167    fn read_block_u32(&mut self) -> Result<Vec<u32>, StorageError> {
168        let _version = self.getstart("Block")?;
169        let count = self.read_u32()?;
170        let mut values = Vec::with_capacity(count as usize);
171        for _ in 0..count {
172            values.push(self.read_u32()?);
173        }
174        self.getend();
175        Ok(values)
176    }
177
178    fn read_block_u64(&mut self) -> Result<Vec<u64>, StorageError> {
179        let _version = self.getstart("Block")?;
180        let count = self.read_u32()?;
181        let mut values = Vec::with_capacity(count as usize);
182        for _ in 0..count {
183            values.push(self.read_u64()?);
184        }
185        self.getend();
186        Ok(values)
187    }
188}

190/// Detect the byte order of in-memory AipsIO data.
191fn detect_aipsio_byte_order(data: &[u8]) -> Result<ByteOrder, StorageError> {
192    if data.len() < 8 {
193        return Err(StorageError::FormatMismatch(
194            "ISM data too short for byte order detection".to_string(),
195        ));
196    }
197    let be_len = u32::from_be_bytes([data[4], data[5], data[6], data[7]]);
198    let le_len = u32::from_le_bytes([data[4], data[5], data[6], data[7]]);
199
200    let be_ok = be_len > 0 && be_len < 4096;
201    let le_ok = le_len > 0 && le_len < 4096;
202
203    match (be_ok, le_ok) {
204        (true, false) => Ok(ByteOrder::BigEndian),
205        (false, true) => Ok(ByteOrder::LittleEndian),
206        (true, true) => Ok(ByteOrder::BigEndian),
207        (false, false) => Err(StorageError::FormatMismatch(format!(
208            "ISM: cannot detect byte order (be_len={be_len}, le_len={le_len})"
209        ))),
210    }
211}

213/// Minimal writer for in-memory AipsIO in the given byte order.
214struct AipsIoWriteBuf {
215    data: Vec<u8>,
216    order: ByteOrder,
217    len_positions: Vec<usize>,
218    level: usize,
219}
220
221impl AipsIoWriteBuf {
222    fn new(order: ByteOrder) -> Self {
223        Self {
224            data: Vec::new(),
225            order,
226            len_positions: Vec::new(),
227            level: 0,
228        }
229    }
230
231    fn into_bytes(self) -> Vec<u8> {
232        self.data
233    }
234
235    fn put_u8(&mut self, val: u8) {
236        self.data.push(val);
237    }
238
239    fn put_u32(&mut self, val: u32) {
240        match self.order {
241            ByteOrder::BigEndian => self.data.extend_from_slice(&val.to_be_bytes()),
242            ByteOrder::LittleEndian => self.data.extend_from_slice(&val.to_le_bytes()),
243        }
244    }
245
246    fn put_i32(&mut self, val: i32) {
247        match self.order {
248            ByteOrder::BigEndian => self.data.extend_from_slice(&val.to_be_bytes()),
249            ByteOrder::LittleEndian => self.data.extend_from_slice(&val.to_le_bytes()),
250        }
251    }
252
253    fn put_u64(&mut self, val: u64) {
254        match self.order {
255            ByteOrder::BigEndian => self.data.extend_from_slice(&val.to_be_bytes()),
256            ByteOrder::LittleEndian => self.data.extend_from_slice(&val.to_le_bytes()),
257        }
258    }
259
260    fn put_bool(&mut self, val: bool) {
261        self.put_u8(if val { 1 } else { 0 });
262    }
263
264    fn put_string(&mut self, s: &str) {
265        self.put_u32(s.len() as u32);
266        self.data.extend_from_slice(s.as_bytes());
267    }
268
269    fn putstart(&mut self, type_name: &str, version: u32) {
270        self.put_u32(AIPSIO_MAGIC);
271        let pos = self.data.len();
272        self.put_u32(0);
273        self.len_positions.push(pos);
274        self.put_string(type_name);
275        self.put_u32(version);
276        self.level += 1;
277    }
278
279    fn putstart_nested(&mut self, type_name: &str, version: u32) {
280        let pos = self.data.len();
281        self.put_u32(0);
282        self.len_positions.push(pos);
283        self.put_string(type_name);
284        self.put_u32(version);
285        self.level += 1;
286    }
287
288    fn putend(&mut self) {
289        if let Some(pos) = self.len_positions.pop() {
290            let obj_len = (self.data.len() - pos) as u32;
291            let bytes = match self.order {
292                ByteOrder::BigEndian => obj_len.to_be_bytes(),
293                ByteOrder::LittleEndian => obj_len.to_le_bytes(),
294            };
295            self.data[pos..pos + 4].copy_from_slice(&bytes);
296        }
297        if self.level > 0 {
298            self.level -= 1;
299        }
300    }
301
302    fn putend_nested(&mut self) {
303        self.putend();
304    }
305}

342// ---------------------------------------------------------------------------
343// ISM-specific canonical element size (Bool = 1 byte, not bit-packed)
344// ---------------------------------------------------------------------------
345
346/// Returns the canonical byte size for one element in the ISM data area.
347///
348/// Unlike SSM (which bit-packs bools), ISM stores each Bool as 1 byte.
349/// Returns 0 for variable-length types (String).
350fn ism_element_size(dt: CasacoreDataType) -> usize {
351    match dt {
352        CasacoreDataType::TpBool => 1,
353        CasacoreDataType::TpUChar | CasacoreDataType::TpChar => 1,
354        CasacoreDataType::TpShort | CasacoreDataType::TpUShort => 2,
355        CasacoreDataType::TpInt | CasacoreDataType::TpUInt | CasacoreDataType::TpFloat => 4,
356        CasacoreDataType::TpDouble | CasacoreDataType::TpInt64 | CasacoreDataType::TpComplex => 8,
357        CasacoreDataType::TpDComplex => 16,
358        CasacoreDataType::TpString => 0, // variable-length
359        _ => 0,
360    }
361}

367fn parse_ism_header(file: &mut (impl Read + Seek)) -> Result<IsmHeader, StorageError> {
368    file.seek(SeekFrom::Start(0))?;
369    let mut header_buf = vec![0u8; ISM_HEADER_SIZE as usize];
370    file.read_exact(&mut header_buf)?;
371
372    let io_order = detect_aipsio_byte_order(&header_buf)?;
373    let mut buf = AipsIoBuf::new(&header_buf, io_order);
374    let version = buf.getstart("IncrementalStMan")?;
375
376    let big_endian = if version >= 5 {
377        buf.read_bool()?
378    } else {
379        true // v4 is always big-endian
380    };
381
382    let bucket_size = buf.read_u32()?;
383    let nr_buckets = buf.read_u32()?;
384    let _pers_cache_size = buf.read_u32()?;
385    let _uniq_nr = buf.read_u32()?;
386    let _nr_free_buckets = buf.read_u32()?;
387    let _first_free_bucket = buf.read_i32()?;
388
389    Ok(IsmHeader {
390        bucket_size,
391        nr_buckets,
392        big_endian,
393        io_order,
394    })
395}

410// ---------------------------------------------------------------------------
411// Index parsing (at end of file, after all buckets)
412// ---------------------------------------------------------------------------
413
414fn parse_ism_index(file: &mut File, header: &IsmHeader) -> Result<IsmIndex, StorageError> {
415    let index_offset = ISM_HEADER_SIZE + (header.nr_buckets as u64) * (header.bucket_size as u64);
416    file.seek(SeekFrom::Start(index_offset))?;
417
418    let mut index_data = Vec::new();
419    file.read_to_end(&mut index_data)?;
420
421    if index_data.is_empty() {
422        return Ok(IsmIndex {
423            rows: vec![0],
424            bucket_nrs: vec![],
425        });
426    }
427
428    let mut buf = AipsIoBuf::new(&index_data, header.io_order);
429    let version = buf.getstart("ISMIndex")?;
430
431    let _nused = buf.read_u32()?;
432
433    let rows = if version == 1 {
434        buf.read_block_u32()?
435            .into_iter()
436            .map(|v| v as u64)
437            .collect()
438    } else {
439        buf.read_block_u64()?
440    };
441
442    let bucket_nrs = buf.read_block_u32()?;
443    buf.getend();
444
445    Ok(IsmIndex { rows, bucket_nrs })
446}

464fn parse_ism_bucket(
465    raw: &[u8],
466    n_cols: usize,
467    big_endian: bool,
468) -> Result<IsmBucket, StorageError> {
469    if raw.len() < 4 {
470        return Err(StorageError::FormatMismatch(
471            "ISM bucket too small".to_string(),
472        ));
473    }
474
475    // Read index_offset from byte 0..3 in table byte order
476    let raw_offset = if big_endian {
477        read_u32_be(raw)
478    } else {
479        read_u32_le(raw)
480    };
481
482    // Bit 31 = 64-bit row numbers flag
483    if raw_offset & 0x80000000 != 0 {
484        return Err(StorageError::FormatMismatch(
485            "ISM 64-bit row numbers not supported".to_string(),
486        ));
487    }
488
489    let index_offset = (raw_offset & 0x7fffffff) as usize;
490
491    // Data area: bytes 4..index_offset
492    let data_end = index_offset.min(raw.len());
493    let data = raw[4..data_end].to_vec();
494
495    // Per-column index: bytes index_offset..end
496    let mut pos = index_offset;
497    let mut col_indices = Vec::with_capacity(n_cols);
498
499    for _ in 0..n_cols {
500        if pos + 4 > raw.len() {
501            break;
502        }
503        let n_values = if big_endian {
504            read_u32_be(&raw[pos..])
505        } else {
506            read_u32_le(&raw[pos..])
507        };
508        pos += 4;
509
510        let n = n_values as usize;
511
512        // Row numbers (u32 each)
513        let mut row_nrs = Vec::with_capacity(n);
514        for _ in 0..n {
515            if pos + 4 > raw.len() {
516                return Err(StorageError::FormatMismatch(
517                    "ISM bucket index overrun reading row_nrs".to_string(),
518                ));
519            }
520            row_nrs.push(if big_endian {
521                read_u32_be(&raw[pos..])
522            } else {
523                read_u32_le(&raw[pos..])
524            });
525            pos += 4;
526        }
527
528        // Offsets (u32 each)
529        let mut offsets = Vec::with_capacity(n);
530        for _ in 0..n {
531            if pos + 4 > raw.len() {
532                return Err(StorageError::FormatMismatch(
533                    "ISM bucket index overrun reading offsets".to_string(),
534                ));
535            }
536            offsets.push(if big_endian {
537                read_u32_be(&raw[pos..])
538            } else {
539                read_u32_le(&raw[pos..])
540            });
541            pos += 4;
542        }
543
544        col_indices.push(IsmBucketColIndex {
545            n_values,
546            row_nrs,
547            offsets,
548        });
549    }
550
551    Ok(IsmBucket { data, col_indices })
552}

558/// Find the interval index for `rel_row` in the per-column index.
559///
560/// Returns the index `k` such that `row_nrs[k] <= rel_row < row_nrs[k+1]`
561/// (or k is the last entry). Row 0 always has an entry, so the search
562/// always succeeds for valid relative row numbers.
563fn get_interval(col_index: &IsmBucketColIndex, rel_row: u32) -> usize {
564    // Binary search for the largest row_nr <= rel_row
565    let pos = col_index.row_nrs.partition_point(|&r| r <= rel_row);
566    if pos == 0 { 0 } else { pos - 1 }
567}

569/// Read a single scalar value from the ISM data area at the given offset.
570fn read_scalar_at(
571    data: &[u8],
572    offset: usize,
573    dt: CasacoreDataType,
574    big_endian: bool,
575) -> Result<casa_types::Value, StorageError> {
576    use casa_types::{ScalarValue, Value};
577
578    let d = &data[offset..];
579    let val = match dt {
580        CasacoreDataType::TpBool => Value::Scalar(ScalarValue::Bool(d[0] != 0)),
581        CasacoreDataType::TpUChar => Value::Scalar(ScalarValue::UInt8(d[0])),
582        CasacoreDataType::TpShort => Value::Scalar(ScalarValue::Int16(if big_endian {
583            read_i16_be(d)
584        } else {
585            read_i16_le(d)
586        })),
587        CasacoreDataType::TpUShort => Value::Scalar(ScalarValue::UInt16(if big_endian {
588            read_u16_be(d)
589        } else {
590            read_u16_le(d)
591        })),
592        CasacoreDataType::TpInt => Value::Scalar(ScalarValue::Int32(if big_endian {
593            read_i32_be(d)
594        } else {
595            read_i32_le(d)
596        })),
597        CasacoreDataType::TpUInt => Value::Scalar(ScalarValue::UInt32(if big_endian {
598            read_u32_be(d)
599        } else {
600            read_u32_le(d)
601        })),
602        CasacoreDataType::TpFloat => Value::Scalar(ScalarValue::Float32(if big_endian {
603            read_f32_be(d)
604        } else {
605            read_f32_le(d)
606        })),
607        CasacoreDataType::TpDouble => Value::Scalar(ScalarValue::Float64(if big_endian {
608            read_f64_be(d)
609        } else {
610            read_f64_le(d)
611        })),
612        CasacoreDataType::TpInt64 => Value::Scalar(ScalarValue::Int64(if big_endian {
613            read_i64_be(d)
614        } else {
615            read_i64_le(d)
616        })),
617        CasacoreDataType::TpComplex => {
618            let (re, im) = if big_endian {
619                (read_f32_be(d), read_f32_be(&d[4..]))
620            } else {
621                (read_f32_le(d), read_f32_le(&d[4..]))
622            };
623            Value::Scalar(ScalarValue::Complex32(casa_types::Complex32::new(re, im)))
624        }
625        CasacoreDataType::TpDComplex => {
626            let (re, im) = if big_endian {
627                (read_f64_be(d), read_f64_be(&d[8..]))
628            } else {
629                (read_f64_le(d), read_f64_le(&d[8..]))
630            };
631            Value::Scalar(ScalarValue::Complex64(casa_types::Complex64::new(re, im)))
632        }
633        CasacoreDataType::TpString => {
634            let s = read_ism_string(d, 1, big_endian)?;
635            Value::Scalar(ScalarValue::String(
636                s.into_iter().next().unwrap_or_default(),
637            ))
638        }
639        _ => {
640            return Err(StorageError::FormatMismatch(format!(
641                "unsupported ISM data type: {dt:?}"
642            )));
643        }
644    };
645    Ok(val)
646}

648/// Read ISM-encoded strings from the data area.
649///
650/// ISM string format (C++ `ISMColumn::fromString` / `toString`):
651///
652/// For scalar strings (nvalues == 1):
653///   `[u32: total_length] [string_bytes]`
654///   where `total_length` = 4 + string_length (includes the u32 itself).
655///
656/// For string arrays (nvalues > 1):
657///   `[u32: total_length] [per-string: [u32: str_len] [str_data]] ...`
658///   where `total_length` includes itself.
659fn read_ism_string(
660    data: &[u8],
661    nvalues: usize,
662    big_endian: bool,
663) -> Result<Vec<String>, StorageError> {
664    let total_length = if big_endian {
665        read_u32_be(data)
666    } else {
667        read_u32_le(data)
668    } as usize;
669
670    let mut pos = 4usize; // skip total_length field
671    let mut strings = Vec::with_capacity(nvalues);
672
673    if nvalues == 1 {
674        // Scalar: total_length includes itself, remaining bytes are the string.
675        let str_len = total_length.saturating_sub(4);
676        let s = String::from_utf8(data[pos..pos + str_len].to_vec())
677            .map_err(|e| StorageError::FormatMismatch(format!("ISM string not UTF-8: {e}")))?;
678        strings.push(s);
679    } else {
680        // Array: each string has its own u32 length prefix.
681        for _ in 0..nvalues {
682            if pos + 4 > total_length {
683                break;
684            }
685            let str_len = if big_endian {
686                read_u32_be(&data[pos..])
687            } else {
688                read_u32_le(&data[pos..])
689            } as usize;
690            pos += 4;
691
692            let s = String::from_utf8(data[pos..pos + str_len].to_vec())
693                .map_err(|e| StorageError::FormatMismatch(format!("ISM string not UTF-8: {e}")))?;
694            pos += str_len;
695            strings.push(s);
696        }
697    }
698
699    Ok(strings)
700}

702/// Read a fixed-shape array value from the ISM data area at the given offset.
703fn read_array_at(
704    data: &[u8],
705    offset: usize,
706    dt: CasacoreDataType,
707    nrelem: usize,
708    shape: &[i32],
709    big_endian: bool,
710) -> Result<casa_types::Value, StorageError> {
711    use casa_types::{ArrayValue, Value};
712    use ndarray::{ArrayD, IxDyn, ShapeBuilder};
713
714    let d = &data[offset..];
715    let shape_usize: Vec<usize> = shape.iter().map(|&s| s as usize).collect();
716
717    let arr_val = match dt {
718        CasacoreDataType::TpBool => {
719            // C++ ISM stores bools bit-packed (1 bit per element).
720            let v: Vec<bool> = (0..nrelem)
721                .map(|i| {
722                    let byte_idx = i / 8;
723                    let bit_idx = i % 8;
724                    (d[byte_idx] >> bit_idx) & 1 != 0
725                })
726                .collect();
727            ArrayValue::Bool(
728                ArrayD::from_shape_vec(IxDyn(&shape_usize).f(), v)
729                    .map_err(|e| StorageError::FormatMismatch(format!("array shape: {e}")))?,
730            )
731        }
732        CasacoreDataType::TpUChar => {
733            let v: Vec<u8> = d[..nrelem].to_vec();
734            ArrayValue::UInt8(
735                ArrayD::from_shape_vec(IxDyn(&shape_usize).f(), v)
736                    .map_err(|e| StorageError::FormatMismatch(format!("array shape: {e}")))?,
737            )
738        }
739        CasacoreDataType::TpShort => {
740            let v: Vec<i16> = (0..nrelem)
741                .map(|i| {
742                    if big_endian {
743                        read_i16_be(&d[i * 2..])
744                    } else {
745                        read_i16_le(&d[i * 2..])
746                    }
747                })
748                .collect();
749            ArrayValue::Int16(
750                ArrayD::from_shape_vec(IxDyn(&shape_usize).f(), v)
751                    .map_err(|e| StorageError::FormatMismatch(format!("array shape: {e}")))?,
752            )
753        }
754        CasacoreDataType::TpInt => {
755            let v: Vec<i32> = (0..nrelem)
756                .map(|i| {
757                    if big_endian {
758                        read_i32_be(&d[i * 4..])
759                    } else {
760                        read_i32_le(&d[i * 4..])
761                    }
762                })
763                .collect();
764            ArrayValue::Int32(
765                ArrayD::from_shape_vec(IxDyn(&shape_usize).f(), v)
766                    .map_err(|e| StorageError::FormatMismatch(format!("array shape: {e}")))?,
767            )
768        }
769        CasacoreDataType::TpFloat => {
770            let v: Vec<f32> = (0..nrelem)
771                .map(|i| {
772                    if big_endian {
773                        read_f32_be(&d[i * 4..])
774                    } else {
775                        read_f32_le(&d[i * 4..])
776                    }
777                })
778                .collect();
779            ArrayValue::Float32(
780                ArrayD::from_shape_vec(IxDyn(&shape_usize).f(), v)
781                    .map_err(|e| StorageError::FormatMismatch(format!("array shape: {e}")))?,
782            )
783        }
784        CasacoreDataType::TpDouble => {
785            let v: Vec<f64> = (0..nrelem)
786                .map(|i| {
787                    if big_endian {
788                        read_f64_be(&d[i * 8..])
789                    } else {
790                        read_f64_le(&d[i * 8..])
791                    }
792                })
793                .collect();
794            ArrayValue::Float64(
795                ArrayD::from_shape_vec(IxDyn(&shape_usize).f(), v)
796                    .map_err(|e| StorageError::FormatMismatch(format!("array shape: {e}")))?,
797            )
798        }
799        CasacoreDataType::TpInt64 => {
800            let v: Vec<i64> = (0..nrelem)
801                .map(|i| {
802                    if big_endian {
803                        read_i64_be(&d[i * 8..])
804                    } else {
805                        read_i64_le(&d[i * 8..])
806                    }
807                })
808                .collect();
809            ArrayValue::Int64(
810                ArrayD::from_shape_vec(IxDyn(&shape_usize).f(), v)
811                    .map_err(|e| StorageError::FormatMismatch(format!("array shape: {e}")))?,
812            )
813        }
814        CasacoreDataType::TpComplex => {
815            let v: Vec<casa_types::Complex32> = (0..nrelem)
816                .map(|i| {
817                    let (re, im) = if big_endian {
818                        (read_f32_be(&d[i * 8..]), read_f32_be(&d[i * 8 + 4..]))
819                    } else {
820                        (read_f32_le(&d[i * 8..]), read_f32_le(&d[i * 8 + 4..]))
821                    };
822                    casa_types::Complex32::new(re, im)
823                })
824                .collect();
825            ArrayValue::Complex32(
826                ArrayD::from_shape_vec(IxDyn(&shape_usize).f(), v)
827                    .map_err(|e| StorageError::FormatMismatch(format!("array shape: {e}")))?,
828            )
829        }
830        CasacoreDataType::TpDComplex => {
831            let v: Vec<casa_types::Complex64> = (0..nrelem)
832                .map(|i| {
833                    let (re, im) = if big_endian {
834                        (read_f64_be(&d[i * 16..]), read_f64_be(&d[i * 16 + 8..]))
835                    } else {
836                        (read_f64_le(&d[i * 16..]), read_f64_le(&d[i * 16 + 8..]))
837                    };
838                    casa_types::Complex64::new(re, im)
839                })
840                .collect();
841            ArrayValue::Complex64(
842                ArrayD::from_shape_vec(IxDyn(&shape_usize).f(), v)
843                    .map_err(|e| StorageError::FormatMismatch(format!("array shape: {e}")))?,
844            )
845        }
846        CasacoreDataType::TpString => {
847            let strings = read_ism_string(d, nrelem, big_endian)?;
848            ArrayValue::String(
849                ArrayD::from_shape_vec(IxDyn(&shape_usize).f(), strings)
850                    .map_err(|e| StorageError::FormatMismatch(format!("array shape: {e}")))?,
851            )
852        }
853        _ => {
854            return Err(StorageError::FormatMismatch(format!(
855                "unsupported ISM array type: {dt:?}"
856            )));
857        }
858    };
859    Ok(Value::Array(arr_val))
860}

881pub(crate) fn read_ism_file_columns(
882    file_path: &Path,
883    dm_blob: &[u8],
884    col_descs: &[(usize, &ColumnDescContents)],
885    total_column_count: usize,
886    nrrow: usize,
887) -> Result<Vec<(String, IsmColumnResult)>, StorageError> {
888    let mut file = File::open(file_path)?;
889    let header = parse_ism_header(&mut file)?;
890    let _dm_name = parse_ism_dm_blob(dm_blob)?;
891    let index = parse_ism_index(&mut file, &header)?;
892    let be = header.big_endian;
893
894    // Pre-compute column info
895    let col_info: Vec<(CasacoreDataType, usize, bool, bool)> = col_descs
896        .iter()
897        .map(|(_, c)| {
898            let scalar_dt =
899                CasacoreDataType::from_primitive_type(c.require_primitive_type()?, false);
900            let nrelem = if c.is_array && !c.shape.is_empty() {
901                c.shape.iter().map(|&s| s as usize).product()
902            } else {
903                1
904            };
905            Ok((
906                scalar_dt,
907                nrelem,
908                c.is_array && c.nrdim > 0 && !c.shape.is_empty(),
909                c.is_array && (c.option & 1) == 0 && c.shape.is_empty(),
910            ))
911        })
912        .collect::<Result<_, StorageError>>()?;
913
914    let mut array_reader = if col_descs.iter().any(|(_, col_desc)| {
915        col_desc.is_array && (col_desc.option & 1) == 0 && col_desc.shape.is_empty()
916    }) {
917        let array_path = PathBuf::from(format!("{}i", file_path.display()));
918        if !array_path.is_file() {
919            return Err(StorageError::MissingDataFile(array_path));
920        }
921        Some(StManArrayFileReader::open(&array_path, be)?)
922    } else {
923        None
924    };
925
926    // Initialize column-major result vectors
927    let ncol = col_descs.len();
928    let mut flat_columns: Vec<Vec<casa_types::Value>> =
929        (0..ncol).map(|_| Vec::with_capacity(nrrow)).collect();
930    let mut indirect_columns: Vec<Option<Vec<Option<casa_types::Value>>>> = col_descs
931        .iter()
932        .map(|(_, col_desc)| {
933            (col_desc.is_array && (col_desc.option & 1) == 0 && col_desc.shape.is_empty())
934                .then(|| Vec::with_capacity(nrrow))
935        })
936        .collect();
937
938    // Iterate over bucket intervals from the ISMIndex
939    let n_intervals = index.bucket_nrs.len();
940    let mut last_bucket_nr: Option<u32> = None;
941    let mut cached_bucket: Option<IsmBucket> = None;
942
943    for interval in 0..n_intervals {
944        let bucket_start = index.rows[interval] as usize;
945        let bucket_end = index.rows[interval + 1] as usize;
946        let rows_in_bucket = bucket_end - bucket_start;
947        let bucket_nr = index.bucket_nrs[interval];
948
949        // Load bucket (cache for repeated access)
950        if last_bucket_nr != Some(bucket_nr) {
951            let raw = read_ism_bucket(&mut file, &header, bucket_nr)?;
952            cached_bucket = Some(parse_ism_bucket(&raw, total_column_count, be)?);
953            last_bucket_nr = Some(bucket_nr);
954        }
955        let bucket = cached_bucket.as_ref().unwrap();
956
957        // For each column, expand the interval values
958        for (out_col_idx, &(source_col_idx, col_desc)) in col_descs.iter().enumerate() {
959            let (dt, nrelem, is_direct_fixed_array, is_indirect_array) = col_info[out_col_idx];
960
961            if source_col_idx >= bucket.col_indices.len() {
962                // Column not in this bucket — fill with defaults
963                for _ in 0..rows_in_bucket {
964                    if is_indirect_array {
965                        indirect_columns[out_col_idx]
966                            .as_mut()
967                            .expect("indirect ISM column storage")
968                            .push(None);
969                    } else {
970                        flat_columns[out_col_idx].push(default_value(dt, false, col_desc));
971                    }
972                }
973                continue;
974            }
975
976            let col_index = &bucket.col_indices[source_col_idx];
977
978            // Expand each row in this bucket's interval
979            for rel_row in 0..rows_in_bucket {
980                let k = get_interval(col_index, rel_row as u32);
981                let data_offset = col_index.offsets[k] as usize;
982
983                if is_indirect_array {
984                    let offset_value =
985                        read_scalar_at(&bucket.data, data_offset, CasacoreDataType::TpInt64, be)?;
986                    let offset = match offset_value {
987                        casa_types::Value::Scalar(casa_types::ScalarValue::Int64(value)) => value,
988                        other => {
989                            return Err(StorageError::FormatMismatch(format!(
990                                "ISM indirect column '{}' expected Int64 offset, got {other:?}",
991                                col_desc.col_name
992                            )));
993                        }
994                    };
995                    let value = array_reader
996                        .as_mut()
997                        .expect("indirect ISM array reader")
998                        .read_array_at(offset, dt)?;
999                    indirect_columns[out_col_idx]
1000                        .as_mut()
1001                        .expect("indirect ISM column storage")
1002                        .push(value);
1003                } else {
1004                    let value = if is_direct_fixed_array {
1005                        read_array_at(&bucket.data, data_offset, dt, nrelem, &col_desc.shape, be)?
1006                    } else {
1007                        read_scalar_at(&bucket.data, data_offset, dt, be)?
1008                    };
1009                    flat_columns[out_col_idx].push(value);
1010                }
1011            }
1012        }
1013    }
1014
1015    // Convert per-column results into the shared storage-layer form.
1016    let mut result = Vec::with_capacity(ncol);
1017    for (col_idx, &(_, col_desc)) in col_descs.iter().enumerate() {
1018        let (dt, _nrelem, is_direct_fixed_array, is_indirect_array) = col_info[col_idx];
1019        if is_indirect_array {
1020            result.push((
1021                col_desc.col_name.clone(),
1022                IsmColumnResult::Indirect(
1023                    indirect_columns[col_idx]
1024                        .take()
1025                        .expect("indirect ISM column storage"),
1026                ),
1027            ));
1028        } else {
1029            let values = &flat_columns[col_idx];
1030            let raw = if is_direct_fixed_array {
1031                values_to_column_raw_array(values, dt, col_desc)?
1032            } else {
1033                values_to_column_raw_scalar(values, dt)?
1034            };
1035            result.push((col_desc.col_name.clone(), IsmColumnResult::Flat(raw)));
1036        }
1037    }
1038
1039    Ok(result)
1040}

1085/// Read selected required scalar columns without expanding every physical
1086/// column or every row in an incremental bucket.
1087pub(crate) fn read_ism_required_scalar_columns_rows(
1088    file_path: &Path,
1089    dm_blob: &[u8],
1090    all_col_descs: &[&ColumnDescContents],
1091    requested_col_descs: &[(usize, &ColumnDescContents)],
1092    selected_rows: &[usize],
1093) -> Result<HashMap<String, RequiredScalarColumnData>, StorageError> {
1094    let _dm_name = parse_ism_dm_blob(dm_blob)?;
1095    let mut outputs = HashMap::with_capacity(requested_col_descs.len());
1096    for (_, col_desc) in requested_col_descs {
1097        let values = match col_desc.data_type {
1098            CasacoreDataType::TpBool => {
1099                RequiredScalarColumnData::Bool(vec![false; selected_rows.len()])
1100            }
1101            CasacoreDataType::TpInt => {
1102                RequiredScalarColumnData::Int32(vec![0; selected_rows.len()])
1103            }
1104            CasacoreDataType::TpFloat => {
1105                RequiredScalarColumnData::Float32(vec![0.0; selected_rows.len()])
1106            }
1107            CasacoreDataType::TpDouble => {
1108                RequiredScalarColumnData::Float64(vec![0.0; selected_rows.len()])
1109            }
1110            other => {
1111                return Err(StorageError::FormatMismatch(format!(
1112                    "required selected-row ISM column '{}' has unsupported type {other:?}",
1113                    col_desc.col_name
1114                )));
1115            }
1116        };
1117        outputs.insert(col_desc.col_name.clone(), values);
1118    }
1119    if selected_rows.is_empty() || outputs.is_empty() {
1120        return Ok(outputs);
1121    }
1122
1123    let mut file = File::open(file_path)?;
1124    let header = parse_ism_header(&mut file)?;
1125    let index = parse_ism_index(&mut file, &header)?;
1126    let contiguous_rows = selected_rows
1127        .windows(2)
1128        .all(|rows| rows[1] == rows[0].saturating_add(1));
1129    if contiguous_rows {
1130        let selected_end = selected_rows[0]
1131            .checked_add(selected_rows.len())
1132            .ok_or_else(|| StorageError::FormatMismatch("ISM row range overflow".to_string()))?;
1133        let mut row_index = selected_rows[0];
1134        let mut output_start = 0usize;
1135        while row_index < selected_end {
1136            let interval = index
1137                .rows
1138                .partition_point(|&start| start <= row_index as u64);
1139            if interval == 0 || interval >= index.rows.len() {
1140                return Err(StorageError::FormatMismatch(format!(
1141                    "ISM index has no bucket for row {row_index}"
1142                )));
1143            }
1144            let interval_idx = interval - 1;
1145            let bucket_start = index.rows[interval_idx] as usize;
1146            let bucket_end = (index.rows[interval_idx + 1] as usize).min(selected_end);
1147            let bucket_nr = *index.bucket_nrs.get(interval_idx).ok_or_else(|| {
1148                StorageError::FormatMismatch(format!(
1149                    "ISM bucket number missing for row interval {interval_idx}"
1150                ))
1151            })?;
1152            let raw_bucket = read_ism_bucket(&mut file, &header, bucket_nr)?;
1153            let bucket = parse_ism_bucket(&raw_bucket, all_col_descs.len(), header.big_endian)?;
1154            let relative_start = row_index - bucket_start;
1155            let row_count = bucket_end - row_index;
1156            for (col_idx, col_desc) in requested_col_descs {
1157                let col_index = bucket.col_indices.get(*col_idx).ok_or_else(|| {
1158                    StorageError::FormatMismatch(format!(
1159                        "required ISM column '{}' missing bucket index {col_idx}",
1160                        col_desc.col_name
1161                    ))
1162                })?;
1163                let output = outputs
1164                    .get_mut(&col_desc.col_name)
1165                    .expect("required ISM output initialized before range decode");
1166                fill_ism_required_scalar_range(
1167                    output,
1168                    &bucket.data,
1169                    col_index,
1170                    relative_start,
1171                    output_start,
1172                    row_count,
1173                    header.big_endian,
1174                )?;
1175            }
1176            row_index = bucket_end;
1177            output_start += row_count;
1178        }
1179        return Ok(outputs);
1180    }
1181    let mut requests: Vec<(usize, usize)> = selected_rows
1182        .iter()
1183        .copied()
1184        .enumerate()
1185        .map(|(out_idx, row_index)| (row_index, out_idx))
1186        .collect();
1187    requests.sort_unstable_by_key(|&(row_index, _)| row_index);
1188
1189    let mut cached_interval = None;
1190    let mut cached_bucket_start = 0usize;
1191    let mut cached_bucket = None;
1192    for (row_index, out_idx) in requests {
1193        let interval = index
1194            .rows
1195            .partition_point(|&start| start <= row_index as u64);
1196        if interval == 0 || interval >= index.rows.len() {
1197            return Err(StorageError::FormatMismatch(format!(
1198                "ISM index has no bucket for row {row_index}"
1199            )));
1200        }
1201        let interval_idx = interval - 1;
1202        if cached_interval != Some(interval_idx) {
1203            let bucket_nr = *index.bucket_nrs.get(interval_idx).ok_or_else(|| {
1204                StorageError::FormatMismatch(format!(
1205                    "ISM bucket number missing for row interval {interval_idx}"
1206                ))
1207            })?;
1208            cached_interval = Some(interval_idx);
1209            cached_bucket_start = index.rows[interval_idx] as usize;
1210            let raw_bucket = read_ism_bucket(&mut file, &header, bucket_nr)?;
1211            cached_bucket = Some(parse_ism_bucket(
1212                &raw_bucket,
1213                all_col_descs.len(),
1214                header.big_endian,
1215            )?);
1216        }
1217        let bucket = cached_bucket
1218            .as_ref()
1219            .expect("ISM bucket loaded before selected scalar decode");
1220        let relative_row = row_index.saturating_sub(cached_bucket_start) as u32;
1221        for (col_idx, col_desc) in requested_col_descs {
1222            let col_index = bucket.col_indices.get(*col_idx).ok_or_else(|| {
1223                StorageError::FormatMismatch(format!(
1224                    "required ISM column '{}' missing bucket index {col_idx}",
1225                    col_desc.col_name
1226                ))
1227            })?;
1228            let value_interval = get_interval(col_index, relative_row);
1229            let offset = *col_index.offsets.get(value_interval).ok_or_else(|| {
1230                StorageError::FormatMismatch(format!(
1231                    "required ISM column '{}' missing value offset",
1232                    col_desc.col_name
1233                ))
1234            })? as usize;
1235            let output = outputs
1236                .get_mut(&col_desc.col_name)
1237                .expect("required ISM output initialized before decode");
1238            match output {
1239                RequiredScalarColumnData::Bool(values) => {
1240                    values[out_idx] = bucket.data[offset] != 0;
1241                }
1242                RequiredScalarColumnData::Int32(values) => {
1243                    values[out_idx] = if header.big_endian {
1244                        read_i32_be(&bucket.data[offset..])
1245                    } else {
1246                        read_i32_le(&bucket.data[offset..])
1247                    };
1248                }
1249                RequiredScalarColumnData::Float32(values) => {
1250                    values[out_idx] = if header.big_endian {
1251                        read_f32_be(&bucket.data[offset..])
1252                    } else {
1253                        read_f32_le(&bucket.data[offset..])
1254                    };
1255                }
1256                RequiredScalarColumnData::Float64(values) => {
1257                    values[out_idx] = if header.big_endian {
1258                        read_f64_be(&bucket.data[offset..])
1259                    } else {
1260                        read_f64_le(&bucket.data[offset..])
1261                    };
1262                }
1263            }
1264        }
1265    }
1266    Ok(outputs)
1267}

1269#[allow(clippy::too_many_arguments)]
1270fn fill_ism_required_scalar_range(
1271    output: &mut RequiredScalarColumnData,
1272    bucket_data: &[u8],
1273    col_index: &IsmBucketColIndex,
1274    source_row_start: usize,
1275    output_row_start: usize,
1276    row_count: usize,
1277    big_endian: bool,
1278) -> Result<(), StorageError> {
1279    let source_end = source_row_start + row_count;
1280    let mut source_row = source_row_start;
1281    let mut output_row = output_row_start;
1282    while source_row < source_end {
1283        let interval = get_interval(col_index, source_row as u32);
1284        let run_end = col_index
1285            .row_nrs
1286            .get(interval + 1)
1287            .map_or(source_end, |row| *row as usize)
1288            .min(source_end);
1289        let run_rows = run_end - source_row;
1290        let offset = *col_index.offsets.get(interval).ok_or_else(|| {
1291            StorageError::FormatMismatch("ISM required scalar run has no offset".to_string())
1292        })? as usize;
1293        match output {
1294            RequiredScalarColumnData::Bool(values) => {
1295                values[output_row..output_row + run_rows].fill(bucket_data[offset] != 0);
1296            }
1297            RequiredScalarColumnData::Int32(values) => {
1298                let value = if big_endian {
1299                    read_i32_be(&bucket_data[offset..])
1300                } else {
1301                    read_i32_le(&bucket_data[offset..])
1302                };
1303                values[output_row..output_row + run_rows].fill(value);
1304            }
1305            RequiredScalarColumnData::Float32(values) => {
1306                let value = if big_endian {
1307                    read_f32_be(&bucket_data[offset..])
1308                } else {
1309                    read_f32_le(&bucket_data[offset..])
1310                };
1311                values[output_row..output_row + run_rows].fill(value);
1312            }
1313            RequiredScalarColumnData::Float64(values) => {
1314                let value = if big_endian {
1315                    read_f64_be(&bucket_data[offset..])
1316                } else {
1317                    read_f64_le(&bucket_data[offset..])
1318                };
1319                values[output_row..output_row + run_rows].fill(value);
1320            }
1321        }
1322        source_row = run_end;
1323        output_row += run_rows;
1324    }
1325    Ok(())
1326}

1328pub(crate) fn read_ism_scalar_column(
1329    file_path: &Path,
1330    dm_blob: &[u8],
1331    col_descs: &[&ColumnDescContents],
1332    target_col_idx: usize,
1333    nrrow: usize,
1334) -> Result<Option<Vec<Option<ScalarValue>>>, StorageError> {
1335    if target_col_idx >= col_descs.len() {
1336        return Err(StorageError::FormatMismatch(format!(
1337            "ISM scalar column index {target_col_idx} is out of range"
1338        )));
1339    }
1340    let target_col_desc = col_descs[target_col_idx];
1341    if target_col_desc.is_array || target_col_desc.is_record() {
1342        return Ok(None);
1343    }
1344
1345    let _dm_name = parse_ism_dm_blob(dm_blob)?;
1346    let mut file = File::open(file_path)?;
1347    let header = parse_ism_header(&mut file)?;
1348    let index = parse_ism_index(&mut file, &header)?;
1349    let big_endian = header.big_endian;
1350    let scalar_dt =
1351        CasacoreDataType::from_primitive_type(target_col_desc.require_primitive_type()?, false);
1352
1353    let mut values = Vec::with_capacity(nrrow);
1354    let mut last_bucket_nr = None;
1355    let mut cached_bucket = None;
1356    for interval in 0..index.bucket_nrs.len() {
1357        let bucket_start = index.rows[interval] as usize;
1358        let bucket_end = index.rows[interval + 1] as usize;
1359        let rows_in_bucket = bucket_end.saturating_sub(bucket_start);
1360        let bucket_nr = index.bucket_nrs[interval];
1361        if last_bucket_nr != Some(bucket_nr) {
1362            let raw_bucket = read_ism_bucket(&mut file, &header, bucket_nr)?;
1363            cached_bucket = Some(parse_ism_bucket(&raw_bucket, col_descs.len(), big_endian)?);
1364            last_bucket_nr = Some(bucket_nr);
1365        }
1366        let bucket = cached_bucket
1367            .as_ref()
1368            .expect("ISM bucket loaded before scalar column extraction");
1369        values.extend(load_scalar_bucket_column_values(
1370            bucket,
1371            target_col_desc,
1372            target_col_idx,
1373            scalar_dt,
1374            rows_in_bucket,
1375            big_endian,
1376        )?);
1377    }
1378
1379    values.truncate(nrrow);
1380    if values.len() < nrrow {
1381        values.resize(nrrow, None);
1382    }
1383    Ok(Some(values))
1384}

1432fn read_ism_scalar_columns_rows_uncached(
1433    file_path: &Path,
1434    dm_blob: &[u8],
1435    col_descs: &[&ColumnDescContents],
1436    selected_rows: &[usize],
1437) -> Result<Vec<Vec<Option<ScalarValue>>>, StorageError> {
1438    let _dm_name = parse_ism_dm_blob(dm_blob)?;
1439    let mut file = File::open(file_path)?;
1440    let header = parse_ism_header(&mut file)?;
1441    let index = parse_ism_index(&mut file, &header)?;
1442    let big_endian = header.big_endian;
1443
1444    let owned_col_descs: Vec<ColumnDescContents> =
1445        col_descs.iter().map(|col| (*col).clone()).collect();
1446    let col_info: Vec<Option<CasacoreDataType>> = owned_col_descs
1447        .iter()
1448        .map(|col_desc| {
1449            if col_desc.is_array || col_desc.is_record() {
1450                return Ok(None);
1451            }
1452            Ok(Some(CasacoreDataType::from_primitive_type(
1453                col_desc.require_primitive_type()?,
1454                false,
1455            )))
1456        })
1457        .collect::<Result<_, StorageError>>()?;
1458
1459    let mut requests: Vec<(usize, usize)> = selected_rows
1460        .iter()
1461        .copied()
1462        .enumerate()
1463        .map(|(out_idx, row_index)| (row_index, out_idx))
1464        .collect();
1465    requests.sort_unstable_by_key(|&(row_index, _)| row_index);
1466
1467    let mut cached_interval = None;
1468    let mut cached_bucket_start = 0usize;
1469    let mut cached_bucket_values: Vec<Vec<Option<ScalarValue>>> = Vec::new();
1470    let mut values_by_column = (0..col_descs.len())
1471        .map(|_| vec![None; selected_rows.len()])
1472        .collect::<Vec<_>>();
1473
1474    for (row_index, out_idx) in requests {
1475        let interval = index
1476            .rows
1477            .partition_point(|&start| start <= row_index as u64);
1478        if interval == 0 || interval >= index.rows.len() {
1479            return Err(StorageError::FormatMismatch(format!(
1480                "ISM index has no bucket for row {row_index}"
1481            )));
1482        }
1483        let interval_idx = interval - 1;
1484        if interval_idx >= index.bucket_nrs.len() {
1485            return Err(StorageError::FormatMismatch(format!(
1486                "ISM bucket number missing for row interval {interval_idx}"
1487            )));
1488        }
1489
1490        if cached_interval != Some(interval_idx) {
1491            cached_interval = Some(interval_idx);
1492            cached_bucket_start = index.rows[interval_idx] as usize;
1493            let bucket_end = index.rows[interval_idx + 1] as usize;
1494            let raw_bucket = read_ism_bucket(&mut file, &header, index.bucket_nrs[interval_idx])?;
1495            let bucket = parse_ism_bucket(&raw_bucket, owned_col_descs.len(), big_endian)?;
1496            let rows_in_bucket = bucket_end.saturating_sub(cached_bucket_start);
1497            cached_bucket_values = owned_col_descs
1498                .iter()
1499                .zip(col_info.iter())
1500                .enumerate()
1501                .map(|(col_idx, (col_desc, scalar_dt))| match scalar_dt {
1502                    Some(scalar_dt) => load_scalar_bucket_column_values(
1503                        &bucket,
1504                        col_desc,
1505                        col_idx,
1506                        *scalar_dt,
1507                        rows_in_bucket,
1508                        big_endian,
1509                    ),
1510                    None => Ok(vec![None; rows_in_bucket]),
1511                })
1512                .collect::<Result<Vec<_>, StorageError>>()?;
1513        }
1514
1515        let bucket_row = row_index.saturating_sub(cached_bucket_start);
1516        for (column_values, bucket_column_values) in
1517            values_by_column.iter_mut().zip(cached_bucket_values.iter())
1518        {
1519            column_values[out_idx] = bucket_column_values
1520                .get(bucket_row)
1521                .cloned()
1522                .unwrap_or(None);
1523        }
1524    }
1525
1526    Ok(values_by_column)
1527}

2188/// Encode string value(s) in ISM format (C++ `ISMColumn::fromString`):
2189///
2190/// Scalar (nrelem == 1): `[u32: total_length] [string_bytes]`
2191///   where total_length = 4 + string.len().
2192///
2193/// Array (nrelem > 1): `[u32: total_length] [per-string: [u32: str_len] [str_data]] ...`
2194///   where total_length includes itself.
2195fn encode_string_value(
2196    value: Option<&casa_types::Value>,
2197    nrelem: usize,
2198    big_endian: bool,
2199) -> Vec<u8> {
2200    use casa_types::{ArrayValue, ScalarValue, Value};
2201
2202    let strings: Vec<String> = if nrelem == 1 {
2203        match value {
2204            Some(Value::Scalar(ScalarValue::String(s))) => vec![s.clone()],
2205            _ => vec![String::new()],
2206        }
2207    } else {
2208        match value {
2209            Some(Value::Array(ArrayValue::String(arr))) => arr.iter().cloned().collect(),
2210            _ => vec![String::new(); nrelem],
2211        }
2212    };
2213
2214    if nrelem == 1 {
2215        // Scalar: total_length = 4 + string.len() (no per-string length prefix)
2216        let s = &strings[0];
2217        let total_length = (4 + s.len()) as u32;
2218        let mut buf = Vec::with_capacity(total_length as usize);
2219        if big_endian {
2220            buf.extend_from_slice(&total_length.to_be_bytes());
2221        } else {
2222            buf.extend_from_slice(&total_length.to_le_bytes());
2223        }
2224        buf.extend_from_slice(s.as_bytes());
2225        buf
2226    } else {
2227        // Array: total_length + per-string [length] [data]
2228        let total_length: usize = 4 + strings.iter().map(|s| 4 + s.len()).sum::<usize>();
2229        let mut buf = Vec::with_capacity(total_length);
2230        if big_endian {
2231            buf.extend_from_slice(&(total_length as u32).to_be_bytes());
2232        } else {
2233            buf.extend_from_slice(&(total_length as u32).to_le_bytes());
2234        }
2235        for s in &strings {
2236            let len = s.len() as u32;
2237            if big_endian {
2238                buf.extend_from_slice(&len.to_be_bytes());
2239            } else {
2240                buf.extend_from_slice(&len.to_le_bytes());
2241            }
2242            buf.extend_from_slice(s.as_bytes());
2243        }
2244        buf
2245    }
2246}

2445        };
2446        bucket_start_rows.push(0);
2447
2448        for row_idx in 0..nrrow {

2465                .col_indices
2466                .iter()
2467                .map(|(row_nrs, _)| 4 + (row_nrs.len() + 1) * 4 + (row_nrs.len() + 1) * 4)

2470                4 + current.data.len() + new_data_estimate + current_index_size + ncol * 12;
2471
2472            if rel_row > 0 && total_estimate > bucket_size as usize {
2473                write_bucket(&mut file, &current)?;

2475                current = BucketBuilder {
2476                    data: Vec::new(),
2477                    col_indices: (0..ncol).map(|_| (Vec::new(), Vec::new())).collect(),
2478                };
2479                bucket_start_rows.push(row_idx);

2486                }

2488            }
2489

2492                    let offset = current.data.len() as u32;
2493                    current.data.extend_from_slice(&encoded);
2494                    current.col_indices[col_idx].0.push(rel_row);
2495                    current.col_indices[col_idx].1.push(offset);

2497                }
2498            }
2499        }
2500
2501        write_bucket(&mut file, &current)?;
2502        nr_buckets += 1;
2503    }

2506    let mut ism_bucket_nrs: Vec<u32> = Vec::with_capacity(nr_buckets);

2523fn write_ism_file_scalar_run_columns(
2524    file_path: &Path,
2525    nrrow: usize,
2526    col_info: &[(CasacoreDataType, usize)],
2527    scalar_runs: &[&[GeneratedScalarValueRun]],
2528    bucket_size: u32,
2529    big_endian: bool,
2530) -> Result<Vec<u8>, StorageError> {
2531    let ncol = col_info.len();
2532    if scalar_runs.len() != ncol {
2533        return Err(StorageError::FormatMismatch(format!(
2534            "ISM scalar run column count mismatch: expected {ncol}, got {}",
2535            scalar_runs.len()
2536        )));
2537    }
2538
2539    #[derive(Debug)]
2540    struct BucketBuilder {
2541        data: Vec<u8>,
2542        col_indices: Vec<(Vec<u32>, Vec<u32>)>,
2543    }
2544
2545    fn new_bucket(encoded_values: &[Vec<u8>]) -> BucketBuilder {
2546        let mut bucket = BucketBuilder {
2547            data: Vec::new(),
2548            col_indices: (0..encoded_values.len())
2549                .map(|_| (Vec::new(), Vec::new()))
2550                .collect(),
2551        };
2552        for (col_idx, encoded) in encoded_values.iter().enumerate() {
2553            let offset = bucket.data.len() as u32;
2554            bucket.data.extend_from_slice(encoded);
2555            bucket.col_indices[col_idx].0.push(0);
2556            bucket.col_indices[col_idx].1.push(offset);
2557        }
2558        bucket
2559    }
2560
2561    fn index_area_len(bucket: &BucketBuilder, extra_entries: &[bool]) -> usize {
2562        bucket
2563            .col_indices
2564            .iter()
2565            .enumerate()
2566            .map(|(col_idx, (row_nrs, _))| {
2567                let entries = row_nrs.len() + usize::from(extra_entries[col_idx]);
2568                4 + entries * 4 + entries * 4
2569            })
2570            .sum()
2571    }
2572
2573    let mut buckets = Vec::<BucketBuilder>::new();
2574    let mut bucket_start_rows = Vec::<usize>::new();
2575
2576    if nrrow > 0 {
2577        let mut current_values = Vec::with_capacity(ncol);
2578        for (col_idx, runs) in scalar_runs.iter().enumerate() {
2579            let Some(first_run) = runs.first() else {
2580                return Err(StorageError::FormatMismatch(format!(
2581                    "ISM scalar run column {col_idx} has no initial run"
2582                )));
2583            };
2584            if first_run.start_row() != 0 {
2585                return Err(StorageError::FormatMismatch(format!(
2586                    "ISM scalar run column {col_idx} starts at row {}, expected 0",
2587                    first_run.start_row()
2588                )));
2589            }
2590            let (dt, nrelem) = col_info[col_idx];
2591            current_values.push(encode_scalar_column_value_bytes(
2592                first_run.value(),
2593                dt,
2594                nrelem,
2595                big_endian,
2596            ));
2597        }
2598
2599        let mut next_run_indices = vec![1usize; ncol];
2600        let mut bucket_start = 0usize;
2601        bucket_start_rows.push(bucket_start);
2602        let mut current = new_bucket(&current_values);
2603
2604        loop {
2605            let next_row = scalar_runs
2606                .iter()
2607                .zip(&next_run_indices)
2608                .filter_map(|(runs, &idx)| runs.get(idx).map(GeneratedScalarValueRun::start_row))
2609                .filter(|&row| row < nrrow)
2610                .min();
2611            let Some(row) = next_row else {
2612                break;
2613            };
2614
2615            let rel_row = row.checked_sub(bucket_start).ok_or_else(|| {
2616                StorageError::FormatMismatch("ISM scalar run bucket row underflow".to_string())
2617            })? as u32;
2618            let mut changed = Vec::new();
2619            let mut extra_entries = vec![false; ncol];
2620            for col_idx in 0..ncol {
2621                let idx = next_run_indices[col_idx];
2622                if scalar_runs[col_idx]
2623                    .get(idx)
2624                    .is_some_and(|run| run.start_row() == row)
2625                {
2626                    let run = &scalar_runs[col_idx][idx];
2627                    let (dt, nrelem) = col_info[col_idx];
2628                    let encoded =
2629                        encode_scalar_column_value_bytes(run.value(), dt, nrelem, big_endian);
2630                    extra_entries[col_idx] = true;
2631                    changed.push((col_idx, encoded));
2632                }
2633            }
2634
2635            let new_data_bytes: usize = changed.iter().map(|(_, encoded)| encoded.len()).sum();
2636            let total_estimate =
2637                4 + current.data.len() + new_data_bytes + index_area_len(&current, &extra_entries);
2638
2639            if rel_row > 0 && total_estimate > bucket_size as usize {
2640                buckets.push(current);
2641                for (col_idx, encoded) in changed {
2642                    current_values[col_idx] = encoded;
2643                    next_run_indices[col_idx] += 1;
2644                }
2645                bucket_start = row;
2646                bucket_start_rows.push(bucket_start);
2647                current = new_bucket(&current_values);
2648                continue;
2649            }
2650
2651            for (col_idx, encoded) in changed {
2652                let offset = current.data.len() as u32;
2653                current.data.extend_from_slice(&encoded);
2654                current.col_indices[col_idx].0.push(rel_row);
2655                current.col_indices[col_idx].1.push(offset);
2656                current_values[col_idx] = encoded;
2657                next_run_indices[col_idx] += 1;
2658            }
2659        }
2660
2661        buckets.push(current);
2662    }
2663
2664    let nr_buckets = buckets.len();
2665    let mut raw_buckets = Vec::with_capacity(nr_buckets);
2666    for bucket in &buckets {
2667        let mut raw = vec![0u8; bucket_size as usize];
2668        let data_len = bucket.data.len().min(bucket_size as usize - 4);
2669        raw[4..4 + data_len].copy_from_slice(&bucket.data[..data_len]);
2670
2671        let mut index_area = Vec::new();
2672        for (row_nrs, offsets) in &bucket.col_indices {
2673            let n = row_nrs.len() as u32;
2674            if big_endian {
2675                index_area.extend_from_slice(&n.to_be_bytes());
2676                for &r in row_nrs {
2677                    index_area.extend_from_slice(&r.to_be_bytes());
2678                }
2679                for &o in offsets {
2680                    index_area.extend_from_slice(&o.to_be_bytes());
2681                }
2682            } else {
2683                index_area.extend_from_slice(&n.to_le_bytes());
2684                for &r in row_nrs {
2685                    index_area.extend_from_slice(&r.to_le_bytes());
2686                }
2687                for &o in offsets {
2688                    index_area.extend_from_slice(&o.to_le_bytes());
2689                }
2690            }
2691        }
2692
2693        let index_start = bucket_size as usize - index_area.len();
2694        let index_offset = index_start as u32;
2695        if big_endian {
2696            raw[0..4].copy_from_slice(&index_offset.to_be_bytes());
2697        } else {
2698            raw[0..4].copy_from_slice(&index_offset.to_le_bytes());
2699        }
2700        raw[index_start..index_start + index_area.len()].copy_from_slice(&index_area);
2701        raw_buckets.push(raw);
2702    }
2703
2704    let mut ism_rows = Vec::with_capacity(nr_buckets + 1);
2705    let mut ism_bucket_nrs = Vec::with_capacity(nr_buckets);
2706    for (idx, &start_row) in bucket_start_rows.iter().enumerate() {
2707        ism_rows.push(start_row as u64);
2708        ism_bucket_nrs.push(idx as u32);
2709    }
2710    ism_rows.push(nrrow as u64);
2711
2712    let index_data = serialize_ism_index(&ism_rows, &ism_bucket_nrs, big_endian);
2713    let header_buf = serialize_ism_header(bucket_size, nr_buckets as u32, big_endian);
2714
2715    let mut file = File::create(file_path)?;
2716    file.write_all(&header_buf)?;
2717    for raw in &raw_buckets {
2718        file.write_all(raw)?;
2719    }
2720    file.write_all(&index_data)?;
2721
2722    serialize_ism_dm_blob("ISM")
2723}

2757fn write_ism_file_impl(
2758    file_path: &Path,
2759    col_descs: &[ColumnDescContents],
2760    rows: &[casa_types::RecordValue],
2761    field_indices: Option<&[usize]>,
2762    big_endian: bool,
2763) -> Result<Vec<u8>, StorageError> {
2764    let nrrow = rows.len();
2765    let ncol = col_descs.len();
2766
2767    // Pre-compute column info
2768    let col_info: Vec<(CasacoreDataType, usize)> = col_descs
2769        .iter()
2770        .map(|c| {
2771            let dt = CasacoreDataType::from_primitive_type(c.require_primitive_type()?, false);
2772            let nrelem = if c.is_array && !c.shape.is_empty() {
2773                c.shape.iter().map(|&s| s as usize).product()
2774            } else {
2775                1
2776            };
2777            Ok((dt, nrelem))
2778        })
2779        .collect::<Result<_, StorageError>>()?;
2780
2781    // Compute fixed bytes per row for bucket sizing
2782    let fixed_bytes_per_row: usize = col_info
2783        .iter()
2784        .map(|&(dt, nrelem)| {
2785            if dt == CasacoreDataType::TpBool {
2786                nrelem.div_ceil(8) // bit-packed
2787            } else {
2788                let elem = ism_element_size(dt);
2789                if elem > 0 {
2790                    elem * nrelem
2791                } else {
2792                    // String: estimate 12 bytes per string value
2793                    nrelem * 12
2794                }
2795            }
2796        })
2797        .sum();
2798
2799    // Target ~32 rows per bucket, plus overhead for per-column index
2800    // Index overhead per row: ncol * (4 + 4) bytes worst case (row_nr + offset per entry)
2801    let index_overhead_per_row = ncol * 8;
2802    let bytes_per_row = fixed_bytes_per_row + index_overhead_per_row;
2803    let bucket_size = if bytes_per_row == 0 {
2804        128u32
2805    } else {
2806        let target = (bytes_per_row * 32 + 4) as u32; // +4 for index_offset field
2807        target.clamp(128, 327680)
2808    };
2809
2810    // Build buckets with delta compression
2811    struct BucketBuilder {
2812        data: Vec<u8>,
2813        col_indices: Vec<(Vec<u32>, Vec<u32>)>, // (row_nrs, offsets) per column
2814    }
2815
2816    let mut buckets: Vec<BucketBuilder> = Vec::new();
2817    let mut bucket_start_rows: Vec<usize> = Vec::new();
2818    let mut last_values: Vec<Vec<u8>> = vec![Vec::new(); ncol];
2819
2820    if nrrow > 0 {
2821        // Start first bucket
2822        let mut current = BucketBuilder {
2823            data: Vec::new(),
2824            col_indices: (0..ncol).map(|_| (Vec::new(), Vec::new())).collect(),
2825        };
2826        bucket_start_rows.push(0);
2827
2828        for (row_idx, row) in rows.iter().enumerate().take(nrrow) {
2829            let rel_row = (row_idx - *bucket_start_rows.last().unwrap()) as u32;
2830
2831            // Check if adding this row would overflow the bucket
2832            // Estimate: current data size + worst-case new data + index size
2833            let mut new_data_estimate = 0usize;
2834            for (col_idx, col_desc) in col_descs.iter().enumerate() {
2835                let (dt, nrelem) = col_info[col_idx];
2836                let value = row_value_for_column(row, field_indices, col_idx, col_desc);
2837                let encoded = encode_value_bytes(value, dt, nrelem, big_endian);
2838                if rel_row == 0 || encoded != last_values[col_idx] {
2839                    new_data_estimate += encoded.len();
2840                }
2841            }
2842
2843            let current_index_size: usize = current
2844                .col_indices
2845                .iter()
2846                .map(|(rn, _)| 4 + (rn.len() + 1) * 4 + (rn.len() + 1) * 4)
2847                .sum();
2848            let total_estimate =
2849                4 + current.data.len() + new_data_estimate + current_index_size + ncol * 12;
2850
2851            if rel_row > 0 && total_estimate > bucket_size as usize {
2852                // Finalize current bucket and start a new one
2853                buckets.push(current);
2854                current = BucketBuilder {
2855                    data: Vec::new(),
2856                    col_indices: (0..ncol).map(|_| (Vec::new(), Vec::new())).collect(),
2857                };
2858                bucket_start_rows.push(row_idx);
2859                // Carry over last values as first entries in new bucket
2860                for (col_idx, col_desc) in col_descs.iter().enumerate() {
2861                    let (dt, nrelem) = col_info[col_idx];
2862                    let bytes = if last_values[col_idx].is_empty() {
2863                        let value = row_value_for_column(row, field_indices, col_idx, col_desc);
2864                        encode_value_bytes(value, dt, nrelem, big_endian)
2865                    } else {
2866                        last_values[col_idx].clone()
2867                    };
2868                    let offset = current.data.len() as u32;
2869                    current.data.extend_from_slice(&bytes);
2870                    current.col_indices[col_idx].0.push(0); // row 0
2871                    current.col_indices[col_idx].1.push(offset);
2872                }
2873                // Now process the current row as rel_row 0
2874                // but we already wrote carried-over values, so check if current
2875                // row differs from carried-over
2876                for (col_idx, col_desc) in col_descs.iter().enumerate() {
2877                    let (dt, nrelem) = col_info[col_idx];
2878                    let value = row_value_for_column(row, field_indices, col_idx, col_desc);
2879                    let encoded = encode_value_bytes(value, dt, nrelem, big_endian);
2880                    // The carried-over value was already written at row 0.
2881                    // If current row differs from carried-over, we need to update.
2882                    if encoded != last_values[col_idx] {
2883                        let offset = current.data.len() as u32;
2884                        current.data.extend_from_slice(&encoded);
2885                        // Row 0 already has an entry from carry-over.
2886                        // Replace it since the actual row 0 value is different.
2887                        let idx = &mut current.col_indices[col_idx];
2888                        idx.0[0] = 0;
2889                        idx.1[0] = offset;
2890                        last_values[col_idx] = encoded;
2891                    }
2892                }
2893                continue;
2894            }
2895
2896            // Add this row's data to current bucket
2897            for (col_idx, col_desc) in col_descs.iter().enumerate() {
2898                let (dt, nrelem) = col_info[col_idx];
2899                let value = row_value_for_column(row, field_indices, col_idx, col_desc);
2900                let encoded = encode_value_bytes(value, dt, nrelem, big_endian);
2901
2902                if rel_row == 0 || encoded != last_values[col_idx] {
2903                    let offset = current.data.len() as u32;
2904                    current.data.extend_from_slice(&encoded);
2905                    current.col_indices[col_idx].0.push(rel_row);
2906                    current.col_indices[col_idx].1.push(offset);
2907                    last_values[col_idx] = encoded;
2908                }
2909            }
2910        }
2911        // Finalize last bucket
2912        buckets.push(current);
2913    }
2914
2915    // Serialize buckets
2916    let nr_buckets = buckets.len();
2917    let mut raw_buckets: Vec<Vec<u8>> = Vec::with_capacity(nr_buckets);
2918
2919    for bucket in &buckets {
2920        let mut raw = vec![0u8; bucket_size as usize];
2921
2922        // Write data area starting at byte 4
2923        let data_len = bucket.data.len().min(bucket_size as usize - 4);
2924        raw[4..4 + data_len].copy_from_slice(&bucket.data[..data_len]);
2925
2926        // Build and write per-column index
2927        let mut index_area = Vec::new();
2928        for (row_nrs, offsets) in &bucket.col_indices {
2929            let n = row_nrs.len() as u32;
2930            if big_endian {
2931                index_area.extend_from_slice(&n.to_be_bytes());
2932                for &r in row_nrs {
2933                    index_area.extend_from_slice(&r.to_be_bytes());
2934                }
2935                for &o in offsets {
2936                    index_area.extend_from_slice(&o.to_be_bytes());
2937                }
2938            } else {
2939                index_area.extend_from_slice(&n.to_le_bytes());
2940                for &r in row_nrs {
2941                    index_area.extend_from_slice(&r.to_le_bytes());
2942                }
2943                for &o in offsets {
2944                    index_area.extend_from_slice(&o.to_le_bytes());
2945                }
2946            }
2947        }
2948
2949        // index_offset = position where per-column index starts
2950        let index_start = bucket_size as usize - index_area.len();
2951        let index_offset = index_start as u32;
2952
2953        // Write index_offset at byte 0
2954        if big_endian {
2955            raw[0..4].copy_from_slice(&index_offset.to_be_bytes());
2956        } else {
2957            raw[0..4].copy_from_slice(&index_offset.to_le_bytes());
2958        }
2959
2960        // Write per-column index at the end of bucket
2961        if index_start + index_area.len() <= raw.len() {
2962            raw[index_start..index_start + index_area.len()].copy_from_slice(&index_area);
2963        }
2964
2965        raw_buckets.push(raw);
2966    }
2967
2968    // Build ISMIndex
2969    let mut ism_rows: Vec<u64> = Vec::with_capacity(nr_buckets + 1);
2970    let mut ism_bucket_nrs: Vec<u32> = Vec::with_capacity(nr_buckets);
2971    for (i, &start) in bucket_start_rows.iter().enumerate() {
2972        ism_rows.push(start as u64);
2973        ism_bucket_nrs.push(i as u32);
2974    }
2975    ism_rows.push(nrrow as u64); // sentinel
2976
2977    let index_data = serialize_ism_index(&ism_rows, &ism_bucket_nrs, big_endian);
2978
2979    // Write header
2980    let header_buf = serialize_ism_header(bucket_size, nr_buckets as u32, big_endian);
2981
2982    // Assemble the file
2983    let mut file = File::create(file_path)?;
2984    file.write_all(&header_buf)?;
2985    for raw in &raw_buckets {
2986        file.write_all(raw)?;
2987    }
2988    file.write_all(&index_data)?;
2989
2990    // Generate the DM data blob
2991    let dm_blob = serialize_ism_dm_blob("ISM")?;
2992
2993    Ok(dm_blob)
2994}
