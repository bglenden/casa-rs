2//! Unit registry: lookup, caching, and user-defined units.
3//!
4//! [`UnitRegistry`] holds all known unit definitions and provides the lookup
5//! that the parser uses to resolve unit names.  It is the Rust counterpart
6//! of C++ `casa::UnitMap`.
7//!
8//! The global registry is initialised once from the built-in tables in
9//! [`registry_data`] and extended at runtime via
10//! [`UnitRegistry::put_user`] and [`UnitRegistry::remove_user`].
11
12use std::collections::HashMap;
13use std::sync::{OnceLock, RwLock};
14
15use crate::quanta::dim::UnitDim;
16use crate::quanta::registry_data;
17use crate::quanta::unit_val::UnitVal;
18
19/// A named unit definition: name, value, and description.
20///
21/// Corresponds to C++ `casa::UnitName`.
22#[derive(Debug, Clone)]
23pub struct UnitName {
24    /// Short name (e.g. `"km"`, `"Jy"`).
25    pub name: String,
26    /// Resolved value (factor + dimensions).
27    pub val: UnitVal,
28    /// Human-readable description.
29    pub description: String,
30}
31
32impl UnitName {
33    /// Creates a new `UnitName`.
34    pub fn new(name: impl Into<String>, val: UnitVal, description: impl Into<String>) -> Self {
35        Self {
36            name: name.into(),
37            val,
38            description: description.into(),
39        }
40    }
41}
42
43/// A prefix entry: name, scale factor, and description.
44#[derive(Debug, Clone)]
45pub struct PrefixEntry {
46    /// The prefix string (e.g. `"k"`, `"da"`).
47    pub name: String,
48    /// The multiplicative factor (e.g. `1e3` for "k").
49    pub factor: f64,
50    /// Human-readable description (e.g. `"kilo"`).
51    pub description: String,
52}
53
54/// The global unit registry.
55///
56/// Contains immutable built-in maps (defining, SI, customary, prefixes) and
57/// a mutable section for user-defined units and parse cache.
58pub struct UnitRegistry {
59    /// SI prefixes, keyed by prefix string.
60    prefixes: HashMap<String, PrefixEntry>,
61    /// Defining (base SI) units.
62    defining: HashMap<String, UnitName>,
63    /// SI derived + extra units.
64    si: HashMap<String, UnitName>,
65    /// Customary (non-SI) units.
66    customary: HashMap<String, UnitName>,
67    /// User-defined units (mutable at runtime).
68    user: RwLock<HashMap<String, UnitName>>,
69    /// Parse cache (mutable at runtime).
70    cache: RwLock<HashMap<String, UnitVal>>,
71}
72
73impl UnitRegistry {
74    /// Builds a new registry from the built-in data tables.
75    fn from_builtin() -> Self {
76        let mut prefixes = HashMap::new();
77        for &(name, factor, desc) in registry_data::PREFIXES {
78            prefixes.insert(
79                name.to_owned(),
80                PrefixEntry {
81                    name: name.to_owned(),
82                    factor,
83                    description: desc.to_owned(),
84                },
85            );
86        }
87
88        let mut defining = HashMap::new();
89        for &(name, factor, desc, dims) in registry_data::DEFINING {
90            defining.insert(
91                name.to_owned(),
92                UnitName::new(name, UnitVal::new(factor, UnitDim::new(dims)), desc),
93            );
94        }
95
96        let mut si = HashMap::new();
97        for &(name, factor, desc, dims) in registry_data::SI_DERIVED {
98            si.insert(
99                name.to_owned(),
100                UnitName::new(name, UnitVal::new(factor, UnitDim::new(dims)), desc),
101            );
102        }
103        for &(name, factor, desc, dims) in registry_data::SI_EXTRA {
104            si.insert(
105                name.to_owned(),
106                UnitName::new(name, UnitVal::new(factor, UnitDim::new(dims)), desc),
107            );
108        }
109
110        let mut customary = HashMap::new();
111        for &(name, factor, desc, dims) in registry_data::CUSTOMARY {
112            customary.insert(
113                name.to_owned(),
114                UnitName::new(name, UnitVal::new(factor, UnitDim::new(dims)), desc),
115            );
116        }
117
118        Self {
119            prefixes,
120            defining,
121            si,
122            customary,
123            user: RwLock::new(HashMap::new()),
124            cache: RwLock::new(HashMap::new()),
125        }
126    }
127
128    /// Looks up a unit by name.
129    ///
130    /// Search order: defining → SI → customary → user → cache.
131    /// The immutable built-in maps are checked first (lock-free) before
132    /// falling back to the `RwLock`-protected user and cache maps.
133    pub fn lookup_unit(&self, name: &str) -> Option<UnitVal> {
134        // Check immutable built-in maps first (no locking needed).
135        if let Some(entry) = self.defining.get(name) {
136            return Some(entry.val);
137        }
138        if let Some(entry) = self.si.get(name) {
139            return Some(entry.val);
140        }
141        if let Some(entry) = self.customary.get(name) {
142            return Some(entry.val);
143        }
144        // Check cache (compound/prefixed expressions that were previously parsed).
145        if let Some(val) = self.cache.read().unwrap().get(name) {
146            return Some(*val);
147        }
148        // User-defined units (rarely used).
149        if let Some(entry) = self.user.read().unwrap().get(name) {
150            return Some(entry.val);
151        }
152        None
153    }
154
155    /// Looks up a prefix by name.
156    pub fn lookup_prefix(&self, name: &str) -> Option<f64> {
157        self.prefixes.get(name).map(|p| p.factor)
158    }
159
160    /// Adds a parsed unit expression to the cache.
161    pub fn cache_put(&self, name: &str, val: UnitVal) {
162        self.cache.write().unwrap().insert(name.to_owned(), val);
163    }
164
165    /// Registers a user-defined unit.
166    ///
167    /// If a unit with the same name already exists in the user map it is
168    /// replaced.  Built-in units are not affected.
169    pub fn put_user(&self, name: impl Into<String>, val: UnitVal, desc: impl Into<String>) {
170        let name = name.into();
171        self.user
172            .write()
173            .unwrap()
174            .insert(name.clone(), UnitName::new(name, val, desc));
175        // Invalidate the cache since user units affect lookup.
176        self.cache.write().unwrap().clear();
177    }
178
179    /// Removes a user-defined unit.
180    ///
181    /// Returns `true` if the unit existed and was removed.
182    pub fn remove_user(&self, name: &str) -> bool {
183        let removed = self.user.write().unwrap().remove(name).is_some();
184        if removed {
185            self.cache.write().unwrap().clear();
186        }
187        removed
188    }
189
190    /// Clears the parse cache.
191    pub fn clear_cache(&self) {
192        self.cache.write().unwrap().clear();
193    }
194}
195
196/// Global singleton registry.
197static REGISTRY: OnceLock<UnitRegistry> = OnceLock::new();
198
199/// Returns a reference to the global [`UnitRegistry`].
200///
201/// The registry is initialised on first access from the built-in data tables
202/// in [`registry_data`].
203pub fn global_registry() -> &'static UnitRegistry {
204    REGISTRY.get_or_init(UnitRegistry::from_builtin)
205}

207#[cfg(test)]
208mod tests {
209    use super::*;
210
211    #[test]
212    fn lookup_base_units() {
213        let reg = global_registry();
214        let m = reg.lookup_unit("m").expect("metre");
215        assert_eq!(m.factor, 1.0);
216        assert_eq!(m.dim.get(crate::quanta::dim::Dimension::Length), 1);
217
218        let kg = reg.lookup_unit("kg").expect("kilogram");
219        assert_eq!(kg.factor, 1.0);
220        assert_eq!(kg.dim.get(crate::quanta::dim::Dimension::Mass), 1);
221    }
222
223    #[test]
224    fn lookup_si_derived() {
225        let reg = global_registry();
226        let hz = reg.lookup_unit("Hz").expect("hertz");
227        assert_eq!(hz.factor, 1.0);
228        assert_eq!(hz.dim.get(crate::quanta::dim::Dimension::Time), -1);
229    }
230
231    #[test]
232    fn lookup_customary() {
233        let reg = global_registry();
234        let jy = reg.lookup_unit("Jy").expect("jansky");
235        assert!((jy.factor - 1e-26).abs() < 1e-40);
236    }
237
238    #[test]
239    fn lookup_prefix() {
240        let reg = global_registry();
241        assert_eq!(reg.lookup_prefix("k"), Some(1e3));
242        assert_eq!(reg.lookup_prefix("da"), Some(1e1));
243        assert_eq!(reg.lookup_prefix("x"), None);
244    }
245
246    #[test]
247    fn user_units() {
248        let reg = global_registry();
249        reg.put_user("test_unit", UnitVal::new(42.0, UnitDim::default()), "test");
250        let val = reg.lookup_unit("test_unit").expect("user unit");
251        assert_eq!(val.factor, 42.0);
252        assert!(reg.remove_user("test_unit"));
253        assert!(reg.lookup_unit("test_unit").is_none());
254    }
255}
