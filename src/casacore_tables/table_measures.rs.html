2//! Measure column support: storing/reading measures in table columns via MEASINFO.
3//!
4//! This module provides the Rust counterparts of C++ `TableMeasDesc`,
5//! `ScalarMeasColumn`, and `ArrayMeasColumn` from the casacore
6//! `measures/TableMeasures` subsystem.
7//!
8//! A "measure column" is a regular numeric array column whose cells carry
9//! physical measure metadata. The metadata is stored in the column's keyword
10//! record under the `MEASINFO` sub-record:
11//!
12//! - **Fixed reference**: `{ type: "epoch", Ref: "UTC" }`
13//! - **Variable Int reference**: `{ type: "epoch", VarRefCol: "TimeRef",
14//!   TabRefTypes: ["LAST","LMST",...], TabRefCodes: [0, 1, ...] }`
15//! - **Variable String reference**: `{ type: "epoch", VarRefCol: "TimeRefStr" }`
16//!
17//! Optional offset fields (`RefOffMsr` for fixed, `RefOffCol` for variable)
18//! encode measure offsets.
19//!
20//! # Relationship to `table_quantum`
21//!
22//! MEASINFO and `QuantumUnits` coexist in the same column keyword record.
23//! A column can have both measure metadata (reference frame) and quantum
24//! metadata (physical units). The `table_quantum` module handles units;
25//! this module handles reference frames.
26//!
27//! # Examples
28//!
29//! ```rust
30//! use casacore_tables::{Table, TableSchema, ColumnSchema, table_measures::*};
31//! use casacore_types::*;
32//!
33//! // Create a table with an epoch column
34//! let schema = TableSchema::new(vec![
35//!     ColumnSchema::array_fixed("TIME", PrimitiveType::Float64, vec![1]),
36//! ]).unwrap();
37//! let mut table = Table::with_schema(schema);
38//!
39//! // Attach fixed-reference MEASINFO
40//! let desc = TableMeasDesc::new_fixed("TIME", MeasureType::Epoch, "UTC");
41//! desc.write(&mut table).unwrap();
42//! assert!(TableMeasDesc::has_measinfo(&table, "TIME"));
43//! ```
44

56const MEASINFO_KW: &str = "MEASINFO";
57const TYPE_KW: &str = "type";
58const REF_KW: &str = "Ref";
59const VAR_REF_COL_KW: &str = "VarRefCol";
60const TAB_REF_TYPES_KW: &str = "TabRefTypes";
61const TAB_REF_CODES_KW: &str = "TabRefCodes";
62const REF_OFF_MSR_KW: &str = "RefOffMsr";
63const REF_OFF_COL_KW: &str = "RefOffCol";

65// ─── MeasureType ────────────────────────────────────────────────────────────

67/// The kind of measure stored in a column.
68///
69/// Corresponds to the `type` field in the `MEASINFO` sub-record.
70#[derive(Debug, Clone, Copy, PartialEq, Eq)]
71pub enum MeasureType {
72    /// Epoch (time instant).
73    Epoch,
74    /// Sky direction.
75    Direction,
76    /// 3D spatial position.
77    Position,
78    /// Spectral frequency.
79    Frequency,
80    /// Doppler shift.
81    Doppler,
82    /// Radial velocity.
83    RadialVelocity,
84}
85
86impl MeasureType {
87    /// Returns the C++ casacore string name for this measure type.
88    pub fn as_str(self) -> &'static str {
89        match self {
90            Self::Epoch => "epoch",
91            Self::Direction => "direction",
92            Self::Position => "position",
93            Self::Frequency => "frequency",
94            Self::Doppler => "doppler",
95            Self::RadialVelocity => "radialvelocity",
96        }
97    }
98
99    /// Parses a measure type from its C++ casacore string name.
100    pub fn from_str_casacore(s: &str) -> Option<Self> {
101        match s.to_lowercase().as_str() {
102            "epoch" => Some(Self::Epoch),
103            "direction" => Some(Self::Direction),
104            "position" => Some(Self::Position),
105            "frequency" => Some(Self::Frequency),
106            "doppler" => Some(Self::Doppler),
107            "radialvelocity" => Some(Self::RadialVelocity),
108            _ => None,
109        }
110    }
111
112    /// Returns the number of Double values per measure for this type.
113    ///
114    /// Epoch=1, Direction=2, Position=3, Frequency=1, Doppler=1,
115    /// RadialVelocity=1.
116    pub fn values_per_measure(self) -> usize {
117        match self {
118            Self::Epoch | Self::Frequency | Self::Doppler | Self::RadialVelocity => 1,
119            Self::Direction => 2,
120            Self::Position => 3,
121        }
122    }
123
124    /// Returns the default unit strings for this measure type.
125    ///
126    /// C++ casacore's `TableMeasDesc::write` writes these via `TableQuantumDesc`
127    /// so that `ArrayMeasColumn` can find them.
128    pub fn default_units(self) -> Vec<String> {
129        match self {
130            Self::Epoch => vec!["d".into()],
131            Self::Direction => vec!["rad".into(), "rad".into()],
132            Self::Position => vec!["m".into(), "m".into(), "m".into()],
133            Self::Frequency => vec!["Hz".into()],
134            Self::Doppler => vec!["m/s".into()],
135            Self::RadialVelocity => vec!["m/s".into()],
136        }
137    }
138}

142/// Describes how the reference frame for a measure column is determined.
143#[derive(Debug, Clone)]
144pub enum MeasRefDesc {
145    /// All rows share a single fixed reference frame.
146    Fixed {
147        /// The reference frame name (e.g. "UTC", "J2000").
148        refer: String,
149    },
150    /// Each row has an integer code stored in a companion column, with
151    /// explicit mapping tables.
152    VariableInt {
153        /// Name of the companion Int column.
154        ref_column: String,
155        /// String names for each code (e.g. `["LAST", "LMST", ..., "UTC", ...]`).
156        tab_ref_types: Vec<String>,
157        /// Integer codes for each entry (e.g. `[0, 1, ..., 6, ...]`).
158        tab_ref_codes: Vec<i32>,
159    },
160    /// Each row has a string reference stored in a companion column.
161    VariableString {
162        /// Name of the companion String column.
163        ref_column: String,
164    },
165}

169/// Describes how the offset for a measure column is determined.
170#[derive(Debug, Clone)]
171pub enum MeasOffsetDesc {
172    /// A fixed offset stored as a serialized measure record.
173    Fixed {
174        /// The serialized measure (same format as `MeasureHolder`).
175        offset: RecordValue,
176    },
177    /// A variable offset read from a companion column.
178    Variable {
179        /// Name of the companion column holding the offset values.
180        offset_column: String,
181    },
182}

186/// Descriptor for a measure column's MEASINFO metadata.
187///
188/// Corresponds to C++ `casa::TableMeasDesc`. This descriptor records the
189/// measure type, reference frame mode (fixed, variable-int, variable-string),
190/// and optional offset for a table column.
191#[derive(Debug, Clone)]
192pub struct TableMeasDesc {
193    column_name: String,
194    measure_type: MeasureType,
195    ref_desc: MeasRefDesc,
196    offset_desc: Option<MeasOffsetDesc>,
197}
198
199impl TableMeasDesc {
200    /// Creates a descriptor with a fixed reference frame.
201    pub fn new_fixed(column: &str, measure_type: MeasureType, refer: &str) -> Self {
202        Self {
203            column_name: column.to_owned(),
204            measure_type,
205            ref_desc: MeasRefDesc::Fixed {
206                refer: refer.to_owned(),
207            },
208            offset_desc: None,
209        }
210    }
211
212    /// Creates a descriptor with a variable integer reference column.
213    ///
214    /// Returns an error if `tab_ref_types` and `tab_ref_codes` have different lengths.
215    pub fn new_variable_int(
216        column: &str,
217        measure_type: MeasureType,
218        ref_column: &str,
219        tab_ref_types: Vec<String>,
220        tab_ref_codes: Vec<i32>,
221    ) -> Result<Self, TableError> {
222        if tab_ref_types.len() != tab_ref_codes.len() {
223            return Err(TableError::Storage(format!(
224                "TabRefTypes length ({}) != TabRefCodes length ({}) for column '{column}'",
225                tab_ref_types.len(),
226                tab_ref_codes.len()
227            )));
228        }
229        Ok(Self {
230            column_name: column.to_owned(),
231            measure_type,
232            ref_desc: MeasRefDesc::VariableInt {
233                ref_column: ref_column.to_owned(),
234                tab_ref_types,
235                tab_ref_codes,
236            },
237            offset_desc: None,
238        })
239    }
240
241    /// Creates a descriptor with a variable string reference column.
242    pub fn new_variable_string(column: &str, measure_type: MeasureType, ref_column: &str) -> Self {
243        Self {
244            column_name: column.to_owned(),
245            measure_type,
246            ref_desc: MeasRefDesc::VariableString {
247                ref_column: ref_column.to_owned(),
248            },
249            offset_desc: None,
250        }
251    }
252
253    /// Adds a fixed offset to this descriptor.
254    pub fn with_fixed_offset(mut self, offset: RecordValue) -> Self {
255        self.offset_desc = Some(MeasOffsetDesc::Fixed { offset });
256        self
257    }
258
259    /// Adds a variable offset column to this descriptor.
260    pub fn with_variable_offset(mut self, offset_column: &str) -> Self {
261        self.offset_desc = Some(MeasOffsetDesc::Variable {
262            offset_column: offset_column.to_owned(),
263        });
264        self
265    }

267    /// Returns `true` if the column has a `MEASINFO` sub-record in its keywords.
268    pub fn has_measinfo(table: &Table, column: &str) -> bool {
269        table
270            .column_keywords(column)
271            .and_then(|kw| kw.get(MEASINFO_KW))
272            .is_some()
273    }
274
275    /// Reconstructs a descriptor from a column's persisted keywords.
276    ///
277    /// Returns `None` if the column has no `MEASINFO` sub-record.
278    pub fn reconstruct(table: &Table, column: &str) -> Option<Self> {
279        let kw = table.column_keywords(column)?;
280        let measinfo = match kw.get(MEASINFO_KW) {
281            Some(Value::Record(r)) => r,
282            _ => return None,
283        };
284
285        // Extract type
286        let type_str = match measinfo.get(TYPE_KW) {
287            Some(Value::Scalar(ScalarValue::String(s))) => s.as_str(),
288            _ => return None,
289        };
290        let measure_type = MeasureType::from_str_casacore(type_str)?;
291
292        // Determine reference mode
293        let ref_desc = if let Some(Value::Scalar(ScalarValue::String(var_col))) =
294            measinfo.get(VAR_REF_COL_KW)
295        {
296            // Variable reference. Check for TabRefTypes/TabRefCodes (Int mode).
297            let tab_ref_types = measinfo.get(TAB_REF_TYPES_KW).and_then(|v| {
298                if let Value::Array(ArrayValue::String(a)) = v {
299                    Some(a.iter().cloned().collect::<Vec<_>>())
300                } else {
301                    None
302                }
303            });
304            let tab_ref_codes = measinfo.get(TAB_REF_CODES_KW).and_then(|v| match v {
305                Value::Array(ArrayValue::Int32(a)) => Some(a.iter().copied().collect::<Vec<i32>>()),
306                Value::Array(ArrayValue::UInt32(a)) => {
307                    Some(a.iter().map(|&x| x as i32).collect::<Vec<i32>>())
308                }
309                Value::Array(ArrayValue::Int64(a)) => {
310                    Some(a.iter().map(|&x| x as i32).collect::<Vec<i32>>())
311                }
312                _ => None,
313            });
314
315            if let (Some(types), Some(codes)) = (tab_ref_types, tab_ref_codes) {
316                MeasRefDesc::VariableInt {
317                    ref_column: var_col.clone(),
318                    tab_ref_types: types,
319                    tab_ref_codes: codes,
320                }
321            } else {
322                MeasRefDesc::VariableString {
323                    ref_column: var_col.clone(),
324                }
325            }
326        } else if let Some(Value::Scalar(ScalarValue::String(refer))) = measinfo.get(REF_KW) {
327            MeasRefDesc::Fixed {
328                refer: refer.clone(),
329            }
330        } else {
331            return None;
332        };
333
334        // Check for offset
335        let offset_desc = if let Some(Value::Record(off_rec)) = measinfo.get(REF_OFF_MSR_KW) {
336            Some(MeasOffsetDesc::Fixed {
337                offset: off_rec.clone(),
338            })
339        } else if let Some(Value::Scalar(ScalarValue::String(off_col))) =
340            measinfo.get(REF_OFF_COL_KW)
341        {
342            Some(MeasOffsetDesc::Variable {
343                offset_column: off_col.clone(),
344            })
345        } else {
346            None
347        };
348
349        Some(Self {
350            column_name: column.to_owned(),
351            measure_type,
352            ref_desc,
353            offset_desc,
354        })
355    }

357    /// Writes the MEASINFO sub-record to the column's keyword record.
358    pub fn write(&self, table: &mut Table) -> Result<(), TableError> {
359        let mut kw = table
360            .column_keywords(&self.column_name)
361            .cloned()
362            .unwrap_or_default();
363
364        let mut measinfo = RecordValue::default();
365        measinfo.upsert(
366            TYPE_KW,
367            Value::Scalar(ScalarValue::String(self.measure_type.as_str().to_owned())),
368        );
369
370        match &self.ref_desc {
371            MeasRefDesc::Fixed { refer } => {
372                measinfo.upsert(REF_KW, Value::Scalar(ScalarValue::String(refer.clone())));
373            }
374            MeasRefDesc::VariableInt {
375                ref_column,
376                tab_ref_types,
377                tab_ref_codes,
378            } => {
379                measinfo.upsert(
380                    VAR_REF_COL_KW,
381                    Value::Scalar(ScalarValue::String(ref_column.clone())),
382                );
383                measinfo.upsert(
384                    TAB_REF_TYPES_KW,
385                    Value::Array(ArrayValue::from_string_vec(tab_ref_types.clone())),
386                );
387                measinfo.upsert(
388                    TAB_REF_CODES_KW,
389                    Value::Array(ArrayValue::from_i32_vec(tab_ref_codes.clone())),
390                );
391            }
392            MeasRefDesc::VariableString { ref_column } => {
393                measinfo.upsert(
394                    VAR_REF_COL_KW,
395                    Value::Scalar(ScalarValue::String(ref_column.clone())),
396                );
397            }
398        }
399
400        if let Some(ref offset) = self.offset_desc {
401            match offset {
402                MeasOffsetDesc::Fixed { offset: off_rec } => {
403                    measinfo.upsert(REF_OFF_MSR_KW, Value::Record(off_rec.clone()));
404                }
405                MeasOffsetDesc::Variable { offset_column } => {
406                    measinfo.upsert(
407                        REF_OFF_COL_KW,
408                        Value::Scalar(ScalarValue::String(offset_column.clone())),
409                    );
410                }
411            }
412        }
413
414        kw.upsert(MEASINFO_KW, Value::Record(measinfo));
415
416        // C++ casacore also writes QuantumUnits via TableQuantumDesc so that
417        // ArrayMeasColumn can reconstruct units. Write default units if not
418        // already present.
419        if kw.get("QuantumUnits").is_none() {
420            kw.upsert(
421                "QuantumUnits",
422                Value::Array(ArrayValue::from_string_vec(
423                    self.measure_type.default_units(),
424                )),
425            );
426        }
427
428        table.set_column_keywords(&self.column_name, kw);
429        Ok(())
430    }
431
432    /// Returns the data column name.
433    pub fn column_name(&self) -> &str {
434        &self.column_name
435    }
436
437    /// Returns the measure type.
438    pub fn measure_type(&self) -> MeasureType {
439        self.measure_type
440    }
441
442    /// Returns the reference descriptor.
443    pub fn ref_desc(&self) -> &MeasRefDesc {
444        &self.ref_desc
445    }
446
447    /// Returns the offset descriptor, if any.
448    pub fn offset_desc(&self) -> Option<&MeasOffsetDesc> {
449        self.offset_desc.as_ref()
450    }

455/// A typed measure value read from a table column.
456///
457/// Wraps the specific measure type so callers can extract the one they need.
458#[derive(Debug, Clone)]
459pub enum MeasureValue {
460    /// An epoch measure.
461    Epoch(MEpoch),
462    /// A direction measure.
463    Direction(MDirection),
464    /// A position measure.
465    Position(MPosition),
466    /// A frequency measure.
467    Frequency(MFrequency),
468    /// A Doppler measure.
469    Doppler(MDoppler),
470    /// A radial velocity measure.
471    RadialVelocity(MRadialVelocity),
472}
473
474impl MeasureValue {
475    /// Returns the epoch, or an error if this is a different type.
476    pub fn as_epoch(&self) -> Result<&MEpoch, TableError> {
477        match self {
478            Self::Epoch(e) => Ok(e),
479            _ => Err(TableError::Storage("expected Epoch measure".to_owned())),
480        }
481    }
482
483    /// Returns the direction, or an error if this is a different type.
484    pub fn as_direction(&self) -> Result<&MDirection, TableError> {
485        match self {
486            Self::Direction(d) => Ok(d),
487            _ => Err(TableError::Storage("expected Direction measure".to_owned())),
488        }
489    }
490
491    /// Returns the position, or an error if this is a different type.
492    pub fn as_position(&self) -> Result<&MPosition, TableError> {
493        match self {
494            Self::Position(p) => Ok(p),
495            _ => Err(TableError::Storage("expected Position measure".to_owned())),
496        }
497    }
498
499    /// Returns the frequency, or an error if this is a different type.
500    pub fn as_frequency(&self) -> Result<&MFrequency, TableError> {
501        match self {
502            Self::Frequency(f) => Ok(f),
503            _ => Err(TableError::Storage("expected Frequency measure".to_owned())),
504        }
505    }
506
507    /// Returns the Doppler, or an error if this is a different type.
508    pub fn as_doppler(&self) -> Result<&MDoppler, TableError> {
509        match self {
510            Self::Doppler(d) => Ok(d),
511            _ => Err(TableError::Storage("expected Doppler measure".to_owned())),
512        }
513    }
514
515    /// Returns the radial velocity, or an error if this is a different type.
516    pub fn as_radial_velocity(&self) -> Result<&MRadialVelocity, TableError> {
517        match self {
518            Self::RadialVelocity(rv) => Ok(rv),
519            _ => Err(TableError::Storage(
520                "expected RadialVelocity measure".to_owned(),
521            )),
522        }
523    }
524}

528/// Resolves the reference string for a given row based on the descriptor.
529fn resolve_ref_string(
530    table: &Table,
531    desc: &TableMeasDesc,
532    row: usize,
533) -> Result<String, TableError> {
534    match &desc.ref_desc {
535        MeasRefDesc::Fixed { refer } => Ok(refer.clone()),
536        MeasRefDesc::VariableInt {
537            ref_column,
538            tab_ref_types,
539            tab_ref_codes,
540        } => {
541            let code = match table.get_scalar_cell(row, ref_column)? {
542                ScalarValue::Int32(v) => *v,
543                ScalarValue::Int64(v) => *v as i32,
544                ScalarValue::UInt32(v) => *v as i32,
545                _ => {
546                    return Err(TableError::Storage(format!(
547                        "ref column '{ref_column}' at row {row}: expected Int"
548                    )));
549                }
550            };
551            // Find this code in the mapping table
552            for (i, &c) in tab_ref_codes.iter().enumerate() {
553                if c == code {
554                    return tab_ref_types.get(i).cloned().ok_or_else(|| {
555                        TableError::Storage(format!(
556                            "TabRefTypes index {i} out of bounds (len={}) for column '{}'",
557                            tab_ref_types.len(),
558                            desc.column_name
559                        ))
560                    });
561                }
562            }
563            Err(TableError::Storage(format!(
564                "ref code {code} not found in TabRefCodes for column '{}'",
565                desc.column_name
566            )))
567        }
568        MeasRefDesc::VariableString { ref_column } => {
569            match table.get_scalar_cell(row, ref_column)? {
570                ScalarValue::String(s) => Ok(s.clone()),
571                _ => Err(TableError::Storage(format!(
572                    "ref column '{ref_column}' at row {row}: expected String"
573                ))),
574            }
575        }
576    }
577}

579/// Creates a MeasureValue from raw f64 values and a reference string.
580fn make_measure(
581    measure_type: MeasureType,
582    values: &[f64],
583    refer_str: &str,
584) -> Result<MeasureValue, TableError> {
585    let err = |msg: String| TableError::Storage(msg);
586    match measure_type {
587        MeasureType::Epoch => {
588            let r: EpochRef = refer_str
589                .parse()
590                .map_err(|e| err(format!("invalid epoch ref '{refer_str}': {e}")))?;
591            Ok(MeasureValue::Epoch(MEpoch::new(
592                MjdHighPrec::from_mjd(values[0]),
593                r,
594            )))
595        }
596        MeasureType::Direction => {
597            let r: DirectionRef = refer_str
598                .parse()
599                .map_err(|e| err(format!("invalid direction ref '{refer_str}': {e}")))?;
600            Ok(MeasureValue::Direction(MDirection::from_angles(
601                values[0], values[1], r,
602            )))
603        }
604        MeasureType::Position => {
605            let r: PositionRef = refer_str
606                .parse()
607                .map_err(|e| err(format!("invalid position ref '{refer_str}': {e}")))?;
608            Ok(MeasureValue::Position(match r {
609                PositionRef::ITRF => MPosition::new_itrf(values[0], values[1], values[2]),
610                PositionRef::WGS84 => MPosition::new_wgs84(values[0], values[1], values[2]),
611            }))
612        }
613        MeasureType::Frequency => {
614            let r: FrequencyRef = refer_str
615                .parse()
616                .map_err(|e| err(format!("invalid frequency ref '{refer_str}': {e}")))?;
617            Ok(MeasureValue::Frequency(MFrequency::new(values[0], r)))
618        }
619        MeasureType::Doppler => {
620            let r: DopplerRef = refer_str
621                .parse()
622                .map_err(|e| err(format!("invalid doppler ref '{refer_str}': {e}")))?;
623            Ok(MeasureValue::Doppler(MDoppler::new(values[0], r)))
624        }
625        MeasureType::RadialVelocity => {
626            let r: RadialVelocityRef = refer_str
627                .parse()
628                .map_err(|e| err(format!("invalid radial velocity ref '{refer_str}': {e}")))?;
629            Ok(MeasureValue::RadialVelocity(MRadialVelocity::new(
630                values[0], r,
631            )))
632        }
633    }
634}

636/// Extracts the raw f64 values from a MeasureValue.
637fn measure_values(mv: &MeasureValue) -> Vec<f64> {
638    match mv {
639        MeasureValue::Epoch(e) => vec![e.value().as_mjd()],
640        MeasureValue::Direction(d) => {
641            let (lon, lat) = d.as_angles();
642            vec![lon, lat]
643        }
644        MeasureValue::Position(p) => p.values().to_vec(),
645        MeasureValue::Frequency(f) => vec![f.hz()],
646        MeasureValue::Doppler(d) => vec![d.value()],
647        MeasureValue::RadialVelocity(rv) => vec![rv.ms()],
648    }
649}

651/// Returns the reference string for a MeasureValue.
652fn measure_ref_string(mv: &MeasureValue) -> String {
653    match mv {
654        MeasureValue::Epoch(e) => e.refer().as_str().to_owned(),
655        MeasureValue::Direction(d) => d.refer().as_str().to_owned(),
656        MeasureValue::Position(p) => p.refer().as_str().to_owned(),
657        MeasureValue::Frequency(f) => f.refer().as_str().to_owned(),
658        MeasureValue::Doppler(d) => d.refer().as_str().to_owned(),
659        MeasureValue::RadialVelocity(rv) => rv.refer().as_str().to_owned(),
660    }
661}
662
663/// Resolves the casacore integer code for a ref string and measure type.
664fn ref_string_to_code(measure_type: MeasureType, refer: &str) -> Result<i32, TableError> {
665    let err = |msg: String| TableError::Storage(msg);
666    match measure_type {
667        MeasureType::Epoch => {
668            let r: EpochRef = refer
669                .parse()
670                .map_err(|e| err(format!("invalid epoch ref '{refer}': {e}")))?;
671            Ok(r.casacore_code())
672        }
673        MeasureType::Direction => {
674            let r: DirectionRef = refer
675                .parse()
676                .map_err(|e| err(format!("invalid direction ref '{refer}': {e}")))?;
677            Ok(r.casacore_code())
678        }
679        MeasureType::Position => {
680            let r: PositionRef = refer
681                .parse()
682                .map_err(|e| err(format!("invalid position ref '{refer}': {e}")))?;
683            Ok(r.casacore_code())
684        }
685        MeasureType::Frequency => {
686            let r: FrequencyRef = refer
687                .parse()
688                .map_err(|e| err(format!("invalid frequency ref '{refer}': {e}")))?;
689            Ok(r.casacore_code())
690        }
691        MeasureType::Doppler => {
692            let r: DopplerRef = refer
693                .parse()
694                .map_err(|e| err(format!("invalid doppler ref '{refer}': {e}")))?;
695            Ok(r.casacore_code())
696        }
697        MeasureType::RadialVelocity => {
698            let r: RadialVelocityRef = refer
699                .parse()
700                .map_err(|e| err(format!("invalid radial velocity ref '{refer}': {e}")))?;
701            Ok(r.casacore_code())
702        }
703    }
704}

708/// Read-only accessor for a scalar measure column.
709///
710/// Corresponds to C++ `casa::ScalarMeasColumn<M>`. A "scalar" measure column
711/// has one measure per row (e.g. one epoch, one direction). The underlying
712/// data column is an `ArrayColumn<Double>` with shape `[N]` where N is the
713/// number of values per measure (1 for epoch, 2 for direction, 3 for position).
714pub struct ScalarMeasColumn<'a> {
715    table: &'a Table,
716    column_name: String,
717    desc: TableMeasDesc,
718}
719
720impl<'a> ScalarMeasColumn<'a> {
721    /// Attaches to a scalar measure column.
722    ///
723    /// Returns an error if the column has no MEASINFO keywords.
724    pub fn new(table: &'a Table, column: &str) -> Result<Self, TableError> {
725        let desc = TableMeasDesc::reconstruct(table, column).ok_or_else(|| {
726            TableError::Storage(format!("column '{column}' has no MEASINFO keywords"))
727        })?;
728        Ok(Self {
729            table,
730            column_name: column.to_owned(),
731            desc,
732        })
733    }
734
735    /// Returns the cached descriptor.
736    pub fn desc(&self) -> &TableMeasDesc {
737        &self.desc
738    }
739
740    /// Reads the measure at `row`.
741    pub fn get(&self, row: usize) -> Result<MeasureValue, TableError> {
742        let refer = resolve_ref_string(self.table, &self.desc, row)?;
743        let values = read_array_values(self.table, row, &self.column_name)?;
744        make_measure(self.desc.measure_type, &values, &refer)
745    }
746
747    /// Convenience: reads the epoch at `row`.
748    pub fn get_epoch(&self, row: usize) -> Result<MEpoch, TableError> {
749        self.get(row)?.as_epoch().cloned()
750    }
751
752    /// Convenience: reads the direction at `row`.
753    pub fn get_direction(&self, row: usize) -> Result<MDirection, TableError> {
754        self.get(row)?.as_direction().cloned()
755    }
756}

760/// Read-only accessor for an array measure column.
761///
762/// Corresponds to C++ `casa::ArrayMeasColumn<M>`. An "array" measure column
763/// has multiple measures per row (e.g. an array of directions). The underlying
764/// data column is an `ArrayColumn<Double>` with shape `[N, M]` where N is
765/// values-per-measure and M is the number of measures per cell.
766pub struct ArrayMeasColumn<'a> {
767    table: &'a Table,
768    column_name: String,
769    desc: TableMeasDesc,
770}
771
772impl<'a> ArrayMeasColumn<'a> {
773    /// Attaches to an array measure column.
774    pub fn new(table: &'a Table, column: &str) -> Result<Self, TableError> {
775        let desc = TableMeasDesc::reconstruct(table, column).ok_or_else(|| {
776            TableError::Storage(format!("column '{column}' has no MEASINFO keywords"))
777        })?;
778        Ok(Self {
779            table,
780            column_name: column.to_owned(),
781            desc,
782        })
783    }
784
785    /// Returns the cached descriptor.
786    pub fn desc(&self) -> &TableMeasDesc {
787        &self.desc
788    }
789
790    /// Reads all measures at `row` as a vector.
791    pub fn get(&self, row: usize) -> Result<Vec<MeasureValue>, TableError> {
792        let refer = resolve_ref_string(self.table, &self.desc, row)?;
793        let all_values = read_array_values(self.table, row, &self.column_name)?;
794        let n = self.desc.measure_type.values_per_measure();
795        let mut result = Vec::with_capacity(all_values.len() / n);
796        for chunk in all_values.chunks(n) {
797            result.push(make_measure(self.desc.measure_type, chunk, &refer)?);
798        }
799        Ok(result)
800    }
801}

805/// Mutable accessor for writing scalar measures to a column.
806///
807/// Corresponds to the write path of C++ `casa::ScalarMeasColumn<M>`.
808pub struct ScalarMeasColumnMut<'a> {
809    table: &'a mut Table,
810    column_name: String,
811    desc: TableMeasDesc,
812}
813
814impl<'a> ScalarMeasColumnMut<'a> {
815    /// Attaches to a scalar measure column for writing.
816    pub fn new(table: &'a mut Table, column: &str) -> Result<Self, TableError> {
817        let desc = TableMeasDesc::reconstruct(table, column).ok_or_else(|| {
818            TableError::Storage(format!("column '{column}' has no MEASINFO keywords"))
819        })?;
820        Ok(Self {
821            table,
822            column_name: column.to_owned(),
823            desc,
824        })
825    }
826
827    /// Returns the cached descriptor.
828    pub fn desc(&self) -> &TableMeasDesc {
829        &self.desc
830    }
831
832    /// Writes a measure at `row`.
833    pub fn put(&mut self, row: usize, mv: &MeasureValue) -> Result<(), TableError> {
834        let vals = measure_values(mv);
835        self.table.set_cell(
836            row,
837            &self.column_name,
838            Value::Array(ArrayValue::from_f64_vec(vals)),
839        )?;
840        self.write_ref(row, mv)?;
841        Ok(())
842    }
843
844    /// Convenience: writes an epoch at `row`.
845    pub fn put_epoch(&mut self, row: usize, epoch: &MEpoch) -> Result<(), TableError> {
846        self.put(row, &MeasureValue::Epoch(epoch.clone()))
847    }
848
849    /// Convenience: writes a direction at `row`.
850    pub fn put_direction(&mut self, row: usize, dir: &MDirection) -> Result<(), TableError> {
851        self.put(row, &MeasureValue::Direction(dir.clone()))
852    }
853
854    fn write_ref(&mut self, row: usize, mv: &MeasureValue) -> Result<(), TableError> {
855        let refer = measure_ref_string(mv);
856        match &self.desc.ref_desc {
857            MeasRefDesc::Fixed { .. } => Ok(()),
858            MeasRefDesc::VariableInt { ref_column, .. } => {
859                let code = ref_string_to_code(self.desc.measure_type, &refer)?;
860                let ref_column = ref_column.clone();
861                self.table
862                    .set_cell(row, &ref_column, Value::Scalar(ScalarValue::Int32(code)))?;
863                Ok(())
864            }
865            MeasRefDesc::VariableString { ref_column } => {
866                let ref_column = ref_column.clone();
867                self.table
868                    .set_cell(row, &ref_column, Value::Scalar(ScalarValue::String(refer)))?;
869                Ok(())
870            }
871        }
872    }
873}

876
877/// Mutable accessor for writing arrays of measures to a column.
878///
879/// Corresponds to the write path of C++ `casa::ArrayMeasColumn<M>`.
880pub struct ArrayMeasColumnMut<'a> {
881    table: &'a mut Table,
882    column_name: String,
883    desc: TableMeasDesc,
884}
885
886impl<'a> ArrayMeasColumnMut<'a> {
887    /// Attaches to an array measure column for writing.
888    pub fn new(table: &'a mut Table, column: &str) -> Result<Self, TableError> {
889        let desc = TableMeasDesc::reconstruct(table, column).ok_or_else(|| {
890            TableError::Storage(format!("column '{column}' has no MEASINFO keywords"))
891        })?;
892        Ok(Self {
893            table,
894            column_name: column.to_owned(),
895            desc,
896        })
897    }
898
899    /// Writes an array of measures at `row`.
900    ///
901    /// All measures must have the same reference frame. For variable-ref
902    /// columns, the reference of the first measure is written.
903    pub fn put(&mut self, row: usize, measures: &[MeasureValue]) -> Result<(), TableError> {
904        // Validate all measures share the same reference
905        if measures.len() > 1 {
906            let first_ref = measure_ref_string(&measures[0]);
907            for (i, mv) in measures.iter().enumerate().skip(1) {
908                let r = measure_ref_string(mv);
909                if r != first_ref {
910                    return Err(TableError::Storage(format!(
911                        "ArrayMeasColumnMut::put: measure[{i}] has reference '{r}' \
912                         but measure[0] has '{first_ref}'; all measures in a row \
913                         must share the same reference"
914                    )));
915                }
916            }
917        }
918        let n = self.desc.measure_type.values_per_measure();
919        let mut all_values = Vec::with_capacity(measures.len() * n);
920        for mv in measures {
921            all_values.extend_from_slice(&measure_values(mv));
922        }
923        self.table.set_cell(
924            row,
925            &self.column_name,
926            Value::Array(ArrayValue::from_f64_vec(all_values)),
927        )?;
928        // Write reference from first measure if variable
929        if !measures.is_empty() {
930            let refer = measure_ref_string(&measures[0]);
931            match &self.desc.ref_desc {
932                MeasRefDesc::Fixed { .. } => {}
933                MeasRefDesc::VariableInt { ref_column, .. } => {
934                    let code = ref_string_to_code(self.desc.measure_type, &refer)?;
935                    let ref_column = ref_column.clone();
936                    self.table.set_cell(
937                        row,
938                        &ref_column,
939                        Value::Scalar(ScalarValue::Int32(code)),
940                    )?;
941                }
942                MeasRefDesc::VariableString { ref_column } => {
943                    let ref_column = ref_column.clone();
944                    self.table.set_cell(
945                        row,
946                        &ref_column,
947                        Value::Scalar(ScalarValue::String(refer)),
948                    )?;
949                }
950            }
951        }
952        Ok(())
953    }
954}

958fn read_array_values(table: &Table, row: usize, column: &str) -> Result<Vec<f64>, TableError> {
959    let arr = table.get_array_cell(row, column)?;
960    match arr {
961        ArrayValue::Float64(a) => Ok(a.iter().copied().collect()),
962        ArrayValue::Float32(a) => Ok(a.iter().map(|&v| v as f64).collect()),
963        ArrayValue::Int32(a) => Ok(a.iter().map(|&v| v as f64).collect()),
964        ArrayValue::Int64(a) => Ok(a.iter().map(|&v| v as f64).collect()),
965        _ => Err(TableError::Storage(format!(
966            "column '{column}' at row {row}: expected numeric array"
967        ))),
968    }
969}
